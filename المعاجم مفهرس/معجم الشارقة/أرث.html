أَرَثَ(فعل) أَرِثَ(فعل) أَرَّثَ(فعل) تَأَرَّثَ(فعل) إِرَاث(اسم) أُرْث(اسم) إِرْث(اسم) أُرْثَة(اسم) أَرِيث(اسم)

المعانى الكلّيّة

  1. إيقادُ النّارِ
01 —

أَرَثَ

فعل
1.1معنى من معاني «أَرَثَ»

* أَرَثَ (بِفَتْحِ الرّاءِ) فلانٌ النَّارَ يَأْرُثُ (بِضَمِّ الرّاءِ) أَرثًا: أَوْقَدَها. فَهُو آرِثٌ، وَهِيَ بِتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ: بَشّارُ بْنُ بُرْدٍ (ت: 167ه=784م) يَتَغَزَّلُ:

تِلْكَ الَّتِي لَوْ نِلْتُها وَالهَوَى
وَالسُّقْمُ بَيْنَ الأَضْلُعِ الْآرِثُ

كَأَنَّما في كَبِدي قُرْحَةٌ
مِن حُبِّها يَفْرِثُها فارِثُ

ديوان بشار بن برد. تح: عباس، ج: 1، ص: 160.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672ه=1273م):

"الأَرْثُ: مَصْدَرُ أَرَثَ النّارَ: أَشْعَلَها".(وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ فِي العَصْرِ الحَديثِ)

ابن مالك، إكمال الإعلام. تح: الغامدي، ج: 1. ص: 104.— المصدر
02 —

أَرِثَ

فعل
2.1معنى من معاني «أَرِثَ»

* أَرِثَتِ(بِكَسْرِ الرّاءِ) الشَّاةُ، تَأْرَثُ (بِفَتْحِ الرّاءِ) أَرْثًا، وَأُرْثَةً: كانَ فيها بَياضٌ وسَوادٌ. فَهُوَ آرَثُ، وَهِي أَرْثاءُ. (ج) أُرْثٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ ابْنُ عَبّاد (ت: 385ه=995م):

"والنَّعْجَةُ الأَرْثاءُ: كالرَّقْطاءِ ببَياضٍ وسَوادٍ، والاسْمُ الأُرْثَةُ، والذَّكَرُ آرَثُ. وعَنْزٌ أَرْثاءُ: وهي التي بَياضُها في مَنْكِبَيْها ولَبانِها". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ فِي بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 169.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672ه=1274م) يذكُرُ مُثَلَّثَ الكلامِ:

وَالنَّعَجَاتُ الرُّقْطُ هُنَّ الْأُرْثُ
أَرْثَاءُ إِحْداها بِلا ارْتِيابِ

ابن مالك، الإعلام بمثلث الكلام، ج: 1، ص: 18.— المصدر
03 —

أَرَّثَ

فعل
3.1معنى من معاني «أَرَّثَ»

* أَرَّثَ فلانٌ النَّارَ: أَوْقَدَها وَحرّكَ جَمْرَها وزادَ ضِرامَها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُ بْنُ زَيْدٍ العبادِيُّ (ت: 35ق هـ=587م) يذكُرُ نارًا أبْصَرَها:

أَرَّثَتْ فِي عَرَفٍ مَوْقِدَهَا
فَأَضَاءَتْ لَمْعَ كَفٍّ بِسِوَارِ

ديوان عدي بن زيد العبادي. تح: المعيبد، ج: 1، ص: 93.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال أَسْلَمُ مَوْلى عُمَرَ بْنِ الخَطّابِ (23ه=644م) يَذْكُرُ مَسِيرَهُ مَعَ مَوْلاهُ:

"خَرَجْتُ مَعَهُ حَتَّى إِذَا كُنَّا بِحَرَّةِ وَاقِم فَإذا نارٌ تُؤَرَّثُ بِصِرارِ".

التميمي، المبعث والمغازي، ج: 2، ص: 809.— المصدر
”وقالَ الكُمَيْتُ الأَسَدِىّ (ت: 126هـ = 744م):

ومِنّا لَقيطٌ وابْنُماهُ وحاجِبٌ
مُؤَرِّثُ نيرانَ المكارمِ لا المُخْبي

شرح هاشميات الكميت. تح: داود سلوم، وآخر، ج: 1، ص: 125.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبُو زَيْدٍ الْأَنْصارِيُّ (ت: 215ه=830م):

"وَيُقَالُ: أَرِّثْ نَارَكَ تَأْرِيثًا؛ إِذَا أَمَرَهُ أَنْ يُعْظِمَهَا بِكَثْرَةِ مَا يُلْقَى عَلَيْهَا مِنَ الحَطَبِ وَالبَعَرِ".

أبو زيد الأنصاري، النوادر في اللغة. تح: محمّد عبد القادر، ج: 1، ص: 404.— المصدر
”وَقال الْآمِدِيُّ (ت: 253ه =936م) ينْتَقِدُ بيْتًا لأبي تَمّامٍ جَعَلَ فيه الوَزيرَ يُؤَرِّثُ النَّارَ والخليفةَ يَشُبُّها:

"وكانُ الأحسَنُ أنْ يكونَ الخليفةُ المبتدئَ بتَأْريثِ النَّارِ وهوَ الذي يَشُبُّ، والخليفةُ القائلَ، وهو الذي يَفْعَلُ".

الآمدي، الموازنة بين شعر أبي تمام والبحتري. تح: المحارب، ج: 3، ص: 35.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ: الـمُسْتَعْصِمِيُّ (ت: 710ه=1310م) يذكرُ نارًا تَشْتَعِلُ:

إِذَا صَدَعَتْهَا الرِّيحُ أَرَّثَ ضَوْءَهَا
مِنَ الأَصْلِ فَرْعٌ يَابِسٌ وَرَطِيبُ

المستعصمي، الدر الفريد وبيت القصيد. تح: الجبوري، ج: 10، ص: 149.— المصدر
3.2معنى من معاني «أَرَّثَ»

وــــــــــبَيْنَ القَوْمِ الحِقْدَ، أو الشَّرَّ وَنَحْوَهُما: أَفْسَدَ وَأَغْرَى.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ هُدْبَةُ بْنُ الْخَشْرَمِ (ت: 57ه=676م) يَمْدَحُ نَفْسَهُ:

وما كُنْتُ مِمَّنْ أَرَّثَ الشَرَّ بَيْنَهُمْ
ولا حِينَ جَدَّ الشَرُّ مِمَّنْ تَخَشَّعَا

ديوان هدبة بن الخشرم. تح: الجبوري، ص: 117.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ عَبْدِاللهِ الْخُزاعِيُّ (ت: 253ه = 867م) يُخاطبُ ابنَه حولَ خلافٍ بين ثَعْلَبَ والـمُبَرِّدِ:

"الناسُ يُخْطِئونَ فاسْمَعْ منهما ولا تُؤَرِّثَنَّ بينهما، ولا تُخْرجْ تَوْقيعي إلى أحَد".

الزجاجي، مجالس العلماء. تح: هارون، ص: 85.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الْمَقْدِسِيُّ (ت: 685ه=1286م):

"وَهَذِهِ فَاتِحَةُ كُلِّ حادِثَةٍ، لِنارِ الفِتَنِ مُؤَرِّثَةٍ".

يوسف بن يحيى،عقد الدرر. تح: البوريني، ج: 1، ص: 314.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ نَجيب مَحْفوظ (ت: 1428ه=2006م):

"وَسَيْطَرَ الْمِصْرِيُّونَ عَلَى الْمَيْدَانِ فِي زَمَنٍ يَسِيرٍ لَا يُصَدَّقُ، بَعْدَ أَنْ قَاتَلُوا بِغَضَبٍ وَحَنْقٍ، وَضَرَبُوا بِسَوَاعِدَ يَشُدُّ أَعْصَابَهَا حِقْدٌ مُؤَرِّثٌ وَسخِيمَةُ مُسْتَعِرَةٌ".

نجيب محفوظ، الأعمال الكاملة، ج: 1، ص: 589.— المصدر
3.3معنى من معاني «أَرَّثَ»

وــــــــــالحَرْبَ: هَيَّجَها وزادَ ضِرامَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبُو تَمّامٍ (ت: 231ه=846م) في إثارَةِ الحَرْبِ بَيْنَ حَيَّيْنِ:

وَأَرَّثَ بَينَ حَيِّ بَنِي جُلاحٍ
سَنَا حَربٍ وَحَيِّ بَنِي مَصَادِ

التبريزي، شرح ديوان أبي تمام. تح: الأسمر، ج: 1، ص: 202.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَباسُ الْعَقّاد (ت: 1383ه=1964م):

ماتَ فِي الْحَرْبِ الَّتِي أَرَّثَها
بَغْيُ فِيلِيب أَبِيكَ الْغَمْشَرِ

ديوان العقاد، ج: 1، ص: 139.— المصدر
3.4معنى من معاني «أَرَّثَ»

وــــــــــالأرْضَيْنِ: جَعَلَ بَيْنَهُما حَدًّا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سِيدَه (ت: 458ه=1066م):

"وأَرَّثَ الأَرْضَيْنِ: جَعَلَ بَيْنَهُما أُرْثَةً". (وهَذِهَ الدَّلَالَةُ سَجَّلَتْهَا الْمَعَاجِمُ فِي بَقِيَّةِ الْعُصُورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 178.— المصدر
04 —

تَأَرَّثَ

فعل
4.1معنى من معاني «تَأَرَّثَ»

* تَأَرَّثَتِ النَّارُ: اتَّقَدَتْ وتَزايَدَ ضِرامُها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الْكُمَيْتُ بْنُ زَيْدٍ (126ه=743م) يمْدَحُ الهاشِمِيِّيْنَ:

هُمُ صَفْوَةُ اللهِ الخِيارُ وَفِيهِمُ
تَأَرَّثُ نِيرَانُ الهُدَى وثُقُوبُهَا

ديوان الكميت. تح: محمّد طريفي، ج: 1، ص: 74.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (231ه=846م، مِنْ شَواهِدِ ابْنِ الْأَعْرابِيِّ):

فَإِنَّ بِأَعْلَى ذِي الْمَجازَةِ سَرْحَةً
طَوِيلًا عَلَى أَهْلِ الْمَجازَةِ عارُها

وَلَوْ ضَرَبُوها بِالفُؤُوسِ وَحَرَّقُوا
عَلَى أَصْلِها حَتَّى تَأَرَّثَ نَارُها

ابن فارس، المقاييس. تح: هارون، ج: 1، ص: 36.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال عَبْدُالقادِرِ بْنُ عُمَرَ الْبَغْدادِيُّ (ت: 1093ه=1682م):

"وَكَذا أُرِّثَتِ، مَجْهُولُ أَرَّثْتُ النّارَ أَوْقَدْتُها، وَتَأَرَّثَتْ: اتّقَدَتْ". (وهَذِهِ الدَّلالةُ سَجَّلَتْهَا الْمَعَاجِمُ فِي العَصْرِ الْحَدِيثِ)

البغدادي، شرح أبيات مغني اللبيب. تح: عبد العزيز رباح، ج: 5، ص: 173.— المصدر
4.2معنى من معاني «تَأَرَّثَ»

وــــــــــالضَّغِينَةُ، وَنَحْوُها: اشْتَدَّتْ وَتَزايَدَتْ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّد رَشِيد رِضا (ت: 1354ه=1935م):

"وَإِنَّما تَكْرَهُهَا لِمَحْضِ التَعَصُّبِ عَلَى السَّوادِ الْأَعْظَمِ مِنْ أَهْلِ وَطَنِها، وَنَاهِيكَ بِما تَأَرَّثَ مِنَ الْأَضْغانِ بَيْنَ مُسْلِمِي الْأَناضُول والرُّومِ وَالْأَرْمَن".

رشيد رضا، الخِلافة. الزّهراء للإعلام العربيّ، ج: 1، ص: 107.— المصدر
05 —

إِرَاث

اسم
5.1معنى من معاني «إِرَاث»

* الإِرَاثُ: النَّارُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو الْخَطّابِ البَهْدَلِيُّ (ت: 247 ه=861 م) يَصِفُ فَرَسًا:

مُحَجَّلُ رِجْلَيْنِ طَلْقُ الْيَدَيْنِ
لَهُ غُرَّةٌ مِثْلُ ضَوْءِ الْإِرَاثِ

ابن معتز، طبقات الشعراء. تح: فراج، ج: 1، ص: 134.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قَالَ ابْنُ مَنْظُورٍ (ت: 711ه=1311م):

"والإِراثُ: ... وقيلَ: هِيَ النَّارُ نَفْسُها". (وهَذِهِ الدَّلالةُ سَجَّلَتْهَا الْمَعَاجِمُ فِي العَصْرِ الْحَدِيثِ.)

ابن منظور، اللسان. تح: أمين عبد الوهاب، ج: 1، ص: 112.— المصدر
5.2معنى من معاني «إِرَاث»

وــــــــــ: ما تُوقَدُ بِهِ النَّارُ مِنْ آلَةٍ أَوْ عُودٍ ونَحْوِهِ. (ج) أُرُثٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ ابْنُ عَبّاد (ت: 385ه=995م):

"والإِرَاثُ: مَا تُوَرَّثُ مِنْهُ النّارُ". (وهَذِهَ الدَّلَالَةُ سَجَّلَتْهَا الْمَعَاجِمُ فِي بَقِيَّةِ الْعُصُورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 169.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672ه=1359م) يذكرُ مُثَلَّثَ الكلام:

مَعْ حَيَّةٍ وَأُرُثٌ قُلْ حُضْبُ
وَإِنْ تُوحِّدْ فَأْتِ بِالحِضَابِ

ابن مالك، الإعلام بمثلث الكلام، ج: 1، ص: 47.— المصدر
5.3معنى من معاني «إِرَاث»

وــــــــــ: الرَّمادُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سِيدَه (ت: 458ه=1066م):

"والإِراثُ: الرَّماد". (وهَذِهَ الدَّلَالَةُ سَجَّلَتْهَا الْمَعَاجِمُ فِي بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 177.— المصدر
06 —

أُرْث

اسم
6.1معنى من معاني «أُرْث»

* الأُرْثُ: شَوْكٌ سَبْطُ الوَرَقِ، لَهُ قَضِيبٌ واحِدٌ في وَسَطِهِ، وَهُوَ مَرْعَى لِلْإِبِلِ.

ع
