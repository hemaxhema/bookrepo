  1. زِيادَةٌ في خِلْقَةٍ
01 —

عَفَلَ

فعل
1.1معنى من معاني «عَفَلَ»

* عَفَلَ (بِفَتْحِ الفاءِ) فُلانٌ الكَبْشَ يَعْفُلُهُ (بِضمِّ الفاءِ) عَفْلًا: مَسَّ عَفْلَهُ لِيَنْظُرَ سِمَنَهُ مِنْ هُزالِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأعْرابِيِّ (ت: 231هـ=845م):

"إذا مَسَّ الرَّجُلُ عَفَلَ الكَبْشِ لِيَنْظُرَ سِمَنَهُ يُقالُ: جَسَّهُ، وغَبَطَهُ، وعَفَلَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 402.— المصدر
02 —

عَفِلَ

فعل
2.1معنى من معاني «عَفِلَ»

* عَفِلَتْ (بِكَسْرِ الفاءِ) فُلانَةُ وغَيْرُها تَعْفَلُ (بِفَتْحِ الفاءِ) عَفَلًا: أَصابَها غِلَظٌ في رَحِمِها كَالأُدْرَةِ لِلرِّجالِ؛ فَهي عَفْلاءُ(ج) عُفْلٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتْ رُهْمُ بِنْتُ الخَزْرَجِ (ت: 202ق هـ=450م) تَسُبُّ ضَرَّتَها:

"يا عَفْلاءُ".

المفضل بن سلمة، الفاخر. تح: الطحاوي، ص: 62.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أبي طالِبٍ (ت: 40ه=661م):

"إذا تَزَوَّجَ الرَّجُلُ المَجْنونَةَ أوِ المَجْذومَةَأوِ العَفْلاءَ؛ فَإنْ لَمْ يَكُنْ دَخَلَ بِهافُرِّقَ بَيْنَهُما، وإنْ دَخَلَ بِها جازَتْ عَلَيْهِ".

البيهقي، الخلافيات. تح: النحال، ج: 6، ص: 128.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"عَفِلَتِ المَرْأَةُ عَفَلًا؛ فَهي عَفْلاءُ. وعَفِلَتِ النّاقِضَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل،العين. تح: المخزومي وآخر، ج: 2، ص: 145.— المصدر
”وقالَ الخِرَقِيُّ (ت: 334هـ=946م)يُبَيِّنُ العُيوبَ المُبيحَةَ لِفَسْخِ النِّكاحِ:

"وأَيُّ الزَّوْجَيْنِ وَجَدَ بِصاحِبِهِ جُنونًا أوْ جُذامًا أوْ بَرَصًا أوْ كانَتِ المَرْأَةُ رَتْقاءَ أوْ قَرْناءَ أوْ عَفْلاءَأوِ الرَّجُلُ مَجْبوبًا؛ فَلِمَنْ وَجَدَ ذَلِكَ مِنْهُما بِصاحِبِهِ الخِيارُ في فَسْخِ النِّكاحِ".

مختصر الخرقي، تح:الشاويش، ص: 142.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ تَيْمِيّةَ (ت: 728هـ=1327م) يُبَيِّنُ العُيوبَ المُبيحَةَ لِفَسْخِ النِّكاحِ:

"فَإذا كانَ الزَّوْجُ صَغيرًا أوْ بِهِ جُنونٌ أوْ جُذامٌ أوْ بَرَصٌ؛ فَالمَسْألَةُ الّتي في الرَّضاعِ تَقْتَضي أنَّ لَها الفَسْخَ في الحالِ، ولا يَنْتَظِرُ وَقْتَ إمْكانِ الوَطْءِ، وعلى قِياسِهِ الزَّوْجَةُ إذا كانَتْ صَغيرَةً أوْ مَجْنونَةً أوْ عَفْلاءَ أوْ قَرْناءَ".

المستدرك على مجموع فتاوى ابن تيمية. تح: ابن قاسم، ج: 4، ص: 177.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَرَفَةَ الدُّسوقِيُّ (ت: 1230هـ=1815م) يُفَصِّلُ في مَسْأَلَةِ الإيلاءِ والوِقاعِ بَيْنَ الشَّيْخِ والمَرْأَةِ:

"يُمْكِنُ الوِقاعُ مِنْ جِهَتِهِ سَواءٌ أَمْكَنَ مِنْ جِهَتِها أمْ لا؟ فَيَنْعَقِدُ الإيلاءُ إذا أَمْكَنَ الوِقاعُ مِنْ جِهَتِهِ، ولَوْ كانَتْ رَتْقاءَ أوْ ‌عَفْلاءَ أوْ صَغيرَةً لا تُطيقُ، أوْ غَيْرَ مَدْخولٍ بِها".

حاشية الدسوقي على الشرح الكبير. دار الفكر، ج: 2، ص: 427.— المصدر
2.2معنى من معاني «عَفِلَ»

وــــــــــفُلانٌ: خَرَجَ مِنْ دُبُرِهِ شَيْءٌ مُدَوَّرٌ كَالبَيْضَةِ؛ فَهُوَ أَعْفَلُ، وهِيَ عَفْلاءُ. (ج) عُفْلٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ خِداشُ بْنُ زُهَيْرٍ (ت: 30ق هـ=592م) يُعَيِّرُ عَبْدَ اللهِ بْنَ جُدْعانَ:

أُرَيْصِعُ حَلّافٌ على كُلِّ بَيْعَةٍ
وآدَرُ مُسْتَلْقٍ بِمَكَّةَ أَعْفَلُ

[الأَرْصَعُ: القَليلُ لَحْمِ العَجُزِ والفَخِذَيْنِ. الآدَرُ: الـمَنْفوخُ الخُصْيَةِ]
أشعار العامريين الجاهليين. تح: عبد الكريم يعقوب، ص: 39.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بْنُ أبي رَبيعَةَ (ت: 93هـ=712م) يَصِفُ بَعْضَ النِّساء اللّائي لا تَجوزُ لَهُنَّ شهادَةٌ:

ولَحى اللهُ كُلَّ عَفْلاءَ زَلّا
ءَ عَبوسًا قَدْ أَذَّنَتْ بِالبَذاءِ

صَرْصَرٍ سَلْفَعٍ رَضيعَةِ غولٍ
لَمْ تَزَلْ في شَصيبَةٍ وشَقاءِ

[الزَلّاءُ: الضَّعيفَةُ الوَرِكَيْنِ. الصَّرْصَرُ: الشَّديدَةُ الصِّياحِ. السَّلْفَعُ: البَذيئَةُ والسَّليطَةُ اللِّسانِ. الشَّصيبَةُ: الجَدْبُ والقَحْطُ]
ديوان عمر بن أبي ربيعة. تح: فايز محمّد، ص: 35.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"العَفْلاءُ مِنَ الشِّفاهِ: التي تَنْقَلِبُ عِنْدَ الضَّحِكِ. والرَّجُلُ أَعْفَلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 52.— المصدر
2.3معنى من معاني «عَفِلَ»

وــــــــــالكَبْشُ: كَثُرَ شَحْمُ خُصْيَتَيْهِ مِنَ السِّمَنِ؛ فَهو أَعْفَلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَيْرُ بْنُ أَفْصى الأَسْلَمِيُّ (ت: 50هـ=669م):

"كَبْشٌ حَوْلِيٌّ أَعْفَلُ".

المديني، المجموع المغيث. تح: العزباوي، ج: 2، ص: 473.— المصدر
03 —

عَفَّلَ

فعل
3.1معنى من معاني «عَفَّلَ»

* عَفَّلَ فُلانٌ المَرْأةَ أوِ النّاقَةَ: عالَجَ عَفَلَها بِالكَيِّ وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"عَفِلَتِ المَرْأَةُ والنّاقَةُ، وهِيَ عَفْلاءُ (...) والتَّعْفيلُ: إصْلاحُها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 52.— المصدر
3.2معنى من معاني «عَفَّلَ»

وــــــــــالعَملَ: أَصْلَحَهُ بِرِفْقٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والتَّعْفيلُ: إصْلاحُها. ويُسْتَعارُ فَيُقالُ: هو يُعَفِّلُ العَمَلَ: إذا كانَ رَفيقًا".

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 52.— المصدر
3.3معنى من معاني «عَفَّلَ»

وــــــــــالشَّيْءَ إِلى كَذا: نَسَبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"التَّعْفيلُ النِّسْبَةُ إلَيْهِ، يُقالُ: عَفَّلَهُ بِهِ: إذا نَسَبَهُ إلَيْهِ".

الزبيدي، تاج العروس. تح: حجازي، ج: 30، ص: 15.— المصدر

عافِل

اسم
4.1معنى من معاني «عافِل»

* العافِلُ: مَنْ يَلْبَسُ الثِّيابَ القِصارَ فَوْقَ الطِّوالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأعْرابِيِّ (ت: 231هـ=845م):

"العافِلُ: مَنْ يَلْبَسُ الثِّيابَ القِصارَ فَوْقَ الطِّوالِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: حجازي، ج: 30، ص: 15.— المصدر
05 —

عَفالِ

اسم
5.1معنى من معاني «عَفالِ»

* عَفالِ (كَقَطامِ): شَتْمٌ لِلمَرْأةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الصَّغانِيُّ (ت: 650هـ=1252م):

"عَفالِ: شَتْمٌ، يُقالُ لِلأَمَةِ: يا عَفالِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: حجازي، ج: 30، ص: 16.— المصدر
5.2معنى من معاني «عَفالِ»

o ابْدَئيهِنَّ بِعَفالِ سُبِيتِ: مَثَلٌ يُضْرَبُ لِمَنْ يُعَيِّرُ صاحِبَهُ بِعَيْبٍ هُوَ فيهِ. (انْظُرْ: ب د أ)

06 —

عُفال

اسم
6.1معنى من معاني «عُفال»

* العُفالُ: الدّاهِيَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"العُفالُ: الدّاهِيَةُ، يُقالُ ابْدَئيهِمْ بِعُفالِ سُبِيتِ".

أبو عمرو،الجيم. تح: الطحاوي، ج: 2، ص: 319.— المصدر
07 —

عَفْل / عَفَل

اسم
7.1معنى من معاني «عَفْل / عَفَل»

* العَفْلُ، والعَفَلُ: كَثْرَةُ شَحْمِ ما بَيْنَ رِجْلَيِ التَّيْسِ والثَّوْرِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ بِشْرُ بنُ أبي خازِمٍ (ت: 32ق هـ=591م) ييَهْجو عُتْبَةَ بْنَ جَعْفَرِ ويُشَبِّهُهُ بِالكَبْشِ السَّمينِ:

دَعا مُعْتِبًا جارَ الثُّبورِ وغَرَّهُ
أَجَمُّ خَدورٌ يَتْبَعُ الضَّأْنَ جَيْدَرُ

جَزيزُ القَفا شَبْعانُ يَرْبِضُ حَجْرَةً
حَديثُ الخِصاءِ وارِمُ العَفْلِ مُعْبَرُ

ديوان ابن أبي خازم. تح:طراد، ص: 72.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ سيدَهْ (ت: 458هـ=1066م):

"العَفَلُ: كَثْرَةُ شَحْمِ ما بَيْنَ رِجْلَيِ التَّيْسِ والثَّوْرِ، ولا يَكادُ يُسْتَعْمَلُ إلّا في الخَصِيِّ مِنْها، ولا يُسْتَعْمَلُ في الأُنْثى". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 163.— المصدر
7.2معنى من معاني «عَفْل / عَفَل»

وــــــــــ: مَجَسُّ الشاةِ بَيْنَ رِجْلَيْها؛ لِيُنْظَرَ سِمَنُها مِنْ هُزالِها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ خِداشُ بْنُ زُهَيْرٍ (ت: 30ق هـ=592م) يَهْجو عَبْدَ اللهِ بْنَ جُدْعانَ:

أَغَرَّكَ أَنْ كانَتْ لِبَطْنِكَ عُكْنَةٌ
وأنَّكَ مَكْفِيٌّ بِمَكَّةَ طاعِمُ

وتَرْضى بِأنْ يُهْدى لَكَ العَفْلُ مُصْلَحًا
وتَحْنَقُ أنْ تُجْنى عَلَيْكَ العَظائِمُ

شعر خداش بن زهير. تح: الجبوري، ص: 95.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسائِيُّ (ت: 189هـ=805م):

"العَفْلُ: المَوْضِعُ الذي يُجَسُّ مِنَ الشّاةِ إذا أرادوا أَنْ يَعْرِفوا سِمَنَها مِنْ غَيْرِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، الغريب المصنف. تح: داوودي، ج: 1، ص: 165.— المصدر
”وقالَ أعْرابِيٌّ (216ه=831م، مِنْ شَواهِدِ الأصْمَعِيِّ) يَهْزَأُ بِصاحِبِهِ:

"اشْتَرِ لي شاةً قَفْعاءَ،كَأَنَّها تَضْحَكُ، مُنْدَلِقَةً خاصِرتاها كَأَنَّها في مَحْمِلٍ، لَها ضَرْعٌ أَرْقَطُ كَأَنَّهُ ضَبٌّ. قالَ: فَكَيْفَ العَفْلُ؟ قالَ: أَوَلِهَذِهِ عَفْلٌ؟!".

الجاحظ،الحيوان. تح: عيون السود، ص: 388.— المصدر
7.3معنى من معاني «عَفْل / عَفَل»

وــــــــــ: شَحْمُ خُصْيَتَيِ الكَبْشِ وما حَوْلَهُ.

قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”وبِهِ فُسِّرَ قَوْلُ خِداشِ بْنِ زُهَيْرٍ (ت: 30ق هـ=592م) يَهْجو عَبْدَ اللهِ بْنَ جُدْعانَ:

أَغَرَّكَ أَنْ كانَتْ لِبَطْنِكَ عُكْنَةٌ
وأَنَّكَ مَكْفِيٌّ بِمَكَّةَ طاعِمُ

وتَرْضى بِأَنْ يُهْدى لَكَ العَفْلُ مُصْلَحًا
وتَحْنَقُ أَنْ تُجْنى عَلَيْكَ العَظائِمُ

ابن قتيبة، الشعر والشعراء. تح: شاكر، ج: 1، ص: 646.— المصدر

عَفَل

اسم
8.1معنى من معاني «عَفَل»

* العَفَلُ: شَيْءٌ يَخْرُجُ مِنْ قُبُلِ النِّساءِ، وحَياءِ النّاقَةِ، ودُبُرِ الرَّجُلِ شِبْهُ الأُدْرَةِ في النّاسِ تَحْتَ الصَّفَنِ. وهي بِتاءٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ اللَّعينُ المِنْقَرِيُّ (ت: 75هـ=695م) يُعَرِّضُ بِرَهْطِ العَجّاجِ:

ما في الدَّوائِرِ مِنْ رِجْلَيَّ مِنْ عَقَلٍ

عِنْدَ الرِّهانِ وما أُكْوى مِنَ العَفَلِ

المفضل الضبي، أمثال العرب. تح: إحسان عباس، ص: 77.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"عَفِلَتِ المَرْأَةُ عَفَلًا؛ فَهي عَفْلاءُ. والعَفَلُ، والعَفَلَةُ الاسْمُ؛ وهو شَيْءٌ يَخْرُجُ في حَياءِ النّاقَةِ شِبْهُ أَدَرَةٍ".

الخليل،العين. تح: المخزومي وآخر، ج: 2، ص: 145.— المصدر
”وقال مُحَمَّدُ بْنُ الحَسَنِ الشَّيبانيُّ (ت: 189هـ=804م) يُعَدِّدُ العُيوبَ في البُيوعِ:

"والحَبَلُ في الجارِيَةِ عَيْبٌ، والحَوَلُ عَيْبٌ، والقَرَنُ عَيْبٌ، والعَفَلُ عَيْبٌ، والبَرَصُ عَيْبٌ، والجُذامُ عَيْبٌ".

محمد بن الحسن،المبسوط. تح: الأفغاني، ج: 5، ص: 176.— المصدر
