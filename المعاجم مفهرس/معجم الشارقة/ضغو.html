المعانى الكلّيّة

  1. ضَرْبٌ مِنَ الأَصْواتِ
01 —

ضَغا

فعل
1.1معنى من معاني «ضَغا»

* ضَغا فُلانٌ وغَيْرُهُ يَضْغو ضَغْوًا، وضُغُوًّا، وضُغاءً: صَوَّتَ وصاحَ. فهُوَ ضاغٍ، وهِيَ ضاغِيَةٌ. (ج) ضَواغٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيْدٍ (ت: 35ق.هـ=588م):

لَمْ يَبْقَ فيهِ إِلّا مَراوِحُ طايا
تٍ وبورٌ تَضْغو ثَعالِبُها

الزمخشري، الفائق. تح: البجاوي، وآخر، ج: 3، ص: 417.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ سَوْدَةُ بِنْتُ زَمْعَةَ العامِرِيَّةُ (11هـ=632م) تُجيبُ النَّبِيَّ ﷺ مُعْتَذِرَةً:

"واللَّهِ يا نَبِيَّ اللَّهِ، ما يَمْنَعُني مِنْكَ أَنْ لا تَكونَ أَحَبَّ البَرِيَّةِ إِلَيَّ، ولَكِنّي أُكْرِمُكَ أَنْ يَضْغُوَ هَؤُلاءِ الصِّبْيَةُ عِنْدَ رَأْسِكَ بُكْرَةً وعَشِيَّةً".

مسند أحمد. تح: الأرناؤوط، وآخرين، ج: 5، ص: 92.— المصدر
”وقالَ الرّاعي النُّميرِيُّ (ت: 90هـ=709م) يَصِفُ كَيْفَ صَرَعَ الثَّوْرُ الوَحْشِيُّ الكِلابَ الَّتي هاجَمَتْهُ:

حَتّى إِذا عَرَّدَتْ عَنْهُ سَوابِقُها
وعانَقَ المَوْتَ مِنْها سَبْعَةٌ عَدَدُ

مِنْها صَريعٌ وضاغٍ فَوْقَ حَرْبَتِهِ
كَما ضَغا تَحْتَ حَدِّ العامِلِ الصُّرَدُ

[الصُّرَدُ: طائِرٌ ضَخْمُ الرَّأْسِ يَصْطادُ العَصافيرَ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجاحِظُ (ت: 255هـ=869م) يَذْكُرُ عَيْبَ البَغْلَةِ:

"وهِيَ في ذَلِكَ أَهْيَجُ مِنْ هِرَّةٍ، وإِنْ كانَتْ لَا تَصيحُ صِياحَها، ولَا تَضْغو ضُغاءَها".

الجاحظ، القول في البغال. تح: شارل بيلا، ص: 97.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الحاجِّ النُّمَيْرِيُّ (ت: 768هـ=1366م) يَمْدَحُ الغَنيَّ بالله محمَّدَ بْنَ يوسُفَ:

هُمامٌ إِذا اصْطَفَّ العَدُوُّ فَدونَهُ
تَلاحٍ بِأَطْرافِ القَنا وَتَلاغِ

شَديدُ قُوًى لَمْ يَضْغَ يَوْمًا وإِنَّهُ
لَهُ كُلُّ مَلْكٍ في المَعاقِلِ ضاغِ

ديوان إبراهيم بن الحاج النميري. تح: عبد الحميد الهرامة، ص: 182.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ القادِرِ المازِنِيُّ (1368هـ=1949م) يَصِفُ فَتاةً تَسْتَنْجِدُ:

"وتُحاوِلُ أَنْ تَصيحَ طَلَبًا لِلنَّجْدَةِ فَيُخْرِسُها الماءُ الَّذي يَمْلَأُ فَمَها، وتومِئُ فَلا يَراها أَحَدٌ، ومَنْ ذا الَّذي يُغيثُ في هَذا الخِضَمِّ الضَّاغي".

عبد القادر المازني، إبراهيم الكاتب. ص: 80.— المصدر
1.2معنى من معاني «ضَغا»

وــــــــــ فُلانٌ: صَرَخَ مِنْ ذُلٍّ أَوْ جوعٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ اللهِ بْنُ الزُّبَيْرِ الأَسَدِيُّ (66هـ=685م) يَهْجو عَبْدَ الرَّحْمنِ بْنَ أُمِّ الحَكَمِ:

صَغيرًا ضَغا في خِرْقَةٍ فَأَمَضَّهُ
مُرَبِّيهِ حَتَّى إِذْ أَهَمَّ وأَفْطَما

[أَمَضَّ فُلانًا: آلَمَهُ وشَقَّ عَلَيْهِ]
شعر عبد الله بن الزبير الأسدي. تح: يحيى الجبوري، ص: 129.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ صَخْرُ بْنُ الجَعْدِ (ت: 140هـ=758م) يَتَحَدَّثُ عَنْ تاجِرٍ استَدانَ مِنْهُ:

أَهْوِنْ عَلَيَّ بِسَيّارٍ وضَغْوَتِهِ
إِذا جَعَلْتُ صِرارًا دونَ سَيّارِ

يُسائِلُ النّاسَ هَلْ أَحْسَسْتُمُ جَلَبًا
مِنْ نَحْوِ يَثْرِبَ أَوْ مِنْ نَحْوِ أَظْفارِ

عبد العزيز محمّد الفيصل، شعراء بني عُقيل. ج: 2، ص: 228.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أَبو السُّعودِ (ت: 951هـ=1544م) يَصِفُ كُفّارَ قُرَيْشٍ الّذينَ ادَّعَوْا عَلى النَّبِيِّ ﷺ الجُنونَ:

"وما مَثَلُهُمْ إِلّا كَمَثَلِ الكَلْبِ إِذا جاعَ ضَغا وإِذا شَبِعَ طَغَى".

تفسير أبي السعود. ج: 8، ص: 61.— المصدر
1.3معنى من معاني «ضَغا»

وــــــــــ المَقْهورُ ضَغْوًا: تَذَلَّلَ، وانْقادَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110هـ=728م) يَهْجو عَيَّاشَ بْنَ الزِّبْرِقانِ:

ضَغا القِرْدُ لَمَّا مَسَّهُ الجَهْدُ، واشْتَكى
بَنو الْقَيْنِ مِنَّا حَدَّ نابٍ وكَلْكَلِ

محمد الصاوي، شرح ديوان جرير. ص: 459.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والضَّغْوُ: الاسْتِخْذاءُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 3، ص: 20.— المصدر
”وقالَ الجاحِظُ (ت: 255هـ=869م) يَتَحَدَّثُ عَنْ حَسَدِ مُدَّعي العِلْمِ:

"وحَمَلَ المُدَّعِيَةَ لِلعِلْمِ المزَوَّرِ الحَسَدُ عَلى بَهْتِ العُلَماءِ المُحِقّينَ وعَضْهِهِمْ والطَّعْنِ عَلَيْهِمْ؛ وجَرَّأَهُمْ عَلى ذلِكَ ما رَأَوْا مِنْ ضَغْوِ ضَعَفَةِ القُلوبِ، وَأَذِلَّةِ النَّاسِ إِلَيْهِمْ".

مجموع رسائل الجاحظ. تح: محمد الحاجري، ص: 162.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَحْمود شاكِر (1372هـ=1953م) تاريخُ الطَّبْعِ، يُخاطِبُ الأُسْتاذَ الزَّياتَ:

"فلَمّا دَعَوْتَني فأَجَبْتُ، انْقَلَبْتُ أُسائِلُ نَفْسي: فيمَ أَكْتُبُ؟ (...) إِنَّ بَيْني وبَيْنَ الأَسْماعِ والأَبْصارِ والقُلوبِ، حِجابًا صاخِبًا مِنْ غَمائِمِ الدَّجاجِلَةِ، وهَمائِمِ الأَفَّاكينَ، وثُغاءِ أَهْلِ الغِشِّ وضُغاءِ أَخْدانِ النِّفاقِ".

مجلة الرسالة، العدد: 1018، ص: 11.— المصدر
1.4معنى من معاني «ضَغا»

وــــــــــ المُقامِرُ ضَغْوًا: غَشَّ ولَمْ يَعْدِلْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو القاسِمِ الخَفّافُ (ت: 384هـ=994م) يَرْوي حيلَةَ أَحَدِ اللُّصوصِ:

"أَنْتَ لَمّا قَمَرْتُكَ الآنَ قُماشَكَ، أَخَذْتَ تَدَّعي عَلَيَّ اللُّصوصِيَّةَ؟ يا غَثُّ، يا بارِدُ، بَيْني وبَيْنَكَ دارُ القِمارِ، الموْضِعُ الَّذي تَعارَفْنا فيهِ، قُلْ بِحِذائِهِمْ وبِحِذاءِ هَؤُلَاءِ الحاضِرينَ، أَنَّكَ قَدْ ضَغَوْتَ حَتَّى أَدَعَ عَلَيْكَ قُماشَكَ".

القاضي التنوخي، نشوار المحاضرة. تح: عبود الشالجي، ج: 1، ص: 157.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"لا تَصْبِنْ لا تَصْبِنْ فَإِنَّهُ طَرَفٌ مِنَ الضَّغْوِ".

ابن منظور، اللسان. تح: عبد الوهاب، وآخر، ج: 7، ص: 282.— المصدر
1.5معنى من معاني «ضَغا»

o وضَغا مِنِّي وَهُوَ ضَغّاءٌ: مَثَلٌ يُضْرَبُ لِكُلِّ مَنْ عَجَزَ عَنْ شَيْءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَتِ العَرَبُ (290هـ=903م) مِنْ شَواهِدِ المُفَضَّلِ الضَّبِيِّ:

"ضَغا مِنِّي وهْوَ ضَغَّاءٌ".

المفضل الضبي، الفاخر في الأمثال. تح: محمد عثمان، ص: 238.— المصدر
02 —

أَضْغَى

فعل
2.1معنى من معاني «أَضْغَى»

* أَضْغَى فُلانٌ فُلانًا: أَرْغَمَهُ عَلى الضُّغاءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الضُّغاءُ: صَوْتُ الذّليلِ إِذا شَقَّ عَلَيْهِ، يُقالُ: ضَغَا يَضْغُو، وأَضْغَيْتُهُ أَنا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 3، ص: 20.— المصدر
03 —

ضَغَّى

فعل
3.1معنى من معاني «ضَغَّى»

* ضَغّىفُلانٌ فُلانًا: حَمَلَهُ عَلَى البُكاءِ حَدَّ الصِّياحِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"ضَغَا يَضْغُو ضُغُوًّا، وأَضْغَاه هُوَ، وضَغَّاهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 34.— المصدر
04 —

تَضاغَى

فعل
4.1معنى من معاني «تَضاغَى»

* تَضاغَىالصِّبْيانُ: تَباكَوْا.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتِ النَّوّارُ امْرأَةُ حاتِمٍ الطّائِيِّ (ت: 46ق.هـ=577م) تَصِفُ حَالَ أَوْلادِها زَمَنَ القَحْطِ:

"أَصابَتْنا سَنَةٌ اقْشَعَرَّتْ لَها الأَرْضُ واغْبَرَّتِ الآفاقُ (...) فَوَاللَّهِ إِنَّا لَفي صُنْبُرٍ (لَيْلَةٍ شَديدَةِ البُرودَةِ) بَعيدَةٍ ما بَيْنَ الطَّرَفَيْنِ إِذا تَضاغَى أَصْبِيَتُنا مِنَ الجُوعِ: عَبْدُ اللهِ وعُدَيٌّ وسِفّانَةُ، فقامَ حَاتِمٌ إِلى الصَّبِيَّيْنِ وقُمْتُ إِلى الصَّبِيَّةِ، فوَاللهِ ما سَكَتوا إِلّا بَعْدَ هَدْأَةٍ مِنَ اللَّيْلِ".

محمد بن حبيب البغدادي، كنى الشعراء. تح: سيد كسروي، ص: 59.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كِنانَةُ بْنُ صُوَيْراءَ (4هـ=626م) يُخاطِبُ أصحابَهُ يَوْمَ غَزْوَةِ بَني النَّضِير:

"قَدْ أُخْبِرَ مُحَمّدٌ مَا هَمَمْتُمْ بِهِ مِنَ الغَدْرِ، فَلا تَخْدَعوا أَنْفُسَكُمْ، واللهِ إنّهُ لَرَسُولُ اللهِ (...) وما يَأْتيكُمْ بِهِ أَوْلى مِنْ مُحارَبَتِهِ إِيّاكُمْ، ولَكَأَنِّي أَنْظُرُ إِلَيْكُمْ ظاعِنينَ، يَتَضاغى صِبْيانُكُمْ، قَدْ تَرَكْتُمْ دورَكُمْ خُلوفًا وأَمْوالَكُمْ، وإِنَّما هِيَ شَرَفُكُمْ".

[الظّاعِنانِ: راحِلانِ هَرَبًا مِنْ دِياركِمُ]ْ
الواقدي، المغازي. تح: مارسدن جونس، ج: 1، ص: 365.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ حَبيبُ بْنُ زَيْدٍ القُشَيْرِيُّ (ت: 175هـ=792م) يَصِفُ صِبْيَةً جائِعينَ:

أَرانِيَ مِنْ جُمْلٍ كَراجي مُخيلَةٍ
حَياهَا لِقَوْمٍ نازِحينَ حُروبُ

أَبى صِبْيَة تَأْوي عَلَيْهِ شَوارِفُ
خَشَوْكَ كِراءٍ كُلُّهُنَّ شَسوبُ

إِذا ما تَضاغوا فابتَغى الدَّرَّ لَمْ يَكُنْ
لَهُمْ بِأَناضِيَّ الشَّقِيِّ حَلوبُ

المستدرك في شعر بني عامر. تح: عبد الرحمن الوصيفي، ج:2، ص: 319.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَهْدي الجَمّ الطَّرابُلْسِيُّ (1353هـ=1934م) تاريخُ الطَّبْعِ، يَقُصُّ:

"حاصَرَ أُورِلُ بِجَيْشِهِ تَدْمُرَ، وطالَ الحِصارُ فَمُلَّ، وَنَفِدَتْ مُؤَنُ المَدينَةِ، وتَضاغَى النَّاسُ جوعًا، فَجَمَعَتْ زَيْنَبُ المَجْلِسَ الحَرْبِيَّ، فَلَقِيَها القادَةُ باكينَ، فقالَتْ: لِمَ البُكاءُ أَيُّها الأَبْطالُ؟ إِمّا المَوْتُ وإِمّا الحَياةُ".

مجلة الرسالة، العدد: 56، ص: 1278.— المصدر
4.2معنى من معاني «تَضاغَى»

وــــــــــ الثَّريدَةُ: تَراجَعَتْ مِنَ الدَّسَمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244هـ=858م):

05 —

ضاغِيَة

اسم
5.1معنى من معاني «ضاغِيَة»

* الضَّاغِيَةُ: الصَّائِحَةُ. (ج) ضَواغٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ قَتادَةُ بْنُ النُّعْمانِ (ت: 23هـ=644م) يَتَحَدَّثُ عَنْ نِهايَةِ قَوْمِ لوطٍ:

"وبَلَغَنا أَنَّ جِبْريلَ أَخَذَ بِعُرْوَةِ القَرْيَةِ الوُسْطى، ثُمَّ أَلْوى بِها إِلى السَّماءِ حَتّى أَسْمَعَ أَهْلَ السَّماءِ ضَواغيَ كِلابِهِمْ، ثُمَّ دَمْدَمَ بَعْضَها عَلى بَعْضٍ فَجَعَلَ عَالِيَها سافِلَها، ثُمَّ تَبِعَتْهُمُ الحِجارَةُ".

[دَمْدَمَ: أَهْلَكَ]
تفسير عبد الرزاق. تح: محمود عبده، ج: 2، ص: 192.— المصدر

06 —

ضُغاء

اسم
6.1معنى من معاني «ضُغاء»

* الضُّغاءُ: صَوْتُ الثَّعْلَبِ ونَحْوِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو حُذَيْفَةَ بْنُ عُتْبَةَ (ت: 12هـ=633م) يَصِفُ ما آلَ إِلَيْهِ قَوْمُ لوطٍ بِسَبَبِ أَفْعالِهِمْ:

"رَفَعَ جِبْريلُ عَلَيْهِ الصَّلَاةُ والسَّلامُ أَرْضَ قَوْمِ لوطٍ، فَأَلْوى بِها، حَتَّى سَمِعَ أَهْلُ السَّماءِ ضُغاءَ كِلابِهِمْ".

أبو موسى المديني، المجموع المغيث. تح: العزباوي، ج: 3، ص: 160.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الضُّغاءُ: ‏صَوْتُ الثَّعْلَبِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 4، ص: 431.— المصدر

وــــــــــ: صَوْتُ الذَّليلِ إِذا شَقَّ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الضُّغاءُ: صَوْتُ الذَّليلِ إِذا شَقَّ عَلَيْهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 4، ص: 431.— المصدر
”وقالَ مالِكُ بْنُ أَنَسٍ (ت: 179هـ=795م) يَسْتَدِلُّ عَلى جَوازِ نِكاحِ الخَصِيِّ بِجارٍ لِعُمَرَ بْنِ الخَطّابِ:

"فَكَانَ عُمَرُ يَسْمَعُ صَوْتَ امْرَأَتِهِ وضُغاءَها مِنْ زَوْجِها هَذا الخَصِيِّ".
