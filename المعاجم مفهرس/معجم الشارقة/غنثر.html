غنثر
‌غَنْثَرَ(فعل) تَغَنْثَرَ(فعل) غَنْثَر/ غُنْثَر/ غُنْثُر(اسم) غَنْثَرة(اسم) ‌مُغَنْثِر(اسم)
01 —

‌غَنْثَرَ

فعل
1.1معنى من معاني «‌غَنْثَرَ»

* غَنْثَرَ فُلانٌ: شَرِبَ الماءَ عَنْ غَيْرِ عَطَشٍ. (ولَمْ يُسْتَعْمَلْ هَذا الفِعْلُ بِهَذا المَعْنَى مِنْ قَبْلُ، واسْتُعْمِلَ المَصْدَرُ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ الخَطَّابِيُّ (ت: 388هـ=998م):

"‌الغَنْثَرَةُ: شُرْبُ الماءِ عَنْ غَيْرِ عَطَشٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخطّابي، غريب الحديث، تح: الغرباوي وآخر، ج: 2، ص: 7.— المصدر
02 —

تَغَنْثَرَ

فعل
2.1معنى من معاني «تَغَنْثَرَ»

* ‌تغَنْثَرَ فُلانٌ بالماءِ: شَرِبَهُ عَنْ غَيْرِ شَهْوَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"يُقالُ: ‌تغَنْثَرَ بالماءِ: إِذا شَرِبَهُ عَنْ غَيْرِ شَهْوَةٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 2، ص: 1131.— المصدر
03 —

غَنْثَر/ غُنْثَر/ غُنْثُر

اسم
3.1معنى من معاني «غَنْثَر/ غُنْثَر/ غُنْثُر»

* الغَنْثَرُ، والغُنْثَرُ، والغُنْثُرُ مِنَ النّاسِ: الجاهِلُ الأَحْمَقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قَالَ أَبُو بَكْرٍ الصّدِّيقُ (11هـ=632م) يُنَادِي ابْنَهُ:

"يا غَنْثَرُ، أَقْسَمْتُ عَلَيْكَ إِنْ كُنْتَ تَسْمَعُ صَوْتِي لَمَا جِئْتَ".

صحيح البخاريّ، تح: العطّار، ص: 1553.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ الخَطّابِيُّ (ت: 388هـ=998م):

"فَأَمّا ‌الغُنْثَرُ -بالغَيْنِ الـمُعْجَمَةِ-: فَهوَ مَأْخوذٌ مِنَ الغَثارَةِ؛ وهِيَ الجَهْلُ، يُقالُ: رَجُلٌ أغَثَرُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخطّابي، أعلام الحديث، تح: آل سعود، ج: 1، ص: 455.— المصدر
3.2معنى من معاني «غَنْثَر/ غُنْثَر/ غُنْثُر»

وــــــــــ: الثَّقيلُ الوَخِمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ أَبي بَكْرٍ الصِّدِّيقِ (11هـ=632م) يُنادِي ابْنَهُ:

"يا ‌غُنْثَرُ، أَقْسَمْتُ عَلَيْكَ إِنْ كُنْتَ تَسْمَعُ صَوْتِي لَمَا جِئْتَ".

صحيح البخاريّ، تح: العطّار، ص: 1553.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبُو عُبَيْدٍ الهَرَوِيُّ (ت: 401هـ=1011م) بَعْدَ أنْ ذَكَرَ قَوْلَ أبي بَكْرٍ الصِّدِّيقِ:

"‌غُنْثَرُ: أَحْسِبُهُ الثَّقِيلَ الوَخِمَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الهروي، الغريبين، تح: المزيدي، ج: 4، ص: 1390.— المصدر
3.3معنى من معاني «غَنْثَر/ غُنْثَر/ غُنْثُر»

وــــــــــ: السَّفِيهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ أَبي بَكْرٍ الصِّدِّيقِ (11هـ=632م) يُنادِي ابْنَهُ:

"يا غَنْثَرُ، أَقْسَمْتُ عَلَيْكَ إِنْ كُنْتَ تَسْمَعُ صَوْتِي لَمَا جِئْتَ".

صحيح البخاريّ، تح: العطّار، ص: 1553.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ القَاضِي عِياضٌ (ت: 544هـ=1149م):

"الغَنْثَرُ: سُفهاءُ النّاسِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

عياض، إكمال المعلم، تح: إسماعيل، ج: 6، ص: 550.— المصدر
3.4معنى من معاني «غَنْثَر/ غُنْثَر/ غُنْثُر»

وــــــــــ: اللَّئِيمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ أَبي بَكْرٍ الصِّدِّيقِ (11هـ=632م) يُنادِي ابْنَهُ:

"يا غَنْثَرُ، أَقْسَمْتُ عَلَيْكَ إِنْ كُنْتَ تَسْمَعُ صَوْتِي لَمَا جِئْتَ".

صحيح البخاريّ، تح: العطّار، ص: 1553.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ القَاضِي عِياضٌ (ت: 544هـ=1149م):

"الغَنْثَرُ: ... مَأْخوذٌ مِنَ الغَثْرِ، وَهُوَ اللُّؤْمُ والسُّقوطُ، كأَنَّهُ قَالَ لهُ: يا لَئِيمُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

عياض، إكمال المعلم، تح: إسماعيل، ج: 6، ص: 550.— المصدر
3.5معنى من معاني «غَنْثَر/ غُنْثَر/ غُنْثُر»

وــــــــ: الذُّبابُ الأَزْرَقُ، وَهُوَ بتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"‌الغَنْثَرُ: ذُبابٌ أَزْرَقُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

عياض، إكمال المعلم، تح: إسماعيل، ج: 6، ص: 550.— المصدر
04 —

غَنْثَرة

اسم
4.1معنى من معاني «غَنْثَرة»

* الغَنْثَرَةُ: كَثْرَةُ شَعْرِ الرَّأْسِ وكَثافَتُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ ابْنُ عَبّادٍ (385هـ=995م):

"الغَنْثَرَةُ: ضُفُوُّ الرَّأْسِ، وكَثْرَةُ الشَّعَرِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط، تح: آل ياسين، ج: 5، ص: 172.— المصدر
05 —

‌مُغَنْثِر

اسم
5.1معنى من معاني «‌مُغَنْثِر»

* المُغَنْثِرُ الرَّأْسِ مِنَ الصِّبيانِ: مَنْ كانَتْ لَهُ شُعَيْفَةٌ حافَّةٌ؛ أَيْ: خُصْلَةٌ مِنَ الشَّعْرِ في أَعْلى الرَّأْسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ ابْنُ عَبّادٍ (385هـ=995م):

"صَبِيٌّ ‌مُغَنْثِرُ الرَّأْسِ: وهُوَ أَنْ تَكونَ لَهُ شُعَيْفَةٌ حافَّةٌ".

ابن عباد، المحيط، تح: آل ياسين، ج: 5، ص: 172.— المصدر
