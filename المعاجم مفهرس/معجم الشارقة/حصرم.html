01 —

حَصْرَمَ

فعل
1.1معنى من معاني «حَصْرَمَ»

*حَصْرَمَ الرَّجُلُ: بَخِلَ وقَلَّ خَيْرُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"ورَجُلٌ مُحَصْرَمٌ: قَليلُ الخَيْرِ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 331 .— المصدر
1.2معنى من معاني «حَصْرَمَ»

و ــــــــ العِنَبُ ونحْوُهُ: كَبُرَ ولَمْ ينضُجْ بَعْدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَلي الهَجَرِيُّ (ت: 300ه=912م):

"وقد حَصْرَمَ العِنَبُ: وآرَقَ، بالألِفِ، وإرْقاقُ العِنَبِ عِنْدَ طُلوعِ الثُّريّا، وذَلكَ أنَّهُ (...) إذا بَدا طيبُهُ".

أبو علي الهجري، التعليقات والنوادر. تح: حمد الجاسر، ج: 3، ص: 1040.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ أيبكَ الدَّوادارِيّ (ت: 736ه= 1337م):

"والعِنَبُ الأبْيَضُ أوَّلُ ما ‌حَصْرَمَ وعَقَدَ (...) ثُمَّ تَرِقُّ بَشْرَتُهُ ويَحْلو مَذاقُهُ".

ابن أيبك الداواداري، كنز الدرر وجامع الغرر. تح: برند راتكه، ج: 1، ص: 316.— المصدر
1.3معنى من معاني «حَصْرَمَ»

وــــــــ النَّسَبُ: اخْتَلَطَ وتَداخَلَ. (ولم يُستعْمَل هذا الفعلُ بهذا المعْنى مِنْ قبْلُ، واستُعْمِلَ المشتقُّ منه)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مِسْحَلٍ الأعْرابيُّ (ت: 230ه=845م):

"يُقالُ: رَجُلٌ ‌مُحَصْرَمُ النَّسَبِ: إذا كان مَدْخولًا".

أبو مسحل الأعرابي، كتاب النوادر. تح: عزة حسن، ص: 385.— المصدر
1.4معنى من معاني «حَصْرَمَ»

وــــــــ الرَّجُلُ في الكَلامِ: لَحَنَ. (ولَمْ يُسْتَعْمَلْ هذا الفِعْلُ بهذا المَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ المُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"‌والحَصْرَمةُ: اللَّحْنُ في الكَلامِ وإفْسادُهُ. كَلامٌ مُحَصْرَمٌ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1141.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الجَزَري، مُحَمَّدُ بْنُ مُحَمّد (ت: 833ه=1430م) يُعيبُ على بَعْضِ القُرّاءِ نُطْقَ الرّاءِ:

"قَدْ يُبالِغُ قَوْمٌ في إخْفاءِ تَكْريرِها مُشَدَّدةً، فَيَأتي بِها مُحَصْرَمَةً شَبيهَةً بِالطّاءِ، وذَلِكَ خَطَأٌ لا يَجوزُ".

ابن الجزري، النشر في القراءات العشر. تح: الضباع، ج: 1، ص: 219.— المصدر
1.5معنى من معاني «حَصْرَمَ»

وــــــــ فُلانٌ الحَبْلَ ونَحْوَهُ: أحْكَمَ فَتَلَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرِو بْنُ العَلاءِ (ت: 154ه=771م):

"الحَصْرَمَةُ: الشُّحُّ. وهُوَ شِدَّةُ إغارةِ الوَتَرِ والحَبْلِ أيْضًا، أَيْ: فَتْلِهِ".

ابن السكيت، كتاب الألفاظ. تح: قباوة، ص: 49.— المصدر
1.6معنى من معاني «حَصْرَمَ»

وــــــــ القَوْسَ: شَدَّ وَتْرَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرِو بْنُ العَلاءِ (ت: 154ه=771م):

"يُقالُ قَدْ حَصْرَمَ قَوْسَهُ: إذا شَدَّ وَتَرَها".

ابن السكيت، كتاب الألفاظ. تح: قباوة، ص: 49.— المصدر
1.7معنى من معاني «حَصْرَمَ»

وــــــــ: الشّيْءَ: ضَيَّقَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216ه=831م):

"حَصْرَمْتُ القِرْبةَ: إذا مَلَأْتَها حَتّى تَضيقَ وكُلُّ مُضَيَّقٍ مُحَصْرَمٌ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

الأزهري، نهذيب اللغة. تح: مرعب، ج: 5، ص: 209.— المصدر
1.8معنى من معاني «حَصْرَمَ»

وــــــــ: الإناءَ ونَحْوَهُ: مَلأهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ الدِّينَوَرِيُّ (ت: 282ه=895م):

"حَصْرَمَ الإناءَ: مَلَأهُ".(وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

ابن سيده، المحكم والمحيط الأعظم. تح: هنداوي، ج: 4، ص: 63.— المصدر
1.9معنى من معاني «حَصْرَمَ»

وــــــــ: القلَمَ: بَراهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَهْ (ت: 458ه=1066م):

"وحَصْرَمَ القَلَمَ: بَراهُ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

ابن سيده، المحكم والمحيط الأعظم. تح: هنداوي، ج: 4، ص: 63.— المصدر
1.10معنى من معاني «حَصْرَمَ»

حَصْرَمَالقارِئُ حَرْفَ الرّاءِ: بالَغَ في شَدِّ حَرْفِ الرّاءِ حَتّى تُشابِهَ الطّاء.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الجَزَرِيِّ (ت: 833هـ=1429م):

"وقَدْ يُبالِغُ قَوْمٌ في إخْفاءِ تَكْريرِها مُشَدّدَةً، فيَأْتي بِها مُحَصْرَمَةً شَبيهَةً بالطّاءِ".

ابن الجزري، النشر. تح: الضبّاع، ج: 1، ص: 219.— المصدر
02 —

تَحَصْرَمَ

فعل
2.1معنى من معاني «تَحَصْرَمَ»

* تَحَصْرَمَ الثَّمَرُ: انعَقَدَ ولمْ يَنْضُجْ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ البوصيرِيُّ (ت: 696ه=1297م):

كَأنَّ ثَناياهُمْ مِنَ الخَبَثِ الّذي
تحَصْرَمَ في نِيّاتِهِمْ وتَزَبَّبا

ديوان البوصيري. تح: أحمد بسج، ص: 50.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ جبران مسعود (1413هـ =1992م تاريخُ الطَّبْعِ):

"تَحَصْرَمَ الشَّيْءُ: صارَ حِصْرِمًا".

جبران مسعود، معجم الرائد اللغوي، ج: 1، ص: 548.— المصدر
2.2معنى من معاني «تَحَصْرَمَ»

وــــــــ الزُّبْدُ: تَفَرَّقَ في شِدَّةِ البَرْدِ فَلَمْ يَجْتَمِعْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَهْ (ت: 458ه=1066م):

"تَحَصْرَمَ الزُّبْدُ: تَفَرَّقَ في شِدّةِ البَرْدِ، فَلمْ يَجْتَمِعْ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

ابن سيده، المحكم والمحيط الأعظم. تح: هنداوي، ج: 4، ص: 63.— المصدر
2.3معنى من معاني «تَحَصْرَمَ»

وــــــــ فُلانٌ: بَخِلَ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفِيروزابادي (ت: 817ه=1414م):

"الرجُلُ البَخِيلُ: ‌المُتَحَصْرِمُ". (وهَذِهِ العِبارَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزآبادي، القاموس المحيط. تح: العرقسوسي، ص 1094.— المصدر
03 —

حِصْرِم

اسم
3.1معنى من معاني «حِصْرِم»

* الحِصْرِمُ: الثَّمَرُ قبلَ النُّضْجِ. واحِدَتُهُ بتاءٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَخْطَلُ (ت: 90ه=708م) يَمْدَحُ عَبْدَ اللهِ بْنَ مُعاوِيةَ بْنِ أَبي سُفْيانَ، ويَصِفُ خَمرًا:

وتَفَصَّدَتْ مِنْ غَيْرِ هَشٍّ عودُهُ
بالٍ ولَيسَ بِحِصْرِمٍ أَبْكارِ

شعر الأخطل. تح: قباوة، ص: 284.— المصدر
”

وقالَ: ابْنُ سيرينَ (ت: 110ه=728م) يُبَيِّنُ الأَصْنافَ التي يَجوزُ فيها بَيْعُ السَّلَمِ: "كانَ شُرَيْحٌ يُسْلِمُ عَلى الحِصْرِمِ".

ابن شيبة، المصنف. تح: الحوت، ج: ٥، ص: ٩.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الـمُعْتَزِّ (ت: 296ه=909م) يَهْجو امْرَأةً:

كَأنَّكَ إذْ جِئْتَها سائِلًا
تُقَطِّرُ في عَينِها حِصْرِمَةْ

ديوان ابن المعتز. تح: محمد شريف، ج: 2، ص: 463.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُحَمَّدُ بْنُ عَبْدِ اللهِ التمرتاشي (ت: 1004=1596م) يَبَيِّنُ حُكْمَ بُسْتانِيٍّ غَفَلَ فَتَلِفَتِ الكُرومُ والحيطانُ:

"يَضْمَنُ الكُرومَ لا الحيطانَ، ولَوْ فيهِ حِصْرِمٌ ‌ضَمِنَ ‌الحِصْرِمَ ‌لا ‌العِنَبَ".

ابن عابدين، حاشية ردّ المحتار . تح: عادل عبد الموجود وآخر، ج: 9، ص: 411.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ التّسولِيُّ (ت: 1258ه=1842م) يَشْرَحُ بَعْضَ شروطِ بَيْعِ الثّمَرِ قَبْلَ النُّضْجِ:

"يَجوزُ بِشُروطٍ ثَلاثَةٍ: أَنْ يُنْتَفَعَ بِهِ ‌في ‌الحينِ، ‌كالحِصْرَمِ والفولِ الأخْضَرِ والفَريكِ".

التسولي، البهجة في شرح التحفة. تح: محمد شاهين، ج: 2، ص: 50.— المصدر
3.2معنى من معاني «حِصْرِم»

وــــــــــ مِنَ الرِّجالِ: البَخيلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَنْظورُ بْنُ مِرْثَدٍ الأسَدِيُّ (ت: 60ه= 681م) يفتخِرُ بالجودِ، ويتَرفَّعُ عنِ البُخلِ:

تُعَيِّرُني الحِظْلانَ أُمُّ مُحَلِّمٍ
فَقُلْتُ لَها لَمْ تَقْذِفيني بِدائِيا

فإنّي رَأيتُ الصّامِرينَ مَتاعُهُمْ
يُذَمُّ ويَفْنى فَارْضَخي مِنْ وِعائيا

فلَنْ تَجِديني في المَعيشةِ عاجِزًا
ولا حِصْرِمًا خَبًّا شَديدًا وِكائِيا

[الحِظْلانُ: المَنْعُ. والصّامِرون: البُخلاء. وارْضَخي، أي: فَرِّقي]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفارابِيُّ (ت: 350ه=961م):

"والحِصْرِمُ: البَخيلُ أيْضًا". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

الفارابي، معجم ديوان الأدب. تح: أحمد عمر، ج: 2، ص: 52.— المصدر
3.3معنى من معاني «حِصْرِم»

وــــــــ: القَصيرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"والحِصْرِمُ: القَصيرُ مِنَ الرِّجالِ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 3، ص: 269.— المصدر
3.4معنى من معاني «حِصْرِم»

وــــــــ: الضَيِّقُ الخُلُقِ.

ع

3.5معنى من معاني «حِصْرِم»

وــــــــ: ثَمْرَةُ شَجَرِ الـمَظِّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"والحِصْرِمُ: جَناةُ شَجَرٍ يُقالُ لَهُ ‌المَظُّ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيّةِ العُصورِ)

[الـمَظُّ: الرُمّانُ البَرِّيُّ]

3.6معنى من معاني «حِصْرِم»

وــــــــ: القِشْرُ الأَعْلى لِثَمَرةِ الدَّوْمَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"ثَمَرُ الدَّوْمَةِ: الفِرْصُ ما دامَ أحْمَرَ، فَإذا احْلَوْلى فَهُوَ الفَضيخُ، فإذا يَبِسَ فَهُوَ البَهْشُ، والحِصْرِمُ قِشْرُهُ الأعْلى وهُوَ القِرْفُ".

أبو عمرو الشيباني، كتاب الجيم. تح: الطحاوي، ج: 2، ص: 278.— المصدر
3.7معنى من معاني «حِصْرِم»
