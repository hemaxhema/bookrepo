زقزق
زَقْزَقَ(فعل) زَقْزاقٌ(اسم) زَقْزاقَة(اسم) زَقْزاقِيّات(اسم) زَقْزَق(اسم) زِقْزِق(اسم) زَقْزقَة(اسم) زُقْزوق(اسم) مُزَقْزَق(اسم)
المقدّمة
النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ز ق ز ق):

في السُّرْيانيّة: وَرَدَ الفِعْلُ: ܙܰܩܙܶܩ (زَقزِق) zaqzeq بِمَعْنى (صَوَّتَ زق زق) ويُوافِقُ الفِعْلَ العَرَبيَّ (زَقْزَقَ) مَبنًى ومَعنًى.

(ܠܟܣܝܩܘܢ ܕܐܘܓܝܢ ܡܢܢܐ: ܙܩܙܩ)

المعانى الكلّيّة

  1. صَوْتُ العُصْفورِ
  2. خِفَّةُ الحَرَكَةِ
  3. نَوْعٌ مِنَ الطُّيورِ
01 —

زَقْزَقَ

فعل
1.1معنى من معاني «زَقْزَقَ»

* زَقْزَقَ فُلانٌ زَقْزَقَةً وزِقْزاقًا: ضَحِكَ ضَحِكًا ضَعيفًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"‌وزَقْزَقَ الرَّجُلُ: ضَحِكَ ضَحِكًا ضَعيفًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 193.— المصدر
1.2معنى من معاني «زَقْزَقَ»

وــــــــــ الطّائِرُ: صَوَّتَ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ نَجْمُ الدّينِ الغَزِّيُّ (ت: 1061هـ=1651م) يُعَدِّدُ أمْثِلَةً مِنَ تَشَبُّهِ بَعْضِ البَشَرِ بالبَهائِمِ:

"ومِنْها: التَّشَبُّهُ -في العَجَلَةِ، والطَّيْشِ، وعَدَمِ التَّرَوّي في العِبارَةِ؛ خُصوصًا في المُناظَرَةِ- بالعُصْفورِ ونَحْوِهِ، في سُرْعَةِ تَقَلُّبِهِ وزَقْزَقَتِهِ".

نجم الدين الغزي، حسن التنبُّه. تح: طالب، ج: 10، ص: 500.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَدُ فارِس الشِّدْياق (1270هـ=1855م) تاريخُ الطَّبْعِ، يَحْكي مُحاوَرَةً:

"ألَمْ أظْهَرْ لَكُمْ في طُلوعِ الشَّمْسِ وغُروبِها، وفي بُزوغِ الكَواكِبِ ومَغيبِها، وفي سُكونِ الرّيحِ وهُبوبِها (...) وفي الأَزْمانِ إذا تَوالَتْ، والأَحْوالِ إذا حالَتْ، والدُّوَلِ إذا دالَتْ، وفي الغِياضِ إذا أبْهَجَتْ، والرِّياضِ إذا دُبِّجَتْ، والأَشْجارِ إذا أَوْرَقَتْ وجُرِّدَتْ، والأَطْيارِ إذا ‌زَقْزَقَتْ وغَرَّدَتْ".

الشدياق،الساق على الساق. باريس، ص: 362.— المصدر
1.3معنى من معاني «زَقْزَقَ»

وــــــــــ الطّائرُ بسَلْحِهِ؛ أي: رَوْثِهِ: رَمى بِهِ. (انْظُرْ: ز ق ق)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيبانيُّ (ت: 206هـ=821م):

"والزَّقْزَقَةُ السَّلْحُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: الأبياري، ج: 2، ص: 61.— المصدر
1.4معنى من معاني «زَقْزَقَ»

وــــــــــ المَرْأَةُ الصَّبِيَّ: رَقَّصَتْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والزِّقزاقُ ‌والزَّقزَقةُ: تَرْقيصُ الأُمِّ وَلَدَها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 14.— المصدر
1.5معنى من معاني «زَقْزَقَ»

وــــــــــ الطّائِرُ فَرْخَهُ: أطْعَمَهُ بمِنْقارِهِ أو مَجَّ في فيهِ. (انْظُرْ: ز ق ق)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"زَقَّ الطّائِرُ فَرْخَهُ وزَقْزَقَهُ إذا مَجَّ في فيهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 201.— المصدر
02 —

زَقْزاقٌ

اسم
2.1معنى من معاني «زَقْزاقٌ»

* الزَّقْزاقُ: طائرٌ ساحِلِيٌّ مُهاجِرٌ مِنَ الفَصيلَةِ القِطْقاطِيَّةِ، حادُّ الصَّوْتِ، أصْغَرُ حَجْمًا مِنَ الحَمامَةِ بقَليلٍ، يَتَمَيَّزُ بطولِ أَجْنِحَتِهِ وأَرْجُلِهِ مَعَ قِصَرِ عُنُقِهِ، لَهُ عِدَّةُ أَلْوانٍ باخْتِلافِ نَوْعِهِ، ويَغْلِبُ عَلَيْهِ اللَّوْنُ البُنِّيُّ، ويُخالِطُهُ -في شَكْلِ طَوْقٍ أو بُقَعٍ أو رَقْطٍ- ألْوانٌ أخْرى كالأسْوَدِ والذَّهبِيِّ، ، يَتَغَذَّى على الحَشَراتِ والبُذورِ وبَعْضِ الكائناتِ البَحْرِيَّةِ الصَّغيرَةِ، وبَعْضُ أنْواعِهِ يَتَغَذَّى على بَقايا الطَّعامِ في فَمِ التِّمْساحِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ المَجيد مُصْطَفى خَليل (1356هـ=1937م) تاريخُ الطَّبْعِ:

"وقَدْ تَعْوي الذِّئابُ فتَنْبَحُها الكِلابُ، فكَأَنَّ أَصْواتَ الرّيحِ والكِلابِ والذِّئابِ صَوْتُ شَيْطانِ الظَّلامِ، وقَدْ يَصيحُ الزَّقْزاقُ وقَلَّما يُسْمَعُ غَيْرُهُ باللَّيْلِ".

عبد المجيد خليل، شاعريات. مطبعة مصر، ص: 41.— المصدر
03 —

زَقْزاقَة

اسم
3.1معنى من معاني «زَقْزاقَة»

* الزَّقْزاقَةُ مِنَ النِّساءِ: الخَفيفَةُ في المَشْيِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"والمَرْأَةُ الزَّقْزاقَةُ: الخَفيفَةُ في المَشْيِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 193.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَدُ فارِس الشِّدْياق (1270هـ=1855م) تاريخُ الطَّبْعِ، يَحْكي مُحاوَرَةً:

"قُلْتُ: وأَيُّ نَفْعٍ في العَرَجِ؟ قالَتِ: الرّاحَةُ مِنَ الجَرْيِ وَراءَ القِرْصافَةِ ‌الزَّقْزاقَةِ".

‌[القِرْصافَةُ: المَرْأَةُ الَّتي تَتَدَحْرَجُ كأَنَّها كُرَةٌ]— المصدر
3.2معنى من معاني «زَقْزاقَة»

وــــــــــ: طائرُ الزَّقْزاقِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيُّ (ت: 1205هـ=1791م):

"والزَّقْزاقَةُ -بالفَتْحِ- طائِرٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزبيدي، تاج العروس. دار الهداية، ج: 25، ص: 306.— المصدر
3.3معنى من معاني «زَقْزاقَة»

وــــــــــ: الرِّيحُ الشَّديدَةُ الَّتي لَها زَقْزَقَةٌ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد كُرْد عَليّ (1333هـ=1914م) تاريخُ النَّشْرِ، يَتَحَدَّثُ عَنْ أوْصافِ الرّيحِ:

"وإذا كانَتْ شَديدَةً فهِيَ العاصِفُ، فإذا كانَتْ شَديدَةً ولَها زَقْزَقَةٌ فهِيَ ‌الزَّقْزاقَةُ".

مجلة المقتبس، العدد: 87، 1914م، ج: 8، ص: 295.— المصدر
04 —

زَقْزاقِيّات

اسم
4.1معنى من معاني «زَقْزاقِيّات»

* الزَّقْزاقِيّاتُ: رُتْبَةٌ مِنَ الطُّيورِ، تَشْمَلُ خَمسينَ وثَلاثَمِئَةِ نَوْعٍ، تَنْتَشِرُ أفْرادُها في جَميعِ أنْحاءِ العالَمِ، يَعيشُ مُعْظَمُها بالقُرْبِ مِنَ المِياهِ، وتَتَغَذّى على اللّافَقارِيّاتِ أو الحَيَواناتِ الصَّغيرَةِ الأُخْرى، وبَعْضُ أنْواعِها يَعيشُ في الصَّحاري، وقِلَّةٌ مِنْها في الغاباتِ الكَثيفَةِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ إدْوار غالِب (1385هـ=1965م) تاريخُ الطَّبْعِ، يُعَرِّفُ بطائرِ الدُّرَّجَةِ:

"طائِرٌ طَويلُ السّاقِ مِنَ الزَّقْزاقِيّاتِ، وَحيدُ الجِنْسِ والنَّوْعِ".

غالب، الموسوعة في علوم الطبيعة. المطبعة الكاثوليكية، ج: 1، ص: 561.— المصدر

زَقْزَق

اسم
5.1معنى من معاني «زَقْزَق»

ع

ابن دريد، الاشتقاق. تح: هارون، ص: 547.— المصدر
06 —

زِقْزِق

اسم
6.1معنى من معاني «زِقْزِق»

* الزِّقْزِقُ: نَوْعٌ مِنَ النَّمْلِ صَغيرُ الحَجْمِ، صَدْرُهُ أقَلُّ سُمْرَةً مِنْ رَأْسِهِ وبَطْنِهِ، يَمْتازُ بنَشاطِهِ وسُرْعَةِ حَرَكَتِهِ.

ع

"والزِّقْزِقُ: ضَرْبٌ من النَّمْلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 193.— المصدر

زَقْزقَة

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الفَضْلِ المَيْدانيُّ (ت: 518هـ=1124م) يَذْكُرُ مَثَلًا لِمَنْ يُكْثِرُ الكَلامَ ويَفْخَرُ بما لَيْسَ عِنْدَهُ، ويأْتي بالباطِلِ:

"بَقْبَقَةٌ في زَقْزَقَةٍ؛ ‌البَقْبَقَةُ: الصَّخَبُ، ‌والزَّقْزَقَةُ: الضَّحِكُ. يُضْرَبُ للنَّفّاجِ الَّذي يَأْتي بالباطِلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الميداني، مجمع الأمثال. تح: عبد الحميد، ج: 1، ص: 104.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ناصِرُ الدّينِ الألبانيُّ (1412هـ=1992م) تاريخُ الطَّبْعِ، يَتَعَقَّبُ مَنْ صَحَّحَ حَديثًا نَبَوِيًّا يَرى أنَّهُ ضَعيفٌ:

"وكَلامُ المُناوِيِّ الَّذي اغْتَرَّ بِهِ ذاكَ الكَوْثَرِيُّ مِمّا لا طائِلَ تَحْتَهُ، بَلْ هُوَ ‌بَقْبَقَةٌ في زَقْزَقَةٍ؛ لأَنَّهُ قائِمٌ على مُجَرَّدِ التَّقْليدِ، الَّذي لَيْسَ فيهِ أَيُّ تَحْقيقٍ".

الألبانيّ، سلسلة الأحاديث الضعيفة. دار المعارف، ج: 6، ص: 139.— المصدر
08 —

زُقْزوق

اسم
8.1معنى من معاني «زُقْزوق»

* الزُّقْزوقُ: صِنْفٌ مِنَ سَمَكِ الشَّبوطِ، صَغيرُ الحَجْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ياقوتٌ الحَمَوِيُّ (ت: 626هـ=1229م) يُعَدِّدُ أنْواعَ السَّمَكِ في بُحَيْرَةِ تِنّيسَ:

"الأَشْبالُ، المَسّاكُ الأَبْيَضُ، ‌الزُّقْزوقُ، أُمُّ عُبَيْدٍ، السِّلَّوْرُ".

ياقوت، معجم البلدان. دار صادر، ج: 2، ص: 54.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحَمْد تَيْمور (ت: 1348هـ= 1930م):

"والكَركورُ: نَوْعٌ مِنَ السَّمَكِ، لَهُ شَوْكَتانِ في جانِبَيْهِ وشَوْكَةٌ في ظَهْرِهِ، يُشْبِهُ الزُّقْزوقَ".

تيمور، معجم تيمور الكبير. تح: نصار، ج: 5، ص: 216.— المصدر
8.2معنى من معاني «زُقْزوق»

وــــــــــ: طائرُ الزَّقْزاقِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيُّ (ت: 1205هـ=1791م):

"والزَّقْزاقَةُ -بالفَتْحِ- طائِرٌ، ‌كالزُّقْزوقِ، بالضَّمِّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزبيدي، تاج العروس. دار الهداية، ج: 25، ص: 306.— المصدر
09 —

مُزَقْزَق

اسم
9.1معنى من معاني «مُزَقْزَق»

* المُزَقْزَقُ: كُلُّ عَمَلٍ يُقْضى سَريعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"والمُزَقْزَقُ: كُلُّ عَمَلٍ يُقْضى على سُرْعَةٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 193.— المصدر
