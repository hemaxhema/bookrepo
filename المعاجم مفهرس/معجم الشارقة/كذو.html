01 —

أَكْذى

فعل
1.1معنى من معاني «أَكْذى»

* أَكْذى الشَّيْءُ: اِحْمَرَّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"أَكْذَى الشَّيْءُ إِذا احْمَرَّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج:10، ص: 336.— المصدر
1.2معنى من معاني «أَكْذى»

وــــــــــ فُلانٌ: اِحْمَرَّ لَوْنُهُ مِنْ خَجَلٍ أَوْ فَزَعٍ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"‌وأَكْذى الرَّجُلُ إِذا اِحْمَرَّ لَوْنُهُ مِنْ خَجَلٍ أَوْ فَزَعٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج:10، ص: 336.— المصدر
02 —

كاذي

اسم
2.1معنى من معاني «كاذي»

* الكاذي: دُهْنٌ عِطْرِيٌّ طَيِّبُ الرّائِحَةِ، يُصْنَعُ مِنْ زَهْرِ شَجَرِ الكاذي.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ عائِشَةُ أُمُّ المُؤْمِنينَ ڤ (ت: 58ه=678م):

"كانَ رَسولُ اللَّهِ ﷺ يَغْسِلُ رَأْسَهُ بِالسِّدْرِ ويَدَّهِنُ ‌بِالكاذي".

الأصبهاني، أخلاق النبي وآدابه. تح: الجميلي، ص: 150.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعِيُّ (ت: 204ه=819م):

"وما طُيِّبَ مِنَ الأَدْهانِ بِالرَّياحينِ فَبَقِيَ طَيِّبًا كانَ طَيِّبًا، وما رُبِّبَ بِها عِنْدي طَيِّبٌ إِذا بَقِيَ طَيِّبًا مِثْلَ: الزَّنْبَقِ والخَيْرِيِّ والكاذي والبانِ المَنْشوشِ﻿".

الشافعي، الأم. تح: النجار، ج:2، ص: 152.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الخَزْرَجِيُّ الزَّبيدِيُّ (ت: 812ه=1410م):

"ومِنْ أَنْواعِ الرَّياحينِ؛ كالفُلِّ والوَرْدِ والنَّرْجِسِ والياسَمينِ والمَنْثورِ والكاذي والأُتْرُجِّ والبَلَحِ وأَشْباهِ ذَلِكَ".

علي الخزرجي، العقود اللؤلؤية. تح: بسيوني، ج:2، ص: 233.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّد الأمينُ الشّنْقيطِيُّ (ت: 1393ه=1973م):

"وخِلافُ العُلَماءِ فيها مِنَ الخِلافِ في تَحْقيقِ المَناطِ؛ كَدُهْنِ البانِ، والزَّنْبَقِ وهُوَ دُهْنُ الياسَمينِ، والكاذي وهُوَ دُهْنٌ ونَبْتٌ طَيِّبُ الرّائِحَةِ (...) ومَذْهَبُ الشّافِعِيِّ أَنَّ الأَدْهانَ المَذْكورَةَ ونَحْوَها طيبٌ، تَجِبَ بِاسْتِعْمالِهِ الفِدْيَةُ".

الشنقيطي، أضواء البيان. دار عطاءات، ج:5، ص: 469.— المصدر
2.2معنى من معاني «كاذي»

وــــــــــ: شَجَرٌ ذو ورْدٍ أَحْمَرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُواسٍ (ت: 198ه=814م):

اِشْرَبْ عَلى الوَرْدِ في نَيْسانَ مُصْطَبِحًا
مِنْ خَمْرِ قُطْرَبُّلٍ حَمْراءَ ‌كالْكاذي

ديوان أبي نواس، تح: فاغنر، ج:3، ص: 125.— المصدر
2.3معنى من معاني «كاذي»

وــــــــــ: ضَرْبٌ من الشَّجَرِ يَضْرِبُ لَوْنُهُ إلى الصُّفْرَةِ، ولَهُ ريحٌ طَيِّبَةٌ، كانَتْ مُلوكُ الهِنْدِ والصّينِ تَتَكاتَبُ فيهِ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو العَبّاسِ الأحولُ (ت: 212ه=828م) يَصِفُ كِتابَ مَلِكِ الهِنْدِ إِلى المَأْمونِ:

"وكانَ الكِتابُ مَكْتوبًا في لِحاءِ شَجَرَةٍ تَنْبُتُ بِالهِنْدِ يُقالُ لَها: الكاذي، لَوْنُهُ إِلى الصُّفْرَةِ، والخَطُّ لازَوَرْدٌ مُفَتَّحٌ بِذَهَبٍ".

الكلبي، تاريخ خلفاء بني العباس. تح: العزاوي، ص: 52.— المصدر
2.4معنى من معاني «كاذي»

وــــــــــ مِنَ النّاسِ: الشَّديدُ الحُمْرَةِ مِنْ خَجَلٍ أَوْ غَيْرِهِ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"رَأَيْتُهُ كاذيا ماتِعًا أَسْلَغَ مُنْسَلِخًا: كُلُّهُ الشَّديدُ الحُمْرَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم، ج:8، ص: 37.— المصدر
2.5معنى من معاني «كاذي»

وــــــــــ: البَقَّمُ، وهُوَ شَجَرٌ يُسْتَخْرَجُ مِنْهُ صِبْغٌ أَحْمَرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"والكاذي والجِرْيالُ: البَقَّمُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج:10، ص: 336.— المصدر
2.6معنى من معاني «كاذي»

وــــــــــ: شَجَرٌ عِظامٌ مِنَ الفَصيلَةِ الكاذِيَّةِ، يُشْبِهُ النَّخْلَ، لِزَهْرِهِ رائِحَةٌ جَميلَةٌ، ويوضَعُ في الثِّيابِ فَتَطيبُ رائِحَتُها. يُصْنَعُ الدُّهْنُ مِنْ طَلْعِهِ، ويُطَيَّبُ بِوَرَقِهِ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ الدّينَوَرِيُّ (ت: 282ه=895م):

"ومِنَ الشَّجَرِ الَّذي يُطَيَّبُ بِهِ الدُّهْنُ الكاذي".

ابن سيده، المخصص. تح: جفال، ج:3، ص: 265.— المصدر
”وقالَ أَبو بَكْرٍ الرّازيُّ (ت: 313ه=925م):

"لِوَجَع اللِّثَةِ مِنْ تَذْكِرَةِ عَبْدوسْ: شَبُّ سُمّاقٍ، جَفْتُ بَلّوطٍ، وَرْدٌ بِأَقْماعِهِ جُلَّنارٌ، رامِكُ ثَمَرِ الطَّرَفا، وَرَقُ الحِنّاءِ المَكِّيِّ، عودُ الكاذي، حَبُّ الآسِ، إِهْليلَجٌ كابُليُّ، مُحْرَّقُ عَفْصٍ، أَقاقِيا؛ يُتَمَضْمَضُ بِهِ بِماءٍ".

الرازي، الحاوي في الطب. تح: طعيمي، ج:1، ص: 439.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوَوِيُّ (ت: 676ه=1277م):

"فَرْعٌ في أَنْواعٍ مِنَ النَّباتِ غَريبَةٌ، ذَكَرَها بَعْضُ الأَصْحابِ؛ مِنْها: الكاذي بِالذّالِ المُعْجَمَةِ".

النووي، المجموع شرح المهذب. تح: المطيعي، ج:7، ص: 291.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّد سُلَيْمانُ الطَّيِّبُ (1431ه=2010م):

"وتَقَعُ تِهامَةُ بَني عَمْرٍو غَرْبَ النِّماصِ، وتُشْتَهَرُ بِخُصوبَةِ أَوْدِيَتِها الثَّلاثَةِ وبِنَباتاتِها؛ ومِنْها: البُنُّ والكاذي والرَّياحينُ والوُرودُ والفَواكِهُ، ويُقَدَّرُ عَدَدُ سُكّانِ هَذِهِ القَبيلَةِ بِسَبْعينَ أَلْفِ نَسَمَةٍ".

محمد الطيب، موسوعة القبائل العربية، تح: المطر وآخر، ج:10، ص: 193.— المصدر

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

النسفي، طلبة الطلبة. تح: خالد العك، ص: 263.— المصدر
2.9معنى من معاني «كاذي»

وــــــــــ: ما يَتَّخِذُهُ راكِبُ السَّفينَةِ مِنَ الأَواني.

ع

السرخسي، المبسوط. مطبعة السعادة، ج:15، ص: 82.— المصدر

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

2.12معنى من معاني «كاذي»

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

النسفي، طلبة الطلبة. تح: خالد العك، ص: 263.— المصدر

03 —

كَذا

اسم
3.1معنى من معاني «كَذا»

* كَذا: كَلِمَةٌ واحِدَةٌ مُرَكَّبَةٌ مِنْ كَلِمَتَيْنِ؛ (كافِ التَّشبيهِ) و(ذا)، يُكْنَى بهَا عَنْ غَيْرِ عَدَدٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الرّبيعُ بنُ زِيادٍ (28ق.ه=595م) يُخاطِبُ النُّعْمانَ بْنَ المُنْذِرِ، ويَسُبُّ لَبيدَ بْنَ رَبيعَةَ:

"كَذَبَ واللَّهِ ابْنُ الفاعِلَةِ، ولَقَدْ فَعَلْتُ بِأُمِّهِ كَذا وكَذا".

الزوزني، شرح المعلقات. دار إحياء التراث، ج:1، ص: 158.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه=632م) يُخاطِبُ عُمَرَ وقَدْ أَرادَ قَتْلَ اليَهودِيِّ الَّذي أَغْلَظَ لِلنَّبِيِّ ﷺ:

«أَنْتَ إِلى غَيْرِ هَذا أَحْوَجُ؛ أَنْ تَأْمُرَهُ فَيُحْسِنَ طَلَبي، وتَأْمُرَني فَأُحْسِنَ قَضاءَهُ. انْطَلِقْ مَعَهُ إِلى حائِطِ كَذا وكَذا (...) فَادْخُلْهُ، فَقُلْ لِفُلانٍ يَكْشِفْ لَهُ عَنِ الطَّعامِ ليرِيَهُ إِيّاهُ، فَإِنْ رَضِيَهُ فَمُرْهُ فَلْيُوَفِّهِ مالَهُ».

سيرة ابن إسحاق. تح: زكار، ج:1، ص: 293.— المصدر

”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=759م) عَلى لِسانِ دِمْنَةَ يَتَحَدَّثُ عَنِ الأَسَدِ:

"هُوَ مَلِكُ السِّباعِ، وهُوَ بِمَكانِ كَذا، ومَعَهُ جُنْدٌ كَثيرٌ مِنْ جِنْسِهِ".

ابن المقفع، كليلة ودمنة. تح: مجموعة محققين، ص: 98.— المصدر

الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ ابنُ عَبْدِ السّلامِ (ت: 660ه=1262م) يُفَسِّرُ قَوْلَ اللهِ : ﵟٱلۡحَقُّ مِن رَّبِّكَ فَلَا تَكُونَنَّ مِنَ ‌ٱلۡمُمۡتَرِينَﵞ [البقرة: 147]:

"المُمْتَرينَ: الشّاكّينَ، خوطِبَ بِهِ والمُرادُ أُمَّتُهُ، امْتَرى بِكَذا: اِعْتَرَضَهُ اليَقينُ تارَةً والشَّكُّ أُخْرى يُدافِعُ أَحَدَهُما بِالآخَرِ".

تفسير العز بن عبد السلام. تح: الوهبي، ج:1، ص: 171.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَجيبةَ (ت: 1224ه=1809م) يُفَسِّرُ قَوْلَ اللهِ : ﴿وَمَآ أَنفَقۡتُم مِّن نَّفَقَةٍ أَوۡ نَذَرۡتُم مِّن نَّذۡرٖ فَإِنَّ ٱللَّهَ يَعۡلَمُهُۥۗ وَمَا لِلظَّٰلِمِينَ مِنۡ أَنصَارٍ﴾ [ البقرة: 270]:

"هُوَ إلزامُ المُكَلَّفِ نَفْسَهُ ما لَمْ يَجِبْ، كَقَوْلِهِ: للهِ عَليَّ أَنْ أَتَصَدَّقَ بِكَذا، أَوْ أُصَلِّيَ كَذا، أَوْ أَنْ أَصومَ كَذا، أَوْ إِنْ شَفَى اللهُ مَريضي فَعَليَّ كَذا".

ابن عجيبة، البحر المديد. تح: رسلان، ج:1، ص: 304.— المصدر
3.2معنى من معاني «كَذا»

وــــــــــ: كَلِمَةٌ واحِدَةٌ مُرَكَّبَةٌ مِنْ كَلِمَتَيْنِ؛ (كافِ التَّشبيهِ) و(ذا)، يُكَنّى بهَا عَنْ العَدَدِ، ويَكونُ تَمْييزُها مُفرَدًا مَنْصوبًا، ولا تَدْخُلُ عَلَيْها أَلْ.﻿

الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأبّارِ (ت: 658ه=1260م) يُتَرْجِمُ لِأَبي عَمْرٍو السَّرَقُسْطِيِّ:

"حَدَّثَ عَنْهُ ابْنُهُ أَبو بَكْرٍ مُحَمَّدُ بْنُ عَبْدِ الصَّمَدِ صاحِبِنا، وتوفِّيَ سَنَةَ ‌كَذا وسِتّمِائَةٍ".

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الطالِبُ البُرْتُلِيُّ (ت: 1219ه=1804م) يُتَرْجِمُ لِأَحَدِ العُلَماءِ:

"أَلَّفَ - رَحِمَهُ اللَّهُ تَعالى- مُقَدِّمَةً في نَحْوِ كَذا مِنْ قالَبِ الثُّمُنِ".

3.3معنى من معاني «كَذا»

o كَذا وكَذا (عِنْدَ الـمُحَدِّثِينْ): أَحَدُ أَلْفاظِ الجَرْحِ الـمَرْويَّةِ عَنِ الإِمامِ أَحْمَدَ بْنِ حَنْبَلٍ وغَيْرِهِ، ويُشارُ بِهِ إِلى أَنَّ الرّاوي فيهِ لِينٌ، أَوْ لَيْسَ ثِقَةً.
