  1. الالْتِواءُ
  2. الغِلَظُ

01 —

تَحَبْجَرَ

فعل
1.1معنى من معاني «تَحَبْجَرَ»

* تَحَبْجَرَتِ الأَمْعاءُ: الْتَوَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّاحِبُ بْنُ عَبّادٍ (ت: 385ه=995م):

"وبِهِ تَحَبْجُرٌ في البَطْنِ: وهُو التِواءٌ في الأَمْعاءِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّادٍ، المحيط في اللغة. تح: آل ياسين، ج: 3، ص: 260.— المصدر
02 —

اِحْبَجَرَّ

فعل
2.1معنى من معاني «اِحْبَجَرَّ»

* احْبَجَرَّ الشَّيْءُ: غَلُظَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المثَنَّى (ت: 209ه=825م):

"إِنَّما قالَ: الحِبَجْرُ: الغَليظُ، وقَدِ احْبَجَرَّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم والمحيط. تح: هنداوي، ج: 4، ص: 55.— المصدر
03 —

احْبَنْجَرَ

فعل
3.1معنى من معاني «احْبَنْجَرَ»

* احْبَنْجَرَ فُلانٌ: انْتَفَخَ غَضَبًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244ه=858م):

"ويُقالُ: احْبَنْجَرَ فُلانٌ: إذا انْتَفَخَ غَضَبًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السّكّيت، كتاب الألفاظ. تح: فخر الدين قباوة، ج: 1، ص: 56.— المصدر
04 —

حُباجِر

اسم
4.1معنى من معاني «حُباجِر»

* الحُباجِرُ: الغَليظُ مِنْ كَلِّ شَيْءٍ. (ج) حَباجِرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عُبَيْدٍ (ت: 224ه=838م):

"الحِبَجْرُ الوَتَرُ الغَليظُ، وهُو الحُباجِرُ".

الأزهريّ، تهذيب اللّغة. تح: عبد الله درويش، ج: 5، ص: 314.— المصدر
”وقالَ الشّاعِرُ (231ه=835م، مِنْ شَواهِدِ ابْنِ الأَعْرابيِّ) يَصِفُ الجَرادَ:

يُخْرِجُ مِنها ذَنَبًا حُباجِرا

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 44.— المصدر
4.2معنى من معاني «حُباجِر»

وــــــــــ: ذَكَرُ الحُبارى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه= 933م):

"حُبْجُرٌ وحُباجِرُ، وهُو ذَكَرُ الحُبارى". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللّغة. تح: رمزي بعلبكي، ص: 1112.— المصدر
4.3معنى من معاني «حُباجِر»

وــــــــــ: عَظيمُ البَطْنِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه= 933م):

"ورَجُلٌ حَبْجَرٌ: عَظيمُ البَطْنِ، وكَذَلِكَ حُباجِرُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللّغة. تح: رمزي بعلبكي، ص: 1112.— المصدر
05 —

حَبْجَر

اسم
5.1معنى من معاني «حَبْجَر»

* الحَبْجَرُ: العَظيمُ البَطْنِ، والـمُنْتَفِخُ الجَنْبَيْنِ. (ج) حَباجِرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه= 933م):

"رَجُلٌ حَبْجَرٌ: عَظيمُ البَطْنِ".

ابن دريد، جمهرة اللّغة. تح: رمزي بعلبكي، ص: 1112.— المصدر
5.2معنى من معاني «حَبْجَر»

وــــــــــ: الغَليظُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"مِمَّا جاءَ في صِفَةِ الأَوْتارِ: وَتَرٌ حُبْجُرٌ وحُباجِرُ وحَبْجَرٌ، وهُو أَغْلَظُها وأَبْقاها وأَصْوَبُها سَهْمًا، ويَمْلَأُ الفوقَيْنِ، والجَمْعُ: حَباجِرُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللّغة. تح: رمزي بعلبكي، ص: 1281.— المصدر
06 —

حُبْجُر

اسم
6.1معنى من معاني «حُبْجُر»

* الحُبْجُرُ: الضَّخْمُ البَطْنِ الذي خَرَجَتْ سُرَّتُهُ. (ج) حَباجِرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (206ه=821م، مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانيِّ) يُخاطِبُ رَجُلًا:

يا عَبْدَ عَبْدِ اللهِ يا بْنَ حَبْتَرِ

هَل لَكَ في بَكْرٍ رَضِيٍّ حُبْجُرِ

وقامَةٍ دَريرَةٍ ومِحْوَرِ

أبو عمرو الشّيبانيّ، الجيم. تح: الإبياري، ج: 1، ص: 211.— المصدر
6.2معنى من معاني «حُبْجُر»

وــــــــــ: ذَكَرُ الحُبارى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"حُبْجُرٌ وحُباجِرُ، وهُو ذَكَرُ الحُبارى". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللّغة. تح: رمزي بعلبكي، ص: 1112.— المصدر
07 —

حِبَجْر

اسم
7.1معنى من معاني «حِبَجْر»

* الحِبَجْرُ: الغَليظُ مِنَ الأَشْياءِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى الكَبيرُ (ت: 7ه=628م) يَصِفُ أَخْلاقَ قَوْمِهِ:

ونَشُدُّ عَقْدَ وَرِينا
عَقْدَ الحِبَجْرِ على الغِفارَهْ

[الغِفارَةُ: الجُلَيْدَةُ الَّتي تَكونُ على فَرْضَةِ القَوْسِ، الوَرِي: الجارُ الذي تُواريهِ بُيوتُنا]

ابن قتيبة، كتاب المعاني الكبير. ج: 1، ص: 1107.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانيُّ (ت: 206ه=821م):

"يُقالُ: سَوْطٌ حِبَجْرٌ: ضَخْمٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو الشّيبانيّ، الجيم. تح: الإبياري، ج: 1، ص: 191.— المصدر
”وقالَ الشّاعِرُ (224ه=838م، مِنْ شَواهِدِ أبي عُبَيْدٍ) يَصِفُ قَوْسًا:

أَرْمي عَلَيْها وهْيَ شَيْءٌ بُجُرُ

والقَوسُ فيها وَتَرٌ حِبَجْرُ

وهْيَ ثَلاثُ أَذْرُعٍ وشِبْرُ

الأزهريّ، تهذيب اللّغة. تح: عبد الله درويش، ج: 5، ص: 315 .— المصدر
