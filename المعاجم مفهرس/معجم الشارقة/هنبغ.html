هنبغ
هَنْبَغَ(فعل) هِنْباغ(اسم) هُنْبُغ(اسم) هُنْبُغ/ هِنْبِغ(اسم) هُنْبوغ(اسم) هَنَيْبَغ(اسم)
01 —

هَنْبَغَ

فعل
1.1معنى من معاني «هَنْبَغَ»

* هَنْبَغَ العَجاجُ: كَثُرَ وثارَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وهَنْبَغَ العَجاجُ: كَثُرَ وثارَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 103.— المصدر
1.2معنى من معاني «هَنْبَغَ»

وــــــــــ الرَّجُلُ ونَحْوُهُ: جاعَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّغّانيُّ (ت: 650ه=1252م):

"هَنْبَغَ: جاعَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، العباب الزاخر. تح: آل ياسين، ج: 1، ص: 90.— المصدر
02 —

هِنْباغ

اسم
2.1معنى من معاني «هِنْباغ»

* الهِنْباغُ مِنَ الجُوعِ: الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَمْرٍو الشَّيْبانيُّ (ت: 206ه=821م):

"جوعٌ هُنْبُغٌ ‌وهِنْباغٌ، وهَلَّقْسٌ، وهَلَّقْتٌ؛ أَيْ: شَديدٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 264.— المصدر
03 —

هُنْبُغ

اسم
3.1معنى من معاني «هُنْبُغ»

* الهُنْبُغُ: شِدَّةُ الجوعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ رُؤْبَةُ (ت: 145ه=762م) يَهْجو:

والعَبْدُ عَبْدُ الخُلُقِ المُدَغْدَغِ

كالفَقْعِ إنْ يُهْمَزْ بوَطْءٍ يُثْلَغِ

صاحِبُ سَوْءاتٍ وجوعٍ ‌هُنْبُغِ

[المُدَغْدَغُ: المَغْمُوزُ في نَسَبِهِ. الفَقْعُ: أَرْدَأُ ما يَكُونُ مِنَ الكَمَأةِ. يُثْلَغُ: يُخْدَشُ]
ديوان رؤبة بن العجاج. تح: البروسي، ص: 99.— المصدر
3.2معنى من معاني «هُنْبُغ»

وــــــــــ: العَجاجُ الَّذي يَطْفُو مِنْ رِقَّتِهِ ودِقَّتِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ رُؤْبَةُ (ت: 145ه=762م) يَصِفُ حِمارًا:

يَشْتَقُّ بعد الطَّرَدِ المُبَغْبِغِ

وبَعْدَ إيغافِ العَجاجِ ‌الهُنْبُغِ

نَدْفًا كإِيغَافِ الغُلَامِ المُرْتَغِي

[يَشْتَقُّ: يَشْتَدُّ. المُبَغْبِغ: القَرِيب. الإِيغَاف: أنْ يَأْخُذَ الغُلَامُ الرَّغْوَةَ فَيَرْفَعَها إِلَى فِيهِ. النَّدُفُ: حَسْرُ الرَّغْوَة. المُرْتَغِي: الذي يَحْسَرُ الرَّغْوَةَ]
ديوان رؤبة بن العجاج. تح: البروسي، ص: 98.— المصدر
3.3معنى من معاني «هُنْبُغ»

وــــــــــ: القَمْلَةُ الصَّغيرَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابيِّ (ت: 231ه=845م):

"يُقالُ لِلْقَمْلَةِ الصَّغيرَةِ: ‌الهُنْبُغُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 264.— المصدر
3.4معنى من معاني «هُنْبُغ»

وــــــــــ: المَرْأَةُ الضَّعيفَةُ البَطْشِ. (ج) هَنابِغُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والهُنْبُغُ: مِنْ أسْماءِ الأسَدِ. وهُوَ مِنَ النِّسَاءِ: ‌الضَّعيفةُ البَطْشِ، وجَمْعُها هَنابِغُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 103.— المصدر
3.5معنى من معاني «هُنْبُغ»

وــــــــــ: المَرْأَةُ الحَمْقاءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والهُنْبُغُ: مِن أسْماءِ الأسَدِ. وهُوَ مِنَ النِّساءِ: ‌الضَّعيفةُ البَطْشِ، وجَمْعُها هَنابِغُ. والمَرْأةُ الحَمْقاءُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 103.— المصدر
3.6معنى من معاني «هُنْبُغ»

وــــــــــ: الأَسَدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والهُنْبُغُ: مِنْ أسْماءِ الأسَدِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 103.— المصدر
04 —

هُنْبُغ/ هِنْبِغ

اسم
4.1معنى من معاني «هُنْبُغ/ هِنْبِغ»

* الهُنْبُغُ والهِنْبِغُ: المَرْأَةُ الفاجِرَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ كُراعُ النَّمْلِ (ت: 309ه=921م):

"والهُنْبُغُ: المَرْأَةُ ‌الفاجِرَةُ، والهِنْبِغُ لُغَةٌ فيهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 454.— المصدر
05 —

هُنْبوغ

اسم
5.1معنى من معاني «هُنْبوغ»

* الهُنْبوغُ: القَمْلَةُ الصَّغيرَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابيِّ (ت: 231ه=845م):

"يُقَالُ لِلقَمْلَةِ الصَّغيرَةِ: الهُنْبُغُ، والهُنْبوغُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 264.— المصدر
5.2معنى من معاني «هُنْبوغ»

وــــــــــ: شِبْهُ الطُّرْثوثِ يُؤْكَلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"‌والهُنْبوغُ: شِبْهُ الطُرْثوثِ يُؤْكَلُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[‌الطُّرْثوثُ: ضَرْبٌ مِنَ البَقْلِ، ويُقالُ مِنَ الشَّجَرِ]
ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 454.— المصدر
5.3معنى من معاني «هُنْبوغ»

وــــــــــ: طائِرٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"‌والهُنْبوغُ: طائِرٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 455.— المصدر
06 —

هَنَيْبَغ

اسم
6.1معنى من معاني «هَنَيْبَغ»

* الهَنَيْبَغُ مِنَ الرِّجالِ: الأَحْمَقُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ الزَّبيديُّ (ت: 1205ه=1790م):

"والهَنَيْبَغُ، كسَمَيْدَعٍ: الأَحْمَقُ ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزبيدي، تاج العروس. تح: حجازي، ج: 22، ص: 600.— المصدر
