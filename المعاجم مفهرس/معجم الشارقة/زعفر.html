01 —

زَعْفَرَ

فعل
1.1معنى من معاني «زَعْفَرَ»

* زَعْفَرَ فُلانٌ الثَّوْبَ وغَيْرَهُ: صَبَغَهُ بالزَّعْفَرانِ، وطَيَّبَهُ بِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ سَعدُ بنُ زيد مناة (ت: 202ق هـ=450م) يُعَرِّضُ بِأَخيهِ الذي انْشَغَلَ بِعِرْسِهِ عَنْ إِبِلِهِ:

يَظَلُّ يَوْمَ وِرْدِها مُزَعْفَرا

وهْيَ خَناطيلُ تَجوسُ الخُضَرا

ابن سلام الجُمَحي، طبقات فحول الشعراء. تح: طه إبراهيم، ص: 35.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابنُ عبّاس (ت: 68هـ=688م):

"انْطَلَقَ النَّبِيُّ ﷺ مِنَ الـمَدينَةِ بَعْدَما تَرَجَّلَ وادَّهَنَ، ولَبِسَ إِزارَهُ ورِداءَهُ هُوَ وأَصْحابُهُ، فَلَمْ يَنْهَ عَنْ شَيْءٍ مِنَ الأَرْدِيَةِ والأُزُرِ تُلْبَسُ إلّا الـمُزَعْفَرَةَ التي تَرْدَعُ عَلى الجِلْدِ".

صحيح البخاري. تح: العطار، ص: 369.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُواس (ت: 198هـ=814م) يَصِفُ طائِرًا:

كَأَنَّما زَعْفَرَهُ مُزَعْفِرُهُ
لا يُوئِلُ الأَبْغَثَ مِنْهُ حَذَرُهُ

ديوان أبي نُواس. تح: الحديثي، ص: 214.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوويُّ (ت: 676هـ=1277م) يَذْكُرُ كَفَنَ الـمَرْأَةِ:

"وأَمّا الـمُزَعْفَرُ والـمُعَصْفَرُ فَلا يَحْرُمُ تَكْفينُها فيهِ".

النووي، روضة الطّالبين. تح: زهير الشاويش، ج: 2، ص: 109.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عُثمان الحنبلي (ت: 1240هـ=1825م) يَذْكُرُ كَفَنَ الـمَيِّتِ:

"وكُرِهَ كَفَنٌ ‌مُزَعْفَرٌ، ومُعَصْفَرٌ ولَوْ لِامْرَأَةٍ؛ لِأَنَّهُ لا يَليقُ بِالحالِ".

عثمان الحنبلي، الفوائد المنتخبات. تح: ابن برجس وآخر، ج: 1، ص: 396.— المصدر
1.2معنى من معاني «زَعْفَرَ»

وــــــــــ الصَّدَأُ الحَديدَ: عَلاهُ. (ولم يُسْتَعْمَلْ هَذا الفِعْلُ بِهَذا المـَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ المـُشْتَقُّ مِنْهُ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أبي طالِبٍ (ت: 40هـ=661م) يُجِيبُ عِندَما سُئِلَ: هَلْ لِلْكِيمياءِ وُجُودٌ؟:

"لَعَمْري إنَّ لَهُ وُجُودًا، وَقَدْ كانَ، وَسَيَكُونُ، وَهُوَ كائِنٌ... إنَّ فِي الأَسْرابِ، وَالزّاجِ، والزِّئْبَقِ، والزُّجاجِ، والحَديدِ الـمُزَعْفَرِ، والزِّنْجارِ، والنُّحاسِ الأَخْضَرِ لَكُنوزًا، لا يُوقَفُ عَلى غابِرِهِنَّ".

رسائل جابر بن حيان. تح: أحمد المزيدي، ص: 90، 91.— المصدر
1.3معنى من معاني «زَعْفَرَ»

وــــــــــ الطَّعامَ ونَحْوَهُ: جَعَلَ فيهِ الزَّعْفَرانَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مالِكُ بنُ أنَس (ت: 179هـ=795م) يَذْكُرُ ما يُكْرَهُ لِلْمُحْرِمِ فِعْلُهُ:

"وَيُكْرَهُ لَهُ أَنْ يَشْرَبَ شَرابًا فِيهِ كافورٌ، وَأَنْ يَأْكُلَ مَرَقَةً مُزَعْفَرَةً؛ فَإِنْ فَعَلَ افْتَدَى".

البراذعي، التهذيب في اختصار المدونة. تح:محمد الأمين، ج: 1، ص: 605.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ الـمُلَقِّن (ت: 804هـ=1402م) يَذْكُرُ حُكْمَ اسْتِعْمالِ الطِّيبِ لِلْحاجِّ:

"وَعندَ المالِكِيَّةِ: أنَّهُ لا شَيءَ علَيهِ في أَكْلِ الخَبيصِ ‌الـمُزَعْفَرِ".

ابن الـمُلَقِّن، الإعلام بفوائد عمدة الأحكام. تح: عبد العزيز المشيقح، ج: 6، ص: 47.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بُطْرُسُ البُسْتانِيُّ (1285هـ=1870م) تاريخُ التَّأْليفِ:

"زَعْفَرَهُ: صَبَغَهُ بالزَّعْفَرانِ، والطَّعامَ: جَعَلَ فيهِ الزَّعْفَرانَ".

البستاني، محيط المحيط ج: 1، ص: 372.— المصدر
02 —

تَزَعْفَرَ

فعل
2.1معنى من معاني «تَزَعْفَرَ»

* تزَعْفَرَ فُلانٌ: تَطَيَّبَ بِالزَّعْفَرانِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أنسُ بنُ مالِك (ت: 93هـ=712م):

"نَهى نَبِيُّ اللهِ ﷺ أَنْ يَتَزَعْفَرَ الرَّجُلُ".

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 19، ص: 40.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعِيّ (ت: 204هـ=819م):

"ونَنْهى الرَّجُلَ حَلالًا بِكُلِّ حالٍ أَنْ ‌يَتَزَعْفَرَ، وَنَأْمُرُهُ إذَا ‌تَزَعْفرَ أَنْ يَغْسِلَ الزَّعْفَرانَ عَنْهُ".

البيهقي، معرفة السنن والآثار. تح: قلعجي، ج: 2، ص: 454.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النّوويُّ (ت: 676هـ=1277م):

"والرَّجُلُ مَنْهِيٌّ عَنِ التَّزَعْفُرِ في كُلِّ الأَحْوالِ".

النّووي، المجموع شرح المهذب. تح: محمد المطيعي، ج: 7، ص: 233.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشَّوكانيُّ (ت: 1250هـ=1834م):

"وقَدْ ثَبَتَ النَّهْيُ عَنْ ‌تَزَعْفُرِ الرَّجُلِ مُطْلَقًا مُحْرِمًا وَغَيْرَ مُحْرِمٍ".

الشّوكاني، نيل الأوطار. تح: محمد حلاق، ج: 9، ص: 182.— المصدر
2.2معنى من معاني «تَزَعْفَرَ»

وــــــــــ الشَّيْءُ: صارَ أَصْفَرَ مائِلًا إلى الحُمْرَةِ كَالزَّعْفَرانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البيرونيُّ (ت: 440هـ=1048م) يَذْكُرُ ما يُصيبُ لَبَنَ الخَشْخاشِ مِنَ التَّحَوُّلِ:

"ويُغْلى في جَوْفِهِ حَديدَةٌ حَتّى يَتَزَعْفَرَ لَوْنُهُ، ويَتَحَلَّلَ، ثُمَّ يُسْتَعْمَلَ".

البيروني، الصيدنة في الطب. تح: عباس زرياب، ص: 251.— المصدر
2.3معنى من معاني «تَزَعْفَرَ»

وــــــــــ الثَّوْبُ ونحوهُ: صُبِغَ بِالزَّعْفَرانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الأثير (ت: 606هـ=1210م):

"التَّزَعْفُرُ: ‏التَّلَطُّخُ بِالزَّعْفَرانِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن الأثير، الشافي. تح: أحمد بن سليمان، ج: 3، ص: 334.— المصدر
03 —

زَعْفَران

اسم
3.1معنى من معاني «زَعْفَران»

* الزَّعْفَرانُ: صِبْغٌ نَباتِيٌّ أَصْفَرُ مائِلٌ إلى الحُمْرَةِ يُتَطَيَّبُ بِهِ. (ج) زَعافيرُ، وزَعافِرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ جَبرُ الـمُعاوي (ت: 86ق هـ=538م) يَفْخَرُ بِبَسالَةِ قَوْمِهِ مُشَبِّهًا دِماءَ الأَعْداءِ بِالزَّعْفَرانِ:

وحَتّى تَرى النَّجْدَ البَسيلَ كَأَنَّما
يُضَرِّجُهُ بِالزَّعْفَرانِ مُضَرِّجُ

مقبل الأحمدي، شعراء مذحج؛ أخبارهم وأشعارهم، ص: 607.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ لَبيدُ بنُ رَبيعة (10هـ=631م) يَرْثي عَمَّه عامِرَ بنَ مالِكٍ مُلاعِبَ الأَسِنَّة:

وَفِتْيَةٍ كَالرَّسَلِ القِماحِ

باكَرْتَهُم بِحُلَلٍ وَراحِ

‌وزَعْفَرانٍ كَدَمِ الأَذْباحِ

ديوان لبيد بن ربيعة. تح: حنا الحتّي، ص: 60.— المصدر
”وقالَ النَّبِيّ ﷺ (ت: 11هـ=632م) يَذْكُرُ لِبْسَ الـمُحْرِمِ في الحَجِّ:

«ولا تَلْبَسوا مِنَ الثِيابِ شَيْئًا مَسَّهُ الزَّعْفَرانُ أَوْ وَرْسٌ».

صحيح البخاري. تح: العطار، ص: 368.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبدُ العَزيزِ الماجَشُون (ت: 164هـ=780م) يَذْكُرُ ثَوْبَ الـمُحْرِمِ في الحَجِّ:

"ومَنْ أَرادَ أَنْ يَلْبَسَ شَيْئًا مِمّا مَسَّهُ ‌الزَّعْفَرانُ، أو الوَرْسُ فَلْيَغْسِلْهُ حَتّى يَذْهَبَ لَوْنُهُ، ويَذْهَبَ ريحُهُ".

الماجشون، الحج. تح: ميكلوش موراني، ص: 191.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوويُّ (ت: 676هـ=1277م):

"ويُكْرَهُ لَطْخُ رَأْسِ الصَّبِيِّ بِدَمِ العَقيقَةِ، ولا بَأْسَ بِلَطْخِهِ ‌بِالزَّعْفَرانِ والخَلوقِ".

النووي، روضة الطّالبين. تح: زهير الشاويش، ج: 3، ص: 232.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الـمُظْهِري (ت: 1225هـ=1810م) يَذْكُرُ ما كانَ عَلَيْهِ مُشْركو قُرَيْشٍ:

"كانوا يَطْلونَ الأَصْنامَ بِالزَّعْفَرانِ ويَضَعونَ بَيْنَ يَدَيْها الطَّعامَ".

ثناء الله المظهري، التفسير المظهري. تح: غلام التونسي، ج: 6، ص: 349.— المصدر
3.2معنى من معاني «زَعْفَران»

وــــــــــ: نَباتٌ بَرِّيٌّ وزِراعيٌّ، صِبْغيٌّ طِبِّيٌّ، طَيِّبُ الرّائِحَةِ، زَهْرُهُ أَصْفَرُ مائِلٌ إلى الحُمْرَةِ، أَوْراقُهُ خَيْطِيَّةٌ. يُسْتَخْدَمُ لِتَطْييبِ بَعْضِ أَنْواعِ الأَطْعِمَةِ والحَلْوَياتِ، يُقَوّي الأَعْصابَ، ويُنَشِّطُ القَلْبَ.

ع

04 —

زَعْفَرانِيَّة

اسم
4.1معنى من معاني «زَعْفَرانِيَّة»

* الزَّعْفَرانِيَّةُ: فِرْقَةٌ مِنَ أَهْلِ الكَلامِ، وافَقتِ الـمُعْتَزِلَةَ والجَهْمِيَّةَ والقَدَرِيَّةَ في بَعْضِ الأُصولِ، وتَناقَضتْ في القَوْلِ بِخَلْقِ القُرْآنِ، ونِسْبَتُها تَعودُ إِلى الزَّعْفَرانيِّ أَحْمَدَ بْنِ مُحَمَّدِ بْنِ أَحْمَدَ الدَّلالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَبْدِ اللهِ البَشّاريُّ (ت: 380هـ=990م) يَذْكُرُ مذاهِبَ الـمُتَكَلِّمينَ ومَواطِنَهُم:

"أَمّا بالرَّيِّ فالغَلَبَةُ لِلحَنِيفِيّينَ؛ وهُمْ نَجّارِيَّةٌ إِلّا رَساتيقَ القَصَبَةِ فإِنَّهُمْ زَعْفَرانِيَّةٌ يَقِفونَ في خَلْقِ القُرْآنِ".

البَشّاريُّ ، أحسن التقاسيم، ص: 395.— المصدر

05 —

مُزَعْفَر

اسم
5.1معنى من معاني «مُزَعْفَر»

* الـمُزَعْفَرُ: الأَسَدُ الوَرْدُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو زُبيد الطائي (ت: 62هـ=682م):

فَأَبْصَرَ رَكْبًا رائِحينَ عَشِيَّةً
فَقالوا: أَبَغْلٌ مائِلُ الرِّجْلِ أَشْقَرُ

أَمِ اللَّيْثُ؟ فَاسْتَنْجوا وأَيْنَ نَجاؤكُمْ
فَهَذا ورَبِّ الرّاقِصاتِ الـمُزَعْفَرُ

رسائل الجاحظ. تح: هارون، ج: 2، ص: 311.— المصدر
ع

تَحِنُّ حَنينَ العودِ في نَغَماتِهِ
وتَزْأَرُ أَحْيانًا زَئيرَ الـمُزَعْفَرِ
