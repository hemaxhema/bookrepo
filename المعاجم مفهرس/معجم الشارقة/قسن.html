قسن
أَقْسَنَ(فعل) اقْسَانَّ(فعل) اقْسَأَنَّ(فعل) قَسانَة(اسم) قُسَأْنينَة(اسم) قَسَن(اسم) قِسْيَنّ(اسم) مُقْسَئِنّ(اسم)
النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذرُ (ق س ن):

في الآراميّة؛ ومنه الاسْمُ:קַשִּׁישׁ (قَشِّيش)qaššīš بمَعْنى "شيخ/ كَبير"؛ وهُوَ يَتَّفِقُ مع الاسْمِ العَربيِّ (قِسيَنّ) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الاسْمَ الآراميَّ خَضَعَ لإبْدالِ السِّين شينًا مضاعَفة، وإسْقاطِ النّون؛ ومن مَجيئِه بمَعْنى "قِسيَنّ" ما وَرَدَ في ترجوم أيوب: "יַעֲקֹבדְאִתְקְרֵידַרְדַקוְאַבְרָהָםדְאִתְקְרֵיקַשִׁישׁ"؛ أَيْ "يعقوبُ الذي يُدعى صغيرًا وإبراهيم الذي يُدعى قِسيَنّا".

(תר'، איוב: ג,יט)

المعانى الكلّيّة

  1. الشِّدَّةُ والصَّلابَةُ
01 —

أَقْسَنَ

فعل
1.1معنى من معاني «أَقْسَنَ»

* أَقْسَنَ فُلانٌ: صَلُبَ بَدَنُهُ عَلى العَمَلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"أقْسَنَ: إِذا صَلُبَ بَدَنُهُ عَلى العَمَلِ والسَّقْيِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 409.— المصدر
02 —

اقْسَانَّ

فعل
2.1معنى من معاني «اقْسَانَّ»

* اقْسانَّ فُلانٌ: كَبِرَ وهَرِمَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأَزْهَرِيُّ (ت: 370ه=981م) يَتَحَدَّثُ عَنِ الهَمْزَةِ في الفِعْلِ (اقْسَأَنَّ):

"هَذِه هَمْزَةٌ تُجْتَلَبُ كَراهَةَ جَمْعٍ بَينِ ساكِنَيْنِ، وكانَ في الأَصْلِ اقْسانّ ‌يَقْسانُّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 409.— المصدر
03 —

اقْسَأَنَّ

فعل
3.1معنى من معاني «اقْسَأَنَّ»

* اقْسَأَنَّ فُلانٌ: كَبِرَ وضَعُفَ وشاخَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ عَمْرُو بْنُ مَسْعودٍ الثَّقَفِيُّ (ت: 50ه=670م) يَشْكو حالَهُ لمُعاويَةَ بنِ أبي سُفْيانَ:

قَدْ قَنْسَرَتْهُ أُمورٌ ‌فاقْسَأَنَّ لَها
وقَدْ حَنى ظَهْرَهُ دَهْرٌ وقَدْ كَبِرا

[في المَصْدَرِ: قَسّرَتْهُ، والتَّصويبُ مِنْ بَقِيَّةِ المَصادِرِ]
ابن أبي الدنيا، مكارم الأخلام. تح: مجدي السيد إبراهيم، ج: 1، ص: 146.— المصدر
3.2معنى من معاني «اقْسَأَنَّ»

وــــــــــ اللَّيْلُ: اشْتَدَّتْ ظُلْمَتُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ العَجّاجُ (ت: 90ه=709م) يَصِفُ أَرَقَهُ في لَيْلَةٍ طَويلَةٍ:

ولَيْلَةٍ مِنَ اللَّيالي مَرَّتِ

بِتُّ لها يَقْظانَ واقْسَأَنَّتِ

إذا رَجَوْتُ أَنْ تُضيءَ اسْوَدَّتِ

دونَ قُدامى الصُّبْحِ فارْجَحَنَّتِ

[قُدامى الصُّبْحِ: أَوائِلُهُ. ارْجَحَنَّتْ: ثَقُلَتْ، فَهِيَ لا تَبْرَحُ]
ديوان العجاج. تح: عزة حسن، ص: 263.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"واقْسَأَنَّ اللَّيْلُ: اشْتَدَّتْ ظُلْمَتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 5، ص: 80.— المصدر
3.3معنى من معاني «اقْسَأَنَّ»

وــــــــــ فُلانٌ: كَبِرَ واشْتَدَّ عودُهُ وصَلُبَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الرّاجِزُ (170ه=787م) مِنْ شَواهِدِ الخَليلِ، يُخاطِبُ رَجُلًا:

إِنْ تَكُ لَدْنًا لَيِّنًا فَإِنّي

ما شِئْتَ مِنْ أَشْمَطَ ‌مُقْسَئنِّ

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 5، ص: 79.— المصدر
”

وقالَ الخَليلُ (ت: 170ه=787م):

"القِسْيَنُّ: الشَّيْخُ القَديمُ، وإذا اشْتَقّوا مِنَ القِسْيَنِّ فِعْلًا هَمَزوا فَقالوا: ‌اقْسَأَنَّ".

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 5، ص: 79.— المصدر
3.4معنى من معاني «اقْسَأَنَّ»

وــــــــــ العُودُ: صَلُبَ واشْتَدَّ وعَسا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ مَعْمَرُ بْنُ المُثَنّى (ت: 209ه=825م):

"القُسَأْنينَةُ: من ‌اقْسَأَنَّ العودُ وغَيْرُهُ، إذا اشْتَدَّ وعَسا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح في اللغة. تح: أحمد عطار، ج: 6، ص: 2182.— المصدر
3.5معنى من معاني «اقْسَأَنَّ»

وــــــــــ: يَبِسَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو زَيْدٍ الأَنْصارِيُّ (ت: 215ه=830م):

"‌اقْسَأَنَّ العودُ: إذ اشْتَدَّ، وعَسا، ويَبِسَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال. تح: حسين شرف، ج: 2، ص: 138.— المصدر
3.6معنى من معاني «اقْسَأَنَّ»

وــــــــــ فُلانٌ في العَمَلِ: مَضى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"‌واقْسَأَنَّ الرَّجُلُ: إذا مَضى ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 291.— المصدر
04 —

قَسانَة

اسم
4.1معنى من معاني «قَسانَة»

* القَسانَةُ: لَفْظٌ يَأْتي إتْباعًا لِلَفْظِ (الحَسانَةِ)، يُقالُ: إنَّه لَبَيِّنُ الحَسانَةِ والبَسانَةِ والقَسانَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو الطَّيِّبِ اللُّغَوِيُّ (ت: 351ه=962م) في ‌‌بابِ (الإتْباعِ الَّذي أَوَّلُهُ القافُ):

"يُقالُ: إنَّهُ لحَسَنٌ بَسَنٌ قَسَنٌ، وإنَّهُ لَبَيِّنُ الحُسْنِ ‌والبَسانةِ والقَسانَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[ذَكَرَ أبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230ه) في كِتابِهِ النّوادِرِ لَفْظَ (الحَسانَةِ) فَقالَ: وحَسَنٌ بَيِّنُ الحُسْنِ والحَسانَةِ]
أبو الطيب اللغوي، الإتباع. تح: التنوخي، ص: 71.— المصدر
05 —

قُسَأْنينَة

اسم
5.1معنى من معاني «قُسَأْنينَة»

* القُسَأْنينَةُ: الشِّدَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الفَرّاءُ النَّحْوِيُّ (ت: 207ه=822م):

"القُسَأْنينةُ: مِنِ اقْسأَنَّ العودُ: إذا اشْتَدَّ وعَسا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 409.— المصدر
06 —

قَسَن

اسم
6.1معنى من معاني «قَسَن»

* قَسَنٌ: لَفْظٌ يَأْتي إتْباعًا لِلَفْظِ (حَسَنٍ)، يُقالُ: حَسَنٌ بَسَنٌ ‌قَسَنٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الكِسائِيُّ (ت: 189ه=805م):

"ومِنَ الإتْباعِ قَوْلُهُمْ: حَسَنٌ ‌قَسَنٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو علي القالي، البارع في اللغة. تح: هشام الطعان، ص: 479.— المصدر
07 —

قِسْيَنّ

اسم
7.1معنى من معاني «قِسْيَنّ»

* القِسْيَنُّ مِنَ الشِّيوخِ: القَديمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"القِسْيَنُّ: الشَّيْخُ القَديمُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 5، ص: 79.— المصدر
7.2معنى من معاني «قِسْيَنّ»

وــــــــــ مِنَ الإبِلِ: الهَرِمُ المُسِنُّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الرّاجِزُ (170ه=787م) مِنْ شَواهِدِ الخَليلِ، يَصِفُ قَوْمًا:

وَهُمْ كَمِثْلِ ‌البازِلِ ‌القِسْيَنِّ

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 5، ص: 79.— المصدر
08 —

مُقْسَئِنّ

اسم
8.1معنى من معاني «مُقْسَئِنّ»

* المُقْسَئِنُّ مِنَ النّاسِ: مَنْ تَوَسَّطَ بينَ الشَّبابِ والكِبَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"والمُقْسَئنُّ: الَّذي قَدِ انْتَهى في سِنِّهِ فلَيْسَ بِهِ ضَعْفُ كِبَرٍ ولا قُوَّةُ شَبابٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 409.— المصدر
