زرفن
زَرْفَنَ(فعل) تَزَرْفَنَ(فعل) زُرْفَن، زِرْفَن(اسم) زُرْفين(اسم) زِرْفين(اسم)

المعانى الكلّيّة

  1. الإِبْزيمُ
  2. الحَلْقَةُ
  3. إِسْدالُ الشَّعْرِ عَلى هَيْئَةِ حَلَقاتٍ
01 —

زَرْفَنَ

فعل
1.1معنى من معاني «زَرْفَنَ»

* زَرْفَنَتْ فُلانَةٌ صُدْغَيْها: سَدَلَتْ عَلَيْهِ شَعْرَها في هَيْئَةِ حَلَقاتٍ. (مو)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نُواسٍ (ت: 198هـ=814م):

مُذَكَّرٌ حينَ يَبْدو
مُؤَنَّثُ الخَلَواتِ!

زَها عَلَيَّ بِصُدْغٍ
مُزَرْفَنِ الحَلَقاتِ!

النويري، نهاية الأرب. تح: قميحة وآخرين، ج: 2، ص: 226.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مارون عَبّود (1382هـ=1962م) تاريخُ الطَّبْعِ:

"وجاءَتِ الآنِسَةُ طابيتا، وقَدْ زَرْفَنَتْ شَعْرَها كَعادَتِها، فَأَكْبَرَ بَعْضُهُمْ وَقاحَتَها، وخُصوصًا زَكَرِيّا".

مارون عبود، أقزام جبابرة، مؤسسة هنداوي، ص: 46.— المصدر
1.2معنى من معاني «زَرْفَنَ»

وــــــــــ السَّرْجَ ونَحْوَهُ: جَعَلَ إبْزيمًا في طَرْفِ حِزامِهِ لِرَبْطِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجاحِظُ (ت: 255هـ=869م) يَصِفُ:

"وكانَ الـمُخارِقُ بْنُ عِفارٍ قَلِعًا، وكانَ خَفيفًا نَحيفًا، وضَئيلًا دَميمًا، وكانَ يُزَرْفِنُ سَرْجَهُ، وكانَ شُجاعًا بَطَلًا".

الجاحظ، البرصان والعرجان. تح: هارون، ص: 267.— المصدر
02 —

تَزَرْفَنَ

فعل
2.1معنى من معاني «تَزَرْفَنَ»

* تَزَرْفَنَ الشَّعْرُ: الْتَوى في شَكْلِ حَلَقاتٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البُرْقُعِيُّ (ت: 290هـ=903م) يَتَغَزَّلُ:

ولاحَظَني بِأَجْفانٍ مِراضٍ
وإيماضٍ يُخالَسُ بِاغْتِماضِ

ومِنْ فَتْلِ العَبيرِ بِعارِضَيْهِ
سَوادٌ قَدْ تَزَرْفَنَ في بَياضِ

السري الرفاء، المحب والمحبوب. تح: غلاونجي، ج: 1، ص: 42.— المصدر
03 —

زُرْفَن، زِرْفَن

اسم
3.1معنى من معاني «زُرْفَن، زِرْفَن»

* الزُّرْفَنُ والزِّرْفَنُ: الإِبْزيمُ، (ج) زَرافِنُ، وزَرافينُ. (مع).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو بَكْرٍ الزُّبَيْدِيُّ (ت: 379هـ=989م):

"ويُقالُ لِلإِبْزيمِ أَيْضًا: زِرْفَنٌ، وزُرْفَنٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، لحن العوام. تح: رمضان عبد التواب، ص: 74.— المصدر
04 —

زُرْفين

اسم
4.1معنى من معاني «زُرْفين»

* الزُّرْفينُ، والزِّرْفينُ: فارِسِيَّةٌ مُعَرَّبَةٌ عَنْ (زورفين: zūrfīn) ومَعْناها: حَلْقةُ البابِ، وحَديدةٌ في طَرَفِ الحِزامِ يُشَدُّ بِها كالإبزيمِ. ونُقِلَتْ بهذَيْنِ المَعْنَيَيْنِ إلى العَرَبِيَّةِ. (ج) زَرافين، زُرافين. (مع).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَد (ت: 170هـ=787م):

"الزِّرْفينُ والزُّرْفينُ، لُغَتانِ: حَلْقَةُ البابِ".

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 400.— المصدر
”قالَ الحُسَيْنُ بْنُ أَحْمَدَ بْنِ مُحَمَّدِ بْنِ الحَجّاج (ت: 391هـ=1001م):

وكُلُّ شاةٍ في غَدٍ
بِرِجْلِها مُعَلَّقَهْ

لا بُدَّ مِنْ أَنْ يَقَعَ الـ
ـزُّرْفينُ جَوْفَ الحَلَقَهْ

الثعالبي، يتيمة الدهر. تح: قميحة، ج: 3، ص: 58.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ نُباتَةَ المِصْرِيُّ (ت: 768هـ=1366م) يَتَذَكَّرُ الدِّيارَ:

وجِنانُ الخُلودِ يَفْتَحُ مِنْها اللَّـ
ـثْمُ صَدْغًا يَظَلُّ كَالزُّرْفينِ

ديوان ابن نباتة المصري، دار إحياء التراث العربي، ص: 495.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عادل زعيتر (1376هـ=1957م) تاريخُ الطَّبْعِ، يُتَرْجِمُ، فَيَذْكُرُ ما بَقِيَ مِنَ الحَضاراتِ السّابِقَةِ عَلى النّيلِ:

"وصَوّانُ النّيلِ وَحْدَهُ الباقي، هُوَ الأَبَدِيُّ، وهُوَ قَدِ احْتَمَلَ الضَّغْطَ وإِزْميلَ النَّقّاشِ وثَقْبَ الزِّرْفينِ والنّاعورَةَ والقَبْرَ وأُلوفَ السِّنينِ".

لودفيغ، النيل حياة نهر، تر: زعيتر، مؤسسة هنداوي، ص: 318.— المصدر

4.2معنى من معاني «زُرْفين»

وــــــــــ: حَلْقَةٌ في طَرَفِ الحِزامِ أَوِ السِّتْرِ الجِلْدِيِّ تُرْبَطُ أَوْ تُعَلَّقُ بِها الأَشْياءُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عامِرٌ الشَّعْبِيُّ (ت: 104هـ=722م) يَصِفُ دِرْعَ النَّبِيِّ ﷺ حينَما عَرَضَهُ عَلِيُّ بْنُ الحُسَيْنِ:

"وأَخْرَجَ إِلَيْنا دِرْعَهُ، فَإِذا هِيَ يَمانِيَّةٌ رَقيقَةٌ ذاتُ زَرافينَ، فَإِذا عُلِّقَتْ ‌بِزَرافينِها سُمِّرَتْ، وإِذا أُرْسِلَتْ مَسَّتِ الأَرْضَ".

ابن أبي شيبة، الكتاب المصنف. تح: الحوت، ج: 5، ص: 198.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ قاسِمُ بْنُ ثابِتٍ السَّرَقُسْطِيُّ (ت: 302هـ=915م):

"واحِدُ الزُّرافينَ: زِرْفينُ وزُرْفينُ؛ وهُوَ: الإِبْزيمُ".

التلمساني، تخريج الدلالات السمعية. تح: إحسان عباس، ص: 426.— المصدر
”وقالَ الأَسْوَدُ الغَنْدَجانِيُّ (ت: 430هـ=1039م) يَذْكُرُ المُصارَعَةَ بَيْنَ عَمْرِو بْنِ الغَوْثِ والأَسْوَدِ بْنِ عَفّارٍ الجَديسِيِّ:

"وكانَتْ مَعَ عَمْرِو بْنِ الغَوْثِ بْنِ طَيِّئٍ قَوْسٌ مَوْصولَةٌ بِزُرافينَ، إِذا شاءَ شَدَّها، فَأَهْوى بِها عَمْرٌو إِلى الجَبَلِ فانْفَتَحَتِ الزُّرافينُ".

الأسود الغندجاني، فرحة الأديب،تح: محمد سلطاني، ص: 56.— المصدر
4.3معنى من معاني «زُرْفين»

وــــــــــ: خُصْلَةُ الشَّعْرِ. (ج) زَرافينُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ اللهِ بْنُ المُعْتَزِّ (ت: 296هـ=909م):

وطافَ بِالدَّنِّ ساقٍ، وَجْهُهُ قَمَرٌ
فَشَكَّهُ بِسَريعِ الحَدِّ مَسْنونِ

ذو طُرَّةٍ نَظَمَتْ في عاجِ جَبْهَتِهِ
مِنْ شَعْرِهِ حِلَقًا سودَ الزَّرافينِ

ديوان ابن المعتز. تح: محمد شريف، ج: 2، ص: 312.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عادل زعيتر (1376هـ=1957م) تاريخُ الطَّبْعِ، يُتَرْجِمُ وَصْفَ بَراعَةِ كِلْيوباتْرا في إِخْفاءِ السُّمِّ في أَزْهارٍ زَيَّنَتْ بِها خُصُلاتِ شَعْرِها تُريدُ قَتْلَ أَنْطونْيوسَ:

"يا لَبَراعَتِها في إِلْقاءِ هَذا الدَّرْسِ حَوْلَ المائِدَةِ بَيْنَ قَدَحَيْنِ ورَياحينَ مَسْمومَةٍ في الزَّرافينِ قُبَيْلَ ما تَتَّخِذُهُ مِنْ قَرارٍ!".

إيميل لودفيغ، كليوباترا، تر: عادل زعيتر، ص: 289.— المصدر
05 —

زِرْفين

اسم
5.1معنى من معاني «زِرْفين»

* الزِّرْفينُ: جَماعَةُ النّاسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"والزِّرْفينُ: جَماعَةُ النّاسِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 9، ص: 121.— المصدر
