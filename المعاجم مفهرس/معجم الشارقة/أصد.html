01 —

أَصَد

فعل
1.1معنى من معاني «أَصَد»

*أَصَدَ (بفتحِ الصّادِ) فُلانٌ لِلْماشِيَةِ يَأْصُدُ (بضمِّ الصّاد) أَصْدًا: عَمِلَ لَها حَظيرَةً.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367ه=977م):

"وَأَصَدَ لِلْغَنَمِ أَصْدًا: ‏عَمِلَ لَها أَصيدَةً".

ابن القوطية، الأفعال. تح: فودة، ص: 10.— المصدر
1.2معنى من معاني «أَصَد»

وــــــــــــ البابَ ونَحْوَه: أَغْلَقَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الطِّرِمّاحُ (ت: 125ه=742م) يَذْكُرُ ارتحالَه بَعيرَه حين تُطْبِقُ عليه الهُمومُ:

فَبِذاكَ أَطَّلِعُ الهُمومَ إِذا دَجَتْ
ظُلَمٌ خَوالِفُها تُخَلُّ وتُؤْصَدُ

ديوان الطرماح. تح: عزة حسن، ص: 146.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367ه=977م):

"وَأَصَدْتُ البابَ: ‏مِثْلُ أَوْصَدْتُ؛ ‏أَيْ أَغْلَقْتُ أَصيدَهُ وَوَصيدَهُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، كتاب الأفعال. تح: فودة، ص: 10.— المصدر
1.3معنى من معاني «أَصَد»

وــــــــــــ الشَّيءَ عَلَى القَوْمِ: أَطْبَقَه عليهم.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"يُقالُ: أَطْبَقَ عَلَيْهِمُ الإِصادُ والوٍصادُ والإِصْدُ. وَأَصَدْتُ عَلَيْهِمْ وَأَوْصَدْتُهُ، والهَمْزُ أَعْرَفُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 145.— المصدر
02 —

آصَد

فعل
2.1معنى من معاني «آصَد»

*آصَدَ فُلانٌ البابَ وَغَيْرَه إيصادًا: أَطْبَقَهُ وَأَحْكَمَ إِغْلاقَهُ. (انظر: و ص د)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ ورقة بْنُ نوفل (ت: 11ق هـ=611م) يرثي عُثْمانَ بْنَ الحُوَيْرِث:

أَمْسَى اِبْنُ جَفْنَةَ في الحَياةِ مُمَلَّكًا
وَصَفِيُّ نَفْسي في ضَريحٍ مُؤْصَدِ

البغدادي، المنمّق في أخبار قريش. تح: خورشيد، ص: 157.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في الْقُرْآنِ الْكَريمِ (11هـ = 632م) قال اللهُ ﷻ:

﴿إِنَّها عَلَيهِم مُّؤْصَدَةٌ﴾. (الهُمَزة: 9)

”قالَ ابْنُ قَيْسِ الرُّقَيَّاتِ (ت: 85ه=704م) يتغَزَّلُ:

إِنَّ في القَصْرِ لَوْ دَخَلْنا غَزالًا
مُؤْصَدًا مُصْفَقًا عَلَيْهِ الحِجابُ

ديوان ابن قيس الرقيات. تح: نجم، ص: 84.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو دُلامَةَ (ت: 161ه=778م) يَشْكو مِنْ الصَّوْمِ في الَحرِّ:

وَلَقيتُ مِنْ أَمْرِ الصِّيامِ وَحَرِّهِ
أَمْرَيْنِ قيسا بِالعَذابِ الـمُؤْصَدِ

ديوان أبي دلامة. تح: يعقوب، ص: 57.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بْنُ عَبْدِ السَّلامِ (ت: 660ه=1262م) يفسر قول الله جل جلاله: ﴿إِنَّها عَلَيْهِمْ مُؤْصَدَةٌ﴾:

"مُّؤْصَدَةٌ: مُطْبَقَةٌ أو مُغْلَقَةٌ بِلُغَةِ قَيْسٍ أو مَسْدودَةُ الجَوانِبِ لا يُفْتَحُ مِنْها جانِبٌ".

العز، تفسيره. تح: الوهبي، ج: 1، ص: 446.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ زاهِرٌ الألمعِيُّ (ت: 1391ه=1971م) يستنهض الشُّعوبَ العَرَبِيَّةَ لتَحْريرِ فِلَسْطينَ:

فَتُلْقي لَهيبًا وَتُلْقي جَحيمًا
يُدَوِّيِ عَلَى السّاحِلِ الـمُؤْصَدِ

زاهر الألمعي، الألمعيات، ص: 64.— المصدر
2.2معنى من معاني «آصَد»

وــــــــــــ الصَّيْدُ الرّامي: أَمْكَنَه مِنْه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230ه=845م):

"ويُقال: أَمْكَنَكَ الصَّيْدُ، وأَكْثَبَكَ، فارْمِهِ، وآصَدَكَ، وأَفْقَرَكَ، وأَفْرَصَكَ، وأَصْقَبَكَ".

أبو مسحل، النوادر. تح: عزة حسن، ص: 187.— المصدر
03 —

أَصَّد

فعل
3.1معنى من معاني «أَصَّد»

*أَصَّدَتِ الفَتاةُ والعَروسُ: لَبِسَتِ الأُصْدَةَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت:321ه=933م):

"وَأَصَّدَتْ إيصادًا، إِذا لَبِسَتِ الـمُؤَصَّدَ والأُصْدَةَ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ص: 1091.— المصدر
3.2معنى من معاني «أَصَّد»

وــــــــــــ البابَ وَنَحْوَهُ: أَطْبَقَهُ وَأَحْكَمَ إِغْلاقَهُ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بُطْرُسُ البُسْتانِيُّ (ت: 1301ه=1883م):

"أَصَّدَ البابَ وَآصَدَهُ لُغَةٌ في أَوْصَدَهُ: أَيْ أَطْبَقَهُ وَأَغْلَقَهُ".

البستاني، محيط المحيط، ج:4، ص: 10.— المصدر
3.3معنى من معاني «أَصَّد»

وــــــــــــ الثَّوْبَ: جَعَلَهُ أُصْدَةً، أي صُدْرَة.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1389ه=1970م):

"أَصَّدَ الثَّوْبَ: جَعَلَهُ أُصْدَةً".

مجمع القاهرة، المعجم الوسيط، ج: 1، ص: 39.— المصدر
04 —

آصِدَة

اسم
4.1معنى من معاني «آصِدَة»

*الآصِدَةُ الغِطاءُ الـمُطْبِقُ. (ج) أَواصِدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللّيْثُ (180ه=796م):

"يُقالُ: أَطْبَقَ عَلَيْهِمُ الإِصادَ والوِصادَ والآصِدَةَ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: هارون، ج: 7، ص: 389.— المصدر
4.2معنى من معاني «آصِدَة»

وــــــــــــ: عَتَبَةُ البابِ. (ج) أُصُدٌ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بُطْرُسُ البُسْتانِيُّ (ت: 1301ه=1883م):

"الإِصادُ: الآصِدَةُ وعَتَبَةُ البابِ. (ج) أُصُدٌ".

البستاني، محيط المحيط، ج: 4، ص: 26.— المصدر
05 —

أَصاد/ إِصاد

اسم
5.1معنى من معاني «أَصاد/ إِصاد»

*الأَصادُ، والإِصادُ: الغِطاءُ الـمُطْبِقُ. (ج) أُصُدٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الطِّرِمّاحُ (125ه=742م) يَذْكُرُ ظَليمًا يَحْتَضِنُ البَيْضَ:

باتَ يَحُفُّ الأُدْحِيَّ مُتَّخِذًا
كِسْرَيْ بِجادٍ مَهْتوكَةٍ أُصُدُهْ

ديوان الطرماح. تح: عزة حسن، ص: 146.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللّيْثُ (ت: 180ه=796م):

"الإِصادُ والأُصُدُ بِمَنْزِلَةِ الـمُطْبَقِ، يُقالُ: أَطْبَقَ عَلَيْهِمُ الإِصادَ والوِصادَ والآصِدَةَ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، التهذيب. تح: البردوني، ج: 12، ص: 222.— المصدر
06 —

إِصاد

اسم
6.1معنى من معاني «إِصاد»

*الإِصادُ: النُّقْرَةُ بَيْنَ الجِبالِ يَجْتَمِعُ فيها الماءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَوْهَرِيُّ (ت: 393ه=1003م):

"والإِصادُ، هِيَ رَدْهَةٌ بَيْنَ أَجْبُلٍ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ج: 1، ص: 442.— المصدر
6.2معنى من معاني «إِصاد»

وــــــــــــ: حَظيرَةُ الغَنَمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والإِصادُ: الـمَوْضِعُ تُحْبَسُ فيهِ الغَنَمُ. وَهُوَ الحَبْسُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عَبّاد، المحيط. تح: آل ياسين، ج: 8، ص: 172.— المصدر
07 —

إِصْد

اسم
7.1معنى من معاني «إِصْد»

*الإِصْدُ: ما يُطْبَقُ مِنْ بابٍ ونَحْوِه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"الإصْدُ، والإِصاد، والوِصادُ اسمٌ والإيصاد المصدر. والإصادُ والإصْدُ هما بِمَنْزِلَةِ الـمُطْبَقِ".

الخليل، العين، تح المخزومي وآخر، ج: 7، ص 145.— المصدر
08 —

أُصْدَة

اسم
8.1معنى من معاني «أُصْدَة»

*الأُصْدَةُ: قِطْعَةُ قُماشٍ يَشُدُّها الرَّجُلُ لِيَسْتُرَ عانَتَهُ وَسِفْلَتَهُ. (ج) أُصَدٌ وَإِصادٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (230ه=845م، مِنْ شَواهِدِ ابْنِ الأَعْرابِيِّ) يَصِفُ سائلا:

مِثْلُ البُرامِ غَدا في أُصْدَةٍ خَلَقٍ
لَمْ يَسْتَعِنْ وَحَوامي الـمَوْتِ تَغْشاهُ

ابن فارس، الصّاحبي. تح: بسج، ص: 144.— المصدر
8.2معنى من معاني «أُصْدَة»

وــــــــــــ: الصُّدْرَةُ، وهي ثَوْبٌ بلا كُمَّيْنِ تَلْبَسُه العَروسُ وَالصَّبِيَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309ه=921م):

"والأُصْدَةُ وجمعها أُصَدٌ: ثوب لا كُمَّيْ له تَلْبَسُه العَروسُ والجارِيَةُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: العمري، ج: 1، ص: 471.— المصدر
09 —

إِصْدَة

اسم
9.1معنى من معاني «إِصْدَة»

*الإِصْدَةُ: مُجْتَمَعُ القَوْمِ. (ج) إِصَدٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وَإِصْدَةُ القَوْمِ: مُجْتَمَعُهُمْ، والجَميعُ الإِصَدُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عَبّاد، المحيط. تح: آل ياسين،ج: 8، ص: 171.— المصدر
9.2معنى من معاني «إِصْدَة»

وــــــــــــ: الحُفْرَةُ ونَحْوُها تُتَّخَذُ كَمينًا لِاصْطِيادِ السِّباعِ. (ج) إِصَدٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"الإِصْدَةُ: زُبْيَةُ السِّباعِ، وَجَمْعُها إِصَدٌ".

ابن عَبّاد، المحيط. تح: آل ياسين،ج: 8، ص: 172.— المصدر
10 —

أَصيد

اسم
10.1معنى من معاني «أَصيد»

*الأَصيدُ: الفِناءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرِو بْنُ العَلاءِ (ت: 154ه=771م) يفسر قول الله جل جلاله: ﴿وَكَلْبُهُمْ باسِطٌ ذِراعَيْهِ بِالوَصيدِ﴾:

"ومَن قالَ: ‏الأَصيدُ، قالَ: ‏آصَدْتُ البابَ، فَهُوَ مُؤْصَدٌ، فَكَأَنَّ مَعْنَى الكَلامِ: وَكَلْبُهُمْ باسِطٌ ذِراعَيْهِ بِفِناءِ كَهْفِهِمْ عِنْدَ البابِ، يَحْفَظُ عَلَيْهِمْ بابَهُ".

الطبري، جامع البيان. تح: التركي، ج: 15، ص: 194.— المصدر
10.2معنى من معاني «أَصيد»

وــــــــــــ: البابُ. (ج) أَصائِدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367ه=977م)
