  1. الصَّوْتُ الخَفِيُّ
01 —

وَسْوَسَ

فعل
1.1معنى من معاني «وَسْوَسَ»

* وَسْوَسَ فُلانٌ: هَمَسَ بِكَلامٍ خَفِيٍّ مُخْتلِطٍ لَمْ يُبَيِّنْهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ عُثْمانُ بْنُ عَفّان (ت: 35هـ=656م) يَذْكُرُ حُزْنَ الصَّحابَةِ عَلى وَفاةِ النَّبيِّ ﷺ:

"حَزِنوا عَلَيْهِ، حَتَّى كادَ بَعْضُهُمْ يُوَسْوِسُ ".

أحمد، المسند. تح: الأرنؤوط وآخرين، ج: 1، ص: 201.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ رُؤْبَةُ (ت: 145ه=762م) يَذْكُرُ صائدًا:

‌وَسْوَسَ ‌يَدْعُو مُخْلِصًا رَبَّ الفَلَقْ

سِرًّا وقَدْ أَوَّنَ تَأْوينَ العُقُقْ

ديوان رؤبة. تح: البروسي، ص: 108.— المصدر
1.2معنى من معاني «وَسْوَسَ»

وــــــــــ الشَّيءُ: صَوَّتَ في خَفاءٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ العَجّاج (ت: 90هـ=708م) يَتَغَزَّلُ:

تَسْمَعُ للحَلْي إِذا ما وَسْوَسا

وارتَجَّ في أَجْيادِها وأَجْرَسا

زَفْزَفَةَ الرِّيحِ الحَصادَ اليُبَّسا

ديوان العجاج. تح: السطلي، ج: 1، ص: 191.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الروميِّ (ت: 283هـ=896م) يَصِفُ رَوْضَةً:

إذا شِئْتُ حَيِّتني رَياحينُ جَنَّةٍ
عَلى سُوقِها في كُلِّ حينٍ تَنفَّسُ

وإِنْ شِئْتُ أَلْهاني سَماعٌ بِمِثْلِهِ
حَمامٌ تَغَنَّى في غُصونٍ تُوَسْوِسُ

ديوان ابن الرومي. تح: نصار، ج: 3، ص: 1231.— المصدر
1.3معنى من معاني «وَسْوَسَ»

وــــــــــ فُلانٌ: اعْتَرَتْهُ الأوهامُ والظُّنونُ وغَلَبَتْ عَلَيْهِ.

العباسي(133ه‍/750م) إلى (656ه‍/1258م)

قالَ الخَليلُ (ت: 170ه=787م):

"‌فُلانٌ ‌مُوَسْوِسٌ، أيْ: غَلَبَتْ عَلَيْهِ الوَسْوَسَةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 335.— المصدر
1.4معنى من معاني «وَسْوَسَ»

وــــــــــ نَفْسُ فُلانٍ بِالأَمْرِ: خالَجَتْهُ وشَغَلَتْهُ بِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

في القرآن الكريم (11ه=632م) قالَ اللهُ ﷻ:

﴿وَلَقَدۡ خَلَقۡنَا ٱلۡإِنسَٰنَ وَنَعۡلَمُ مَا تُوَسۡوِسُ بِهِۦ نَفۡسُهُۥۖ وَنَحۡنُ أَقۡرَبُ إِلَيۡهِ مِنۡ حَبۡلِ ٱلۡوَرِيدِ﴾ [ق: 16]

”

وقالَ النَّبِيّ ﷺ (ت: 11ه=632م) يُبَيِّنُ ما يُتَسامَحُ فيهِ لِلْمُسْلِمينَ:

«تُجُوِّزَ لِأُمَّتِي عَمّا حَدَّثَتْ فِي أَنْفُسِها، أَوْ وَسْوَسَتْ بِهِ أَنْفُسُها، ما لَمْ تَعْمَلْ بِهِ، أَوْ تَكَلَّمْ بِهِ».

أحمد، المسند. تح: الأرنؤوط وآخرين، ج: 12، ص: 38.— المصدر
1.5معنى من معاني «وَسْوَسَ»

وــــــــــ الشَّيْطانُ إلى فُلانٍ، ولَهُ، وفي صَدْرِهِ: حَدَّثَهُ بما لا نَفْعَ فيه ولا خَيْرَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

في القرآن الكريم (11ه=632م) قالَ اللهُ ﷻ:

﴿‌فَوَسۡوَسَ إِلَيۡهِ ٱلشَّيۡطَٰنُ قَالَ يَٰٓـَٔادَمُ هَلۡ أَدُلُّكَ عَلَىٰ شَجَرَةِ ٱلۡخُلۡدِ وَمُلۡكٖ لَّا يَبۡلَىٰ﴾ [طه: 120]

”

وقالَ النَّبِيّ ﷺ (ت: 11ه=632م):

«إِنَّ الشَّيْطانَ يأْتي أَحَدَكُمْ في صَلاتِهِ ‌فَيُوَسْوِسُ لَهُ حَتَّى يُخَيَّلَ لَهُ أَنَّهُ قَدْ أَحْدَثَ لِيَقْطَعَ صَلاتَهُ، فَلا يَقْطَعْ أَحَدُكُمْ صَلاتَهُ حَتَّى يَجِدَ رِيحًا أَوْ يَسْمَعَ صَوْتًا».

القاسم بن سلام، الطهور. تح: سلمان، ص: 403.— المصدر
”

وقالَ رَجُلٌ مِنَ الأَنْصارِ(11ه=632م) يَسْأَلُ النَّبِيّ ﷺ:

"يا نَبِيَّ اللهِ، أَرَأَيْتَ أَشْياءَ يُوَسْوِسُ بِها الشَّيْطانُ في صُدورِنا، لَأَنْ يَخِرَّ أَحَدُنا مِنَ الثُّرَيّا أَحَبُّ إِلَيْهِ مِنْ أَنْ يَبُوحَ بِهِ؟!".

عبد الرزاق، المصنف. تح: الأعظمي، ج: 11، ص: 244.— المصدر

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"الشَّيْطانُ ‌يُوَسْوِسُ في القَلْبِ، فإذا ذُكِرَ اللهُ خَنَسَ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 199.— المصدر

قالَ أبو شامَة (ت: 665هـ=1267م) يَذْكُرُ اجْتِهادَ الفُقَهاء في الوقائِعِ والنَّوازِلِ:

"قاسوا واجْتَهَدوا في إِلْحاقِ غَيْرِها بِها، فَتَضاعَفَتْ مَسائِلُ الفِقْهِ، وشَكَّكَهُمْ إِبْلِيسُ ووَسْوَسَ فِي صُدورِهِمْ".

"إذا تَوَجَّهَ لِحَضْرَةِ الرَّبِّ، واشْتَغَلَ بأَوْرادِ الطَّريقِ وأَعْمالِهِ، تَعَطَّلَتْ عَلَيْهِ أَسْبابُ مَعاشِهِ في الغالِبِ، فَيَأْتِيْهِ الشَّيْطانُ ويُوَسْوِسُ لَهُ ... ودَواءُ ذلِكَ كَثْرَةُ الذِّكْرِ والرِّياضَةِ".

تَوَسْوَسَ

فعل
3.1معنى من معاني «تَوَسْوَسَ»

* تَوَسْوَسَ القَوْمُ: تَحَرَّكوا وهَمَسَ بَعْضُهُمْ إِلى بَعْضٍ بكَلامٍ خَفيٍّ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)

قالَ ابْنُ مَسْعود (ت: 32ه=653م) يَذْكُرُ ما كانَ مِنَ القَوْمِ حِينَ زادَ النَّبيُّ ﷺ في صَلاتِهِ سَهْوًا:

"صَلَّى بِنا رَسولُ اللهِ ﷺ خَمْسًا، فَلَمَّا انْفَتَلَ ‌تَوَسْوَسَ ‌القَوْمُ بَيْنَهُمْ، فَقالَ: ما شأْنُكُمْ؟ قالوا: يا رَسولَ اللهِ، هَلْ زِيدَ في الصَّلاةِ؟! قالَ: لا".

ابن خزيمة، الصحيح. تح: الأعظمي، ج: 1، ص: 526.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ القَطّاع (ت: 515ه=1121م):

"تَوَسْوَسَ القَوْمُ: تَحَرَّكوا وهَشَّ بَعْضُهُمْ إلى بَعْضِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

قالَ ابْنُ حَزْمٍ (ت: 456ه=1064م) في سِياقِ الجَوابِ عَلى مَنْ قالَ: هَلِ اللهُ قَادِرٌ عَلى أنْ يَمْسَخَ الكافِرَ قِرْدًا وكَلْبًا؟:

"أَيُمْكِنُ أنْ يَكونَ إنْسانٌ في النَّاسِ قَدْ ‌تَوَسْوسَ، وأَوْهَمَتْهُ ظُنونُهُ الكاذِبَةُ، وتَخَيُّلُهُ الفاسِدُ وهَوَسُهُ أنَّ الأشْياءَ عَلى خِلافِ ما هِيَ عَلَيْهِ؟".

قالَ مُقاتِلُ بنُ سُلَيْمانَ (ت: 150ه=767م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿أَلَا يَعۡلَمُ ‌مَنۡ ‌خَلَقَ ‌وَهُوَ ٱللَّطِيفُ ٱلۡخَبِيرُ﴾ [الملك: 14]:

"يَعْني: لُطْفُ عِلْمِهِ بِما في القُلوبِ، خَبيٌر بِما فيها مِنَ السِّرِّ والوَسْوَسَةِ".

قالَ الرّازيُّ (ت:666هـ=1268م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿وَإِن تُبۡدُواْ مَا فِيٓ أَنفُسِكُمۡ أَوۡ تُخۡفُوهُ يُحَاسِبۡكُم بِهِ ٱللَّهُۖ فَيَغۡفِرُ لِمَن يَشَآءُ وَيُعَذِّبُ مَن يَشَآءُۗ وَٱللَّهُ عَلَىٰ كُلِّ شَيۡء قَدِيرٌ﴾ [البقرة: 284]:

"ما يُحَدِّثُ بِهِ الإِنْسانُ نَفْسَهُ لا يَأْثَمُ بِهِ...وإِنَّما المُرادُ ما يُمْكِنُ الاحْتِرازُ عَنْهُ، وهو العَزْمُ القاطِعُ والاعْتِقادُ الجازِمُ لا مُجَرَّدَ حَديثِ النَّفْسِ والوَسْوَسَةِ".
