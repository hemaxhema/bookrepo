  1. التَّمَهُّلُ
01 —

راثَ

فعل
1.1معنى من معاني «راثَ»

*راثَ فُلانٌ أو غَيْرُه يَريثُ رَيْثًا: أَبْطَأَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أبو المُثَلَّم الهُذَليّ (ت: 50ق ه=573م) يُجيبُ صَخْرَ الغَيّ:

لَعَلّي إِنْ دَعَوْتُكَ مِنْ قَريبٍ
إِلَى خَيْرٍ لِتَأْتِيَهُ تَريثُ

السّكّرِيّ، شرح أشعار الهذليين. تح: فراج، ج: 1، ص: 264.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (ت: 11ه=632م):

"اللَّهُمَّ اسْقِنا غَيْثًا مُغيثًا، مَريعًا طَبَقًا غَدَقًا غَيْرَ رائِثٍ، نافِعًا غَيْرَ ضارٍّ".

مصنّف ابن أبي شيبة. تح: كمال الحوت، ج: 6، ص: 324.— المصدر
”وقالَ أَبو بكْرٍ الصِّدّيق (ت: 13ه = 634م) يَذْكُرُ الأَطْلالَ:

عَرَفْتُ دِيارًا بِالحِمَى فَشَرائِثِ
تَعَفَّتْ فَدَمْعُ العَيْنِ لَيْسَ بِرائِثِ

ديوان أبي بكر الصّدّيق، تح: الأسمر، ص: 46.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو العَتاهِيةِ (ت: 211ه=826م) يَذْكُرُ أَحْوالَ الإِنسانِ:

وَرُبَّ رَيْثِ امْرِئٍ أَقْوَى لِمَأْخَذِهِ
لِما أَرادَ وأَوْحَى فيهِ مِنْ عَجَلِهْ

ديوان أبي العتاهية. ص: 371.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ الجيّابِ الغَرْناطِيّ (ت: 749ه=1348م):

ثَبَتَّ عَلَى عَهْدِ الصِّبا غَيْرَ ناكِثِ
وبادَرْتَ مَهْواةَ الهَوَى غيْرَ رائِثِ

ديوان ابن الجياب، تح: فوزي عيسى. ص: 123.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ سامي الباروديّ (ت: 1323ه=1904م) يَهْجو أَهْلَ سَرَنْديبَ:

لاتَلْبَثُ الحِكْمَةُ ما بَيْنَهُمْ
ولايَريثُ الفَضْلُ حَتَّى يَتُوهْ

ديوان محمود سامي البارودِيّ. ص: 294.— المصدر
1.2معنى من معاني «راثَ»

وــــــــــ عَلَى فُلانٍ، وعَنْهُ: أَبْطَأَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ تأَبَّط شَرًّا (ت: 82ق ه=542م) يَذْكُرُ ما فَعَلَهُ بالأعداءِ:

وَخَدْتُ بِهِم، حَتَّى إِذا طالَ وخْدُهُم
وراثَ عَلَيْهِم مَضْجَعي ومَقيلي

مَهَدْتُ لَهُمْ حَتّى إِذا طالَ رَوْعُهُم
إِلى المَهْدِ خاتَلْتُ الضِّيا بِخَتيلِ

ديوان تأبّط شرًّا وأخباره. تح: ذي الفقار، ص: 304.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ عائِشةُ أُمُّ المُؤمِنينَ (ت: 58ه=678م):

"واعَدَ جِبْريلُ رَسولَ اللَّهِ صَلَّى اللهُ عَلَيْهِ وسَلَّمَ أَنْ يَأْتِيَهُ في ساعَةٍ فَراثَ عَلَيْهِ، فَخَرَجَ رَسولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وسَلَّمَ فَلَقِيَهُ قائِمًا بِالبابِ".

مسند إسحاق بن راهويه. تح: البلوشي، ج: 2، ص: 496.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"يُقالَ: راثَ عَلَيْنا فُلانٌ يَريثُ رَيْثًا، وراثَ عَلَيْنا خَبَرُهُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 8، ص: 235.— المصدر
”وقالَتِ امرَأَةٌ مِنْ طَيّىئ (231ه =864م) مِنْ شَواهِدِ أبي تمّام، تَرْثي:

تأوَّبَ عَيْني نُصْبُها واكْتِئابُها
ورَجَّيْتُ نَفْسًا راثَ عَنْها إيابُها

التبريزي، شرح ديوان الحماسة، ص: 461— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الذَّهَبيُّ (ت: 748ه=1347م):

"فَقالَ رَسول اللَّهِ - صَلَّى اللَّهُ عَلَيْهِ وسَلَّمَ - كما ذُكر - حين راثَ عَلَيْهِ أبو لُبابة: أَما فَرَغَ أبو لُبابةَ مِنْ حُلَفائِهِ؟".

الذهبي، تاريخ الإسلام. تح: بشّار عواد، ج: 1، ص: 205.— المصدر
1.3معنى من معاني «راثَ»

o رُبَّ رَيْثٍ يُعْقِبُ فَوْتًا: مَثَلٌ يُضْرَبُ في مَعْنى رُبَّما أُخِّرَ أَمْرٌ فَفاتَ صاحِبَه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الخَيْرِ الهاشِمِيُّ (400ه=1010م):

"رُبَّ رَيْثٍ يُعْقِبُ فَوْتًا أي رُبّما احْتُبِسَ المَرْءُ عَنْ أَمْرِ يُريدُهُ فَفاتَه".

الأمثال للهاشمِيّ. دار سعد الدّين ص: 135.— المصدر
02 —

أَراثَ

فعل
2.1معنى من معاني «أَراثَ»

* أَراثَ فُلانٌ أَوْ غَيْرُهُ: أَبْطَأَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَعْقِلُ بنُ خُوَيْلِدٍ (ت: 5ه=626م):

لَعَمْرُكَ لَلْيَأْسُ غَيْرُ الـمُريـ
ــثِ خَيْرٌ مِنَ الطَّمَعِ الكاذِبِ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 1، ص: 392.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّريفُ المُرْتَضى (ت: 423ه=1031م) يَرْثي ابْنَ شُجاع الصّوفِيَّ:

إِذا ما أُريثُ أُريثُ الرِّحالَ
وإِمّا قَرِبْتُ قَرِبْتُ الغُرورا

ديوان الشّريف المرتضى. ج: 2، ص: 28.— المصدر
2.2معنى من معاني «أَراثَ»

وــــــــــ فُلانًا: أَبْطَأهُ وأَخَّرَه.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بنُ أبي طالب (ت: 40ه=661م) يُخاطبُ جَريرَ بنَ عَبْدِ اللهِ:

"أَرادَ مُعاويَةُ أَلّا يَكونَ لي في عُنُقِهِ بَيْعَةٌ، وأَنْ يَخْتارَ مِنْ أَمْرِهِ ما أَحَبَّ، وأَرادَ أَنْ يُريثَكَ ويُبْطِئَكَ حَتَّى يَذوقَ أَهْلَ الشّامِ".

ابن مزاحم، وقعة صفّين. تح: هارون، ص: 52.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البُحْتريّ (ت: 284ه=897م) يَهْجو:

وَما كادَتِ الأَيّامُ عُمْرًا يُريثُهُ
ولا الدَّهْرُ يُبْلي ما أَجَدَّتْ عَجائِبُهْ

ديوان البحتري، تح: الصّيرفِيّ، ص: 221.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفِيروزاباديُّ (ت: 817ه=1414م):

"وما أراثَكَ: ما أبْطَأَكَ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزابادِيّ، القاموس المحيط. تح: العرقسوسِيّ، ص: 170.— المصدر
03 —

رَيَّثَ

فعل
3.1معنى من معاني «رَيَّثَ»

* رَيَّثَ فُلانٌ وغَيْرُهُ: تَعِبَ وأَعْيا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385ه=995م):

"ورَيَّثَ الفَرَسُ والرَّجُلُ: إِذا أَعْيَيا أَوْ كادا".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، المحيط في اللّغة. تح: آل ياسين، ج: 10، ص: 168.— المصدر
3.2معنى من معاني «رَيَّثَ»

وــــــــــ عَمّا كانَ عَلَيْهِ: قَصَّرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سيدَهْ (ت: 458ه=1066م):

"ورَيَّثَ عَمّا كانَ عَلَيْهِ: قَصَّرَ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 190.— المصدر
3.3معنى من معاني «رَيَّثَ»

وــــــــــ فُلانًا أَوِ الأَمْرَ: أَبْطَأَهُ وأَخَّرَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو بكر الصّدّيقُ (ت: 12ه=633م) يوصي يَزيدَ بْنَ أَبي سُفْيانَ حينَ وجَّهَهُ لِفَتْحِ الشّامِ:

"وإِذا قَدِمَ عَلَيْكَ رُسُلُ عَدُوِّكَ فَأَكْرِمْهُمْ، وأَقْلِلْ لُبْثَهُمْ حَتَّى يَخْرُجُوا مِنْ عَسْكَرِكَ وهُمْ جاهِلونَ بِهِ، ولا تُرَيِّثْهُمْ فَيَرَوْا خَلَلَكَ، ويَعْلَمُوا عِلْمَكَ".

البلاذري، أنساب الأشراف. تح: سهيل زكّار وآخر، ج: 10، ص: 113.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الدُّقَيش الأعْرابيّ (ت: 170ه=787م) يَصِفُ الكِسائيَّ:

"إِنَّهُ لَيُرَيِّثُ النَّظَرَ".

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 190.— المصدر
3.4معنى من معاني «رَيَّثَ»

وــــــــــ الشَّيْءَ: لَيَّنَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَمْرو بنُ العاصِ (ت: 43ه=664م) يُخاطِبُ معاويَةَ ويَتَعَهَّدُ بإسْكاتِ ابنِ الزُبَيْر:

"ضَمِنْتُ لَكَ يا أَميرَ المُؤْمِنينَ، أَنْ أُلَيِّنَ عَريكَتَه، وأُذْهِبَ نَخْوَتَه، وأُخْرِسَ لِسانَه، وأُعْدِمَهُ بَيانَه، حَتَّى أَدَعَهُ أَلْيَنَ مِنْ خَميرَةٍ مُرَيَّثَةٍ، وأَذَلَّ مِنْ نَقْدَةٍ".

البلاذري، أنساب الأشراف. تح: زكّار وآخر، ج: 5، ص: 78.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385ه=995م):

"والتَّرْييثُ: التَّلْيينُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، المحيط في اللّغة. تح: آل ياسين، ج: 10، ص: 168.— المصدر
”وقالَ ابنُ سينا (ت: 418ه= 1027م) يُعَدِّدُ الأَدْويَةَ المُسَكِّنَةَ لِلْوَجَعِ:

"وأَمّا الأَدْوِيَةُ المُسَكِّنَةُ لِلْوَجَعِ فَمِثْلُ بَزْرِ الكَتّانِ ولُعابِهِ، ومِثْلُ الجِلَّوْزِ، والفُنْدُقِ، وبَزْرِ الخِطْمِيِّ، ولَها تَرْييثٌ أَيْضًا لِلْأَدْوِيَةِ الحَصَوِيَّةِ، مُوافِقَةٌ لِجِرْمِ الكُلْيَةِ".

ابن سينا، القانون في الطب، تح: الضناوي، ج: 2، ص: 692.— المصدر
3.5معنى من معاني «رَيَّثَ»

وــــــــــ الأَمْرَ: تَراخَى فيهِ وتَهاوَنَ. (انظر: ر ب ث)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ الرّحمَنِ الهَمَذانيّ (ت: 320ه=932م):

"ويُقالَ أَيْضًا: فَتَّرَ، وونَى (الاسْمُ الوِنْيَةُ)، وتَراخَى، وفَشَلَ، وتَهاوَنَ (مِنَ الهُوَينى)، وثَبَّطَ الأُمورَ، ورَيَّثَها ورَبَّثَها".

الهمذاني، الألفاظ الكتابيّة. ص: 35.— المصدر
3.6معنى من معاني «رَيَّثَ»

وــــــــــ فُلَانًا عَنِ الشَّيْءِ: بَطَّأَهُ وأَخَّرَهُ عَنْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سيبَوَيْهِ (ت: 180ه=796م):

"ومِثْلُ ذَلِكَ [ يَعْني تَحَلَّمَ ]: تَقَعَّدْتُهُ، أَيْ: رَيَّثْتُهُ عَنْ حاجَتِهِ وعُقْتُهُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

سيبويه، الكتاب. تح: هارون، ج: 4، ص: 71.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ العَديم (ت: 660ه=1262م) يَذْكُرُ خَسارَةَ المُسْلمينَ أَمامَ الفِرِنْج:

"ونَزَلَتِ العَساكِرُ السُّلْطانِيَّةُ أَرْضَ شيزَرَ، وجَعَلَ أَتابِكُ يُرَيِّثُ الفِرِنْجَ عَنْ اللِّقاءِ؛ خَوْفًا مِنَ الفِرِنْجَ أَنْ يَكْسِروا العَساكِرَ السُّلْطانِيَّةَ فَيَأْخُذوا الشّامَ جَميعَهُ".

ابن العديم، تاريخ حلب. ج: 1، ص: 264.— المصدر
04 —

ارْتاثَ

فعل
4.1معنى من معاني «ارْتاثَ»

* ارْتاثَ الأَمْرُ عَلَى فُلانٍ: اشْتَدَّ وتَعَسَّرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ الرَّحْمنِ الهَمَذانيّ (ت: 320ه=932م):

"قَدِ اعْتاصَ عَلَيْهِ الأَمْرُ أَيْ التَوَى، فَهُوَ مُعْتاصٌ، وتَوَعَّرَ فَهو مُتَوَعِّرٌ(...) وتَعَسَّرَ، والتاثَ، وارْتاثَ، وتَشَدَّدَ".

الهمذاني، الألفاظ الكتابية. ص: 40.— المصدر
05 —

تَرَيَّثَ

فعل
5.1معنى من معاني «تَرَيَّثَ»

* تَرَيَّثَ فُلانٌ وغَيْرُهُ: تَباطَأَ وتَوَقَّفَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ العَزيزِ السَّكونيّ (ت: 250ه=864م) يُخاطِبُ يَحْيى بنَ أَكْثَمَ:

"أَيُّها القاضي قَدْ وقَفَتِ الأُمورُ وتَرَيَّثَتْ".

المزي، تهذيب الكمال. تح: بشّار عواد، ج: 31، ص: 219.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أبو شامةَ المَقْدِسيُّ (ت: 665ه=1267م) يَذْكُرُ أَسْبابَ إِعْجابِ صَلاحِ الدّينِ بالعاضدِ:

"وكانَ صَلاحُ الدّينِ قَدْ وقَعَ مِنَ العاضِدِ بِمَوْقِعٍ، وأَعْجَبَهُ عَقْلُهُ وسَدادُ رَأْيِهِ وشَجاعَتُهُ وإِقْدامُهُ عَلَى شاوَر في مَوْكِبِهِ، وأَنَّه قَتَلَهُ حينَ جاءَهُ أَمْرُهْ، ولَمْ يَتَرَيَّثْ ولا تَوَقَّف".

أبو شامة، أخبار الدّولتين. تح: الزيبق، ج: 2، ص: 115.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ المُخْتارُ بنُ يبّ (ت: 1270ه=1854م) يَذْكُرُ جِمالًا:

سَلَكْنَ السَّيْلَ لا مُتَرَيِّثاتٍ
حِذارًا مِنْ مُعالَجَةِ الرِّمالِ

محمّد المختار ولد أباه، الشّعر والشُّعراء في موريتانيا. ص: 154.— المصدر
5.2معنى من معاني «تَرَيَّثَ»

وــــــــــ: احْتَبَسَ ومَكَثَ. (انظر: ر ب ث)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو العَبّاسِ الشَّريشيّ (ت: 619ه=1222م):

"تَرَبَّثَ، إِذا احْتَبَسَ ومَكَثَ، ويُقالَ: تَرَيَّثَ بِنُقْطَتَيْنِ وتَرَبَّثَ بِواحِدَةٍ، والمَعْنَى فيهِما واحِدٌ".

الشريشي، شرح مقامات الحريري. تح: شمس الدين، ج: 3، ص: 386.— المصدر
5.3معنى من معاني «تَرَيَّثَ»

وــــــــــ عَلَى القَوْمِ:صَبَرَ عَلَيْهِم وأَمْهَلَهُمْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ إسْحاقَ (ت: 151ه=768م) يَذْكُرُ بَني إِسرائيلَ:

"ثُمَّ عَدَّدَ عَلَيهِمْ أَحْداثَهُمْ، فَذَكَرَ لَهُمْ العِجْلَ، ما صَنعُوا فيهِ وتَرَيُّثَهُ عَلَيهِمْ، وإِقالَتَهُ إِيّاهُمْ".

تفسير ابن إسحاق. تح: أبو صعيليك، ج: 1، ص: 25.— المصدر
5.4معنى من معاني «تَرَيَّثَ»

وــــــــــ الأَمْرَ، وفيهِ: تَأَنَّى فيهِ وتَدَبَّرَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحَسَنُ بنُ عليّ (ت: 43ه=663م) يُخاطِبُ عَمْرَو بنَ العاصِ:

"واللهِ إِنَّكَ لَتَعْلَمُ أَنَّ عَلِيًّا لَمْ يَتَرَيَّثْ في الأَمْرِ، ولَمْ يَشُكَّ في اللهِ طَرْفَةَ عَيْنٍ".

الجاحظ، المحاسن والأضداد. تح: عطوي، ج: 1، ص: 98.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ إِسْحاقَ (ت: 151ه=768م) يَذْكُرُ ورقَةَ بنَ نَوْفَل:

"فَكانَ فيما يَذْكُرونَ يَقولُ أَشْعارًا يَسْتَبْطِئُ فيها خَبَرَ خَديجَةَ، ويَتَرَيَّثُ ما ذَكَرَتْ لَهُ".

ابن اسحاق، السير والمغازي. تح: زكّار، ج: 1، ص: 115.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ رَشيدُ رِضا (ت: 1354ه=1935م) يَذْكُرُ رَجُلين يُحاولان تَسَلُّقَ جِدارٍ مُرْتَفِعٍ:

"ولَكِنَّهُما تَكَلَّفا ذَلِكَ وتَرَيَّثا فيهِ حَتَّى وصَلا إِلَى مَكانِ طَمْرِ النُّقودِ".

رشيد رضا، تفسير القرآن، ج: 9، ص: 140.— المصدر
06 —

اِسْتَراثَ

فعل
6.1معنى من معاني «اِسْتَراثَ»

* اسْتَراثَ فُلانٌ وغَيْرُهُ: أَبْطَأَ. ورُبَّما قيلَ: اسْتَرْيَثَ، بِتَرْكِ الإِعْلالِ. (عن ابنِ فارِس).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ عائِشةُ أُمُّ المُؤْمِنينَ (ت: 58ه=678م) تَذْكُرُ حادِثَةَ الإِفْكِ:

«ثُمَّ أَصْبَحْتُ أَبْكي، فَدَعا رَسولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وسَلَّمَ عَلِيَّ بنَ أَبي طالِبٍ وأُسامَةَ بنَ زَيْدٍ حينَ اسْتَراثَ الوَحْيُ فَشاوَرَهُما في فِراقِ أَهْلِهِ».

الطبراني، المعجم الكبير. ج: 23، ص: 58.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو تَمّامٍ الطّائيّ (ت: 231ه=846م) يُخاطِبُ مَمْدوحَه:

طالَ المَدَى فاعْتَراكَ عَتْبٌ
مِنْ صادِقِ الوُدِّ مُسْتَريثِ

التّبريزيّ، شرح ديوان أبي تمّام. تح: الأسمر، ج: 1، ص: 177.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ هُتَيْمِلٍ (ت: 696ه=1297م) يُخاطبُ صاحِبيْهِ:

لا تَعْقِلا نِضْوَيْكُما أَوْ تَنْزِلا
أوْ تَنْشُدا عَقْلي بِبُرقَةِ عاقِلِ

وإِذا اسْتَراثَ النّازِلونَ فَعَرِّضا
بَكْرَيْكُما لِلْهَضْبِ هَضْبَ حُلاحِلِ

ديوان ابن هتيمل. تح: العقيلي، ج: 1، ص: 803.— المصدر
6.2معنى من معاني «اِسْتَراثَ»

وــــــــــ فُلانًا أو الشَّيْءَ: اسْتَبْطَأَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو سَهْمٍ الهُذَليّ (ت: 30ه=651م) يَصِفُ حِمارًا وحْشيًّا يَتَربَّصُ بِهِ صائِدٌ:

فَلَمّا تَوَلَّى صادِرًا واسْتَراثَهُ
غَبِيُّ سَفاهٍ في المَقاتِرِ صائِدُ

مَقيتٌ إِذا لَمْ يَرْمِ لا هُوَ يائِسٌ
ولا هُوَ حَتّى يَخفِقَ النَّجْمُ راقِدُ

أُخيفَ بِهَمٍّ فاحزَأَلَّ فُؤادُهُ
فَرامَ بِهَمٍّ أَيُّها هُوَ عامِدُ

ابن المبارك، منتهى الطّلب. تح: طريفي، ج: 1، ص: 221.— المصدر
”وقالَ عَبْدُ الرَّحْمَنِ بنُ سَمُرةَ (ت: 50ه=670م) يَذْكُرُ أَشْراطَ السّاعَةِ:

"فَبَيْنَما مَنْ فيها مِنَ المُسْلِمينَ يَسْتَريثونَ نُزولَ عيسَى وقَدْ نَزَلَ الدَّجّالُ بِالمُؤْمِنينَ المُعْتَصِمينَ بِجَبَلِ الخَليلِ(...) إِذْ نَزَلَ عيسَى بنُ مَرْيَمَ عِنْدَ المَنارَةِ البَيْضاءِ شَرْقِيَّ دِمَشْقَ".

السلمي، أشراط الساعة. تح: الغماري، ص: 145.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (216ه=831م) مِنْ شَواهِدِ الأصْمَعيّ:

إِذا اسْتَراثَتْ ساقِيًا مُسْتَوْفِزا

بَجَّتْ مِنَ البَطْحاءِ نَهْرًا هُزْهُزا

الجوهري، الصحاح. تح: أحمد عطار، ج: 3، ص: 902.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ السَّيِّدُ الجُمَيْليُّ (ت: 1405ه=1984م) يَذْكُرُ ما صَنَعَهُ أَبو لُبابةَ بِنَفْسِهِ بَعْدَ فِعْلَتِهِ معَ بَني قُرَيْظَةَ:

"فَلَمّا اسْتَراثَهُ واسْتَبْطَأَهُ رَسولُ اللهِ صَلَّى اللهُ عَلَيْهِ وسَلَّمَ، وعَرَفَ جَلِيَّةَ الأَمْرِ قالَ عَلَيْهِ الصَّلاةُ والسَّلامُ: «أَما إنَّهُ لَوْ جاءَني لاسْتَغْفَرْتُ لَهُ".

السيد الجميلي، نساء النبي صلى الله عليه وسلّم، ج: 1، ص: 88.— المصدر
07 —

ارْتِياث

اسم
7.1معنى من معاني «ارْتِياث»

* الارْتِياثُ: التَّأَخُّرُ والبُطْءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سينا (ت: 428ه=1037م) يَذْكُرُ فَسادَ الجَمْعِ بَيْنَ الأَطْعِمَةِ:

"وفَسادُ الجَمْعِ إِمّا لِتَهْيِئَةٍ لِلْعُفونَةِ، أَوْ لِلِانْعِقادِ، أَوْ لِلارْتِياثِ والمُكْثِ، أَوِ الانْحِدارِ، بِأَنْ يُزَلَّقَ أَوْ يُعَصَّرَ أَوْ يُسَهَّلَ".

من مؤلفات ابن سينا الطبية. ص: 16.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ اللَّوّاحُ الخَروصيّ (ت: 920ه=1514م) يَذْكُرُ مَنْ يَرْثيه بَعْدَ مَوْتِهِ:

لَقَدْ أَكْثَرْتُ في التِّرْبِ المَراثي
فَمَنْ لي إِنْ لَحِقْت التِّرْبَ راثي

فَيَذْكُرَ مِنْ خِلالي ما حَلا لي
عَلَى عَجَلٍ بِها وعَلى ارْتياثِ

ديوان اللَّوّاح ، تح: الصليبي، ص: 295.— المصدر
08 —

تَرَيُّث

اسم
8.1معنى من معاني «تَرَيُّث»

* التَّريُّثُ: الإِبْطاءُ والتَّمَهُّلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بنُ الخَطّابِ (ت: 23ه=644م) يُخاطِبُ عَمْرو بنَ العاصِ حينَ اسْتبْطَأ خَراجَه:

"ولَقَدْ أَكْثَرْتُ في مُكاتَبَتِكَ في الَّذي عَلَى أَرْضِكَ مِنَ الخَراجِ، وظَنَنْتُ أَنَّ ذَلِكَ سَيَأْتينا عَلَى غَيْرِ تَرَيُّثٍ".

أحمد زكي، جمهرة رسائل العرب، ج: 198، ص: 1.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعيّ (ت: 216ه=831م):

"سايَرْتُ عَبْدَ اللهِ بْن مُحَمَّدِ بْن عِمْرانَ الطَّلْحِيّ، وكانَ لا يَضْحَكُ مَعَ المُروءَةِ والتَّرَيُّثِ".

وكيع، أخبار القضاة. تح: اللحام، ص: 148.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ تَيْمِيَّةَ (ت: 728ه=1327م) يَشْرَحُ قَوْلَ عُمَرَ: "كانَتْ بَيْعَةُ أَبي بَكْرٍ فَلْتَةً":

"ومَعْناهُ أَنَّ بَيْعَةَ أَبِي بَكْرٍ بودِرَ إِلَيْها مِنْ غَيْرِ تَرَيُّثٍ وَلا انْتِظارٍ".

ابن تيميّة، منهاج السنة النبوية. تح: محمد سالم، ج: 5، ص: 469.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّدُ السَّنوسيّ (ت: 1294ه=1876م) يَرْثي:

حَتَّى دَعاهُ الموتُ عَنْ هَرَمٍ فَلَمْ
يُبْدِ التَّرَيُّثَ عَنْ مَوارِدِ شُرْبِهِ

السنوسي، مسامرات الظريف، ج: 1، ص: 163.— المصدر
09 —

رَيْث

أداة
9.1معنى من معاني «رَيْث»

* رَيْثَ (عند النُّحاةِ): مَصْدَرٌ عومِلَ مُعامَلَةَ ظَرْفِ الزَّمانِ المَبْنِيِّ عَلَى الفَتْحِ، وهو مُلازِمٌ للإِضافةِ إلى الجُمْلَةِ الفِعْلِيَّة التي فِعْلُها مُتَصَرِّفٌ مُثْبَتٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الشَّنْفَرى (ت: 70ق ه=554م) يَفْخَرُ:

وَلَوْلا اجْتِنابُ الذَّأْمِ لَم يُلْفَ مَشْرَبٌ
يُعاشُ بِهِ إِلّا لَدَيَّ ومَأْكَلُ

وَلَكِنَّ نَفْسًا مُرَّةً لا تُقيمُ بي
عَلَى الذَّأْمِ إِلّا رَيْثَما أَتَحَوَّلُ

ديوان الشّنفرى. تح: إميل يعقوب، ص: 63.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ عائِشةُ أُمُّ المُؤْمِنينَ (ت: 58ه=678م) تَذْكُرُ ما فَعَلَهُ النَّبِيُّ ـــــــــ صَلَّى اللَّهُ عَلَيْهِ وسَلَّمَ ـــــــ في لَيْلَتِها:

"لَمّا كانَتْ لَيْلَتي الَّتي هُوَ عِنْدي انْقَلَبَ، فَوَضَعَ نَعْلَيْهِ عِنْدَ رِجْلَيْهِ، ووضَعَ رِداءَهُ، وبَسَطَ طَرَفَ إِزارِهِ عَلَى فِراشِهِ، فَلَمْ يَلْبَثْ إِلّا رَيْثَما ظَنَّ أَنّي قَدْ رَقَدْتُ".

سنن النسائي. تح: شلبي، ج: 8، ص: 160.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ المُقَفَّع (ت: 142ه=760م) عَلَى لِسانِ أَرْنَبٍ تُخاطِبُ الوُحوشَ:

"قالَتْ: تَأْمُرْنَ الَّذي يَنْطَلِقُ بي إِلَى الأَسَدِ أَنْ يُمْهِلَني رَيْثَما أُبْطِئُ عَلَيْهِ بَعْضَ الإِبْطاءِ".

ابن المقفع، كليلة ودمنة. ص: 117.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
