ضغبس
ضُغْبوس(اسم)
01 —

ضُغْبوس

اسم
1.1معنى من معاني «ضُغْبوس»

* الضُّغْبوسُ مِنَ الرِّجالِ: الضَّعيفُ الذَّليلُ. (ج) ضَغابيسُ وضَغابِسُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُتَلَمِّسُ (ت: 50ق هـ=573م) يَفْخَرُ بنَسَبِهِ:

لَوْ كانَ مِنْ آلِ وَهْبٍ بَيْنَنا عُصَبٌ
ومِنْ نَذيرٍ ومِنْ عَوْفٍ مَحاميسُ

أَوْدى بِهِمْ مَنْ يُرادِيني وأَعْلَمُهُمْ
جودَ الأَكُفِّ إِذا ما اسْتَعْسَرَ البوسُ

ياحارِ إِنِّي لَمِنْ قَوْمٍ أُولي حَسَبٍ
لا يَجْهَلونَ إِذا طاشَ الضَّغابيسُ

القرشي، جمهرة أشعار العرب. تح: البجاوي، ص: 449.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّجاشيُّ (40هـ=660م) يَفْخَرُ بمُوالاةِ عَلِيِّ بْنِ أَبي طالِبٍ:

ما بالُ كُلِّ أَميرٍ يُسْتَرابُ بِهِ
دِينٌ صَحيحٌ ورَأْيٌ غَيْرُ مَلْبوسِ

والَى عَلِيًّا بِقَدْرٍ  بَذَّ  مِنْهُ إِذا
ما صَرَّحَ الغَدْرُ عَنْ رَدِّ الضَّغابيسِ

ديوان النّجاشيّ الحارثيّ. تح: البكاري وآخرين، ص: 42.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ شُبْرُمَةَ (ت: 144هـ=762م) يَصِفُ قَضاءَهُ:

أَقْضي بِما في كِتابِ اللَّهِ مُجْتَهِدًا
وبِالنَّظائِرِ أَقْضي والمَقاييسِ

إِذا قَضَيْتُ بِمُرِّ الحَقِّ مُجْتَهِدًا
فَلَسْتُ أَجْهَلُ أَقْوالَ الضَّغابيسِ

وكيع القاضي،أخبار القضاة. تح: المراغي، ج: 3، ص: 97.— المصدر
1.2معنى من معاني «ضُغْبوس»

وــــــــــ مِنَ القِثّاءِ وَنَحْوِهِ: الصَّغيرُ. (ج) ضَغابيسُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُجيبُ جابِرَ بْنَ عَبْدِ اللهِ عَنْ وَصْفِ أَهْلِ النَّارِ بَعْدَ الشَّفاعَةِ لَهُمْ:

«يَخْرُجُ مِنَ النّارِ بِالشَّفاعَةِ كَأَنَّهُمُ الثَّعاريرُ، قُلْتُ: ما الثَّعارِيرُ؟ قالَ: الضَّغابيسُ».

البخاري، الصحيح. تح: الناصر، ج: 8، ص: 115.— المصدر
”وقالَ أبو أُسَيْدٍ السّاعِدِيُّ (ت: 30هـ=651م) يَصِفُ ما أُهْدِيَ للنَّبِيّ ﷺ في الحُدَيْبِيَةِ:

"أُهْدِيَ يَوْمَئِذٍ لِرَسولِ اللَّهِ ﷺ مِنْ وَدّانَ ثَلاثَةُ أَشْياءَ: مُعيشًا، وعِتْرًا وضَغابيسَ، وجَعَلَ رَسولُ اللَّهِ ﷺ يَأْكُلُ مِنَ الضَّغابيسِ والعِتْرِ وأَعْجَبَهُ".

الواقدي، المغازي. تح: جونس، ج: 2، ص: 577.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"الضَّغابيسُ: شِبْهُ صِغارِ القِثّاءِ تُؤْكَلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 229.— المصدر
”وقالَتِ امْرَأَةٌ (216هـ=831م) مِنْ رُواةِ الأَصْمَعِيِّ، تَتَحَدَّثُ عَنْ طَعامِهِمْ:

"طَعامُنا الحَارُّ والقارُّ وإِنْ ذُكِرَتِ الضَّغابيسُ فَإِنِّي ضَغِبَةٌ".

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 229.— المصدر
1.3معنى من معاني «ضُغْبوس»

وــــــــــ: نَبْتٌ لَهُ ساقٌ يَنْبُتُ فِي أَصْلِ الثُّمامِ يُسْلَقُ وَيُجعَلُ بِالخَلِّ والزَّيْتِ وَيُؤكَلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رَجُلٌ مِنْ وَفْدِ عَبْدِ القَيْسِ (9هـ=630م) يُخْبِرُ النَّبِيَّ ﷺ عَنْ قُسِّ بْنِ ساعِدَةَ الإِيادِيِّ، ويَصِفُ طَريقَهُ إِلَيْهِ:

"خَرَجْتُ في جاهِلِيَّتي أَبْغي بَعيرًا شَرَدَ مِنِّي، أَقْفو أَثَرَهُ في تَنائِفَ حِقافٍ ذاتِ ضَغابيسَ، وعَرَصاتِ جَثْجاثٍ".

ابن عساكر، تاريخ دمشق. تح: العمروي، ج: 73، ص: 255.— المصدر
”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«لا بَأْسَ باجْتِناءِ الضَّغابيسِ فِي الحَرَمِ».

الهروي، الغريبين في القرآن والحديث. تح: المزيدي، ج: 1، ص: 1129.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخلِيلُ (ت: 170هـ=787م):

"الضَّغابيسُ: شِبْهُ العَراجينِ تَنْبُتُ بِالغَوْرِ في أُصولِ الثُّمامِ، طِوالٌ حُمْرٌ رَخْصَةٌ تُؤْكَلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 460.— المصدر
”‌وقالَ ابْنُ قُتَيْبَةَ (ت: 276ه=889) يَذْكُرُ رَأْيَ عَطاءٍ في قَطْعِ شَجَرِ الشِّبْرِقِ وَالضَّغابيسِ لِلتَّداوي:

"لَمْ يَرَ بَأْسًا أَنْ يُؤْخَذَ هَذانِ مِنَ الحَرَمِ لِيُتَداوى بِهِما كَما لَمْ يَرَ بَأْسًا بِأَخْذِ الشِّبْرِقِ ‌وَالضَّغابِيسِ".

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 3، ص: 665.— المصدر
1.4معنى من معاني «ضُغْبوس»

وــــــــــ: وَلَدُ الثَّعْلَبَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الضُّغْبوسُ: وَلَدُ الثُّرْمُلَةِ، وهِيَ الثَّعْلَبَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 461.— المصدر
1.5معنى من معاني «ضُغْبوس»

وــــــــــ مِنَ الإِبِلِ: البَعيرُ لا هُوَ بسَمِينٍ ولا مُسِنٍّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"الضُّغْبوسُ: البَعيرُ الَّذي لَيْسَ بِسَمِينٍ ولا مُسِنٍّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 160.— المصدر
1.6معنى من معاني «ضُغْبوس»

وــــــــــ مِنَ الشَّيَاطينِ ونَحْوِها: الخَبيثُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَهْ (ت: 458هـ=1066م):

"الضُّغْبوسُ: الخَبيثُ مِنَ الشَّيَاطينِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 79.— المصدر
