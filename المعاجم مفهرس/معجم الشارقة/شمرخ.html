شمرخ
شَمْرَخَ(فعل) شِمْراخ(اسم) شُمْروخ(اسم)

المعانى الكلّيّة

  1. المعاني الكلية,العُلُوُّ والارتِفاعُ
01 —

شَمْرَخَ

فعل
1.1معنى من معاني «شَمْرَخَ»

* شَمْرَخَ الشَّيْءُ وَجْهَ فُلانٍ: جَعَلَهُ وَضيئًا مُشْرِقًا. (ولَمْ يُسْتَعْمَلِ الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ واسْتُعْمِلَ المُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الرّومِيِّ (ت: 283هـ=896م) يَمْدَحُ:

هُوَ الطِّرْفُ أَجْرتْهُ المُلوكُ ومَسَّحَتْ
قَديمًا لَهُ وَجْهًا أَغَرَّ مُشَمْرَخا

ديوان ابن الرومي، تح: بسج، ج: 1، ص: 367.— المصدر
1.2معنى من معاني «شَمْرَخَ»

وـــــــــ الأَمْرُ فُلانًا: سَيَّدَهُ ورَفَعَ مكانَتَهُ. (ولَمْ يُسْتَعْمَلِ الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ واسْتُعْمِلَ المُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الرُّومِيِّ (ت: 283هـ=896م) يَمْدَحُ أحَدَ الوُلاةِ:

أَغَرُّ لا تُنْكِرُهُ مُشَمْرَخُ
آباؤُهُ في المُلْكِ قِدْمًا تُنَّخُ

ديوان ابن الرومي، تح: بسج، ج: 1، ص: 369.— المصدر
1.3معنى من معاني «شَمْرَخَ»

وـــــــــ فُلانٌ النَّخْلَةَ: انْتَزَعَ بُسْرَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"شَمْرَخَ النَّخْلَةَ: إِذَا خَرَطَ بُسْرَها". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 2، ص: 1145.— المصدر
1.4معنى من معاني «شَمْرَخَ»

وــــــــــ العِذْقَ: خَرَطَ عَثَاكِلَهُ بِالمِخْلَبِ قَطْعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبُو صَبْرَةَ السَّعْدِيُّ (ت: 340هـ=952م):

"شَمْرِخِ العِذْقَ؛ أَيِ: اخْرُطْ شَمارِيخَهُ بِالمِخْلَبِ قَطْعًا". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: سرحان، ج: 7، ص: 647.— المصدر
02 —

شِمْراخ

اسم
2.1معنى من معاني «شِمْراخ»

* الشِّمْراخُ مِنَ الجَبَلِ: رَأْسٌ مُسْتَدِيرٌ طَويلٌ دَقيقٌ في أعْلاهُ، (ج) شَماريخُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ لَقيطُ بْنُ يَعْمُرَ (ت: 249ق هـ=380م) يَصِفُ جَيْشًا:

لَوْ أَنَّ جَمعَهُمُ راموا بِهَدَّتِهِ
شُمَّ الشَّماريخِ مِنْ ثَهْلانَ لانصَدَعا

ديوان لقيط بن يعمر، تح: ألتونجي، ص: 78.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الشِّمْراخُ مِنَ الجَبَلِ: مُسْتَدِقٌّ طَويلٌ في أَعْلاهُ".

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 367.— المصدر
”وقالَ ابْنُ مُناذِرٍ (ت: 198هــ=813م) يَذْكُرُ فِعْلَ الدَّهْرِ بِجَبَلِ رَضْوى:

يَقْدَحُ الدَّهْرُ فِي شَمارِيخِ رَضْوى
وَيَحُطُّ الصُّخُورَ مِنْ هَبُّودِ

شعر ابن مناذر، تح: محمد غريب، ص: 141.— المصدر
2.2معنى من معاني «شِمْراخ»

وــــــــــ مِنَ النَّخْلِ: الغُصْنُ الَّذي يَنْبُتُ عَلَيْهِ البُسْرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80 ق هـ=544م) يَتَغَزَّلُ:

فَلَمّا تَنازَعْنا الحَديثَ وأَسْمَحَتْ
هَصَرْتُ بِغُصْنٍ ذي شَماريخَ مَيّالِ

ديوان امرئ القيس، تح: عليان وآخر، ج: 1، ص: 329.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُخَفِّفُ عَنْ رَجُلٍ ضَعيفٍ وَجَبَتْ عَلَيْهِ مِئَةُ جَلْدَةٍ:

«فَخُذوا لَهُ عِثْكالًا فيهِ مِئَةُ شِمْراخٍ، فَاضْرِبوهُ بِهِ ضَرْبَةً واحِدَةً، وخَلّوا سَبيلَهُ»

مسند أحمد، تح: الأرنؤوط وآخرين، ج: 36، ص: 363.— المصدر
”وقالَ عُقْبَةُ بْنُ النُّعْمانِ (ت: 20هـــ=641م) يَفْخَرُ:

وَفَيْنا وَفِينا يَفيضُ الوَفاءُ
وفينا يُفَرِّخُ أَفْراخَهُ

كَذاكَ الوَفاءُ يَزينُ الرِّجالَ
كَما زَيَّنَ الصَّدْقُ شِمْراخَهُ

ابن حجر، الإصابة، تح: عادل عبد الموجود وآخر، ج: 5، ص: 101.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الشِّمْراخُ: عِسْقِبَةٌ مِنْ عِذْقٍ أَوْ عُنْقُودٍ".

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 325.— المصدر
”وقالَ أَبو نُواسٍ (ت: 198هــ=813م) يَتَغَزَّلُ:

وحَسْبُكَ الحُسْنُ في ضَفائِرِها
مِثْلُ الشَّماريخِ في العَراجينِ

ديوان أبي نواس، تح. الحديثي، ص: 658.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بْنُ عَبْدِ السَّلامِ (ت: 660هـ=1262م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿وَخُذۡ بِيَدِكَ ‌ضِغۡثٗا فَٱضۡرِب بِّهِۦ وَلَا تَحۡنَثۡۗ﴾ [ص: 44]:

"... أَوْ مِلْءَ الكَفِّ مِنَ الحَشيشِ أَوِ الشَّجَرِ، أَوِ الشَّماريخِ، وذلِكَ خاصٌّ لِأَيّوبَ، عَلَيْهِ الصَّلاةُ والسَّلامُ".

ابن عبد السلام، تفسير القرآن، تح: الوهبي، ج: 3، ص: 86.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجَبَرْتِيُّ (ت: 1237هـ=1822م) يُقَدِّمُ لِكِتابِهِ:

"وهذا أَوانُ انْشِقاقِ كَمائِمِ طَلْعِ الشَّماريخِ، عَنْ زَهْرِ مُجْمَلِ التّاريخِ".

الجبرتي، عجائب الآثار، ج: 1، ص: 24.— المصدر
2.3معنى من معاني «شِمْراخ»

وــــــــــ مِنَ السَّحابِ: أَعَاليهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80 ق هـ=544م) يُخاطِبُ صاحِبَهُ:

أَعِنّي عَلى بَرْقٍ أَراهُ وَميضِ
يُضيءُ حَبِيًّا ذي شَماريخَ بيضِ

ديوان امرئ القيس، تح: عليان وآخر، ج: 2، ص: 458.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"شِمْرَاخُ السَّحابِ: أَعَاليهِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم، تح: هنداوي، ج: 5، ص: 328.— المصدر
2.4معنى من معاني «شِمْراخ»

وــــــــــ مِنَ الفَرَسِ: غُرَّتُهُ إِذا دَقَّتْ وطالَتْ، وجَلَّلَتِ الخَيْشومَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بْنُ حَجَرٍ (ت: 2ق هـ=620م) في مَقامِ رِثاءِ فَضالَةَ بْنَ كِلْدَةَ:

أَوْهَبَ مِنهُ لِذي أَثرٍ وسابِغَةٍ
وقَيْنَةٍ عِندَ شَرْبٍ ذاتِ أَشْكالِ

وخارِجِيٍّ يَزُمُّ الأَلْفَ مُعتَرِضًا
وهَوْنَةٍ ذاتِ شِمراخٍ وأَحجالِ

ديوان أوس بن حجر، تح: نجم، ص: 102.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُرَيْثٌ الطَّائِيُّ (ت: 37هـ=658م) يَصِفُ فَرَسًا:

تَرَى الجَوْنَ ذا الشِّمْراخِ والوَرْدَ يُبْتَغى
لَيالِيَ عَشْرًا وَسْطَنا وهْوَ عائِرُ

ابن منظور، لسان العرب، أمين محمد، ج: 7، ص: 192.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الشِّمْراخُ مِنَ الغُرَّةِ: ما سالَ عَلى الْأَنْفِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 325.— المصدر
2.5معنى من معاني «شِمْراخ»

وــــــــــ مِنَ الصُّبْحِ: ما لاحَ مِنْ نُورِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُعْتَزِّ (ت: 296هـ=909م):

يا صاحِ غادِ الخَنْدَريسَ فَقَد بَدا
شِمْراخُ صُبْحٍ لاحَ في الظُّلُماتِ

ديوان ابن المعتز، دار صادر، ص: 114.— المصدر
03 —

شُمْروخ

اسم
3.1معنى من معاني «شُمْروخ»

* الشُّمْروخُ مِنَ العِنَبِ: غُصْنٌ دَقيقٌ رَخْصٌ يَنْبُتُ في أَعْلى الغُصْنِ الغَليظِ، خَرَجَ مِنْ سَنَتِه رَخْصًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الشُّمْرُوخُ: غُصْنٌ دَقيقٌ في أَعْلى الغُصْنِ الغَليظِ، خَرَجَ مِنَ سَنَتِهِ دَقيقًا رَخْصًا". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 325.— المصدر
3.2معنى من معاني «شُمْروخ»

وــــــــــ مِنَ النَّخْلِ: العِثْكالُ، وهُوَ الَّذي عَلَيْهِ البُسْرُ وأَصْلُهُ في العِذْقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"الشِّمْرَاخُ: هُوَ الَّذي عَلَيْهِ البُسْرُ وأصْلُهُ في العِذْقِ، ويُقال لَهُ: الشُّمْرُوخُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: سرحان، ج: 7، ص: 646.— المصدر
3.3معنى من معاني «شُمْروخ»

وــــــــــ: أُنْبوبٌ مَعْدِنيٌّ، لَهُ غِطاءٌ مَحْشُوٌّ بِمادَّةٍ قابِلَةٍ لِلاشْتِعالِ تَنْفَجِرُ مُحْدِثَةً صَوْتًا قَويًّا؛ يُسْتَعْمَلُ في استِغاثَةِ السُّفُنِ حينَ الغَرَقِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ اللهِ النَّديمُ (1298هـ=1881م)، تاريخُ الطَّبْعِ، يَنْتَقدُ الشَّعْبَ:

"ومَتى نُفيقُ مِنْ سَكْرَةِ الغَفْلَةِ؟!، وما بَقِيَ عَلى حُكومَتِنا إِلّا أَنْ تُنَبِّهَنا بِالعَصا أَوْ توقِظَنا بِالشُّمْروخِ!".

ابن النديم، التجارة البائرة، مجلة التنكيت والتبكيت، عدد 11، ص: 172.— المصدر
