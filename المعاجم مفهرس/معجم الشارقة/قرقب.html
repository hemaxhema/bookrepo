قرقب
قَرْقَب / قُرْقُب / قُرْقُبّ(اسم) قُرْقُب(اسم) قَرْقَبة(اسم) قُرْقُبَّة(اسم) قُرْقُبِيّ(اسم)
النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذرُ (ق ر ق ب):

في الآراميّة؛ ومنه الاسْمُ:קוֹרְקְבָן (ق۫ۈرقْڤَان)qōrqəḇān بمَعْنى "مَعِدَة/ حَوصَلَةُ الطَّائِر"؛ وهُوَ يَتَّفِقُ مع الاسْمِ العَرَبيِّ (قُرقُب) مَبْنًى ومَعْنًى؛ ومن مَجيئِهِ بمَعْنى "قُرقُب" ما وَرَدَ في التَّلمودِ البابِليِّ: "והאההיאבראווזאדהוהבימרשמואלדלאהוהקאמקלףקורקבניהואותביהבשימשא"؛ أَيْ "كانَتْ هُناكَ إِوَزَّةٌ في بيتِ السَّيدِ صَموئيل ولَمْ يُمْكِنْ نَزْعُ قَرْقبِها فَوَضَعوها في الشَّمس".

(חול': סב,ב)

المعانى الكلّيّة

  1. صَوْتُ البَطْن
  2. صِغارُ الطَّيْر
01 —

قَرْقَب / قُرْقُب / قُرْقُبّ

اسم
1.1معنى من معاني «قَرْقَب / قُرْقُب / قُرْقُبّ»

* القَرْقَبُ والقُرْقُبُ والقُرْقُبُّ: البَطْنُ. (ج) قَراقِبٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ اللَّيْثُ بن المُظَفَّر (ت: 180ه=796م):

"والقُرْقُبُ: البَطْنُ".

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 419.— المصدر
02 —

قُرْقُب

اسم
2.1معنى من معاني «قُرْقُب»

* القُرْقُبُ: الصِّغارُ مِن الطَّيرِ كالصِّعاءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ اللَّيْثُ بن المُظَفَّر (ت: 180ه=796م):

"‌القُرْقُبُ: الصِّغارُ مِن الطَّيرِ، نَحوٌ مِن الصَّعْوِ".

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 418.— المصدر
”

وقالَ الراجز (206ه=821م) من شَواهِدِ أبي عَمْرو، يَذْكُرُ اجْتِماعَ الطُّيورِ:

اجْتَمَعَ اليَومَ عَلى شأْنِ الحُمَرْ

القُرقُبُ الجُونُ الصِّغارُ والقُبَرْ

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 113.— المصدر
2.2معنى من معاني «قُرْقُب»

وــــــــــ من الرِّجالِ: الشَّروبُ طَويلًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابن عَبّاد (ت: 385ه=995م):

"‌وإنَّهُ ‌لَشَروبٌ قُرْقُبٌ: إذا كانَ شَروبًا طَويلًا".

ابن سيده، المحكم، تح: هنداوي، ج: 5، ص: 406.— المصدر
03 —

قَرْقَبة

اسم
3.1معنى من معاني «قَرْقَبة»

* القَرْقَبَةُ: صَوتُ البَطْنِ إذا اشْتَكى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو عَمْرو الشَّيبانيّ (ت: 206ه=821م):

"القَرْقَبَةُ: صَوتُ ‌البَطْنِ إذا اشْتَكى".

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 419.— المصدر
04 —

قُرْقُبَّة

اسم
4.1معنى من معاني «قُرْقُبَّة»

* القُرْقُبَّةُ: لَحْمَةُ الصَّيدِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ الفِيروزابادي (ت: 817ه=1414م):

"‌القُرْقُبُ، كقُنْفُذٍ وجَعْفَرٍ وزُخْزُبٍّ: البَطْنُ (.....) وكزُخْزُبَّةٍ: ‌لَحْمَةُ الصَّيدِ".

الفيروزابادي، القاموس. تح: العرقسوسي، ص: 124.— المصدر
05 —

قُرْقُبِيّ

اسم
5.1معنى من معاني «قُرْقُبِيّ»

* القُرْقُبيُّ: ثيابٌ بيضٌ مِن كَتَّانٍ. وهي بهاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"قُرْقُبِيٌّ -بقافَينِ- (...) ‌ثَوبٌ ‌مِن ‌الكَتَّانِ ‌الأبْيَضِ".

اليفرنّي، الاقتضاب في غريب الموطأ. تح: عبد الرحمن العثيمين، ج: 2، ص: 215.— المصدر
”

وقالَ مالكُ بن أنَس (ت: 179ه=795م) في جَوازِ تَسْليفِ ثيابِ القُطْنِ بعضِهِ في بَعْضٍ:

"وكَذَلِكَ الكَتّانُ رَقيقُهُ كُلُّهُ واحِدٌ؛ ‌القُرْقُبيُّ والشَّطَويُّ والقَصَبيُّ كُلُّهُ واحِدٌ، ولا بأسَ بِهِ في الزّيَقَةِ والـمَريسيَّةِ، وذَلِكَ أنَّها غِلاظٌ كُلُّها".

سحنون، المدونة الكبرى. ج: 3، ص: 73.— المصدر
