01 —

قافَ

فعل
1.1معنى من معاني «قافَ»

* قافَ فُلانٌ فُلانًا، وعَلَيْهِ يَقوفُ قَوْفًا وقِيافَةً: قَذَفَهُ بالظَنِّ، وقالَ فيه كَلامًا بِغَيْرِ عِلْمٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجزُ (315ه=927م) مِنْ شَواهِدِ الأخْفَش الأصْغَر:

أَعوذُ بِاللهِ الجَليلِ الأَعْظَمِ

مِنْ قَوْفِيَ الشَّيْءَ الَّذي لَمْ أَعلمِ

[ابنُ عبّادٍ: وقُفْتُ عليهِمْ أَقوفُ قَوْفًا وقِيافَةً: وهُوَ الظَّنُّ ورَجْمُ الغَيْبِ]
الأزهري، تهذيب اللغة. تح: عبد السلام هارون، ج: 9، ص: 326.— المصدر
1.2معنى من معاني «قافَ»

وــــــــــ الشَّيْءَ: تَتَبَّعَ أثرَهُ، فهُوَ قائِفٌ وقَوّافٌ وقَيّافٌ. (انْظُرُ:ق ف و)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الأسْوَدُ بن يَعْفُر (ت: 22ق.ه=600م) يُغْري مُخاطَبَهُ باتِّباعِهِ:

كَذَبْتُ عَلَيْكَ لا تَزالُ تَقوفُني
كَما ‌قافَ آثارَ الوَسيقَةِ قائِفُ

[الوَسيقَةُ: الطَّريدَةُ من الإِبِلِ]
ديوان الأسود بن يعفر. تح: نوري القيسي، ص: 50.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مِسْكينُ الدّارميّ (ت: 89ه=708م) في مَعادِنِ الرِّجالِ:

وإِنِّي لأَعْرِفُ سيْما الرِّجالِ
كَما يَعْرِفُ ‌القائِفونَ الأَثَرْ

ديوان مسكين الدارميّ. تح: كارين صادر، ص: 42.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"ومَعَ هذا أنَّهُمْ أَدْخَلوا الفِعالَ، كَما قالوا: النِّفارُ والنُّفورُ، وشَبَّ شِبابًا وشُبوبًا، فهذا نَظيرُهُ من العِلَّةِ، وقالوا: ناحَ يَنوحُ نِياحةً، وعافَ يَعوفُ عِيافَةً، وقافَ يَقوفُ قِيافَة؛ فِرارًا مِن الفُعولِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

سيبويه، الكتاب. تح: عبد السلام هارون، ج: 4، ص: 51.— المصدر
”وقالَ الـمُتَنبّي (ت: 354ه=965م) يَصِفُ رحلتَهُ إلى مِصْرَ:

ولا اتَّبَعَتْ آثارَنا عَيْنُ قائِفٍ
فَلَمْ تَرَ إِلّا حافِرًا فَوْقَ مَنْسِمِ

أبو العلاء المعري، اللامع العزيزي. تح: محمد سعيد المولوي، ص: 1419.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الصَّيْقَل الجَزَريّ (ت: 701ه=1301م) في ِالمَقامَةِ البَغْداديَّةِ يَصِفُ عُرْسَ صَديقٍ لهُ:

"فأَقْبَلْتُ بَعْدَما كَمَّلَ وَطَرَهُ، وحَمَّلَ حِقَّ الحَيْلِ وقَطَرَهُ، ‌أَقوفُ أَثرَهُ، وأذمُّ مَخْبَرَهُ".

ابن الصيقل الجزري، المقامات الزينية. تح: عباس الصالحي، ص: 91.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الـخَضِر حُسَيْن (ت: 1377ه=1958م) يَصِفُ القُرْآنَ الكَريمَ:

عِصْمَةٌ مِنْ كُلِّ ما تَكْبو بِهِ
نَظَراتُ الفَيْلَسوفِ القائِفَهْ

الخضر حسين، موسوعة الأعمال الكاملة تح: علي الحسيني، ج: 13، ص: 142.— المصدر
1.3معنى من معاني «قافَ»

وــــــــــ فُلانًا قِيافَةً: تَتَبَّعَ شَبَهَهُ بأَبيهِ أو أَخيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بن الخَطّاب (ت: 23ه=644م) يَنسُبُ رَجُلاً إِلى أَهْلِ نَجْرانَ من خِلالِ قِيافَةِ مَلامحِهِ:

"مَهُ، إِنَّا ‌نَقوفُ ‌الآثارَ".

عبد الرزاق الصنعاني، المصنف. تح: الأعظمي. ج: 7، ص: 448.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجاحظ (ت: 255ه=869م) يَصِفُ مَعْرِفَةَ النّبيِّ ﷺ بمَعارِفِ العَرَبِ:

"ويَقْصُرُهُ على مَعْرِفَةِ مَصالِحِ الدّينِ دونَ ما تَتَباهى بهِ العَرَبُ من قِيافَةِ الأَثَرِ والبَشَرِ، ومن العِلْمِ بالأَنْواءِ وبالخَيْلِ وبالأَنْسابِ وبالأَخْبارِ ".

الجاحظ، البيان والتبيين. تح: عبد السلام هارون، ج: 4، ص: 32.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أكمَلُ الدّين البابرتي (ت: 786ه=1384م) يَشْرَحُ رأيَ الشافِعيِّ في حُكْمِ القيافَةِ في النَّسَبِ شَرْعًا:

"ولهَذا تَمَسَّكَ الشّافعيُّ -رَحِمَهُ اللهُ- في القيافَةِ بالاسْتِبْشار ِوتَرْكِ الإِنْكارِ لقَوْلِ الـمُدْلِجيِّ".

البابرتي، الردود والنقود. تح: ضيف الله العمري، ج: 1، ص: 502.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الرّافعيّ (ت: 1356ه=1937م):

"وكانَ العَرَبُ يَعْرِفونَ هذِهِ القِيافَةَ، ولكِنَّهُمْ يَسْتَعْمِلونَها في تَحْقيقِ الأَنْسابِ وتَمَيُّزِ القّبائِلِ".

الرافعي، تاريخ الأدب العربي، ج: 2، ص: 28.— المصدر
02 —

اقْتافَ

فعل
2.1معنى من معاني «اقْتافَ»

* اقْتافَ فُلانٌ الشَّيْءَ: تَتَبَّعَ أَثَرَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَوْفُ بن الأحْوَص (ت: 22ق.ه=601م) يَفْخَرُ بحِمايتِهِ لعِرْضِهِ، ويُشَبِّهُ نَفْسَهُ بالـمُقْتافِ الماهِرِ:

أَلَمْ أَظْلِفْ عَنِ الشُّعَراءِ عِرْضي
كَما ظُلِفَ الوَسيقَةُ بِالكُراعِ

فَلا ‌أَقْتافُ إِلّا فَوْقَ قُفٍّ
﻿ يَزِلُّ بِذي الـحَوافِرِ أَوْ بِقاعِ

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص. ج: 1، ص: 25.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الرّوميِّ (ت: 283ه=896م) يهجو الشّاعِرَ ابنَ الخَبّازَةِ، ويَصِفُ رداءَةَ شِعْرِهِ:

شِعْرُك لا تثْبُتُ آثارُهُ
من غُرَّةِ اليومِ إلى اللَّيْلِ

مَدَبُّ ذَرٍّ فى نَقا هائلٍ
مَرَّتْ بهِ مُعْصِفَةُ الذّيلِ

عَفا فما يَسْطيعُ ‌يَقْتافُهُ
ناظِرُ لُقْمانَ ولا قَيْلِ

[الأزْهَريّ: واقتافَ أَثَرَهُ اقتِيافاً: إِذا تَبِع أَثَرَهُ]
المرزباني، الموشح. تح: علي محمد البجاوي، ص: 475.— المصدر
03 —

تَقَوَّف

فعل
3.1معنى من معاني «تَقَوَّف»

* تَقَوَّفَ فُلانٌ: قالَ الباطِلَ والظنَّ الذي لا يُفْضي إلى حَقيقَةٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَرير (ت: 110ه=728م) يَشْكو انْقِطاعَ وَصْلِهِ لأحْبابِهِ خَشْيَةَ الرُّقَباءِ:

وطالَ حذاري غُرْبَةَ البَيْنِ والنَّوى
وأُحْدوثَةً من كاشِحٍ يَتَقَوَّفُ

شرح ديوان جرير. تح: محمد الصاوي، ص: 380.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجصّاص (ت: 370ه=981م):

"وقَدْ كانَ هَذا الِاسْمُ مَوْضوعًا عِنْدَهُمْ لِما يُخْبِرُ بِهِ الإِنْسانُ مِنْ غَيْرِ حَقيقَةٍ، يَقولونَ: ‌تَقَوَّفَ الرَّجُلُ إذا قالَ الباطِلَ".

الجصاص، أحكام القرآن. تح: محمد الصادق قمحاوي، ج: 5، ص: 28.— المصدر
3.2معنى من معاني «تَقَوَّف»

وــــــــــ على المالِ ونَحْوِه: حَجَرَ عليهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110ه=728م) يَهْجو ابنَ عيسى لِتَحْريضِهِ ابْنَي حُجَينٍ ليَمْنَعا عن الفَرَزْدَقِ العَطاءَ:

تَقَوَّفتَ مالَ ابْنَي حُجَينٍ وما هُما
كَذي حَطْمَةٍ فانٍ ولا ضَرَعٍ غَمْرِ

ديوان الفرزدق. تح: علي فاعور. ج: 1، ص: 263.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بن شُمَيْل (ت: 203ه=819م):

" فُلانٌ يَتَقَوَّفُ على مالي؛ أي: يَحْجُرُ عليَّ فِيهِ".

الأزهري، تهذيب اللغة. تح: عبد السلام هارون، ج: 9، ص: 330.— المصدر
3.3معنى من معاني «تَقَوَّف»

وــــــــــ فُلانًا: أَخَذَ عليهِ في كَلامِهِ وقالَ لهُ: قُلْ كَذا وكَذا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بن شُمَيْل (ت: 203ه=819م):

"وهُوَ يَتَقَوَّفُني في الـمَجْلِسِ؛ أَيْ: يَأْخُذُ عليَّ في كَلامي ويَقولُ: قُلْ كَذا وكَذا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: عبد السلام هارون، ج: 9، ص: 330.— المصدر
3.4معنى من معاني «تَقَوَّف»

وــــــــــ الشَّيءَ: تَتَبَّعَ أَثَرَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ نَهْشَلُ الدّارميّ (ت: 45ه=665م) يَصِفُ حالَهُ بَعْدَ أنْ مَسَّهُ الكِبَرُ:

وقَدْ كُنْتُ بالبيدِ القَليلِ أَنيسُها
أَقوفُ وأمضي قَبْلَ منْ ‌يَتَقَوَّفُ

فأَصْبَحْتُ ممَّا يُحْدِثُ الدَّهرُ لِلْفتى
أقُصُّ العَلاماتِ التي كُنْتُ أَعْرِفُ

محمّد بن المبارك، منتهى الطّلب. تح: طريفي، ص: 3034.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"وقافَ الأَثَرَ قِيافَةً، واقْتافَهُ وتَقَوَّفَهُ: تَتَبَّعَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 576.— المصدر
”وقالَ أبو الـمُظَفَّر الأَبِيوَرْديّ (ت: 507ه=1113م) يَصِفُ رَكْبًا:

وركْبٍ على الأكْوارِ غِيدٍ من الكرى
تَداوَلَـهُمْ سَيْرٌ حَثيثٌ ونَفْتَفُ

ترى العِتْقَ مِنْهُمْ في وُجوهٍ شَواحِبٍ
يُرَدِّدُ فيها لَـحْظَهُ الـمُتَقَوِّفُ

ديوان الأبيوردي. ص: 208.— المصدر
04 —

قائِف

اسم
4.1معنى من معاني «قائِف»

* القائِفُ: العارِفُ بتَتَبُّعِ الآثارِ وتَشابُهِ النّسَبِ. (ج) قافَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُرْوةُ بن الزُّبَيْر (ت: 94ه=713م):

"إنَّ رَجُلَيْنِ ادَّعَيا وَلَدًا: فدَعا عُمَرُ ‌القافَةَ، واقْتَدى في ذَلِكَ بِبَصَرِ ‌القافَةِ، وأَلْحَقَهُ أَحَدَ الرَّجُلَيْنِ".

الصنعاني، المصنف. ج: 7، ص: 305.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُقاتلُ بن سُلَيْمان (ت: 150ه=767م) يُفَسِّرُ قَوْلَ الله سُبْحانَهُ وتَعالى: ﴿إِذۡ يَقُولُ ‌لِصَٰحِبِهِۦ لَا تَحۡزَنۡ إِنَّ ٱللَّهَ مَعَنَا﴾ [التوبة: 40]:

"قال لأبي بكرٍ: لا تَحْزَنْ إِنَّ الله مَعَنا في الدَّفْعِ عَنّا، وذَلِكَ حينَ خافَ القافَةَ حولَ الغارِ، فقالَ أبو بَكْرٍ: أُتينا يا نَبيَّ اللهِ، وحَزِنَ أَبو بَكْرٍ".

مقاتل بن سليمان، تفسيره. تح: عبد الله شحاتة، ج: 2، ص: 171.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوَويّ (ت: 676ه=1277م):

"لَوْ تَداعى اثْنانِ طِفْلًا ولَيْسَ هُناكَ قائِفٌ، فانْتَسَبَ بَعْدَ بُلوغِهِ إِلى أَحَدِهِما، ثُمَّ وَجَدْنا قائِفًا؛ فإِنّا نُقَدِّمُ القائِفَ على إِخْبارِهِ".

النووي، المجموع شرح المهذب. تح: المطيعي، ج: 2، ص: 57.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُصْطَفى السُّيوطيّ (ت: 1243ه=1827م) يَشْرحُ ما يُشْتَرَطُ في القائِفِ كي يُعْتَمَدَ حكمُهُ في نسبةِ الوَلَدِ الذي يَتَنازَعُهُ أَكْثَرُ من أَبٍ:

"ويُعْتَبَرُ عَدالَةُ القائِفِ وذكوريَّتُهُ وكثرةُ إِصابتِهِ، ولا تُعْتَبَرُ حُرِّيَّتُهُ كالشّاهِدِ، ويَكْفي قائِفٌ واحِدٌ؛ لأَنَّهُ يُنَفَّذُ ما يَقولُهُ فيه كالحاكِمِ، ولا يُبْطَلُ قَوْلُ القافَةِ بقَوْلِ قافَةٍ أُخْرى".

مصطفى السّيوطيّ، مطالب أولي النّهى، ج: 5، ص: 551.— المصدر
4.2معنى من معاني «قائِف»

وــــــــــ: (في اصْطِلاحِ الفُقَهاءِ): الَّذي يَعْلَمُ النَّسَبَ بِفِراسَتِهِ ونَظَرِهِ إِلى أَعْضاءِ المـَوْلودِ، ويَعْرِفُ شَبَهَ الرَّجُلِ بِأَخيهِ وأَبيهِ، ويَتَتَبَّعُ الآثارَ ويَعْرِفُها، والقيافَةُ اسْمٌ لِهَذا العِلْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعي (ت: 204ه=820م) يُبَيِّنُ أَنَّهُ يُقْبَلُ القائِفُ الواحِدُ:

"ويَكْفي ‌القائِفُ الواحِدُ؛ لِأَنَّ هَذا مَوْضِعُ حُكْمٍ بِعِلْمٍ لا مَوْضِعُ شَهادَةٍ، ولَوْ كانَ إِنَّما حُكْمُهُ حُكْمُ الشَّهاداتِ ما أَجَزْنا غَيْرَ اثْنَيْنِ".

الشّافعي، الأمّ. ج: 6، ص:265.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عَبْدِ السَّلامِ (ت: 660ه=1262م) يُبَيِّنُ الحُكْمَ إِذا انْعَدَمَ القائِفُ في الاسْتِدْلالِ عَلى النَّسَبِ:

"إِذا عُدِمَ ‌القائِفُ، أَوْ تَحَيَّرَ، أَوْ غابَ، أَوْ جُرِّبَ، فَغَلِطَ، وُقِفَ الأَمْرُ إِلى أَنْ يَبْلُغَ الصَّبِيُّ فَيَنْتَسِبَ إِلى أَحَدِهِما ...".

عز الدين بن عبد السَّلام، الغاية. ج: 8، ص: 152.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ البُجَيْرَمِيّ (ت: 1221ه=1806 م) يُبَيِّنُ جَوازَ القيافَةِ، وأَنَّها حَقٌّ لِفَرَحِ رَسولِ اللهِ ﷺ بِشَهادَةِ القائِفِ:

"فَإِقْرارُهُ ﷺ وسُرورُهُ بِهِ يَدُلُّ عَلى أَنَّ القيافَةَ حَقٌّ، ووَجْهُ الرَّدِّ عَلى الـمُنافِقينَ أَنَّهُمْ كانوا يُسَلِّمونَ الحُكْمَ ‌بِالقائِفِ؛ لِأَنَّهُ كانَ أَمْرًا مَعْروفًا عِنْدَهُمْ".

سليمان البُجَيْرَمِيّ، التجريد لنفع العبيد. ج: 4، ص: 411.— المصدر
05 —

قاف

اسم
5.1معنى من معاني «قاف»

* القافُ: الحَرْفُ الحادي ‌والعِشْرونَ مِنْ حُروفِ الهِجاءِ حَسَبَ التَّرْتيبِ الأَلِفْبائيِّ، والتّاسِعَ عَشَرَ في التَّرْتيبِ الأَبْجَديِّ، يُساوي في حِسابِ الجُمَّلِ الرَّقْمَ مِئَةً، وهُوَ حَرْفٌ لَهَويٌّ مَجْهورٌ شَديدٌ مُفَخَّمٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"القُدّامُ: القادِمونَ مِنْ سَفَرٍ، جَمْعُ قادِمٍ. وقيلَ: القَدّامُ، بِفَتْحِ ‌القافِ".

الخليل، العين. تح: المخزومي وآخر، ج: 1، ص: 172.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّارِ (ت: 658ه= 1260م) يُتَرْجِمُ لِلوَزيرِ أَبي جَعْفَرٍ:

"وهُوَ مِنْ بَيْتِ القاضي أَبي الوَليدِ هِشامِ بْنِ أَحْمَدَ الوَقَّشيِّ، وهِيَ قَرْيَةٌ بِنَواحي ‌طَلَبيرَةَ، مُشَدَّدَةُ ‌القافِ".

ابن الأبّار، الحلة السيراء. تح: مؤنس، ج: 2، ص: 257.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ البُجَيْرَميُّ (ت:1221ه=1806م) يَشْرَحُ مَعْنى اللُّقَطَةِ:

"(كِتابُ اللُّقَطَةِ) هِيَ -بِضَمِّ اللّامِ، وفَتْحِ ‌القافِ وإِسْكانِها-لُغَة-: الشَّيْءُ المَلْقُوطُ".

البجيرمي، التجريد لنفع العبيد. ج: 3، ص: 221.— المصدر
06 —

قافِيَّة

اسم
6.1معنى من معاني «قافِيَّة»

* القافِيَّةُ (في الشِّعْرِ): القَصيدَةُ أَوِ المـَقْطوعَةُ الشِّعْرِيَّةُ الَّتي رَويُّها حَرْفُ القافِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَطيبُ البَغْداديُّ (ت: 463ه=1072م) يُتَرْجِمُ لصالِحِ بْنِ عَبْدِ القُدّوسِ:

"ومِنْ مُسْتَحْسَناتِ قَصائِدِ صالِحٍ ‌القَصيدَةُ ‌القافِيَّةُ (...)".

الخطيب البغدادي، تاريخ بغداد. تح: بشار معروف، ج: 10، ص: 413.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ سَعيدٍ المَغْرِبيُّ (ت: 685ه=1286م) يُتَرْجِمُ لأَبي بَكْرٍ عيسى بْنِ وَكيلٍ:

"وأَنْشَدَ لَهُ ‌القَصيدَةَ ‌القافِيَّةَ الـمَشْهورَةَ الَّتي قالَها في ابْنِ عَشْرَةٍ حينَ خَلَّصَهُ مِنْ السِّجْنِ (...)".

ابن سعيد المغربي، المغرب في حلى المغرب. ج: 2، ص:236.— المصدر
07 —

قُوف

اسم
7.1معنى من معاني «قُوف»

* قُوفُ الأُذُنِ: أعْلاها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والشَّنَفُ، مَجْزومٌ ومُتَحرِّكٌ: مِعْلاقٌ في ‌قُوفِ الأُذُنِ؛ أي: في أَعْلى الأُذُنِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 6، ص: 267.— المصدر
7.2معنى من معاني «قُوف»

وــــــــــ: الشَّعْرُ الـمُتَدَلّي في نُقْرَةِ القَفا، وهُوَ بهاءٍ أَيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأعْرابِيِّ (ت: 231ه=845م):

"يُقالُ: خُذْ بقُوفِ قَفاهُ وبقُوفَةِ قَفاهُ، وبقافِيَةِ قَفاهُ، (...)، كُلُّهُ بِمَعْنى قَفاهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد السلام هارون، ج: 9، ص: 330.— المصدر
7.3معنى من معاني «قُوف»

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"يُقالُ: أَخْذْتُهُ بقُوفِ رقبَتِهِ؛ أي: أَخَذْتُهُ كُلَّهُ".

الأزهري، تهذيب اللغة. تح: عبد السلام هارون، ج: 9، ص: 330.— المصدر
7.4معنى من معاني «قُوف»

o ودَفَعَهُ بقُوفِ رَقَبتِه: دَفَعَهُ بِنَفْسِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
