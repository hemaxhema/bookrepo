01 —

شِمْشار

اسم
1.1معنى من معاني «شِمْشار»

* الشِّمْشارُ: شَجَرَةٌ يُشْبِهُ وَرَقُها وَرَقَ الآسِ، عودُها أَصْفَرُ صُلْبٌ، لَها حَبٌّ أَسْوَدُ. (مع)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ عَبّاسٍ (ت: 68هـ=688م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿وَقَالَ ‌لَهُمۡ ‌نَبِيُّهُمۡ إِنَّ ءَايَةَ مُلۡكِهِۦٓ أَن يَأۡتِيَكُمُ ٱلتَّابُوتُ فِيهِ سَكِينَةٞ مِّن رَّبِّكُمۡ﴾. [البَقَرَة: 248]:

"كانَ التّابوتُ مِنْ عودِ الشِّمْشارِ".

ابن الجوزي، زاد المسير. ص: 152.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ السّائِبِ الكَلْبِيُّ (ت: 146هـ=763م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿وَقَالَ ‌لَهُمۡ ‌نَبِيُّهُمۡ إِنَّ ءَايَةَ مُلۡكِهِۦٓ أَن يَأۡتِيَكُمُ ٱلتَّابُوتُ فِيهِ سَكِينَةٞ مِّن رَّبِّكُمۡ﴾ [البقرة: 248]:

"وكانَ التّابوتُ مِنْ عودِ الشِّمْشارِ الَّذي يُتَّخَذُ مِنْهُ الأَمْشاطُ".

السمرقندي، بحر العلوم، تح: علي معوض وآخرَين، ج: 1، ص: 219.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العِبْرِيِّ (ت: 685هـ=1286م) يَرْوي بَعْضَ مَزاعِمِ بَني إِسْرائيلَ:

"وقال اللهُ لِموسى: قُلْ لِبَني إِسْرائيلَ يُفْرِدونَ لي ذَهَبًا وفِضَّةً ونُحاسًا ... وأَديمًا وخَشَبَ شِمْشارٍ".

ابن العبري، تاريخ مختصر الدول، تح: المنصور، ص: 22.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ يوسُف شِلْبُ الشّامِ (1420هـ=1999م)، تاريخُ الطَّبْعِ، يحكي على لِسانِ فتاةٍ تَصِفُ حالَها:

"أَصْبَحْتُ أَرْضًا جافَّةً يابِسَةً ... شِمْشارًا مَتْروكًا مُصْفَرًّا مِنَ الدّاخِلِ".

الطاهر بن جلون، نزل المساكين. تر: شلب الشّام، ص: 170.— المصدر
1.2معنى من معاني «شِمْشار»

وــــــــــ: السِّكّينُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ عَبّاسٍ (ت: 68هـ=688م) يَذْكُرُ زَعْمَ فارِسَ أَنَّ المَيِّتَ مِنَ الأنْعامِ ذَبَحَهُ اللهُ:

"بِشِمْشارٍ مِنْ ذَهَبٍ ...".

الطّبري، جامع البيان، تح: التركي، ج: 9، ص: 520، 521.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نُواسٍ (ت: 198هـ=814م) يَتَغَنّى بِالخَمْرِ:

أَسالَها مِنْ بُزالٍ
وَجاه بِالشِّمْشارِ

فسالَ مِنْها مُحاكٍ
مَوادِجَ البَيْطارِ

ديوان أبي نواس، تح: فاغنر، ج: 3، ص: 420.— المصدر
02 —

شِمْشير

اسم
2.1معنى من معاني «شِمْشير»

* الشِّمْشيرُ: السِّكّينُ. (مع)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

"بِشِمْشيرٍ مِنْ ذَهَبٍ ...".

الحميدي، تفسير ابن عبّاس، ص: 388.— المصدر
2.2معنى من معاني «شِمْشير»

وــــــــــ: ضَرْبٌ مِنَ السُّيوفِ، ذو نَصْلٍ ضَيِّقٍ وسَميكٍ، ولَهُ حَدٌّ واحِدٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمْسُ الدّينِ أَيْدَغْمُشُ (ت: 610هـ=1214م) يُخاطِبُ رسولَ قائِدِ بعضِ قِلاعِ الإسماعيليةِ لمّا عَرَضَ عليه النَّجدةَ:

"قُلْ لَهُ: إِنْ كُنْتَ مُسْلِمًا فأَريهِ، وإِنْ كُنْتَ كافِرًا فما لَكَ عِنْدي إِلّا شِمْشيرٌ".

الصفدي، الوافي بالوفيات، تح: الأرنؤوط وآخر، ج: 9، ص: 273.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمن زَكِي (1376هـ=1956م)، تاريخ الطَّبْعِ، يَتَحَدَّثُ عَنِ السُّيوفِ:

"ونُلاحِظُ أَنَّ أَوْجُهَ الاخْتِلافِ بَيْنَ القَليجِ التُّرْكِيِّ والشِّمْشيرِ الإيرانِيِّ يَنْحَصِرُ في الجُزْءِ النِّهائِيِّ مِنَ النَّصْلِ".

عبد الرّحمن زكي، السَّيف، ص: 76.— المصدر
2.3معنى من معاني «شِمْشير»

وــــــــــ: نَباتٌ هِنْدِيٌّ رائِحَتُهُ عَطِرَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ البَيْطارِ (ت: 646هـ=1248م):

"شِمْشيرٌ: هُوَ القاقُلَّةُ الصَّغيرَةُ".

ابن البيطار، الجامع، ج: 3، ص: 92.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ العَزيزِ العَظَمَةُ (ت: 1362هـ=1943م) يُعَدِّدُ أَنْواعَ الزُّهورِ الشّامِيَّةِ:

"ثُمَّ القَناديلُ ... والشِّمْشيرُ، والمَرْجانُ".

عبد العزيز العظمة، مرآة الشام، ص: 129.— المصدر
2.4معنى من معاني «شِمْشير»

وــــــــــ: نَباتٌ شَجَرِيٌّ قَصيرٌ، يُسْتَعْمَلُ لِلزّينَةِ المِعْمارِيَّةِ الدّاخِلِيَّةِ والخارِجِيَّةِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَبْدُ الصَّمَدِ المَوْزَعِيُّ (ت: 1031هـ=1622م) يَصِفُ قَصْرَ الأَميرِ حُسَين المومِيِّ:

"وجَعَلَ رَواشِنَهُ مِنَ الشِّمْشيرِ والسّاجِ".

الموزعي، دخول العثمانيّين، تح: الحبشي، ص: 72.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ حَسيب كَيّالِي (ت: 1413هـ=1993م) يَصِفُ الدّيارَ القَديمَةَ:

"وحَوْلَها أُصُصُ الأَزْهارِ كُلُّ عِرْقٍ بِلَوْنٍ، والشِّمْشيرُ الَّذي يَصْلُحُ مَخْبَأً في لُعْبَةِ أُمِّ عَميشٍ".

حسيب كيالي، الأعمال الروائية والقصصية، ج: 2، ص: 521.— المصدر
2.5معنى من معاني «شِمْشير»

وــــــــــ: شَجَرَةُ الصَّمْغِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ إِسماعيلُ القونَوِيُّ (ت: 1195هـ=1781م):

"شَجَرُ شِمْشيرٍ: شَجَرُ الصَّمْغِ".

حاشية القونوي على تفسير البيضاوي، تح: عبد الله عمر، ج: 5، ص: 349.— المصدر
