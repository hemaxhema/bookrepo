خردل
خَرْدَلَ(فعل) خَرادِل(اسم) خَراديل(اسم) خَرْدَل(اسم) خَرْدَلِيَّة(اسم) خُرْدولة(اسم)
المقدّمة
النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (خ ر د ل):

في الآراميّة: وَرَدَ الاسْمُ: חַרְדְּלָא (حَردْلا) ḥardəlā بِمَعْنى (نَباتٌ لَهُ حَبٌّ صَغيرٌ جدًّا) ويُوافِقُ الاسْمَ العَرَبِيَّ (خَرْدلٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسْمَ الآراميَّ خَضَعَ لِإبْدالِ الخاءِ حاءً، ومِنْ مَجيئِهِ بِمَعْنى (خَردَلٌ) ما وَرَدَ في التَّلْمودِ البابليِّ: "דאכלה חרדלא הוו לה בני זלזלני" أيْ: "مَنْ تَأكُلِ الخَرْدَلَ فَسَيَكونُ لَها أولادٌ شَرِهِينَ".

(כת': ס, ב)

في العِبْريّة: وَرَدَ الاسْمُ: חַרְדָּל (حَردال) ḥardāl بِمَعْنى (نَباتٌ تُستخدَمُ حُبوبُهُ في التَّوْبَلةِ والتَّداوي) ويُوافِقُ الاسْمَ العَرَبيَّ (خَرْدَلٌ) معنًى ومبنًى، ومِنْ مَجيئِهِ بِمَعْنى (خَرْدَل) ما وَرَدَ في المشنا: "אֵין סוֹמְכִין לִשְׂדֵה תְבוּאָה חַרְדָּל" أيْ: "لا تَجوزُ مُجاوَرةُ حَقلِ غَلّةٍ بِشَجَرِ خَرْدَلٍ".

(כלאים, ב', ח')

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܚܰܪܕ̊ܠܳܐ (حَردْلا) ḥardəlā بِمَعْنى (نَباتٌ لَهُ حَبٌّ صَغيرٌ جدًّا) ويُوافِقُ الاسْمَ العَرَبِيَّ (خَرْدَلٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسْمَ السُّرْيانِيَّ خَضَعَ لِإبْدالِ الخاءِ حاءً، ومِنْ مَجيئِهِ بِمَعْنى (خَرْدَلٌ) ما وَرَدَ في إنْجيلِ مَتّى: "ܕ̊ܳܡܝܳܐ ܡܰܠܟ̊ܽܘܬ̥ܳܐ ܕ̊ܰܫܡܰܝܳܐ܆ ܠܰܦ̥ܪܶܕ̊ܬ̊ܳܐ ܕ̊ܚܰܪܕ̊ܠܳܐ" أيْ: "يُشْبِهُ مَلَكُوتُ السَّماواتِ حَبّةَ خَرْدَل".

(ܡܬܝ، ܝܓ: 31)

المعانى الكلّيّة

  1. نَوْعٌ مِنَ النَّباتِ
  2. العُضْوُ الوافِرُ اللَّحْمِ
  3. اللَّحْمُ الـمُقَطَّعُ
01 —

خَرْدَلَ

فعل
1.1معنى من معاني «خَرْدَلَ»

*خَرْدَلَتِ النَّخْلَةُ: كَثُرَ نَفْضُها وعَظُمَ ما بَقِيَ مِنْ بُسْرِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"خَرْدَلَتِ النَّخْلَةُ، فهِيَ مُخَرْدِلٌ: إِذا كَثُرَ نَفْضُها، وعَظُمَ ما بَقِي مِنْ بُسْرِها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال، تح: حسين شرف، ج: 1، ص: 508.— المصدر
1.2معنى من معاني «خَرْدَلَ»

وــــــــــ: لَمْ تَقْبَلِ اللِّقاحَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م):

"‏ونَظَمَتِ النَّخْلَةُ: قَبِلَتِ اللِّقاحَ، وخَرْدَلَتْ: إِذا لَمْ تَقْبَلْ".

الزمخشري، أساس البلاغة، تح: عيون السود، ج: 2، ص: 284.— المصدر
1.3معنى من معاني «خَرْدَلَ»

وــــــــــ فُلانٌ فُلانًا: صَرَعَهُ ورَماهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يَصِفُ أَهْلَ النّارِ:

«ومِنْهُمُ الموبَقُ بِعَمَلِهِ، ومِنْهُمُ الـمُخَرْدَلُ أَوِ الـمُجازى».

صحيح البخاري، تح: العطار، ج: 1، ص: 1862.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَد (ت: 170هـ=786م):

"والـمُخَرْدَلُ: الـمَصْروعُ الـمَرْمِيُّ، في بَعْضِ الحَديثِ".

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 334.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ قَيِّمِ الجَوْزِيَّةِ (ت: 751هـ=1350م) يَصِفُ:

"وكَما أَرى الأَصْنامَ في الدُّنْيا، صُوَرُها مُكَسَّرَةٌ مُخَرْدَلَةٌ مُلْقاةٌ بِالأَمْكِنَةِ القَذِرَةِ".

ابن القيم، الضوء المنير، تح: الصالحي، ج: 6، ص: 220.— المصدر
1.4معنى من معاني «خَرْدَلَ»

وــــــــــ فُلانٌ اللَّحْمَ: قَطَّعَهُ وفَرَّقَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ القَتّالُ الكِلابِيّ (ت: 70هـ=689م) يَصِفُ مُرافَقَتَهُ نَمِرًا:

تَضَمَّنَتِ الأَرْوى لَنا بِشِوائِنا
فكُلٌّ لَهُ مِنْها نَصيبٌ مُخَرْدَلُ

الوطواط، مباهج الفكر، ص: 22.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَد (ت: 170هـ=787م):

"وخَرْدَلْتُ اللَّحْمَ: فَصَّلْتُ أَعْضاءَهُ مُوَفَّرَةً". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 334.— المصدر
1.5معنى من معاني «خَرْدَلَ»

وــــــــــ فُلانٌ الطَّعامَ: أَكَلَ خِيارَهُ وأَطايِبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَد (ت: 170هـ=786م):

"وخَرْدَلْتُ الطَّعامَ: أَكَلْتُ خِيارَهُ وأَطايِبَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 334.— المصدر
02 —

خَرادِل

اسم
2.1معنى من معاني «خَرادِل»

* الخَرادِلُ: اللَّحْمُ الـمُقَطَّعُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَبو طالِبٍ بْنُ عَبْدِ المُطَّلِبِ (3ق هـ=619م) يَفْتَخِرُ بِقُوَّةِ بَني هاشِمٍ حينَ ضُرِبَ عَلَيْهِ حِصارُ شِعْبِ بَني هاشِمٍ مَعَ المُسْلِمينَ:

دَفَعْناهُمُ حَتّى تَبَدَّدَ جَمْعُهُمْ
وحَسَّرَ عَنّا كُلُّ باغٍ وجاهِلِ

...

بِضَرْبٍ تَرى الفِتْيانَ فيهِ كَأَنَّهُمْ
ضَواري أُسودٍ فَوْقَ لَحْمٍ خَرادِلِ

ديوان أبي طالب، تح: التونجي، ص: 73.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبيدُ بْنُ أَيّوبَ (ت: 100هـ=719م) يَصِفُ حالَهُ بَعْدَما قَلى الإِنْسَ وهَجَرَ طَريقَهُمْ:

فإِنّي وبُغْضي الإِنْسَ مِنْ بَعْدِ حُبِّها
ونَأْيِيَ مِمَّنْ كُنْتُ ما إِنْ أُزايِلُهْ

لَكَالصَّقْرِ جَلّى بَعْدَما صادَ قُـــنْيَةً
قَديرًا ومَشْوِيًّا تَـــــــــــــــــرُفُّ خَرادِلُهْ

عبيد بن أيوب العنبري حياته وما تبقى من شعره، تح: القيسي، ص: 130.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نُواسٍ (ت: 198هـ=814م) يَصِفُ بازِيًّا:

وبَيْنَ مَفْرِيِّ القِرا خَرادِلُ

كَأَنَّهُ في جِلْدِهِ الرَّعابِلُ

ديوان أبي نواس، تح: الحديثي، ص: 237.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)

"كَقَوْلِهِمْ:‏ لَحْمٌ خَرادِلُ وخَراذِلُ".

خَراديل

اسم
3.1معنى من معاني «خَراديل»

* الخَراديلُ: اللُّحومُ الـمُقَطَّعَةُ. (لا واحِدَ لَها).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ساعِدَةُ بْنُ جُؤَيَّةَ الهُذَلِيّ (ت: 10هـ=631م) يَفْتَخِرُ بِقَبيلَتِهِ:

يُجَدِّلونَ مُلوكَاً في طَوائِفِهِمْ
ضَرْبًا خَراديلَ كالتَّشْقيقِ في الأُدُمِ

شعر ساعدة بن جؤية الهذلي، تح: ميساء قتلان، ص: 253.— المصدر
ع

"لَحْمٌ خَراديلُ ومُخَرْدَلٌ؛ أَيْ: مُقَطَّعٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، الألفاظ، تح: قباوة، ص: 435.— المصدر
04 —

خَرْدَل

اسم
4.1معنى من معاني «خَرْدَل»

*الخَرْدَلُ: نَباتٌ يَطولُ عَنِ الذِّراعِ قَليلًا، أَوْراقُهُ دَقيقَةٌ خَضْراءُ مُسَنَّنَةٌ، لَهُ حَبٌّ صَغيرٌ جِدًّا أَصْفَرُ، ويُتَبَّلُ بِهِ الطَّعامُ، ويُسْتَفادُ بِهِ في الطِّبِ. واحِدَتُهُ خَرْدَلَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11هـ=632م) قالَ اللهُ ﷻ:

﴿يَٰبُنَىَّ إِنَّهَآ إِنْ تَكُ مِثْقَالَ حَبَّةٍۢ مِّنْ خَرْدَلٍۢ فَتَكُن فِى صَخْرَةٍ أَوْ فِى ٱلسَّمَٰوَٰتِ أَوْ فِى ٱلْأَرْضِ يَأْتِ بِهَا ٱللَّهُ ۚ إِنَّ ٱللَّهَ لَطِيفٌ خَبِيرٌ﴾. (لقمان:16)

”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُنْبِئُ عَمّا سَيَكونُ مِنْ نُقْصانِ الدّينِ بَعْدَ تَمامِهِ:

«إِنَّهُ سَيَكونُ مِنْ ذلِكَ ما شاءَ اللهُ، ثُمَّ يَبْعَثُ اللهُ ريحًا طَيِّبَةً، فيُتَوَفّى مَنْ كانَ في قَلْبِهِ مِثْقال ُحَبَّةٍ مِنْ خَرْدَلٍ مِنْ خَيْرٍ، فيَبْقى مَنْ لا خَيْرَ فيهِ، فيَرْجِعونَ إِلى دينِ آبائِهِمْ».

مرويات أم المؤمنين عائشة في التفسير، تح: الفنيسان، ص: 205.— المصدر
”وقالَ إِياسُ بْنُ سَهْمٍ الهُذَلِيُّ (ت: 75هـ=694م) يُخاطِبُ شاعِرًا هَجاهُ:

فإِنَّ الَّذي أَسْدَيْتَ قَدْ عادَ مَغْرَمًا
عَلَيْكَ ولَمْ يُغْنِمْكَ حَبَّةَ خَرْدَلِ

السكري، شرح أشعار الهذليين، تح: فراج، ج: 1، ص: 527.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَد (ت: 170هـ=787م):

"الثَّفاءُ: الخَرْدَلُ بِلُغَةِ أَهْلِ الغَوْرِ، والواحِدَةُ بِالهاءِ. وقيلَ: بلِ الخَرْدَلُ الـمُعالَجُ بِالصِّباغِ".

الخليل، العين، تح: هنداوي، ج: 1، ص: 202.— المصدر
”وقالَ بَهاءُ الدّين زُهَيْر (ت: 656هـ=1258م):

لَكَ يا صَديقي بَغْلَةٌ
لَيْسَتْ تُساوي خَرْدَلَهْ

ديوان بهاء الدين زهير، دار صادر، ص: 294.— المصدر

الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)

خَرْدَلِيَّة

اسم
5.1معنى من معاني «خَرْدَلِيَّة»

* الخَرْدَلِيَّةُ: ثَمَرَةٌ يابِسَةٌ مُسْتَطيلَةٌ مُتَفَتِّحَةٌ تُشْبِهُ ثِمارَ النَّباتاتِ القَرَنِيَّةِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ لُويس مَعْلوف (ت: 1326هـ=1908م):

"الخَرْدَليَّةُ: ثَمَرَةٌ يابِسَةٌ مُسْتَطيلَةٌ مُتَفَتِّحَةٌ تُشْبِهُ ثِمارَ النَّباتاتِ القَرَنِيَّةِ. وهِيَ مُرَكَّبَةٌ مِنْ خِباءَيْنِ يَجْمَعُهُما شِبْهُ حاجِزٍ يُقَسِّمُ الثَّمَرَةَ قِسْمَيْنِ عَنْدَ نُضْجِها".

لويس معلوف، المنجد في اللغة والأدب، ص: 173.— المصدر
