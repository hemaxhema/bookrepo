  1. المع,السُّمُّ النّاقِعُ

01 —

ذَأَفَ

فعل
1.1معنى من معاني «ذَأَفَ»

*ذَأَفَ (بِفَتْحِ الهَمْزَةِ) فُلانٌ يَذْأَفُ (بِفَتْحِ الهَمْزَةِ) ذَأْفانًا: ماتَ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزابادي (ت:817هـ=1414م):

"وذَأَفَ، كَمَنَعَ، ذَأْفانًا: ماتَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في العَصْرِ الحَديثِ)

الفيروزابادي، القاموس المحيط. تح: العرقسوسي، ج: 1، ص: 811.— المصدر
1.2معنى من معاني «ذَأَفَ»

وــــــــــ الجَرِيحَ وَنَحْوَه، وعليه ذَأْفًا، وَذَأَفًا: أَجْهَزَ عَلَيْهِ، وأسْرَعَ قَتْلَه.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جُرْهُم بن عبد الله (ت: 65هـ=685م) يفْتَخِرُ بِفَتْكِ جَيْشِ الأزْدِ بِقاتِلي مَسْعودِ بنِ عَمْرو العَتْكي:

جاءَتْ عُمانُ دَغَرى لا صَفّا
بَكْرٌ وَجَمْعُ الأزْدِ حينَ اصْطَفّا

في حارَةِ المَوْتِ يَدِفُّ دَفّا
ضَرْبًا بِكُلِّ صارِمٍ مُصَفَّى

إنْ أخْطَأ الرَّأْسَ أَصابَ الكَفّا
وَلَّوْا خَزايا قَدْ أقِصّوا الحَتْفا

وَأُمُّ مَسْعودٍ تنادي لَهْفا
قَدْ ذَأَفَ المَوْتُ عَلَيْهِ ذَأْفا

معمر بن المثنى، شَرْح نقائض جرير والفرزدق. تح: محمد حُوَّر، ج: 1، ص: 285.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن دُرَيْد (ت: 321هـ=933م):

"والذَّأْفُ: الإجْهازُ عَلى الجَريح".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 1108.— المصدر
1.3معنى من معاني «ذَأَفَ»

وــــــــــ السُّمُ فُلانًا: قَتَلَهُ سَريعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رؤبة (ت: 145هـ=762م) يَمْدَحُ الوَليدَ بْنَ يَزيدَ بْنِ عَبْدِ المَلِكِ:

ذو مِرَّةٍ أَنْيابُهُ صَوارِفُ

بِالمُحْسِنينَ مُحْسِنٌ مُلاطِفُ

وَهْوَ لِمَنْ شاوَسَ سُمٌّ ذائِفُ

ديوان رؤبة بن العجاج. تح: البروسي، ص: 103.— المصدر
1.4معنى من معاني «ذَأَفَ»

وــــــــــ القَوْمَ: طَرَدَهُمْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن سيده (ت: 458هـ=1066م):

"ومَرَّ يَذْأَفُهُم أَي يَطْرُدُهم". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 100.— المصدر
02 —

أَذْأَفَ

فعل
2.1معنى من معاني «أَذْأَفَ»

* أَذْأَفَ فُلانٌ الأَسيرَ أو الجَريحَ، وعليه: أَجْهَزَ عَلَيْهِ وأَسْرَعَ قَتْلَه.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ خالد بن الوليد (21هـ=642م) يَأْمُرُ بِقَتْلِ الأَسْرَى في غزوة جَذيمة:

"مَنْ كانَ مَعَهُ أَسِيرٌ فَلْيُذْئِفْ عَلَيْهِ".

ابن الأثير، النهاية. تح: علي الحلبي، ج: 1، ص: 323.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الأثيرِ (ت: 606هـ=1210م):

"أَذْأَفْتُ الأَسيرَ وَذَأَفْتُهُ: إِذا أَجْهَزْتَ عَلَيْهِ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن الأثير، النهاية. تح: علي الحلبي، ج: 1، ص: 323.— المصدر
03 —

ذَأَّفَ

فعل
3.1معنى من معاني «ذَأَّفَ»

*ذَأَّفَ فُلانٌ عَلى الجَريحِ ونَحْوِه: أَجْهَزَ عَلَيْهِ وأَسْرَعَ قَتْلَه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن القُوطِيَّة (ت: 367هـ=977م):

"وَذَأَّفْتُ عَلَيْهِ مُشَدّدًا أَيْضًا: أَجْهَزْتُ عَلَيْهِ".

السرقسطي، كتاب الأفعال. تح: حسين شرف، ج: 3، ص: 603.— المصدر
04 —

انْذَأَفَ

فعل
4.1معنى من معاني «انْذَأَفَ»

* انْذَأَفَ فُلانٌ: انْقَطَعَ فُؤادُهُ. (وانظر: ذ ع ف).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عَبّاد (ت: 385هـ=995م):

"وانْذَأَفَ: انْقَطَعَ فُؤادُهُ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 106.— المصدر
05 —

ذَأْف

اسم
5.1معنى من معاني «ذَأْف»

*الذَّأْفُ: سُرْعَةُ المَوْتِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"والذَّأْفُ: سُرْعَةُ المَوْتِ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 8، ص: 201.— المصدر
06 —

ذَأْفان

اسم
6.1معنى من معاني «ذَأْفان»

*الذَّأْفانُ، والذُّؤْفانُ، والذِّئْفانُ: السُّمُّ النّاقِعُ أَوِ القاتِلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النعمان بن بشير (ت: 65هـ=685م) يَمْدَحُ قَوْمَه:

إذا المَوْتُ أَدْلَفَ ذِئْفانَهُ
وَكانتْ أَجِلَّتُهُ كالظُّلَلْ

يُبادِرُهُ كُلُّ مُسْتَبْسِلٍ
كَحَدِّ السِّنانِ شُجاعٍ بَطَلْ

ديوان النعمان بن بشير. تح: يحيى الجبوري، ص: 103.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"الذِّيفانُ والذِّئْفانُ: السُّمُّ الذي يُذْأَفُ ذَأْفًا".

الخليل، العين. تح: المخزومي، ج: 8، ص: 201.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزابادي (ت: 817هـ=1414م):

"والذَّأْفانُ والذِّئْفانُ والذُّؤْفانُ والذَّيْفانُ والذُّوفانُ والذِّيفانُ والذَّيَفانُ محرَّكةً، والذُّوافُ، كغُرابٍ: السَّمُّ الناقِعُ، أو القاتِلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في العَصْرِ الحَديثِ)

الفيروزابادي، القاموس المحيط. تح: العرقسوسي، ج: 1، ص: 811.— المصدر
6.2معنى من معاني «ذَأْفان»

وــــــــــ: المَوْتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عَبّاد (ت: 385هـ=995م):

"والذَّأْفانُ والذِّئْفانُ: المَوْتُ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 106.— المصدر
07 —

ذُؤاف

اسم
7.1معنى من معاني «ذُؤاف»

*الذُّؤافُ: السُّمُّ النّاقِعُ أو القاتِلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِم بن سلام الهروي (ت: 224هـ=838م):

"الذِّيفان، بِكَسْر الذّال وَفتحها، والذُّؤاف، كُلُّه السُّمُّ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، محمد علي النجار، ج: 15، ص: 20.— المصدر
7.2معنى من معاني «ذُؤاف»

وــــــــــ: سُرْعَةُ المَوْت.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عباد(ت: 385هـ=995م):

"والذَّأْفُ: سُرْعَةُ المَوْتِ، وكذلك الذُّؤافُ".

(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 106.— المصدر
7.3معنى من معاني «ذُؤاف»

0مَوْتٌ ذُؤافٌ: مُجْهِزٌ سَرِيعٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكسائي (ت:189هـ=805م):

"مَوْتٌ ذُؤافٌ وذُعافٌ".
(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد علي النجار، ج: 2، ص: 320.— المصدر
