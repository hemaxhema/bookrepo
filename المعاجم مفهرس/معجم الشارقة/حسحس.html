01 —

حَسْحَسَ

فعل
1.1معنى من معاني «حَسْحَسَ»

* حَسْحَسَ فُلانٌ لفُلانٍ: تَوَجَّعَ، وتَشَكّى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عبّاد (ت: 385ه=995م):

"وحَسْحَسَ الرَّجُلُ: تَوَجَّعَ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، الـمحيط في اللغة. تح: محمد حسن آل ياسين، ج: 2، ص: 301.— المصدر
1.2معنى من معاني «حَسْحَسَ»

وــــــــــاللَّحْمَ: جَعَلَهُ على الجَمْرِ، وقلّبَهُ عَلَيْهِ. (انْظُرْ: ح س س)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أوْسُ بْنُ حَجَرٍ التَّميمِيُّ (ت: 2ق. ه=620م) يَهْجو أبا المُهَوِّشِ الأسَدِيَّ:

أعَيَّرْتَنا تَمْرَ العِراقِ وبُرَّهُ
وزادُكَ أيْرُ الكَلْبِ ‌حَسْحَسَهُ ‌الجَمْرُ

أبو عمرو الشيباني، كتاب الجيم. تح: إبراهيم الإبياري، ج: 1، ص: 204.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ غَيْلانُ بْنُ سَلَمَةَ (ت: 23ه=644م): يَهْجو بادِّعاءِ الكَرَمِ:

لِتُكْذِبَ نَفْسَها نَصْرٌ وجَسْرٌ
تُحَسْحِسُ بالشَّوِيِّ عَنِ الجَميمِ

شعر ثقيف حتى نهاية العصر الأموي. جمع: إسلام ماهر عمارة، ص: 474.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأنْصارِيُّ (ت: 215ه=830م):

"إنْ جَعَلْتَ اللَّحْمَ على ‌الجَمْرِ قُلْتَ: ‌حَسْحَسْتُهُ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 2، ص: 449.— المصدر
”وقالَ أعْرابيٍّ (248ه=863م) يُجيبُ ابْنَ أبي كَريمَةَ وقَدْ سَألَهُ عِنْدَ عَمْرِو بْنِ كَرْكَرةَ: أتَعْرِفُ القَرَنْبى؟

"وما لي لا أعْرِفُ القَرَنْبى؟! فَواللهِ لَرُبَّما لَـمْ يَكُنْ غَدائي إلّا ‌القَرَنْبى ‌يُحَسْحَسُ ‌لي".

الجاحظ، الحيوان. تح: محمد باسل، ج: 3، ص: 256.— المصدر
1.3معنى من معاني «حَسْحَسَ»

وــــــــــالخَبَرَ: تَلَمَّسَهُ وتَحَرّى عَنْهُ، أو تَوَقّعَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ غَيْلانُ بْنُ سَلَمَةَ (ت: 23ه=644م) يَهْجو:

وَدِّعْ بِذَمٍّ إذا ما حانَ رِحْلَتُنا
أهْلَ الحَظائِرِ مِنْ عوفٍ ودُهْمانا

القائِلينَ وقَدْ حَلَّتْ بِساحَتِهِمْ
جَسْرٌ تُحَسْحِسُ عَنْ أوْلادِ هِصّانا

والقائِلينَ وقَدْ رابَتْ وِطابُهُمُ
أسَيْفَ عَوْفٍ تَرى أمْ سَيْفَ غَيْلانا

أبو الفرج الأصبهاني، الأغاني. تح: إحسان عباس وآخرينَ، ج: 13، ص: 142.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّغانِيُّ (ت: 650ه=1252م):

"حَسْحَسَ: إذا تَوَقَّعَ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الصّغاني، التكملة والذيل والصلة. تح: محمد أبو الفضل إبراهيم، ج: 3، ص: 338.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ بَسمة الشّواليّ (1434ه=2013م) في إحْدى قِصَصِها:

"ذُهِلَ فَرَحاتُ (...) أرْخى مِنَ النّافِذةِ رَوْمَةَ أُذُنٍ يُحَسْحِسُ في جَوْفِها الـمرْتَعِبِ الخَنّاسِ لِيَتَسَقّطَ الخَبَرَ مِنْ مَصْدَرٍ مَوْثوقٍ".

بسمة الشوالي، قناديل الـمطر. ص: 62.— المصدر
1.4معنى من معاني «حَسْحَسَ»

وــــــــــ اللَّحْمَ: قَشَرَ عَنْهُ الجَمْرَ أو الرَّمادَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيّ (ت: 216ه=831م): في مَعْنى حَسْحَسَةِ اللَّحْمِ:

"حَسْحَسَتُهُ: ‌هُوَ ‌أنْ ‌يُقْشَرَ ‌عَنْهُ ‌الرَّمادُ بَعَدما يُخْرَجُ مِنَ الجَمْرِ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 223.— المصدر
1.5معنى من معاني «حَسْحَسَ»

وــــــــــ الجِلْدَ ونَحْوَهُ: حَرَقَهُ بالنّارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بْنُ شُمَيْلٍ (ت: 203ه=819م):

"الـمُضْطَرِمُ: الـمُغْتَلِمُ مِنَ الجِمالِ، تَراهُ كَأنَّهُ قَدْ حُسْحِسَ بِالنّارِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: أحمد البردوني، ج: 12، ص: 32.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عُثمان بن عبد العزيز (ت: 1282ه=1865م) يَذُمُّ مُنْتَقِصَ أهْلِ الحَديثِ:

وأنْتَ كَما الشَّيْطانُ تَرْميكَ شُهْبُهُم
مُحَسْحَسُ ريشِ الدُّفَّتيَنِ بَعيدُ

عثمان بن عبد العزيز، فتح الحميد في شرح التوحيد. تح: سعود العريفي وآخرينَ، ص: 1567.— المصدر
02 —

تَحَسْحَسَ

فعل
2.1معنى من معاني «تَحَسْحَسَ»

* تَحَسْحَسَ الشَّيْءُ: انْقَبَضَ والْتَوى مِنْ يُبْسٍ ونَحْوِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ حُنَيْفُ الحَناتِمِ (نحو 50 ق. ه=573م) عِنْدَما سُئِلَ: ما النشر؟:

"نَدى الشَّمالِ في قَصَبِ الوَسْمِيِّ، يَسْتَأخِرُ المَطَرُ بَعْدَ الوَسْمِيِّ فَيَيْبَسُ البَقْلُ حتّى ‌تَتَحَسْحَسَ جَوانِبُهُ، وتَلْتَوي عُروقُهُ، ويَمْصَحَ ماءُ البَقْلِ وتَذْوي نَوْرَتُهُ".

أبو حيان التوحيدي، البصائر والذخائر. تح: وداد القاضي، ج: 2، ص: 135.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أحْمَدُ بْنُ طَيْفورٍ (ت: 280ه=893م):

"تَحَسْحَسَتِ اللَحْمَةُ في النّارِ: إذا انْقَبَضَتْ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أحمد بن أبي طاهر، بلاغات النساء. ص: 124.— المصدر
2.2معنى من معاني «تَحَسْحَسَ»

وــــــــــ فُلانٌ: تَحَرَّكَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ ‌‌عَبْدُ عَمْرِو بْنُ عَمّارٍ الطائِيُّ (ت: 28 ق.ه=595م) يَهْجو الأُبَيْرِدَ الغَسّانِيَّ:

كَأنَّ ثَناياهُ إذا افتَرَّ ضاحِكًا
رُؤوسُ جَرادٍ في أرونٍ تَحَسْحَسُ

[الأرونُ: جمع أرةٍ، وهي الحفرةُ وفيها النارُ]
شعر طيئ وأخبارها في الجاهلية والإسلام. تح: وفاء السنديوني، ص: 445.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ بَعْضُ الصَّحابَةِ (3ه = 625م):

"بَيْنَما النّاسُ جُلوسٌ في مَسْجِدِ رَسولِ اللهِ ﷺ إذْ خَرَجَ مُنادٍ فَنادى: أيُّها النّاسُ سُدّوا أبْوابَكُمْ! ‌فَتَحَسْحَسَ ‌النّاسُ لِذلِكَ، ولَمْ يَقُمْ أحَدٌ، ثُمَّ خَرَجَ الثّانِيَةَ فَقال: أيُّها النّاسُ سُدّوا أبْوابَكُمْ! فَلَمْ يَقُمْ أحَدٌ، فَقال النّاسُ: ما أرادَ بِهذا؟ فَخَرَجَ فَقال: أيُّها النّاسُ سُدّوا أبْوابَكُمْ قَبْلَ أنْ يَنْزِلَ العَذابُ! فَخَرَجَ النّاسُ مُبادِرينَ، وخَرَجَ حَمْزةُ بْنُ عَبْدِ المُطَّلِبِ يَجُرُّ كِساءَهُ".

علي بن عبد الله السمهودي، وفاء الوفا بأخبار دار المصطفى. ج: 2، ص: 65.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزْهَرِيُّ (ت: 370ه = 981م):

"وَيُقالُ: اقْتُصَّ مِنْ فُلانٍ فَما ‌تَحَسْحَسَ، أيْ: ما تَحَرَّكَ وما تَضَوَّرَ".

الأزهري، تهذيب اللغة. تح: محمد مرعب، ج: 3، ص: 264.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711ه=1311م) يَتَحَدَّثُ عَنْ أبي العُرْيانِ المَخْزومِيِّ:

"فَمَرَّ بِهِ زِيادٌ في الغَدِ وهُوَ يَتَكَلَّمُ، ‌وتَحْسَحَسَ لَهُ النّاسُ، فَقال: مَنْ هذا؟ قالوا: زِيادٌ".

ابن منظور، مختصر تاريخ دمشق. ج: 29، ص: 69.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عُثْمانُ بْنُ عَبْدِ العَزيزِ (ت: 1425ه=2004م) يَهْجو مَنْ يَنالُ مِنْ أهْلِ الحَديثِ:

فَلَوْ قُلِبَتْ جَنْباكَ مِنْكَ رَأيْتَ ما
تَحَسْحَسَ بِالوَحْيَيْنِ مِنْكَ بَديدُ

عثمان بن عبد العزيز، فتح الحميد في شرح التوحيد. تح: سعود العريفي وآخرين، ص: 1567.— المصدر
2.3معنى من معاني «تَحَسْحَسَ»

وــــــــــ أوْبارُ الإبِلِ: انْقَشَرَتْ وتَحاتَّتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

ابن عبّاد، الـمحيط في اللغة. تح: محمد حسن آل ياسين، ج: 2، ص: 301.— المصدر
03 —

حَسْحاس

اسم
3.1معنى من معاني «حَسْحاس»

* الحَسْحاسُ: الرَّجُلُ الجَوادُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسّانُ بْنُ ثابِتٍ (54ه=674م):

واذْكُرْ حُسَيْنًا في النَّفيرِ وقَبْلَهُ
حَسَنًا وعُتْبَةَ ذا النَّدى ‌الحَسْحاسا

ابن فارس، مجمل اللغة. تح: زهير عبد المحسن. ص: 213.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الراجِزُ (393ه=1003م، مِنْ شَواهِدِ الجَوْهَرِيِّ):

مَحَبَّةَ الأبْرامِ للحَسْحاسِ

[الأبرام: جمع بَرَمٍ، وهُوَ الذي لا يَدْخُلُ مَعَ القَومِ في المَيْسِرِ]
الجوهري، الصحاح. تح: عطار، ص: 918.— المصدر
”وقالَ الجوهريّ (ت: 393ه=1003م):

"ورُبَّما سَمّوا الرَّجُلَ الجوادَ حَسْحاسًا". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهريّ، الصحاح. تح: عطار، ص: 918.— المصدر
3.2معنى من معاني «حَسْحاس»

وــــــــــ: الخَفيفُ الحَرَكَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"رَجُلٌ حَسْحاسٌ: خَفيفُ الحَرَكَةِ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي. ج: 1، ص: 186.— المصدر
3.3معنى من معاني «حَسْحاس»

وــــــــــ: السَّيْفُ الـمُبيرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"والحَسْحاسُ: ‏السَّيْفُ الـمُبيرُ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، الـمحيط في اللغة. تح: محمد حسن آل ياسين، ج: 2، ص: 300.— المصدر
04 —

حَسْحَس

اسم
4.1معنى من معاني «حَسْحَس»

* الحَسْحَسُ: ذَهابُ الـمالِ حَتّى لا يَبْقى مِنْهُ شَيْءٌ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزابادي (ت: 817ه=1414م):

"لَأخَلِّفَنَّهُ بِحَسْحَسِهِ، أيْ: ذَهابِ مالِهِ حتّى لا يَبْقى مِنْهُ شَيءٌ". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزابادي، القاموس الـمحيط. تح: محمد نعيم العرقسوسي، ص: 538.— المصدر
05 —

حَسْحَسَة

اسم
5.1معنى من معاني «حَسْحَسَة»

* الحَسْحَسَةُ: صَوْتُ نَشيشِ اللَّحْمِ على النّارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَهْ (ت: 458ه=1066م):

"الحُساسُ: أنْ تَضَعَ اللَّحْمَ على الجَمْرِ (...) ‌وحَسْحَسَتُهُ: صَوْتُ نَشيشِهِ، وقَدْ ‌حَسْحَسَتْهُ النارُ".

ابن سيده، المحكم والمحيط الأعظم. تح: عبدالحميد هنداوي، ج: 2، ص: 498.— المصدر
