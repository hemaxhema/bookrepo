المعانى الكلّيّة

  1. المعاني الكلية,القِلَّةُ والخِفَّةُ

01 —

سِنْدَأْو

اسم
1.1معنى من معاني «سِنْدَأْو»

* السِّنْدَأْوُ: الجَريءُ المُقْدِمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بنُ بُرْدٍ (ت: 167هـ=784م) يَمْدَحُ:

بَدْرُ لَيْلٍ يَخافُ سِنْدَأْوُهُ الغِـــ
ـيلَ وعَيْنًا مِنْ صَيْدَنِيٍّ مُغيرِ

ديوان بشار. تح: إحسان عباس، ص: 334.— المصدر
1.2معنى من معاني «سِنْدَأْو»

وــــــــــ مِنَ الجِمالِ: الصُّلْبُ الشَّديدُ الخَلْقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل بنُ أحمدَ (ت: 170هـ=787م):

"القِنْدَأْوَةُ: ... النّاقَةُ الصُّلْبَةُ الشَّديدَةُ الخَلْقِ، وَجَمَلٌ قِنْدَأْوٌ وسِنْدَأْوٌ كَذَلِكَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 195.— المصدر
1.3معنى من معاني «سِنْدَأْو»

وــــــــــ مِنَ الرِّجالِ: الخَفيفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسائِيُّ (ت: 189هـ=805م):

"رَجُلٌ سِنْدَأْوٌ وقِنْدَأْوٌ: وهُوَ الخَفيفُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السّخاويّ، سفر السّعادة. تح: الدّاليّ، ج: 1، ص: 306.— المصدر
1.4معنى من معاني «سِنْدَأْو»

وــــــــــ: الجَسيمُ الحَسَنُ الخَلْقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأَنْصارِيُّ (ت: 215هـ=830م):

"رَجُلٌ سِنْدَأْوٌ: جَسيمٌ حَسَنُ الخَلْقِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 234.— المصدر
1.5معنى من معاني «سِنْدَأْو»

وــــــــــ: العَظيمُ الرَّأْسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"ورَجُلٌ قِنْدَأْوٌ، وَسِنْدَأْوٌ: عَظيمُ الرَّأْسِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: العمري، ص: 188.— المصدر
1.6معنى من معاني «سِنْدَأْو»

وــــــــــ: القَصيرُ. (كَأَنَّهُ ضِدٌّ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السِّيرافيُّ (ت: 368هـ=979م):

"رجُلٌ سِنْدَأْوٌ: ... هُوَ الْقَصِيرُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 540.— المصدر
1.7معنى من معاني «سِنْدَأْو»

وــــــــــ مِنَ الجِمالِ ونَحْوِها: المـُتَّسِعُ الخَطْوِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزهريُّ (ت: 370هـ=981م):

"جَمَلٌ سِنْدَأْوٌ: ... الفَسيحُ مِنَ الإِبِلِ في مِشْيَتِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج: 10، ص: 141.— المصدر
1.8معنى من معاني «سِنْدَأْو»

وــــــــــ: السَّيِّئُ الخُلُقِ والغِذاءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّاد (ت: 385هـ=995م):

"السِّنْدَأْوُ: ... السَّيِّئُ الخُلُقِ والغِذاءِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 8، ص: 286.— المصدر
02 —

سِنْدَأْوَة

اسم
2.1معنى من معاني «سِنْدَأْوَة»

* السِّنْدَأْوَةُ مِنَ الرِّجالِ: الدَّقيقُ الجِسْمِ، العَريضُ الرَّأْسِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جرير (ت: 110هـ=728م) يَهْجُو:

يَسُوقُ بِهِمْ سِنْدَأْوَةٌ مُتَقاعِسٌ
عَدُوُّ صَديقِ الصّالِحينَ لَعينُ

ابن بري، شرح شواهد الإيضاح. تح: عيد درويش، ص: 625.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السِّيرافيُّ (ت: 368هـ=979م):

"رجُلٌ سِنْدَأْوَةٌ: ... الدَّقيقُ الجِسْمِ مَعَ عِرَضِ الرَّأْسِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 540.— المصدر
2.2معنى من معاني «سِنْدَأْوَة»

وــــــــــ (وقَدْ تُخَفَّفُ الهَمْزَةُ فَيُقالُ: سِنْداوَة): الذِّئْبَةُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَعْفَرُ بنُ بَشّارٍ (ت: 126هـ=744م) يُخاطِبُ رَجُلًا:

أَبِنْ لِي ما الحِظاءُ وما الفَظــ
ـاءُ مَعًا، وما الغِرْيَلْ

وما العِسْبارُ والسِّنْدَأْ
وَةُ الطَّحْلاءُ والجَيْئَلْ

دراسات عربية وإسلامية مهداة إلى محمود محمد شاكر، ص: 198.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"السِّنْدَأْوَةُ: الذِّئْبَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 88.— المصدر
”وقالَ أبو عَدْنانَ السُّلَمِيُّ (ت: 250هـ=864م) يَبْكي عَلى الأَطْلالِ:

أَيا باكِيَ الأَطْلالِ في رَسْمِ دِمْنَةٍ
تَرُودُ بِها عَيْنُ المَها والجآذِرُ

وعاناتُ جَوَّالٍ، وَهَيْقٌ سَفَنَّجٌ
وَسِنْداوَةٌ فَضْفاضَةٌ وَحَضاجِرُ

الجاحظ، الحيوان. تح: عيون السود، ج: 1، ص: 121.— المصدر
2.3معنى من معاني «سِنْدَأْوَة»

وــــــــــ مِنَ النّاسِ وغَيْرِهِمْ: الجَريءُ المـُقْدِمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"السِّنْدَأْوَةُ: الجَريءُ الشَّديدُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 230.— المصدر
”وقالَ الرّاجِزُ (170هـ=787م)، من شواهد الخليل، يَصِفُ رَجُلًا:

سِنْدَأْوَةٌ مِثْلُ الفَنِيقِ الحافِرِ

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 230.— المصدر
2.4معنى من معاني «سِنْدَأْوَة»

وــــــــــ: الخَفيفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسائِيُّ (ت: 189هـ=805م):

"سِنْدَأْوَةٌ وقِنْدَأْوَةٌ: وهُوَ الخَفيفُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 246.— المصدر
2.5معنى من معاني «سِنْدَأْوَة»

وــــــــــ مِنَ النِّساءِ: الحَسَنةُ الخَلْقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأَنْصارِيُّ (ت: 215هـ=830م):

"رَجُلٌ سِنْدَأْوٌ: جَسيمٌ حَسَنُ الخَلْقِ، وامْرَأَةٌ سِنْدَأْوَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 234.— المصدر
2.6معنى من معاني «سِنْدَأْوَة»

وــــــــــ: الخِرْقَةُ تَكونُ وِقايةً تَحْتَ العِمامَةِ مِنَ الدُّهْنِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السُّكَّريُّ (ت: 275هـ=888م):

" السِّنْدَأْوَةُ: الخِرْقَةُ تَكونُ وِقايةً تَحْتَ العِمامَةِ مِنَ الدُّهْنِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 365.— المصدر
2.7معنى من معاني «سِنْدَأْوَة»

وــــــــــ: أُنْثَى النَّمِرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"وَيُقالُ لِلْأُنْثَى مِنَ النُّمُورِ: ... السِّنْدَأْوَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: العمري، ص: 126.— المصدر
2.8معنى من معاني «سِنْدَأْوَة»

وــــــــــ: القَصيرُ.
