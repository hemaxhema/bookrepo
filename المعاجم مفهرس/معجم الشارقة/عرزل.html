عرزل
عَرازيل(اسم) عِرْزال(اسم)
النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذْرُ (ع ر ز ل):

في الآرامِيّةِ: ومِنْهُ الاسْمُ: עַרְסַל/ אוּרְזִילָא (عَرسَل/ أُورزِيلَا) ʼūrzīlā /ʽarsal بمَعْنى "كوخٍ"؛ وهُوَ يَتَّفِقُ معَ الاسْمِ العَرَبيِّ (عِرْزالٍ) مَبْنًى ومَعْنًى؛ غَيْرَ أَنَّ الاسْمَ الآراميَّ خَضَعَ لإِبْدالِ الزّايِ سينًا أَوْ لإِبْدالِ العَيْنِ هَمْزَةً؛ ومن مَجيئِهِ بمَعْنى "عِرْزال" ما وَرَدَ في تُرْجومِ يوناثانَ: "וְאִשְׁתַּאְרַת כְּנִשְׁתָּא דְּצִיּוֹן כִּמְטַלַּלְתָּא בְּכַרְמָא בָּתַר דְקַטְפוֹהִי כְּעַרְסַל מִבְּתּוֹתָא בְּמִקְשַׁיָא"؛ أَيْ: "وبَقيَتْ جَمَاعَةُ صِهْيَونَ مِثْلَ المِظَلَّةِ في الكَرْمِ بَعْدَ أَنْ قَطَفوهُ، مِثْلَ عِرْزالِ مَبيتٍ في المَقْثَأَةِ"

(ת''י، יש': א, ח)

وفي العِبْريّةِ: ومِنْهُ الاسْمُ: עַרְסוֹל (عَرس۫ۈل) ʽarsōl يُطْلَقُ على "الخِرَقِ والأَحْبالِ الّتي تُمَدُّ بَيْنَ شَجَرَتَيْنِ يَمْتَهِدُها ويَضْطَجِعُ عَلَيْها المَرْءُ"، وهُوَ يَتَّفِقُ معَ الاسْمِ العَرَبيِّ (عِرزالٍ) مَبْنًى ومَعْنًى؛ غَيْرَ أَنَّ الاسْمَ العِبْريَّ خَضَعَ لإِبْدالِ الزّايِ سينًا؛ ومن مَجيئِهِ بمَعْنى "عِرْزالٍ" ما وَرَدَ في بَعْضِ كُتُبِ الأَدَبِ: "היה תלוי ערסול"؛ أَيْ: "كانَ هُناكَ عِرْزالٌ مُعَلَّقٌ".

(בוריס שץ, ירושלים הבנויה: חלום בהקיץ, יג')

وفي السُّرْيانيّةِ: ومِنْهُ الاسْمُ: ܥܰܪܙܶܠܳܐ (عَرز۪لَا) ʽərzelā بمَعْنى "كوخٍ/ سَقيفَةٍ/ مُجْتَنى الثِّمارِ"، وهُوَ يَتَّفِقُ معَ الاسْمِ العَرَبيِّ (عِرْزالٍ) مَبْنًى ومَعْنًى؛ ومن مَجيئِهِ بمَعْنى "عِرْزالٍ" ما وَرَدَ في سِفْرِ إشْعيا: "ܘܶܐܫܬܰܚܪܰܬ݂ ܒܰܪܬ݂ ܨܶܗܝܽܘܢ: ܐܰܝܟ݂ ܡܛܰܠܬ݂ܳܐ ܒܟܰܪܡܳܐ: ܘܰܐܝܟ݂ ܥܰܪܙܶܠܳܐ ܒܡܰܩܛܝܳܐ"؛ أَيْ: "فَبَقِيَتِ ابْنَةُ صِهْيَوْنَ كَمِظَلَّةٍ فِي كَرْمٍ، كعِرْزالٍ فِي مَقْثَأَةٍ"

(ܐܶܫܰܥܝܳܐ- ܐ: 8)

01 —

عَرازيل

اسم
1.1معنى من معاني «عَرازيل»

* العَرازيلُ: القَوْمُ المُجْتَمِعُونَ على لُصوصِيَّةٍ، وشَرٍّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سيدَه (ت: 458هـ=1066م):

"قَوْمٌ ‌عَرَازِيلُ: مُجْتَمِعُونَ، وأُرى أَنَّهُمُ المُجْتَمِعُونَ في لُصُوصِيَّةٍ، وخِرابَةٍ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 451.— المصدر
1.2معنى من معاني «عَرازيل»

o عَرازيلُ الثُّمامِ: عيدانُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (231هـ=846م، مِنْ شَواهِدِ ابنِ الأَعْرابِيِّ) يَصِفُ إِبِلًا:

لا تَرِدُ الماءَ بعَظْمٍ تَعْجُمُهْ

ولا ‌عَرازيلِ ثُمامٍ تَكْدُمُهْ

[تَعْجُمُهْ: تَعَضُّهُ. الثُّمامُ: نَبْتٌ. تَكْدُمُهُ: تَعَضُّهُ]
ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 451.— المصدر
02 —

عِرْزال

اسم
2.1معنى من معاني «عِرْزال»

* العِرْزالُ: مَأْوى الأَسَدِ. (ج) عَرازِلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو زُبَيْدٍ الطّائِيُّ (ت: 62هـ=682م) يَرْثي كَلْبًا لَهُ:

حَتَّى إِذا وَرَدَ العِرْزالَ وانْتَبَهَتْ
لِحِسِّهِ أُمُّ أَجْرٍ سِتَّةٍ شَزَنِ

[أجرٍ: جَمْعُ جَرْوٍ. الشَّزَنُ: الغَليظُ مِنَ الأَرْضِ]
الجاحظ، الحيوان. تح: عيون السود، ج: 2، ص: 395.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"العِرْزالُ: مَوْضِعُ الحَيَّةِ، ومَوْضِعُ ‌الأَسَدِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1150.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ الصَّيْقَلِ الجَزَرِيُّ (ت: 701هـ=1302م):

"وأَمّا بَنو سَعْدٍ فَلا نَصْبِرُ عَلى نِزالِها، ‌ومُبارَزَةِ أُسودِ ‌عِرْزالِها".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ص: 131.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ شَكيب أَرسْلان (ت: 1365هـ=1946م) يَرْثي لِحالِ فِلَسْطينَ:

مُتَبَدِّلًا عِرْزالَهُ وزَئيرَهُ
مِنْ حائِطِ المَبْكَى وَجَهْشِ المُعْوِلِ

أرسلان، اللاميات الثلاث. تح: حمادة وآخر، ص: 26.— المصدر

وــــــــــ: جُحْرُ الحَيَّةِ.

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 334.— المصدر
”وقالَ ابنُ الرّومِيِّ (ت: 283هـ=896م) يَمْدَحُ عَلِيَّ بْنَ يَحْيى النَّديم:

يَسْتَثيرُ المَكائِدَ الصُّمْعَ مِنْها
أيَّ صِلٍّ هُناكَ في العِرْزالِ

ديوان ابن الرّوميّ. تح: بسج، ج: 3، ص: 184.— المصدر
2.3معنى من معاني «عِرْزال»

وــــــــــ: ما يَجْمَعُه الأَسَدُ في مَأْواهُ، يُمَهِّدُهُ لِأَشْبالِهِ كَالعُشِّ.

العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"العِرْزالُ: ما يَجْمَعُه الأَسَدُ في مَأواهُ مِنْ شَيْءٍ، يُمَهِّدُه لِأَشْبالِهِ كَالعُشِّ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 334.— المصدر

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

2.5معنى من معاني «عِرْزال»

ع

احْتَذِرُوا لا يَلْقَكُمْ طَمَالِيلْ

قَلِيلَةٌ أَمْوَالُهُمْ عَرَازِيلْ

[هَذَاليلُ: مُسرِعونَ يَتْبَعُ بَعْضُهُمْ بَعْضًا. نَوْكى: حَمْقى. الطُّمْلُوْلُ: الخَلَقُ مِنَ الثِّيَابِ]

وــــــــــ: ما يَبْقى، ويُخَبَّأُ مِنَ اللَّحْمِ.

البكري، معجم ما استعجم. تح: السقا، ج: 2، ص: 442.— المصدر

”وقالَ الشّاعِرُ (282هـ=895م، مِنْ شَواهِدِ أبي حَنيفَةَ الدّينَوَرِيِّ):

لَقَدْ ساءَنِي والنّاسُ لا يَعْلَمونَهُ
عَرازيلُ كَمّاءٍ بِهِنَّ مُقيمِ

[الكَمّاءُ: جامِعُ الكَمْأَةِ]
ابن سيده، المخصص. تح: جفال، ج: 3، ص: 279.— المصدر
2.13معنى من معاني «عِرْزال»

وــــــــــ: كُلُّ ما جُمِعَ ووُطِّئَ لِلنَّوْمِ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"وكُلُّ شَيْءٍ جَمَعْتَهُ وَوَطَّأْتَهُ لِتَنامَ عَلَيْهِ؛ فَهو ‌عِرْزالٌ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1150.— المصدر
2.14معنى من معاني «عِرْزال»

وــــــــــ: بَيْتٌ صَغيرٌ يُتَّخَذُ لِلمَلِكِ إِذا قاتَلَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385هـ=995م):

"مَوْضِعُ النَّاظِرِ أيْضًا: ‌عِرْزَالٌ. ‌وَبَيْتٌ صَغيرٌ كانَ يُتَّخَذُ لِلمَلِكِ إِذا قاتَلَ يَكونُ فيهِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 254.— المصدر
2.15معنى من معاني «عِرْزال»

وــــــــــ: الحانُوتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ فارِسٍ (ت: 395هـ=1005م):

"العِرْزالُ: ‌حانوتُ الرَّجُلِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مجمل اللغة. تح: زهير سلطان، ص: 677.— المصدر
2.16معنى من معاني «عِرْزال»

وــــــــــ: غُصْنُ الشَّجَرَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سيدَه (ت: 458هـ=1066م):

"العِرْزالُ: غُصْنُ الشَّجَرَةِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 451.— المصدر
2.17معنى من معاني «عِرْزال»

وــــــــــ: الفِرْقَةُ مِنَ النَّاسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سيدَه (ت: 458هـ=1066م):

"العِرْزَالُ: الفِرْقَةُ مِنَ النَّاسِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 451.— المصدر
2.18معنى من معاني «عِرْزال»

o عِرْزالُ الصَّيّادِ: ما يَتَّخِذُهُ مِنْ أَهْدامٍ، وَخِرَقٍ، يَضْطَجِعُ عَلَيْها في القُتْرَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (170هـ=787م) :

"عِرْزالُ ‌الصَّيَّادِ: أَهْدامُهُ وَخِرَقُهُ الَّتي يَمْتَهِدُها وَيَضْطَجِعُ عَلَيْها في القُتْرَةِ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 334.— المصدر
