01 —

أَمَلَ

فعل
1.1معنى من معاني «أَمَلَ»

: * أَمَلَ (بِفَتْحِ الميمِ) فُلانٌ الشَّيْءَ يَأْمُلُ (بِضَمِّ الميمِ) أَمْلًا، وأَمَلًا: رَجاهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أُحَيْحَةُ بْنُ الجُلاحِ (ت: 129ق ه=496م):

فَعَمٌّ لِعَمِّكُمُ نافِعٌ
وطِفْلٌ لِطِفْلِكُمُ يُؤْمَلُ

ديوان ابن الجلاح. تح: باجودة، ص: 72.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كَعْبُ بْنُ زُهَيْرٍ (ت: 26هـ=645م):

وقالَ كُلُّ خَليلٍ كُنْتُ آمُلُهُ
لا أُلْفِيَنَّكَ إِنّي عَنْكَ مَشْغولُ

ديوان كعب بن زهير. تح: الحتي، ص: 37.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142هـ=759م):

"وَلَيْسَ غايَةُ التَّواصُلِ مِنْ كُلٍّ مِنَ الـمُتَواصِلَيْنِ إلّا طَلَبَ عاجِلِ الَّنفعِ وبُلوغَ مأْمولِهِ".

آثار ابن المقفع. ج: 1، ص: 192.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ (671ه=1273م):

"وإِذا آلَ البُخْلُ إلى هَذِهِ الأَخْلَاقِ الـمَذْمومَةِ، والشِّيَمِ اللَّئيمَةِ؛ لَمْ يَبْقَ مَعَهُ خَيْرٌ مَرْجُوٌّ، ولا صَلاحٌ مَأْمولٌ".

تفسير القرطبي. تح: التركي، ج: 7، ص: 166.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد أَمين الزَّلَلِيُّ (ت: 1241ه=1825م) مُلْتَمِسًا مِنْ وَالي جُدَّةَ جَلْبَ ابْنِهِ إِلَيْهِ:

فَحَقِّقْ رَعَاكَ اللهُ ظَنّي وَرَاعِني
بِإِنْجَاحِ مَا أَبْغي وَأَرْجو وَآمُلُ

ديوان الزللي. تح: الخطراوي، ص: 128.— المصدر
02 —

أَمَّلَ

فعل
2.1معنى من معاني «أَمَّلَ»

: * أَمَّلَ فُلانٌ الشّيْءَ: رَجَاهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتْ لَيْلى العَفيفَةُ الأَسَدِيَّةُ (ت: 143ق ه=483م) تَرْثي زَوْجَها:

بَرَّاقُ سَيِّدُنا وفارِسُ خَيْلِنا
وهُوَ الـمُطاعِنُ في مَضيقِ الجَحْفَلِ

وعِمادُ هَذا الحَيِّ في مَكْــروهِـــهِ
ومُـــؤَمَّـــلٌ يَرْجـــوهُ كُــــــلُّ مُـــؤَمِّـــــــلِ

بشير يموت، شاعرات العرب. المكتبة الأهلية، ص: 34.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَجْنونُ لَيْلى (ت: 68هـ=687م):

وقَدْ أَصْبَحَ الوُدُّ الّذي كانَ بَيْنَنَا
أَمَانِيَّ نَفْسٍ والمؤَمِّلُ حائِرُ

ديوان مجنون ليلى. تح: فراج، ص: 99.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو جَعْفَرٍ المَنْصورُ (ت: 158ه=775م):

"إِنّي قَدْ أَمَّلْتُكَ لِأَمْرٍ، فَكَيْفَ تَكونُ فيهِ؟".

الأصفهاني، الأغاني. تح: عباس وآخرين، ج: 10، ص: 71.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرَّضِيُّ الأَسْتَرَاباذِيُّ (ت: 686ه=1287م):

"واللّهُ تَعالى الـمُؤَمَّلُ لِإِرْشادِ السَّبيلِ".

شرح الرضيّ على الكافية. تح: يوسف حسن، ج: 1، ص: 18.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد البَسّامُ (ت: 1233ه=1817م) يَذْكُرُ صِفاتِ قَبائِلِ العِراقِ:

"والأَخْذُ الوَبيلُ لِمَنْ ناوَأَهُمْ، والرُّكْنُ الـمَنيعُ لِمَنْ والاهُمْ، ومآلُ الـمُؤَمِّلِ رِفْدَهُمْ".

محمد البسام، الدرر المفاخر. تح: العجمي، ج: 1، ص: 351.— المصدر
03 —

تَأَمَّلَ

فعل
3.1معنى من معاني «تَأَمَّلَ»

: * تَأَمَّلَ فُلانٌ: تَثَبَّتَ، وتَحَقَّقَ مِنْ أَمْرِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80 ق ه=543م):

عَقيلَةُ أَتْرابٍ لَها لا ذَميمَةٌ
ولا ذاتُ خَلْقٍ إِنْ تَأَمَّلْتَ جَأْنَبِ

ديوان امرئ القيس. تح: أبو الفضل، ص: 41.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو سُفْيانَ بْنُ أُمَيَّةَ (ت: 31هـ=652م):

تَأَمَّلْ فَإِنَّ القَوْمَ في سَرَوَاتِهِم
صَريحُ لُؤَيٍّ لا شَماطيطُ جُرْهُمِ

ابن إسحاق، السيرة. تح: المزيدي، ص: 321.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ إِخوانُ الصَّفا (ت: 360هـ=971م):

"وإِنَّما يَحْتاجُ الإِنْسانُ إلى التَّأَمُّلِ بِالقُوَّةِ الفِكْرِيَّةِ حَسْبُ".

رسائل إخوان الصفا. ص: 84.— المصدر
3.2معنى من معاني «تَأَمَّلَ»

وــــــــــ الشَّيْءَ وفيهِ: أَمْعَنَ نَظَرَهُ فيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ خِفافُ بْنُ نُدْبَةَ (ت: 20هـ=641م):

أَقولُ لَهُ والرُّمْحُ يَأْطُرُ مَتْنَهُ
تَأَمَّلْ خُفافًا إِنَّني أَنا ذَلِكا

جاد المولى وآخرون: أيام العرب في الجاهلية. ج: 1، ص: 286.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ إِسْحاقَ (ت: 151ه=768م):

"فَلَمّا تَأَمَّلَهُ القَوْمُ؛ قالوا: يا رَسولَ اللهِ! هُوَ واللهِ أَبو ذَرٍّ".

إسماعيل التيمي، المبعث والمغازي. ص: 684.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ (ت: 671ه=1272م):

"وَلَمّا تَأَمَّلَ يَعْقوبُ ﷺ القَميصَ، فَلَمْ يَجِدْ فيهِ خَرْقًا ولا أَثَرًا؛ اسْتَدَلَّ بِذلِكَ على كَذِبِهِمْ".

تفسير القرطبي. تح: التركي، ج: 11، ص: 684.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ المُفْتي فَتْحُ اللهِ (ت: 1260هـ=1844م) يَمْدَحُ النَّبِيَّ ﷺ:

وَفيكَ ابْنَ فَتْحِ اللهِ حُسْنُ خِتامِهِ
تَأَمَّلَهُ يا خَيْرَ مَنْ عُرْفَهُ أَسْدَى

ديوان المفتي فتح الله. تح: زهير فتح الله، ج: 2، ص: 877.— المصدر
3.3معنى من معاني «تَأَمَّلَ»

وــــــــــ الشَّيْءَ: تَدَبَّرَهُ وَأَعْمَلَ فيهِ فِكْرَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المَرّارُ الفَقْعَسِيُّ(ت: 110ه=728م):

تَأَمَّلْ ما تَقولُ وكُنْتَ قِدْمًا
قُطامِيًّا تَأَمُّلُهُ قَليلُ

شعر المرار الفقعسي. تح: نوري القيسي، مجلة المورد، ص: 172.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

"فَتَأَمَّلْ كَيْفَ تَجَرَّأَ هَذا الواضِعُ على اللهِ".

تفسير مقاتل. تح: شحاتة، ج: 5، ص: 229.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شُعْلَةُ المَوْصِلِيُّ (ت: 656ه=1258م):

"لَيْسَ ذلكَ الأَمْرُ صَعْبًا على مَنْ تَأَمَّلَه".

شعلة الموصلي، كنز المعاني. تح: المشهداني، ص: 286.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ البُجَيْرَمِيُّ (ت: 1221هـ=1807م):

"تَأَمَّلْ هَذا الإيرادَ وجَوابَهُ".

البجيرمي، تحفة الحبيب. ج: 4، ص: 322.— المصدر
3.4معنى من معاني «تَأَمَّلَ»

وــــــــــ الشَّيْءَ: حَذِرَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الشَّمّاخُ (ت: 22هـ=643م):

أَلا مَنْ مُبْلِغٌ خاقانَ عَنِّي
تَأَمَّلْ حينَ يَضْرِبُكَ الشِّتَاءُ

ديوان الشماخ. تح: الهادي، ص: 427.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بْنُ بُرْدٍ (ت: 167هـ=784م):

يا عَبْدَ حَيّي عَنْ قَريبِ
وتَأَمَّلي عَيْنَ الرَّقيبِ

الأصفهاني، الأغاني. تح: عباس وآخرين، ج: 6، ص: 175.— المصدر
04 —

آمِل

اسم
4.1معنى من معاني «آمِل»

: * الآمِلُ: عَوْنُ الرَّجُلِ. (ج) أَمَلَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"التّواثيرُ: الشُّرَطُ، وهُمُ العَتَلَةُ، والفَرَعَةُ، والأَمَلَةُ؛ واحِدُهُمْ: آمِلٌ، مِثْلُ: كافِرٌ وكَفَرَةٌ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 15، ص: 85— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الزَّبيدِيُّ (ت: 1205هـ=1790م):

"والأَمَلَةُ -مُحَرّكَةً-: أَعْوانُ الرَّجُلِ، واحِدُهُمْ: آمِلٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزَّبيدي، تاج العروس. دار الهداية، ج: 28، ص: 28.— المصدر
05 —

آمِلة

اسم
5.1معنى من معاني «آمِلة»

: * الآمِلَةُ: البُكاءُ والعَويلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ صَنَّانُ بْنُ عَبّادٍ اليَشْكُرِيُّ (نحو: 50ه=670م):

لَوْ كانَ يُشْكى إلى الأَمْواتِ ما لَقِيَ الـ
أَحْياءُ بَعْدَهُمُ مِنَ شِدَّةِ الكَمَدِ

ثُـَّم اشْـتَكَــــيـــْـتُ لَأَشْـكـانـي بِآمِلَـةٍ
قَبْرٌ بِسِنْجَارَ أَوْ قَبْــــرٌ عــلــــى قَــهَــدِ

المعرّي، شرح الحماسة. تح: نقشة، ج: 1، ص: 489.— المصدر
06 —

أَمَل

اسم
6.1معنى من معاني «أَمَل»

: * الأَمَلُ، والأَمْلُ، والإِمْلُ: الرَّجاءُ. (ج): آمالٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَبْدُ اللهِ بْنُ ثَعْلَبَةَ اليَشْكُرِيُّ (ت: 86 ق. ه=538م):

إِنَّ الَّتي عَلِقَتْ بِها آمَالُنا
دارٌ تَصَرَّفُ كالظِّلالِ الأُفَّلِ

ابن ميمون، منتهى الطلب. تح: طريفي، ج: 1، ص: 86.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11هـ = 632م) قالَ اللهُ ﷺ:

﴿ذَرْهُمْ يَأْكُلوا ويَتَمَتَّعوا وَيُلْهِهِمُ الْأَمَلُ فَسَوْفَ يَعْلَمونَ﴾ (الحِجْر: 3)

”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) وَقَدْ خَطَّ في الأَرْضِ خُطوطًا:

«هَذا الأَمَلُ، وهَذا أَجَلُهُ».

صحيح البخاري. تح: الناصر، ج: 8، ص: 89.— المصدر
”وقالَ أَبو بَكْرٍ الصِّدّيقُ (ت: 13هـ 634م):

فَأَظْفَرَ كُلَّ ذي أَمَلٍ
يُسَرُّ بِهِ بِما أَمَلا

ديوان أبي بكر الصديق. تح: الأسمر، ص: 79.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=759م):

"ومَنْ رُزِقَ العَقْلَ وأُعينَ على صِدْقِ قَريحَتِهِ بِالأَدَبِ حَرَصَ على طَلَبِ سَعْدِ جَدِّهِ وأَدْرَكَ في الدُّنْيا أَمَلَهُ".

آثار ابن المقفع. ج: 1، ص: 32.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ (671ه=1273م):

"﴿وَأَمْلَى لَهُمْ﴾؛ أَيْ: مَدَّ لَهُمُ الشَّيْطانُ في الأَمَلِ".

تفسير القرطبي. تح: التركي، ج: 19، ص: 279.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجَبَرْتِيُّ (1221ه=1806م):

"إِنِ امْتَدَّ الأَجَلُ وأَسْعَفَ الأَمَلُ".

الجبرتي: عجائب الآثار. تح: عبد الرحيم عبد الرحمن، ج: 3، ص: 110.— المصدر

08 —

أُمُلَّة

اسم
8.1معنى من معاني «أُمُلَّة»

: * الأُمُلَّةُمِنَ النُّوقِ: الـمُسِنَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الطّائي (206هـ=821م، مِنْ رُواةِ الشَّيْبانِيِّ):

"ناقَةٌ أُمُلَّةٌ وإِبِلٌ أُمُلّاتٌ؛ أَيْ: الجِلَّةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: الأبياري، ج: 3، ص: 251.— المصدر
09 —

أَميل

اسم
9.1معنى من معاني «أَميل»

: * الأَميلُ: قِطْعَةٌ مِنَ الرَّمْلِ تَسْتَطيلُ وتَسْتَعْرِضُ فَلا يَقْطَعُهَا الـمَاشي إِلَّا في أيَّامٍ. (ج) أُمُلٌ، وأَمَايِل.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طَرَفَةُ بْنُ العَبدِ (ت: 60ق ه=563م):

سَما لَكَ مِن سَلْمَى خَيالٌ ودونَها
سَوادُ كَثيبٍ عَرْضُهُ فَأَمَايِلُهْ

ديوان طرفة. تح: درية الخطيب وآخر، ص: 63.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=708م):

وأُمُلٍ مَوْصولَةٍ بِأُمُلِ

تَنْحى بِطولِ أَحْبُلٍ وأَحْبُلِ

ديوان العجاج. تح: عزة حسن، ص: 213.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سِيْبَوَيْهِ (ت: 188ه=803م):

"وقالوا: ‏السُّبُلُ، وأَمْيُلٌ وَأُمُلٌ".

سيبويه، الكتاب. تح: هارون، ج: 3، ص: 403.— المصدر
10 —

التَّأمُّل

اسم
10.1معنى من معاني «التَّأمُّل»

: * التَّأَمُّلُ(في الفَلْسَفَةِ): إِنْعامُ النَّظَرِ والتَّفْكيرِ في رَوِيَّةٍ، وهُوَ ضَرْبٌ مِنَ التَّفْكيرِ الَّذي يَنْصَبُّ على الـمُجَرَّداتِ ويُجاوِزُ الـمَحْسوساتِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الدَّبوسِيُّ (ت: 430هـ=1038م):

"وذَلِكَ بِالتَّأَمُّلِ في باطِنِ أَمْرِهِ".

الدبوسي، تقويم الأدلة. تح: الميس، ص: 186.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ القَيِّمِ (ت: 751هـ=1351م):

"وأَمْثالُ ذلِكَ مِنَ الإِشاراتِ الصَّحيحَةِ لا تُنالُ إِلّا بِصَفاءِ الباطِنِ وصِحَّةِ البَصيرَةِ وحُسْنِ التَّأَمُّلِ".

ابن القيم، مدارج السالكين. ج: 2، ص: 435.— المصدر
11 —

مُؤَمَّل

اسم
11.1معنى من معاني «مُؤَمَّل»

: * الـمُؤَمَّلُ: الثّامِنُ مِنْ خَيْلِ الحَلَبَةِ العَشَرَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ قُتَيْبَةَ (ت: 276ه=889م): في أَسْماءِ الخَيْلِ في السِّباقِ:

"ثُمَّ الحَظِيُّ، ثُمَّ الـمُؤَمَّلُ، ثُمَّ اللَّطيمُ".

ابن قتيبة: الجراثيم. تح: الحميدي، ج: 2، ص: 138.— المصدر
11.2معنى من معاني «مُؤَمَّل»

: وــــــــــ: عَلَمٌ على غَيرِ واحِدٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَحْمَدُ بْنُ حَنْبَلٍ (ت: 241هـ=855م):

"قالَ مُؤَمَّلٌ: قُلْتُ لِسُفْيانَ: هَذِهِ القِرَاءَةُ في الحَديثِ؟ قالَ: نَعَم".

مسند أحمد. تح: شعيب الأرناؤوط وآخرين، ج: 35، ص: 74.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَديمِ (ت: 660هـ=1261م):

"وأَبو الـمَعالي أَحْمَدُ بْنُ مَنصورِ بْنِ مُؤَمَّلٍ الغَزّالُ".

ابن العديم، بغية الطلب. تح: زكار، ج: 1، ص: 374.— المصدر
