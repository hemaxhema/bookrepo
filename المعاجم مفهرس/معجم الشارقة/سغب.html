  1. الم,شِدَّةُ الجُوعِ
01 —

سَغَبَ

فعل
1.1معنى من معاني «سَغَبَ»

* سَغَبَ (بِفَتْحِ الغَيْنِ) فُلانٌ، يَسْغُبُ (بِضَمِّ الغَيْنِ) سَغْبًا، وسُغوبًا، ومَسْغَبَةً: جاعَ مَعَ تَعَبٍ. فهو ساغِبٌ. وهيَ بِتاءٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امرؤُ القيس (ت: 80ق هـ=544م) يَصِفُ فَرَسَهُ ويُشَبِّهُها بِلَقْوَةٍ تُطْعِمُ فَرْخَها:

تُطْعِمُ فَرْخًا لَها ساغِبًا
أَزْرى بِهِ الجوعُ والإِحْثالُ

ابن منظور، لسان العرب. تح: أمين عبد الوهاب وآخر، ج: 3، ص: 49.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (ت: 11هـ=632م) يُخاطِبُ رَجُلًا ضَرَبَ آخَرَ أَكَلَ مِنْ زَرْعٍ لَهُ:

«ما أَطْعَمْتَهُ إِذْ كانَ جائِعًا أَوْ ساغِبًا، وَلا عَلَّمْتَهُ إِذْ كانَ جاهِلًا».

ابن ماجه، السنن. تح: عبد الباقي، ج: 2، ص: 770.— المصدر
”وقالَ القَتّال الكِلابيُّ (ت: 70هـ=690م) يَمْدَحُ:

جَليدٌ كَريمٌ خيمُهُ وطِباعُهُ
عَلى خَيْرِ ما تُبْنى عَلَيْهِ الضَّرائِبُ

إِذا جاعَ لَمْ يَفْرَحْ بِأَكْلَةِ ساعَةٍ
ولَمْ يَبْتَئِسْ مِنْ فَقْدِها وهْوَ ساغِبُ

ديوان القتال الكلابي. تح: إحسان عباس، ص: 29.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

" السَّاغِبُ: الجائِعُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 380.— المصدر
”وقالَ الرَّسِّي (ت: 246هـ=860م) يَتَمَنَّى ظُهورَ المَهْدِيِّ:

عَسى فَرَجٌ يَأْتي بِهِ اللهُ عاجِلًا
بِدَوْلَةِ مَهْدِيٍّ يَقومُ فيَظْهَرُ

يُقَسَّمُ أَعْشارٌ ويُشْبَعُ ساغِبٌ
ويُنْصَفُ مَظْلومٌ ويوسَعُ مُقْتِرُ

مجموع كتب ورسائل الرّسي. تح: عبد الكريم جدبان، ص: 26.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أحمد عبدالعزيز العجميُّ (664هـ=1265م) فيما كَتَبَ، مُبَشِّرًا ابْنَ خَلِّكان بِفَتْحِ المَلِكِ الظّاهِرِ لِمَدينَةِ "صَفَدَ":

"وهِيَ وإِنْ كانَتْ غارَةً عَظيمَةً شُنَّتْ في يَوْمٍ واحِدٍ عَلى جَميعِ سَواحِلِهِ... إِلّا أَنَّها كانَتْ بَيْنَ يَدَيْ عَزائِمِنا المَنْصورَةِ نَشيطَةً، نَشَّطْنا بِها الغازينَ، واسْتَرْهَفْنا بِها هِمَمَ المُجاهِدينَ، وقَدَّمْناها لَهُمْ كاللُّهْنَةِ قَبْلَ الطَّعامِ لِلسّاغِبينَ".

اليونيني، ذيل مرآة الزّمان. ج: 2، ص: 339.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ محمد عبده (ت: 1323هـ=1905م) يُمَثِّلُ لِمَنْ يَأْمُرُ النّاسَ بِالبِرِّ ويَنْسى نَفْسَهُ:

"مَثَلُ مَنْ كانَتْ هَذِهِ حالَتُهُ كمَثَلِ ... أوْ مَثَلِ ساغِبٍ يَدْعو النّاسَ إِلى المائِدَةِ الشَّهِيَّةِ، ويَبيتُ عَلى الجوعِ والطَّوى".

محمد عبده، الأعمال الكاملة، ج: 4، ص: 157.— المصدر
02 —

سَغِبَ

فعل
2.1معنى من معاني «سَغِبَ»

* سَغِبَ(بِكَسْرِ الغَيْنِ) فُلانٌ يَسْغَبُ (بِفَتْحِ الغَيْنِ) سَغَبًا، وسَغابَةً: جاعَ مَعَ تَعَبٍ. فهُوَ سَغِبٌ. وهيَ بِتاءٍ. (ج) سِغابٌ. ولغَيْرِ العاقِلِ: سَواغِبُ، وهُوَ سَغْبانُ، وهيَ سَغْبَى. (ج) سِغابٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ شاعِرٌ مِنْ بَني قُريم (80ق هـ=544م) يُجيبُ تَأَبَّطَ شَرًّا:

تَأَبَّطُ سَوْأَةً وحَمَلْتَ شَرًّا
لَعَلَّكَ أَنْ تَكونَ مِنَ الـمُصابِ

لَعَلَّكَ أَنْ تَجيءَ بِكَ المَنايا
تُساقُ لِفِتْيَةٍ مِنّا غِضابِ

فتُصِبِحَ في مَكَرِّهِمُ صَريعًا
وتُصْبِحَ طُرْفَةَ الضُّبُعِ السِّغابِ

أبو سعيد السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 848.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُجَيَّة الكِنْدي (ت: 1هـ=623م) يُخاطِبُ زَوْجَتَهُ:

فقُلْتُ لِعَبْدَيْنا أَريحا عَلَيْهِمُ
سَأَجْعَلُ بَيْتي مِثْلَ آخَرَ مُعْزِبِ

بَنِيَّ أَحَقُّ أَنْ يَنالوا سَغابَةً
وأَنْ يَشْرَبوا رَنْقًا لَدى كُلِّ مَشْرَبِ

التّبريزيّ، شرح ديوان الحماسة لأبي تمّام. تح: غريد الشّيخ وآخرين، ص: 721.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بنُ بُرْدٍ (ت: 167هـ=784م) يُخاطِبُ مَحْبوبَتَهُ:

عامَتْ سُلَيْمى ومَسَّها سَغَبُ
بَلْ مالَها لا تَزالُ تَكْتَئِبُ

تَذَكَّرَتْ عيشَةً بِذي سَلَمٍ
عِشْنا بِها نَجْتَني ونَحْتَلِبُ

ديوان بشار. تح: إحسان عباس، ص: 76.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القرطبيُّ (ت: 671هـ=1273م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ﴿ أَوۡ إِطۡعَٰمٞ فِي يَوۡمٖ ذِي مَسۡغَبَةٖ ﴾] البَلَد: [:

"وإِطْعامُ الطَّعامِ فَضيلَةٌ، وهُوَ مَعَ السَّغَبِ الذي هُوَ الجوعُ أَفْضَلُ".

القرطبي، الجامع لأحكام القرآن. تح: التركي، ج: 22، ص: 303.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ناصيف اليازجي (ت: 1287هـ=1870م) يَرْوي في المَقامَةِ الصورِيَّةِ، عَلى لِسانِ لَيْلى تُخاطِبُ آخَرَ:

"يا فُلُ؛ قَدْ أَنْهَكَني اللَّغَبُ، وأَهْلَكَني السَّغَبُ".

ناصيف اليازجي، مجمع البحرين. ط 4، ص: 131.— المصدر
03 —

أَسْغَبَ

فعل
3.1معنى من معاني «أَسْغَبَ»

* أَسْغَبَ فُلانٌ: دَخَلَ في المَجاعَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَبّاسُ بنُ مِرْداسٍ (3هـ=625م) يَهْجو خَوّاتَ بْنَ جُبَيْرٍ؛ لِرِثائِهِ يَهودَ بَني النَّضيرِ:

فصِرْتَ كَمَنْ أَمْسى يُقَطِّعُ رَأْسَهُ
لِيَبْلُغَ عِزًّا كانَ فيهِ مُرَكَّبا

فبَكِّ بَني هارونَ واذْكُرْ فَعالَهُمْ
وقَتْلَهُمُ لِلجوعِ إِذْ كُنْتَ مُسْغِبا

أبو الفرج الأصفهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 14، ص: 202.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الحائِكِ الهَمْدانِيُّ (ت: 334هـ=946م) يَصِفُ دِيارَ أَعْدائِهِمْ بَعْدَ اجْتِياحِهِمْ لَها:

فنَتْرُكُ دارَ مَنْ سِرْنا إِلَيْهِ
تَئِنُّ لَدى القُفولِ بِنا أَنينا

وقَدْ عُرِكَتْ فساوى الحُزْنُ مِنْها
سَباسِبَها بِكَلْكَلِ فاحِسينا

كَما دارَتْ رَحًى مِنْ فَوْقِ حَبٍّ
تَداوَلُـهَا أَكُفُّ الـمُسْغِبينا

مقبل التّام عامر الأحمديّ، مجلّة التّراث العربيّ - الدّامغة - اتّحاد الكتّاب العرب - العدد: 95. ص: 213.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عبد الحميد يونس (1408هـ=1988م) تاريخُ الطَّبْعِ، يُتَرْجِمُ عَنْ ويليام ديورانت ما كَتَبَهُ عَنْ ثَوْرَةِ "سيكينْجِنْ" وحِصارِهِ مَدينَةَ "تيريزَ":

"لَقَدْ كانَ الفُرْسانُ المُسْغِبونَ يَنْتَظِرونَ في صَبْرٍ نافِذٍ فُرْصَةً مُواتِيَةً لِلثَّوْرَةِ عَلى الأُمَراءِ".

وِل ديورَانت، قصة الحضارة، ج: 24، ص: 72.— المصدر
3.2معنى من معاني «أَسْغَبَ»

وــــــــــــــ فُلانًا: جَوَّعَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رَجُلٌ مِنْ بَني نَهْشَلٍ (35هـ=656م) يَرْثي عَبْدَ اللهِ بْنَ أَبي رَبيعَةَ:

نُعيتُ ابْنَ أَسْماءَ الذي هَدَّ يَوْمُهُ
بُيوتَ بَني كَعْبٍ وأَسْغَبَ دارِما

فلَوْ كُنْتَ يابْنَ النَّهْشَلِيَّةِ شاهِدًا
لَأَبْرَمْتَ مَيْمونًا مِنَ الأَمْرِ حازِما

البلاذري، أنساب الأشراف. تح: زكّار وآخر، ج: 10، ص: 186.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشريف المرتضى (ت: 436هـ=1044م) يمْدَحُ فَخْرَ المُلْكِ:

ولَوْلا الهَوى لَمْ تَلْقَني بِمَنازِلٍ
نَواحِلَ يَسْتَدْعي نُحولي نُحولُها

أُغَنّي بِها مَجْهولَةً لَمْ تَبِنْ لَنا
ويُذْكِرُني غَضَّ الوِصالِ مَحيلُها

مِنَ اللّاتي يُسْغِبْنَ النِّطاقَ هَضامَةً
ويَمْشينَ بِالبَطْحاءِ خِرْشًا حُجولُها

ديوان الشريف المرتضى، ج: 3، ص: 101.— المصدر
04 —

سَغَّبَ

فعل
4.1معنى من معاني «سَغَّبَ»

* سَغَّبَ فُلانٌ الشَّيْءَ لِفُلانٍ: سَوَّغَهُ لَهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزاباديُّ (ت: 817هـ=1414م):

"وهُوَ مُسَغَّبٌ لَهُ كَذا، ومُسَغَّبٌ: مُسَوَّغٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزاباديّ، القاموس المحيط. تح: العرقسوسي، ص: 97.— المصدر
05 —

اسْتَغَبَ

فعل
5.1معنى من معاني «اسْتَغَبَ»

* اسْتَغَبَ القَوْمُ: أُصيبوا بمَجاعَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شاعِرٌ إياديٌّ (400هـ=1010م) مِنْ شَواهِدِ أبي حَيّانَ التَّوحيدِيِّ، يَمْدَحُ:

فإِنَّكَ ضَحّاكٌ إِلى كُلِّ صاحِبٍ
وأَنْطَقُ مِنْ قُسٍّ غَداةَ عُكاظِها

إِذا اسْتَغَبَ المَوْلى مَساغِبَ مَعْشَرٍ
فعُذْرَةُ فيها آخِذٌ بكِظاظِها

أبو حيان التوحيدي، البصائر والذخائر. تح: وداد القاضي، ج: 4، ص: 120.— المصدر
06 —

تَسَغَّبَ

فعل
6.1معنى من معاني «تَسَغَّبَ»

* تَسَغَّبَ فُلانٌ: تَلَوّى مِنْ شِدَّةِ الجوعِ أَوِ العَطَشِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مهيار الدَّيْلَمي (ت: 428هـ=1037م) يُواسي نَفْسَهُ، ويَحُضُّها عَلى الصَّبْرِ:

ولا تَتَسَغَّبي أَسَفًا ويَأْسًا
فعِنْدَ مَغالِقِ الأَمْرِ انْفِتاحُ

ديوان مهيار الدّيلميّ. تح: أحمد نسيم، ج: 1، ص: 208.— المصدر
07 —

ساغِب

اسم
7.1معنى من معاني «ساغِب»

o ساغِبٌ لاغِبٌ: مُصابٌ بالإعْياءِ مِنْ شِدَّةِ الجوعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207ه=822م):

"‌وأَسْغَبَ ‌الرَّجُلُ فهُوَ مُسْغِبٌ، إذا دَخَلَ في المَجاعَةِ، ورَجُلٌ سَغْبانُ لغْبانُ، وساغِبٌ لاغِبٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 71.— المصدر
”وقالَ ابْنُ الرُّومِيِّ (ت: 283هـ=896م) في أبي شَيْبَةَ بْنِ الحاجِبِ وكانَ قَدْ دَعاهُ واسْتَتَرَ عَنْهُ:

لَهْفي وقَدْ جاءَتْكَ جَفّالَةٌ
كُلٌّ مُغِذٌّ ‌ساغِبٌ ‌لاغِبُ

ألّا يُلاقوكَ فتَلْقَى بِهِمْ
أكْلَ يَتامى ما لَهُمْ كاسبُ

ديوان ابن الرومي. تح: الطباع، ج: 1، ص: 229.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَلِيٌّ الطَّنْطاوِيُّ (1367ه=1947م، تاريخُ النَّشْرِ):
