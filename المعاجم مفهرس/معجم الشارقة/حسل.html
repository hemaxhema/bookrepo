المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السَّاميَّاتُ بِهذا الجَذْرِ مِنَ العَرَبِيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (ح س ل):

في الآراميّة: وَرَدَ الاسْمُ: חֲשִׁילְתָּא (حَشِيلتَا) ḥašīltā بِمَعْنى (حَشَفُ النَّخْلِ/ تَمْرٌ مَنْزوعُ النَّوَى) ويُوافِقُ الاسْمَ العَرَبِيَّ (حَسيلةٌ) مَبْنًى ومَعْنًى غَيرَ أنَّ الاسْمَ الآراميَّ خَضَعَ لِإبْدالِ السِّينِ شينًا، ومِنْ مَجيئِهِ بِمَعْنى (حَسيلةٌ) ما وَرَدَ في التَّلْمودِ البابِليِّ: "אמר ליה חשילתא ודאי קא אמרת"، أيْ: "قالَ لَهُ مِنَ المُؤَكَّدِ أنَّكَ تَتَحَدَّثُ عَنِ الحَسيلةِ".

(ברכ': לח, א)

في العِبريّة: وَرَدَ الاسْمُ: נֶחְשָׁל (نِحشَال) neḥšāl بِمَعْنى (مَجْرُورٌ، مَسْحوبٌ، مَسُوقٌ) ويُوافِقُ الاسْمَ العَرَبيَّ (حَسيلٌ / مَحْسُولٌ) مَعْنًى ومَبْنًى غَيْرَ أنَّ الاسْمَ العِبْرِيَّ خَضَعَ لِإبْدالِ السِّينِ شينًا، ومِنْ مَجيئِهِ بِمَعْنى (مَحْسولٌ) ما وَرَدَ في سِفْرِ التَّثْنيةِ: "וַיְזַנֵּב בְּךָ כָּל-הַנֶּחֱשָׁלִים אַחֲרֶיךָ"، أيْ: "وقَطَعَ عَنْكَ كُلَّ المَحْسولينَ خَلْفَكَ".

(דב' 25: 18)

في السُّرْيانيّة: وَرَدَ الفِعْلُ: ܚܣܰܠ (حْسَل) ḥəsal بِمَعْنى (صارَ حِصْرِمًا) ويُوافِقُ في أصولِهِ أصولَ الاسْمِ العَرَبِيِّ (حسيلةٌ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (صارَ حِصْرِمًا) ما وَرَدَ في سِفْرِ إشعيا: "ܘܗܰܒܳܒܐ ܚܣܰܠ. ܘܢܶܗܘܶܐ ܥܽܘܦܝܳܐ. ܘܢܶܦܣܽܘܩ ܙܰܠܺܝܠܳܐ ܒܡܰܓܠܳܐ"، أيْ: "عِنْدَمَا يَصِيرُ الزَّهْرُ حِصْرِمًا نَضِيجًا، يَقْطَعُ القُضْبَانَ بِالمَنَاجِلِ".

(ܐܶܫܰܥܝܳܐ، ܝܚ: 5)

في الجِعزيّة: وَرَدَ الفِعْلُ: ሐሰለ (حَسَلَ) ḥasala بِمَعْنى (ساقَ الدَّابّةَ بِالرَّسَنِ- لَجَمَ- حَمَّلَ) ويُوافِقُ الفِعْلَ العَرَبِيَّ (حَسَلَ) مَبْنًى وفي بَعْضِ مَعانيهِ، ومِنْ مَجيئِهِ بِمَعْنى (قَيَّدَ) ما وَرَدَ في سِفْرِ الأمْثالِ: ኃጢአት ያሤግር ብእሴ ወበኅብለ ኃጢአት ኵሉ ይትሐሰል። أيْ: "آثامُ المُنافِقِ تَتَصَيَّدُهُ، ويُقَيَّدُ بِحبالِ خَطيئَتِهِ".

(ምሳሌ: 5:22)

01 —

حَسَلَ

فعل
1.1معنى من معاني «حَسَلَ»

* حَسَلَ (بِفَتْحِ السّينِ) فُلانٌ بَقِيّةً مِنَ الشَّيءِ يَحْسُلُ (بَضَمِّ السّينِ) حَسْلًا، وحُسولًا: أبْقَى مِنْهُ بَقِيّةً رُذالًا.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ شدّادُ بْنُ مُعاويَةَ العَبْسيُّ (ت: 50 ق. ه=573م) يفْتخِرُ بِفَرَسِهِ جَرْوَةَ:

ألا أبْلِغْ بَني العُشَراءِ عَنّي
عَلانِيَةً وما يُغني السِّرارُ

قَتَلْتُ سَراتَكُمْ وحَسَلْتُ مِنْكُمْ
حَسيلًا مِثْلَ ما حُسِلَ الوِبارُ

المفضل الضبي، أمثال العرب. تح: إحسان عباس، ج: 1، ص: 97.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"والمَحْسولُ: المَرْذولُ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخَليلُ، العَيْن. تح: المخزومي وآخر، ج: 43، ص: 197.— المصدر
1.2معنى من معاني «حَسَلَ»

وــــــــــ الرّاعي الإبِلَ: ساقَها سَوْقًا شديدًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانيّ (ت: 206 هـ=821م):

"والحَسْلُ: سَوقٌ شَديدٌ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبْو عَمْرو الشَّيْبانيّ، الجيم. تح: الإبياري، ج: 1، ص: 190.— المصدر
”وقالَ الراجز (370ه=981م، مِنْ شَواهِدِ الأزْهَرِيِّ):

كَيْفَ رَأيْتَ نَجْعتي وحَسْلي

الأزهري، تهذيب اللغ. تح: محمد مرعب، ج: 4، ص: 177.— المصدر
1.3معنى من معاني «حَسَلَ»

وــــــــــ فُلانٌ الشَّيْءَ: رَذَلَهُ واسْتَخَسَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367 هـ=977م):

"وحَسَلَ الشَّيْءَ حَسْلًا: رَذَلَهُ" (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْن القوطِيَّة، كِتاب الأفْعال. تح: فودة، ج: 1، ص: 209.— المصدر
1.4معنى من معاني «حَسَلَ»

وــــــــــ فُلانٌ الجِلْدَ: قَشَرَهُ وكَشَطَهُ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَد رِضا (ت: 1372 هـ=1953م):

"حَسَلَ الجِلْدَ حَسْلًا: قَشَرَهُ".

أحْمَد رِضا العامِليّ، مُعْجَم مَتْنِ اللُّغَةِ. ج: 2، ص: 89.— المصدر
02 —

حُسِلَ

فعل
2.1معنى من معاني «حُسِلَ»

*حُسِلَ بهِ: أُخِسَّ حَظُّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ محمد بن عمران القاضي (ت: 154ه=771م):

"شَبابُ المَرْأةِ مِنْ خَمْسَ عَشْرةَ سَنةً إلى ثَلاثينَ سَنةً، وفيها مِنَ الثَّلاثينَ إلى الأرْبعيَن مُسْتَمْتَعٌ، وإذا اقْتَحَمَتِ العَقَبةَ الأُخْرى حُسِلَتْ".

ابن قتيبة، عيون الأخبار. تح: منذر أبو شعر،ج:3، ص:236.— المصدر
”قالَ الجَوْهَريُّ (ت: 393 هـ=1003م):

"وحُسِلَ بِهِ، أيْ: أُخِسَّ حَظُّهُ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ص: 1669.— المصدر

04 —

احْتَسَلَ

فعل
4.1معنى من معاني «احْتَسَلَ»

* احْتَسَلَ فُلانٌ: اصْطادَ الحِسْلَ، وهو ولَدُ الضَّبِّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانيُّ (ت: 650 هـ=1252م):

"واحْتَسَلَ: إذا اصْطادَ الحُسولَ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصَّغانيّ، التَّكْمِلَة والذَّيل والصِّلَة. تح: الإبياري، ج: 5، ص: 316.— المصدر
05 —

حُسالَة

اسم
5.1معنى من معاني «حُسالَة»

* الحُسالَةُ: ما تَساقَطَ مِنْ مُعالَجَةٍ لشَيْءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سيبَوَيْهِ (ت: 180 هـ= 797م) يُمَثِّلُ لِما جاءَ على فُعالَةٍ مِمّا يَسْقُطُ مِنْ مُعالجَةٍ لِشَيءٍ:

"وذلِكُ نحوُ: القُلامةِ والقُوارةِ والقُراضَةِ والنُّفايةِ والحُسالَةِ".

سيبويه، الكتاب. تح: هارون، ج: 4، ص: 13.— المصدر
5.2معنى من معاني «حُسالَة»

وــــــــــ: الرَّذْلُ الرَّديءُ مِنْ كُلِّ شَيْءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207 هـ=822م):

"الحُسالَةُ: الرَّذْلُ مِنْ كُلِّ شَيْءٍ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزْهَريّ، تَهْذيب اللُّغَة. تح: درويش، ج: 4، ص: 303.— المصدر
5.3معنى من معاني «حُسالَة»

وــــــــــ مِنَ الفِضَّةِ: ما سَقَطَ مِنْها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللِّحْيانيُّ (ت: 220 هـ=835م):

"الحُسالَةُ مِنَ الفِضَّةِ كالسُّحالَةِ، وهُوَ ما سَقَطَ مِنْها". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْن سِيدَه، المُحْكَم. تح: هنداوي، ج: 3، ص: 189.— المصدر
5.4معنى من معاني «حُسالَة»

وــــــــــ: ما تَكَسَّرَ مِنْ قِشْرِ الشَّعيرِ وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنِيفَةَ الدّينَوَريّ (ت: 282 هـ=895م):

"الحُسالَةُ: ما تَكَسَّرَ مِنْ قِشْرِ الشَّعيرِ وغَيْرِهِ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْنُ سِيدَه، المُحْكَم. تح: هنداوي، ج: 3، ص: 189.— المصدر
5.5معنى من معاني «حُسالَة»

وــــــــــ مِنَ النّاسِ: أسافِلُهُمْ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيديُّ (ت: 1205 هـ=1790م):

"وحُسالَةُ النّاسِ: خُشارَتُهُمْ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزَّبيديّ، تاجُ العَرْوس. تح: الطناحي، ج: 28، ص: 300.— المصدر
06 —

حَسْل/ حَسَل/ حِسْل

اسم
6.1معنى من معاني «حَسْل/ حَسَل/ حِسْل»

* الحَسْلُ، والحَسَلُ، والحِسْلُ: النَّبِقُ الأخْضَرُ. واحدِتُها حَسْلَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عبّاد (ت: 385 هـ=995م):

"والحَسْلُ: السَّوْقُ الشَّديدُ. والنَّبِقُ الأخْضَرُ، الواحِدَةُ: حَسْلَةٌ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، المحيط. تح: آل ياسين، ج: 2، ص: 483.— المصدر
6.2معنى من معاني «حَسْل/ حَسَل/ حِسْل»

وــــــــــ: عُشْبٌ شُجَيْريٌّ دائمُ الخُضرَةِ، تُسْتَعْمَلُ أطرافُهُ الغَضَّةُ كتابِلٍ، ويُستخرَجُ منه زيتٌ طيّارٌ يُسْتعمَلُ في تَقْطيرِ بعضِ المشروباتِ الرُّوحيَّةِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1421ه=2000م، تاريخُ الطَّبْعِ):

"الحَسْلُ والحَسَلُ والحِسْلُ: عُشْبٌ شُجَيْريٌّ دائمُ الخُضرَةِ، تُسْتَعْمَلُ أطرافُهُ الغَضَّةُ كتابِلٍ، ويُستخرَجُ منه زيتٌ طيّارٌ يُسْتعمَلُ في تَقْطيرِ بعضِ المشروباتِ الرُّوحيَّةِ".

مجمع القاهرة، المعجم الكبير. ج: 5، ص: 340.— المصدر
07 —

حِسْل

اسم
7.1معنى من معاني «حِسْل»

* الحِسْلُ: ولَدُ الضَّبِّ حينَ يخرجُ مِنْ بَيْضَتِهِ. (ج) أحْسالٌ وحِسَلَةٌ وحِسْلانٌ وحُسولٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَمْرُو بْنُ كُلْثومٍ (ت: 39 ق. ه=584م) يتَوَعَّدُ سُلَيْمًا وسَيِّدَهُ النُّعمانَ بْنَ المُنْذِرِ:

فَإنْ كانَ جِدٌّ فاسْعَيا ما وسِعْتُما
وإنْ كانَ لَعْبٌ آخِرَ الدَّهْرِ فالعَبا

ومِنْ يَعْدِلُ اللَّيْثَ المُجَرَّبَ وقْعُهُ
بِحِسْلَيْنِ لَمّا يَعْدُوا أنْ تَضَبَّبا

شعراء تغلب في الجاهليَّة. تح: علي أبو زيد، ج: 2، ص: 11.— المصدر
