01 —

قَمَأَ

فعل
1.1معنى من معاني «قَمَأَ»

* قَمَأَ (بِفَتْحِ الميمِ) فُلانٌ وغَيْرُهُ يَقْمَأُ (بِفَتْحِ الميمِ) قَمْأةً وقُمْأةً وقِمْأةً وقَماءةً وقُماءً وقِماءً: ذَلَّ وصَغُرَ. فهُوَ قَميءٌ، وقَمِيٌّ. (ج) قِماءٌ وقُماءٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ قُرادُ بن حَنَش (ت: 13ق.ه=609م) يَهْجو:

وأَلْقَوْا مَقاليدَ الأُمورِ إِلَيْهِمُ
جَميعًا قِماءً كارِهينَ وطُوَّعا

ابن السكيت، إصلاح المنطق. تح: شاكر وهارون، ص: 400.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"رَجُلٌ قَميءٌ، وامْرَأةٌ بِالهاءِ؛ أَيْ: قَصيرٌ ذَليلٌ".

الخليل، العين. تح: السامرائي وآخر، ج: 5، ص: 235.— المصدر
”وقالَ ابْنُ السِّكّيت (ت: 244ه=858م):

"رَجُلٌ جُعْشوشٌ وجُعْسوسٌ. وكُلُّ ذَلِكَ إِلى قَماءةٍ وصِغَرٍ".

[الجُعْسوسُ: اللَّئيمُ الخِلْقةِ والخُلُقِ، ويُقالُ: اللَّئيمُ القَبيحُ]
ابن السكيت، الألفاظ. تح: قباوة، ص: 163.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ لسانُ الدّين بن الخَطيب (ت: 776ه=1374م) يَصِفُ مَثالِبَ الوَزيرِ إِبْراهيمَ بْنِ أَبي الفَتْحِ الأَصْلَعِ وابْنِ عَمِّهِ مُحَمّد بن إِبْراهِيمَ العَقْرَبِ:

"وفِي العَشْرِ الأُوَلِ مِنْ رَمَضانَ عامَ واحِدٍ وسِتّينَ وسَبْعِمائةٍ تَقَبَّضَ عَلى الوَزيرِ الـمَشْؤومِ، وابْنِ عَمِّهِ الغَويِّ الغَشومِ (...) يُعاقِرونَ النَّبيذَ في السِّكَكِ الغاصّةِ، ووَلَدُ العَقْرَبِ الرَّدِيِّ بِضِدِّهِ قَماءةً وتَقَطُّبًا، تَنْبُو عَنْهُما العُيونُ، ويَبْكي مِنْهُما الخَزُّ".

المقري، نفح الطيب. تح: إحسان عباس، ج: 5، ص: 139.— المصدر
1.2معنى من معاني «قَمَأَ»

وــــــــــ فُلانٌ وغَيْرُهُ قَماءةً: قَصُرَتْ قامَتُهُ وصَغُرَ جِسْمُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَوّاسُ بن ثابِت (ت: 70ه=690م) مُتَحَدِّثًا عَنْ أَصْلِ خِلْقَتِهِ:

وأَوْرَثَها شَرَّ التُّراثِ أَبوهُمُ
قَماءَةَ جِسْمٍ والرُّواءُ دَميمُ

المرزوقي، شرح ديوان الحماسة. تح: أحمد أمين وآخر، ص: 1016.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"دِقالُ الغَنَمِ: قِماؤُها وصِغارُها أَجْسامًا".

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 164.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عَبْدِ الـمَلِكِ الـمَراكُشيّ (ت: 703ه=1303م) مُتَرْجِمًا للشَّيْخِ أَحْمَدَ بْنِ خَلَصَةَ الحِمْيَريِّ الكُتاميِّ:

"وكانَ -مَعَ قَماءةِ خَلْقِهِ- جَهيرَ الصَّوتِ فَصيحًا يُسْمِعُ عَلى شاخَتِهِ مَنْ في أُخْرَياتِ الجَامِعِ الأَعْظَمِ عَلى بُعْدِ مَسافةِ ما بَيْنَهُما".

ابن عبد الملك، الذيل والتكملة. تح: إحسان عباس وآخرين، ج: 1، ص: 570.— المصدر
1.3معنى من معاني «قَمَأَ»

وــــــــــ الـماشِيةُ وغَيْرُها قُموءًا وقُموءَةً وقَماءً وقَمْئًا وقَمَأً: كَثُرَتْ وامْتَلَأَتْ سِمَنًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَمْرو بن أحْمَر (ت: 75ه=695م) يَصِفُ صوفَ شَعَرِ جُرْدٍ:

وجُرْدٍ طارَ باطِلُها نَسيلًا
وأَحْدَثَ قَمْؤُها شَعَرًا قِصارا

[باطِلُها: الشَّعَرُ أَوِ الوَبَرُ أَوِ الصّوفُ عَلى الدّابّة عِنْدَ وِلادَتِهَا أَوْ هُزالِها. نَسيلًا: ساقطًا مُتَقَطِّعًا]
الشريف المرتضى، الأمالي. تح: محمد أبو الفضل إبراهيم، ج: 1، ص: 467.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وقَمَأَتِ الـمَاشِيةُ تَقْمَأُ قُموءًا، فَهِيَ قامِئةٌ؛ أَيِ: امْتَلَأَتْ سِمَنًا".

الخليل، العين. تح: السامرائي وآخر، ج: 5، ص: 235.— المصدر
”وقالَ أبو زَيْد الأنْصاريّ (ت: 215ه=830م):

"وهَذا زَمانٌ يَقْمَأُ فيهِ الـمالُ؛ أَيْ: يَسْمُنُ ويَحْسُنُ".

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 363.— المصدر
1.4معنى من معاني «قَمَأَ»

وــــــــــ الماشيةُ بالمكانِ: أَقامَتْ بِهِ لِخِصْبِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مَهْدي الباهليّ (ت: 160ه=777م):

"ويُقالُ: قَمَأتِ الـماشِيةُ بمَكانِ كَذا وكَذا حَتّى سَمِنَتْ".

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 363.— المصدر
1.5معنى من معاني «قَمَأَ»

وــــــــــ فُلانٌ بِالـمَكانِ قَمَأً: دَخَلَهُ وأَقامَ بِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو موسى المَدينيُّ (ت: 581ه=1185م):

"في الحَديثِ:كانَ يَقْمَأُ إِلى مَنْزِل عائِشَةَ ’ كَثيرًا".

أبو موسى المديني، المجموع المغيث. تح: العزباوي، ج: 2، ص: 750.— المصدر
1.6معنى من معاني «قَمَأَ»

وــــــــــ فُلانًا: عابَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ الرَّحْمن بن عَوْف (ت: 32ه=653م) يَرُدُّ عَلى عُمَرَ بْنِ الخَطّابِ بَعْدَ أَنْ خَطَبَ فِي النّاسِ:

"ما زِدْتَ عَلى أَنْ قَمَأْتَ نَفْسَكَ".

[في المطبوع: (قَمَّأتَ) بتشديد الميم، ولعلَّهُ تَصْحيفٌ، فالأَصْلُ التَّخْفِيفُ، لا سِيَّما والمَعاجِمُ لم تَذْكُرْ غَيْرَهُ]
المتقي الهندي، كنز العمال. تح: بكري حياني وآخر، ج: 12، ص: 655.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الـمَرْزُبان (ت: 330ه=942م):

"بابُ قَمَأْتُ كِبْرَهُ: ووَقَمْتُ تِيهَهُ، وخَسَأْتُ عُنْجُهِيَّتَهُ وأَذْلَلْتُ عِزَّهُ، وقَذَعْتُ أُبَّهَتَهُ وأَصْدَأْتُ صَوْتَهُ وقَمَعْتُ صَلَفَهُ، وخَفَضْتُ رِفْعَتَهُ وكَبَحْتُ تَطاوُلَهُ".

ابن المَرْزُبان، الألفاظ. تح: حامد صادق قنيبي، ص: 186.— المصدر
02 —

قَمُؤَ

فعل
2.1معنى من معاني «قَمُؤَ»

* قَمُؤَ (بِضَمِّ المِيمِ) فُلانٌ وغَيْرُهُ يَقْمُؤُ (بِضَمِّ الميمِ) قَماءةً وقُماءً وقِماءً: ذَلَّ وصَغُرَ. فهُوَ قَميءٌ، وقَمِيٌّ. (ج) قِماءٌ وقُماءٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”وبه فُسّر ما قالَ قُرادُ بن حَنَش (ت: 13ق.ه=609م) يَهْجو:

وأَلْقَوْا مَقاليدَ الأُمورِ إِلَيْهِمُ
جَميعًا قِماءً كارِهينَ وطُوَّعا

ابن السكيت، إصلاح المنطق. تح: شاكر وهارون، ص: 400.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ وَهْبُ بن مُنَبِّه (ت: 114ه=732م) ناصِحًا:

"احْتِمالُ بَعْضِ الذُّلِّ خَيْرٌ مِنِ انْتِصارٍ يَزيدُ صاحِبَهُ قَماءةً".

ابن أبي الدنيا، الإشراف في منازل الأشراف. تح: نجم عبد الرحمن خلف، ص: 151.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"﻿ قَمُؤَ الرّجُلُ قَماءةً: الصّاغِرُ القَميءُ، يُصَغَّرُ بِذَلِكَ وإِنْ لَمْ يَكُنْ قَصيرًا".

الخليل، العين. تح: السامرائي وآخر، ج: 5، ص: 235.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ السِّغْناقي (ت: 711ه=1311م) فِي بابِ الإِحْرامِ:

"أَوْ يُشيرُ إِلى أَنَّهُ ائْتَمَرَ بِأَوامِرِ الشَّرْعِ، وقَمُؤَ فِي سَبيلِهِ قَماءةَ النَّزْعِ، والنَّقْعِ غَيْرَ راجِعٍ إِلى عَقْلِهِ، ومَسْكَنَتِهِ، وحَوْلِهِ وقُوَّتِهِ".

السغناقي، النهاية في شرح الهداية. جامعة أم القرى، ج: 6، ص: 41.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ العقاد (ت: 1384ه=1964م):

وإِذا التَّنافُسُ لَمْ يَكُنْ لِغَنيمةٍ
رَجَحَ القَميءُ بِهِ عَلى الجَبّارِ

العقاد، وهج الظهيرة. ج: 1، ص: 71.— المصدر
2.2معنى من معاني «قَمُؤَ»

وــــــــــ فُلانٌ قَماءةً وقِماءً: صَغُرَ جِسْمُهُ وهُزِلَ. (ضِدُّ).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أُنَيْفُ بن حَكيم (ت: 40ه=660م) مُتَحَسِّرًا عَلَى قِصَرِ قامَتِهِ:

تَبيَّنَ لي أَنَّ القَماءةَ ذِلَّةٌ
وأنَّ أَعِزّاءَ الرِّجالِ طِوالُهَا

أبو الحسن البصري، الحماسة البصرية. تح: مختار الدين أحمد، ج: 1، ص: 35.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وغُلامٌ قَصْعٌ وقَصيعٌ: إِذا كانَ قَميئًا لا يَشِبُّ".

الخليل، العين. تح: السامرائي وآخر، ج: 1، ص: 128.— المصدر
2.3معنى من معاني «قَمُؤَ»

وــــــــــ القَوْمُ قَمْأةً: أَخْصَبوا واتَّسَعَ حالُهُمْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”روى الـحَسَنُ البَصْريّ (ت: 110ه=728م):

"أَنَّ ثَقِيفًا قالَتْ: سَنُعْطيكَهَا عَلى قَمْأةٍ فِيها".

أبو داود الطياليسي، مسنده. تح: محمد التركي، ج: 2، ص: 249.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَر بن المُثَنّى (ت: 209ه=825م):

"إِنَّهُمْ لَفي قَمْأةٍ؛ أَيْ: فِي خِصْبٍ وسَعَةٍ مِنَ العَيْشِ ودَعةٍ".

[ابن القطاع: وقَمُؤَ القَوْمُ قَمْأَةً: اتَّسَعَ حالُهُمْ]
ابن السكيت، الألفاظ. تح: قباوة، ص: 13.— المصدر
2.4معنى من معاني «قَمُؤَ»

وــــــــــ الإِبِلُ وغَيْرُها: سَمِنَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ الدّينَوَريّ (ت: 282ه=895م):

"قَمَأَتِ المـاشِيةُ تَقْمأُ قُموءًا وَقُموءةً وقَمُؤَتْ قَمْأً: سَمِنَتْ".

ابن سيده، المخصص. تح: خليل جفال، ج: 2، ص: 164.— المصدر
03 —

أَقْمَأَ

فعل
3.1معنى من معاني «أَقْمَأَ»

* أَقْمَأَ القَوْمُ: كَثُرَتْ إِبِلُهُمْ وماشِيَتُهُمْ وسَمِنَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ قُتَيْبَة الدِّينَوَرِيّ (ت: 276ه=889م):

"فإِنْ سَمِنَتِ الإِبِلُ وكَثُرَتْ مَعَ سِمَنِها؛ قيلَ: قَمَأَتْ، وأَقْمَأَ القَوْمُ إِذا كانَ ذَلِكَ في إِبِلِهِمْ".

ابن قتيبة، كتاب الجراثيم. تح: محمد الحميدي، ج: 2، ص: 193.— المصدر
3.2معنى من معاني «أَقْمَأَ»

وــــــــــ فُلانٌ فُلانًا: صَغَّرَهُ وأَذَلَّهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (3ه=623م) داعِيًا عَلى ابْنِ قَمِئةَ يَوْمَ أُحُدٍ:

«ما لَكَ؟ أَقْمَأَكَ اللهُ».

الطبراني، المعجم الكبير، ج: 8، ص: 154.— المصدر
”وقالَتْ نُسَيْبةُ بنتُ كَعْب (ت: 13ه=634م) عِنْدَما سُئِلَتْ عَمَّنْ جَرَحَها يَوْمَ أُحُدٍ:

"ابْنُ قَمِئةَ. أَقْمَأَهُ اللهُ".

ابن كثير، البداية والنهاية. تح: عبد الله التركي، ج: 5، ص: 409.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيت (ت: 244ه=858م):
