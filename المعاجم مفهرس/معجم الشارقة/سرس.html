سرس
سَرِسَ(فعل) سَرَّسَ(فعل) سَريس(اسم)
المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (س ر س):

في الآراميّة: وَرَدَ الاسْمُ: סָרִיסָא (ساريسا) sārīsā بِمَعْنى (الذي لا يأتي النِّساءَ، العِنّينُ مِنَ الرِّجالِ، مَخْصيٌّ) يُوافِقُ الاسمَ العَرَبيَّ (سَرِسٌ، سريسٌ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (سَريسٌ) ما وَرَدَ في تَرجومِ يوناثانَ: "וְלָא יֵימַר סָרִיסָא הָא אֲנָא חֲשִׁיב כַּאֲע יָבִישׁ"، أي: "ولا يقلْ السَّريسُ: ها أنا مِثلُ خَشبٍ يابسٍ"

(ת''י، יש': נו, ג)

في العِبْريّة: وَرَدَ الاسْمُ: סָרִיס (ساريس) sārīs بِمَعْنى (مَنْ أُزيلَتْ خِصيتاهُ فلا يَأتي النِّساءَ، خَصيّ، طَواشي) يُوافِقُ الاسمَ العَرَبيَّ (سَريسٌ) معنًى ومبنًى، ومِنْ مَجيئِهِ بِمَعْنى (سَريسٌ) ما وَرَدَ في سِفرِ التَّكوينِ: "וְהַמְּדָנִים מָכְרוּ אֹתוֹ אֶל מִצְרָיִם לְפוֹטִיפַר סְרִיס פַּרְעֹה"، أي: "وباعَ المديانيّونَ يوسفَ في مِصرَ لفوطيفارَ سَريسِ الفِرْعَوْنِ".

(בר' 37: 36)

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܣܪܺܝܼܣ (سْريسَ) sərīs بِمَعْنى (العِنّينُ، الخَصيُّ) يُوافِقُ الاسمَ العَرَبيَّ (سَريسٌ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (سريس): "ܐܝܬ ܓܪ ܣܪ̈ܝܣܝܢ ܗܠܝܢ ܕܡܢ ܟܪܣܗܿ ܕܐܝܡ̈ܗܘܢ ܐܬܿܝܠܕܘ ܟܕܢ ܣܪܝܣܝܢ ܘܐܝܬܿ ܣܪ̈ܝܣܝܢ ܕܐܣܬܪ̈ܣܘ ܡܢ ܒܢ̈ܝܢܫ"، أيْ: "هُناكَ سِراسٌ وُلِدُوا مِنْ بُطُونِ أُمَّهاتِهِمْ سِراسًا، وهناك سِراسٌ سَرِسُوا بِفِعلِ النّاسِ".

(Lag(295) Matt19:12)

المعانى الكلّيّة

  1. الضَّعْفُ والعَجْزُ
  2. التَّفَطُّنُ والحَزْمُ
  3. نَباتٌ يُتَداوَى به
01 —

سَرِسَ

فعل
1.1معنى من معاني «سَرِسَ»

* سَرِسَ (بِكَسْرِ الرّاءِ) فُلانٌ يَسْرَسُ (بِفَتْحِ الرّاءِ) سَرَسًا: ضَعُفَ، فهُوَ سَرِسٌ، وسَريسٌ. (ج) سُرَساءُ وسِراسٌ. (لُغَةُ طَيِّئٍ). (ولَمْ يُسْتَعْمَل هذا الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ الـمُشْتَقُّ مِنْهُ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو زُبَيْدٍ الطَائِيُّ (ت: 62هـ=682م) يُعاتِبُ قَوْمًا:

أَفي حَقٍّ مُواساتي أَخاكُمْ
بِمالي ثُمَّ يَظْلِمُني السَّريسُ

شعر أبي زبيد الطائي. تح: القيسي، ص: 101.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والسَّريسُ: العِنِّينُ، والجَميعُ: السُّرَساءُ والسِّراسُ... وقيلَ: الضَّعيفُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 8، ص: 241.— المصدر
1.2معنى من معاني «سَرِسَ»

وــــــــــ: عَقَلَ وحَزُمَ بَعْدَ جَهْلٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرو الشَّيْبانيُّ (ت: 206هـ=821م):

"السَّرِيسُ: الكَيِّسُ الحافِظُ [ما] في يَدَيْهِ".

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 289.— المصدر
”وقالَ الرّاجِزُ (231هـ=846م) ، من شواهِدِ ابْنِ الأَعْرابيِّ:

يا رُبَّ قَرْمٍ ‌سَرِسٍ عَنَطْنَطِ

لَيْسَ بِجُعْسوسٍ ولا بِأَذْوَطِ

الأزهري، تهذيب اللغة. تح: درويش، ج: 1، ص: 334.— المصدر
1.3معنى من معاني «سَرِسَ»

وــــــــــ: عَجَزَ عَنْ مُعاشَرَةِ النِّساءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرو الشَّيْبانيُّ (ت: 206هـ=821م):

"السَّريسُ: الكَيِّسُ الحافِظُ [ما] فِي يَدَيْهِ... وهُوَ العِنِّينُ أَيْضًا".

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 289.— المصدر
1.4معنى من معاني «سَرِسَ»

وــــــــــ: ساءَ خُلُقُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"سَرِسَ الرَّجُلُ -بِالكَسْرِ-: إِذا ساءَ خُلُقُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: أبي الفضل، ج: 3، ص: 365.— المصدر
1.5معنى من معاني «سَرِسَ»

وــــــــــالجَمَلُ ونَحْوُهُ: كانَ لا يُلْقِحُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَوْهَرِيُّ (ت: 393هـ=1003م):

"وفَحْلٌ سَريسٌ: بَيِّنُ السَّرَسِ، إِذا كانَ لا يُلْقِحُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ج: 3، ص: 938.— المصدر
02 —

سَرَّسَ

فعل
2.1معنى من معاني «سَرَّسَ»

* سَرَّسَ فُلانٌ المُصْحَفَ ونَحْوَهُ: حَبَكَهُ، أَوْ شَدَّ بَعْضَهُ إلى بَعْضٍ ولَمْ يُضَمَّ طَرَفاهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَثيرِ (ت: 630هـ=1233م) يَذْكُرُ أَمْوالَ حبشيِّ بْنِ مُعِزِّ الدَّوْلَةِ الَّتي وَقَعَتْ في يَدِ وَزيرِ أَخيهِ بَخْتِيارَ بِالْبَصْرَةِ لَمَّا أَظْهَرَ عِصْيانَهُ لَهُ:

"ومِن جُمْلَةِ ما أَخَذَ لَهُ: خَمْسَةَ عَشَرَ مُجَلَّدًا، سِوى الأجْزاءِ والمُسَرَّسِ وما لَيْسَ لَهُ جِلْدٌ".

ابن الأثير الجزري، الكامل في التاريخ. تح: تدمري، ص: 357.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَدُ رضا العاملي (ت 1372هـ=1953م):

"يُقالُ: مُصْحَفٌ مُسَرَّسٌ؛ أَيْ: مَشْدودٌ بَعْضُهُ إِلى بَعْضٍ ولم يُضَمَّ طَرَفاهُ فإنْ ضُمّا فَهُوَ مُشَرَّزٌ".

أحمد رضا، متن اللغة، ج: 3، ص: 140.— المصدر
03 —

سَريس

اسم
3.1معنى من معاني «سَريس»

* السَّريسُ: ضَرْبٌ مِنَ الهِنْدِبا، نَباتٌ يُتَداوَى بِه، وهُوَ نَوْعانِ؛ بَرِّيٌّ وبُسْتانِيٌّ. (مع)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سُحْنونُ (191هـ= 807م) يَسْأَلُ عَبْدَ الرَّحْمنِ بْنَ القاسِمِ عَنْ بَعْضِ أَحْكامِ البَيْعِ:

"هَلْ يَجوزُ بَيْعُ فَدّانِ كُرَّاثٍ بِفَدّانَيْ كُرّاثٍ أَوْ سَريسٍ أَوْ خَسٍّ أَوْ سِلْقٍ؟".

مالك بن أنس، المدونة، ج: 4، ص: 269.— المصدر
