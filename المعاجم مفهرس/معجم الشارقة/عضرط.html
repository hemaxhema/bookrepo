01 —

عَضْرَط

اسم
1.1معنى من معاني «عَضْرَط»

*العَضْرَطُ: الاسْتُ أَوِ الدُّبُرُ. (ج) عَضاريطُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بْنُ لَجَأٍ (ت: 105هـ=724م) يَهْجو:

ما بالُ أُمِّكَ بِالمَنْحاةِ إِذْ كَشَفَتْ
عَنْ عَضْرَطٍ وارِمٍ قدْ غَمَّهُ الشَّعَرُ

ديوان عمر بن لجأ التيمي. تح: يحيى الجبوري، ص: 95.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"العَضْرَطُ: الدُّبُرُ".

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 1153.— المصدر
1.2معنى من معاني «عَضْرَط»

وــــــــــ: الفَرْجُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110هـ=728م) يَهْجُو أَحَدَ قَوْمِ الفَرَزْدَقِ:

وحُجْزَةُ لَوْ تَبَيَّنَ ما رَأَيْتُمْ
بِعَضْرَطِها لَماتَ مِنَ الفُحامِ

[حُجْزَةُ: هو حُجْزَةُ بْنُ جَعْثَنٍ. فَحِمَ يَفْحَمُ فُحومًا: بَكى حَتّى يَنْقَطِعَ صَوْتُهُ، ويُدَلَّهَ عَقْلُهُ]
ديوان جرير بشرح محمد بن حبيب. تح: نعمان محمد أمين طه، ج: 1، ص: 198.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ حَبيبٍ الهاشِمِيُّ (ت: 245هـ=860م) يَشْرَحُ قَوْلَ جَريرٍ السّابِقَ:

"يَقولُ: لَوْ رَأى ما رَأَيْتُمْ لَبَكى حَتّى يَفْحَمَ. يُقالُ: فَحِمَ يَفْحَمُ فُحومًا إذا بَكى حَتّى يَنْقَطِعَ صَوْتُهُ ويُدَلَّهَ عَقْلُهُ. وعَضْرَطُها: فَرْجُها".

ديوان جرير بشرح محمد بن حبيب. تح: نعمان محمد أمين طه، ج: 1، ص: 198.— المصدر
1.3معنى من معاني «عَضْرَط»

وــــــــــ: الرَّجُلُ الكَثيرُ شَعْرِ الجَسَدِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"ورَجُلٌ أَهْلَبُ عَضْرَطٌ: كَثيرُ شَعَرِ الجَسَدِ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

ابن عباد، المحيط في اللغة. تح: محمد حسن آل ياسين، ج: 2، ص: 241.— المصدر
1.4معنى من معاني «عَضْرَط»

وــــــــــ: عَظْمُ الدُّبُرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المَعَرّيُّ (ت: 449هـ=1057م):

"وإِنَّما أُخِذَ العُضْروطُ مِنَ العَضْرَطِ وهُوَ عَظمُ الدُّبُرِ".

أبو العلاء المعرّي، اللامع العزيزي شرح ديوان المتنبي. تح: محمد سعيد المولوي، ج: 1، ص: 424.— المصدر
1.5معنى من معاني «عَضْرَط»

وــــــــــ: العَفْلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَه (ت: 458هـ=1066م):

"العَضْرَطُ والعِضْرِطُ وهُوَ العَفْلُ

1-6: o إِيَّاكَ ‌وأهْلَبَ ‌الْعَضْرَطِ: مَثَلٌ يُضْرَبُ في التَّحذيرِ مِنَ المُعْجَبِ بِنَفْسِهِ. (انْظُرْ: ه ل ب)".

ابن سيده، المخصص. تح: خليل جفال، ج: 1، ص: 170.— المصدر
1.7معنى من معاني «عَضْرَط»

o تَرَكْتُهُ على مِثْلِ عَضْرَطِ العَيْرِ: مَثَلٌ يُضْرَبُ لِعَدَمِ تَرْكِ شَيءٍ للآخَرِ. (انْظُرْ: ت ر ك)

02 —

عُضْرُط

اسم
2.1معنى من معاني «عُضْرُط»

*العُضْرُطُ: هُوَ الخادِمُ الذي يَخْدُمُ بِطَعامِ بَطْنِهِ. (ج) عَضاريطُ وعَضارِطَةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الأَفْوَه الأَوْديُّ (ت: 54ق.هـ=569م) يَصِفُ نِساءَ قَوْمٍ دَعَوْهُمْ إلى المَعْرَكَةِ:

تُناغي العَضاريطَ المُشاةَ خَرائِدٌ
تُمَسِّحُ أَطْرافَ القِلاصِ ذُيولُها

[تُناغي: تَناجي. الخَرائِدُ: جَمْعُ الخَريدَةِ، وهِيَ البِكْرُ التي لَمْ تُمَسَّ قَطُّ. القَلائِصُ: جَمْعُ القَلوصُ، وهِيَ الأَنْثى الشّابَّةُ مِنَ الإِبِلِ]
الأفوه الأودي، ديوان الأفوه الأوديّ. تح: محمّد ألتونجي، ج: 1، ص: 102.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسَّانُ بْنُ ثابِتٍ (ت: 54هـ=674م) يَهْجو:

مَهاجِنَةٌ إذا نُسِبوا عَبيدٌ
عَضاريطٌ مَغالِثَةُ الزِّنادِ

حسان بن ثابت، ديوان حسّان بن ثابت. تح: عبد أ. مهنّا، ج: 1، ص: 90.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللَّيْثُ بْنُ سَعْدٍ (ت: 180هـ=796م):

"والعُضْرُطُ، كقُنْفُذٍ، وعُلابِط، وعُصْفورٍ: الخادِمُ على طَعامِ بَطْنِه ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

مرتضى الزبيدي، تاج العروس. تح: عبد العليم الطحاوي، ج: 19، ص: 475.— المصدر
2.2معنى من معاني «عُضْرُط»

وــــــــــ: التّابِعُ والصُّعْلوكُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ النَّابِغَةُ الذُّبْيانيُّ (ت: 18ق.هـ=605م) يَصِفُ:

تَشْكو ‏العَضاريطُ من عَوْذى ومِنْ عَمَمٍ
أجْنَ المِياهِ وَقَدْ جاوَزْنَ أَوْرالا

[عَوْذى وعَمَم: مِنْ لَخْمٍ، وأَوْرَل: جَبَلٌ]
النّابغة الذّبيانيّ، ديوان النّابغة الذبيانيّ. تح: محمد أبو الفضل إبراهيم،ص: 180.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الطِّرِمّاحُ (ت: 125هـ=743م) يَهْجو:

هُمُ الفاتِقونَ الرّاتِقونَ وأَنْتُمُ
عَضاريطُ للسَّوْءاتِ حَيْثُ اسْتُحِلَّتِ

مجموعة شعراء، ديوان حروب الردة. تح: محمود أبو الخير، ج: 1، ص: 78.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بِلالُ بْنُ جَريرِ بْنِ عَطِيَّةَ (ت: 140=757م) يُجيبُ الفَرَزْدَقَ:

دَعاكمْ حَوارِيُّ الرَّسولِ فَكُنْتُمُ
عَضاريطَ يا خُشْبَ الخِلافِ المُصَرَّعا

محمّد بن المبارك، منتهى الطّلب من أشعار العرب. تح: محمد نبيل طريفي، ج: 1، ص: 68.— المصدر
2.3معنى من معاني «عُضْرُط»

وــــــــــ: الخَطُّ الذي مِنَ الذَّكَرِ إلى الدُّبُرِ، ويُسمَّى: العِجانَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَه (ت: 458هـ=1066م):

"العُضْرُطُ: العِجانُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم والمحيط الأعظم. تح: عبد الحميد هنداوي، ج: 2، ص: 441.— المصدر
03 —

عِضْرِط

اسم
3.1معنى من معاني «عِضْرِط»

*العِضْرِطُ: الاسْتُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110هـ=728م) يَهْجو:

إذا سَعَلَتْ فَتاةُ بَني تَميمٍ
تَلَقَّمَ بابُ عِضْرِطِها التُّرابا

معمر بن المثنى، كتاب النقائض. تح: محمد إبراهيم حور، وآخر، ج: 1، ص: 607.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصاريّ (ت: 215هـ=830م):

"يُقال إذا أَلْزَقَ اسْتَهُ بالأَرْضِ: قَدْ أَلْزَقَ بُعْثُطَهُ بالأَرْضِ، وقَدْ أَلْزَقَ عِضْرِطَهُ بالصَّلَّةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

قاسم بن ثابت السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 1، ص: 1083.— المصدر
3.2معنى من معاني «عِضْرِط»

وــــــــــ: اللَّئيمُ مِنَ الرِّجالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"العِضْرِطُ: اللَّئيمُ مِنَ الرِّجالِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد، العين. تح: المخزومي وآخر، ج: 2، ص: 326.— المصدر
3.3معنى من معاني «عِضْرِط»

وــــــــــ: الخَطُّ الذي مِنَ الذَّكَرِ إلى الدُّبُرِ، ويُسمى: العِجانَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بْنُ شُمَيْلٍ (ت: 203هـ=819م):

"العِضْرِطُ: العِجانُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب. تح: أمين محمد عبد الوهاب وآخر، ج: 9، ص: 254.— المصدر
3.4معنى من معاني «عِضْرِط»

وــــــــــ: الخُصْيَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بْنُ شُمَيْلٍ (ت: 203هـ=819م):

"العِضْرِطُ العِجانُ والخُصْيَةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب. تح: أمين محمد عبد الوهاب وآخر، ج: 9، ص: 256.— المصدر
3.5معنى من معاني «عِضْرِط»

وــــــــــ: العَفْلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَه (ت: 458هـ=1066م):

"والعِضْرِطُ: وهُوَ العَفْلُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[العَفْلُ: شَيْءٌ مُدَوَّرٌ يَخْرُجُ مِنْ رَحِمِ المَرْأَةِ وحَياءِ النّاقَةِ شِبْهُ الأُدْرَةِ في الرَّجُلِ]
ابن سيده، المخصص. تح: خليل جفال، ج: 1، ص: 170.— المصدر
3.6معنى من معاني «عِضْرِط»

وـــــــــ: عَجْبُ الذَّنَبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ بَرِّي (اللُّغَويُّ) (ت: 582هـ=1187م):

"ويُقالُ: العِضْرِطُ: عَجْبُ الذَّنَبِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن بري، التنبيه والإيضاح عما وقع في الصحاح. تح: رجب عبد الجواد، ج: 3، ص: 129.— المصدر
3.7معنى من معاني «عِضْرِط»

o إِيَّاكَ وكُلَّ قِرْنٍ أَهْلَبَ العِضْرِطِ: مَثَلٌ يُضْرَبُ لعَدَمِ تَرْكِ شَيءٍ للآخَرِ. (انْظُرْ: ق ر ن)

04 —

عَضارِط/عُضارِط

اسم
4.1معنى من معاني «عَضارِط/عُضارِط»

*العَضارِطُ والعُضارِطُ: الأَجيرُ. (ج) عَضارِطُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ القُطَاميُّ التَّغْلِبيُّ (ت: 130هـ=748م) يَذْكُرُ رَجُلًا:

تَصُدُّ عَضارِطُ الرُّكْبانِ عَنْهُ
وشهْرًا مِنْ تَخَمُّطِهِ يَصومُ

ديوان القطامي التغلبي. تح: محمود الربيعي، ج: 1، ص: 99.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (216هـ=831م، مِنْ شَواهِدِ الأَصْمَعيّ):

أَذاكَ خَيْرٌ أَيُّها العُضارِطُ

وأَيُّها اللَّمْعَــظَةُ العُمارِطُ؟

الصغاني، العباب الزاخر. تح: محمد حسن آل ياسين، ج: 1، ص: 127.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)

لا يشْمَتنَّ فُؤادُ الذِّمرِ إِنْ لَمَزا
عَضارِطُ السَّوْءِ أَوْ هَمُّازُهُمْ هَمَزا

[الذِّمْرُ: الشُّجاعُ. لَمَزَ: عابَ]

05 —

عَضارِطَة

اسم
5.1معنى من معاني «عَضارِطَة»

*العَضارِطَةُ: الخَدَمُ بِطَعامِ بُطُونِهِم.

ع
