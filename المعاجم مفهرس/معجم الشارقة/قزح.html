احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذرُ (ق ز ح):

في الأُوجارِيتِيَّةِ؛ ومنه الاسْمُ: qS; (قصح)qṣḥ بمَعْنى "نوعٌ من الكمّون"، وهُوَ يَتَّفِقُ مع الاسْمِ العَرَبيِّ (قَزْح) مَبْنًى ومَعْنًى؛ غَيرَ أَنَّ الاسْمَ الأوجاريتي خَضَعَ لإبْدال الزَّاي صادًا؛ ومن مَجيئِه بمَعْنى "الكَمّون": mrb' qS;m mrb' dblt mrb' Smqm؛ أَيْ "ربعٌ من الكمّون، ربعٌ من التِّينِ الجافِّ، ربعٌ من الزّبيب".

(DULAT, p. 705)

في الآراميّة؛ ومنه الاسْمُ:קִצְחָא(قِصحَا)qiṣḥā بمَعْنى "تابل/ كمّون أسود"؛ وهُوَ يَتَّفِقُ مع الاسْمِ العَرَبيِّ (قَزْح) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الاسْمَ الآراميّ خَضَعَ لإبْدالِ الزَّاي صادًا؛ ومن مَجيئِه بمَعْنى "قَزْح" ما وَرَدَ في ترجوم يوناثان:

"אֲרֵילָאבְּמוֹרִיגֵיבַּרְזְלָאמַדִירְכִיןקִיצְחָאוְגִילְגַלֵיעֲגָלָאעַלכַּמוֹנָאמְחַזְרִיןאֲרֵיבְחוּטְרָאחָבְטִיןקִצְחָאוְכַמוֹנָאבְמַחְבוֹטָא"؛

أَيْ "إن القَزحَ لا يُداسُ بنورجٍ حديدي ولا تُدار بَكَرةُ العَجلةِ على الكمّون وإنَّما يُخبط القَزْح بالقضيبِ والكمّون بالعصا".

(ת''י، יש': כח,כז)

في العِبْرِيَّةِ؛ ومنه الاسْمُ: קֶצַח (ق۪صَح)qeṣaḥ بمعنى "ما يُطْرَحُ في القِدْرِ مع الطَّعامِ من توابل، أبزارُ القِدْر"، وهُوَ يُوافِقُ الاسْمَ العَرَبيَّ (قِزَح) مَبْنًى ومَعْنًى؛ ومن مَجيئِه بمَعْنى "قِزْح" ما وَرَدَ في سِفْرِ إشعيا: "וְהֵפִיץ קֶצַח וְכַמֹּן יִזְרֹק וְשָׂם חִטָּה"؛ أَيْ "نَثَر القِزْحَ وذَرَّ الكَمُّونَ وأَلْقى الحِنْطَةَ".

(יש' 28: 25)

في السُّرْيانِيَّةِ؛ ومنه الاسْمُ:ܟܶܣܚܳܐ (ك۪سحَا)kesḥā يُطْلَقُ على "التابِل/ أبْزار القِدر/ البصل الصغار"، وهُوَ يَتَّفِقُ مع الاسْمِ العَرَبيِّ (قِزْح/ قَزْح) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الاسْمَ السُّريانيَّ خَضَعَ لإبْدالِ القافِ كافًا، والزّاي سينًا.

(R. Payne Smith's Thesaurus Syriacus. P. 1785)

01 —

قَزَحَ

فعل
1.1معنى من معاني «قَزَحَ»

* قَزَحَتْ (بِفَتْحِ الزّايِ) القِدْرُ تَقْزَحُ (بِفَتْحِ الزّايِ) قَزْحًا وقَزَحانًا: أَقْطَرَتْ ما خَرَجَ مِنْها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأَنْصارِيُّ (ت: 215ه=830م):

"قَزَحَتِ القِدْرُ تَقْزَحُ قَزَحًا وقَزَحانًا: إِذا أَقْطَرَتْ ما خَرَجَ مِنْها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 4، ص: 29.— المصدر
1.2معنى من معاني «قَزَحَ»

وــــــــــ الشَّيْءُ قَزْحًا: ارْتَفَعَ. (انْظُرْ: ق ح ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المُبَرَّدُ (ت: 286ه=899م):

"قَزَحَ الشَّيْءُ، وقَحَزَ: إِذا ارْتَفَعَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزَّمخشري، الفائق في غريب الحديث. تح: البجاوي وإبراهيم، ج: 3، ص: 190.— المصدر
1.3معنى من معاني «قَزَحَ»

وــــــــــ السِّعْرُ: عَلا. (انْظُرْ: ق ح ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الصَّغانِيُّ (ت: 650ه=1252ه):

"وقَزَحَ الشَّيْءُ، وقَحَزَ: إذا ارْتَفَع. ويُقالُ: سِعْرٌ قازِحٌ، وقاحِزٌ؛ أيْ: غالٍ".

الصغاني، التكملة. تح: الأبياري، ج: 2، ص: 87.— المصدر
1.4معنى من معاني «قَزَحَ»

وــــــــــ الكَلْبُ ونَحْوُهُ بَوْلَهُ وبِهِ قَزْحًا وقُزوحًا: رَفَعَ رِجْلَهُ وبالَ. وقيلَ: رَمى بِهِ ورَشَّهُ أَوْ أَرْسَلَهُ دَفْعًا. (انْظُرْ: ق ح ز)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَرْطاةُ بْنُ سُهَيّةَ (ت: 86ه=705م) يَذْكُرُ حالَهُ إِذا اعْتَرَضَهُ كَلْبٌ شَرِسٌ:

أَبْذى إِذا بوذيتُ مِنْ كَلْبٍ ذَكَرْ
أَسْوَدَ قَزّاحٍ يُعَوّي في السَّحَرْ

[أَبْذى: أَفْحَشَ في مَنْطِقِهِ]
الجاحظ، الحيوان. تح: عيون السُّود، ج: 1، ص: 186.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"قَزَحَ الكَلْبُ بَوْلَهُ يَقْزَحُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 85.— المصدر
1.5معنى من معاني «قَزَحَ»

وــــــــــ فُلانٌ القِدْرَ وغَيْرَها قَزْحًا: جَعَلَ فيها الأَقْزاحَ، وهِيَ التَّوابِلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"قَزَحْتُ القِدْرَ، وقَزَّحْتُها. ومِنْهُ مَليحٌ قَزيحٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 426.— المصدر
02 —

قَزِحَ

فعل
2.1معنى من معاني «قَزِحَ»

* قَزِحَ (بِكَسْرِ الزّايِ) الكَلْبُ ونَحْوُهُ بِبَوْلِهِ يَقْزَحُ (بِفَتْحِ الزّايِ): رَفَعَ رِجْلَهُ وبالَ. وقيلَ: رَمى بِهِ ورَشَّهُ أَوْ أَرْسَلَهُ دَفْعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِمُ بْنُ سَلّامٍ (ت: 224ه=839م):

"قَزِحَ الكَلْبُ بِبَوْلِهِ وقَزَحَ يَقْزَحُ فيهِما ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 293.— المصدر
2.2معنى من معاني «قَزِحَ»

وــــــــــ فُلانٌ فُلانَةَ: وَطِئَها بِذَكَرِهِ الصُّلْبِ. (ولَمْ يُسْتَعْمَلْ هذا الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ المَصْدَرُ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (206ه=822م) مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانِيِّ، يَذْكُرُ وَطْءَ أَحَدِهِمْ إِحْداهُنَّ:

ثُمَّ اعْتَلاها قَزَحًا وارْتَهَزا

وساقَها ثَمَّ سِياقًا بَزْبَزا

[ارْتَهَزَ: اهْتَزَّ. البَزْبَزُ: شِدّةُ السَّوْقِ]
الأزهري، تهذيب اللغة. تح: أحمد البردوني، ج: 13، ص: 173.— المصدر

قَزُحَ

فعل

* قَزُحَ (بِضَمِّ الزّايِ) فُلانٌ يَقْزُحُ (بِضَمِّ الزّايِ) قَزاحَةً: تَنَظَّفَ. وقيلَ: نَظُفَ.

ع

ابن القطّاع، كتاب الأفعال. عالم الكتب، ج: 3، ص: 36.— المصدر
04 —

قَزَّحَ

فعل
4.1معنى من معاني «قَزَّحَ»

* قَزَّحَ رَأْسُ الشَّجَرَةِ ونَحْوِها: انْشَعَبَ شُعَبًا كَبُرْثُنِ الكَلْبِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ ابْنِ عَبّاسٍ (ت: 68ه=688م) يَتَكَلَّمَ عَلى كَراهَةِ الصَّلاةِ إِلى صِنْفٍ مِنَ الشَّجَرِ:

"كَرِهَ أَنْ يُصَلِّي الرَّجُلُ إِلى الشَّجَرةِ المُقَزَّحةِ".

الهرويّ، الغريبين في القرآن والحديث. تح: المزيدي، ج: 5، ص: 1537.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والتَّقْزيحُ في رَأْسِ شَجَرةٍ أَوْ نَبْتٍ: إِذا انْشَعَبَتْ شُعَبًا مِثْلَ بُرْثُنِ الكَلْبِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، كتاب العين. تح: المخزومي وآخر، ج: 3، ص: 38.— المصدر

”قالَ الحَسَنُ بْنُ مُحَمَّدٍ الكَرْكِيُّ الصَّفَدِيُّ (ت: 723ه=1323م) يَسْتَبْشِرُ بِالمَطَرِ:

وكَمْ أَرَتْنا القِسِيُّ عَنْ قُزَحٍ
مُبَشِّراتٍ بِواكِفٍ سَرِبِ

[القِسِيُّ: جَمْعُ القَوْسِ. الواكِفُ: المَطَرُ المُنْهَلُّ. السَّرِبُ: السّائِلُ]
الصَّفديّ، الوافي بالوفيات. تح: جلال الأسيوطي، ج: 10، ص: 95.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحمد شوقي (ت: 1351ه=1932م) يُنْشِئُ عَلى لِسانِ أَحَدِ المُجّانِ في مَسْرَحِيّةٍ:

القَدَحا القَدَحا
الخَمْرُ تَنْفي التَّرَحا

قَصْرًا أَرى أَمْ فَلَكًا
وشَجَرًا أَمْ قُزَحا

أحمد شوقي، الأعمال الكاملة المسرحيّات. تح: سعد درويش، ص: 375.— المصدر
14 —

قِزْح

اسم
14.1معنى من معاني «قِزْح»

* القِزْحُ: ما تُخْرِجُهُ الحَيّةُ مِنَ الفَضَلاتِ. (ج) أَقْزاحٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"والأَقْزاحُ أَيْضًا: خُرْءُ الحَيّاتِ، واحِدُها: ‌قِزْحٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 4، ص: 28.— المصدر
14.2معنى من معاني «قِزْح»

وــــــــــ بِزْرُ البَصَلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458ه=1066م):

"‌القِزْحُ: بَزْرُ البَصَلِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي ج: 2، ص: 564.— المصدر
15 —

قُزَحِيَّة

اسم
15.1معنى من معاني «قُزَحِيَّة»

* القُزَحِيّةُ (في التَّشْريحِ): حِجابٌ مُسْتَديرٌ مُلَوَّنٌ خَلْفَ القَرْنِيّةِ يَتَوَسَّطُهُ ثُقْبٌ يُسَمّى: إِنْسانَ العَيْنِ أَوِ البُؤْبُؤَ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ يوحنّا عنحوري (1250ه=1835م) يُتَرْجِمُ كَلامًا عَلى الْتِهابِ القُزَحِيّةِ:

"قَدْ وَضَعْنا شَرْحَ الْتِهابِ القُزَحِيّةِ عَقِبَ شَرْحِ الْتِهابِ الأَغْشِيةِ المَصْلِيّةِ، وإِنْ كانَتِ القُزَحِيّةُ غَيْرَ داخِلةٍ في الأَغْشِيةِ المَصْلِيّةِ، لِأَنَّ هذِهِ الالْتِهاباتِ كَثيرًا ما تَكونُ مُصاحِبةً لِبَعْضِها".

لوئي شارل روش، منتهى الأغراض في علم الأمراض. تر: يوحنّا عنحوري ج: 2، ص: 8.— المصدر
16 —

قَزّاح

اسم
16.1معنى من معاني «قَزّاح»

* القَزّاحُ: بائِعُ الأَقْزاحِ، وهِيَ التَّوابِلُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ مَنْظورٍ (ت: 711ه=1311م):

"والقِزْحُ والقَزْحُ: التّابَلُ، وجَمْعُهُما أَقْزاحٌ؛ وبائِعُهُ قَزّاحٌ".

ابن منظور، لسان العرب. تح: أمين عبد الوهاب وآخر، ج: 11، ص: 150.— المصدر
17 —

قَزيح

اسم
17.1معنى من معاني «قَزيح»

* القَزيحُ: المُزَيَّنُ المُحَسَّنُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ مَكّي الصَّقلِّيُّ (ت: 501ه=1107م):

"وكَذلِكَ قَوْلُهُمْ: قَزيحٌ، وهُوَ عِنْدَ العَرَبِ: ‌المُزَيَّنُ ‌المُحَسَّنُ".

ابن مكِّي الصِّقلِّي، تثقيف اللِّسان وتلقيح الجنان. تح: مصطفى عطا، ص: 192.— المصدر
18 —

قَوازِح

اسم
18.1معنى من معاني «قَوازِح»

* القَوازِحُ: نُفّاخاتُ الماءِ الّتي تَنْتَفِخُ فَتَذْهَبُ. والمُفْرَدُ: قازِحةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو وَجْزَة السَّعْدِيُّ (ت: 130ه=747م) يَفْخَرُ بِقَوْمِهِ بَني سَعْدِ بْنِ بَكْرٍ:

لَهُمْ حاضِرٌ لا يُجْهَلونَ وصارِخٌ
كَسَيْلِ الغَوادي يَرْتَمي بِالقَوازِحِ

[الغَوادي: جَمْعُ غادِيةٍ، وهِيَ السَّحابةُ تَنْشَأُ وتُمْطِرُ غُدْوةً. الصّارِخُ: المُغيثُ]
محمّد بن المبارك، منتهى الطّلب من أشعار العرب. تح: طريفي، ص: 234.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

الأزهري، تهذيب اللغة. تح: درويش، ج: 4، ص: 28.— المصدر
19 —

مَقْزَحَة / مِقْزَحَة

اسم
19.1معنى من معاني «مَقْزَحَة / مِقْزَحَة»

* المَقْزَحَةُ، والمِقْزَحَةُ: ما يوضَعُ فيهِ القِزْحُ، وهُوَ التّابِلُ. (ج) مَقازِحُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نَصْرٍ الجَوْهَرِيُّ (ت: 393ه=1003م):

"والمِقْزَحةُ: نَحْوٌ مِنَ المِمْلَحةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ص: 396.— المصدر

مُقَزَّح

ع
