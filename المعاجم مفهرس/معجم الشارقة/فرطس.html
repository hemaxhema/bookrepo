فرطس
فَرْطَسَ(فعل) فِرْطاس(اسم) فُرْطوس(اسم) فُرْطوسَة(اسم) فِرْطيسَة(اسم)
01 —

فَرْطَسَ

فعل
1.1معنى من معاني «فَرْطَسَ»

* فَرْطَسَ الخِنْزيرُ أَوِ الفيلُ: مَدَّ أَنْفَهُ أَوْ خُرْطومَهُ. (آرامِيَّة)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"فِنطيسَةُ الخِنْزيرِ: خَطْمُهُ، وهِيَ الفِرْطيسَةُ ، والفَرْطَسَةُ فِعْلُهُ: إِذا مَدَّ خُرْطومَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 338.— المصدر
1.2معنى من معاني «فَرْطَسَ»

وــــــــــ: مَدَّ قَضيبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّغّانيُّ (ت: 650هـ=1252م):

"‌فَرْطَسَ الخِنْزيرُ: إِذا مَدَّ فُرْطوسَتَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، العباب الزاخر. تح: فير حسن، ج: 4، ص: 547.— المصدر
1.3معنى من معاني «فَرْطَسَ»

وــــــــــ فُلانٌ فُلانًا بِالكَلِمَةِ: قالَها في أَنْفِهِ. (ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"وفَرْطَسْتُهُ بالكَلِمَةِ، أَيْ: قُلْتُها في ‌فِرْطيسَتِهِ".

ابن عباد، المحيط. تح: آل ياسين، ج: 8، ص: 423.— المصدر
02 —

فِرْطاس

اسم
2.1معنى من معاني «فِرْطاس»

* الفِرْطاسُ مِنَ الأُنوفِ: العَريضُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"الفِرْطاسُ، بالكَسْرِ: العَريضُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: محمد أبي الفضل ، ج: 3، ص: 401.— المصدر
2.2معنى من معاني «فِرْطاس»

وــــــــــ مِنَ الرِّجالِ: السَّريعُ في مَشْيِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"الفِرْطاسُ: السَّريعُ ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1203.— المصدر
03 —

فُرْطوس

اسم
3.1معنى من معاني «فُرْطوس»

* الفُرْطوسُ: طَرَفُ أَنْفِ الرَّجُلِ. (ج) فَراطيسُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ عَبْدُ اللهِ بْنُ بُسْرٍ (ت: 96هـ=715م) يُحَذِّرُ عَبْدَ الرَّحْمَنِ بْنَ الجُنْدِيِّ أَلّا يَفْتَحَ بابَهُ لأَهْلِ المَعاصي:

"أُحَذِّرُكَ أَنْ تَسْتَفْتِحَ بابَ بَيْتِ أَهْلِكَ، ‌ولَكَ ‌فُرْطوسٌ كَفُرْطوسِ الخِنْزيرِ أَوْ خَطْمٌ كَخَطْمِ القِرْدِ".

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 1، ص: 855.— المصدر
3.2معنى من معاني «فُرْطوس»

وــــــــــ: خُرْطومُ الخِنْزيرِ أَوِ الفيلِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ عَبْدُ اللهِ بْنُ بُسْرٍ (ت: 96هـ=715م) يُحَذِّرُ عَبْدَ الرَّحْمَنِ بْنَ الجُنْدِيِّ أَلّا يَفْتَحَ بابَهُ لأَهْلِ المَعاصي:

"أُحَذِّرُكَ أَنْ تَسْتَفْتِحَ بابَ بَيْتِ أَهْلِكَ، ‌ولَكَ ‌فُرْطوسٌ كَفُرْطوسِ الخِنْزيرِ أَوْ خَطْمٌ كَخَطْمِ القِرْدِ".

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 1، ص: 855.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ قاسِمٌ السَّرَقُسْطِيُّ (ت: 302هـ=915م):

"‌الفُرْطوسُ: خَطْمُ الخِنْزيرِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 2، ص: 858.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ ابْنُ أَيْبَكَ الدّوادارِيُّ (ت: 713هـ=1313م) يَصِفُ دابَّةً ظَهَرَتْ عَلى نيلِ مِصْرَ:

"دَوْرُ حافِرِها قَدْرُ السُّكُرُجَّةِ، بِأَرْبَعَةِ أَظافِرَ شِبْهِ أَظافِرِ الجَمَلِ، وعَرْضُ ظَهْرِها ذِراعَانِ ونِصْفٌ، وطولُها مِنْ ‌فُرْطوسِها إلى ذَنَبِها خَمْسَةَ عَشَرَ قَدَمًا".

الدواداري، كنز الدرر وجامع الغرر. تح: هانس روبرت، ج: 9، ص: 81.— المصدر
3.3معنى من معاني «فُرْطوس»

وــــــــــ: قَضيبُ الخِنْزيرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو حاتِمٍ السِّجِسْتانِيُّ (ت: 255هـ=868م) يُعَدِّدُ أَسْماءَ الذَّكَرِ:

"والقَضيبُ يَجوزُ في كُلِّ ذي ذَكَرٍ ... ويُقالُ لَهُ مِنَ الخِنْزيرِ خَاصَّةً: فُرْطوسٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السجستاني، الفرق. تح: حاتم الضامن، ص: 235.— المصدر
3.4معنى من معاني «فُرْطوس»

وــــــــــ: الكَمَرَةُ الغَليظَةُ؛ وهِيَ رَأْسُ الذَكَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"الفَرَاطِيْسُ: ‏الكَمَرُ الغِلاظُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 8، ص: 423.— المصدر
04 —

فُرْطوسَة

اسم
4.1معنى من معاني «فُرْطوسَة»

* الفُرْطوسَةُ: خُرْطومُ الفيلِ أَوِ الخِنْزيرِ. (ج) فَراطيسُ. (آرامِيَّة)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"‌الفُرْطوسَةُ، والفِرْطيسَةُ: خَطْمُ ‌الفيلِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 277.— المصدر
4.2معنى من معاني «فُرْطوسَة»

وــــــــــ: طَرَفُ أَنْفِ الإِنْسانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الفُرْطوسَةُ: طَرَفُ أَنْفِ الرَّجُلِ".

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 61.— المصدر
4.3معنى من معاني «فُرْطوسَة»

وــــــــــ: فَمُ الأَسَدِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الشّاعِرُ (ت: 284هـ=897م، مِنْ شَواهِدِ البَنْدَنيجِيِّ):

فأَسْمَحَتْ نَفْسُهُ بِالسَّيْرِ مُعْتَصِمًا
ولَوْ تَجَرْثَمَ في ‌فُرْطوسَةِ الأَسَدِ

البندنيجي، التفقية في اللغة. تح: العطية، ص: 474.— المصدر
4.4معنى من معاني «فُرْطوسَة»

وــــــــــ: قَضيبُ الخِنْزيرِ. (ج) فَراطيسُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"‌‌‌الفُرْطُوسَةُ، خَطْمُ الخِنْزِيرِ... وقيلَ: هُوَ قَضيبُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 286.— المصدر
