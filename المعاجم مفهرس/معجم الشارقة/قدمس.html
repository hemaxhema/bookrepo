01 —

قُدامِس

اسم
1.1معنى من معاني «قُدامِس»

* القُدامِسُ: القَديمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ كُراعُ النَّمْل (ت: 309هـ=921م):

"القُدْموسُ والقُدامِسُ، والعادِيُّ: القَديمُ المَنْسوبُ إلى عاد ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: محمد العمري، ج: 1، ص: 372.— المصدر
1.2معنى من معاني «قُدامِس»

وــــــــــ: السَّيِّدُ الكَريمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وقُدامِسٌ: سَيِّدٌ كَريمٌ، وهو القُدْموسُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 1209.— المصدر
1.3معنى من معاني «قُدامِس»

وــــــــــ: الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"والقُدْموسُ والقُدامِس: الشَّديدُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 612.— المصدر

03 —

قُدْموس

اسم
3.1معنى من معاني «قُدْموس»

* القُدْموسُ: القَديمُ. (ج) قَداميسُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”

قالَ عَبِيدُ بن الأبْرص (ت: 25ق.هـ=598م) يَصِفُ مَنْزِلًا قَديمًا وَرِثَهُ:

ولَنا دارٌ وَرِثْناها عَنِ الْـ
أَقْدَمِ القُدْموسِ مِنْ عَمٍّ وخالِ

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 612.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ جَريرٌ (ت: 110هـ=728م) يَصِفُ جَيْشًا:

مَنْ يَتَّبِعْ غَيْرَ مَتْبوعٍ فإِنَّ لنا
في ابْنَيْ نِزارٍ نَصيبًا غَيْرَ مَخْسوسِ

وابْنا نِزارٍ أحَلّاني بِـمَنْزِلَةٍ
في رأْسِ أَرْعَنَ عادِيِّ القَدَاميسِ

شرح ديوان جرير. تح: محمد الصاوي، ج: 1، ص: 323.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ القاسِمُ بن سَلّام (ت: 224هـ=839م):

"العُدْمُلُ والعُدْمُلِيُّ: القَديمُ، والقُدْموسُ مِثْلُهُ".

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 280.— المصدر
”

وقالَ دِعْبِلٌ الخُزاعيُّ (ت: 246هـ=860م) يَحُثُّ أَسْعَدَ أَبا كَرْبٍ على التَّمَسُّكِ بوَصِيَّةِ أَخيهِ إفريقيس:

أوصيكَ خَيْرًا بالأَنامِ فإِنَّـما
لَكَ مُلْكُهُمْ والـمَنْصِبُ القُدْموسُ

دعبل الخزاعي، وصايا الملوك وأبناء الملوك. تح: نزار أباظة، ج: 1، ص: 48.— المصدر
3.2معنى من معاني «قُدْموس»

وــــــــــ: مُقَدَّمُ الجَيْشِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”

قالَ سلامةُ بن جَنْدلٍ (23ق.هـ=599م) يَمْدَحُ الحوفَزانَ وجَيْشَهُ العَظيمَ:

تَكادُ له الأَرْضُ، مِنْ رِزِّهِ
إذا سارَ تَرْجُفُ أَرْكانُـها

قَداميسُ يَقْدُمُها الحوفَزانُ
وأَبْـجَرُ تَـخْفِقُ عِقْبانُـها

[رِزُّ الجَيْشِ : صَوْتُ الجَيْشِ تَسْمَعُهُ مِنْ بَعِيدٍ،أَوْ هُوَ الصَّوْتُ تَسْمَعُهُ وَلا تَدْرِي مَا هُوَ]
ديوان سلامة بن جندل. تح: قباوة، ص: 254.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ العَجّاجُ (ت: 90هـ=709م) يَصِفُ عَسْكَرًا كَثيرًا:

بإِفْكِهِ حَتَّى رَأى الصُّبْحَ جَشَرْ

عَنْ ذي قَداميسَ لُهامٍ لو دَسَرْ

برُكْنِهِ أَرْكانَ دَمْخٍ لانْقَعَرْ

ديوان العجاج. تح: عزّة حسن، ج: 1، ص: 73.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ السِّكِّيتِ (244هـ=858م):

"القُدْموسُ: مُقَدَّمُ الجَيْشِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 2، ص: 118.— المصدر
3.3معنى من معاني «قُدْموس»

وــــــــــ: الـمَلِكُ الضَّخْمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"القُدْموسُ: الـمَلِكُ الضَّخْمُ ... والجميعُ: القَداميسُ".

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 251..— المصدر
3.4معنى من معاني «قُدْموس»

وــــــــــ: الصَّخْرَةُ العَظيمَةُ. وهو بهاءٍ أَيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"والقُدْموسَةُ: الصَّخْرَةُ العَظيمَةُ، والجَميعُ: القَداميسُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[القُدْموسُ: الصَّخْرَةُ العَظيمَةُ]
الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 251.— المصدر
3.5معنى من معاني «قُدْموس»

وــــــــــ: السَّيِّدُ الكَريمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وقُدامِسٌ: سَيِّدٌ كَريمٌ، وهو القُدْموسُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 1209.— المصدر
3.6معنى من معاني «قُدْموس»

وــــــــــ: الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّنَوْبَريُّ (ت: 334هـ=946م) يَصِفُ رَجُلًا عَظيمًا:

ناموسُ حُلْمٍ به انْفَرَدْتَ ولا
تُخْدَعُ عنه والحُلْمُ ناموسُ

قَدَّمَهُ مِنْكَ حينَ قَدَّمَهُ
أَبٌ قَديمُ الأَرومِ قُدْموسُ

ديوان الصّنوبريّ. تح: إحسان عبّاس، ج: 1، ص: 1449.— المصدر
3.7معنى من معاني «قُدْموس»

وــــــــــ: العَظيمُ منَ الإِبِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّاد (ت: 385هـ=995م):

"القُدْموسُ: العَظيمُ منَ الإِبِلِ".

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 86.— المصدر
3.8معنى من معاني «قُدْموس»

وــــــــــ: من الأشْياءِ: العَظيمُ الكَبيرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"وجَيْشٌ قُدْموسٌ: عَظيمٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ص: 612.— المصدر
