المقدّمة

النَّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (س ط م):

في العِبْريّة: وَرَدَ الفِعْلُ: סָתַם (ساتَم) sātam بِمَعْنى (سَدَّ، أغْلَقَ، رَدَمَ) يوافِقُ الفِعلَ العَربيَّ (سَطَمَ) معنًى ومبنًى غَيرَ أنَّ الفِعلَ العِبريَّ خَضَعَ لإبدالِ الطاء تاءً، ومِنْ مَجيئِهِ بِمَعْنى (سَطَمَ) ما وَرَدَ في سِفرِ الملوكِ الثاني: "וְכָל מַעְיְנֵי מַיִם תִּסְתֹּמוּ"، أي: "وكلَّ عَيْنِ ماءٍ تسْطُمُونَ".

(מל"ב 3: 19)

في السُّرْيانيّة: ومِنْهُ الفِعْلُ: ܣܛܰܡ (سْطَم) səṭam بِمَعْنى (سَدّ، ردّ) يُوافِقُ الفِعْلَ العَرَبيَّ (سطم) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (سَطَمَ): "ܣܛܘܡܝܗܝ ܠܦܘܡܗ ܫܿܩܪܐ"، أيْ: "اسطُمْ فَمَ الكَذّابِ".

(ActsThom 293:10)

ومِنهُ الاسمُ: ܐܶܣܛܳܡܳܐ/ ܐܶܣܛܡܽܘܼܡܳܐ (إ۪سطاما/ إ۪سطْمُوما) ʼesṭāmā\ ʼesṭəmūmā بِمَعْنى (ذُكرةُ الحَديدِ، الفولاذُ) يُوافِقُ الاسمَ العَرَبيَّ (إسطامٌ) لفظًا ومَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (إسطامٌ): "ܠܫܘܪܐ... ܕܐܣܛܡܘ̈ܡܐ"، أيْ: "لسور... مِنْ قِطعِ الإسطامِ".

(SyrAm2:52)

في الجِعزيّة: وَرَدَ الفِعْلُ: ሰጠመ (سَݖَمَ) sat'ama بِمَعْنى (غَطَسَ، غَمَرَ، غَرَقَ) يُوافِقُ الفِعْلَ العَرَبيَّ (سَطَمَ) مَبْنًى وفي بَعضِ مَعانيهِ غَيرَ أنَّ الفِعلَ الجِعزيَّ خَضَعَ لإبدالِ الطاءِ تاءً دَفعيّةً، ومِنْ مَجيئِهِ بِمَعنى (غَرقَ) ما وَرَدَ في سِفرِ الخروجِ: ወተመይጠ ባሕር ወተሰጥሙ ምስለ ሰረገላቲሆሙ ወመስተጽዕናኒሆሙ ወኵሉ ኀይሉ ለፈርዖን። أيْ: "وارتدَّتِ المياهُ وأغرَقَتِ المَركباتِ والفُرسانَ وكُلَّ جَيشِ فِرعونَ".

(ኦሪት ዘፀአት: 14:28)

01 —

سَطَمَ

فعل
1.1معنى من معاني «سَطَمَ»

* سَطَمَ (بِفَتْحِ الطّاءِ) فُلانٌ البابَ يَسْطُمُهُ (بِضَمِّ الطّاءِ) سَطْمًا: رَدَّهُ. (انظر: س د م).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيّ (ت: 231هـ=845م):

"وسَدَمْتُ البابَ وسَطَمْتُهُ واحِدٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 374.— المصدر
02 —

أَسْطام

اسم
2.1معنى من معاني «أَسْطام»

* الأَسْطامُ: حَديدَةٌ تَكونُ في طَرَفِ السَّهْمِ حَيْثُ يُعَلَّقُ حَجَرُ الرَّمْيِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخُوارَزْمِيُّ (ت: 387هـ=997م):

"الأَسْطامُ: حَديدَةٌ تَكونُ في طَرَفِ السَّهْمِ حَيْثُ يُعَلَّقُ حَجَرُ الرَّمْيِ".

الخوارزمي، مفاتيح العلوم. تح: الأبياري، ص: 270.— المصدر
2.2معنى من معاني «أَسْطام»

وــــــــــ: الفولاذُ الطَّبيعِيُّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ أَسْلَمَ الغافِقِيُّ (ت: 560هـ=1165م) يُعَرِّفُ الشّابُرْقانَ:

"فَالشّابُرْقانُ: هُوَ الفولاذُ الطَّبيعِيُّ، وهُوَ الذَّكَرُ، وهُوَ الأَسْطامُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن البيطار، الجامع لمفردات الأدوية والأغذية، ج: 2، ص: 265.— المصدر
03 —

إِسْطام

اسم
3.1معنى من معاني «إِسْطام»

* الإِسْطامُ: المِسْعارُ؛ وهُوَ حَديدَةٌ عَريضَةُ الرَّأْسِ تُحَرَّكُ بِها النّارُ وتُسَعَّرُ. (ج) أَساطيمُ. (مع). (وانْظُرْهُ أيْضًا في رَسْمِهِ)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ غُلامٌ لِكِسْرى (50ق ه=572م) يُجيبُ قَيْسَ بْنَ مَسْعودٍ - حينَ بُشِّرَ بِميلادِ ابْنِهِ (بِسْطام) وهُوَ واقِعٌ بِأَسْرِ كِسْرى - فَسَأَلَ الغُلامَ عَمّا في يَدِهِ يُقَلِّبُ بِهِ النّارَ:

"إِسْطامٌ".

ابن السائب الكلبي، جمهرة النسب. تح: ناجي حسن، ص: 506.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُحَذِّرُ مِنْ أَكْلِ أَمْوالِ النّاسِ بِالباطِلِ:

«... فَمَنْ قَضَيْتُ لَهُ مِنْ حَقِّ أَخيهِ شَيْئًا، فَلا يَأْخُذْهُ، فَإِنَّما أَقْطَعُ لَهُ قِطْعَةً مِنَ النّارِ يَأْتي بِها ‌إِسْطامًا في عُنُقِهِ يَوْمَ القِيامَةِ».

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 44، ص: 308.— المصدر
”وقالَ رَجُلٌ (125هـ=743م) يُخاطِبُ هِشامَ بْنَ عَبْدِ المَلِكِ:

"يا أَميرَ الـمُؤْمِنينَ، إِنَّ لِكُلِّ شَيْءٍ ‌إِسْطامًا، وإِنَّ ‌إِسْطامَ الـمَوالي العَرَبُ".

ابن منظور، مختصر تاريخ دمشق. تح: إبراهيم صالح، ج: 23، ص: 378.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السّائِبِ الكَلْبِيُّ (ت: 146هـ=763م) يَذْكُرُ سَبَبَ تَسْمِيَةِ بِسْطامِ بْنِ قَيْسِ بْنِ مَسْعودٍ بِهَذا الاسْمِ:

"بِسْطامُ، وهُوَ أَوَّلُ مَنْ سُمِّيَ مِنَ العَرَبِ بِسْطامًا، كانَ أَبوهُ في حَبْسِ كِسْرى، فَبُشِّرَ بِهِ - وبَيْنَ يَدَيْهِ غُلامٌ يُؤَرِّثُ النّارَ بِإِسْطامٍ حَديدٍ - فَقالَ: "أَيُّ شَيْءٍ هَذا؟" قالَ: "إِسْطامٌ"، فَسَمّاهُ بِسْطامَ".

ابن السائب الكلبي، جمهرة النسب. تح: ناجي حسن، ص: 506.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُحَمَّد بْنُ مَنْكِلي النّاصِرِيُّ (ت: 770هـ=1369م) يُعَدِّدُ صُنوفًا مِنَ اللَّعِبِ بِالنّارِ:

"تَحْمي إِسْطامًا، أَوْ سِكّينًا، أَوْ سِكَّةَ حَديدٍ، أَوْ ضَبَّةَ حَديدٍ؛ حَتّى تَصيرَ كَأَنَّها النّارُ، ثُمَّ الْحَسْها بِلِسانِكَ وأَسْرِعْ. وهَذا بابٌ مِنَ الخِفَّةِ أَيْضًا".

محمد بن منكلي ، الحيل في الحروب. تح: نبيل محمد، ص: 142.— المصدر
3.2معنى من معاني «إِسْطام»

وــــــــــ: القِطْعَةُ مِنَ الشَّيْءِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

«فَمَنْ قَضَيْتُ لَهُ مِنْ حَقِّ أَخيهِ شَيْئًا، فَلا يَأْخُذْهُ، فَإِنَّما أَقْطَعُ لَهُ قِطْعَةً مِنَ النّارِ يَأْتي بِها ‌إِسْطامًا في عُنُقِهِ يَوْمَ القِيامَةِ».

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 44، ص: 308.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م) يَشْرَحُ قَوْلَهُ ﷺ: «فَإِنَّما أقْطَعُ لَهُ ‌إسْطامًا مِنَ النّارِ»:

"أَرادَ ‌بِالإِسْطامِ: القِطْعَةَ مِنْها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 350.— المصدر
3.3معنى من معاني «إِسْطام»

وــــــــــ مِنَ السَّيْفِ: حَدُّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو تَمّامٍ (ت: 231ه=846م) يَمْدَحُ أَبا سَعيدٍ مُحَمَّدَ بْنَ يوسُفَ:

يا واهبَ العيسِ الهَموسِ بِرَحْلِها
والأَعْوَجِيِّ بِسَرْجِهِ ولِجامِهِ

والواهِبَ الصَّمْصامَةَ السَّيْفَ الَّذي
يَجْري زُعافُ الـمَوْتِ في إِسْطامِهِ

الخطيب التبريزي، شرح ديوان أبي تمام. تح: الأسمر، ج: 2، ص: 123.— المصدر
3.4معنى من معاني «إِسْطام»

وــــــــــ: خَشَبَةٌ بارِزَةٌ في مُقَدَّمِ السَّفينَةِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ المَقْريزِيُّ (ت: 845هـ=1441م) يَذْكُرُ حَفْرَ بِئْرِ مَسْجِدٍ:

"فَقالَ الحَفّارُ: إِنّي أَجِدُ في البِئْرِ شَيْئًا ... فَلَمّا قَلَعَهُ فارَ الماءُ وأَخْرَجَهُ، وإِذا هُوَ ‌إِسْطامُ مَرْكَبٍ، وهُوَ الخَشَبَةُ الَّتي تُبْنى عَلَيْها السَّفينَةُ".

المقريزي، المواعظ والاعتبار. تح: أيمن فؤاد سيد، ج: 4، ص: 891.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمَنِ البَرْقوقِيُّ (1338هـ=1919م) تاريخُ الطَّبْعِ، يَتَحَدَّثُ عَنِ الأُسْطولِ الأَنْدَلُسِيِّ:

"ومِنْ حِيَلِهِمْ أَنَّهُمْ يَجْعَلونَ في مُقَدَّمِ الـمَرْكَبِ هَناةً كَالفَأْسِ يُسَمّونَها اللِّجامَ، وهِيَ حَديدَةٌ طَويلَةٌ مُحَدَّدَةُ الرَّأْسِ، وأَسْفَلُها مُجَوَّفٌ كَسِنانِ الرُّمْحِ تَدْخُلُ في أَسْفَلِها في خَشَبَةٍ كَالقَناةِ بارِزَةٍ في مُقَدَّمِ الـمَرْكَبِ يُقالُ لَها ‌الإسْطامُ".

عبد الرحمن البرقوقي، مجلة البيان، القاهرة، ع: 54، ج: 54، ص: 5.— المصدر
04 —

أُسْطُمّ

اسم
4.1معنى من معاني «أُسْطُمّ»

* الأُسْطُمُّ: مُجْتَمَعُ الشَّيْءِ ووَسَطُهُ. (ج) أَساطِمُ. وهُوَ بِتاءٍ أَيْضًا. (مع). (وانْظُرْهُ أيْضًا في رَسْمِهِ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=709م) يَفْتَخِرُ بِكَثْرَةِ جَيْشِ قَوْمِهِ:

كَباذِخِ اليَمِّ سَقاهُ اليَمُّ

لَهُ نَواحٍ، ولَهُ أُسْطُمُّ

ديوان العجاج. تح: عزة حسن، ص: 377.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يُقَدِّمُ لِقَصيدَتِهِ:

وَصَلْتُ مِنْ ‌حَنْظَلَةَ ‌الأُسْطُمّا

والعَدَدَ الغُطامِطَ الغِطَمّا

ديوان رؤبة. تح: البروسي، ص: 183.— المصدر
4.2معنى من معاني «أُسْطُمّ»

وــــــــــ: أَصْلُ الشَّيْءِ وأَساسُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110هـ=728م) يَمْدَحُ سُلَيْمانَ بْنَ عَبْدِ المَلِكِ وعَبْدَ العَزيزِ:

قَدْ رَضِيَ النّاسُ بِهِ فَسَمِّهِ

يا لَيْتَها قَدْ خَرَجَتْ مِنْ فُمِّهِ

حَتّى يَعودَ ‌الـمُلْكُ ‌في أُسْطُمِّهِ

ديوان جرير. تح: نعمان أمين طه، ج: 3، ص: 1038.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"وعادَ الـمُلْكُ في أُسْطُمِّهِ: في أَصْلِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ع

أُسْطُمَّة

اسم
5.1معنى من معاني «أُسْطُمَّة»

وــــــــــ مِنَ القَوْمِ: خِيارُهُمْ وأَشْرافُهُمْ. (انْظُرْ: ص ط م، ط س م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"أُسْطُمَّةُ البَحْرِ: لُغَةٌ في أُصْطُمِّهِ؛ وهِيَ مُجْتَمَعُهُ ووَسَطُهُ ... وأُسْطُمَّةُ الحَسَبِ كَذَلِكَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 221.— المصدر
06 —

سِطام

اسم
6.1معنى من معاني «سِطام»

* السِّطامُ: الـمِسْعارُ؛ وهُوَ حَدِيدةٌ عَريضَةُ الرَّأْسِ تُحَرَّكُ بِهَا النَّارُ وتُسَعَّرُ. (ج) سُطُمٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«مَنْ قَضَيْتُ لَهُ بِحُجَةٍ أَراها، فَاقْتَطَعَ بِهَا مِنْ مالِ أَخيهِ، فإنمَّا يَقْتَطِعُ بِهَا قِطْعَةً مِنَ النَّارِ، تَكونُ ‌ سِطامًا في رَقَبتِهِ يَوْمَ القِيامَةِ».

أبو يعلى، المسند. تح: السناري، ج: 9، ص: 440.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م):

"ويُقالُ لِلحَديدَةِ الَّتي تُحْرَثُ بِها النّارُ: سِطامٌ ‌وإِسْطامٌ، إِذا فُطِحَ طَرَفُها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 350.— المصدر
6.2معنى من معاني «سِطام»

وــــــــــ: حَدُّ السَّيْفِ وغَيْرِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«العَرَبُ سِطامُ النّاسِ».

”[قالَ ابنُ الأَثيرِ في"النهاية": أَيْ هُمْ في شَوْكَتِهِمْ وحِدَّتِهِمْ كالحَدِّ مِنَ السَّيْفِ]الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 178.وقالَ: كَعْبُ بْنُ جُعَيْلٍ (ت: 86هـ=705م) يَصِفُ عُدَّتَهُ لِلحَرْبِ:

أَعِنّي بِخَوّارِ العِنانِ تَخالُهُ
إِذا راحَ يَرْدي بِالـمُدَجَّجِ أَحْرَدا

وأَبْيَضَ مَصْقولَ ‌السِّطامِ مُهَنَّدًا
وذا حَلَقٍ مِنْ نَسْجِ داوُدَ مُسْرَدا

سيبويه، الكتاب. تح: هارون، ج: 1، ص: 170.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رَجُلٌ مِنْ مَقاوِلِ حِمْيَرَ يُدْعى عَمْرًا (154هـ=771م) مِنْ شَواهِدِ أَبي عَمْرِو بْنِ العَلاءِ، يُجيبُ أَباهُ حينَ سَأَلَهُ عَنْ أَحَبِّ السُّيوفِ إِلَيْهِ:

"الصَّقيلُ الحُسامُ، الباتِرُ المِجْذامُ، الماضي ‌السِّطامِ".

القالي، الأمالي. ج: 1، ص: 192.— المصدر
6.3معنى من معاني «سِطام»

وــــــــــ: الدَّرَوَنْدُ؛ وهُوَ الَّذي يُرَدُّ بِهِ البابُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"ويُقالُ لِلدَّرَوَنْدِ: ‌سِطامٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 350.— المصدر
6.4معنى من معاني «سِطام»

وــــــــــ: صِمامُ القارورَةِ وسَدادُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيّ (ت: 231هـ=845م):

"يُقالُ لِسَدادِ ‌القِنّينَةِ: الفِدامُ والسِّطامُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 349.— المصدر
6.5معنى من معاني «سِطام»

وــــــــــ: أَصْلُ الشَّيْءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"السُّطُمُ: الأُصولُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 12، ص: 350.— المصدر
07 —

سَطْم

اسم
7.1معنى من معاني «سَطْم»

* السَّطْمُ: حَدُّ السَّيْفِ وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"‌‌السَّطْمُ والسِّطامُ: حَدُّ السَّيْفِ وغَيْرِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 837.— المصدر
08 —

سُطُمَّة

اسم
8.1معنى من معاني «سُطُمَّة»

* السُّطُمَّةُ: وَسَطُ الشَّيْءِ ومُجْتَمَعُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"‌وسُطُمَّةُ البَحْرِ ... وأُسْطُمَّتُهُ ‌وأُسْطُمُّهُ: وَسَطُهُ ومُجْتَمَعُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 442.— المصدر
8.2معنى من معاني «سُطُمَّة»

وــــــــــ مِنَ القَوْمِ: خِيارُهُمْ وأَشْرافُهُمْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"‌وسُطُمَّةُ البَحْرِ والحَسَبِ وأُسْطُمَّتُهُ ‌وأُسْطُمُّهُ: وَسَطُهُ ومُجْتَمَعُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 442.— المصدر
