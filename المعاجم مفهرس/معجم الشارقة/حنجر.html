01 —

حَنْجَرَ

فعل
1.1معنى من معاني «حَنْجَرَ»

*حَنْجَرَتْ عَيْنُ فُلانٍ: غارَتْ فَهُوَ مُحَنْجِرٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النّابِغَةُ الجعدِيُّ (ت: 50ه=670م) يَهجو رَجُلاً من بَني سَعْدٍ:

ويَغْمِزُ مِنْهُ الفائِقَيْنِ كِليهِما
على شَهْوَةٍ غَمْزَ الطَبِيبِ المُحَنْجِرا

ديوان النّابغة الجعديّ. تح: واضح الصّمد،، ص: 77.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458ه=1066م):

"حَنْجَرَتْ عَيْنُهُ: غارَتْ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم ، تح: هنداوي، ج: 4، ص: 54.— المصدر
1.2معنى من معاني «حَنْجَرَ»

وــــــــــفُلانٌ: أصابَهُ داءٌ في البَطْنِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"المُحَنْجِرُ: داءٌ يُصِيبُ في البَطْنِ، ...، يُقالَ: حَنْجَرَ الرَّجُلُ، فهُوَ مُحَنْجِرٌ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ).

ابن منظور، لسان العرب. تح: أمين عبد الوهاب، وآخر، ج: 3، ص: 355.— المصدر
1.3معنى من معاني «حَنْجَرَ»

وــــــــــ الحَيَوانَ وغَيْرَهُ: ذَبَحَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"حَنْجَرْتُ الرَّجُلَ: إِذا ذَبَحْتُهُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْنُ دُرَيْدٍ، جَمْهَرَةُ اللُّغَةِ. تح: رمزي بعلبكي، ج: 2، ص: 1134.— المصدر
02 —

حُنْجار

اسم
2.1معنى من معاني «حُنْجار»

*الحُنْجارُ: داءٌ يُصِيبُ البَطْنَ. (انظر: م ح ن ج ر).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المُغيرَةُ بْنُ شُعْبَةَ (ت: 50ه=670م) يُكَذِّبُ جارِيَةً اشتراها فادَّعَتْ أنَّها حاملٌ:

"هذا الحُنْجارُ مِنْكِ، ووَطِئَهَا قَبْلَ الاسْتِبْراءِ".

البلاذري، أنساب الأشراف. تح: سهيل زكّار، وآخر، ج: 13، ص: 349.— المصدر
03 —

حَنْجَرة

اسم
3.1معنى من معاني «حَنْجَرة»

*الحَنْجَرَةُ: باطِنُ الحُلْقُومِ. (ج) حَنْجَرٌ وحَناجِرُ وحَناجيرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ ‌‌‌المُفَضَّلُ ‌النُّكْرِيُّ (ت: 108ق.ه=517م) يَفْخَرُ بِبَأْسِ قَوْمِهِ:

رَمَيْنا في وُجُوهِهِمُ بِرَشْقٍ
تَغَصُّ بِهِ الحَناجِرُ والحُلوقُ

البحتري، الحماسة. تح: محمد إبراهيم حور، وآخر، ص: 123.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وفي القُرْآنِ الكَريمِ (11هـ=632م) قالَ اللهُﷻ:

﴿وإذْ زَاغَتِ الأَبْصارُ وَبَلغَتِ القُلُوبُ الحَناجِرَ﴾. (الأحزاب: 10)

”وقالَ النَّبِيُّ ﷺ(ت 11ه=632م):

«لا يُجاوِزُ إيمانُ البَرْبَرِيِّ حَنْجَرَتَهُ».

التاريخ الكبير، البخاري، دائرة المعارف العثمانية، ج8، ص: 378.— المصدر
”وقالَ رُؤْبَةُ بْنُ العَجَّاجِ (127 هـ=745م) يَمْدَحُ القاسِمَ بْنَ مُحمَّدٍ الثَّقفيّ:

تَهوي رُؤوسُ القَاحِراتِ القُحَّرِ

إِذا هَوَتْ بَيْنَ اللَّهى والحَنْجَرِ

ديوان رؤبة بن العجاج. تح: وليم البروسي، ص: 60.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ جُريجٍ (ت: 150هـ=766م): مُفَسِّرًا قَولَه تَعالى: ﴿وَيَأْتِيهِ ٱلْمَوْتُ مِن كُلِّ مَكَانٍۢ وَمَا هُوَ بِمَيِّتٍۢ ۖ ﴾. (إبراهيم:17):

"تُعَلَّقُ نَفْسُهُ عِنْدَ حَنْجَرَتِه ولا تَخْرُجُ مِنْ فِيهِ فَيَمُوتُ ، ولا تَرْجِعُ إلى مَكانِها مِنْ جَوْفِهِ فَتَنْفَعُهُ الحَياةُ".

 تفسير ابن جريج. تح:علي حسن عبدالغني، ص:185.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أحمد بنُ محمَّد القُرْطُبِيّ (ت: 671هـ=1272م) يُبَيِّنُ أحْوالَ المَوْتى:

"ومِنَ المَوْتى مَنْ تُجْذَبُ نَفْسُهُ رُوَيْدًا حَتَّى تَنْحَصِرَ في الحَنْجَرَةِ".

القرطبي، التذكرة بأحوال الموتى وأمور الآخرة. تح: الصادق بن محمد، ج: 1، ص: 253.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عُثمانُ بْنُ عَبْدِ العَزيزِ (ت: 1282ه=1865م) يُبَيِّنُ عَظيمَ خَلْقِ اللهِ تَعالى:

"وخالَفَ سُبْحانَهُ بَيْنَ أَشْكالِ الحَناجِرِ في الأَصْواتِ".

ابن منصور التّميميّ، فتح الحميد. تح: العريفي، وآخرين، ج: 1، ص: 2086.— المصدر
3.2معنى من معاني «حَنْجَرة»

0الحَنْجَرَةُ الذَّهَبِيَّةُ: الصَّوْتُ الحَسَنُ المُطْرِبُ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بنيلوبي مري (ت: 1422ه=2000م):

" ... فَإِنَّهُ عَلَيْنا أَنْ نَتَذَكَّرَ أَنَّ اخْتِيارَها لِمِثْلِ هذِهِ الحَنْجَرةِ الذَّهَبِيَّةِ في كُلِّ مَرَّةٍ، هُوَ الذي يَجْعَلُ مِنْها آلِهَةً".

بنيلوبي مري، سلسلة عالم المعرفة - العدد 208. ج: 1، ص: 247.— المصدر
3.3معنى من معاني «حَنْجَرة»

0بَلَغَتِ القُلُوبُ الحَناجِرَ: اشْتَدَّ الخَوْفُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وفي القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ ﷻ:

﴿وإِذْ زاغَتْ الأبْصارُ وَبَلَغَتِ القُلُوبُ الحَناجِرَ﴾. (الأحزاب: 10).

”قالَ أَبو سَعيدٍ الخُدريُّ (5هـ=627م) يَوْمَ الخَنْدَقِ يُخاطِبُ النَّبيَّ ﷺ:

"قُلْنَا يَوْمَ الْخَنْدَقِ: يَا رَسُولَ اللَّهِ، زَاغَتِ الْأَبْصَارُ ‌وَبَلَغَتِ ‌الْقُلُوبُ ‌الْحَنَاجِرَ، فَهَلْ مِنْ شَيْءٍ نَقُولُهُ؟ قالَ: " قُولُوا: اللَّهُمَّ آمِنْ رَوْعَتَنَا، وَاسْتُرْ عَوْرَاتِنَا".

مُعجم ابن الأعرابي، تح: عبد المحسن إبراهيم، ج: 1، ص: 406.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الثَّعالِبيُّ (ت: 429 هـ=1039م) فِي حَدِيثِهِ عَنْ أَبِي الطَيِّبِ المُتَنَبِّي:

"دارَتْ كَأْسُ المَوْتِ دِهاقًا، وعادَ القِرْنُ لِلْقِرْنِ عِناقًا، ‌بَلَغَتِ ‌القُلوبُ ‌الحَناجِرَ، وشافَهَتِ السُّيوفُ المَناحِرَ".

الثعالبي، سحر البلاغة وسر البراعة. تح: عبدُ السلام الحوفي، ص: 162.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الصَّيْقَلِ في المقَامَةِ الرَّسْعَنِيَّةِ (ت: 701 هـ=1302م):

"...وَ أَجْزَلَتْ وَسَاوِسَ الإِعْدامِ، وَأَهْزَلَتْ بِها مَعالِمُ الأَعْلامِ، حَتَّى ‌بَلَغَتِ ‌القُلوبُ ‌الحَناجِرَ، وشامَ المَلَأُ عَلى الِجيَفِ الخَناجِرَ".

ابْنُ الصَّيْقلِ الجَزريّ، المَقاماتُ الزَّيْنِيَّةُ. تح: عباس مصطفى الصالحي، ج: 1، ص: 197.— المصدر
04 —

حُنْجور

اسم
4.1معنى من معاني «حُنْجور»

*الحُنْجورُ: الحَلْقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجَّاجُ (ت: 90 هـ=708م) يَصِفُ جَمَلًا:

في شَعْشَعانِ عُنُقٍ يَمْخورِ

حابي الحُيودِ فارِضِ الحُنْجورِ

[الحُيُودُ:واحدتها حَيْدَةٌ، وهي العقدة في قرن الوعل ،وكل نتوء في القرن والجبل وغيرهما .والمراد بها هنا أطراف عظام الجمل]

ديوان العجاج. تح: عبد الحفيظ السطلي، ص: 348.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والهِبْلَعُ: الأَكُولُ، العَظيمُ اللَّقْمِ، الواسِعُ ‌الحُنْجُورِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 282.— المصدر
4.2معنى من معاني «حُنْجور»

وــــــــــ: قارورَةٌ لِمَسْحوقِ الطّيبِ. (وبِتاءٍ في لِسانِ أَهْلِ اليَمامَةِ ، عنِ النَّضْرِ).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206 هـ=821م):
‌"والحُنْجورُ: القارورَةُ".

أَبْو عَمْرو الشَّيْبانيّ، الجيم. تح: الإبياري، ج: 1، ص: 215.— المصدر
”وقالَ الشّاعِرُ ( 231ه=846م، مِنْ شَواهِدِ ابْنِ الأعْرابِيِّ):

لَوْ كانَ خَزُّ واسِطٍ وسَقَطُهْ

حُنْجُورُهُ وحُقُّهُ وسَفَطُهْ

وعالِجٌ نَصِيُّهُ وسَبَطُهْ

والشّامُ طُرًّا زَيْتُهُ وحِنَطُهْ

يَأْوِي إلَيها أَصْبَحَتْ تُقَسِّطُهْ

الصغاني، التكملة والذيل والصلة. تح: إبراهيم الإبياري، ج: 2، ص: 466.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيُّ (ت: 1205ه=1790م):

"والحُنجودُ: كَزُنْبُورٍ: الحُنْجُرَةُ، كالحُنْجورِ، بِالرَّاءِ، نَقَلَهُ الصّاغانِيُّ، وقارورَةٌ طَويلَةٌ للذّريرَةِ، ووِعاءٌ كالسَّفَطِ الصَّغِيرِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ).

مرتضى الزبيدي، تاج العروس. تح: عبد العزيز مطر، ج: 8، ص: 46.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ زَيْنَب العامِلي (ت: 1332ه=1914م) تَصِفُ صُورَةً مُعَلّقَةً بِأَحَدِ حيطانِ الغُرْفَةِ:

"وأَمامَها حُنْجورانِ مِنَ الطُّيوبِ".

زينب العاملي، الدر المنثور في طبقات ربّات الخدور. مؤسسة هنداوي، ص: 237.— المصدر
05 —

مُحَنْجِر

اسم
5.1معنى من معاني «مُحَنْجِر»

*المُحَنْجِرُ: داءٌ يُصيبُ البَطْنَ أَوِ العُنُقَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رِياحُ الدُّبَيْريُّ (ت: 150ه=767م):

أَفائِقٌ بالحَـلْقِ أَمْ مُحَنْجِرُ

باللَّقْمِ ثَبْتٌ غَدَرِيّ مُقَعِّرُ

أبو عمرو الشيباني، الجيم، تح: العزباوي، وآخر، ج3.، ص: 109.— المصدر
”وقالَ أَبو عَمْرٍو الشَّيْبانيّ (ت: 206 هـ=821م):

‌"والمُحَنْجِرُ: داءٌ في العُنُقِ، يَرِمُ مِنْهُ".

أَبْو عَمْرو الشَّيْبانيّ، الجيم. تح: الإبياري، ج: 1، ص: 211.— المصدر
5.2معنى من معاني «مُحَنْجِر»

وــــــــــ: الأسَدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"المُحَنْجِرُ: ‏الأَسَدُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

مرتضى الزبيدي ، تاج العروس. تح: إبراهيم الترزي، ج: 10، ص: 549.— المصدر
