النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ذ ع ر):في السُّرْيانيّة: وَرَدَ الفِعلُ: ܕܥܰܪ (دْعَر) dəʽar بِمَعْنى (فَزِعَ/ خافَ مِنَ التَّوبيخِ) ويُوافِقُ الفِعلَ العَرَبيَّ (ذَعِرَ) مَبْنًى ومَعْنًى، ومِن مَجيئِهِ بِمَعْنى (ذَعَرَ): "ܠܩܫܝܫܐ ܠܐ ܬܕܥܘܪ"، أيْ: "لا تُذعِرْ مُسنًّا".

(1TimH 5:1)

01 —

ذَعَرَ

فعل
1.1معنى من معاني «ذَعَرَ»

* ذَعَرَ (بِفَتْحِ العَيْنِ) فُلانٌ في القَوْمِ يَذْعَرُ (بِفَتْحِ العَيْنِ) ذَعْرًا، وذُعْرًا، وذَعرَةً: انْتَهَكَ أَعْراضَهم.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بْنُ لَجَأٍ (ت: 105ه=724) يَهْجو قَوْمًا:

لَقَدْ ذَعَرْنا قَديمًا في نِسائِكُمُ
فلَمْ تَغاروا ولَمْ تُسْتَنْكَرِ الذُّعَرُ

شعر عمر بن لجأ التيمي. تح: الجبوري، ص: 102.— المصدر
1.2معنى من معاني «ذَعَرَ»

وــــــــــ فُلانًا: أَخافَهُ وأَفْزَعَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيسِ (ت: 80ق ه=544م):

فإمّا تَرَيْني اليَوْمَ في رَأْسِ شاهِقٍ
فقَدْ أَغْتَدي أَقودُ أَجْرَدَ تائِقا

‌وقَدْ ‌أذْعَرُ الوَحْشَ الرِّتاعَ بِغِرَّةٍ
وقَدْ أَجْتَلي بيضَ الخُدورِ الرَّوائِقا

ديوان امرئ القيس. تح: محمد أبو الفضل إبراهيم، ص: 198.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسّانُ بْنُ ثابِتٍ (4ه=643م) يَذْكُرُ هُذَيلًا لما أَصابَتْ مِنْ أَصْحابِ النَّبِيِّ ﷺ يَوْمَ الرَّجيعِ:

فإنْ لا أَمُتْ أَذْعَرْ هُذَيلًا بِغارَةٍ
كَغادي الجِهامِ المُغتَدي بِإفاءِ

سيرة ابن هشام. تح: السقا، ج: 2، ص: 182.— المصدر
”وقالَ النَّبِيُّ ﷺ (ت: 11ه=632م) يُخاطِبُ رَجُلًا بَعَثَهُ خَلْفَ عَلِيِّ بْنِ أَبي طالِبٍ:

"الْحَقْهُ ولا ‌تَذْعَرْهُ مِنْ خَلفِهِ".

العسقلاني، المطالب العالية. تح: صاولي، ج: 9، ص: 412.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=759م) على لِسانِ رَجُلٍ في مَثَلٍ ضَرَبَهُ:

"واللهِ لَأَسْكُتَنَّ حَتّى أَنْظُرَ ماذا يَصْنَعُ، ولا ‌أَذْعَرُهُ، ولا أُعْلِمُهُ أَنّي قَدْ عَلِمْتُ بِهِ".

ابن المقفع، كليلة ودمنة. المطبعة الأميرية، ص: 61.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711ه=1311م):

"ذَعَرَهُ يَذْعَرُهُ ذَعْرًا فانْذعَرَ، وهو مُنْذَعِرٌ، وأذْعَرَه، كِلاهُما: أَفْزَعَهُ وصَيَّرَهُ إلى الذُّعْرِ".

ابن منظور، لسان العرب. دار صادر، ج: 4، ص: 306.— المصدر
”وقالَ ابْنُ الفُوَطِيِّ (ت: 723= 1886م) يَذْكُرُ الخَليفةَ المُقْتَفي:

"وفَوَّضَ إلى فَخْرِ الدّينِ إيلاجِكْ سورا وأَعْمالَها إلى بَوادي الكُوفَةِ لا ‌يَذْعَرُها ذاعِرٌ خَوْفًا مِنَ العَساكِرِ".

ابن الفوطي، مجمع الآداب. تح: الكاظم، ج: 2، ص: 571.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ حَيدَر الحِلّيُّ (ت: 1304ه= 1886م) يَذْكُرُ أميرَ المُؤمِنينَ عَلِيًّا:

أَصَبْرًا وسِرْبُ العِدى راتِعٌ
يَروحُ ويَغْدو بلِا ذاعِرِ

ديوان حيدر الحلي. تح: الخاقاني، ج: 1، ص: 76.— المصدر
02 —

ذَعِرَ

فعل
2.1معنى من معاني «ذَعِرَ»

*ذَعِرَ(بِكَسْرِ العَيْنِ) فُلانٌ وغَيْرُهُ يَذْعَرُ (بِفَتْحِ العَيْنِ) ذَعَرًا: خافَ وفَزِعَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه=632م):

«لا يَزالُ الشَّيْطانُ ذَعِرًا مِنَ المُؤْمِنِ ما حافَظَ على الصَّلَواتِ الخَمْسِ فإذا ضَيَّعَهُنَّ تَجَرَّأ عَلَيْهِ وأوْقَعَهُ في العَظائِمِ».

الزمخشري، ربيع الأبرار. مؤسسة الأعلمي، ج: 2، ص: 270.— المصدر
”وقالَ عُبَيْدُ اللهِ بْنُ الحُرِّ الجُعْفِيُّ (ت: 68ه=687م):

ويَوْمٍ بِباجِسْرى هَزَمْتَ وغودِرَتْ
جَماعَتُهُم صَرْعى لَدى جانِبِ الجِسْرِ

فوَلَّوْا سِراعًا هارِبينَ كَأَنَّهُم
رَعيلُ نَعامٍ بِالفَلا شُرَّدٍ ذُعْرِ

ياقوت الحموي، معجم البلدان. دار صادر، ج: 1، ص: 314.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م) في رِوايَةِ ابْنِ سِيدَهْ:

"ورجُلٌ ‌ذَعِرٌ: ‌مُنْذَعِرٌ".

ابن سيده، المخصص. تح: جفال، ج: 3، ص: 357.— المصدر
2.2معنى من معاني «ذَعِرَ»

وــــــــــ: دَهِشَ مِنَ الحَياءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"الذَّعَرُ: الدَّهَشُ مِنَ الحَياءِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 314.— المصدر
03 —

ذُعِرَ

فعل
3.1معنى من معاني «ذُعِرَ»

* ذُعِرَفُلانٌ: فَزِعَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق ه=544م) يَصِفُ حِصانًا:

لَهُ أُذُنانِ تَعْرِفُ العِتْقَ فيهِما
كَسامِعَتَيْ مَذْعورَةٍ وسْطَ رَبْرَبٍ

ديوان امرئ القيس. تح: أنور أبو سويلم وآخرين، ص: 92.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه= 632م) في خَبَرِ موسى والعَبْدِ الصّالِحِ:

«فانْطَلَقَ إلى أَحَدِهِمْ بادِيَ الرَّأْيِ فَقَتَلَهُ، ‌فَذُعِرَ عِنْدَها مُوسى- عَلَيْهِ السَّلامُ- ذَعْرَةً مُنْكَرَةً».

صحيح مسلم. تح: عبد الباقي، ج:4، ص: 1850.— المصدر
”وقالَ كَعْبُ بْنُ زُهَيرٍ (ت: 26ه=645م):

وإذا ما أَشاءُ أَبْعَثُ مِنْها
مَطْلِعَ الشَّمْسِ ناشِطًا مَذْعورا

السكري، شرح ديوان كعب بن زهير. دار الكتب المصرية، ص: 161.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=759م) يَذْكُرُ غُرابًا:

"فبَيْنَما هو ذاتَ يَومٍ ساقِطٌ في وَكْرِه إذْ بَصُرَ بِصَيّادٍ قَبيحِ المَنْظَرِ سَيِّئِ الخَلْقِ، على عاتِقِهِ شَبَكَةٌ، وفي يَدِهِ عَصا مُقْبِلًا نَحْو الشَّجَرَةِ؛ ‌فذُعِرَ مِنْهُ الغُرابُ".

ابن المقفع، كليلة ودمنة. المطبعة الأميرية، ص: 178.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَديمِ (ت: 660ه=1262م) يَحْكي خَبَرَإيقاعِ امْرِئِ القَيْسِ بِبَني أَسَدٍ:

"ونَزَلَ بِالقُرْب مِنْهم في اللَّيْلِ ‌‌فذُعِرَ ‌القَطا ‌فَطارَ ‌عَنْ ‌مَجاثِمِهِ، فمَرَّ بِبني أَسَدٍ".

ابن العديم، بغية الطلب. تح: زكار، ج: 4، ص: 2018.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ الطُّلْبَةِ (ت: 1270ه=1856م) يَرْثي خالَهُ:

جودي عَلَيْهِ وبَكّيهِ إذا ذُعِرَتْ
أبْناءُ عامِرَ مِنْ أمْرٍ بمِفْظاعِ

ديوان ابن الطلبة. تح: ابن الشبيه، ص: 273.— المصدر
04 —

أَذْعَرَ

فعل
4.1معنى من معاني «أَذْعَرَ»

*أَذْعَرَالسَّبْعُ ونَحْوُهُ فُلانًا: أَفْزَعَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ وَرَقةُ بْنُ نَوْفَلٍ (ت: 1 ق ه=611م) يَذْكُرُ النَّبِيَّ ﷺ:

إنّي رَأَيْتُ أَمينَ اللهِ واجَهَني
في صورَةٍ أُكْمِلَتْ في أَحْسَنِ الصُّوَرِ

ثُمَّ اسْتَمَرَّ فكادَ ‌الخَوْفُ ‌يُذْعِرُني
مِمّا يُسَلِّمُ ما حَوْلي مِنَ الشَّجَرِ

الفاكهي، أخبار مكة. تح: ابن دهيش، ج: 4، ص: 54.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه= 632م):

"مَنْ ‌أَذْعَرَ ‌مُسْلِمًا ‌أَطالَ ‌اللهُ ‌ذُعْرَهُ يَوْمَ القِيامَةِ».

ابن منده، معرفة الصحابة. تح: صبري، ص: 733.— المصدر
”وقالَ عُبَيْدُ بْنُ أَيّوبَ العَنْبَرِيُّ (ت: 100ه=719م) يُخاطِبُ بَعيرَهُ:

وكَيْفَ تُرَجّيها وقَدْ حالَ دونَها
مِنَ الأَرْضِ مَخْشِيُّ التَّنائِفِ مُذْعِرُ

مجلة المورد. العدد: 2، 1974م، ج: 3، ص: 127.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ القاسِمِ (ت: 179ه=795م):

"رَأَيْتُ مِنْ قَوْلِ مالِكٍ أَنَّ الفارِسَ إذا جَمَحَ بِهِ فَرَسُهُ، إنّما ذَلِكَ شَيْءٌ فَعَلَهُ، أَمّا إذا أَذْعَرَهُ أَوْ خافَ فجَمَحَ، فسَبَبُ جَمْحِهِ مِنْ قِبَلِ فارِسِهِ فهْوَ ضامِنٌ لِما أَصابَ".

سحنون، المدونة. دار الكتب العلمية، ج: 3، ص: 500.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأبّارِ (ت: 658ه=1260) يَذْكُرُ المُعتَضِدَ بْنَ عَبّادٍ:

"فجاءَ مِنْها بِمَهولاتٍ ‌تُذْعِرُ مَنْ سَمِعَها فَضْلًا عَمَّنْ عايَنَها".

ابن الأبار، الحلة السيراء. تح: مؤنس، ج: 2، ص: 41.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ جَعْفرٌ الحِلّيُّ (1298ه=1880) يَرْثي جَعْفَرًا القَزوينيَّ:

اللهُ أَكْبَرُ ما جَرى
أَمْ أَيُّ خَطْبٍ قَدْ عَرا

اللهُ أَيُّ مُلِمَّةٍ
طَرَقَتْ فأَذْعَرَتِ الوَرى

جعفر الحلي، سحر بابل. مطبعة العرفان، ص: 193.— المصدر

ذَعَّرَ

* ذعَّرَفُلانٌ فُلانًا: أَفزَعَهُ وَخَوَّفَهُ.

ع

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)

تَولَّتْ تَتَّقي بِالفَرِّ بَأْسي

06 —

انْذَعَرَ

فعل
6.1معنى من معاني «انْذَعَرَ»

* انْذَعَرَ القَوْمُ: فَزِعوا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ دُرَيْدُ بْنُ الصِّمّةِ (ت: 8ه=630م) يَذْكُرُ خَوْفَ الفُرْسِ مِنْ قَوْمِهِ:

ويَوْمَ طَعْنِ القَنا الخَطِّيِّ تَحْسَبُهُم
عاناتِ وَحْشٍ دَهاها صَوْتُ مُنْذَعِرِ

ذُعَرَة

اسم
16.1معنى من معاني «ذُعَرَة»

* الذُّعَرَةُ مِنَ النّاسِ: ذو العُيوبِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو الدَّرْداءِ (ت: 32ه=653م) يُخاطِبُ رَجُلًا لَقِيَهُ في بَيْتِ المَقْدِسِ:

"‏يا عَبْدَ اللهِ، أَذُعَرَةٌ أَنا؟"

تفسير يحيى بن سلام. تح: هند شلبي، ج: 2، ص: 788.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّملِ (ت: 309ه=921م):

"‌ورَجُلٌ ‌ذاعِرٌ وذُعَرَة، وذُعَراتٌ للجَميعِ: إذا كانَ ذا عُيوبٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في عَصْرِ الدّولِ والإِماراتِ)

كراع، المنجد. تح: أحمد مختار عمر، ص: 206.— المصدر
16.2معنى من معاني «ذُعَرَة»

وــــــــــ: طائِرٌ صَغيرٌ كَثيرُ الفَزَعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309ه=921م):

"والذُّعَرَةُ: طُوَيْرَةٌ تَكونُ في الشَّجَرِ تَدْخُلُ فيهِ تَهُزُّ ذَنَبَها، لا تُرى إلّا مَذْعورَةً". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

كراع، المنجد. تح: أحمد مختار عمر، ص: 206.— المصدر
