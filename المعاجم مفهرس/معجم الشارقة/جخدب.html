01 —

جَخْدَبَ

فعل
1.1معنى من معاني «جَخْدَبَ»

* جَخْدَبَالأَسَدُ، وغَيْرُهُ: أَسْرَعَ وجَرُؤَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ نَشْوانُ الحِمْيَرِيُّ (ت: 573ه=1178م):

"جَخْدَبَ: الجَخْدَبَةُ: السُّرْعَةُ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

نشوان الحميريّ، شمس العلوم. تح: العمري و آخرين، ص: 1006.— المصدر
02 —

جُخادِب

اسم
2.1معنى من معاني «جُخادِب»

* الجُخادِبُ: ضَرْبٌ مِنَ الجَنادِبِ، وهُوَ الجَرادُ الضَّخْمُ الأَخْضَرُ الطَّويلُ الرِّجْلَيْنِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ يَزيدُ بْنُ مَفَرِّغٍ الحِمْيَرِيُّ (ت: 69ه=688م):

تَحْسِبُ الجُنْدُبَ الـجُخادِبَ فيها يَبْتَغي الظِّلَّ زائِفًا مَنْقودا

أبو حنيفة الدّينوري، كتاب النبات. تح: برنهارد ليفين، ص: 70.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170ه=787م):

"وأَبو جُخادِبٍ: مِنَ الجَنادِبِ".

الخليل بن أحمد، العين. تح: هنداوي، ج: 1، ص: 221.— المصدر
”وقالَ أَبو زِيادٍ الكِلابِيُّ (ت:200هـ=816م):

إِذا صَنَعَتْ أُمُّ الفُضَيْلِ طَعامَها إِذا خُنْفُساءُ ضَخْمَةٌ وجُخادِبُ

أبو حنيفة الدّينوري، كتاب النبات، تح: برنهارد ليفين، ص: 69.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الشَّريفُ بَرَكاتُ (ت: 1024ه=1615م) يَصِفُ حَرَّ مَوْسِمِ الجَوْزاءِ:

بِيَوْمٍ مِنَ الجَوْزاءِ مُسْتَوْقِدِ الحَصى تَلُوذُ بِأَعْضادِ المَطايا جُخادِبُهْ

ماجد المطيريّ، ديوان الأمراء وتحفة الشّعراء. مطابع الفرزدق التجارية، ج: 1، ص: 39.— المصدر
2.2معنى من معاني «جُخادِب»

وــــــــــ: الضَّخْمُ الجَسيمُ مِنَ الرِّجالِ والإِبِلِ. (ج): جَخادِبُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ بْنُ العَجّاجِ (ت: 145ه=762م) يَصِفُ جَمَلًا عَظيمَ الخَلْقِ:

إِذا رَأَيْنَ خَلْفَهُ الجَخادِبا

وزَبَدًا مِنْ هَدْرِهِ زُغادِبا

تُحْسَبُ في أَرْآدِهِ غَنادِبا

مجموع أشعار العرب. تح: وليم البروسيي، ج: 3، ص: 170.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711ه=1311م):

"والجُخادِبُ، والجُخادِيُّ، كُلُّهُ: الضَّخْمُ الغَليظُ مِنَ الرِّجالِ والجِمالِ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثْ)

ابن منظور ، لسان العرب. تح: أمين عبد الوهاب وآخر،، ج: 2، ص: 192.— المصدر
2.3معنى من معاني «جُخادِب»

وــــــــــ: دابَّةٌ نَحْوُ الحِرْباءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسَائِيُّ (ت: 189ه=805م):

"هَذا أَبو جُخادِبٍ قَدْ جاءَ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 384.— المصدر
2.4معنى من معاني «جُخادِب»

وــــــــــ: الحُمْطوطُ، وهُوَ دُوَيْبَّةٌ مُلَوَّنَةٌ تَعيشُ بَيْنَ الأعْشابِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"أَبو جُخادِبٍ: دابَّةٌ، واسْمُهُ الحُمْطوطُ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب. تح: أمين عبد الوهاب وآخر، ج: 2، ص: 193.— المصدر
2.5معنى من معاني «جُخادِب»

وــــــــــ: الذَّكَرُ مِنَ الجَرادِ والجِعْلانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وجُخْدُبٌ وجُخادِبٌ، وهُوَ الذَّكَرُ مِنَ الجَرادِ والجِعْلانِ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ص: 1112.— المصدر
2.6معنى من معاني «جُخادِب»

وــــــــــ: ضَرْبٌ مِنَ الخُنْفُساءِ ضَخْمٌ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفَيْروزآبادِيُّ (ت: 817ه=1414م):

"والجُخادِبَةُ...ضَرْبٌ مِنَ الجَنادِبِ ومِنَ الجَرادِ، ومِنَ الخُنْفُساءِ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروز ابادي، القاموس. تح: العرقسوسي، ص: 66.— المصدر
03 —

جُخادِبَى

اسم
3.1معنى من معاني «جُخادِبَى»

*الجُخادِبَى: ضَرْبٌ مِنَ الجَنادِبِ، وهُوَ الجَرادُ الضَّخْمُ الأَخْضَرُ الطَّويلُ الرِّجْلَيْنِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سِيبَوَيْهِ (ت: 180ه=796م) يَذْكُرُ الرُّباعيَّ المَزيدَ بِالأَلِفِ:

"ويَكونُ على مِثالِ (فُعالِلَى)، وهُوَ قَليلٌ: قالوا: جُخادِبى، وهُوَ اسْمٌ".

سيبويه، الكتاب. تح: عبد السلام هارون، ج: 4، ص: 294.— المصدر
”وقالَ الشاعر (ت: 370ه=981م. من شواهد الأزهري):

وعانَقَ الظِّلَّ أَبو جُخادِبا

الأزهري، التهذيب. تح: عبد السلام سرحان، ج: 7، ص: 636.— المصدر
3.2معنى من معاني «جُخادِبَى»

وــــــــــ: الضَّخْمُ الجَسيمُ مِنَ الرِّجالِ والإِبِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458ه=1066م):

"والجُخادِبُ والجُخادِبَى، كُلُّهُ الضَّخْمُ الغَليظُ مِنَ الرِّجالِ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. دار إحياء التراث العربي، ج: 1، ص: 190.— المصدر
3.3معنى من معاني «جُخادِبَى»

وــــــــــ: دابَّةٌ نَحْوُ الحِرْباءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ (ت: 515ه=1122م) يَذْكُرُ أَوْزانَ الرُّباعِيِّ المَزيدِ:

"وعَلى (فُعالِلى) نَحْوَ جُخادِبى لِدابَّةٍ كَالعَظايَةِ".

ابن القطاع الصقلي، أبنية الأسماء والأفعال والمصادر. تح: أحمد عبد الدايم، ص: 299.— المصدر

06 —

جَخْدَب / جُخْدَبُ

اسم
6.1معنى من معاني «جَخْدَب / جُخْدَبُ»

*الجَخْدَبُوالجُخْدَبُ مِنَ الخَيْلِ، أَوْ الجِمالِ: الضَّخْمُ الجِسْمِ، العَريضُ الصَّدْرِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ: العَجّاجُ (ت: 90ه=708م) يَصِفُ جَمَلًا:

إِنّ لَنا شَدّاخَةً مُعارِكا

قَرْمَ قُرومٍ صَلْهَبًا ضُبارِكا

مِنْ آلِ مُرٍّ جَخْدَبًا مُماحِكا

ديوان العجاج. تح: عزّة حسن، ص: 126.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ بْنُ العَجّاجِ (ت: 145ه=762م) يَصِفُ فَرَسًا:

شَدّاخَةً ضَخْمَ الضُّلوعِ جُخْدَبا

الخليل بن أحمد، العين. تح: هنداوي، ج: 1، ص: 221.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عُصْفورٍ الإِشْبيلِيُّ (ت: 669ه=1271م) يَذْكُرُ أَبْنِيَةَ رُباعِيّ الأُصولِ:

"أَمّا جُخْدَبٌ وبُرْقَعٌ وجُؤْذَرٌ فَلا حُجَّةَ فيها؛ لِأَنَّهُ يُقالُ: جُخْدُبٌ وبُرْقُعٌ وجُؤْذُرٌ بِالضَّمِّ، فَيُمْكِنُ أَنْ يَكونَ الفَتْحُ تَخْفيفًا". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن عصفور الإشبيلي، الممتع الكبير في التصريف، تح: قباوة، ص: 54.— المصدر

”قالَ أَبو الجَرّاحِ نَوْفَلُ العُقَيْلِيُّ (ت: 150ه=767م) يَذْكُرُ أَشْباهَ العَظاءِ والحِرْباءِ:

"والجُخْدُبُ: دابَّةٌ نَحْوُ ذلِكَ أَيْضًا". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 384.— المصدر
7.3معنى من معاني «جُخْدَب»

وــــــــــ مِنَ الرِّجالِ والجِمالِ: الضَّخْمُ الجَسيمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"ورُبَّما سُمِّيَ الرَّجُلُ الضَّخْمُ جُخْدُبًا". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ص: 1135.— المصدر
7.4معنى من معاني «جُخْدَب»

وــــــــــ: الأَسَدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الحُسَيْنُ ابْنُ خَالَوَيْهِ (ت: 370ه=980م) يَذْكُرُ أَسْماءَ الأَسَدِ:

"والجُخْدُبُ، وهُوَ الجَرادُ أَيْضًا". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن خالويه، أسماء الأسد. تح: محمود الدرويش، ص: 11.— المصدر
08 —

جُخْدُب

اسم
8.1معنى من معاني «جُخْدُب»

*الجُخْدُبُ: الذَّكَرُ مِنَ الجَرادِ والجِعْلانِ.(ج): جَخادِب.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دَرَيْدٍ (ت: 321ه=933م):

"وجُخْدُبٌ وجُخادِبٌ، وهُوَ الذَّكَرُ مِنَ الجَرادِ والجِعْلانِ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ص: 1112.— المصدر
8.2معنى من معاني «جُخْدُب»

وــــــــــ مِنَ الرِّجالِ: الضَّخْمُ الوَجْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت 385ه=995م):

"الجُخْدُبُ: الوارِمُ الضَّخْمُ ". (وهذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 448.— المصدر
