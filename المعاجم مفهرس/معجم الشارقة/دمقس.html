دمقس
دَمْقَسَ(فعل) دِمْقاس(اسم) دِمَقْس(اسم)
المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (د م ق س):

في العِبريّةِ: وَرَدَ الاسْمُ: דְּמֶשֶׁק (دْم۪ش۪ق) dəmešeq بِمَعْنى (نَوعٌ مِنَ الحَريرِ، ديباجٌ، إبْرَيْسَمٌ) ويُوافِقُ الاسْمَ العَرَبيَّ (دِمَقْسٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسمَ العِبريَّ خَضَعَ لإبدالِ السِّينِ شينا، كَما خَضَعَ لِقَلبٍ مَكانيٍّ فتقدم رابعُهُ على ثالثه، ومِن مَجيئِهِ بِمَعْنى (دِمَقْسٌ) ما وَرَدَ في سِفرِ عاموس: "בְּנֵי יִשְׂרָאֵל הַיֹּשְׁבִֿים בְּשֹׁמְרוֹן בִּפְאַתֿ מִטָּה וּבִֿדְֿמֶשֶׁק עָרֶשׂ"، أيْ: "بَنو إسرائيلَ الجالِسونَ في السَّامرةِ في زاويةِ السَّريرِ وعلى دِمَقْسِ الفِراشِ".

(עמ' 3: 12)

المعانى الكلّيّة

  1. نوعٌ منَ الحَرير
01 —

دَمْقَسَ

فعل
1.1معنى من معاني «دَمْقَسَ»

* دَمْقَسَفلانٌ الثَّوْبَ: نَسَجَهُ مِنَ الدِّمَقْسِ، وهُوَ الحَريرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُتَلَمِّسُّ (ت: 50 ق ه=573م):

مَحْبوكَةً حُبِكَتْ مِنْها نَمانِمُها
مِنَ المُدَمْقَسِ، أَوْ مِنْ فاخِرِ الطُّوطِ

ديوان المتلمس ، تح: الصيرفي، ص: 303.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجَّاجُ (ت: 90ه=708م) يَصِفُ فَتاةً:

خَوْدًا تَخالُ رَيْطَها المُدَمْقَسا

ومَيْسَنانِيًّا لَها مُمَيَّسا

ديوان العجاج، تح: السطلي، ج: 1، ص: 190.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ظافِر الحَدّاد (ت: 528ه=1134م) يُشَبِّه ماءَ النّيلِ بالحَريرِ النّاعِمِ:

ودَرَّجَتْ ماءَهُ الصَّبا، فَحَكى
ثَوْبَ حَريرٍ مُدَمْقَسٍ أَزْرَقْ

ديوان ظافر الحداد، تح: حسين نصار، ص: 36.— المصدر
02 —

دِمْقاس

اسم
2.1معنى من معاني «دِمْقاس»

* الدِّمْقاسُ: الحَريرُ. (مع).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَليٍّ الفارِسيُّ (ت: 377ه=987م):

"دِمَقْسٌ، ودِمْقاسٌ، ومِدَقْسٌ، وثَوْبٌ مُدَمْقَسٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص، تح: خليل جفال، ج: 1، ص: 384.— المصدر
03 —

دِمَقْس

اسم
3.1معنى من معاني «دِمَقْس»

* الدِّمَقْسُ: الحَريرُ. (مع).

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق ه=544م) يَصِفُ شَحْمَ الجَزورِ الَّتي نَحَرَ لِلْعَذارى:

فَظَلَّ العَذارى يَرْتَمينَ بِلَحْمِها
وشَحْمٍ كَهُدّابِ الدِّمَقْسِ المُفَتَّلِ

ديوان امرئ القيس، تح: عليان، وآخرين،ص: 182.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أمَيَّةُ بنُ أبي الصَّلْتِ (ت: 5هـ = 626م) يذكرُ أهلَ الجَنَّة:

وتَحتَهُمُ نَمارِقُ مِن دِمَقْسٍ
ولا أَحَدٌ يُرى فيهِمْ سَئيمُ

ديوان أمية بن أبي الصَّلْت،تح: الجبيلي، ص: 123.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارٌ (ت:167ه=784م) يَتَغَزَّلُ بِعَبْدَةَ:

وتُريكَ عَيْنَيْ جُؤْذَرٍ فَرِقٍ
بِالرَّوْضِ لَمْ تُكْحَلْ مِنَ الرَّمَدِ

أَحْوى المَدامِعِ زانَ قامَتَهُ
حُلَلُ الدِّمَقْسِ تَظَلُّ في أَوَدِ

ديوان بشار ، تح: إحسان عباس، ج: 1، ص: 258.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ حازِمُ القَرْطاجَنّي (ت: 684ه=1285م) يمدح النبيّﷺ:

صِحابٌ بَرَوا هامَ العُداةِ، وكَمْ قَرَوا
صَفيفَ شِواءٍ أَوْ قَديرٍ مُعَجَّلِ!

وكَمْ أَكْثَروا ما طابَ مِنْ لَحْمِ جَفْرَةٍ
وشَحْمٍ كَهُدّابِ الدِّمَقْسِ المُفَتَّلِ!

ديوان حازم القرطاجني، تح: عثمان الكعاك، ص: 95.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ الطُّلْبَةِ (ت: 1272ه=1856م):

وكَشْحٍ لَطيفٍ كَالجَدائِلِ طَيُّهُ
كَلَمْسِ الدِّمَقْسِ ذاتُ خَلْقٍ مُعَذْلَجِ

ديوان محمد الطلبة، تح: محمد بن ابّوه ص: 184.— المصدر
3.2معنى من معاني «دِمَقْس»

وــــــــــ: الثَّوْبُ المَنْسوجُ مِنَ الحَريرِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْريُّ (ت: 90ه=709م) يَصِفُ حِسانًا نَزَلْنَ بِأَرْضِهِ:

وما مالَ النَّهارُ وهُنَّ فيها
يُخَدِّرْنَ الدِّمَقْسَ ويَحْتَوينا

ديوان الراعي النميري، تح: راينهرت فايبرت، ص: 271.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ شُهَيْدٍ الأندلسِيُّ (ت:426ه=1035م) يَصِفُ إوَزَّةً:

"كَأَنَّما ذُرَّ عَلَيْها الكافورُ، أَوْ لَبِسَتْ غِلالَةً مِنْ دِمَقْسِ الحَريرِ".

ابن شُهيد الأندلسي، رسالة التوابع والزوابع. تح: بطرس البستاني، ص: 146.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النُّوَيْريّ (ت:733ه=1333م) يَصِفُ قَطيعًا مِنَ النِّعاجِ:

"وكَأَنَّما لَبِسْنَ الدِّمَقْسَ سِرْبالاً، واتَّخّذْنَ السُّنْدُسَ سِرْوالاً".

النويري، نهاية الأرب، تح: مفيد قميحة وآخرين، ج: 9، ص: 197.— المصدر
3.3معنى من معاني «دِمَقْس»

وــــــــــ: الثَّوبُ منَ الكَتّان.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنّى (ت:209ه=825م):

"الدِّمَقْسِ مِنَ الكَتّانِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: هارون، ج: 9، ص: 392.— المصدر
