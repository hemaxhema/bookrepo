المعانى الكلّيّة

  1. الانْحِدارُ في تَتابُعٍ
01 —

دَحْرَجَ

فعل
1.1معنى من معاني «دَحْرَجَ»

* دَحْرَجَالسُّكْرُ بِفُلانٍ: أَخَذَهُ كُلَّ مَأْخَذٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّريفُ العَقِيلِيُّ (ت: 450ه=1058م):

فلَمْ تَزَلْ تَعْمَلُ كاساتُنا
حَتّى اكْتَسَتْ وَجْنَتُهُ وَرْدا

ودَحْرَجَ السُّكْرُ بِنا فانْثَنى
مِنْ فَوْقِ خَدِّي واضِعًا خَدّا

ديوان الشريف العقيلي، تح: المحاسني، ص: 129.— المصدر
1.2معنى من معاني «دَحْرَجَ»

وــــــــــ فُلانٌ الشَّيْءَ: حَرَّكَهُ فانْدَفَعَ مُنْحَدِرًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«مَنْ دَحْرَجَ عَنْ طَريقِ المُسْلِمينَ شَيْئًا يُؤْذيهِمْ كَتَبَ اللهُ - تَبارَكَ وتَعالى- لَهُ بها حَسَنَةً، ومَنْ كُتِبَتْ ‏لَهُ حَسَنَةٌ وَجَبَ لَهُ بها الجَنَّةُ».

الخرائطي، مكارم الأخلاق، تح: الحميري، ج: 1، ص: 982.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170ه=787م):

"الدُّحْروجَةُ: كُلُّ ما دَحْرَجْتَهُ مِنْ طِينٍ أَو غَيْرِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: هنداوي، ج: 2، ص: 10.— المصدر
”وقالَ ابْنُ الرُّومِيِّ (ت: 283هـ=896م) يَذُمُّ أَهْلَ زَمانِهِ:

كَمَنْ دَحْرَجَ دُرَّ البَحْـ
ــرِ في بَحْرِ البَلاليعِ

ديوان ابن الرّوميّ، تح: بسج، ج: 2، ص: 329.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبَّار (ت: 658هـ=1260م) يَذْكُرُ حَفْلًا - أَشْبَهَ بالسِّيركِ - يُلْعَبُ فيه بالكُرَةِ:

يُدَحْرِجُها ماشِيًا ثِنْيَها
عَلى حَذَرٍ مِشْيَةَ الأَنْكَبِ

ديوان ابن الأبار، تح: الهرّاس، ص: 105.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الرّافِعيُّ (ت: 1356هـ=1937م) يَتَحَدَّثُ عَنْ تَلاعُبِ الشَّيْطانِ بالإِنْسانِ:

"إذا أَرادَ الشَّيْطانُ أنْ يَسْخَرَ مِنْهُ دَحْرَجَهُ في الشَّوارِعِ".

الرافعي، وحي القلم، تح: الجويدي، ج: 1، ص: 192.— المصدر
1.3معنى من معاني «دَحْرَجَ»

وــــــــــ الوَتَرَ: مَلَّسَهُ وشَدَّ فَتْلَهُ. (انظر: ح د ر ج)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"حِدْرِجانُ: فِعْلِلانُ مِنْ قَوْلِهِمْ: حَدْرَجْتُ السَّوْطَ وغَيْرَهُ، إذا فَتَلْتَهُ فَتْلًا شَديدًا. أَوْ يَكونُ مِنَ المَقْلوبِ مِنْ قَوْلِهِمْ: حَدْرَجَ ودَحْرَجَ".

ابن دريد، الاشتقاق، تح: هارون، ص: 522.— المصدر
1.4معنى من معاني «دَحْرَجَ»

وــــــــــ الجُعَلُ فَضَلاتِهِ: دَوَّرَها كَالبُنْدُقَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السَّرَقُسطِيُّ (ت:400هـ=1010م):

"دَحْرَجَ الجُعَلُ دُحْروجَتَهُ: وهُوَ ما يُدَحْرِجُهُ مِنَ القَذَرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، كتاب الأفعال، تح: شرف، ج: 3، ص: 343.— المصدر
02 —

تَدَحْرَجَ

فعل
2.1معنى من معاني «تَدَحْرَجَ»

* تَدَحْرَجَالشَّيْءُ: تَحَرَّكَ فانْدَفَعَ مُنْحَدِرًا.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ قَيْسٌ بْنُ الخَطيمِ (ت: 2ق هـ=620م) يَصِفُ الحَرْبَ:

لَوَ انَّكَ تُلْقي حَنْظَلًا فَوقَ بَيْضِنا
تَدَحْرَجَ عَنْ ذِي سامِهِ المُتَقارِبِ

ديوان قيس بن الخطيم، تح: السامرائي وآخر، ص: 33.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو النَّجمِ العِجْلِيُّ (ت: 130هـ=748م) يَتَغَزَّلُ:

وكَفَلًا يَرْتَجُّ في تَبَجُّجِ

رَيّانَ لَمْ يَزْنَجْ ولَمْ يُزَنَّجِ

إِذا مَشَتْ سالَتْ ولَمْ تَدَحْرَجِ

ديوان أبي النجم العجلي، تح: جمران، ص: 117.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ حِماشُ بْنُ الأبْرَشِ المُقْعَدُ الكِلابِيُّ(195هـ=811م) يَمْدَحُ أبا بَكْرِ بْنَ عَبْدِ اللهِ بْنِ مُصْعَبٍ:

إِنِّي لَآتيكَ ولَوْ تَدَحْرُجا

زَحْفًا عَلى كوعِ يَدي أَوْ زَلَجا

الزبير بن بكار، جمهرة نسب قريش، تح: شاكر، ص: 174.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القَرَافِيُّ (684هـ=1285م) يَتَحَدَّثُ عَن حُكْمِ التَّطَيُّبِ في الإحْرامِ:

"لَوْ سَقَطَ عَلَيْهِ طيبٌ أَوْ تَدَحْرَجَ، ثُمَّ اسْتَيْقَظَ فنَزَعَهُ، فإنْ اسْتَدامَ افْتَدى".

القرافي، الذخيرة، تح: حجي، ج: 3، ص: 308.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ البَهْكَليُّ (ت: 1248هـ=1832م) يَصِفُ حالَ الأَميرِ:

"تَدَحْرَجَ في الإِمارَةِ بَيْنَ إِخْوانِهِ تَدَحْرُجَ الكُرَةِ بَيْنَ أَقْدامِ اللّاعِبينَ".

البهكلي، نفح العود في سيرة دولة الشريف حمود، تح: العقيلي، ص: 70.— المصدر
2.2معنى من معاني «تَدَحْرَجَ»

وــــــــــ اللَّيْلُ: اشْتَدَّتْ ظَلْماؤُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (421ه=1030م، مِنْ شَواهِدِ المَرْزوقيِّ):

حَتَّى إِذا ما لَيْلُهُ تَدَحْرَجا

وانْجابَ لَوْنُ الْأُفُقِ اليَرَنْدَجا

المرزوقي، الأزمنة والأمكنة، تح: المنصور، ص: 429.— المصدر
2.3معنى من معاني «تَدَحْرَجَ»

وــــــــــ الشَّيْءُ مِنَ الشَّيْءِ: انْحَدَرَ مَنْ أَصْلِهِ وانْتَمى إِلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافعيُّ (200ه=815م، في كِتابِ الأُمِّ):

"ولا يُضَمُّ الدُّخْنُ إلَى الْجُلْبانِ، ولا الْحِمَّصُ إلى الْعَدَسِ ... ويُضَمُّ كُلُّ صِنْفٍ مِنْ هَذا كَبُرَ إلى ما هُوَ أَصْغَرُ مِنْهُ، وكُلُّ صِنْفٍ اسْتَطالَ إلى ما تَدَحْرَجَ مِنْهُ".

الشافعي، الأم، تح: رفعت فوزي، ج: 3، ص: 90.— المصدر
03 —

ادَّحْرَجَ

فعل
3.1معنى من معاني «ادَّحْرَجَ»

* ادَّحْرَجَالشَّيْءُ: انْحَدَرَ مُتَتابِعًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ إِحْدى بَناتِ قَيْلَةَ بِنْتِ مَخْرَمَةَ (ت: 11هـ=632م) تُخاطِبُ أُمَّها:

"اقْلِبي ثِيابَكِ ظُهورَها لِبُطونِها، وادَّحْرِجي ظَهْرَكِ لِبَطْنِكِ، واقْلِبي أَحْلاسَ جَمَلِكِ".

ابن سعد، الطبقات الكبير، تح: علي عمر، ج: 1، ص: 275.— المصدر
04 —

دُحْرُجاء

اسم
4.1معنى من معاني «دُحْرُجاء»

* الدُّحْرُجاءُ:لُعْبَةٌ يَلْعَبُ بِها الصِّبْيانُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=993م):

"والدُّحْرُجاءُ: لُعْبَةٌ يَلْعَبُ بِها الصِّبْيانُ".

ابن دريد، الاشتقاق، تح: هارون، ص: 522.— المصدر
05 —

دَحْرَجَة

اسم
5.1معنى من معاني «دَحْرَجَة»

* الدَّحْرَجَةُ(في الجُيولوجْيا): تَحَرُّكُ الحُطامُ الصَّخْرِيّ على قَعْرِ المَجْرى.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1425هـ=2004م، تاريخُ الطَّبْعِ):

"الدَّحْرَجَةُ: تَحَرُّكُ الحُطامِ الصَّخْرِيِّ على قَعْرِ المَجْرى بِسَبَبِ الاحْتِكاكِ بِأَرْضِيَّةِ المَجْرى وضَرْبِ التَّيّارِ لِقِمَمِ الأَجْسامِ بِسُرْعَةٍ أَكْبَرَ مِنْ سُرْعَةِ ضَرْبِهِ لِقَواعِدِها".

مجمع القاهرة، المعجم الكبير، ج: 7، ص: 122.— المصدر
06 —

دُحْروج

اسم
6.1معنى من معاني «دُحْروج»

* الدُّحْروجُ:ما يُدَوِّرُهُ الجُعَلُ مِنَ فَضَلاتِهِ كَالبُنْدُقَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (421ه=1030م، مِنْ شَواهِدِ المَرْزوقيِّ):

كالجُعُلَيْن رَكِبا دُحْروجا

دَمامَةً ومَنْظَرًا سَميجا

المرزوقي، شرح ديوان الحماسة، تح: غريد الشيخ، ص: 1030.— المصدر
6.2معنى من معاني «دُحْروج»

0 دُحْرُوجُ الوَليدِ:لُعْبَةٌ يَتَقاذَفُها الصِّبْيانُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الأَسْعَرُ الجَعْفِيُّ (ت: 94ق هـ=529م) يَصِفُ خَيْلًا في المَعْرَكَةِ:

ظَلَّتْ سَنابِكُها على جُثْمانِهِ
يَلْعَبْنَ دُحْروجَ الوَليدِ وقَدْ قَضَى

الأصمعيات، تح: شاكر وآخر، ص: 143.— المصدر
07 —

دُحْروجَة

اسم
7.1معنى من معاني «دُحْروجَة»

* الدُّحْروجَةُ:ما تَحَرَّكَ فانْدَفَعَ مُنْحَدِرًا مِنَ الطّينِ وغَيْرِهِ. (ج) دَحاريجُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النابِغَةُ الجَعْديُّ (ت: 50هـ=670م) يُشَبِّهُ صِبْيَةً تَجَمَّعوا على ناقَتِهِ بِالدَّحارِيجِ:

أَضْحَتْ يُنَفِّرُها الوِلْدَانُ مِنْ سَبَإٍ
كَأَنَّهُمْ تَحْتَ دَفَّيْها دَحاريجُ

ديوان النّابغة الجعديّ، تح: واضح الصّمد، ص: 50.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170ه=787م):

"الدُّحْروجَةُ: كُلُّ ما دَحْرَجْتَهُ منْ طِينٍ أَوْ غَيْرِهِ، مِثْل البُنْدُقَةِ المُدَوَّرَةِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 3، ص: 328.— المصدر
7.2معنى من معاني «دُحْروجَة»

وــــــــــ: ما يُدَوِّرُهُ الجُعَلُ مِنْ فَضَلاتِهِ كَالبُنْدُقَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العُجَيْرُ السَّلولِيُّ(ت: 90هـ=709م) يَصِفُ جَمَلًا شَديدًا:

سَمينُ المَطايا يَشْرَبُ السُّؤْرَ والحِسا
قِمَطْرٌ كَحَوّازِ الدَّحاريجِ أبْتَرُ

الزبيدي، لحن العوام، تح: رمضان عبد التواب، ص: 307.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نَصْر ٍالجَوْهَرِيُّ (ت: 393هـ=1003م) :

"الدُّحْروجَةُ: ما يُدَحْرِجُهُ الجُعَلُ مِنَ البَنادِقِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح في اللغة، تح: عطار، ج: 1، ص: 313.— المصدر
08 —

دُحْرَيْج

اسم
8.1معنى من معاني «دُحْرَيْج»

* الدُّحْرَيْجُ:نَباتٌ ضَعيفُ السّاقِ مِنَ الفَصيلَةِ القَرْنِيَّةِ، واحِدُهُ بتاءٍ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ شَفيقٌ غربال وآخَرونَ (1384هـ=1964م، تاريخُ الطَّبْعِ):

"دُحْرَيْجٌ: نَباتٌ مِنَ الفَصيلَةِ القَرْنِيَّةِ، ضَعيفُ السّاقِ".

محمد شفيق غربال وآخرون، الموسوعة العربية الميسرة، ج: 1، ص: 1491.— المصدر
8.2معنى من معاني «دُحْرَيْج»

وــــــــــ حَبٌّ كُرِّيٌّ صَغيرٌ قاتِمٌ، وهُوَ مِنْ غَلَثِ القَمْحِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1425هـ=2004م، تاريخُ الطبْعِ):

"الدُّحْرَيْجُ: حَبٌّ كُرِّيٌّ صَغيرٌ قاتِمٌ، وهُوَ مِنْ غَلَثِ القَمْحِ ".

مجمع القاهرة، المعجم الكبير، ج: 7، ص: 122.— المصدر
09 —

دُحَيْرِيجاءُ

اسم
9.1معنى من معاني «دُحَيْرِيجاءُ»

* الدُّحَيْرِيجاءُ: لُعْبَةٌ يَلْعَبُ بِها الصِّبْيانُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (321هـ=933م، مِنْ شَواهِدِ ابْنِ دُرَيْدٍ):

عَلَيْكَ الدُّحَيْرِيجاءَ فاتْبَعْ صِحابَها
سَيَكْفيكَ زَبْنَ الحَرْبِ أَرْوَعُ ماجِدُ

ابن دريد، الاشتقاق، تح: هارون، ص: 522.— المصدر
10 —

مُدَحْرَج

اسم
10.1معنى من معاني «مُدَحْرَج»

* المُدَحْرَجُ:المُدَوَّرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الشَّنْفَرى (ت: 70ق هـ=554م) يَتَحَدَّثُ عَنْ شَجاعَةِ فارِسٍ:

ومُسْتَبْسِلٍ ضافي القَميصِ ضَمَمْتُهُ
بِأَزْرَقَ لا نِكْسٍ ولا مُتَعَوِّجِ

عَلَيْهِ نُسارِيٌّ على خُوطِ نَبْعَةٍ
وَفُوقٍ كَعُرْقوبِ القَطاةِ مُدَحْرَجِ

ديوان الشّنفرى، تح: إميل يعقوب، ص: 40.— المصدر
11 —

مُدَحْرِج

اسم
11.1معنى من معاني «مُدَحْرِج»

* المُدَحْرِجُ:الجُعَلُ، وهُوَ حَشَرَةُ تُشْبِهُ الخُنْفُساءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابيِّ (ت: 231هـ=846م):

"يُقالُ لِلْجُعَلِ: المُدَحْرِجُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

مرتضى الزبيدي، تاج العروس، تح: حجازي، ج: 5، ص: 553.— المصدر
11.2معنى من معاني «مُدَحْرِج»

0 مُدَحْرِجُ البَعَرِ:مِنْ أَسْماءِ أَيّامِ الشِّتاءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَريُّ (ت: 370هـ=981م):

"ومِنْ أَيّامِ الشِّتاءِ: هالِبُ الشَّعَرِ، ومُدَحْرِجُ البَعَرِ".

الأزهري، تهذيب اللغة، تح: خفاجي وآخر، ج: 6، ص: 305.— المصدر
