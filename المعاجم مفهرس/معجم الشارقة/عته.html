  1. الحُمْقُ ونَقْصُ العَقْلِ
01 —

عَتِهَ

فعل
1.1معنى من معاني «عَتِهَ»

عَتِهَ (بِكَسْرِ التّاءِ) فُلانٌ يَعْتَهُ (بفَتْحِ التّاءِ) عَتَهًا وعَتاهًا: نَقَصَ عَقْلُهُ أو اضْطَرَبَ مِنْ غَيْرِ جُنُونٍ أو دَهَشٍ. فَهُوَ عَتِهٌ وعَتيهٌ، وهِيَ بتاءٍ. (ج) عُتَهاءُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتْ حَليمَةُ السَّعْدِيَّةُ (50ق.هـ=568م) تُخاطِبُ الكاهِنَ الَّذي أشارَ بقَتْلِ النَّبيِّ ﷺ حينَ كانَ مُسْتَرْضَعًا عِنْدَها:

"أَنْتَ أَعْتَهُ وأَجَنُّ مِنِ ابْني هذا، ولَوْ عَلِمْتُ أَنَّ هذا يَكونُ مِنْ قَوْلِكَ ما أَتَيْتُكَ بِهِ، فَاطْلُبْ لِنَفْسِكَ مَنْ يَقْتُلُكَ، فَإِنّا غَيْرُ قاتِلي هذا الغُلامَ".

الآجري، الشريعة. تح: الدميجي، ج: 3، ص: 360.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعِيُّ (ت: 204هـ=819م):

"وإذَا ارْتَدَّ الرَّجُلُ عَنِ الإسْلامِ ولَحِقَ بِدارِ الحَرْبِ أَوْ هَرَبَ فلَمْ يَدْرِ أَيْنَ هُوَ أَوْ خَرِسَ أَوْ ‌عَتِهَ أَوْقَفْنا مالَهُ فَلَمْ نَقْضِ فيهِ بِشَيْءٍ".

الشافعي، الأم. دار الفكر، ج: 4، ص: 310.— المصدر
1.2معنى من معاني «عَتِهَ»

وــــــــــ: ضَلَّ وحَمُقَ؛ فَهو عَتاهِيَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأخْفَشُ الأَوْسَطُ (ت: 215هـ=830م):

"عَتِهَ، كَفَرِحَ، عَتَهًا؛ فَهو عَتاهِيةٌ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

مرتضى الزبيدي، تاج العروس. تح: العزباوي، ج: 36، ص: 435.— المصدر
02 —

عُتِهَ

فعل
2.1معنى من معاني «عُتِهَ»

* عُتِهَ فُلانٌ عَتْهًا وعُتْهًا وعُتاهًا وعَتاهَةً ‌وعَتاهِيَةً: نَقَصَ عَقْلُهُ أَوْ فُقِدَ أَوْ دُهِشَ مِنْ غَيْرِ مَسٍّ وجُنونٍ. فهُوَ مَعْتوهٌ وعَتيهٌ، وهِيَ بِتاءٍ. (ج) عُتُهٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) في أَصْحابِ الحُجَّةِ والعُذْرِ:

«أَرْبَعَةٌ كُلُّهُمْ يُدْلي عَلى اللهِ بِحُجَّةٍ وَعُذْرٍ: رَجُلٌ ماتَ في الفَتْرَةِ، ورَجُلٌ ماتَ هَرِمًا، ورَجُلٌ مَعْتوهٌ، ورَجُلٌ أَصَمُّ أَبْكَمُ، فَيَقولُ اللهُ لَهُمْ: إنّي أُرْسِلُ إِلَيْكُمْ رَسولًا فَأَطيعوهُ، فيَأْتيهِمْ فيَتَأَجَّجُ لَهُمْ نارًا فيَقولُ: اقْتَحِموها. مَنْ دَخَلَها كانَتْ عَلَيْهِ بَرْدًا وسَلامًا، وَمَنْ لَمْ يَقْتَحِمْها حَقَّتْ عَلَيْهِ كَلِمَةُ العَذابِ».

ابن راهويه، المسند. تح: البلوشي، ج: 1، ص: 445.— المصدر
”وقالَ نافِعٌ مَوْلى ابْنِ عُمَرَ (ت:117هـ=735م):

"إِنَّ المُغيرَةَ بْنَ عَبْدِ الرَّحْمنِ طَلَّقَ امْرَأَتَهُ وَهُوَ مَعْتوهٌ فَأَمَرَ ابْنُ عُمَرَ أَنْ تَعْتَدَّ فَقيلَ لَهُ: إِنَّهُ مَعْتوهٌ فَقالَ: إِنّي لَمْ أَسْمَعِ اللهَ اسْتَثْنى لِمَعْتوهٍ طلاقًا ولا غَيْرِهِ".

ابن أبي شيبة، المصنف. تح: الحوت، ج: 4، ص: 72.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ (الإمامُ) (ت: 150هـ=767م) في رَجُلٍ قُتِلَ، وَلَهُ وَلِيَّانِ ابْنانِ صَغِيرٌ وَكَبِيرٌ ولا وارِثَ لَهُ غَيْرُهُمَا:

"أَقْبَلُ البَيِّنَةَ مِنَ الكَبيرِ وأَقْضِي لَهُ بِالقِصاصِ، ولا أنْتَظِرُ إلى كِبَرِ الصَّغيرِ (...) أَرَأَيْتَ لَوْ كَبِرَ هَذا ‌مَعْتُوهًا أَكُنْتُ أَحْبِسُ هَذا".

أبو يوسف. الخراج. تح: سعد، ص: 175.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ (المُفَسِّرُ) (ت: 671هـ=1272م) يَتَحَدَّثُ عَنْ نُزولِ القُرْآنِ على النَّبِيِّ ﷺ:

"فإنْ أنْكَرَ مُنْكِرٌ أَنْ يَكونَ جاءَ بالقُرْآنِ فقَدْ تَبَيَّنَ عِنادُهُ وسَقَطَ اسْتِرْشادُهُ ... ولَنْ يُنازِعَ في ذَلِكَ عَاقِلٌ مُنْصِفٌ، بلْ إمَّا ‌مَعْتوهٌ أو مُتَعَسِّفٌ".

القرطبي، الإعلام. تح: السَّقّا، ص: 323.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُصْطَفى السُّيوطِيُّ (ت: 1243هـ=1827م):

"وإنْ كانَ الْمَوْلَى حَيًّا، وهُوَ رَجُلٌ عاقِلٌ موسِرٌ؛ فعَلَيْهِ مِنَ العَقْلِ، ولَهُ مِنَ الميراثِ؛ لِأَنَّهُ عَصَبَةُ مُعْتِقِهِ، وإنْ كانَ صَبِيًّا أَوْ امْرَأَةً أَوْ ‌مَعْتوهًا فالعَقْلُ على عَصَباتِهِ، والميراثُ لَهُ؛ لأنَّهُ لَيْسَ مِنْ أهْلِ العَقْلِ".

الرحيباني، مطالب أولي النّهى. ج: 4، ص: 684.— المصدر
2.2معنى من معاني «عُتِهَ»

وــــــــــ بشَيْءٍ أو أمْرٍ، وفيهِ: أُوِلعَ بِهِ وحَرَصَ عَلَيْهِ. فهُوَ عاتِهٌ وعَتيهٌ، وهِيَ بِتاءٍ. (ج) عُتَهاءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وعُتِهَ بِهِ: أولِعَ بِهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 1، ص: 104.— المصدر
2.3معنى من معاني «عُتِهَ»

وــــــــــ في فُلانٍ، وعَلَيْهِ: حاكاهُ في كَلامِهِ وحَرَكاتِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وكُلُّ مَن حَاكَى غَيْرَهُ فيما قد عُتِهَ فهو عَتيهٌ بمَعْنى مَعْتوهٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 1، ص: 104.— المصدر
03 —

تَعَتَّهَ

فعل
3.1معنى من معاني «تَعَتَّهَ»

* تَعَتَّهَ فُلانٌ: خَفَّ عَقْلُهُ أو دُهِشَ مِنْ غَيْرِ مَسٍّ أو جُنونٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ بنُ العَجّاجِ (ت: 145هـ=762م) يَصِفُ نَفْسَهُ:

للهِ دَرُّ الغانِياتِ المُدَّهِ

سَبَّحْنَ واسْتَرْجَعْنَ مِنْ تَألُّهي

أَنْ كادَ أَخْلاقي مِنَ التَّنَزُّهِ

يُقْصِرْنَ عَنْ زَهْوِ الشَّبابِ المُزْدَهي

بَعْدَ لَجاجٍ لا يَكادَ يَنْتَهي

عَنِ التَّصابي وَعَنِ التَّعَتُّهِ

[التَّألُّهُ: التَّنَسُّك. الــلَّجاجُ: التَّمادي في العِناد]
ديوان رؤبة بن العجاج. تح: البروسي، ص: 165.— المصدر
3.2معنى من معاني «تَعَتَّهَ»

وــــــــــ: التَّحَيُّرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”وبِهِ فُسِّرَ قَوْلُ رُؤْبَةَ (ت: 145هـ=762م) يَصِفُ نَفْسَهُ:

للهِ دَرُّ الغانِياتِ المُدَّهِ

سَبَّحْنَ واسْتَرْجَعْنَ مِنْ تَألُّهي

أَنْ كادَ أَخْلاقي مِنَ التَّنَزُّهِ

يُقْصِرْنَ عَنْ زَهْوِ الشَّبابِ المُزْدَهي

بَعْدَ لَجاجٍ لا يَكادَ يَنْتَهي

عَنِ التَّصابي وَعَنِ التَّعَتُّهِ

[قالَ نَشْوانُ الحِمْيَرِيُّ في شمس العلوم (7/4365): التَّعَتُّهُ: التَّحَيُّرُ]
ديوان رؤبة بن العجاج. تح: البروسي، ص: 165.— المصدر
3.3معنى من معاني «تَعَتَّهَ»

وــــــــــ: تنظَّفَ ونَظَّفَ ثِيابَهُ وتأنَّقَ فيها وتَزَيَّنَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والتْعَتُّهُ حُسْنُ اللِّبْسَةِ والتَّنَظُّفُ فيها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 402.— المصدر
”وقالَ مَيْمونُ بْنُ هارونَ الكاتِبُ (ت: 277هـ=890م) يَحْكي عَنْ بَعْضِ مَشايِخِهِ سَبَبَ كُنْيَةِ الشّاعِرِ أَبي العَتاهِيَةِ:

"‏كُنِّيَ بأَبي العَتاهِيَةِ، أَنْ كانَ يُحِبُّ الشُّهْرَةَ والمُجونَ والتَّعَتُّهَ".

الأصبهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 4، ص: 6.— المصدر
3.4معنى من معاني «تَعَتَّهَ»

وــــــــــ: تَوَلَّهَ بالنِّساءِ وسُلِبَ بِهِنَّ عَقْلُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المَهْدِيُّ العَبّاسِيُّ (ت: 169هـ=786م) يُعاتِبُ الشّاعِرَ أبا العَتاهِيَةِ حينَ وَجَدَهُ مُتَوَلِّهًا بجارِيَةٍ:

"أَراكَ مُتَخلِّطًا مُتَعتِّهًا!".

ابن منظور، لسان العرب. تح: دار صادر، ج: 13، ص: 513.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م) يَتَحَدَّثُ عَنِ الشّاعِرِ أبي العَتاهِيَةِ:

"واسْمُهُ إسْماعيلُ بْنُ القاسِمِ، ولُقِّبَ بِذَلِكَ لأَن المَهْدِيَّ قالَ لَهُ: أَراكَ مُتَخلِّطًا مُتَعتِّهًا! وَكَانَ قَدْ ‌تعَتَّهَ ‌بِجارِيَةٍ لِلْمَهْدِيِّ واعتُقِلَ بِسَبَبِها".

ابن منظور، لسان العرب. تح: دار صادر، ج: 13، ص: 513.— المصدر
3.5معنى من معاني «تَعَتَّهَ»

وــــــــــ: ذَهَبَ في الباطِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"والتَّعَكُّظُ والتَّعَتُّهُ: الذَّهابُ في الباطِلِ، وَكذلِكَ التَّمَتُّهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: محمد العمري، ج: 1، ص: 348.— المصدر
3.6معنى من معاني «تَعَتَّهَ»

وــــــــــ في المأْكَلِ أَوِ المَلْبَسِ وغَيْرِهِما: أَسْرَفَ فيهِ وبالَغَ، وزادَ على قَدْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وتَعَتَّهَ في كَذا: أَسْرَفَ فيهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 1، ص: 104.— المصدر
3.7معنى من معاني «تَعَتَّهَ»

وــــــــــ في أمْرٍ أو عَمَلٍ: تَنَوَّقَ وبالغَ في تَجْويدِهِ وإتْقانِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ العُذْرِيُّ (195هـ=811م، مِنْ رُواةِ الشَّيْبانِيِّ):

"تَعَتَّهَ فُلانٌ أو فُلانَةُ في صَنْعَتِهِ: إذا تَنَوَّقَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: الأبياري، ج: 2، ص: 277.— المصدر
3.8معنى من معاني «تَعَتَّهَ»

وــــــــــ لِفُلانٍ عَنِ الأَمْرِ: تغافَلَ عَنْهُ فيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"وفُلانٌ يَتَعَتَّهْ لَكَ عَنْ كَثيرٍ مِمّا تَأْتيهِ؛ أي: يَتَغافَلُ عَنْكَ فيهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 1، ص: 139.— المصدر
04 —

عُتاه

اسم
4.1معنى من معاني «عُتاه»

* العُتاهُ: ضَعْفٌ ونَقْصٌ يُصيبُ العَقْلَ، يؤَثِّرُ في تَفْكيرِ الشَّخْصِ وسُلوكِهِ، فيَصيرُ مُخْتَلِطَ العَقْلِ، فَيُشْبِهُ بَعْضُ كَلامِهِ كَلامَ العُقَلاءِ وَبَعْضُهُ كَلامَ المَجانينِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"عُتِهَ الرَّجُلُ فَهُوَ مَعْتوهٌ، والاسْمُ العُتاهُ، وهُوَ اخْتِلاطُ العَقْلِ، شَبيهٌ بِالبَلَهِ  ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 403.— المصدر
4.2معنى من معاني «عُتاه»

o والعُتاهُ الباكِرُ (أو المُبَكِّرُ): الفِصامُ، وهُوَ ضَعْفٌ عَقْليٌّ يُصيبُ المُراهِقينَ، يؤَثِّرُ في تَفْكيرِ الشَّخْصِ وسُلوكِهِ فيَجْعَلُهُ مُنْفَصِلًا عَنِ الواقِعِ، وقَدْ يَحْدُثُ لِمَنْ هُوَ في مَراحِلَ عُمْرِيَّةٍ مُتَأَخِّرَةٍ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ المُنْعِمِ الحِفْنيُّ (ت: 1402هـ=1992م):

"العُتاهُ الباكِرُ هُوَ الاسْمُ القَديمُ للفِصامِ، وكانَ المَظْنونُ أنَّ هذا النَّوْعَ مِنَ العَتَهِ لا يُصيبُ إلا المُراهِقينَ".

الحفني، موسوعة الطب النفسي. ص: 840.— المصدر
4.3معنى من معاني «عُتاه»

o والعُتاهُ الشَّلَليُّ: مَرَضٌ زُهْريٌّ في المُخِّ مَصْحوبٌ بِارْتِعاشٍ واضْطِرابٍ في النُّطْقِ وضَعْفٍ عَقْليٍّ مُتزايِدٍ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ ( 1374هـ=1955م، تاريخُ النَّشْرِ):

"العُتاهُ الشَّلَليُّ مَرَضٌ في المُخِّ زُهْرِيٌّ مَصْحوبٌ بِارْتِعاشٍ واضْطِرابٍ في النُّطْقِ وَضَعْفٍ عَقْليٍّ مُتَزايِدٍ".

مجلّة مجمع اللغة العربيّة بالقاهرة. العدد: 8، ص: 514.— المصدر

عَتاهة

اسم
5.1معنى من معاني «عَتاهة»

* العَتاهَةُ: ضُلّالُ النَّاسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

" والعَتاهَةُ والعَتاهِيَّةُ: ضُلّالُ النّاسِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 1، ص: 267.— المصدر
5.2معنى من معاني «عَتاهة»

وــــــــــ: ضَعْفٌ ونَقْصٌ يُصيبُ العَقْلَ، يؤَثِّرُ في تَفْكيرِ الشَّخْصِ وسُلوكِهِ، فيَصيرُ مُخْتَلِطَ العَقْلِ، فَيُشْبِهُ بَعْضُ كَلامِهِ كَلامَ العُقَلاءِ وَبَعْضُهُ كَلامَ المَجانينِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الحكيمُ التِّرْمِذِيُّ (ت: 320هـ=932م) يَذْكُرُ حالَ العَروسِ تَرَكَتْ تَقْليمَ أظْفارِها وتَنْظيفَ نَفْسِها:

"فإنْ لَمْ تَفْعَلْ ذَلِكَ وتَرَكَتْ هَذِهِ الأَظْفارَ والدَّرَنَ والأَوْساخَ على جَسَدِها، وحُلِّيَتْ بالحُلِيِّ وزُيِّنَتْ بالثِّيابِ كانَ ذَلِكَ كاللَّعِبِ، ويُنْسَبُ ذَلِكَ إلى فِعْلِ الجُنونِ ‌والعَتاهَةِ".

الحكيم الترمذي، الأمثال. تح: الجميلي، ص: 93.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ وِداد سَكاكيني،(1357هـ=1939م، تاريخُ النَّشْرِ) في أحْداثِ رِوايَةٍ عَنْ رَجُلٍ مُتَزَوِّجٍ مِنِ اثْنَتَيْنِ:

"ولَمّا جَلَسَ طاهِرٌ إلى الطَّعامِ ذَرَّتْ لَهُ الَّتي هُوَ عِنْدَها ذَرورَ السّاحِرِ، فلَمّا أَكَلَ انْقَلَبَ كالَّذي خامَرَهُ امْتعاضٌ، وظَلَّ أيّامًا كُلَّما أكَلَ عِنْدَ واحِدَةٍ مِنْهُما ذَرَّتْ لَهُ قَليلًا مِنَ الذَّرورِ، فلَمْ يَمْضِ شَهْرانِ حَتَّى غَدا كالمَخْبولِ المَمْرورِ لا يَدْري ما يَفْعَلُ. فقالَ النّاسُ: قَدْ أصابَتْهُ ‌عَتاهَةٌ وبَلاهَةٌ!".

مجلة الرسالة. ‌‌العدد: 290، 1939م، ص: 187.— المصدر

اسم
6.1معنى من معاني «عُتاهِيّ»

* العُتاهِيُّ: ناقِصُ العَقْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو المُشْمَعِلِّ(ت:184هـ=800م) يَرْثي مَوْلاهُ عَبْدَ اللهِ بْنَ مُصْعَبٍ الزُّبَيْرِيَّ:

فأنَّى كعَبْدِ اللهِ يُرْجَى لكُرْبَةٍ
وأنَّى كعَبْدِ اللهِ للضَّيْمِ مَدْفَعَا

يُنيلُكَ ما لا يُدْرِكُ النّاسُ بَذْلَهُ
هَنيئًا وللعاتي ‌العُتَاهِيّ مِرْدَعا

الزبير بن بكار، جمهرة نسب قريش. تح: شاكر، ص: 155.— المصدر

7.4معنى من معاني «عَتاهِيَة»

وــــــــــ: الدَّهاءُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أسْعَدُ بْنُ إبْراهيمَ النَّشَّابِيُّ (ت: 657هـ=1258م):

"وتَقولُ العَرَبُ: ‏رَجُلٌ عَتاهِيَةٌ، بِغَيْرِ أَلِفٍ ولامٍ. ومَعْنى عَتاهِيَةٍ مِنَ الدَّهاءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

7.5معنى من معاني «عَتاهِيَة»

o وأَبو العَتاهِيَةِ: لَقَبٌ لواحِدٍ مِنْ أشْهَرِ شُعَراءِ العَصْرِ العَبّاسِيِّ، في آخرِ القَرْنِ الثّاني الهِجْرِيِّ، اسْمُهُ إسْماعيلُ بْنُ القاسِمِ، لُقِّبَ بذلِكَ لأنَّهُ ‌تعَتَّهَ ‌بِجارِيَةٍ لِلْخَليفَةِ المَهْدِيِّ واعتُقِلَ بِسَبَبِهَا، أو لأنَّهُ كانَ يَميلُ إلى المُجونِ والحُمْقِ والضَّلالِ في أوَّلِ حَياتِهِ، ثُمَّ تَرَكَ ذَلِكَ وتَنَسَّكَ وصارَ أكْثَرُ شِعْرِهِ في الزُّهْدِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بْنُ بُرْدٍ (ت: 167هـ=784م) يُخاطِبُ أبا العَتاهِيةِ:

"يا أبا العَتاهِيَةِ واللهِ إنِّي لأسْتَحْسِنُ ‌اعْتِذارَكَ ‌في ‌البُكاءِ إذْ تَقولُ:

كَمْ مِنْ صَديقٍ لي أُسا
رِقُهُ البُكاءَ مِنَ الحياءِ".

البكري، سمط اللآلي. تح: الميمني، ج: 1، ص: 197.— المصدر

09 —

عَتَه

اسم
9.1معنى من معاني «عَتَه»

* العَتَهُ: آفَةٌ تُوجِبُ خَلَلًا في العَقْلِ، وتُؤَثِّرُ في تَفْكيرِ الشَّخْصِ وسُلوكِهِ، فيَصيرُ مُخْتَلِطَ العَقْلِ، فَيُشْبِهُ بَعْضُ كَلامِهِ كَلامَ العُقَلاءِ وَبَعْضُهُ كَلامَ المَجانينِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّافِعيُّ (ت: 204هـ=819م) في الغَلَبَةِ على العَقْلٍ في غَيرِ المَعْصِيَةِ:

"وإِذا غُلِبَ الرَّجُلُ عَلى عَقْلِهِ بِعارِضِ جِنٍّ أَوْ عَتَهٍ أَوْ مَرَضٍ ما كانَ المَرَضُ؛ ارْتَفَعَ عَنْهُ فَرْضُ الصَّلاةِ، ما كانَ المَرَضُ بِذَهابِ العَقْلِ عَلَيْهِ قائِمًا؛ لأَنَّهُ مَنْهيٌّ عَنِ الصَّلاةِ حَتّى يَعْقِلَ".

الشافعي، الأم. تح: النجار، ج: 1، ص: 70.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)

"البِرْسامُ ‌والعَتَهُ نَوْعانِ مِنِ اخْتِلالِ العَقْلِ والجُنونِ".

[البِرْسامُ: وَرَمٌ يُصيبُ الدِّماغَ أو الصَّدْرَ، فإذا كانَ في الدِّماغِ أحْدَثَ في العَقْلِ اخْتلالًا]
النووي، تحرير ألفاظ التنبيه. تح: الدقر، ص: 241.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عابِدينَ (ت: 1252هـ=1836م):

"اعْلَمْ أَنَّ اللهَ تبارَكَ وتَعالى جَعَلَ بَعْضَ البَشَرِ ذَوي النُّهَى، وجَعَلَ مِنْهُمْ أَعْلامَ الدّينِ وأَئِمَّةَ الْهُدَى ومَصابيحَ الدُّجَى، وابْتَلَى بَعْضَهُمْ بِما شاءَ مِنْ أَسْبابِ الرَّدَى، كالجُنونِ الموجِبِ لِعَدَمِ العَقْلِ، والصِّغَرِ ‌والعَتَهِ المُوجِبَيْنِ لِنُقْصانِهِ، فجَعَلَ تَصَرُّفَهُما غَيْرَ نافِذٍ بِالحَجْرِ عَلَيْهِما".

o والعَتَهُ المُبْكِرُ: الفِصامُ، وهُوَ ضَعْفٌ عَقْليٌّ يُصيبُ المُراهِقينَ، يؤَثِّرُ في تَفْكيرِ الشَّخْصِ وسُلوكِهِ فيَجْعَلُهُ مُنْفَصِلًا عَنِ الواقِعِ، وقَدْ يَحْدُثُ لِمَنْ هُوَ في مَراحِلَ عُمْرِيَّةٍ مُتَأَخِّرَةٍ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بِنْيامين بَهْمَن (1357هـ=1939م، تاريخ النَّشْرِ) يَتَحَدَّثُ عَنْ أَحَدِ الأَمْراضِ العَصَبِيَّةِ:

"ذلِكَ النَّوْعُ الَّذي يَزْدادُ انْتِشارًا عِنْدَنا في السَّنَواتِ الأَخيرَةِ، وهُوَ العَتَهُ المُبْكِرُ أَوْ جُنونُ الشَّبابِ، يَكونُ المَريضُ بِهِ مُنْذُ طُفولَتِهِ هادِئًا مُؤَدَّبًا، مُطيعًا لِوالِدَيْهِ خَجولًا، قَليلَ الاخْتِلاطِ بِالغَيْرِ، مَيَّالًا لِلْعُزْلَةِ".

مجلة الثّقافة. العدد: 2، 1939م، مج: 1، ص: 40.— المصدر
