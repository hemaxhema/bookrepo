المقدّمة

النَّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ش ل ح):

في الآراميّة: وَرَدَ الفِعْلُ: שְׁלַח (شْلَح) šəlaḥ بِمَعْنى (خَلَعَ، تَجرَّدَ) يوافِقُ الفِعْلَ العَرَبيَّ (شَلَحَ) مَبنًى ومَعنًى، ومِنْ مَجيئِهِ بِمَعْنى (شَلَحَ) ما وَرَدَ في تَرجومِ يوناثانَ: "שְלַחוּ וְאִתעַרטַלוּ‏"، أي: "اشلَحنَ مَلابِسَكُنَّ وتَعرَّينَ".

(ת''י יש' לב, יא)

في العِبْريّة: وَرَدَ الفِعْلُ: שִׁלַּח (شِلّح) šillaḥ بِمَعْنى (طَرَدَ، خَلَعَ، نَحّى، أزاحَ، أبعَدَ) يوافِقُ الفِعْلَ العَرَبيَّ (شَلَحَ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (شَلَحَ) ما وَرَدَ في سِفرِ التَّكوينِ: "וַיְשַׁלְּחֵהוּ יְהוָה אֱלֹהִים, מִגַּן-עֵדֶן"، أي:"فَشَلَحَهُ الرَّبُّ الإلهُ مِنْ جَنّةِ عَدْنٍ ليَعْمَلَ الأرْضَ".

(בר' 3: 22)

في السُّرْيانيّة: وَرَدَ الفِعْلُ: ܫܠܰܚ (شْلَح) šəlaḥ بِمَعنى (سَلَخَ، تَجرَّدَ، خَلَعَ) يوافِقُ الفِعْلَ العَرَبيَّ (شَلَحَ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (شَلَحَ) ما وَرَدَ في سِفرِ اللّاويينَ: "ܘܢܶܫܠܰܚ ܠܒܽܘ̈ܫܶܐ ܕܒܽܘܨܳܐ ܕܠܳܒܶܫ ܡܳܐ ܕܥܳܐܶܠ ܠܩܽܘܕ݂ܫܳܐ"، أي: "ويَشلَحُ ثيابَ الكَتّانِ الَّتي لَبِسَها عِنْدَ دُخولِهِ إلى القُدْسِ".

(ܟܳܗ̈ܢܶܐ- ܝܘ: 23)

01 —

شَلَحَ

فعل
1.1معنى من معاني «شَلَحَ»

* شَلَحَ (بفَتْحِ اللّامِ) فُلانٌ الثَّوْبَ يَشْلَحُهُ (بِفَتْحِ اللّامِ) شَلْحًا: خَلَعَهُ ونَزَعَهُ. (مح)

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ لُويس شيخو (ت: 1346هـ=1927م) يَحْكي عَلى لِسانِ رَجُلٍ لوحِقَ فاخْتَبَأَ عِنْدَ آخَرَ:

"وأَشارَ لي إِلى بابٍ فَدَخَلْتُهُ، ومَضى مُسْرِعًا وأقْفَلَ البابَ، ودَخَلَ حَرَمَهُ وأَتاني مِنْ ثِيابِهِنَّ وقالَ لي: قُمِ اشْلَحْ ما عَلَيْكَ والْبَسْ هذِهِ الثِّيابَ؛ لأنّي رأَيْتُ الطَّلَبَ عَلَيْكَ شَديدًا".

لويس شيخو، مجاني الأدب، ج: 3، ص: 221.— المصدر
02 —

شَلَّحَ

فعل
2.1معنى من معاني «شَلَّحَ»

* شَلَّحَ فُلانٌ فُلانًا: نَهَبَهُ وسَلَبَ ما مَعَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (ت: 40هـ=661م) يَصِفُ الشُّراةَ:

"خَرَجوا لُصوصًا مُشَلِّحِينَ".

ابن الأثير، النهاية، تح: الطناحي وآخر، ج: 2، ص: 498.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو المَكارِمِ الأعْرابِيُّ (ت: 210هـ=826م):

"قَدْ جَرَّدَهُ: إذا شَلَّحَهُ".

الهروي، كتاب الغريبين، تح: الطناحي، ج: 1، ص: 342.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَلَمُ الدّينِ البِرْزالِيُّ (ت: 739هـ=1338م) يَذْكُرُ حالَ النّاسِ عِنْدَ غَزْوِ التَّتارِ:

"وخافَ النّاسُ مِنَ الخُروجِ، ولَزِموا البُيوتَ، وقلَّ النّاسُ في البَلَدِ ... وقَلَّ النّاسُ بالجامِعِ، وقَلَّ مَنْ يَحْضُرُ الصَّلَواتِ؛ خَوْفًا مِنَ ‌التَّشْليحِ".

البرزالِي، المقتفي ، تح: تدمري، ج: 3، ص: 42.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجَبَرتِيُّ (ت: 1237هـ=1822م) يَذْكُرُ قَوْمًا:

"... وناموا، فَلَمّا اسْتَيْقَظوا فلَمْ يَجِدوا ثِيابَهُمْ وسَطا عَلَيْها السُّرّاقُ ‌فشَلَّحوهُمْ ...".

الجبرتي، تاريخ عجائب الآثار، دار الجيل، ج: 3، ص: 209.— المصدر
2.2معنى من معاني «شَلَّحَ»

وــــــــــ فُلانًا الثَّوْبَ: نَزَعَهُ عَنْهُ وعَرّاهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ الفَضْلِ الجَرْجَرائِيُّ (ت: 251هــ =867م) يَحْكي ما فَعَلَهُ المُعْتَصِمُ بِهِ:

"فَلَمّا نَظَرَ إلَيَّ شَتَمَني، وقالَ: أخْرَبْتَ الضِّياعَ، ونَهَبْتَ الارْتِفاعَ، وقالَ: واللهِ لأقْتُلَنَّكَ، هاتوا السِّياطَ، فَأُحْضِرَتْ، وشُلِّحْتُ للضَّرْبِ".

التنوخي، الفرج بعد الشدة، تح: الشالجي، ج: 2، ص: 27.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ القَيِّمِ (ت: 751هـ=1350م) يُمَثِّلُ لبَعْضِ الدَّعاوى الَّتي يَنْفيها العُرْفُ:

"أَوْ يَدَّعي رَجُلٌ مَعْروفٌ بالشِّحَاذَةِ وسُؤالِ النّاسِ: أَنَّهُ أَقْرَضَ تاجِرًا مِنْ أَكْبَرِ التُّجّارِ مِئَةَ أَلْفِ دِينارٍ، أَوْ أَنَّهُ غَصَبَها مِنْهُ، أَوْ أَنَّ ثِيابَ التّاجِرِ الَّتي هِيَ عَلَيْهِ مِلْكُ ‌الشَّحّاذِ ‌شَلَّحَهُ ‌إِيّاها، أَوْ غَصَبَها مِنْهُ".

ابن القيم، الطرق الحكمية، تح: نايف الحمد، ج: 1، ص: 245.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجَبَرتِيُّ (ت: 1237هـ=1822م) يَذْكُرُ أحْداثَ عامِ 1218هـ:

"وفيهِ أخْرَجوا الإنْكِشارِيَّةَ الَّذين بقَلْعَةِ الظّاهِرِ، وسَفَّروهُمْ إلى جِهَةِ الصّالِحِيَّةِ ... بَعْدَ ما أخَذوا سِلاحَهُمْ ومَتاعَهُمْ، بَلْ ‌وشَلَّحوهُمْ ثِيابَهُمْ".

الجبرتي، تاريخ عجائب الآثار، دار الجيل، ج: 2، ص: 582.— المصدر
03 —

شُلِّحَ

فعل
3.1معنى من معاني «شُلِّحَ»

* شُلِّحَ فُلانٌ: خَرَجَ عَلَيْهِ قُطَّاعُ الطَّريقِ فسَلَبُوه ثِيَابَه وعَرَّوْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الأزْهَرِيُّ (ت: 370هـ=981م) يَحْكي عَنْ أهْلِ السَّوادِ:

"سَمِعْتُهُمْ يَقولونَ: ‌شُلِّحَ فُلانٌ، إذا خَرَجَ عَلَيْهِ قُطّاعُ الطَّريقِ فَسَلَبوهُ ثِيابهُ وعَرَّوْهُ، وأحْسَبُها نَبَطِيَّةً". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: عبد الله درويش، ج: 4، ص: 183.— المصدر
04 —

تَشَلَّحَ

فعل
4.1معنى من معاني «تَشَلَّحَ»

* تَشَلَّحَ فُلانٌ: خَلَعَ بَعْضَ ثِيابِهِ، أَوْ شَمَّرَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مُسْلِمٍ الكَجِّيُّ (ت: 292هـ=905م) يَحْكي على لِسانِ طُلّابٍ ساعَدوا شَيْخَهُمْ في سَقْيِ بُسْتانِهِ:

"فتَشَلَّحْنا وأَدَرْنا الدّولابَ حَتَّى سَقَيْنا البُسْتانَ".

البغدادي، تاريخ بغداد، تح: مصطفى عطا، ج: 6، ص: 120.— المصدر
4.2معنى من معاني «تَشَلَّحَ»

وــــــــــ: تَعَرَّضَ للسَّلْبِ والنَّهْبِ مِنْ قِبَلِ قُطّاعِ الطَّريقِ ونَحْوِهِمْ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شِهابُ الدّينِ الحَلّاقُ (ت: 1175هـ=1762م) يَحْكي خَبَرَ حُجّاجٍ تَعَرَّضَ لَهُمْ قُطّاعُ طُرُقٍ:

"فَناسٌ سَلِموا وناسٌ قُتِلوا، ومِنْهُمْ مَنْ تَشَلَّحَ مِرارًا".

الحلاق، حوادث دمشق اليومية، تح: القاسمي، ص: 208.— المصدر
05 —

شِلْح

اسم
5.1معنى من معاني «شِلْح»

* الشِّلْحُ: اللِّصُّ، (ج): شُلوحٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ حُسامُ الدّينِ الإسْكنْدرانِيُّ (ت: 560هـ=1165م) يَهْجو ابْنَ قَلاقِسَ:

اسْأَلوا عَنِّي فُتوحَ ابْنِ قَلاقِسْ

كَيْفَ رَأَى ضَرْبَ الشُّلوحِ بِالدَّرافِسْ

الأزدي، بدائع البدائه، تح: عطا، ص: 227.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ نَجْمُ الدّينِ الطّوفِيُّ (ت: 716هـ=1316م) يَتَحَدَّثُ عَنِ ادِّعاءِ النّمْرودِ القُدْرَةَ عَلى الإحْياءِ والإماتَةِ:

"والنّمْرودُ إنَّما أثْبَتَ لِنَفْسِهِ مَجازَ ذلِكَ، وهُوَ اسْتِبْقاءُ الحَياةِ في الإحْياءِ، وتَفْويتُها بالعِلاجِ المَحْسوسِ في الإماتَةِ، وذلِكَ مِمّا يَقْدِرُ علَيْهِ اللُّصوصُ ‌والشُّلوحُ وقُطّاعُ الطَّريقِ".

الطوفي، الإشارات الإلهية، تح: محمد حسن، ص: 107.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الطَّيِّبُ بْنُ عَبْدِ السَّلامِ (ت: 1250هـ=1834م) يَمْدَحُ السُّلْطانَ عَبْدَ الرَّحْمنِ بْنَ هِشامٍ:

وجَيَّشَ للشِّقاقِ عَظيمَ جُنْدٍ
يَضيقُ بِها لكَثْرَتِها الفَسيحُ

وحَصَّنَ بِالمَدافِعِ والعَوالِي
حُصونًا في دَواخِلِها شُلوحُ

السجلماسي، إتحاف أعلام الناس، تح: علي عمر، ج: 3، ص: 116.— المصدر

08 —

شُلوح

اسم
8.1معنى من معاني «شُلوح»

* الشُّلوحُ: طَوائِفُ مِنَ البَرْبَرِ، يَتَكَلَّمونَ بِأَلْسِنَةٍ مُخْتَلِفَةٍ، ويَسْكُنونَ أَقْصى بَوادي المَغرِبِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الزَّبيديُّ (ت: 1205هـ=1790م):

"‌والشُّلوحُ: طَوائفُ مِنَ البَرْبَرِ يَتَكَلَّمونَ بأَلْسِنَةٍ مُخْتَلِفَةٍ، ومَساكِنُهُمْ بأَقْصى بَوادي المَغْرِبِ".

الزبيدي، تاج العروس، تح: حسين نصار، ج: 6، ص: 510.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَجيبَةَ (ت: 1224هـ=1809م) يَذْكُرُ ما فَعَلُهُ داوُدُ، عَلَيْهِ السَّلامُ، بقَوْمِ جالوتَ:

"ولَمّا تَمَكَّنَ داوُدُ -عَلَيْهِ السَّلامُ- مِنَ المُلْكِ، أَجْلى مَنْ بَقِيَ مِنْ قَوْمِ جالوتَ إلى المَغرِبِ، فَمِنْ بَقِيَّتِهِمُ البَرابِرَةُ مِنَ الشُّلوحِ وسائِرِ الأَرْيافِ".

ابن عجيبة، البحر المديد، تح: القرشي، ج: 1، ص: 280.— المصدر
09 —

مَشْلَح

اسم
9.1معنى من معاني «مَشْلَح»

* المَشْلَحُ: حُجْرَةٌ في الحَمّامِ تُخْلَعُ فيها الثِّيابُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ هانِئٍ النَّيْسابورِيُّ (ت: 275هـ=889م) يَذْكُرُ سُؤالًا وُجِّهَ إلى الإمامِ أحْمَدَ بْنِ حَنْبَلٍ:

"وسُئِلَ عَنِ الرَّجُلِ يُصَلّي في ‌مَشْلَحِ الحَمّامِ؟ قالَ: مَكْروهُ، لا يُصَلَّى فيهِ".

خالد الرباط وآخر، الجامع لعلوم الإمام أحمد، ج: 5، ص: 598.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ تَيْمِيَّةَ (ت: 728هـ=1327م):

"فأَمّا ‌الـمَشْلَحُ الَّذي تُوضَعُ فيهِ الثِّيابُ وهُوَ بارِدٌ لا يُغْتَسَلُ فيهِ ولا يَقْعُدُ فِيهِ إلّا المُتَلَبِّسُ، فَلَيْسَ هُوَ مَكانَ حَمّامٍ".

ابن تيمية، مجموع الفتاوى، تح: عبد الرحمن بن محمد، ج: 21، ص: 303.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ إبْراهيم خورشيد (ت: 1407هـ=1987م):

"فَإِنَّ الحَمّامَ يَتَكَوَّنُ بصِفَةٍ أَساسِيَّةٍ مِنْ عَدَدٍ بعَيْنِهِ مِنَ الغُرَفِ، كُلٌّ مِنْها لَها وَظيفَةٌ خاصَّةٌ، فَفي البِدايَةِ غُرْفَةٌ لِخَلْعِ المَلابِسِ والرّاحَةِ، تُعْرَفُ عادَةً بِالمُشَلَّحِ أَو الـمَشْلَحِ في الشَّرْقِ".

خورشيد وآخرون، موجز دائرة المعارف الإسلاميّة، ج: 14، ص: 4312.— المصدر
10 —

مِشْلَح

اسم
10.1معنى من معاني «مِشْلَح»

* المِشْلَحُ: ضَرْبٌ مِنَ الثِّيابِ، مَفْتوحٌ مِنَ الأَمامِ كالجُبَّةِ، يَلْبَسُهُ الرِّجالُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ خَلدونَ (ت: 808هـ=1405م) يَتَحَدَّثُ عَنْ قَتْلِ سَيْفِ الدَّولَةِ إيّازَ السُّلْجوقِيَّ:

"وقَدْ أَعَدَّ جَماعَةً مِنْ خَواصِّهِ لقَتْلِ إيازٍ، فلَمّا دَخَلوا ضُرِب إيّازُ فقُطِعَ رَأْسُهُ ولُفَّ شِلْوُهُ في ‌مِشْلَحٍ وأُلْقِيَ على الطَّريقِ".

ابن خلدون، تاريخ ابن خلدون، تح: شحادة، ج: 3، ص: 610.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ كامِلٌ الغَزِّيُّ (ت: 1352هـ=1933م) يَذْكُرُ خِلافًا بَيْنَ أحَدِ الدَّراويشِ وأَحَدِ عَساكِرِ السُّلْطانِ:

"تَنازَعَ أَحَدُ الفُقَراءِ الَّذينَ يَصْنَعونَ الكَراسِيَّ مَعَ واحِدٍ مِنْ عَساكِرِ الدَّراويشِ المَوْلَوِيَّةِ بِسَبَبِ ‌مِشْلَحٍ".

الغزي، نهر الذهب، ج: 3، ص: 254.— المصدر
11 —

مُشَلَّح

اسم
11.1معنى من معاني «مُشَلَّح»

* المُشَلَّحُ: حُجْرَةٌ في الحَمّامِ تُخْلَعُ فيها الثِّيابُ.
