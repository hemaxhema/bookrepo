غذم
غَذَمَ(فعل) غَذِمَ(فعل) أَغْذَمَ(فعل) اغْتَذَمَ(فعل) تَغَذَّمَ(فعل) غَنْذَمَ(فعل) غَذائم(اسم) غُذامَة(اسم) غُذّامَة(اسم) غَذَم(اسم) غَذِم(اسم) غَذْمَة(اسم) غُذْمَة(اسم) غُذَمَة(اسم) غَذَمْذَم(اسم) غَذيمَة(اسم) مِغْذَم(اسم)
النقوش العربيّة

وَرَدَ الجَذْرُ (غ ذ م غ ذ م ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَينِها نَقْشٌ جرافيتيٌّ سبئيُّ اللَّهجةِ غَيْرُ مُؤرَّخٍ (Ja 1012 i)، وَرَدَ فيهِ الاسْمُ (غذمت Ġaḏimat ) عَلَمًا على أحَدِ الأفرادِ، فَجاءَ في النَّقْشِ: "غذمت"، أيْ: "غَذِمةُ".

(Jamme, 1966: 20)

المعانى الكلّيّة

  1. طَريقَةٌ في الأكْلِ أو الشُّرْبِ
  2. الإعْطاءُ
  3. الأَخْذُ
  4. الكَلِمَةُ
  5. الغُبْرَةُ
01 —

غَذَمَ

فعل
1.1معنى من معاني «غَذَمَ»

* غَذَمَ (بِفَتْحِ الذّالِ) فُلانٌ وغَيْرُهُ يَغْذُمُ ويَغْذِمُ (بِضَمِّ الذّالِ وكَسْرِها) غَذْمًا: أَكَلَ بِجَفاءٍ وشِدَّةِ نَهَمٍ. 

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"غَذَمَ غَذْمًا، أَيْ أَكَلَ بِجَفاءٍ وشِدَّةِ نَهَمٍ".

الخليل، العين. تح: هنداوي، ج: 3، ص: 270.— المصدر
”وقالَ الـمَعَرّيُّ (ت: 449هـ=1057م) يَعِظُ مَنْ يَتَشَدَّدُ ويَعَضُّ لِيَأْخُذَ الشَّيْءَ بِكَثْرَةٍ مِنْ مَأْكولٍ وغَيْرِهِ:

"أنْتَ تَعْذِمُ ‌لِتَغْذِمَ، وشَفْرَتُكَ تَهْذِمُ، لِتَكونَ النُّحُرُ تَقْذِمُ، والجِفانُ تَرْذِمُ، فَإنَّ السَّبَبَ مُنْجَذِمٌ".

أبو العلاء المعرّي، الفصول والغايات. ص: 312.— المصدر
1.2معنى من معاني «غَذَمَ»

وــــــــــ لِفُلانٍ مِنْ مالِهَ: أَعْطاهُ مِنْهُ شَيْئًا كَثيرًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ هِنْدُ بِنْتُ عُتْبَة (ت: 14هـ=635م) تُخاطِبُ ابْنَها مُعاوِيَةَ:

"قَدْ قَدِمَ عَلَيْكَ أَبوكَ وأَخَواكَ، فَلا ‌تَغْذِمْ ‌لَهُمْ، فَيَعْزِلَكَ عُمَرُ".

الجاحظ، رسائل الجاحظ. تح: هارون، ج: 2، ص: 347.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعيّ (ت: 216هـ=831م):

"ويُقَالَ: قَثَمَ لَهُ مِنْ مالِهِ وقَذَمَ، ‌وغَذَمَ ‌لَهُ مِنْ مالِهِ وغَثَمَ: إذا دَفَعَ إلَيْهِ دُفْعَةً فَأَكْثَرَ".

أبو علي القالي، الأمالي. ج: 2، ص: 134.— المصدر
1.3معنى من معاني «غَذَمَ»

وــــــــــ الشَّيْءَ: أَكَلَهُ بِشدَّة ونَهَمٍ. 

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ خَلَفٌ الأحْمَر (ت: 180هـ=796م):

"يُقالُ: اغْتَذَمَ الحُوارُ ما في ضَرْعِ أُمِّهِ، وذَلِكَ إِذا اسْتَوْعَبَهُ فَلَمْ يُبْقِ فِيهِ شَيْئًا، وهُوَ مِنَ الأوَّلِ؛ يُقالُ: ‌غَذَمَ واغْتَذَمَ".

أبو عبيد، غريب الحديث. تح: خان، ج: 4، ص: 35.— المصدر
1.4معنى من معاني «غَذَمَ»

وــــــــــ غُذْمَةً: وجَدَ واقِعَةً من عُشْبٍ أو بَقْلٍ مُنْكَرَةً.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو صاعِدٍ الكِلابيُّ (ت: 216هـ=831م):

"يُقالُ ‌غَذَموا ثَمَّ غُذْمَةً مُنْكَرَةً -بِضَمِّ الغَيْنِ وسُكونِ الذَّالِ-، وغَذيمَةً -عَلى مِثالِ فَعيلَةٍ-: إذا وَجَدوا في الأَرْضِ واقِعَةً مِنْ عُشْبٍ أوْ بَقْلٍ مُنْكَرَةً ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القالي، البارع. تح: الطعان، ص: 371.— المصدر
1.5معنى من معاني «غَذَمَ»

وــــــــــ الفَصيلُ ما في ضَرْعِ أمِّهِ: شَرِبَ جَميعَ ما فيه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عُبَيْدٍالقاسِمُ بْنُ سَلّامٍ (ت: 224هـ=838م) يُعَلِّقُ على قَوْلِ خَلَفٍ الأحْمَر: "يُقالُ: اغْتَذَمَ الحُوارُ ما في ضَرْعِ أمِّهِ، وذَلِكَ إِذا اسْتَوْعَبَهُ فَلَمْ يُبْقِ فِيهِ شَيْئا":

"وكَذَلِكَ امْتَكَّهُ، وكُلُّ مَن أكَلَ شَيْئًا أَو شَرِبَهُ برَغَبٍ ونَهَمٍ فَقَدْ ‌غَذَمَهُ ‌واغْتَذَمَهُ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، غريب الحديث. الهندية، ج: 4، ص: 35.— المصدر
02 —

غَذِمَ

فعل
2.1معنى من معاني «غَذِمَ»

* غَذِمَ (بِكَسْرِ الذّالِ) فُلانٌ وغَيْرُهُ يَغْذَمُ (بِفَتْحِ الذّالِ) غَذْمًا: أَكَل بِشدَّة ونَهْمَةٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو ذَرّ الغِفاريُّ (ت: 32هـ=652م) يُخاطِبُ عُثْمانَ بنَ عَفّانَ حينَ عَرَضَ عَلَيْهِ الإقامَةَ مَعَهُ فَرَفَضَ واسْتَأذنَهُ إلى الرَّبْذَةِ:

"عَلَيْكُمْ مَعْشَرَ قُرَيْشٍ بِدُنْياكُمْ ‌فاغْذُموها".

القالي، البارع. تح: الطعان، ص: 375.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال اللَّيْثُ بْنُ الـمُظَفَّرِ الكِنانيُّ (ت: 180ه=796م):

"الغَذَمُ: ‌الأَكْلُ ‌بِـجَفاءٍ ‌وشِدَّةِ ‌نَـهَمٍ، وقَدْ غَذِمْتُ أَغْذَمُ غَذَمًا ". (وهذه الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بِقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 100.— المصدر
2.2معنى من معاني «غَذِمَ»

وــــــــــ الطَّعامَ: لَقِمَ مِنْهُ لُقْمَةً.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مِسْحَلٍ الأعْرابيُّ (ت: 230هـ=845م):

"يُقال: زَهِمْتُ زُهْمَةً، وخَضِمْتُ خُضْمةً، وغَذِمتُ ‌غُذمةً؛ بِمَعْنى: لَقِمْتُ لُقْمةً (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الأزهري، تهذيب اللغة. تح: مجموعة من المحققين، ج: 6، ص: 167.— المصدر
03 —

أَغْذَمَ

فعل
3.1معنى من معاني «أَغْذَمَ»

* أغْذَمَ فُلانٌ فُلانًا: أَطْعَمَهُ ما يُشْبِعُهُ.‌

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وأَغْذَمْتُهُ: أَطْعَمْتُهُ ما ‌يُغْذِمُ".

الخليل، العين. تح: هنداوي، ج: 3، ص: 270.— المصدر
3.2معنى من معاني «أَغْذَمَ»

وــــــــــ الإناءَ وَنَحْوَهُ: مَلَأَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو عَمْرٍو الشّيْبانيُّ (ت: 206هـ=818م):

"الإغْذامُ: الـمَلْءُ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

أبو عمرو، الجيم. تح: الأبياري، ج: 3، ص: 9.— المصدر
”وقال الرّاجِزُ (206هـ=821م، من شَواهِدِ أبي عَمْرو الشَّيبانيّ) يَذْكُرُ كَيْلًا زَمَنَ الجاهِلِيَّةِ:

إِذا أُنيخَتْ والتَقَوْا بالأَهْجامْ

‌أَوْفَتْ ‌لَهُمْ ‌كَيْلًا سَريعَ الإغْذامْ

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 9.— المصدر
3.3معنى من معاني «أَغْذَمَ»

وــــــــــ الشّيْءَ: أَخَذَ مِنْهُ أَخْذًا كَثيرًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”وبه فُسّرِ قَوْلُ الرّاجِز (206هـ=821م، من شَواهِدِ الشَّيبانيّ) يَذْكُرُ كَيْلًا زَمَنَ الجاهِلِيَّةِ:

إِذا أُنيخَتْ والتَقَوْا بالأَهْجامْ

‌أَوْفَتْ ‌لَهُمْ ‌كَيْلًا سَريعَ الإغْذامْ

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 9.— المصدر
3.4معنى من معاني «أَغْذَمَ»

وــــــــــ لِفُلانٍ شَيْئًا: أعْطاهُ عَطاءً حَسَنًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو عَمْرٍو الشّيْبانيُّ (ت: 206هـ=818م):

"أَغْذَمْتُ لَهُ ‌غُذمَةً حَسَنَةً؛ أيْ: عَطاءً حَسَنًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 12.— المصدر
3.5معنى من معاني «أَغْذَمَ»

وــــــــــ الفَصيلُ ما في ضَرْعِ أمِّهِ: اشْتَفَّهُ كُلَّهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزابادي (ت: 817هـ=1414م):

"‌وأغْذَمَ ‌الفَصيلُ ما في ضَرْعِ أُمِّه، وغَنْذَمَه واغْتَذَمَه: شَرِبَ جَمِيعَهُ ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزابادي، القاموس. تح: العرقسوسي، ص: 1142.— المصدر
04 —

اغْتَذَمَ

فعل
4.1معنى من معاني «اغْتَذَمَ»

* اغْتَذَمَ فُلانٌ وغَيْرُهُ الشَّيْءَ: اسْتَوْعَبَهُ كُلَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"واغْتَذَمَ الحُوارُ ما فى ضَرْعِ أُمِّهِ؛ أيْ: اسْتَوْعَبَهُ كُلَّهُ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الخليل، العين. تح: هنداوي، ج: 3، ص: 270.— المصدر
”وقالَ عَبْدُ الله بْنُ هِبة (250هـ=864م) يَمْدَحُ أبا المُغِيرَةَ بنَ عيسى المَخْزومِيَّ:

وقُمْتَ باليُمْنِ فانْساحَتْ مذَاهِبُهُمْ
وَراحَ ما كانَ مِنْ خَوْفٍ ومِنْ عَدَمِ

وَدَرَّ لِلنّاسِ والأَنْعامِ عاقِبَةٌ
بِسَاكِبٍ من ضُروعِ الرِّزْقِ ‌مُغْتَذِمِ

الهجري، التعليقات والنوادر. تح: الجاسر، ج: 2، ص: 716.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الله الكَوْكَنيّ (ت: 1320هـ=1902م) يُوَجِّهُ خِطابَهُ إلى نَدْوَةِ العُلَماءِ:

أَلَمْ تَرَوْا كَيْفَ فاتَتْ مِنْ تَشاجُرِكُمْ
فَوائِدٌ واسْتَفادَ الغَيْرُ مُغْتَذِمَا

الحسني، الإعلام بمن في تاريخ الهند من الأعلام. ج: 1، ص: 1292.— المصدر
05 —

تَغَذَّمَ

فعل
5.1معنى من معاني «تَغَذَّمَ»

* تَغَذَّمَ فُلانٌ: لَـمْ يَتَعاظَمْهُ شَيْءٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النَّضْرُ بْنُ شُمَيلٍ (ت: 203هـ=819م):

"هُوَ سَيِّدٌ ‌مُتَغَذِّمٌ لا يُمْنَعُ مِنْ كُلِّ ما أَرادَ ولا يَتَعاظَمُهُ شَيْءٌ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

ابن منظور، لسان العرب. تح: أمين محمد، وآخر، ج: 10، ص: 29.— المصدر
5.2معنى من معاني «تَغَذَّمَ»

وــــــــــ فُلانٌ: امْتَلَأَ غَضَبًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّاد (ت: 385هـ=995م):

"وجاءَني ‌مُتَغَذِّمًا: قَدِ امْتَلَأَ مِنَ الغَضَبِ".

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 54.— المصدر
5.3معنى من معاني «تَغَذَّمَ»

وــــــــــ فُلانٌ وغَيْرُهُ الشَّيْءَ، وَبِهِ: مَضَغَهُ وأَلْقاهُ مِنْ فِيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو ذُؤَيبٍ الهُذَليُّ (ت: 27هـ=648م) يَصِفُ سَحابًا يُرْعِدُ بِشِدَّةٍ مُشَبِّهًا صَوْتَهُ بِهَديرِ الفُحولِ تَمْضَغُ الزَّبَدَ:

كأَنَّ مَصاعيبَ زُبَّ الرُّؤو
سِ في دارِ صِرْمٍ تُلاقى مُريحا

‌تَغَذَّمْنَ في جانِبَيْهِ الخَبيــــ
ـــــرَ لَمّا وَهى مُزْنُهُ واسْتُبيحَا

ديوان أبي ذؤيب الهذلي. تح: أحمد الشّال، ص: 110.— المصدر
