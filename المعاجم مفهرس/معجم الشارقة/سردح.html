01 —

سَرْدَحَ

فعل
1.1معنى من معاني «سَرْدَحَ»

* سَرْدَحَ فلانٌ الشَّيْءَ: أَهْمَلَهُ. (انظر: س ر د ج)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ (ت: 515هـ=1122م):

"وسَرْدَحَ الشيءَ: أهْمَلَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القطاع، كتاب الأفعال، ج: 2، ص: 174.— المصدر
”وقالَ الرَّاجزُ (573هـ=1178م)، من شواهِدِ نَشْوانَ الحِمْيري:

وتَرَكَتْكَ اليَوْمَ كالْمُسَرْدَحِ

نشوان الحميري، شمس العلوم. تح: العمري وآخرين، ج: 5، ص: 3068.— المصدر
02 —

سِرْداح

اسم
2.1معنى من معاني «سِرْداح»

* السِّرْداحُ: الْمَكانُ المُتَّسِعُ مِنَ الوادِي، أو الوادي نَفْسُه. (ج) سَرادِحُ، وسَرادِيحُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسّانُ بْنُ ثابِتٍ (3هـ=624م) يَبْكي الأَطْلالَ مِنْ قَصيدَةٍ يَرْثِي بِها حَمْزَةَ بنَ عَبْدِ المُطَّلِبِ، رَضِيَ اللهُ َعْنُه:

أَتَعْرِفُ الدَّارَ عَفَا رَسمَها
بَعْدَكَ صَوْبُ المُسْبِلِ الهاطِلِ

بَيْنَ السَّرادِيحِ فَأُدْمــــــــــانَةٍ
فَمَدْفَــــــعِ الرَّوْحـــــاءِ في حــــــائِلِ

ديوان حسّان بن ثابت. تح: مهنّا، ص: 194.— المصدر
2.2معنى من معاني «سِرْداح»

وــــــــــ: النّاقَةُ الكَرِيمَةُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُمَيْدُ بْنُ ثَوْرٍ (ت: 30هـ=651م) يَصِفُ نَاقَتَهُ:

أَتَنْسَى عَدُوًّا سَارَ نَحْوَكَ لَمْ يَزَلْ
ثَمَانِينَ عامًا قَبْضَ نَفْسِكَ يَطْلُبُ

وتَذْكُرُ سِرْداحًا مِنَ الوَصْلِ باقِيًا
طَويلَ القَرَا أَنْضَيْتَهُ وَهْوَ أَحْدَبُ

ديوان حميد بن ثور. تح: البيطار، ص: 270.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ مَيّادَةَ (ت: 149هـ=766م) يَتَغَزَّلُ:

وكَواعِبٍ قَدْ قُلْنَ يَوْمَ تَواعُدٍ
قَوْلَ الْمُجِدِّ وهُنَّ كَالْمُزَّاحِ:

يا لَيْتَنا في غَيْرِ أَمْرٍ فادِحٍ
طَلَعَتْ عَلَيْنا العِيــــسُ بِالرَّمَّاحِ

بَيْنا كَذَاكَ رَأَيْنَنِي مُتَعَصِّبًا
بِالخَزِّ فَوْقَ جُـــــــــــــــــلالَةٍ سِرْداحِ

شعر ابن ميادة. تح: حداد، ص: 99.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرَّضِيُّ الإِسْتَراباذِيُّ (ت: 686هـ=1287م) يَشْرَحُ قاعِدَةً في التَّصْغيرِ والتَّكْسيرِ:

"وقَوْلُهُمْ: الرَّابِعُ الوَسَطُ يَتَحَرَّكُ فِي التَّصْغِيرِ والتَّكْسِيرِ إِذَا حُذِفَ الخامِسُ، لَيْسَ بِمُسْتَقيمٍ؛ لأَنَّ الأَلِفَ تُقْلَبُ إِذَنْ ياءً ساكِنَةً كَسُرَيْدِيحٍ، وسَراديحَ فِي سِرْداحٍ".

الإستراباذي، شرح شافية ابن الحاجب. تح: الحسن وآخرين، ص: 57.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ الطُّلْبَة الشِّنْقيطِيّ (ت: 1272هـ=1856م)َ يُخاطِبُ صاحِبَيْهِ:

يا خَلِيلَيَّ ما شَفَى النَّفْسَ شافٍ
كاعْتِمالِ الجُلالَةِ السِّرْداحِ

ديوان محمد الطلبة الشنقيطي، ص: 186.— المصدر
2.3معنى من معاني «سِرْداح»

وــــــــــ: الفَرَسُ القَوِيَّةُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37هـ=658م) يَصِفُ خَيْلَ قَوْمِهِ مُفْتَخِرًا:

جُرْدٌ تُباري الشَّبا أُرْقٌ مَراكِلُها
مِثْلُ السَّراحينِ مِنْ أُنْثَى ومِنْ ذَكَرِ

مِنْ كُلِّ أَهْوَجَ سِرْداحٍ ومُقْرَبَةٍ
تُقاتُ يَوْمَ لِـــــــكاكِ الـــــــوِرْدِ بِالغُــــــــــــمَرِ

ديوان ابن مقبل. تح: عزة حسن، ص: 78.— المصدر
2.4معنى من معاني «سِرْداح»

وــــــــــ: جَمَاعَةُ الطَّلْحِ، واحِدُها: سِرْداحَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"السِّرْداحُ: جَمَاعَةُ الطَّلْحِ،‏ وَاحِدُها: سِرْداحَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 3، ص: 232.— المصدر
2.5معنى من معاني «سِرْداح»

وــــــــــ: الرَّمْلَةُ العَظيمَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشاعرُ (206هـ=821م)، من شواهدِ الشَّيْبانيِّ:

مِنَ الرَّمْلِ في تَيْهُورَةٍ حَفَّ جَوْفَهُ
أَكِلَّةُ سِرْدَاحٍ مُنِيفٍ غوَارِبُهْ

أبو عمرو الشيباني، الجيم. تح: الطحاوي وآخر، ج: 2، ص: 119.— المصدر
2.6معنى من معاني «سِرْداح»

وــــــــــ مِنَ الأَراضي: اللَّيِّنَةُ الْمُسْتَوِيَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجزُ (216هـ=831م)، مِنْ شَواهِدِ الأَصْمَعِيِّ، يَصِفُ مَكَانًا مِنَ الأَرْضِ:

عَلَيْكَ سِرْداحًا مِنَ السّرادِحِ

 ذا عَجْلَةٍ وذا نَصِيٍّ واضِحِ

الأزهري، تهذيب اللغة. تح: درويش، ج: 5، ص: 322.— المصدر
2.7معنى من معاني «سِرْداح»

وــــــــــ: الأَسَدُ القَوِيُّ الشَّديدُ التَّامُّ الخِلْقَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشاعرُ (216هـ=831م) ، مِنْ شَواهِدِ الأَصْمَعِيِّ، يَذْكُرُ لِصًّا يَفْتَخِرُ بِجُرْأَتِهِ:

وكَأَنِّي في فَحْمَةِ ابْنِ جَمِيرٍ
فِي نِقَابِ الأُسَامةِ السِّرْداحِ

الصغاني، التكملة. تح: الأبياري، ج: 2، ص: 44.— المصدر
2.8معنى من معاني «سِرْداح»

وــــــــــ مِنَ الأَراضي: الواسِعَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السّرّاجِ النَّحْوِيُّ (ت: 316هـ=929م):

"والصِّفَةُ: ‌سِرْداحٌ وهيَ الأَرضُ الواسعةُ".

ابن السراج، الأصول في النحو. تح: الفتلي، ج: 3، ص: 218.— المصدر
2.9معنى من معاني «سِرْداح»

وــــــــــ: البَعيدَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"وأَرْضٌ سِرْداحٌ: بَعيدَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 1202.— المصدر
2.10معنى من معاني «سِرْداح»

وــــــــــ مِنَ النَّخْلِ: الصَّفِيَّةُ الكَثيرَةُ الحَمْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"نَخْلَةٌ سِرْداحٌ:‏ كَريمَةٌ صَفِيَّةٌ".

ابن سيده، المخصص. تح: جفال، ج: 3، ص: 218.— المصدر
2.11معنى من معاني «سِرْداح»

وــــــــــ مِنْ كُلِّ شَيْءٍ: الضَّخْمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السِّيرافيُّ (ت: 368هـ=979م):

"ويُقالُ لِكُلِّ ضَخْمٍ أيْضًا: سِرْداحٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السيرافي، شرح كتاب سيبويه. تح: مهدلي وآخرين، ج: 5، ص: 190.— المصدر
03 —

سَرْدَح

اسم
3.1معنى من معاني «سَرْدَح»

* السَّرْدَحُ: الأَرْضُ الليِّنَةُ أو المُسْتَوِيَةُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أُحَيْحَةُ بْنُ الجُلّاحِ (ت: 129ق.هـ=496م) يَصِفُ نَاقَتَهُ التِى تَغوصُ أقْدامُها في رِمالِ الصَّحْراءِ:

ولْتَبْكِنِي ناقَةٌ إِذَا رَحَلَتْ
وغابَ في سَرْدَحٍ مَناكِبُها

ديوان أحيحة بن الجلاح. تح: باجوده، ص: 34.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جُهَيْشُ بْنُ أَوْسٍ النَّخَعيُّ (ت: 15هـ=636م) عِنْدَ قُدُومِهِ مَعَ وَفْدٍ مِن قَوْمِهِ:

"وكَأَيِّن قَطَعْنَا إلَيْكَ مِنْ دَوِّيَّةٍ سَرْبَح، ودَيْمومَةٍ سَرْدَحٍ".

المديني، المجموع المغيث. تح: العزباوي، ج: 2، ص: 253.— المصدر
3.2معنى من معاني «سَرْدَح»

وــــــــــ: البَعيرُ الضَّخْمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الراجزُ (216هـ=831م) ، مِنْ شَواهِدِ الأَصْمَعِيِّ، يَصِفُ راعيًا:

بُلِّيَ فِي إثْرِ الجِمــــــــــــــالِ الوُقَّـــــحِ

وإِثْرِ كُلِّ دَرْدَبِيـــــــــــــــــــــسٍ سَــــرْدَحِ

أبو الطيب اللغوي، الإبدال. تح: التنوخي، ج: 1، ص: 44.— المصدر
