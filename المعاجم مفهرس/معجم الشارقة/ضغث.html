ضَغَثَ(فعل) ضَغِثَ(فعل) ضُغِث(فعل) أَضْغَثَ(فعل) ضاغَثَ(فعل) ضَغَّثَ(فعل) اضْطَغَثَ(فعل) تَضْغيث(اسم) ضاغِث(اسم) ضَغاثة(اسم) ضَغْث/ ضَغَث(اسم) ضِغْث(اسم) ضَغوث(اسم) ضَغِيث(اسم) ضَغِيثَة(اسم)
النقوش العربيّة

وَرَدَ الجَذْرُ (ض غ ث ضغث ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَينِها نَقْشٌ سَبَئِيٌّ يَعودُ تاريخُهُ إلى فَتْرةِ ما بَينَ القَرنَينِ الأوَّلِ قَبْلَ الميلادِ والثّاني المِيلادِيِّ (UAM 508/2)، وَرَدَ فيهِ الاسْمُ (ضغيثت ḍaġyṯat ) عَلَمًا على إحدى القبائِلِ، فَجاءَ في النَّقْشِ: "عميثع/ و الشرح/ ذتو/ ضغيثت"،) أيْ: ( عَم يثعَ وإيلِ شَرحَ اللّذانِ مِنْ قَبيلةِ ضَغيثةَ".

(باطايع والحسني، 2015م: 84)

المعانى الكلّيّة

  1. الحُزْمَةُ مِنَ الشَّيْءِ
  2. الاخْتِلاطُ والالتِباسُ
01 —

ضَغَثَ

فعل
1.1معنى من معاني «ضَغَثَ»

* ضَغَثَ (بِفَتْحِ الغَيْنِ) الوَرَلُ، وَهُوَ حَيوانٌ شَبِيهُ الضَّبِّ، يَضْغَثُ (بِفَتْحِ الغَيْنِ) ضَغْثًا: صَوَّتَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207هـ=822م):

"ضَغَثَ ‌الوَرَلُ: صَوَّتَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ).

الزبيدي، تاج العروس. تح: حجازي، ج: 5، ص: 288.— المصدر
1.2معنى من معاني «ضَغَثَ»

وــــــــــ فُلانٌ الشَّعْرَ وَنَحْوَهُ: دَلَكَهُ بِيَدَيْهِ، لِيَصِلَ الماءُ إِلى فَرْوَةِ الرَّأْسِ؛ فَيَسْهُلَ غَسْلُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ أُمُّ المُؤْمِنينَ عائِشَةُ (ت: 58هـ=678م) تُبَيِّنُ طَريقَةَ غُسْلِ المَرْأَةِ:

"لِتَحْفِنْ عَلى رَأْسِها ثَلاثَ حَفَناتٍ مِنَ الماءِ، ولْتَضْغَثْ رَأْسَها بِيَدَيْها".

مالك بن أنس، الموطأ. تح: الأعظمي، ج: 2، ص: 63.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"ضَغَثَ رَأْسَهُ: أيْ دَلَكَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 364.— المصدر
”وقالَ مالِكٌ (ت: 179هـ=795م) يُوَضِّحُ كَيْفِيَّةَ غَسْلِ الشَّعَرِ لِلجُنُبِ، رَجُلًا كانَ أَوِ امْرَأَةً:

"ولْيَضْغَثاهُ في الجَنابَةِ، وإِنْ كانَ مَضْفورًا".

ابن أبي زيد القيراوني، النوادر. تح: الحلو، ج: 1، ص: 40.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ خليلٌ المالِكيُّ (ت: 776هـ=1375م) يَذْكُرُ واجِباتِ الغُسْلِ مِنَ الجَنابَةِ:

"تَخْليلُ شَعَرٍ، وضَغْثُ مَضْفورِهِ، لا نَقْضُهُ".

الحطّاب، مواهب الجليل. تح: الشنقيطي وآخر، ص: 481.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَرَفَةَ الدّسوقيُّ (ت: 1230هـ=1815م) يَذْكُرُ واجِباتِ الغُسْلِ مِنَ الجَنابَةِ:

"لا بُدَّ مِنَ التَّخْليلِ (...) فِي ‌ضَغْثِ ‌المَضْفورِ ‌مِنَ ‌الشَّعْرِ".

حاشية الدسوقي على الشرح الكبير. ج: 1، ص: 134.— المصدر
1.3معنى من معاني «ضَغَثَ»

وــــــــــ النّاقَةَ وَنَحْوَها: جَسَّ سَنامَها بِيَدِهِ؛ لِيَعْرِفَ سِمَنَها مِنْ هُزالِها، فَهِيَ ضَغوثٌ. (ج) ضُغَثٌ وضُغُثٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"ناقَةٌ ضَغوثٌ: ‏لا يُدْرى سِمَنُها حَتّى تُضْغَثَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 364.— المصدر
1.4معنى من معاني «ضَغَثَ»

وــــــــــ الطَّعامَ وَنَحْوَهُ: لاكَهُ بِالأَنْيابِ والنَّواجِذِ. (انْظُرْ: ض غ ت).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الضَّغْثُ: اللَّوْكُ بِالأَنْيابِ والنَّواجِذِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 364.— المصدر
1.5معنى من معاني «ضَغَثَ»

وــــــــــ الشَّيْءَ: خَلَطَ بَعْضَهُ بِبَعْضٍ. (ز).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَلِيلُ (ت: 170هـ=787م):

"الضَّغْثُ: التِباسُ الشَّيْءِ بَعْضِهِ بِبَعْضٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 364.— المصدر
”وقالَ أبو زيادٍ الكِلابيُّ (ت: 200هـ=815م) يَنْعى عَلى النّاسِ خَلْطَهُمُ الحَديثَ:

"‏النّاسُ يَضْغَثونَ أَشْياءَ عَلى غَيْرِ وَجْهِها، قيلَ: ‏وما يَضْغَثونَ؟ قال: يَقولونَ الشَّيْءَ حِذاءَ الشَّيْءِ، ولَيْسَ بِهِ".

السمين الحلبي، عمدة الحفاظ. تح: عيون السود، ج: 2، ص: 381.— المصدر
1.6معنى من معاني «ضَغَثَ»

وـــــــــــ الثَّوْبَ: غَسَلَهُ وَلَمْ يُنَقِّهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"ضَغَثْتُ ‌الثَّوْبَ ضَغْثًا: أَيْ غَسَلْتُهُ وَلَمْ أُنَقِّهِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 1، ص: 394.— المصدر
1.7معنى من معاني «ضَغَثَ»

وــــــــــ الحَشيشَ وغَيْرَهُ: حَزَمَهُ حَزْمًا.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ السَّمينُ الحَلَبيُّ (ت: 756هـ=1355م):

"ضَغَثَ الحَشيشَ ضَغْثًا، أَيْ حَزَمَهُ حَزْمًا". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

السمين الحلبي، عمدة الحفاظ. تح: عيون السود، ج: 2، ص: 381.— المصدر
02 —

ضَغِثَ

فعل
2.1معنى من معاني «ضَغِثَ»

* ضَغِثَ (بِكَسْرِ الغَيْنِ) فُلانٌ يَضْغَثُ (بِفَتْحِ الغَيْنِ) ضَغَثًا: حَمُقَ، فَهُوَ ضَغيثٌ. (ولَمْ يُسْتَعْمَلِ الفِعْلُ بِهَذِهِ الدَّلالَةِ مِنْ قَبْلُ، واسْتُعْمِلَ المُشْتَقُّ مِنْهُ).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأُمَوِيُّ (ت: 203هـ=818م):

"يُقالُ لِرَجُلٍ ضَغيثٍ: أَحْمَقُ".

تاريخ ابن معين. تح: نور سيف، ج: 3، ص: 419.— المصدر
2.2معنى من معاني «ضَغِثَ»

وــــــــــ الوَرَلُ، وَهُوَ حَيوانٌ شَبِيهُ الضَّبِّ: صَوَّتَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207هـ=822م):

"ضَغِثَ ‌الوَرَلُ: صَوَّتَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: حجازي، ج: 5، ص: 288.— المصدر
03 —

ضُغِث

فعل
3.1معنى من معاني «ضُغِث»

* ضُغِثَ فُلانٌ: غُمَّ قَلْبُهُ. (ولَمْ يُسْتَعْمَلِ الفِعْلُ بِهَذِهِ الدَّلالَةِ مِنْ قَبْلُ، واسْتُعْمِلَ المُشْتَقُّ مِنْهُ).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385هـ=995م):

"رَجُلٌ مَهْمُوصُ الفُؤادِ، أَيْ ‌مَضْغُوثُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 1، ص: 291.— المصدر
04 —

أَضْغَثَ

فعل
4.1معنى من معاني «أَضْغَثَ»

* أضْغَثَ الحالِمُ الرُّؤْيا: خَلَّطَ في قَصِّها والْتَبَسَتْ عَلَيْهِ أَحْداثُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الأَضْغاثُ: أَحْلامٌ مُلْتَبِسَةٌ، ويُقالُ لِلحالِمِ: أَضْغَثْتَ الرُّؤْيا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 364.— المصدر
”وقالَ ابْنُ دَرّاجٍ القَسْطَلِّيُّ (392ه=1002م) يَمْدَحُ المَنْصورَ بْنَ أَبي عامِرٍ:

وَدَعْوى النُّهَى وَالحِلْمِ في غَيْرِ مُنْذِرٍ
خَيالٌ مِنَ الأَحْلامِ أَضْغَثَ حَالِمُهْ

ديوان ابن دراج. تح: مكي، ص: 195— المصدر
05 —

ضاغَثَ

فعل
5.1معنى من معاني «ضاغَثَ»

* ضاغَثَ فُلانٌ الحُلْمَ: خَلَّطَ في قَصِّهِ والْتَبَسَتْ عَلَيْهِ أَحْداثُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مِهْيارٌ الدَّيْلَميُّ (ت: 428هـ=1037م) يَمْدَحُ قَوْمًا:

عِيافةٌ ما كَذَبَتْني فيكُمُ
قَطُّ ولا زَجَرْتُ مِنْها أَشْأَما

وواعِدٌ مِنَ الأماني لَمْ يَكُنْ
خالًا ولا ضاغَثْتُ مِنْهُ حُلُما

ديوان مهيار الديلمي. تح: نسيم، ج: 3، ص: 287.— المصدر
06 —

ضَغَّثَ

فعل
6.1معنى من معاني «ضَغَّثَ»

* ضَغَّثَ فُلانٌ النَّباتَ: جَعَلَهُ حُزَمًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37هـ=657م) يَصِفُ الكَلَأَ:

 ضَغَّثَ أَوْساطَهُ خالٍ وخَلَّطَهُ
مِنَ الخُزامى بِأَحْدابٍ ومُهْتَضَمِ

القالي، البارع. تح: الطعان، ج: 1، ص: 258.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانيُّ (ت: 206هـ=821م):

"ضَغَّثَ النَّبْتَ تَضْغيثًا: جَعَلَهُ حُزَمًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القالي، البارع. تح: الطعان، ج: 1، ص: 258.— المصدر

وــــــــــ الشَّعْرَ وَنَحْوَهُ: دَلَكَهُ بِيَدَيْهِ، لِيَصِلَ الماءُ إلى فَرْوَةِ الرَّأْسِ؛ فَيَسْهُلَ غَسْلُهُ.

ع

07 —

اضْطَغَثَ

فعل
7.1معنى من معاني «اضْطَغَثَ»

* اضْطَغَثَ فُلانٌ الحَشيشَ وَنَحْوَهُ: جَمَعَهُ وحَزَمَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ(216هـ=831م) مِنْ شَواهِدِ الأَصْمَعِيِّ، يَصِفُ احْتِطابَ العُشْبِ:

إِنْ يَخْلِهِ بِعِرْقِهِ أَوْ يَجْتَثِثْ

لا يَخْلِ حَتّى اللَّيْلِ ضِغْثَ المُضْطَغِثْ

الصغاني، التكملة. تح: الطحاوي، ج: 1، ص: 370.— المصدر

ضاغِث

اسم
9.1معنى من معاني «ضاغِث»

* الضَّاغِثُ مِنَ النّاسِ: الَّذي يَتَنَكَّرُ وَيُفْزِعُ الصِّبْيانَ بِمِثْلِ صَوْتِ الوَحْشِ. (انْظُرْ: ض غ ب).

ع
