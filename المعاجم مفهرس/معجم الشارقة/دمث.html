  1. اللِّينُ والسُّهولَةُ
01 —

دَمِثَ

فعل
1.1معنى من معاني «دَمِثَ»

* دَمِثَ (بِكَسْرِ الميمِ) الـمَكانُ وغَيْرُهُ يَدْمَثُ (بِفَتْحِ الميمِ) دَمَثًا، ودَمْثًا، ودَماثَةً: سَهُلَ ولانَ، فهُوَ دَمِثٌ، ودَمْثٌ، ودَميثٌ. (ج) دِماثٌ، وأَدْماثٌ. وهي بِتاءٍ (ج) دَمِثاتٌ، ودِماثٌ، ودُمْثٌ ودَمائِثُ، وهي أيضًا دَمْثاءُ، وهو وهي دَمَثٌ (ج) أَدْماثٌ، دِماثٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ قِلابَةُ الهُذَلِيّ (146 ق ه=480م) يَصِفُ صاحِبَتَه:

خَوْدٌ ثَقَالٌ فِي القِيَامِ كَرَمْلَةٍ
دَمَثٍ يُضِيءُ لَهَا الظَّلَامُ الحِنْدِسُ

السكري، شرح أشعار الهذليين. تح: فرّاج، ج: 2، ص: 714.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ساعِدَةُ بْنُ جُؤَيَّة (ت: 11هـ=632م):

بِشَرَبَّةٍ دَمِثِ الكَثيبِ بِدورِهِ
أَرْطى يَعوذُ بِهِ إِذا ما يُرْطَبُ

شعر ساعدة بن جؤية. تح: قتلان، ص: 147.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (نَحْوَ: 170هـ=786م، مِنْ شَواهِدِ الخَليلِ) يصِفُ مَكانًا:

فلَوْ عَلَوْهُ جَرِلًا هَراسا

لتَرَكوهُ دَمِثًا دَهاسا

الخليل، العين. تح: المخزومي وآخر، ج: 6، ص: 101.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ التّورِبِشْتِيّ (ت: 661هـ=1263م) يُوَجِّهُ التَّفْريقَ بَيْنَ اللَّحْدِ والشَّقِّ:

"لِأَنَّ النّاسَ في كَثيرٍ مِنَ البُلْدانِ مُضْطَرونَ إِلَى الشَّقِّ إِذا كانَتِ الأَرْضُ رَخْوَةً أَوْ ‌دَمِثَةً ذاتَ رَمْلٍ، وإِذا كانَتْ صُلْبَةً: فالاخْتِيارُ اللَّحْدُ".

التُّوربشْتي، الميسر في شرح مصابيح السنة. تح: هنداوي، ج: 2، ص: 399.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد بْنُ الطُّلْبَة (ت: 1272هـ=1856م) يَصِفُ ثَوْرَ الوَحْشِ:

فكَأَنّي إِذا الهَواجِرُ شَبَّتْ
كُلَّ حَزْنٍ عَلى شَبوبٍ لَياحِ

مُفْرَدٍ باِللِّوى يَرودُ دِماثًا
لَمْ يَرُدْهُنَّ غَيْرُ هوجِ الرِّياحِ

ديوان محمد الطلبة. تح: الشبيه، ص: 188.— المصدر
1.2معنى من معاني «دَمِثَ»

وــــــــــ الـمَرْأَةُ: امتَلَأتْ ولانَ جِسْمُها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الـمُتَوَكِّلُ اللَّيْثِيّ (ت: 85هـ=704م) يَتَغَزَّلُ:

وإِذا خَلَوْتَ بِها خَلَوْتَ بِحُرَّةٍ
رَيّا العِظامِ دَمِيثَةٍ مِكْسالِ

شعر المتوكل الليثي. تح: يحيى الجبور، ص: 172.— المصدر
1.3معنى من معاني «دَمِثَ»

وــــــــــ الـعَيْشُ: طابَ ورَغِدَ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الباروديُّ (ت: 1323هـ=1904م) يَشْكو الزَّمانَ وأَهْلَهُ:

بَرِمْتُ بِهِمْ حَتَّى سَئِمِتُ مَكانَتي
وأَنْكَرْتُ طيبَ العَيْشِ وهْوَ دَميثُ

ديوان محمود سامي البارودي. ج: 1، ص: 67.— المصدر
02 —

دَمُثَ

فعل
2.1معنى من معاني «دَمُثَ»

* دَمُثَ (بِضَمِّ المِيمِ) فُلانٌ يَدْمُثُ (بِضَمِّ الميمِ) دَماثَةً، ودُموثَةً: لانَ خُلُقُهُ وحَسُنَ. فهُوَ دَمِثٌ، ودَمْثٌ، ودَميثٌ. (ج) دِماثٌ، وهِيَ بِتاءٍ (ج) دَمائِثُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ هِنْدُ بْنُ أَبي هالَةَ (ت: 36هـ=656م) يَصِفُ النَّبِيَّ ﷺ:

"يَتَكَلَّمُ بِجَوامِعِ الكَلِمِ...دَمِثًا لَيْسَ بالجافي ولا الـمَهينِ".

ابن قتيبة، غريب الحديث. تح: عبد الله الجبوري، ج: 1، ص: 488.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نُوَاس (ت: 198هـ=814م) يَصِفُ ساقِيَ الخَمْرِ:

يَسْعَى بِها خَنِثٌ فِي خُلْقِهِ دَمَثٌ
يَسْتَأْثِرُ العَيْنَ فِي مُسْتَدْرَجِ الرَّائِي

ديوان أبي نواس. تح: واصف، ص: 237.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّار (ت: 658هـ=1260م) يَصِفُ الرَّشيدَ:

"وكَانَ دَمِثًا رَقِيقَ حَاشِيَةِ الطَّبْعِ".

ابن الأبار، الحلة السيراء. تح: مؤنس، ص: 68.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشَّوْكانِيّ (ت: 1250هـ=1834م) يُعَرِّفُ بِالشَّيْخِ أَحْمَدَ بْنِ الحَسَنِ الصَّنْعانيّ:

"وكانَتْ لَهُ اليَدُ الطّولى في تَعْبيرِ الرُّؤْيا والتَّفَرُّسِ في حالِ الرّائي، وكانَ دَمِثَ الأَخْلاقِ".

الشوكاني، البدر الطالع. ج: 2، ص: 26.— المصدر
2.2معنى من معاني «دَمُثَ»

وــــــــــ الحَيوانُ: كَرُمَ أَصْلُهُ وجاد نَوْعُه.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الوَطْواطُ (ت: 718هـ=1318م):

"والنَّوّاحُ: طائِرٌ كالقُمْري، وحالُهُ كَحالِهِ، إِلّا أَنَّهُ أَحَرُّ مِنْهُ مِزاجًا، وأَرْطَبُ، وأَدْمَثُ، وأَشْرَفُ، قالوا: يَكادُ يَكونُ النَّوّاحُ لِلأَطْيارِ ‌الدَّمِثَةِ مَلِكًا، وهُوَ بِصَوْتِهِ يُهَيِّجُها إِلَى التَّصْويتِ".

النويري، مباهج الفكر ومناهج العبر، ج:1، ص: 88.— المصدر
03 —

أَدْمَثَ

فعل
3.1معنى من معاني «أَدْمَثَ»

* أَدْمَثَ القَوْمُ: نَزَلوا السُّهولَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّة (ت: 367هـ=977م):

"وأَدْمَثْنا: نَزَلْنا الدِّماثَ، وهِيَ السُّهولُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: فودة، ص: 125.— المصدر
3.2معنى من معاني «أَدْمَثَ»

وــــــــــ الشَّيْءُ: انْخَفَضَ وتَوارى. (وَهذا الفِعْلُ لَمْ يُسْتَعْمَلْ بِهَذَه الدِّلالَةِ مِنْ قَبْلُ واسْتُعْمِلَ الـمُشْتَقُّ مِنْهُ)

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العَفِيفُ التِّلِمْسانيُّ (ت: 690هـ=1291م) يَتَغَزَّلُ:

ثَرائي وجاهي فيهِ وَجْدٌ وأَدْمُعٌ
فقَلْبِيَ في الحالَيْنِ ضاحٍ ومُدْمِثُ

ديوان عفيف الدّين التّلمسانيّ. تح: يوسف زيدان، ص: 145.— المصدر
04 —

دَمَّثَ

فعل
4.1معنى من معاني «دَمَّثَ»

* دَمَّثَ فُلانٌالمكانَ أَوِ الـمَوْضِعَ ونَحْوَه: سَهَّلَه ومَهَّدَه ووَطَّأَه.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَت عُفَيْراءُ الحِمْيَرِيَّة الكاهِنَةُ (ت: 110ق هـ=512م) تَصِفُ أَنْصارَ النَّبِيِّ ﷺ:

"أَعْضادُهُ غَطارِيفُ يَمانونَ، طائِرُهُمْ بِهِ مَيْمونٌ، يُغْزيهِمْ فيَغْزونَ، ويُدَمِّثُ بِهِمُ الحُزونَ، إلى نَصْرِهِ يَعْتزونَ".

النويري، نهاية الأرب. تح: قميحة وآخرين، ج: 16، ص: 113.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (ت: 40هـ=660م):

"مَنْ كَذَبَ عَلَى رَسولِ اللهِ ﷺ فإِنَّما يُدَمِّثُ مَجْلِسَهُ مِنَ النّارِ".

ابن قتيبة، المسائل والأجوبة. تح: العطية وآخر، ص: 76.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ديكُ الجِنِّ (ت: 235ه = 849م):

عادَ تَدْمِيثُكَ المـَضَاجِعَ للجَنْـ
ـبِ فَعَالَ الخَريدَةِ المِكْسالِ

ديوان ديك الجن. تح: مطلوب وآخر، ص: 122.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ السَّرّاج الوَرّاق (ت: 695هـ=1296م) يَرْثي حِمارًا:

خَلَتْ مِنْهُ مَراغَتُهُ وكانَتْ
تُعَشِّرُهُ وتَأْلَفُهُ الـمَلابِدْ

تُدَمَّثُ تَحْتَ جَنْبيْهِ الحَشايا
وتُلْقى تَحْتَ خَدَّيْهِ الوَسائِدْ

العمري، مسالك الأبصار. تح: كامل الجبوري، وآخر، ج: 19، ص: 74.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الزَّيّات (1352هـ=1934م، تاريخ الطَّبْعِ):

"ولَكِنَّ مَجْدَنا الَّذي تَحَدَّى القُرونَ وغَبَّرَ وَجْهَ الفَلَكِ، لا يَزالُ جَيّاشَ الغَضَبِ عَلَى غَدْرَةِ الجَوِّ بِشَهيدَيْهِ في فَرَنْسا! فهُوَ ‌يُدَمِّثُ لِنُسورِهِ مَثْوَى الضِّيافَةِ، ويَعْقِدُ غَيْبِ ضَميرِهِ عَلَى الثَّأْرِ".

الزيات، مجلة الرسالة، العدد 27، ص: 1.— المصدر
4.2معنى من معاني «دَمَّثَ»

وــــــــــ الشَّيءُ الأَخْلاقَ: حَسَّنَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البَهاءُ زُهَير (ت: 656هـ=1258م):

وما العِشْقُ في الإِنْسانِ إِلّا فَضيلَةٌ

ديوان البهاء زهير. تح: أبي الفضل إبراهيم، ص: 168.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّد كُرْد عَليّ (1326هـ=1909م، تاريخُ الطَّبْعِ):

"والكُرّاثُ ‌يُدَمِّثُ الخُلُقَ، ويُكْثِرُ في مُتَناوِلِهِ الصَّبْرَ".

محمد كرد عَلي، مجلة المقتبس، العدد 45، ص: 50.— المصدر
4.3معنى من معاني «دَمَّثَ»

وــــــــــ فُلانٌالأَمْرَ: طَوَّعَهُ وذَلَّلَهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ خَلِّكان (ت: 681هـ=1282م) يَصِفُ مَحْمودًا الغَزْنَويَّ:

"لَعِبُهُ مَعَ الأَتْرابِ جِدٌّ، وجِدُّهُ مُسْتَكِدٌّ، يَأْلَمُ لِما لا يَعْلَمُ حَتَّى يَقْتُلَهُ خُبْرًا، ويَحْزَنُ لِما يَحْزُنُ حَتَّى يُدَمِّثَهُ قَسْرًا وقَهْرًا".

ابن خَلِّكان، وفيات الأعيان. تح: عباس، ج: 5، ص: 180.— المصدر
4.4معنى من معاني «دَمَّثَ»

وــــــــــ الـمَطَرُ التُّرْبَةَ: بَلَّلَها ولَيَّنَها.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الـمَقَّريُّ (ت: 1041هـ=1631م) يَصِفُ مَوْقِعَ تِلِمْسانَ:

"اقْتَعَدَتْ بِسَفْحِ جَبَلٍ (...)، ويُطَلُّ مِنْها عَلَى فَحْصٍ أَفْيَحَ مُعَدٍّ لِلفِلاحَةِ، تَشُقُّ ظُهورَهُ الأَسْلِحَةُ عَنْ مِثْلِ أَسْنِمَةِ الـمَهاري، وتَبْقُرُ في بُطونِهِ عِنْدَ ‌تَدْميثِ الغَمائِمِ بُطونَ العَذارى".

المقّري التلمساني، نفح الطيب من غصن الأندلس الرطيب. تح: عباس، ج: 7، ص: 134.— المصدر

وــــــــــ التُّرابَ لِلخُبْزَةِ: حَفَرَهُ ولَيَّنَهُ لِيُلْقيَها فيهِ.

ع

ابن عباد، المحيط. تح: آل ياسين، ج: 9، ص: 284.— المصدر
4.6معنى من معاني «دَمَّثَ»

وــــــــــ الحَديثَ لِفُلانٍ: ذَكَرَ لَهُ أَوَّلَهُ؛ لِيَعْرِفَ وَجْهَهُ ويأخُذَ فِيه.

ع
