01 —

ناطَ

فعل
1.1معنى من معاني «ناطَ»

* ناطَ فُلانٌ الشَّيْءَ، وبِهِ، وعَلَيْهِ، وإِلَيْهِ، يَنوطُ نَوْطًا: عَلَّقَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الـمُرَقِّش الأكبر (ت: 75ق.ه=550م) يَصِفُ ناقَتَهُ:

وتُصْبِحُ كالدَّوْداةِ ناطَ زِمامَها
إِلى شُعَبٍ فيها الجَواري العَوانِسُ

المفضل الضبي، المفضليات. تح: شاكر وهارون، ص: 226.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بن الخَطّاب (ت: 23ه=644م) يَتَحَدَّثُ لَمّا أُتِيَ بِمالٍ كَثيرٍ:

"إِنِّي لَأَظُنُّكُمْ قَدْ أَهْلَكْتُمُ النّاسَ ... بِلا سَوْطٍ ولا ‌نَوْطٍ".

القاسم بن سلام، الأموال. تح: سيد بن رجب، ج: 1، ص: 100.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رؤبة (ت: 145ه=762م) يَصِفُ ناقةً:

أَلْقى عَلَيْها صِلْدَمًا مُعَرَّقا

كَأَنَّ نَوْطًا ناطَهُ مُعَلَّقَا﻿

مجموع أشعار العرب. تح: وليم البروسي، ج: 3، ص: 111.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الصّاحِبُ شَرَفُ الدّين الأنْصاريّ (ت: 662ه=1264م) يُخاطِبُ أَحَدَ أَصْحابِهِ:

يا ذا الأَيادي الغُرِّ والـمَنْبِتِ الـ
حُلْوِ الجَنى والـمَنْهَلِ الـمُسْتَطابْ

ومَنْ حَوى مِنْ كُلِّ فَـنٍّ فقَدْ
ناطَ إِلى الحِكْمةِ فَصْلَ الخِطابْ

موسى اليونيني، ذيل مرآة الزمان. ج: 2، ص: 289.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ شِهابُ الدّين الآلوسيُّ (ت: 1270ه=1854م):

"إِنَّ تَنْويعَ الوَعْدِ للمُتَّقي وتَكْريرَ الحَثِّ عَلَيْهِ بَعْدَ الدَّلالةِ عَلى أَنَّ التَّقْوى مَلاكُ الأَمْرِ عِنْدَ اللهِ تَعالى ‌ناطَ بِهِ سُبْحانَهُ سَعادةَ الدّارَيْنِ؛ يَدُلُّ عَلى أَنَّ أَمْرَ الطَّلاقِ والعِدّةِ مِنَ الأُمورِ الَّتي تَحْتاجُ إِلى فَضْلِ تَقْوى".

الآلوسي، روح المعاني. تح: زهير القاسم وآخرين، ج: 27، ص: 214.— المصدر
02 —

نيطَ

فعل
2.1معنى من معاني «نيطَ»

* نيطَ فُلانٌ وغَيْرُهُ: أَصابَتْهُ نَوْطةٌ، وهِيَ وَرَمٌ في الصَّدْرِ. ﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعيّ (ت: 216ه=831م):

"يُقالُ للبَعيرِ إِذا وَرِمَ نَحْرُهُ وأَرْفاغُهُ: قَدْ نيطَ لَهُ نَوْطةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: يعقوب عبد النبي، ج: 14، ص: 30.— المصدر
2.2معنى من معاني «نيطَ»

وــــــــــ ﻿ الشَّيْءُ بالشَّيْءِ وعَلَيْهِ: عُلِّقَ بهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الحارثُ بن حِلّزَةَ (ت: 50ق.ه=573م) يَسْتَنْكِرُ تَحْميلَ قَوْمِهِ جَريرةَ غَيْرِهِمْ:

أَمْ عَلَيْنا جَرَّى العِبادِ كَمَا نيـ
ـطَ بِجَوْزِ الـمُحَمَّلِ الأَعْبَاءُ

أبو بكر الأنباري، شرح القصائد السبع. تح: هارون، ص: 481.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه=632م):

«أُرِيَ اللَّيْلةَ رَجُلٌ صالِحٌ أَنَّ أَبا بَكْرٍ ‌نيطَ برَسولِ اللهِ، ونيطَ عُمَرُ بأَبي بَكْرٍ، ونيطَ عُثْمانُ بعُمَرَ».

أحمد بن حنبل، المسند. تح: الأرنؤوط وآخرين، ج: 23، ص: 124.— المصدر
”وقالَ الأغْلَبُ العِجْليّ (ت: 21ه=642م) يَصِفُ ناقةً:

‌كأَنَّ ‌تحتَ ‌خُفِّها ‌الوَهّاصِ

‌ميظَبَ ‌أُكْمٍ ‌نيطَ ‌بالـمِلاصِ

الأزهري، تهذيب اللغة. تح: هارون، ج: 6، ص: 365.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ مَيّادة (ت: 149ه=766م) يُخاطِبُ الوَليدَ بْنَ يَزيدَ:

أَلا لَيتَ شِعري هَل أَبيتَنَّ لَيْلةً
بِحَرَّةِ لَيلى حَيثُ رَبَّتَني أَهْلي

بِلادٌ بِها نيطَتْ عَلَيَّ تَمائِمي
وقُطِّعْنَ عَنّي حينَ أَدْرَكَني عَقْلي

ابن قتيبة، الشعر والشعراء. تح: أحمد محمد شاكر، ص: 772.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ حازم القَرْطاجَنِّي (ت: 684ه=1285م):

"فضُروبُ التَّقابُلِ الأَرْبَعةُ إِنَّما يَصِحُّ مِنْها ما لَمْ يَتَوافَ الـمُتَقابِلانِ فيهِ مِنْ جِهةٍ واحِدةٍ، ولَكِنْ ‌نيطَ هذا بجِهةٍ وهذا بجِهةٍ".

القرطاجنيّ، منهاج البلغاء. تح: بلخوجة، ج: 1، ص: 121.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال ابْنُ عَجيبة (ت: 1224ه=1809م) يُفَسِّرُ قَوْلَ اللهِ تعالى: ﴿وَيَسۡـَٔلُونَكَ عَنِ ٱلرُّوحِۖ قُلِ ٱلرُّوحُ مِنۡ أَمۡرِ رَبِّي وَمَآ أُوتِيتُم مِّنَ ٱلۡعِلۡمِ إِلَّا قَلِيلٗا﴾ [الإسراء: 85]

"فإِنَّ مِنَ الحِكْمةِ الإِنْسانِيّةِ أَنْ يُعَلِّمَ مِنَ الخَيْرِ ما تَسَعُهُ الطّاقةُ البَشَرِيّةُ، بَلْ ما ‌نيطَ بِهِ المَعاشُ والمَعادُ، وذَلِكَ بالإِضافةِ إِلى ما لا نِهايةَ لَهُ مِنْ مُتَعَلَّقاتِ عِلْمِهِ سُبْحانَهُ، قَليلٌ يُنالُ بهِ خَيْرٌ: كَثيرٌ في نَفْسِهِ".

ابن عجيبة، البحر المديد. تح: القرشي، ج: 3، ص: 228.— المصدر
2.3معنى من معاني «نيطَ»

وــــــــــ فُلانٌ بفُلانٍ: أحَبَّهُ وتَعلَّقَ بهِ. ﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وفُلانٌ مَنوطٌ بفُلانٍ: إذا أَحَبَّهُ وتَعَلَّقَ بحَبْلِهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 456.— المصدر
”وقال ابْنُ طَباطبا (ت: 322ه=934م) يَمْدَحُ صاحِبَهُ:

لي صاحِبٌ لا غابَ عَنّي شَخْصُهُ
أَبَدًا وظَلْتُ مُمَتَّعًا بوِدادِهِ

فَطِنٌ بما يوحى إِلَيْهِ كَأَنَّما
قَدْ ‌نيطَ هاجِسُ فِكْرَتي بفُؤادِهِ

ياقوت الحموي، معجم الأدباء. تح: إحسان عباس، ص: 2829.— المصدر
03 —

أَناطَ

فعل
3.1معنى من معاني «أَناطَ»

* أَناطَ البَعيرُ: أَصابَتْهُ نَوْطةٌ، وهِيَ وَرَمٌ في الصَّدْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرو الشَّيْبانيّ (ت: 206ه=821م):

"‌أناطَتِ ‌الإِبِلُ: ‌أَصابَها ‌وَرَمٌ ‌في ‌نُحورِها؛ ‌مثل ‌نيطَتْ".

الصغاني، الشوارد. تح: حجازي، ص: 202.— المصدر
3.2معنى من معاني «أَناطَ»

وــــــــــ فُلانٌ وغَيْرُهُ الشَّيْءَ بالشَّيْءِ: عَلَّقَهُ بهِ. ﻿

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ تُبَّعٌ الأصغرُ (ت: 214ق.ه=430م) يَمْدَحُ جَدَّهُ ذا القَرْنَيْنِ:

ورَأى مَسيرَ الشَّمْسِ عِنْدَ غُروبِها
في عَيْنِ ذي خُلُبٍ وثَأْطٍ حِرْمِدِ

فلَقَدْ أَذَلَّ الصَّعْبَ صَعْبَ زَمانِهِ
‌وأَناطَ ‌قُوّةَ ‌عِزِّهِ ‌بالفَرْقَدِ

ابن هشام الحميري، التيجان في ملوك حمير. ص: 123.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الحسن الأبياريُّ (ت: 618ه=1221م):

"فإِذا ‌أَناطَ الشَّرْعُ حُكْمًا بحِكْمةٍ؛ وَجَبَ اعْتِبارُ تِلْكَ الحِكْمَةِ في تَعْدِيةِ ذَلِكَ الحُكْمِ".

الأبياري، التحقيق والبيان في شرح البرهان. تح: علي بن عبد الرحمن الجزائري، ج: 3، ص: 527.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شهابُ الدّين القَرافيُّ (ت: 684ه=1285م):

"مَقْصودُ الشّارِعِ ضَبْطُ الأَمْوالِ عَلى العِبادِ؛ لأَنَّهُ ‌أَناطَ بها مَصالِحَ دُنْياهُمْ وأُخْراهُمْ؛ فمَنَعَ لِذَلِكَ مِنْ تَسْليمِ الأَمْوالِ للسُّفَهاءِ".

القرافي، الذخيرة. تح: محمد حجي، ج: 5، ص: 240.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال ابْنُ عَرْفة الدُّسوقيّ (ت: 1230ه=1815م):

"لأَنَّ الشّارِعَ ‌أَناطَ العُشْرَ ونِصْفَهُ بالسَّقْيِ بالآلةِ وغَيْرِها".

نَوَّطَ

فعل
4.1معنى من معاني «نَوَّطَ»

* نَوَّطَ الجَمَلُ: خَرَجَ بنَحْرِهِ خُراجٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ دُكَيْنٌ الطّائي (ت: 190ه=806م):

"قَدْ ‌نَوَّطَ جَمَلُ بني فُلانٍ فماتَ، وهُوَ أَنْ يخْرُجَ بِنَحْرِهِ خُراجٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

4.2معنى من معاني «نَوَّطَ»

وــــــــــ فُلانٌ الشَّيْءَ وغَيْرَهُ: علَّقَهُ. ﻿

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَمْرو بن مَعْدي كَرِب (ت: 21ه=642م) يُعَيِّرُ قَيْسَ بنَ عَبْدِ يَغوث:

غَدَرْتَ ولَمْ تُحْسِنْ وَفاءً ولَمْ يَكُنْ
لِيَحْتَمِـلَ الأَسـْبابَ إِلَّا الـمُعَـوَّدُ

وكَيـْفَ لِقَيْـسٍ أَنْ يُنَـوِّطَ نَفْسـَهُ
إذا ما جَرى والـمَضْرَحِيُّ الـمُسَوَّدُ

الطبري، تاريخ الرسل والملوك. تح: محمد أبو الفضل، ج: 3، ص: 328.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُواسٍ (ت: 198ه=814م) يَصِفُ ضَوْءًا:

لِضَوءِ بَرْقٍ ظَلِلْتَ مُكْتَئِبًا

شَقَّ سَناهُ في الجَوِّ والْتَهَبا

يومِضُ فـي ضاحِكِ النَّواجِذِ مَحـ
ـذُوٌّ بِريحَينِ شَمْأَلٍ وصَبا

نَوَّطَ في الأُفْقِ عِبْءَ فُرَّقِهِ
وجَرَّ مِنهُ عَلى الرُّبا ذَنَبا

ديوان أبي نُواس. تح: بهجت الحديثي، ص: 62.— المصدر
4.3معنى من معاني «نَوَّطَ»

وــــــــــ فُلانٌ القِرْبةَ: أَثْقَلَها لِيَدْهُنَها. ﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

انْتاطَ

فعل

* انْتاطَتِ الدّارُ وغَيْرُها: بَعُدَتْ.

س

18.1معنى من معاني «نياط»

* النّياطُ: بُعْدُ الـمَفازَةِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ الكِنْديُّ (ت: 80ق.ه=544م) يَفْتَخِرُ بِنَفْسِهِ:

وخَرْقٍ بَعيدٍ قد قَطَعْتُ نِياطَهُ
على ذاتِ لَوْثٍ سَهْوةِ المَشْيِ مِذعانِ

البطليوسي، رسائل في اللّغة. تح: وليد محمّد السراقبيّ، ص: 140.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأعْشى الكَبير (ت: 7ه=628م) يَصِفُ ناقةً:

وجَزورِ أيْسارٍ دَعَوْتُ إلى النَّدى
‌ونِياطِ ‌مُقْفِرَةٍ ‌أَخافُ ضَلالَها

الطبري، جامع البيان. تح: التركي، ج: 3، ص: 677.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وإِنَّما قيلَ لبُعْدِ المَفازةِ: نِياطٌ؛ لأَنَّها مَنوطةٌ بِفَلاةٍ أُخْرى تَتَّصِلُ بها لا تَكادُ تَنْقَطِعُ".

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 456.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ناصيفُ اليازجي (ت: 1287ه=1870م):

"فجَلَسَ مَجْلِسَ الفَقيهِ، وأَخَذَ يَنْثُرُ اللّآلِئَ مِنْ فيهِ، حَتّى إِذا تَمادَتْ بهِ الأَشْواطُ في شُقَّةٍ بَعيدةِ النِّياطِ، تَصَدّى لَهُ رَجُلٌ قَصاقِصُ كَأَنَّهُ فَرافِصُ".

ناصيف اليازجي، مجمع البحرين. ج: 1، ص: 398.— المصدر
18.2معنى من معاني «نياط»

وــــــــــ: الفُؤادُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال ابْنُ عَبّاسٍ (ت: 68ه=688م) يُخْبِرُ نافِعَ بنَ الأَزْرَقِ عَنْ قَوْلِ اللهِ عَزَّ وجَلَّ: ﴿قَدۡ ‌شَغَفَهَا ‌حُبًّاۖ ﴾ [يوسف: 30]:

"الشِّغافُ في القَلْبِ في ‌النِّياطِ، يَقولُ: قَدِ امْتَلَأَ قَلْبُها مِنْ حُبِّ يوسُفَ".

نافع بن الأزرق، غريب القرآن. تح: محمد عبد الرحيم وآخر، ص: 186.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رؤبةُ (ت: 145ه=762م) يَسْتَعْطِفُ الحارِثَ بْنَ سُلَيْمٍ:

إِلّا تَضَعْ دَيْني فدَيْني لابِثُ

وأَنا مَجْهودُ النِّياطِ لاهِثُ

[النِّياطُ: الفُؤادُ]
مجموع أشعار العرب. تح: وليم البروسي، ج: 3، ص: 30.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ الصَّيْقل الجَزَريّ (ت: 701ه=1302م):

"قَرُمْتُ إِلى مُناهَزَةِ الابْتِهاجِ، أَيّامَ ضَيْقِ الانْتِهاجِ، وسَئِمْتُ مَرارةَ العِلاجِ، أَيّانَ انْزِعاجِ المِزاجِ، وكُنْتُ حينَئِذٍ ضَئيلَ ‌النِّياطِ، جَزيلَ الاحْتِياطِ، أَلِجُ مِنْ شِدّةِ العِياطِ، في سَمِّ الخِياطِ".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ص: 355.— المصدر
18.3معنى من معاني «نياط»

وــــــــــ مِنْ كُلِّ شَيْءٍ: مُعَلَّقُهُ.﻿

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110ه=728م) يُعَرِّضُ بأَخْوالِ سُلَيْمانَ بْنِ عَبْدِ المَلِكِ:

فسَيْفُ بَني عَبْسٍ وقَدْ ضَرَبوا بهِ
نَبا بِيَدَيْ وَرْقاءَ عَنْ رَأْسِ خالِدِ

كَذاكَ سُيوفُ الهِنْدِ تَنْبو ظُباتُها
ويَقْطَعْـنَ أَحْيانًا نِياطَ القَلائِدِ

ديوان الفرزدق. تح: علي فاعور، ص: 143.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ سيدَه (ت: 458ه=1066م):

"ونيِاطُ كُلِّ شَيْءٍ: مُعَلَّقُهُ، كنِياطِ القَوْسِ والقِرْبةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 9، ص: 240.— المصدر
18.4معنى من معاني «نياط»

وــــــــــ: عِرْقٌ غَليظٌ نيطَ بهِ القَلْبُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ الله بنُ الحَسَن الهاشِميّ (ت: 145ه=762م) يُخاطِبُ أَبا جَعْفَرٍ المَنْصورَ:

وكَيْفَ أُريدُ ذاك وأَنْتَ مِنّي
وزِنْدُكَ حينَ تَقْدَحُ مِنْ زِنادي

وكَيْفَ أُريدُ ذاك ‌وأَنْتَ ‌مِنّي
‌بِمَنْزِلةِ ‌النِّياطِ ‌مِنَ ‌الفُؤادِ

[النِّياطُ: عِرْقٌ غَليظٌ قَدْ عُلِّق بِهِ القَلْبُ مِنَ الوَتينِ]
البلاذري، أنساب الأشراف. تح: زكّار وآخر، ج: 3، ص: 310.— المصدر
18.5معنى من معاني «نياط»

وــــــــــ: كَوْكَبانِ بَيْنَهُما قَلْبُ العَقْرَبِ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ قُتَيْبَة الدِّينَوَرِيّ (ت: 276ه=889م):

"ثُمَّ القَلْبُ قَلْبُ العَقْرَبِ. وهُوَ الكَوْكَبُ الأَحْمَرُ وَراءَ الإِكْليلِ بَيْنَ كَوْكَبَيْنِ يُقالُ لَهُما: ‌النِّياطُ".

ابن قتيبة، الأنواء في مواسم العرب، ج: 1، ص: 70.— المصدر
18.6معنى من معاني «نياط»

وــــــــــ: عِرْقٌ مُسْتَبْطِنُ الصُّلْبِ تَحْتَ المَتْنِ.﻿

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 152.— المصدر

نَيِّط

اسم
19.1معنى من معاني «نَيِّط»

* النَّيِّطُ مِنَ الآبارِ: الوَسَطُ بَيْنَ الغَزيرِ والقَليلِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رَجُلٌ من الـمُسْلمين (95ه=714م) يُجيبُ عَنْ سُؤالِ الحَجّاجِ حينَ قالَ لَهُ: يا عُضَيدةُ، لَقَدْ تَخَطَّيْتَ بها مِياهًا عِذابًا، أَأَخْسَفْتَ أَمْ أَوْشَلْتَ؟

"لا واحِدَ مِنْهما، ‌ولَكِنْ ‌نَيِّطًا ‌بَيْنَ ‌الماءَيْنِ".

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 3، ص: 703.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانيُّ (ت: 650ه=1252م):

"بِئْرٌ ‌نَيِّطٌ، مِثالُ هَيِّنٍ: إِذا كانَتْ قَدْرَ قامَةٍ؛ أَيْ: وَسَطًا، ومِنْهُ حَديثُ عُضَيْدةَ: ولَكِنْ نَيِّطًا بَيْنَ الماءَيْنِ؛ أَيْ: بَيْنَ الغَزيرِ والقَليلِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الطحاوي، ج: 4، ص: 186.— المصدر

وــــــــــ: يَجْري ماؤُها مُعَلَّقًا يَنْحَدِرُ مِنْ جَوانِبِها إِلى مَجَمِّها.
