نفرج
نَفْرَجَ(فعل) نِفْراج(اسم) نِفْرِج(اسم) نِفْرِجاء(اسم) نِفْريج(اسم)

المعانى الكلّيّة

  1. كَثْرَةُ الكَلامِ
  2. الضَّعْفُ
  3. الانْكِشافُ
01 —

نَفْرَجَ

فعل
1.1معنى من معاني «نَفْرَجَ»

* نَفْرَجَ فُلانٌ: أَكْثَرَ الكَلامَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والنَّفْرَجَةُ: كَثْرَةُ الكَلامِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 235.— المصدر
02 —

نِفْراج

اسم
2.1معنى من معاني «نِفْراج»

* النِّفْراجُ مِن الرِّجالِ: الجَبانُ الضَّعيفُ. وهو بِهاءٍ أيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مِسْحَلٍ الأَعْرابيُّ (ت: 230ه=845م):

"ويُقالُ: رَجُلٌ نِفْرِجٌ، ‌ونِفْراجٌ، ونِفْرِجاءُ، إذا كانَ جَبانًا".

أبو مسحل، النوادر. تح: عزة حسن، ص: 29.— المصدر
2.2معنى من معاني «نِفْراج»

وــــــــــ: الـمُنْكَشِفُ عِنْدَ الحَرْبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَليٍّ القاليُّ (ت: 356ه=967م):

"ويُقالُ: رَجُلٌ نِفْرِجٌ، بالنّونِ والفاءِ، ونِفْرِجاءُ، ونِفْراجٌ، ونِفْرِجَةٌ؛ وهو الَّذي يَنْكَشِفُ عِنْدَ الحَرْبِ".

القالي، البارع. تح: الطعّان، ص: 669.— المصدر
2.3معنى من معاني «نِفْراج»

وــــــــــ: كَثيرُ الكَلامِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والنَّفْرَجَةُ: كَثْرَةُ الكَلامِ، وهو نِفْراجٌ ونِفْريجٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 235.— المصدر
2.4معنى من معاني «نِفْراج»

وــــــــــ: كَثيرُ انكِشافِ الفَرْجِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ الصَّقَلّيُّ (ت: 515ه=1121م):

"وعَلى نِفْعالٍ، نَحْوُ: نِفْراجٍ؛ للَّذي يَنْكَشِفُ فَرْجُهُ".

ابن القطاع، أبنية الأسماء والأفعال والمصادر. تح: عبد الدايم، ص: 203.— المصدر
03 —

نِفْرِج

اسم
3.1معنى من معاني «نِفْرِج»

* النِّفْرِجُ مِنَ الرِّجالِ: الجَبانُ الضَّعيفُ. وهو بِهاءٍ أَيْضًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُرَيْثُ بنُ زَيْدِ الخَيْلِ (ت: 60ه=680م) يَهْجو:

نِفْرِجَةُ القَلْبِ قَليلُ النَّيْلْ

يُلْقى عَلَيهِ النَّيْدُلانُ بِاللَّيْلْ

[النَّيْدُلانُ: الكابوسُ]
أبو مسحل، النوادر. تح: عزة حسن، ص: 30.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مِسْحَلٍ الأَعْرابيُّ (ت: 230ه=845م):

"ويُقالُ: رَجُلٌ نِفْرِجٌ، ‌ونِفْراجٌ، ونِفْرِجاءُ: إذا كانَ جَبانًا".

أبو مسحل، النوادر. تح: عزة حسن، ص: 29.— المصدر
3.2معنى من معاني «نِفْرِج»

وــــــــــ: مُنْكَشِفُ الفَرْجِ. وهو بِهاءٍ أيضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصاريُّ (ت: 215ه=830م):

"رَجُلٌ نِفْرِجٌ ونِفْرِجَاءُ: يَنْكَشِفُ فَرْجُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: حسين نصار، ج: 6، ص: 249.— المصدر
3.3معنى من معاني «نِفْرِج»

وــــــــــ: القَصّارُ. (ج) نَفاريجُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):
ابن سيده، المخصص. تح: خليل جفال، ج: 2، ص: 462.— المصدر
3.4معنى من معاني «نِفْرِج»

وــــــــــ: الـمُنْكَشِفُ عِنْدَ الحَرْبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَليٍّ القاليُّ (ت: 356ه=967م):

"ويُقالُ: رَجُلٌ نِفْرِجٌ، بالنّونِ والفاءِ، ونِفْرِجاءُ، ونِفْراجٌ، ونِفْرِجَةٌ؛ وهو الَّذي يَنْكَشِفُ عِنْدَ الحَرْبِ".

القالي، البارع. تح: الطعّان، ص: 669.— المصدر
3.5معنى من معاني «نِفْرِج»

وــــــــــ مِن السَّحابِ: الحافِلُ الَّذي قَدْ فَرَّغَ ماءَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والنِّفْرِجُ مِن السَّحابِ: الحافِلُ الَّذي قَدْ فَرَّغَ ماءَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 235.— المصدر
3.6معنى من معاني «نِفْرِج»

وــــــــــ: مَن لا يَكْتُمُ سِرًّا. وهو بِهاءٍ أَيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ جِنّي (ت: 392ه=1002م):

"الَّذي لا يَكْتُمُ سِرًّا، هو في مَعْنى نِفْرِجَةٍ، ومِثالُهُ: نِفْعِلَةٌ".

ابن جنّي، سرّ صناعة الإعراب. تح: حسن هنداوي، ج: 1، ص: 444.— المصدر
04 —

نِفْرِجاء

اسم
4.1معنى من معاني «نِفْرِجاء»

* النِّفْرِجاءُ: مُنْكَشِفُ الفَرْجِ.

ع

"وعلى تَعْفِلاءَ، نَحْوُ: نِفْرِجاءَ، للَّذي يَنْكَشِفُ فَرْجُهُ".

وــــــــــ مِن الرّجالِ: الـجَبانُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مِسْحَلٍ الأَعْرابيُّ (ت: 230ه=845م):

"ويُقالُ: رَجُلٌ نِفْرِجٌ، ونِفْراجٌ، ‌ونِفْرِجاءُ، إذا كانَ جَبانًا".
