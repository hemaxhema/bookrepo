النقوش العربيّة

وَرَدَ الجَذْرُ (س ب ه) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَينِها نَقْشٌ صخري سبئي اللهجة غَيرُ مُؤرَّخٍ (Ry 548/5)، وَرَدَ فيهِ الاسْمُ (سبهم Sbhm ) رُبَّما كانَ عَلَمًا على أحَدِ الأفْرادِ، فَجاءَ في النَّقْشِ"سبهم"، أيْ: "سَبْهَم".

(Müller, 1978: 131; http://dasi.cnr.it/)

النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (س ب ه):

في السُّرْيانيّة: وَرَدَ الفِعْلُ: ܫܒܗ (شْڤَه) šəḇah بِمَعْنى (ذَهَبَ عَقلُهُ مِنَ الهَرمِ) يُوافِقُ الفِعْلَ العَرَبيَّ (سَبهَ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الفِعلَ السُّريانيَّ خَضَعَ لإبدالِ السّينِ شينًا

(ܠܟܣܝܩܘܢ ܕܐܘܓܝܢ ܡܢܢܐ: ܫܒܗ)

المعانى الكلّيّة

  1. ذَهاب العَقْلِ
  2. الكِبْرُ
01 —

سَبِهَ

فعل
1.1معنى من معاني «سَبِهَ»

* سَبِهَ (بِكَسْرِ الباءِ) فُلانٌ، يَسْبَهُ (بِفَتْحِ الباءِ) سَبَهًا: ذَهَبَ عَقْلُهُ مِنْ هَرَمٍ؛ فَهُوَ سَبِهٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"السَّبَهُ: ذَهابُ العَقْلِ مِنْ هَرَمٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 10.— المصدر
02 —

سُبِهَ

فعل
2.1معنى من معاني «سُبِهَ»

* سُبِهَفُلانٌ سَبْهًا: ذَهَبَ عَقْلُهُ؛ فَهُوَ مَسْبوهٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المُفَضَّلُ الضَّبّيُّ (ت: 168هـ=785م):

"السُّباهُ: سَكْتَةٌ تأْخُذُ الإِنْسانَ يَذْهَبُ مِنْها عَقْلُهُ، وهُوَ مَسْبوهٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب. تح: أمين عبد الوهاب، وآخر، ج: 6، ص: 165.— المصدر
03 —

سُبِّهَ

فعل
3.1معنى من معاني «سُبِّهَ»

* سُبِّهَ فُلانٌ: ذَهَبَ عَقْلُهُ مِنَ الكِبَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَذْكُرُ امْرَأَةً:

قالَتْ: أُبَيْلى لي ولَمْ أُسَبَّهِ

ما السِّنُّ إِلَّا غَفْلَةُ الـمُدَلَّهِ

ديوان رؤبة. تح: البروسي، ص: 165.— المصدر
04 —

سَباه

اسم
4.1معنى من معاني «سَباه»

* السَّباهُ: الـمُضَلَّلُ مِنَ الرِّجالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والسَّباهُ: الـمُضَلَّلُ، عَلى مِثالِ كَهَامٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 3، ص: 417.— المصدر
05 —

سُباه

اسم
5.1معنى من معاني «سُباه»

* السُّباهُ: سَكْتَةٌ تَأْخُذُ الإِنْسانِ يَذْهَبُ مِنْها عَقْلُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المُفَضَّلُ الضَّبّيُّ (ت: 168هـ=785م):

"السُّباهُ: سَكْتَةٌ تَأْخُذُ الإِنْسانَ يَذْهَبُ مِنْها عَقْلُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد خفاجي وآخر، ج: 6، ص: 137.— المصدر
5.2معنى من معاني «سُباه»

وــــــــــ مِنَ النّاسِ: الذّاهِبُ العَقْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"والسُّباهُ... الذّاهِبُ العَقْلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النمل، المنتخب. تح: العمري، ج: 1، ص: 158.— المصدر
5.3معنى من معاني «سُباه»

وــــــــــ: ذَهابُ العَقْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"السُّباهُ: ذَهابُ العَقْلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 224.— المصدر
5.4معنى من معاني «سُباه»

وــــــــــ: النَّشاطُ الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"السُّباهُ:... نَشاطُ ‌الَّذي ‌كَأَنَّهُ ‌مَجْنُون".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 224.— المصدر
06 —

سَباهي

اسم
6.1معنى من معاني «سَباهي»

* السَّباهِي مِنَ الأَفْئِدَةِ: الغافِلُ إِلّا مِنَ الـمَرَحِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (231هـ=846م) مِنْ شَواهِدِ ابْنِ الأَعْرابِيِّ، يَصِفُ فَرَسًا:

ومُنْتَخَبٍ كَأَنَّ هالَةَ أُمُّهُ
سَباهِي الفُؤادِ ما يَعيشُ بِمَعْقولِ

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 224.— المصدر
6.2معنى من معاني «سَباهي»

وــــــــــ مِنَ النّاسِ: الـمُتَكَبِّرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"ويُقالُ: رَجُلٌ ... سَباهٍ... إِذا كانَ مُتَكَبِرًا".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 341.— المصدر
6.3معنى من معاني «سَباهي»

وــــــــــ: ذاهِبُ العَقْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"ورَجُلٌ... سَباهٍ: مُدَلَّهٌ ذَاهِبُ العَقْلِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 223.— المصدر
07 —

سَباهِيَة

اسم
7.1معنى من معاني «سَباهِيَة»

* السَبَاهِيَةُ مِنَ النّاسِ: الـمُتَكَبِّرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”وقالَ الشَّاعِرُ (256هـ=870م) مِنْ شَواهِدِ الأُشْنانْدانِيِّ، يَصِفُ امْرَأَةً:

سَباهِيَةٌ تَخالَ الشَّمْسَ أُمًّا
وَتَـحْسَبُ أَنَّ والِدَها ثَميرُ

تَرُدُّ عَلى الحَصى طَرْفًا كَليلًا
ونَاظِرُهُ بِما تَهْوى بَصيرُ

السخاوي، سفر السعادة. تح: الدالي، ج: 2، ص: 670.— المصدر
08 —

سَباهِيّ

اسم
8.1معنى من معاني «سَباهِيّ»

* السَّباهِيُّ مِنَ العُقولِ: الضَّعيفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللّحيانِيُّ (ت: 220هـ=835م):

"رَجُلٌ سَباهِيُّ العَقْلِ: إِذا كانَ ضَعيفَ العَقْلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: العزباوي، ج: 36، ص: 392.— المصدر
09 —

سَباهِيَّة

اسم
9.1معنى من معاني «سَباهِيَّة»

* السَّباهِيَّةُ: فِئَةُ من الخَيّالَةِ في الجَيْشِ العُثْمانِيِّ، مُهِمَّتُها حِفْظُ الأَمْنِ والـمُمْتَلَكاتِ، ويُعْفَوْنَ مُقابِلَ ذلك مِنْ كُلِّ التَّكاليفِ، الواحِدُ: سَباهِيٌّ (مع).

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ حاجي خَليفَة (ت: 1067هـ=1657م) يُتَرْجِمُ:

"(تيغي): شاعِرٌ عُثْمانِيٌّ كانَ يَنْتَسِبُ إِلى طائِفَةِ ‌السَّباهِيَّةِ في عَصْرِ السُّلْطانِ سَليمٍ الثّاني".

حاجي خليفة، سلم الوصول، محمود الأرناؤوط، ج:4، ص: 297.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عابِدينَ (ت: 1252هـ=1836م) يُشيرُ إِلى إِحْدى المَسائِلِ الفِقْهِيَّةِ:

"السَّباهِيُّ لا يَنْتَصِبُ خَصْمًا لِمُدَّعي الأَرْضِ مِلْكًا أَوْ وَقْفًا".

ابن عابدين، حاشية رد المحتار. تح: مكتبة الحلبي، ج: 5، ص: 570.— المصدر
10 —

سَبَه/سَبِه

اسم
10.1معنى من معاني «سَبَه/سَبِه»

* السَّبَهُ والسَّبِهُ مِنَ الرِّجالِ: الـمُتَكَبِّرُ، (ج) سَوابِهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَرْجِيُّ (ت: 120هـ=737م) عَلى لِسانِ امْرَأَةٍ تَذُمُّ رَجُلًا:

ولا هَناهُ حَتّى يَشُوبَ بِهِ
وُدًّا أُراهُ لِوُدِّنا دَخَلا

هُوَ الـمَلولُ الَّذي سَمِعْتِ بِهِ
ولا أُحِبُّ السَوابِهَ الـمُلُلا

ديوان العرجي. تح: الطائي وآخر، ص: 81.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"ويُقالُ: رَجُلٌ سَبِهٌ... إِذا كانَ مُتَكَبِّرًا".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 341.— المصدر

مُسَبَّه

اسم

* المُسَبَّهُ مِنَ الرِّجالِ: الفَصيحُ اللِّسانِ السَّمْحُ.

ع

ابن عباد، المحيط. تح: آل ياسين، ج: 3، ص: 417.— المصدر
