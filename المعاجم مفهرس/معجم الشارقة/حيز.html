01 —

حازَ

فعل
1.1معنى من معاني «حازَ»

* حازَ فُلانٌ يَحيزُ حَيْزًا: سارَ رُوَيْدًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الـحَيْزُ: السَّيْرُ الرُّوَيْدُ" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ).

الأزهري، التهذيب. تح: درويش، ج: 5، ص: 177.— المصدر
1.2معنى من معاني «حازَ»

وــــــــــ: سارَ سَيْرًا شَديدًا. (عَنِ الفارابيِّ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفارابِيُّ (ت: 350ه=961م):

"الـحَيْزُ: السَّيْرُ الشَّديدُ".

الفارابي، ديوان الأدب. تح: مختار، ج: 3، ص: 406.— المصدر
1.3معنى من معاني «حازَ»

وــــــــــ فُلانٌ الإِبِلَ: ساقَها سَوْقًا رُوَيْدًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الحَيْزُ: السَّيْرُ الرُّوَيْدُ، وقَدْ حِزْتُها أَحيزُها" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، التهذيب. تح: درويش، ج: 5، ص: 177.— المصدر
1.4معنى من معاني «حازَ»

وــــــــــ الشَّيْءَ: ضَمَّهُ إِلى نَفْسِهِ وجَمَعَهُ. لُغَةٌ في حاز يَحوزُ. (عَنِ الفَيّوميِّ)

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفَيّوميُّ (ت: 770ه=1368م):

"حُزْتُ الشَّيْءَ أَحوزُهُ حَوْزًا وحيازَةً: ضَمَمْتُهُ وجَمَعْتُهُ، وكُلُّ مَنْ ضَمَّ إِلى نَفْسِهِ شَيْئًا فقَدْ حازَهُ. وحازَهُ حَيْزًا -مِنْ بَابِ سارَ- لُغَةٌ فيهِ"

الفيومي، المصباح المنير.، ج: 1، ص: 156.— المصدر
1.5معنى من معاني «حازَ»

وــــــــــ: ساقَها سَوْقًا شَديدًا. (ضِدٌّ)

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفَيْروزابادِيُّ (ت: 817هـ=1414م):

"الـحَيْزُ: السَّوْقُ الشَّديدُ" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الفيروزآبادي، القاموس. تح: العرقسوسي، ج: 1، ص: 510.— المصدر
02 —

تَحَيَّز

فعل
2.1معنى من معاني «تَحَيَّز»

تَحَيَّزَ الشَّيْءُ: جاوَزَ ما حَوْلَهُ وبَرَزَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ النّابِغَةُ الذُّبْيانِيُّ (ت: 18ق.هـ=605م) يَصِفُ فَرْجَ امْرَأَةٍ:

وإذا لَمَسْتَ لَمَسْتَ أَجْثَمَ جاثِمًا
‌مُتَحَيِّزًا ‌بِمَكانِهِ مِلْءَ اليَدِ

ديوان النّابغة الذبيانيّ.، تح: محمد أبو الفضل، ج: 1، ص: 96.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ قُتَيْبَةَ (ت: 276ه=889م) يَشْرَحُ بَيْتِ النّابِغَةِ الذُّبْيانيِّ:

"أَيْ: هُوَ مُنْبَسِطٌ عَريضٌ في ارْتِفاعٍ، مُتَحَيِّزٌ قَدْ مَلَأَ مَكانَهُ، لا جِهَةَ لَهُ يَمْضي فيها" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، كتاب المعاني الكبير. ص: 510.— المصدر
2.2معنى من معاني «تَحَيَّز»

وــــــــــ الشَّيْءُ: تَفَرَّقَ وتَمَيَّزَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو ذُؤَيْبٍ الهُذَليُّ (ت: 27هـ=648م) يَصِفُ ما فَعَلَهُ مُشْتارُ العَسَلِ بالنَّحْلِ:

فلمّا جَلاها ‌بِالإيامِ ‌تَحَيَّزَتْ
ثُباتٍ عَلَيْها ذُلُّها واكْتِئابُها

ابن قتيبة، كتاب المعاني الكبير. ص: 619.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ قُتَيْبَةَ (ت: 276ه=889م) يَشْرَحُ مَعْنى (تَحَيَّزَتْ) في بَيْتِ أَبي ذُؤَيْبٍ:

"تَحَيَّزَتْ: انْحازَتْ وتَمَيَّزَتْ قِطَعًا قِطَعًا" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، كتاب المعاني الكبير. ص: 619.— المصدر
2.3معنى من معاني «تَحَيَّز»

وــــــــــ فُلانٌ وغَيْرُهُ: تَلَوّى وتَقَلَّبَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ القُطامِيُّ (ت: 101ه=719م) يَصِفُ عَجوزًا تَخْشى أَنْ يَنْزِلَ الشّاعِرُ عَلَيْها ضَيْفًا:

‌تَحَيَّزُ ‌مِنّي خَشْيَةً أَنْ أَضيفَها
كَما انْحازَتِ الأَفْعى مَخافَة ضارِبِ

الجوهري، الصحاح. تح: أحمد عطار، ج: 3، ص: 875.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّنَوْبَريُّ (ت: 334ه=946م) يَهْجو سَعيدَ بْنَ مُحْرِزٍ:

كَمْ راغَ مِنْ رَوْغَةٍ فأَدْرَكَهُ
حَدٌّ كَحَدِّ الـمِهْمازِ يَهْمِزُهُتَحَيُّزَ الكَلْبِ حينَ أَخْسَأَهُ
كَأَنَّهُ نافِعٌ تَحَيُّزُهُ

ديوان الصّنوبريّ.، تح: إحسان عبّاس، ج: 1، ص: 133.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (711ه=1311م):

"والتَّحَيُّزُ والتَّحَوُّزُ: التَّلَوّي والتَّقَلُّبُ. وخَصَّ بَعْضُهُمْ بِهِ الـحَيَّةَ، يُقالُ: ‌تَحَوَّزَتِ ‌الـحَيَّةُ ‌وتَحَيَّزَتْ؛ أَيْ: تَلَوَّتْ".

ابن منظور، لسان العرب.، تح: أمين محمد، وآخر، ج: 3، ص: 388.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الطّوَيْرانيُّ (1300=1882م، تاريخُ الطَّبْع) يَصِفُ حالَ الإنْسانِ:

والمَرْءُ بَيْنَ حَياتِهِ ومَماتِهِ
مَتَحَيِّرٌ مَتَحَيِّزٌ مُتَقَلِّبُ

الطويراني، ثمرات الحياة. مطبعة إدارة الوطن، ص: 356.— المصدر
2.4معنى من معاني «تَحَيَّز»

وــــــــــ: أَرادَ القِيامَ فأَبْطَأَ عَلَيْهِ ذلِكَ. (انْظُرْ: ح و ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَهْ (ت: 458هـ=1066م):

"وتَحَيَّزَ الرَّجُلُ: أَرادَ القيامَ فأَبْطَأَ ذلِكَ عَلَيْهِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 3، ص: 426.— المصدر
2.5معنى من معاني «تَحَيَّز»

وــــــــــ إلى القَوْمِ: انْضَمَّ إلَيْهِمْ ووافَقَهُمُ الرَّأْيَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11ه=632م) قال الله ﷻ:

﴿وَمَن يُوَلِّهِمۡ يَوۡمَئِذ دُبُرَهُۥٓ إِلَّا مُتَحَرِّفا لِّقِتَالٍ أَوۡ ‌مُتَحَيِّزًا إِلَىٰ فِئَة فَقَدۡ بَآءَ بِغَضَب مِّنَ ٱللَّهِ وَمَأۡوَىٰهُ جَهَنَّمُۖ وَبِئۡسَ ٱلۡمَصِيرُ﴾ (الأنفال: 16)﴾.

”وقالَ عُمرُ بنُ الخَطّابِ (13ه=634م) حينَ لَقيَ أبو عبيدِ بْنُ مَسْعودٍ الثَّقَفيُّ حَتْفَهُ في وَقْعَةِ القَرْقَسِ (الجِسْرِ):

"عِبادَ اللَّهِ، اللَّهُمَّ إِنَّ كُلَّ مُسْلِمٍ في حِلٍّ مِنّي، أَنا فِئَةُ كُلِّ مُسْلِمٍ، يَرْحَمُ اللَّهُ أَبا عُبَيْدٍ، لَوْ كانَ عَبَرَ فاعْتَصَمَ بِالخَيْفِ، ‌أَوْ ‌تَحَيَّزَ ‌إِلَيْنا ولَمْ يَسْتَقْتِلْ، لَكُنّا لَهُ فِئَةً".

الطبري، تاريخ الرسل والملوك. تح: محمد أبو الفضل، ج: 3، ص: 455.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّافِعِيُّ (ت: 204ه=819م) يُحَدِّثُ عَنِ النّاجينَ مِنْ سَخَطِ اللهِ ﷻ:

"فإنْ كانَ اللَّهُ عَزَّ وَجَلَّ يَعْلَمُ أَنَّهُ إنَّما تَحَرَّفَ لِيَعودَ ‌لِلْقِتالِ، ‌أَوْ ‌تَحَيَّزَ لِذلِكَ، فهُوَ الَّذي اسْتَثْنى اللَّهُ فأَخْرَجَهُ مِنْ سَخَطِهِ في التَّحَرُّفِ والتَّحَيُّزِ".

الشافعي، الأم. تح: محمد النجار، ج: 4، ص: 257.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَديمِ (ت: 660ه=1262م) يُحَدِّثُ عَنِ الخَوارِزْميَّةِ:

"وساروا إِلى شَيْزَرَ، وتَـحَيَّزَ أَهْلُها إلى الـمَدينَةِ الَّتي تَحْتَ القَلْعَةِ".

ابن العديم، زبدة الحلب من تاريخ حلب. ج: 1، ص: 506.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ السَّلاويُّ (ت: 1315هـ=1897م):

"واجْتَمَعَ نَحْوُ العِشْرينَ مِنْ كِبارِ الشَّرارِدَةِ وتَقَدَّموا إلى السُّلْطانِ،‌ فقالوا: يا مَوْلانا، تَحَيَّزْ إِلَيْنا لِئَلّا تُصيبَكَ العامَّةُ".

الناصري، الاستقصا.، تح: جعفر الناصري وآخر، ج: 8، ص: 162.— المصدر
03 —

حِيازَة

اسم
3.1معنى من معاني «حِيازَة»

* الـحِيازَةُ: تَمَلُّكُ الشَّيْءِ والتَّسَلُّطُ عَلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ قَتادَةُ بْنُ دِعامَةَ (ت: 117ه=736م):

"لَيْسَ بَيْنَ الرَّجُلِ وامْرَأَتِهِ ‌حِيازَةٌ إذا وَهَبَتْ لَهُ أَوْ وَهَبَ لَهَا".

مصنف عبد الرزاق. دار التأصيل، ج: 7، ص: 375.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مالِكُ بْنُ أَنَسٍ (ت: 179ه=795م) في تَبْويبٍ لَهُ في كِتابِهِ المُوَطَّأِ:

"بابُ الوَصِيَّةِ لِلْوارِثِ ‌والحِيازَةِ".

موطأ مالك. تح: الأعظمي، ج: 4، ص: 1110.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بْنُ عَبْدِ السَّلامِ (ت: 660ه=1262م):

"وتَرِقُّ الـمَرْأَةُ والصَّبِيُّ والـمَجْنونُ ‌بالحِيازَةِ وانْتِهاءِ الـحَرْبِ، ولَيْسَ للإمامِ قَتْلُهُمْ، ولا قَتْلُ العَبْدِ".

العز بن عبد السلام، الغاية. تح: الطبّاع، ج: 5، ص: 34.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَرَفَةَ الدُّسوقيُّ (ت: 1230هـ=1815م):

"ومَحَلُّ الِاحْتِياجِ لِبَيِّنَةِ ‌الحِيازَةِ إذا كانَتْ بَيِّنَةُ الـمِلْكِ شَهِدَتْ بِأَنَّ لَهُ دارًا بِمَحَلِّ كَذا، ولَمْ تَذْكُرْ حُدودَها ولا جيرانَها عَلى وَجْهِ الشَّهادَةِ بِهِ، وإلّا فلا يُحْتَاجُ لِبَيِّنَةِ ‌الحِيازَةِ".

الدسوقي، الحاشية على الشرح الكبير. دار الفكر، ج: 2، ص: 521.— المصدر
04 —

حَيْزِ

اسم
4.1معنى من معاني «حَيْزِ»

* حَيْزِ: اسْمُ صَوْتٍ لِزَجْرِ الـمِعْزى، وقيلَ: لزَجْرِ الحِمارِ أَيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (244ه=858م، مِنْ شَواهِدِ ابْنِ السِّكّيتِ):

شَمْطاءُ جاءَتْ مِنْ أَعالي البَرِّ
قَدْ تَرَكَتْ حَيْزِ وقالَتْ حَرِّ

ابن سيده، المخصص.، تح: خليل جفال، ج: 2، ص: 247.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظُورٍ (ت: 711هـ=1311م):

"وحَيْزِ حَيْزِ: مِنْ زَجْرِ الـمِعْزى". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، اللسان. دار صادر، ج: 5، ص: 343.— المصدر
05 —

حَيْز

اسم
5.1معنى من معاني «حَيْز»

* الحَيْزُ: كُلُّ ناحيَةٍ عَلى حِدَةٍ. (ج) أَحْيازٌ.

ع

الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الآلوسيّ (ت: 1270ه=1854م) يَتَحَدَّثُ عَنْ حَرَكَةِ الأَرْضِ:

"أمّا السُّكونُ فلِأَنَّ ‌الجِسْمَ ‌الحاصِلَ ‌في ‌الـحَيِّزِ الطَّبيعيِّ لا يَتَحَرَّكُ حَرَكَةً طَبيعيَّةً آنيَةً؛ لاسْتِلْزامِها الخروجَ عَنِ الـحَيِّزِ الطَّبيعيِّ، ولا يُتَصَوَّرُ مِنَ الأَرْضِ الحَرَكَةُ الإِراديَّةُ؛ لِكَوْنِها عَديمَةَ الشُّعورِ".

الآلوسيّ، روح المعاني. تح: زهير القاسم، ج: 14، ص: 60.— المصدر
