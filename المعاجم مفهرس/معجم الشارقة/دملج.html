المعانى الكلّيّة

01 —

دَمْلَجَ

فعل
1.1معنى من معاني «دَمْلَجَ»

* دَمْلَجَ فُلانٌ الشَّيْءَ دَمْلَجَةً، ودِمْلاجًا: ضَمَّهُ، وسَوّاهُ، وأَحْسَنَ صَنْعَتَهُ. (انظر: د م ل ق، د م ل ك).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ زُرارَةُ بْنُ عَمْرٍو النَّخَعِيُّ (11هـ=632م) يُخْبِرُ رَسولَ اللهِ ﷺ بِما رَآهُ في رِحْلَتِهِ إِلَيْهِ:

"يا رَسولَ اللهِ، ورَأَيْتُ النُّعْمانَ بْنَ الـمُنْذِرِ عَلَيْهِ قُرْطانِ مُدَمْلَجانِ ومَسْكَتانِ".

ابن قيم الجوزية، زاد المعاد، تح: الأرنؤوط وآخر، ج: 3، ص: 600.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ ما أَصابَهُ بِسَبَبِ كِبَرِ سِنِّهِ:

قَدْ عَجِبَتْ نَضْرَةُ مِنْ تَهْداجي

مُخْتَضِعًا أَهُمُّ بِالهِمْلاجِ

إِذْ رَقَّ بَعْدَ مُدْمَجِ الإِدْماجِ

وَدُمْلُجِيٍّ حَسَنِ الدِّمْلاجِ

ديوان رؤبة، تح: البروسي، ص: 30.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَعْتوقٍ الموسَوِيُّ (ت: 1087هـ=1676م) يَمْدَحُ:

ولَكُمْ بِهِ شَمْسٌ تَقَلَّدَ جيدُها
شُهُبًا وبَدْرٌ بِالهِلالِ مُدَمْلَجُ

ديوان ابن معتوق، تح: الشرتوني، ص: 118.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ الطُّلْبَة الشِّنْقيطِيُّ (ت: 1272هـ=1856م) يَتَغَزَّلُ:

رَمَتْني بِوَضّاحٍ ظِماءٍ عُمورُهُ
بَرودِ الثَّنايا ذي غُروبٍ مُفَلَّجِ

وكَشْحٍ لَطيفٍ كَالجَدائِلِ طَيُّهُ
كَلَمْسِ الدِّمَقْسِ ذاتُ خَلْقٍ مُعَذْلَجِ

وتُشْجي رَحيباتِ الدَّمالِجِ والبُرى
بِما شِئْتَ مِنْ غَيْلٍ رَواءٍ مُدَمْلَجِ

ديوان محمد بن الطلبة، تح: بن أبوه، ص: 184.— المصدر
02 —

دُمْلِجَ

فعل
2.1معنى من معاني «دُمْلِجَ»

* دُمْلِجَ الجِسْمُ: طُوِيَ واكْتُنِزَ لَحْمُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَجْنونُ لَيْلى (ت: 68هـ=688م) يَتَغَزَّلُ:

مُدَمْلَجَةُ السّاقَيْنِ بَضٌّ بَضيضَةٌ
مُفَلَّجَةُ الأَنْيابِ مَصْقولَةُ الخَمْرِ

ديوان قيس بن الملوح، تح: يسري عبد الغني، ص: 62.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ حَمْديسَ (ت: 527هـ=1133م) يَتَغَزَّلُ:

لَقَدْ أَسْقَتِ الأَضْدادُ مِنْكِ مَلاحَةً
فَتًى روحُهُ في الحُبِّ غَيْرُ مُرَوَّحِ

سَخاءٌ بِهَجْرٍ مِنْ سَمينٍ مُدَمْلَجٍ
وشُحٌّ بِوَصْلٍ مِنْ هَزيلٍ مُوَشَّحِ

ديوان ابن حمديس، تح: إحسان عباس، ص: 109.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ هُتَيْمِلٍ (ت: 696هـ=1297م) يَتَغَزَّلُ:

ومُدَمْلَجٍ حَضَرَ الرَّقيبُ وَلَمْ يَغِبْ
عَنْهُ، فَكانَ عَنِ الزِّيارَةِ عائِقا

ديوان ابن هتيمل، تح: العقيلي، ص: 672.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ زَيْنُ العابِدينَ الكُوَيْتِيُّ (ت: 1369هـ=1950م) يَتَغَزَّلُ:

مُدَمْلَجَةُ السّاقَيْنِ، تُرْكِيَّةُ القَفا
مُقَرْطَفَةُ الأُذْنَيْنِ، تُبْدي الـمَكارِما

زين العابدين الكويتي، ديوان كوكبة السعودية، تح: الغنيم، ص: 144.— المصدر
2.2معنى من معاني «دُمْلِجَ»

وــــــــــ الشَّيْءُ: صارَ مُدْرَجًا أَمْلَسَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (393هـ=1003م، مِنْ شَواهِدِ الجَوْهَرِيِّ):

كَأَنَّ مِنْها القَصَبَ الـمُدَمْلَجا

سوقٌ مِنَ البَرْدِيِّ ما تَعَوَّجا

الجوهري، الصحاح، تح: عطار، ج: 1، ص: 216.— المصدر
03 —

دُمْلَج، دُمْلُج

اسم
3.1معنى من معاني «دُمْلَج، دُمْلُج»

* الدُّمْلَجُ، والدُّمْلُجُ: الـمِعْضَدُ؛ وهُوَ ما يُحيطُ بِالعَضُدِ مِنَ الحُلِيِّ، (ج) دَمالِجُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الفِنْد الزِّمّانِيُّ (ت: 90ق هـ=530م) يُخاطِبُ عاذِلَةً:

أَيا تَمْلِكُ يا تَمْلي
ذاتُ الدَّلِّ والشَّكْلِ

وذاتُ الطَّوْقِ والدُّمْلُـ
ـجِ والتَّقْصارِ والحِجْلِ

ذَريني وذَري عَذْلي
فَإِنَّ العَذْلَ كَالقَتْلِ

شعر الفند الزماني، تح: الضامن، ص: 22.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الشَّمّاخُ (ت: 22هـ=643م) يَتَغَزَّلُ:

هَضيمُ الحَشا لا يَمْلَأُ الكَفَّ خَصْرُها
ويُمْلَأُ مِنْها كُلُّ حِجْلٍ ودُمْلُجِ

ديوان الشماخ، تح: صلاح الدين الهادي، ص: 75.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَتَغَزَّلُ:

قَدْ عَجِبَتْ نَضْرَةُ مِنْ تَهْداجي

مُخْتَضِعًا أَهُمُّ بِالهِمْلاجِ

إِذْ رَقَّ بَعْدَ مُدْمَجِ الإِدْماجِ

ودُمْلُجِيٍّ حَسَنِ الدِّمْلاجِ

ديوان رؤبة، تح: البروسي، ص: 30.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّارِ (ت: 658هـ=1260م) يَتَغَزَّلُ:

لَمْ يَزِنْ دُمْلُجُها مِعْصَمَها
ذَلِكَ المِعْصَمُ زانَ الدُّمْلُجا

ديوان ابن الأبار، تح: الهراس، ص: 115.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ الطُّلْبَةِ الشِّنْقيطِيُّ (ت: 1272هـ=1856م) يَصِفُ السِّمْطَ:

عَلَيْها سُموطٌ مِنْ مَحالٍ مُلَوَّبٍ
مِنَ التِّبْرِ أَوْ مِنْ لُؤْلُؤٍ وزَبَرْجَدِ

يُفَصَّلُ بِالـمَرْجانِ والشَّذْرِ بَيْنَهُ
وقَدْ غَصَّ مِنْها كُلُّ حِجْلٍ ودُمْلُجِ

ديوان محمد بن الطلبة، تح: بن أبوه، ص: 157.— المصدر
04 —

دُمْلُج

اسم
4.1معنى من معاني «دُمْلُج»

* الدُّمْلُجُ: الحَجَرُ الأَمْلَسُ، (ج) دَمالِجُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَبّانُ (ت: 416هـ=1025م):

"والدُّمْلُجُ والدُّمْلوجُ: الحَجَرُ الأَمْلَسُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

المديني، المجموع المغيث، تح: العزباوي، ج: 1، ص: 673.— المصدر
05 —

دُمْلوج

اسم
5.1معنى من معاني «دُمْلوج»

*الدُّمْلوجُ: المِعْضَدُ؛ وهُوَ ما يُحيطُ بِالعَضُدِ مِنَ الحُلِيِّ، (ج): دَماليجُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طَرَفَةُ بْنُ العَبْدِ (ت: 60ق هـ=564م) يَتَغَزَّلُ:

كَأَنَّ البُرينَ وَالدَّماليجَ عُلِّقَتْ
عَلى عُشَرٍ، أَوْ خِرْوَعٍ لَمْ يُخَضَّدِ

ديوان طرفة، تح: درية الخطيب وآخر، ص: 47.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37هـ=658م) يَتَغَزَّلُ:

وبيضٍ مَباهيجٍ كَأَنَّ خُدودَها
خُدودُ مَهًا آلَفْنَ مِن عالِجٍ هَجْلا

ثِقالِ الخُطى، غيدِ السَّوالفِ لَمْ تَقُمْ
عَلى الخَسْفِ يَمْلَأْنَ الدَّماليجَ والحَجْلا

ديوان ابن مقبل، تح: عزة حسن، ص: 158.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو الشَّيْصِ الخُزاعِيُّ (ت: 196هـ=812م) يَتَغَزَّلُ:

لَوْلا التَّمَنْطُقُ والسِّوارُ مَعًا
والحِجْلُ والدُّمْلوجُ في العَضْدِ

لَتَزايَلَتْ مِنْ كُلِّ ناحِيَةٍ
لَكِنْ جُعِلْنَ لَها عَلى عَمْدِ

ديوان أبي الشيص، تح: الجبوري، ص: 53.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ هُتَيْمِلٍ (ت: 696هـ=1297م) يَتَغَزَّلُ:

مُرْهَفٌ مَجْرى الوِشاحِ مُفْعَمَةٌ
مِنْهُ الدَّماليجُ وَالخَلاخيلُ

ديوان ابن هتيمل، تح: العقيلي، ص: 730.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد مُحَرَّم (ت: 1364هـ=1945م) يَذْكُرُ تَبَرُّعَ إِحْدى الصَّحابِيّاتِ بِحُلِيِّها في غَزْوَةِ تَبوكَ:

فَما أَمْسَكَ البُخْلُ دُمْلوجَها
ولا مَلَكَ الحِرْصُ خَلْخالَها

أحمد محرم، ديوان مجد الإسلام، ص: 320.— المصدر
5.2معنى من معاني «دُمْلوج»

وــــــــــ: الأَرْضُ الصُّلْبَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"الدَّماليجُ: الأَرَضونَ الصِّلابُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: محمد أبي الفضل إبراهيم، ج: 11، ص: 252.— المصدر
5.3معنى من معاني «دُمْلوج»

وــــــــــ: الحَجَرُ الأَمْلَسُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَبّانُ (ت: 416هـ=1025م):

"والدُّمْلُجُ والدُّمْلوجُ: الحَجَرُ الأَمْلَسُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

المديني، المجموع المغيث، تح: العزباوي، ج: 1، ص: 673.— المصدر
5.4معنى من معاني «دُمْلوج»

o الدُّمْلوجُ الرُّسوبِيُّ (في الجُيولوجْيا): صَخْرٌ رُسوبِيٌّ يَتَكَوَّنُ مِنْ كُسارَةٍ صَخْرِيَّةٍ في حَجْمِ الحَصى أَوِ الجَمَراتِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1425هـ=2004م، تاريخُ الطَّبْعِ):

"والدُّمْلوجُ الرُّسوبِيُّ (في الجُيولوجْيا) ...: صَخْرٌ رُسوبيٌّ، يَتَكَوَّنُ مِنْ كُسارَةٍ صَخْرِيَّةٍ في حَجْمِ الحَصى أَوِ الجَمَراتِ، وأَحْيانًا في حَجْمِ الجَلاميدِ، مُتَلاحِمٌ بَعْضُها مَعَ بَعْضٍ بِمادَّةٍ لاحِمَةٍ كَالسِّليكا أَوْ مُرَكَّباتِ الحَديدِ. ويَتَمَيَّزُ هَذا الصَّخْرُ بِأَنَّ مُكَوِّناتِهِ تَكونُ مُسْتَديرَةَ الشَّكْلِ، ومُهَذَّبَةَ الأَطْرافِ. وبَعْضُ أَنْواعِهِ تُعْطي بِالصَّقْلِ مَناظِرَ جَميلَةً تُسْتَعْمَلُ في أَغْراضِ الزّينَةِ ونَحْتِ الأَواني والأَدَواتِ. ويُسَمّى أَيْضًا: الرَّصيصَ".

مجمع القاهرة، المعجم الكبير، ج: 7، ص: 552.— المصدر
5.5معنى من معاني «دُمْلوج»

o الدُّمْلوجُ النَّارِيُّ (في الجُيولوجْيا): كُتْلَةٌ مِنْ فُتاتِ الحَجَرِ الجَرْشِيِّ، أَوْ عَرْمَةٍ مِنَ الحُمَمِ، تَصْدُرُ عَنْ هِياجٍ بُرْكانِيٍّ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1425هـ=2004م، تاريخُ الطَّبْعِ):

"الدُّمْلوجُ النّارِيُّ (في الجُيولوجْيا)...:‏كُتْلَةٌ مِنْ فُتاتِ الحَجَرِ الجَرْشِيِّ، أَوْ عَرْمَةٍ مِنَ الحُمَمِ (الـمَوادِّ الـمَصْهورَةِ)، تَصْدُرُ عَنْ هِياجٍ بُرْكانِيٍّ. وهَذا الفُتاتُ يَكونُ في العادَةِ مَزْوِيًّا (ذا زَوايا)، ويَخْتَلِفُ عَنِ الرَّمادِ البُرْكانِيِّ بِكِبَرِ حَجْمِهِ وضَخامَتِهِ".

مجمع القاهرة، المعجم الكبير، ج: 7، ص: 552.— المصدر
