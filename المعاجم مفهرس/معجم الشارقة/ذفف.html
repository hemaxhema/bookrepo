01 —

ذَفَّ

فعل
1.1معنى من معاني «ذَفَّ»

* ذَفَّ (كَضَرَبَ) الشَّيءُ يَذِفُّ (بِكَسرِ الذّالِ) ذَفًّا، وذَفيفًا، وذَفافَةً: أَسْرَعَ وخَفَّ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيدٍ (ت: 80ق ه=544م) يَصِفُ جَوادًا:

يَسَرٌ في القِيادِ نَهْدٌ ذَفيفُ الـ
ـعَدْوِ عَبْلُ الشَّوى أَمينُ العُراقِ

ديوان عدي بن زيد. تح: المعيبد، ص: 153.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كَعْبُ بْنُ مالِكٍ (5ه=626م) يَتَوَعَّدُ المُشْرِكينَ يَوْمَ الخَنْدَقِ:

نُصَبِّحُكُمْ بِكُلِّ أَخي حُروبٍ
وكُلِّ مُطَهَّمٍ سَلِسِ القِيادِ

وكُلِّ طِمَّرَةٍ خَفِقٍ حَشاها
تَذِفُّ ذَفيفَ صَفْراءِ الجَرادِ

ديوان كعب بن مالك. تح: العاني، ص: 193.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبةُ بْنُ العَجّاجِ (ت: 145ه=762م):

تَرْكَبُ قُطْرَيْ ‌وَثَبى ‌ذَفوفِ

ابن سيده، المخصص. تح: جفال، ج: 4، ص: 489.— المصدر
1.2معنى من معاني «ذَفَّ»

وــــــــــ الأَمْرُ ذَفيفًا، وذَفافَةً: أَمْكَنَ وتَيَسَّرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"خُذْ ما ذَفَّ لَكَ، ودَفَّ لَكَ، أيْ: خُذْ ما تَيَسَّرَ لَكَ وتَهَيَّأَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، العباب الزاخر-الفاء. تح: آل ياسين، ص: 198.— المصدر
1.3معنى من معاني «ذَفَّ»

وــــــــــ الغَنَمُ ذَفًّا: أَصابَها داءٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وذَفَّتِ الغَنَمُ ذَفًّا: أَصابَها داءٌ".

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 61.— المصدر
1.4معنى من معاني «ذَفَّ»

وــــــــــ على فُلانٍ ذَفًّا، وذَفافًا، وذَفَفًا: أَجْهَزَ عَلَيْهِ وأَتَمَّ قَتْلَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه= 632م):

«سُلِّطَ عَلَيْهِم آخِرَ الزَّمانِ مَوْتُ طاعونٍ ذَفيفٍ يُحَرِّفُ القُلوبَ».

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 10.— المصدر
”وقالَ عَليُّ بْنُ أَبي طالِبٍ (36ه=656م) بَعْدَ وَقْعَةِ الجَمَلِ:

"وأَمَرْتُ أَنْ لا يُذَفَّ على جَريحٍ".

ابن عبد ربه، العقد الفريد. تح: الترحيني، ج: 5، ص: 78.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عُبَيْدٍ (ت: 224ه=838م):

"الذَّفيفُ هو المُجْهِزُ الذي يُذَفِّفُ عَلَيْهِمِ فيَقْتُلُهُمْ كما يُذَفَّفُ على الجَريحِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، غريب الحديث. تح: خان، ج: 4، ص: 498.— المصدر
02 —

أَذَفَّ

فعل
2.1معنى من معاني «أَذَفَّ»

*أَذَفَّفُلانٌ الجَريحَ، و عَلَيْهِ: أَجْهَزَ عَلَيْهِ وأَتَمَّ قَتْلَه.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤبةُ بْنُ العَجّاجِ (ت: 145ه=762م):

قَدْ كانَ جَمًّا شاؤُهُ ونَعَمُهُ

فعَضَّهُ دَهْرٌ مُذِفٌّ مَحْطِمُهُ

البغدادي، خزانة الأدب. تح: هارون، ج: 4، ص: 454.— المصدر
03 —

ذافَّ

فعل
3.1معنى من معاني «ذافَّ»

* ذافَّفُلانٌ الجَريحَ، و عَلَيْهِ، ولَهُ: أَجْهَزَ عَلَيْهِ وأَتَمَّ قَتْلَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (2ه= 623م) وقَدْ سُئِلَ مَنْ قَتَلَ أَبا جَهْلٍ مَع ابْنَيْ عَفْراءَ:

«المَلائِكَةُ، وذافَّ ابْنُ مَسْعودٍ، فكُلٌّ شَرِكَ في قَتْلِهِ».

مغازي الواقدي. تح: م. جونس، ج: 1، ص: 91.— المصدر
”وقالَ خالِدُ بْنُ الوَليدِ (8ه= 629م) في غَزْوَةِ بَني جَذيمَةَ:

"مَنْ كانَ مَعَهُ أَسيرٌ ‌فلْيُذافَّهُ".

مغازي الواقدي. تح: م. جونس، ج: 3، ص: 877.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"ذفَّفَهُ بِالسَّيْفِ، ‌وذافَّ لَهُ، وذافَّهُ: إذا أَجْهَزَ عَلَيْهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: يعقوب عبد النبي، ج: 14، ص: 411.— المصدر
04 —

ذَفَّفَ

فعل
4.1معنى من معاني «ذَفَّفَ»

* ذَفَّفَتِالدَّوابُّ: أَسْرَعَتْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

"وإنْ ذَفَّفَتْ بِهِمُ الهَماليجُ".

ابن الأثير، النهاية. تح: علي الحلبي، ص: 327.— المصدر
4.2معنى من معاني «ذَفَّفَ»

وــــــــــ فُلانٌ في الأَمْرِ: أَسْرَعَ وخَفَّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُريدٍ (ت: 321ه=933م):

"فأَمّا الذَّفُّ فهْوَ السُّرْعَةُ في كُلِّ ما أَخَذَ فيه، ذَفَّ في أَمْرِه، ‌وذَفَّفَ فيه". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 117.— المصدر
4.3معنى من معاني «ذَفَّفَ»

وــــــــــ فُلانًا، وعَلَيْهِ: أَجْهَزَ عَلَيْه وأتمَّ قَتلَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه= 632م) يُبَيِّنُ حُكْمَ مَنْ بَغى مِنْ أُمَّتِهِ:

«فإنَّ حُكْمَ اللهِ فيهِمْ أَنْ لا يُتْبَعَ مُدْبِرُهُمْ، ولا يُقْتَلَ أَسيرُهُمْ، ولا يُذَفَّفَ على جَريحِهِمْ».

مستدرك الحاكم. تح: عطا، ج: 2، ص: 168.— المصدر
”وقالَ ابنُ مَسْعودٍ (ت: 32ه=653م) يَصِفُ قَتْلَ أَبي جَهْلٍ:

"فذَفَّفْتُ على أَبي جَهْلٍ".

ابن الأثير، النهاية. تح: علي الحلبي، ص: 327.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ يونُسُ بْنُ حَبيبٍ (ت: 182ه=798م):

"ذَفّفَه بالسَّيْفِ، وذافّهُ وذَفَّهُ، ‌وذَفَّفَ عَلَيْهِ، إذا أَجْهَزَ، أَيْ: قَتَلَه".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 1282.— المصدر
”وقالَ القاضي أَبو يوسُفَ (ت: 182ه=798م) يُخبِرُ عَنْ عَلِيِّ بْنِ أَبِي طالِبٍ في حَربِهِ:

"ولَمْ يَقْتُلْ مِنْهُم أَسيرًا، ولَمْ يُذَفِّفْ على جَريحٍ، ولَمْ يَتْبَعْ مُدْبِرًا".

القاضي أبو يوسف، الخراج. تح: طه سعد وآخر، ص: 233.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بنُ عَبْدِ السَّلامِ (ت: 660ه=1262م):

"ولا يَحِلُّ لنا القَتْلُ إنْ أَمْكَنَ الأَسْرُ، ولا ‌التَّذْفيفُ إن أَمْكَنَ الإثْخانُ".

ابن عبد السلام، الغاية. تح: الطبّاع، ج: 7، ص: 55.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ نوَوي الجاويّ (ت: 1316ه=1898م) في الرَّجُلَيْنِ يُتَّهَمانِ في قَتْلِ رَجُلٍ:

"وإنْ كانَ أَحَدُهما مُذَفِّفًا دُونَ الآخَرِ فالمُذَفِّفُ ‌هو ‌القاتِل ‌فلا ‌يُقْتَلُ الآخَرُ".

نووي الجاوي، نهاية الزين. دار الفكر، ص: 340.— المصدر
4.4معنى من معاني «ذَفَّفَ»

وــــــــــ الشَّيْءَ: خَفَّفَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عبّادٍ (ت: 385ه=995م):

"وذَفِّفْ جَهازَ راحِلَتِكَ، أَيْ: خَفِّفْ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 61.— المصدر
4.5معنى من معاني «ذَفَّفَ»

وــــــــــ السَّهْمَ: قَزَّعَهُ فَصارَ خَفيفًا سَريعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وسَهْمٌ ‌مُذَفَّفٌ، أيْ: مُقَزَّعٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج:10، ص: 61.— المصدر
05 —

اسْتَذَفَّ

فعل
5.1معنى من معاني «اسْتَذَفَّ»

* اسْتَذَفَّالأَمْرُ: تَهَيَّأَ وتَيَسَّرَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتِ الخَنْساءُ (10 ق ه=612م) تَرْثي صَخْرًا:

شُدّوا المَآزِرَ حَتّى يَسْتَذِفَّ لَكُم
وشَمِّروا إنَّها أَيّامُ تَشْمارِ

ديوان الخنساء. تح: أنور أبو سويلم، ص: 297.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (ت: 40ه=660م):

"جَعَلَ اللهُ طاعَتَهُ غَنيمَةَ هَؤُلاءِ، إذا فَرَّطَ فيها العَجَزَةُ المَخْذولونَ مِنَ النّاسِ، كصَيْدٍ اسْتَذَفَّ لرَجُلَيْنِ، أَحَدُهما جَلْدٌ والآخَرُ عاجِزٌ".

ابن أبي الحديد، شرح نهج البلاغة. تح: محمد أبو الفضل إبراهيم، ج: 19، ص: 243.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصارِيُّ (ت: 215ه=830م):

"ذَفَّ الأمْرُ، واسْتَذَفَّ: إذا تَهَيَّأَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، الغريب المصنف. تح: داوودي، ج: 1، ص: 341.— المصدر
06 —

ذَفاف/ ذُفاف/ ذِفاف

اسم
6.1معنى من معاني «ذَفاف/ ذُفاف/ ذِفاف»

*الذَّفافُ، والذُّفافُ، والذِّفافُ مِنَ الماءِ وغَيْرهِ: القَليلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو ذُؤَيْبٍ الهُذَليُّ (ت: 27ه=648م):

يَقولونَ لَمّا جُشَّتِ البِئْرُ أَوْرِدوا
فلَيْسَ بها أَدْنى ذَفافٍ لوارِدِ

ديوان أبي ذؤيب الهذلي. تح: الشّال، ص: 107.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"ماءٌ ذُفافٌ، وجَمْعُهُ ذُفُفٌ وأَذِفَّةٌ، أَي: قَليلٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 177.— المصدر
6.2معنى من معاني «ذَفاف/ ذُفاف/ ذِفاف»

وــــــــــ: البَللُ، وهو أَوَّلُ ما تَيَسَّرَ للرَّضيعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عُبَيْدٍ (ت: 224ه=838م):

"الذَّفافُ: البَلَلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: يعقوب عبد النبي، ج: 14، ص: 411.— المصدر
6.3معنى من معاني «ذَفاف/ ذُفاف/ ذِفاف»

* الذَّفافُ، والذِّفافُ: القَتْلُ والهَلاكُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رُؤبةُ بْنُ العَجّاجِ (90ه=709م) يُعاتِبُ أَباهُ:

حَوْلًا بِحَوْلِ اللهِ لا اعْتِصافي

ذاكَ الذي يَزْعُمُه ذِفافي

شرح ديوان رؤبة بن العجاج. تح: محروس، ج: 3، ص: 294.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (300ه=912م، مِنْ شَواهِدِ الهَجَريِّ):

وهَلْ ‌أَشْرَبَنْ ‌مِنْ ‌ماءِ ‌حَلْيَةَ ‌شَرْبَةً
تَكونُ شِفاءً أوْ ذَفافًا لما بِيا

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 52.— المصدر
6.4معنى من معاني «ذَفاف/ ذُفاف/ ذِفاف»

* الذُّفافُ، والذِّفافُ: السُّمُّ القاتِلُ. (ج) أَذِفَّةٌ، وذُفُفٌ.

ع
