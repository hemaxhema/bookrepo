01 —

خَلْخَلَ

فعل
1.1معنى من معاني «خَلْخَلَ»

* خَلْخَلَفُلانٌ الشَّيْءَ: حَرَّكَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أيّوبُ بْنُ سَعْنَةَ النَّخَعِيُّ (75هـ=695م) يَهْجو عَبْدَ اللهِ بْنَ الزَّبِيرِ الأسَدِيَّ:

رَمى اللهُ عَيْنَ ابْنَ الزَّبِيرِ بلَقْوَةٍ
فخَلْخَلَها حتَّى يَطولَ شُهودُها

البلاذري، أنساب الأشراف. تح: زكار وآخر، ج: 6، ص: 411.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ خالَوَيْهِ (ت: 370هـ=980م) يَتَحَدَّثُ عَنْ أَسْماءِ الأفْعالِ:

"إنَّما هِيَ حُروفٌ وأَفْعالٌ وُضِعَتْ لِلْأَمْرِ فَقَطْ،‍ فَجَرى كالمَثَلِ، لا ‌يُخَلْخَلُ عَنْ مَواضِعِهِ".

ابن خالويه، إعراب القراءات السبع. تح: العثيمين، ج: 1، ص: 326.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شَمْسُ الدّينِ الذَّهَبِيُّ (ت: 748هـ=1347م) يَذْكُرُ أَحْداثَ حِصارٍ للصَّليبِيّينَ على مَدينَةِ عَكّا:

"ودَفَعوا البُرجَ بِبَكَرٍ تَحْتَهُ حَتَّى أَلصَقوهُ بِسورِ عَكّا، وبَقِيَ أَعْلَى مِنْها بِكَثيرٍ، فَسلَّطَ عَلَيْهِ أَهْلُ عكّا الـمَجَانِيقَ حَتَّى ‌خَلْخَلوهُ، ثُمَّ رَمَوْهُ بقِدْرَةِ نَفْطٍ، فَاشْتَعَلَ".

الذهبي، سير أعلام النبلاء. تح: بشار عواد، ج: 22، ص: 210.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَد فارِس الشِّدْياق (1270هـ=1855م، تاريخُ الطَّبْعِ) يَحْكي على لِسانِ أَحَدِ شَخْصِيّاتِ قِصَّةٍ:

"لقَدْ ‌خَلْخَلَتْ هذِهِ الكَسْرَةُ سِنّي، فلأَقْلَعَنَّ سِنًّا مِنْ أسْنانِ هذا الدَّيْرِ".

الشدياق، الساق على الساق. مكتبة الحياة، ص: 136.— المصدر
1.2معنى من معاني «خَلْخَلَ»

وــــــــــ المَرْأَةَ أو غَيْرَها: ألْبَسَها الخَلْخالَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142هـ=760م):

"والرَّجُلُ الَّذي لا مُروءَةَ لَهُ يُهانُ، وإنْ كَثُرَ مالُهُ، كالكَلْبِ الَّذي يَهونُ على النّاسِ، وإن هُوَ طُوِّقَ ‌وخُلْخِلَ".

ابن المقفع، الأدب الصغير. دار صادر ص: 58.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُحَمَّدُ بْنُ مُحَمَّدِ بْنِ عَبْدِ اللهِ الحِمْيَرِيُّ (ت: 727هـ=1327م) يَصِفُ جامِعًا في مَدينَةِ المَوْصِلِ:

"وفي وسَطِ هَذا الجامِعِ الجَديدِ ساريَةُ رُخامٍ قائِمَةٌ، قَدْ خُلْخِلَ جيدُها بِخَمْسَةِ خَلاخِلَ مَفْتولَةٍ فَتلَ السِّوارِ مِن جِرْمِ رُخامِها".

الحميري، الروض المعطار. تح: عباس، ص: 563.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَحْمود خَيْرَت (1353هـ=1934م، تاريخُ الطَّبْعِ) يَتَغَزَّلُ:

‌ومُخَلْخَلاتٍ يَسْتَبينَكَ كُلَّما
يَخْطُرْنَ في رَوْضِ الـمَلاحَةِ عِينَا

مجلة الرسالة. ‌‌العدد: 49، 1934م، ص: 981.— المصدر
1.3معنى من معاني «خَلْخَلَ»

وــــــــــ الشَّيْءَ أو الأمْرَ: فَكَّكَهُ وجَعَلَهُ غَيرَ مُتَضامٍّ ولا مُتَماسِكٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو بَكْرٍ الرّازيُّ (ت: 313هـ=925م):

"وإنْ كانَتْ حَرارَةُ الحُمَّى لَيسَتْ قَوِيَّةً، والقُوَّةُ قَوِيَّةٌ والنُّضْجُ بَيِّنٌ، فَإنَّهُ يُنتَفَعُ جِدًا بالحَمّامِ وشُربِ الشَّرابِ والأدْهانِ المُخَلخِلَةِ للجَسَدِ".

الرازي، الحاوي في الطب. تح: طعيمي، ج: 4، ص: 261.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ بَدْرُ الدّينِ الشِّبْلِيُّ (ت: 769هـ=1367م) في أَحْكامِ الجانِّ:

"فَإنْ قيلَ: يَجوزُ أَنْ يَدْخُلوا في الأحْجارِ؟ قيلَ: نَعَمْ؛ إذا كانَتْ ‌مُخَلْخَلَةً كما يَجوزُ دُخولُ الهَواءِ فيها".

الشبلي، آكام المرجان. تح: الجمل، ص: 160.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمنِ البَهْكَليُّ (ت: 1224هـ=1809م) في إمارة ناصِرِ بْنِ مُحَمَّدٍ الحَسَنِيِّ:

"فاصْطَدَمَ بالنُّفوذِ المَورُوثِ للأُمَراءِ الخَواجِيّينَ، فأخَذَ في المُحاوَلَةِ لتَفريغِ مَكانَتِهِمْ مِنْ نُفوسِ أهْلِ صَبْيَا، وخَلْخَلَةِ نُفوذِهِمْ مِنْ قُلوبِ أهلِ الجِهَةِ".

البهكلي، نفح العود. تح: العقيلي، ج: 1، ص: 77.— المصدر
1.4معنى من معاني «خَلْخَلَ»

وــــــــــ العَظْمَ: أخَذَ ما عَلَيْهِ من اللَّحْمِ حَتَّى يَنْجَرِدَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحجّاجُ بْنُ يوسُفَ (ت: 95هـ=714م) لأبي حَرْبِ بْنِ أبي الأسْوَدِ الدّيليِّ:

"العامُ عامُكَ ‌تُخَلْخَلُ فيهِ عِظامُكَ".

البلاذري، أنساب الأشراف. تح: زكار وآخر، ج: 11، ص: 118.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"خَلْخَلْتُ العِظامَ: إذا أَخَذْتُ ما عَلَيْها مِنَ اللَّحْمِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ج: 1، ص: 190.— المصدر
1.5معنى من معاني «خَلْخَلَ»

وــــــــــ الحَرارَةُ الهَواءَ ونَحْوَهُ: جَعَلَتْهُ أقَلَّ كَثَافَةً وتَمَاسُكًا.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ خَلْدونَ (808هـ=1405م):

"وتَقَرَّرَ أنَّ الحَرارَةَ مُفْشِيَةٌ للهَواءِ والبُخارِ، مُخَلخِلَةٌ لَه، زائِدَةٌ في كَمِّيَّتِهِ".

ابن خلدون، المقدمة. تح: وافي، ص: 397.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ زكي نَجيب مَحْمود (1352هـ=1933م، تاريخُ الطَّبْعِ) يُعَلِّلُ سَبَبَ كَوْنِ الإنسانِ في الجِهاتِ البارِدَةِ يَتَنَفَّسُ أُكْسِجِينًا أكْثَرَ مِمَّن هُوَ في الجِهاتِ الدّافِئَةِ:

"لأنَّ الهَواءَ أكْثَفُ في الجِهاتِ البارِدَةِ، فيَكونُ مِقْدارُ الأُكْسِجِينِ في الشَّهْقَةِ الواحِدَةِ أكْبَرَ مِمّا لَوْ كانَ الهَواءُ ‌مُخَلْخَلًا خَفيفًا".

مجلة الرسالة. ‌‌العدد: 8، 1933م، ص: 16.— المصدر
02 —

تَخَلْخَلَ

فعل
2.1معنى من معاني «تَخَلْخَلَ»

* تَخَلْخَلَ فُلانٌ أو الشَّيْءُ: تَحَرَّكَ واضْطَرَبَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو كَبيرٍ الهُذَليُّ (ت: 11هـ=632م) يَصِفُ سِهامًا:

فَإذا تُسَلُّ تَخَلخَلَتْ أَرْياشُها
خَشْفَ الجَنوبِ بِيابِسٍ مِنْ إسْحِلِ

الشنقيطي، ديوان الهذليين. الدار القومية، ج: 2، ص: 99.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُقاتِلُ بْنُ سُلَيْمانَ (ت: 150هـ=896م) يَتَحَدَّثُ عَن كَشْفِ خازِنِ نارِ جَهَنَّمَ شيئًا مِنْها:

"فكَشَفَ عَنْ مِثْلِ مِنْخَرِ الثَّوْرِ مِنْها، فتَخَلْخَلَتْ فجاءَتْ بأمْرٍ عَظيمٍ".

مقاتل، تفسير القرآن. تح: شحاتة، ج: 4، ص: 604.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شِهابُ الدّينِ النُّوَيْرِيُّ (ت: 733هـ=1332م) يَتَحَدَّثُ عَن أحَدِ مُلوكِ مِصْرَ في وَقْتِ الطّوفانِ:

"وكانَ فَرْعانُ سَكْرانَ فَلَمْ يَقُمْ إلّا بخَريرِ الماءِ، فوَثَبَ مُبادِرًا يُريدُ الهرب إلى الهَرَمِ، ‌فتَخَلْخَلَتِ الأَرْضُ بِهِ".

النويري، نهاية الأرب. دار الكتب والوثائق، ج: 15، ص: 43.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمنِ الجَبَرْتِيُّ (ت: 1237هـ=1821م) يَذْكُرُ أَحْداثَ سَنَةِ 1213ه:

"وفيهِ سَقَطَتِ البَوّابَةُ المَصْنوعَةُ ببِرْكَةِ الأُزْبَكِيَّةِ المُقابِلَةِ لبابِ الهَواءِ الَّتي كانوا وَضَعوها في يَوْمِ عيدِهِمْ، وقَدْ تَقَدَّمَ شَرْحُها ووَصْفُها، وسَبَبُ سُقوطِها أنَّهُمْ لَمّا مَنَعوا الماءَ مِنْ دُخولِهِ للبِرْكَةِ، وسَدّوا القَنْطَرَةَ كما تَقَدَّمَ، عَلا الماءُ في أرْضِ البِرْكَةِ، ‌وتَخَلْخَلَتِ الأَرْضُ، فسَقَطَتْ تِلْكَ البَوّابَةُ".

الجبرتي، تاريخ عجائب الآثار. دار الجيل، ج: 2، ص: 214.— المصدر
2.2معنى من معاني «تَخَلْخَلَ»

وــــــــــ المَرْأةُ: لَبِسَتْ الخَلْخالَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 142.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"وتَخَلْخَلَتِ المرأَةُ: لَبِسَتِ الخَلْخَالَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، اللسان. دار صادر، ج: 11، ص: 221.— المصدر
2.3معنى من معاني «تَخَلْخَلَ»

وــــــــــ الشَّيْءُ أو الأمْرُ: تَفَكَّكَ وتَباعَدَتْ أجْزاؤُهُ.

ع

الجاحظ، الحيوان. تح: عيون السود، ج: 5، ص: 27.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ خَلِّكان (ت: 681ه=1282م):

"ثُمَّ إنَّ عَبْدَ المَلِكِ بْنَ مَرْوانَ وَلَّى الحَجّاجَ مَكَّةَ واليَمَنَ واليَمامَةَ، فنَقَضَ الحَجّاجُ بُنْيانَ الكَعْبَةِ الَّذي بَناهُ ابْنُ الزُّبَيْرِ؛ لأنَّهُ كانَ تَخَلْخَلَ مِنْ حِجارَةِ المِنْجَنيقِ".

ابن خلكان، وفيات الأعيان. تح: عباس، ج: 3، ص: 75.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)

"قُلْنا إنَّهُ مِنْ جِنْسِهِ، لكِنْ لأجْلِ رُطوبَتِهِ ‌وتَخَلْخُلِ أجْزائِهِ لا يُدْرَكُ المُماثَلَةُ بالكَيْلِ، فصارَ كالمُجازَفَةِ".

محمد ثناء المظهري، تفسير المظهري. تح: التونسي، ج: 1، ص: 403.— المصدر
2.4معنى من معاني «تَخَلْخَلَ»

وــــــــــ الثَّوبُ: رَقَّ، وصارَ قِطَعًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللِّحيانيُّ (ت: 220هـ=835م):

"تَسَلْسَلَ ‌الثَّوْبُ ‌وتَخَلْخَلَ: إذا لُبِسَ حَتَّى رَقَّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 12، ص: 208.— المصدر
”وقالَ أبو جَعْفَرٍ النَّحّاسُ (ت: 338هـ=950م):

"إنَّ السَّفَهَ كُلُّ ما يَقْبُحُ فِعْلُهُ؛ أي: هُوَ فِعْلٌ لَيْسَ بِمُحْكَمٍ، مِنْ قَوْلِهِمْ: ‌ثَوْبٌ سَفيهٌ؛ إذا كانَ ‌مُتَخَلْخِلًا".

النحاس، معاني القرآن. تح: الصابوني، ج: 1، ص: 316.— المصدر

* الخَلْخَلَةُ (في الكيمياءِ): عَمَلِيَّةُ تَقْلِيلِ كَثافَةِ الغازاتِ، بزِيادةِ حَجْمِها دونَ تَغْييرِ كَمِّيَّتها، أو بالعَكْسِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد شَفيق غِرْبال (ت: 1381ه=1961م) في الفَيْلَسوفِ أناكسمانس:

"وواحِدٌ مِن الفَلاسِفَةِ الأَيُونِيِّينَ، يَرَى أنَّ الأشياءَ كُلَّها تَرتَدُّ إلى عُنصُرِ الهَواءِ، ومِن خَلخَلَةِ الهَواءِ أو تَكاثُفِهِ تَتَكَوَّنُ العَناصِرُ المُختَلِفَةُ كما تَتَكَوَّنُ مَادَّةُ النَّفْسِ الإنسانِيَّةِ".

غربال وآخرون، الموسوعة العربية الميسرة. ج: 1، ص: 452.— المصدر
08 —

مُخَلْخَل

اسم
8.1معنى من معاني «مُخَلْخَل»

* المُخَلْخَلُ: مَوضِعُ الخَلْخالِ مِنَ السّاقِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤ القَيْسِ (ت: 80ق ه=543م) يَذْكُرُ مَحْبوبَتَهُ:

هَصَرْتُ بِفَوْدَيْ رَأْسِها فَتَمايَلَتْ
عَلَيَّ هَضِيمَ الكَشْحِ رَيَّا المُخَلْخَلِ

امرؤ القيس، ديوان امرئ القيس. تح: عليان وآخرين، ص: 212.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحارِثُ بْنُ خالِدٍ المَخْزومِيُّ (ت: 80هـ=699م) يَتَغَزَّلُ:

لفّاءُ ‌مَمْلوءٌ ‌مُخَلْخَلُها
عَجْزاءُ لَيْسَ لعَظْمِها حَجْمُ

الأصفهاني، الأغاني. تح: جابر، ج: 9، ص: 259.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشْارُ بْنُ بُرْدٍ (ت: 170هـ=787م) يَتَغَزَّلُ:

هَيفاءُ لَفّاءُ جِردَحْلٌ مُخَلْخَلُها
تُحْيي وتَقتُلُ مَن شاءَت بِما تَعِدُ

بشار بن برد، ديوان بشار. تح: أمين، ج: 2، ص: 281.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ التَّلَّعْفَريُّ (ت: 675هـ=1276م) يَتَغَزَّلُ:

حَسُنَتْ فَمَلَّكَها الجَمالَ مُخَلْخَلٌ
ومُسَوَّرٌ ومُدَمْلَجٌ ومُقَلَّدُ

التّلّعفريّ، ديوان التّلّعفريّ. تح: رضا رجب، ص: 98.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الباقي العُمَريُّ (ت: 1278هـ=1861م) يَرْثي رَجُلًا:

يَقُولُ: مَنِ العَلْيا سَتُبْدِي نُواحَها
عَلَيَّ هَضِيمَ الكَشْحِ رَيّا المُخَلخَلِ

عبد الباقي العمري، التّرياق. دار النعمان، ص: 369.— المصدر
8.2معنى من معاني «مُخَلْخَل»

وــــــــــ (في عِلْمِ الأصْواتِ): الحَلْقُ ذو الصَّوْتِ العَالي والنَّغَمِ الحادِّ بِحَلاوَةٍ وجَهارَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الحُسَيْنِ ابْنُ الطَّحّانِ (ت: 450ه=1058م) يُعَدِّدُ أسْماءَ الحُلوقِ وصِفاتِها:

"الـمُخَلْخَلُ: وهُوَ العَالي الحادُّ النَّغَمِ بِحَلاوَةٍ وجَهارَةٍ".

ابن الطّحّان، حاوي الفنون. تح: زكريا يوسف، ص: 33.— المصدر
