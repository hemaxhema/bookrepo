ضنك
ضَنَكَ(فعل) ضَنُكَ(فعل) ضَنِكَ(فعل) ضُنِكَ(فعل) أَضْنَكَ(فعل) ضانَكَ(فعل) تَضَنَّكَ(فعل) ضَناك/ ضِناك(اسم) ضُناك(اسم) ضِناك(اسم) ضَنْك(اسم) ضُنْكَة(اسم) ضَنيك(اسم)
النقوش العربيّة

وَرَدَ الجَذْرُ (ض ن ك ضنك ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَينِها نَقْشٌ حَضرَميٌّ غَيرُ مؤرَّخٍ (Ja 2879/2)، وَرَدَ فيهِ الاسْمُ (ضنك ḍinʾâk  ) عَلَمًا على أحَدِ الأفْرادِ، فَجاءَ في النَّقْشِ: "ضنك/ بن/ أكوك"، أيْ: "ضِنَكُ بْنُ أكوك".

(Jamme, 1979: 95)

المعانى الكلّيّة

  1. 1ــ الضِّيقُ والشِّدَّةُ
  2. 2ــ المَرَضُ والهُزالُ
  3. الاكْتِنازُ والامْتِلاءُ
01 —

ضَنَكَ

فعل
1.1معنى من معاني «ضَنَكَ»

* ضَنَكَ (بِفَتْحِ النّونِ) فُلانٌ فُلانًا يَضْنُكُ (بِضَمِّ النّونِ) ضَنْكًا: غَمَزَ يَدَيْهِ بِأَطْرافِ أَصابِعِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن القَطّاع (ت:515هـ=1121م):

"وضَنَكْتُ الرَّجُلَ: إِذا غَمَزْتَ يَدَيْهِ بِأَطْرافِ أَصابِعِكَ".

ابن القَطَّاع، الأفعال، ج: 2، ص: 275.— المصدر
1.2معنى من معاني «ضَنَكَ»

وــــــــــ: ضَيَّقَ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ نَشْوانُ الحِمْيَرِيُّ (ت:573هـ=1178م):

"ضَنَكَهُ: إِذا ضَيَّقَ عَلَيْهِ، مِنَ الضَّنْكِ: وَهُوَ الضِّيقُ".

نشوان، شمس العلوم، ج: 6، تح: العمري وآخرين، ص: 4007.— المصدر
02 —

ضَنُكَ

فعل
2.1معنى من معاني «ضَنُكَ»

* ضَنُكَ (بِضَمِّ النّونِ) العَيْشُ والمَكانُ وغَيْرُهُما يَضْنُكُ (بِضَمِّ النّونِ)، ضَنْكًا وضَناكَةً وضُنوكَةً: ضاقَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُهَلْهِل (ت: 100ق.هـ=525م) يَصِفُ قِلَّةَ حيلَةِ قاتِلِ أَخيهِ:

مَنْ شاءَ وَلّى النَّفْسَ في مَهْمَهٍ
ضَنْكٍ ولَكِنْ مَنْ لَهُ بِالمَضيقْ

ديوان المهلهل. تح: القوال، ص: 54.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القرآن الكريم (11هـ=632م) قالَ اللهُ ﷻ:

﴿وَمَنۡ أَعۡرَضَ عَن ذِكۡرِي فَإِنَّ لَهُۥ مَعِيشَةٗ ضَنكٗا وَنَحۡشُرُهُۥ يَوۡمَ ٱلۡقِيَٰمَةِ أَعۡمَىٰ ﴾. ]طه:124[

”وقالَ أبو حرب بن خُوَيْلد (ت:11هـ=632م) يَصِفُ فَوارِسَ المُسْلِمينَ زَمَنَ الرِّدَّةِ ويُحَذِّرُ قَوْمَهُ مِنْهُمْ:

تَقَحَّمَها في غَمْرَةِ المَوْتِ خالِدٌ
بِمُعْتَرَكٍ ضَنْكٍ أَحَرَّ مِنَ الجَمْرِ

ديوان حروب الردة. تح: محمود أبو الخير، ص: 238.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت:170هـ=787م):

"وقَدْ ضَنُكَ عَيْشُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 302.— المصدر
”وقالَ السيد الحِمْيَري (ت:173هـ=789م) يَرْثي أَخاهُ:

ولَعَمْري لَئِنْ تَرَكْتُكَ مَيْتًا
رَهْنَ رَمْسٍ ضَنْكٍ عَلَيْكَ مُهالِ

ديوان السيد الحميري. تح: الأعلمي، ص: 159.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرَّسْعني (ت:661هـ=1263م) يَذْكُرُ حالَ المُنافِقينَ مِنْ أَهْلِ المَدينَةِ قَبْلَ هِجْرَةِ النَّبِيِّ ﷺ:

"كانوا حينَ قَدِمَ رَسولُ اللهِ ﷺ إِلى المَدينَةِ في ضيقٍ وضَنْكِ مَعيشَةٍ".

الرّسعني، رموز الكنوز. تح: دهيش، ج: 2، ص: 548.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الآلوسي (ت:1270هـ=1854م) يُفَسِّرُ قَوْلَهُ تَعالى عَلى لِسانِ يوسُفَ: ﴿ يَٰصَٰحِبَيِ ٱلسِّجۡنِ ءَأَرۡبَابٞ مُّتَفَرِّقُونَ خَيۡرٌ أَمِ ٱللَّهُ ٱلۡوَٰحِدُ ٱلۡقَهَّارُ  ﴾ [يوسف: 39]:

"وَلَعَلَّ في نِدائِهِما بِذَلِكَ عَلى هَذا الوَجْهِ حَثًّا لَهُما عَلى الإِقْرارِ بِالحَقِّ، كَأَنَّهُ قالَ لَهُما: يا ساكِنَيْ هَذا المَكانِ الشّاقِّ، والمَحَلِّ الضَّنْكِ، إِنّي ذاكِرٌ لَكُمْ أَمْرًا، فَقولوا الحَقَّ فيهِ، ولا تَزيغوا عَنْ ذَلِكَ".

الآلوسيّ، روح المعاني. تح: حبّوش وآخرين، ج: 12، ص: 338.— المصدر
2.2معنى من معاني «ضَنُكَ»

وــــــــــ فُلانٌ وغَيْرُهُ: ضَعُفَ بَدَنُهُ ورَأْيُهُ، فهو ضانِكٌ (ج) ضَوانِكُ، وضَنِكٌ، وضَنيكٌ (ج) ضُنْكٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفرزدق (ت:110هـ=728م) يُعاتِبُ خَالِدَ بْنَ عَبْدِ اللهِ القَسْرِيَّ عَلى هَدْرِ المالِ:

أَإِنْفَاقَ مَالِ اللهِ فِي غَيْرِ كُنْهِهِ
وَمَنْعًا لِحَقِّ المُرْمَلَاتِ الضَّوَانِكِ

ديوان الفرزدق. تح: فاعور، ص: 415.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصارِيُّ (ت:215هـ=830م):

"وضَنُكَ أَيْضًا: إِذا ضَعُفَ في بَدَنِهِ، ورَأْيِهِ ونَفْسِهِ، فهُوَ ضَنِكٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال. تح: شرف، ج: 2، ص: 233.— المصدر
2.3معنى من معاني «ضَنُكَ»

وــــــــــ: السَّحابُ: عَظُمَ وتَراكَمَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عَبّاد (ت:385هـ=995م):

"وضَنُكَ السَّحابُ: غَلُظَ والْتَفَّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 173.— المصدر
03 —

ضَنِكَ

فعل
3.1معنى من معاني «ضَنِكَ»

* ضَنِكَتِ (بِكَسْرِ النّونِ) المَرْأَةُ تَضْنَكُ (بِفَتْحِ النّونِ) ضَنَكًا: امْتَلَأَتْ شَحْمًا، فَهِيَ ضَناك.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن القَطّاع (ت:515هـ=1121م):

" وضَنِكَتِ المَرْأَةُ ضَنَكًا: امْتَلَأَتْ شَحْمًا، فَهِيَ ضَناك".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القَطَّاع، الأفعال، ج: 2، ص: 275.— المصدر
04 —

ضُنِكَ

فعل
4.1معنى من معاني «ضُنِكَ»

* ضُنِكَ (بِالبِناءِ لِلْمَجْهولِ) فُلانٌ ضَنْكًا وضُنْكَةً: أُصيبَ بِالزُّكامِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت:170هـ=787م):

"والضُّناكُ: الزُّكامُ، ‌ضُنِكَ فهُوَ مَضْنوكٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 302.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القَرافي (ت:684هـ=1285م) يَذْكُرُ تَشْميتَ العاطِسِ:

"وَكانَتِ الجاهِلِيَّةُ تَتَطَيَّرُ بِالعُطاسِ إِلى ثَلاثِ مَرَّاتٍ وَتَجْعَلُها شُؤْمًا، فَأَعْلَمَ صاحِبُ الشَّرْعِ أَنَّها رَحْمَةٌ مِنَ اللَّهِ تَعالى. وَاقتَصَرَ بِقَوْلِنا: يَرْحَمُكَ اللَّهُ عَلى الثَّلاثَةِ التِي كانَتِ الجاهِلِيَّةُ تَتَشاءَمُ بِها إِثْباتًا لِلضِّدِّ، ولِهَذا السِّرِّ قيلَ لَهُ في الرّابِعَةِ إِنَّهُ مَضْنوكٌ".

القرافي، الذخيرة. تح: حجي، ج: 13، ص: 301.— المصدر
4.2معنى من معاني «ضُنِكَ»

وــــــــــ ضُناكًا: لَزِمَهُ الزُّكامُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن القوطية (ت:367هـ=977م):

"وضُنِكَ ضُنْكَةً: زُكِمَ، وضُناكًا: إِذا لَزِمَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: فودة، ص: 246.— المصدر
05 —

أَضْنَكَ

فعل
5.1معنى من معاني «أَضْنَكَ»

* أَضْنَكَ اللهُ فُلانًا: أَزْكَمَهُ.فَهُوَ مُضْنَكٌ، ومَضْنوكٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسم بن سلّام (ت:224هـ=839م):

"واللهُ أَضْنَكَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج: 10، ص: 41.— المصدر
06 —

ضانَكَ

فعل
6.1معنى من معاني «ضانَكَ»

* ضانَكَ فُلانٌ وغَيْرُهُ فُلانًا وَغَيْرَهُ: زاحَمَهُ وضايَقَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ غالبٌ الجُعْفِيُّ (132هـ=750م) كانَ مُعاصِرًا لِـمَرْوانَ بْنِ مُحَمَّدٍ:

يَقولَ لَها الزُّوّارُ أَيْنَ خِيارُها
أَهاتِيكَ أَمْ هاتي الَّتي لَمْ تُضانِكِ

المعري، اللامع العزيزي. تح: المولوي، ص: 859.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المَعَرِيُّ (449هـ=1057م) يَتَخَيَّلُ مُحاوَلَةَ وُقوفِهِ بِجِوارِ خازِنِ الجَنَّةِ لِيُسْمِعَهُ ما نَظَمَهُ فيهِ:

"ثُمَّ ‌ضانَكْتُ النَّاسَ حَتّى وَقَفْتُ مِنْهُ بِحَيْثُ يَسْمَعُ وَيَرى، فَما حَفَلَ بي، ولا أَظُنُّهُ أَبِهَ لِما أَقولُ"

المعري، رسالة الغفران. تح: بنت الشاطئ، ص: 249.— المصدر

08 —

ضَناك/ ضِناك

اسم
8.1معنى من معاني «ضَناك/ ضِناك»

* الضَّناكُ والضِّناكُ: المَرْأَةُ المُكْتَنِزَةُ اللحْمِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حميد بن ثور (ت:60هـ=680م) يَصِفُ امْرَأَةً:

وَقَدْ أُناجي الرَّشَأَ المُرَبَّبا

ذا الرَّعَثاتِ البادِنَ المُخَضَّبا

خَوْدًا ضِناكًا لا تَمُدُّ العُقَبا

يَهْتَزُّ مَتْناها إِذا ما اضْطَرَبا

مجالس ثعلب. تح: هارون، ج:1، ص: 372.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت:170هـ=787م):

"وامْرَأَةٌ ضِناكٌ، أَيْ: مُكْتَنِزَةٌ تَارَّةٌ صُلْبَةُ اللحْمِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 302.— المصدر

تَبْدو ويَرْفَعُها السَّرابُ كَأَنَّها
مِنْ عُمِّ مَوْثِبَ أَوْ ضِناكِ خِدادِ

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
