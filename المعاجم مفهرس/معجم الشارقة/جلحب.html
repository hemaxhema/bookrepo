   مُجْلَحِبّ                                                     (اسم) جُلاحِب(اسم) جِلْحاب(اسم) جِلْحابة(اسم) جَلْحَب(اسم) جِلْحَب(اسم)
01 —

   مُجْلَحِبّ                                                     

اسم
1.1معنى من معاني «   مُجْلَحِبّ                                                     »

* الـمُجْلَحِبُّ: الـمُمْتَدُّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابن سِيدَه (ت: 458ه=1066م):

"وَالـمُجْلَحِبُّ: الـمُمْتَدُّ، وَلَا أُحِقُّهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 56.— المصدر
01 —

جُلاحِب

اسم
1.1معنى من معاني «جُلاحِب»

* الجُلاحِبُ: الشَّيْخُ الكَبيرُ الهَرِمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

حَتّى رَأى الشَّيْبَةَ في العِنَاسِ

وَعــــادِمَ الــــجُــلاحِــبِ العَوّاسِ

العِنَاسُ: الـمِرْآةُ
ابن سيده، المحكم. تح: هنداوي، ج: 1، ص: 494.— المصدر
1.2معنى من معاني «جُلاحِب»

وــــــــــ: الرّجُلُ الضَّخْمُ الأَجْلَحُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ ‌السِّكِّيتِ (ت: 244ه=858م):

"رَجُلٌ جِلْحابٌ وجِلْحابَةٌ ‌وجُلاحِبٌ وجَلْحَبٌ، وهُوَ الضَّخْمُ الأجْلَحُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل، تح: الطحاوي، ج: 1، ص: 91.— المصدر
02 —

جِلْحاب

اسم
2.1معنى من معاني «جِلْحاب»

* الجِلْحابُ: الرَّجُلُ يَبْلُغُ الغَايَةَ في الكِبَرِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَسْلَمَةُ بْنُ عَبْدِ الـمَلِكِ (121ه=739م) بَعْدَ سَماعِ قَصِيدٍ للْكُمَيْتِ:

"سُبْحَانَ اللهِ، مَنْ هذَا الهِنْدِكِيُّ الجِلْحَابُ الَّذِي أقْبَلَ مِنْ أُخْرَياتِ النّاسِ، فبَدَأَ بالسَّلامِ ثُمّ أَمَّا بَعْدُ، ثُمَّ بِالشِّعْرِ؟ قيلَ لَهُ: ‏هَذا الكُمَيْتُ".

ابن عبد ربه، العقد الفريد. تح: قميحة، ج: 2، ص 56.— المصدر
2.2معنى من معاني «جِلْحاب»

وــــــــــ: الـمُسِنُّ الـمُـتَهالِكُ؛ ضَعيفُ الجِسْمِ والعَقْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الثَّعالِبِيُّ (430ه=1039م):

"إِذا شاخَ الرَّجُلُ وَعَلَتْ سِنُّهُ؛ فَهُوَ قَحْرٌ وَقَحْبٌ؛ فَإِذا وَلّى وَساءَ عَلَيْهِ أَثَرُ الْكِبَر؛ فَهُوَ يَفَنٌ وَدِرْدَحٌ؛ فَإِذا زَادَ ضَعْفُهُ وَنَقَصَ عَقْلُهُ؛ فَهُوَ: جِــلْـحابٌ ‌وَمُـــهْتَــرٌ".

الثعالبي، فقه اللغة. تح: عبد الرزاق المهدي، ص 79.— المصدر
2.3معنى من معاني «جِلْحاب»

وــــــــــ: الطَّويلُ.

—
غير محدد
”(ق. س) قالَ ‌نِجَادٌ ‌الخيْبري (86ق.هـ=538م) يَهْجو أَحَدَهُمْ:

عِضٌّ لَئيمُ الـمُنْتَمى والعُنْصُرِ


لَيْسَ بِـجِـِلْـحــــابٍ ولا ‌هَــقَـــــوَّرِ


لَـكِـنَّــهُ الــبُـــــهْــــــتُــــــــرُ وابْنُ البُهْتُــــرِ


[العِضُّ: الرَّجُلُ الدَّاهِي الـمُنْكَرُ.]
ابن السِّكِّيت، كتاب الألفاظ. تح: قباوة، ص: 161.— المصدر
2.4معنى من معاني «جِلْحاب»

وــــــــــ: الضَّخْمُ الأَصْلَعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو ‌عَمْرٍو ‌الشَّيْبانِيُّ (ت: 206ه=821م): 

"وَرَجُلٌ ‌جِلْحابٌ، وَجِلْحابةٌ، إِذا كانَ ضَخْمًا أجْلَح". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو بكر الأنباري، المذكر والمؤنث. تح: عضيمة، وآخر، ج: 2، ص 171.— المصدر
2.5معنى من معاني «جِلْحاب»

وــــــــــ: فُحّالُ النَّخْل، وَهُوَ ذَكَرُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الْأَعرَابيّ (ت: 231ه=845م):

"الجِلْحَابُ: فُحَّالُ النَخْلِ".

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 5، ص: 315.— المصدر
03 —

جِلْحابة

اسم
3.1معنى من معاني «جِلْحابة»

* الـجِلْحابةُ: الثَّقيلُ الأَحْمَقُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الفَارابِي (350ه=961م):

"يُقالُ: شَيْخٌ ‌جِلْحابَــةٌ وَجِلْحابٌ، بِمَعْنى الـهِلْـبـاجَةِ الأَحْمَقِ"

الفارابي، معجم ديوان الأدب. تح: أحمد مختار، ج: 2، ص 72.— المصدر
3.2معنى من معاني «جِلْحابة»

وــــــــــ: الـمُسِنُّ الفاني.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيّ (216 ه/831م):

"ومِثْلُهُ: شَيْخٌ ‌جِلْحابَةٌ وَجِلْحابٌ".

القاسم بن سلام، الغريب. تح: داوودي، ج: 1، ص 153.— المصدر
3.3معنى من معاني «جِلْحابة»

وــــــــــ: الرَّجُلُ الضَّخْمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو ‌عَمْرٍو ‌الشَّيْبانِيُّ (ت: 206ه=821م) قال: 

"يُقالُ رَجُلٌ شِهْدارَةٌ، إِذا كانَ قَصيرًا، وَرَجُلٌ جِلْحابٌ وجِلْحابةٌ: إِذا كانَ ضَخْمًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأنباري، المذكر والمؤنث. تح: عضيمة، وآخر، ج: 2، ص 171.— المصدر
04 —

جَلْحَب

اسم
4.1معنى من معاني «جَلْحَب»

* الـجَلْـحَـبُ: الشَّيْخُ الـمُسِنُّ العَظيمُ الجِسْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيدٍ (ت: 321ه=933م):

"وَرَجُلٌ ‌جَلْحَبٌ وجِلْحابٌ وَجُلاحِبٌ، وَهُوَ الشَّيْخُ الْعَظِيمُ الْجِسْمِ، وفيه بَـقِيَّةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص 1112.— المصدر
4.2معنى من معاني «جَلْحَب»

وــــــــــ: الضَّخْمُ الأَصْلَعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

"رَجُلٌ جِلْحابٌ وَجِلْحابَةٌ وَجُلاحِبٌ وَجَلْحَبٌ، وَهُوَ الضَّخْمُ الأَجْلَحُ"

الصغاني، التكملة والذيل. تح: الطحاوي، ج: 1، ص 91.— المصدر
05 —

جِلْحَب

اسم
5.1معنى من معاني «جِلْحَب»

* الـجِلْحَبُّ: الرَّجُلُ الطّويلُ القامَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الأَزْهَرِيُّ أبو عَمْرٍو الشَّيْبانِيّ (ت: 206ه=821م):

"الـجِلْحَبُّ: الرَّجُلُ الطَّوِيلُ القَامَةِ، وَأنْشَدَ:

وَهْي تُرِيدُ العَزَبَ الــجِــلْـــحَــــبَّـــا

يَسْكُبُ ماءَ الظَّهرِ فِيهَا سَكْبَا"

(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)
الأَزْهَرِي، تهذيب اللغة. تح: عبد الله درويش، ج: 5، ص: 315.— المصدر
5.2معنى من معاني «جِلْحَب»

وــــــــــ: القَوِيُّ الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابن سِيدَه (ت: 458ه=1066م): 

"وَالجِلْحَبُّ: الْقَوِيُّ الشَّديدُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 56.— المصدر
