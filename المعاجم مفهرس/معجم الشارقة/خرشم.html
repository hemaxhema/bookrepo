01 —

خَرْشَم

فعل
1.1معنى من معاني «خَرْشَم»

* خَرْشَمَ فُلانٌ: جَعَلَ وَجْهَهُ كَريهًا. (وانْظُرْ: ج ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"وَخَرْشَمَ الرَّجُلُ، إِذا كَرَّهَ وَجْهَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 1145.— المصدر
1.2معنى من معاني «خَرْشَم»

وــــــــــ: مَرِضَ أَوْ هُزِلَ ثُمَّ بَرَأَ. (وانْظُرْ: ج ر س م، ج ر ش ب، ج ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ (ت: 515هـ=1122م):

"جَرْشَمَ الرَّجُلُ: إِذا مَرِضَ أَوْ هُزِلَ، ثُمَّ انْدَمَلَ (...)، وخَرْشَمَ بالخاءِ والشّيْنِ كَذلِكَ".

ابن القطاع، الأفعال، ج: 1، ص: 196.— المصدر
1.3معنى من معاني «خَرْشَم»

وــــــــــ فُلانًا: لَكَمَهُ، فأَصابَ أَنْفَهُ أَوْ وَجْهَهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيديُّ (ت: 1205هـ=1790م):

"وخَرْشَمَهُ خَرْشَمَةً: أَصابَ أَنْفَهُ (عامِّيَّةٌ)". (وهذِهِ الدّلالَةُ سَجَّلَتْها الـمَعاجِمُ في العَصْرِ الحَديثِ)

الزبيدي، تاج العروس، تح: العزباوي وآخرين، ج: 32، ص: 76.— المصدر
02 —

اخْرَنْشَمَ

فعل
2.1معنى من معاني «اخْرَنْشَمَ»

* اخْرَنْشَمَ فُلانٌ وغَيْرُهُ: انْقَبَضَ، وتَقارَبَ خَلْقُ بَعْضِهِ مِنْ بَعْضٍ. (وانْظُرْ: ج ر ش م)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَدِيُّ بنُ الرِّقاعِ (ت: 95هـ=714م) يَتَغَزَّلُ:

إِذا دَجا اللَّيْلُ وَلّاها مَقاتِلَهُ
فقَدْ بَرى لَحْمَهُ مِنْ حُبِّها العَجَفُ

مُخْرَنْشِمًا لِعَماءٍ باتَ يَضْرِبُهُ
مِنْهُ الرُّضابُ ومِنْهُ الـمُسْبِلُ الهَطِفُ

الأزهري، تهذيب اللغة، تح: خفاجي وآخرين، ج: 6، ص: 181.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (231هـ=845م، مِنْ شَواهِدِ ابْنِ الأَعْرابِيِّ):

وفَخِذٍ طالَتْ ولَمْ تَخْرَنْشِمِ

الأزهري، تهذيب اللغة، تح: محمد أبي الفضل وآخر، ج: 11، ص: 261.— المصدر
2.2معنى من معاني «اخْرَنْشَمَ»

وــــــــــ: غَضِبَ أَوِ اغْتاظَ. (وانْظُرْ: خ ر ط م، خ ر ف ش)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والـمُخْرَنْفِشُ، والـمُخْرَنشِمُ، هُوَ كالـمُغْتاظِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 325.— المصدر
2.3معنى من معاني «اخْرَنْشَمَ»

وــــــــــ السِّقاءُ: ذَهَبَ ثُلُثُهُ أَوْ رُبُعُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُؤَرِّجٌ السَّدوسِيُّ (ت: 195هـ=811م):

"يُقالُ لِلسِّقاءِ: قَدِ اخْرَنْشَمَ، إذا ذَهَبَ ثُلُثُهُ أَوْ رُبُعُهُ".

السدوسي، الأمثال، تح: رمضان عبد التواب، ص: 69.— المصدر
2.4معنى من معاني «اخْرَنْشَمَ»

وــــــــــ الحَيَوانُ: اضْطَرَبَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو الدُّرُسْتِ السَّدوسِىُّ (195هـ=811م، من رُواةِ مُؤَرِّجٍ السَّدوسِيِّ) يُجيبُ من سَأَلَهُ عَنْ إِبِلِهِ:

"تُمْسي مُخْرَنْشِماتٍ".

السدوسي، الأمثال، تح: رمضان عبد التواب، ص: 69.— المصدر
2.5معنى من معاني «اخْرَنْشَمَ»

وــــــــــ فُلانٌ: تَغَيَّرَ لَوْنُهُ، وذَهَبَ شَحْمُهُ ولَحْمُهُ. (وانْظُرْ: ح ر ش م، ج ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"والـمُخْرَنْشِمُ أَيْضًا: الـمُتَغَيِّرُ اللَّوْنِ الذّاهِبُ اللَّحْمِ".

الزبيدي، تاج العروس، تح: العزباوي وآخرين، ج: 32، ص: 75.— المصدر
2.6معنى من معاني «اخْرَنْشَمَ»

وــــــــــ: تَكَبَّرَ وتَعاظَمَ في نَفْسِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207هـ=822م):

"مُخْرَنْشِمٌ: وهُوَ الـمُتَعظِّمُ الـمُتَكَبِّرُ في نَفْسِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، الغريب المصنف، تح: داوودي، ج: 1، ص: 646.— المصدر
03 —

خَرْشَم

اسم
3.1معنى من معاني «خَرْشَم»

* الخَرْشَمُ: الجَماعَةُ مِنَ النَّحْلِ. (وانْظُرْ: خ ش ر م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو القاسِمِ الزَّجّاجِيُّ (ت: 337هـ=947م) يُعَدِّدُ أَسْماءَ جَماعَةِ النَّحْلِ:

"ويُقالُ: هِيَ النَّحْلُ، والثَّوْلُ، والدَّبْرُ، والخَشْرَمُ، والخَرْشَمُ، والرَّضَعُ، والدُّخا، بِتَخْفيفِ الخاءِ والقَصْرِ، واليَعاسيبُ، والنّوبُ، كلُّه بِمَعْنًى واحِدٍ".

الزجاجي، أمالي الزَّجاجي، تح: هارون، ص: 27.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَحْمود شُكْري الآلوسِيُّ (1326هـ=1909م، تاريخُ الطَّبْعِ):

"ويُفَرِّقونَ في الجَماعاتِ، فيَقولونَ: كَوْكَبَةٌ مِنَ الفُرْسانِ، وكَبْكَبَةٌ مِنَ الرِّجالِ، وجَوْقَةٌ مِنَ الظِّباءِ، وعَرْجَلَةٌ مِنَ السِّباعِ، وعِصابَةٌ مِنَ الخَيْلِ، وصِرْمَةٌ مِنَ الإِبِلِ، وقَطيعٌ مِنَ الغَنَمِ، وسِرْبٌ مِنَ الظِّباءِ، وعَرْجَلَةٌ مِنَ السِّباعِ، وعِصابَةٌ مِنَ الطَّيْرِ، ورِجْلٌ مِنَ الجَرادِ، وخَرْشَمٌ مِنَ النَّحْلِ".

محمود الآلوسي، المقيم المقعد، مجلة المقتبس، ع: 40، ص: 5.— المصدر
04 —

خِرْشَمّ

اسم
4.1معنى من معاني «خِرْشَمّ»

* الخِرْشَمُّ: الحَجَرُ الرِّخْوُ. (وانْظُرْ: ه ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"والهِرْشَمُّ مِتْلُ الخِرْشَمِّ، وقَدْ مَرَّ ذِكْرُهُ، وهُوَ الحَجَرُ الرِّخْوُ".

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 1152.— المصدر
05 —

خِرْشَمّ / خِرْشِمّ

اسم
5.1معنى من معاني «خِرْشَمّ / خِرْشِمّ»

* الخِرْشَمُّ، والخِرْشِمُّ: الحَجَرُ، أَوِ الجَبَلُ اليابِسُ الصُّلْبُ. (وانْظُرْ: ه ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (321هـ=933م، مِنْ شَواهِدِ ابْنِ دُرَيْدٍ) يصِفُ بِئْرًا:

خِرْشَمّةٌ في جَبَلٍ خِرْشَمِّ

تُبْذَلُ لِلجارِ ولِابْنِ العَمِّ

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 1145.— المصدر

اسم

* الخِرْشَمَّةُ: الأَرْضُ ذاتُ الحِجارَةِ الرِّخْوَةِ. (وانْظُرْ: ه ر ش م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"وأَرْضٌ خِرْشَمَّةٌ، وهِي ذاتُ الحِجارَةِ الرِّخْوَةِ".

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 1145.— المصدر

08 —

خُرْشوم

اسم
8.1معنى من معاني «خُرْشوم»

(الخُرْشُومُ : ẖar- som ) فارِسِيّةٌ مُعَرَّبةٌ، أصْلُها في الفارِسِيّةِ (خُرْسُم) وتُطْلَقُ على أنْفِ الجَبَلِ المُشْرِفِ على وادٍ أو قاعٍ.

—
غير محدد
”

وقَدْ وَرَدَتْ في الفارِسِيّةِ حَيْثُ يَقولُ دهخدا: "خرشوم: بينى كوه بر وادى يا زمين هموار – كوه بزرﮒ - زمين سخت ودرشت "


والمَعْنى: الخُرْشُومُ : هُوَ أنْفُ الجَبَلِ المُطِلِّ على وادٍ أو أرْضٍ مُسْتَويةٍ، كَما يُطْلَقُ على الجَبَلِ الضَّخْمِ والأرْضِ شَديدةِ الوُعورةِ.
(لغت نامهءدهخدا – جلد 7– تأليف علي اكبر دهخدا –ص 9690 – 1377ه) — المصدر
”

وقَدِ انْتَقَلَتْ إلى العَرَبِيّةِ بِنَفْسِ المَعْنى، يَقولُ أدي شيرُ: "الخُرْشُومُ: مُرَكَّبةٌ مِنْ (خر وهِيَ أداة بها يوصف الشَّيْء بالغلاظة + سم: حافر)، وتُطْلَقُ على أنْفِ الجَبَلِ المُشْرِفِ على وادٍ أو قاعٍ".

معجم الألفاظ الفارسية المعربة – أدي شيرُ: – ص 53— المصدر
8.2معنى من معاني «خُرْشوم»

* الخُرْشومُ: الجَبَلُ العَظيمُ، أَوْ أَنْفُهُ الـمُشْرِفُ عَلى وادٍ أَوْ قاعٍ. (ج) خَراشِمُ، وخَراشيمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العُدَيْلُ بْنُ الفُرْخِ العِجْلِيُّ (ت: 100هـ=719م) يَصِفُ إِبِلًا:

يَنْوينَ فَرْجَ ثَقيفٍ في أَرومَتِها
إِذا ثَقيفٌ سَمَتْ مِنْها الخَراشيمُ

ابن المبارك، منتهى الطلب، تح: طريفي، ج: 7، ص: 91.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بنُ أَحْمَدَ (ت: 170هـ=787م):

"الخُرْشومُ: أَنْفُ الجَبَلِ الـمُشْرِفُ عَلى وادٍ أو قاعٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 325.— المصدر
8.3معنى من معاني «خُرْشوم»

وــــــــــ: ما غَلُظَ مِنَ الأَرْضِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"الخُرْشومُ: ما غَلُظَ مِنَ الأَرْضِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

8.4معنى من معاني «خُرْشوم»

وــــــــــ: نَوعٌ مِن الخَضْراواتِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد فَريد (ت: 1338ه=1919م):

"والحالُ أَنَّ ما كانَ مِثْلَ الوَرْدِ والخُرْشومِ مِنَ الأَزْهارِ والخَضْراواتِ والفَواكِهِ، الَّتي يَتَلاحَقُ ظُهورُ مَحْصولاتِها، يَصِحُّ فيه البَيْعُ إٍذا كانَ بَعْضُ مَحْصولاتِها ظَهَرَ وبَعْضُها لَمْ يَظْهَرْ".

محمد فريد، تاريخ الدولة العلية العثمانية، تح: إحسان حقي، ص: 552.— المصدر
