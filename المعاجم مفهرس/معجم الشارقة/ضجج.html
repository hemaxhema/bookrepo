01 —

ضَجَّ

فعل
1.1معنى من معاني «ضَجَّ»

* ضَجَّ(كَضَرَبَ) فُلانٌ يَضِجُّ (بِكَسْرِ الضّادِ) ضَجًّا، وضَجيجًا، وضَجاجًا، وضُجاجًا: جَلَّبَ وصاحَ مِنْ مَشَقَّةٍ أَوْ جَزَعٍ أَوْ نَحْوِهِما، فَهُوَ ضاجٌّ، وضَجّاجٌ، وضَجوجٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَنْتَرَةُ (ت: 22 ق هـ=601م) يَفْخَرُ بِثَباتِهِ في الحَرْبِ حينَ فَزِعَ النّاسُ:

ضَجّوا فَصِحْتُ عَلَيْهِمُ فَتَجَمَّعوا
ودَنا إِلَيَّ خَميسُ ذاكَ العَسْكَرِ

التبريزي، شرح ديوان عنترة، تح: طراد، ص: 85.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحُطَيْئَةُ (ت: 45هـ=665م) يَصِفُ قَوْمَهُ:

مَطاعينُ في الهَيْجاءِ بيضٌ وُجوهُهُمْ
إِذا ضَجَّ أَهْلُ الرَّوْعِ ساروا وهُمْ وُقْرُ 

ديوان الحطيئة، تح: قميحة، ص: 84.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بْنُ بُرْدٍ (ت: 167هـ=783م) يُبَيِّنُ حالَ بَعْضِ النّاسِ الّذينَ لا مَنْفَعَةَ مِنْهُمْ:

وفي القَوْمِ ميلاعٌ ولَيْسَ بِنافِعٍ
يَضِجُّ كَما ضَجَّ القَعودُ المُحَدَّجُ

ديوان بشار، تح: إحسان عباس، ص: 169.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ خَلِّكانَ (ت: 681هـ=1282م) يَصِفُ تَبَرُّمَ ابْنِ دُرَيدٍ بالزُّوّارِ في مَرَضِهِ:

"فَكانَ إِذا دَخَلَ عَلَيْهِ الدَّاخِلُ ضَجَّ".

ابن خلكان، وفيات الأعيان، تح: إحسان عباس، ج: 4، ص: 326.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجَبَرْتِيُّ (ت: 1237هـ=1822م) يُعَلِّقُ عَلى سَرِقاتِ الأَمْتِعَةِ بِخُطَّةِ الجامِعِ الأَزْهَرِ:

"وتَكَرَّرَ ذلِكَ حَتّى ‌ضَجَّ النَّاسُ وكَثُرَ لَغَطُهُمْ".

الجبرتي، عجائب الآثار، دار الجيل، ج: 3، ص: 347.— المصدر
1.2معنى من معاني «ضَجَّ»

وــــــــــ النّاقَةُ: ضَجِرَتْ عِنْدَ الحَلْبِ، فَهِيَ ضَجوجٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو صَخْرٍ الهُذَلِيُّ (ت: 80هـ=699م) يَذْكُرُ سَحابَةً شَبّهَها بِالنّاقَةِ:

حَدَتْ مُزْنَةٌ مِنْ حَضْرَموتَ مَرِيَّةٌ
ضَجوجٌ لَهُ مِنْها مُدِرٌّ وحالِبُ

البكري، معجم ما استعجم، تح: السقا، ص: 455.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَوْهَرِيُّ (ت: 393هـ=1003م):

"الضَّجُوجُ مِنَ النّوقِ: الّتي تَضِجُّ إِذا حُلِبَتْ". (وهذِهِ الدّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح، تح: عطار، ج: 1، ص: 326.— المصدر
1.3معنى من معاني «ضَجَّ»

وــــــــــ الجَمَلُ ونَحْوُهُ بِالحِمْلِ، ومِنْهُ: أَطَّ وأَعْيا وناءَ بِهِ.

ق

”قالَ امْرُؤُ القَيْسِ (ت: 80ق هـ=544م) يَصِفُ سَيْرَهُمُ المُتَعَجِّلَ:

بِسَيرٍ يَضِجُّ العَوْدُ مِنْهُ يَمُنُّهُ
أَخو الجَهْدِ لا يُلْوي عَلى مَن تَعَذَّرا

ديوان امرئ القيس، تح: أبو الفضل، ص: 62.— المصدر

"فَكَأَنّي بِكَ وقَدْ رَأَيْتُكَ تَضِجُّ مِنَ الحَرْبِ إِذا عَضَّتْكَ ‌ضَجيجَ ‌الجِمالِ ‌بِالأَثْقالِ ...".

وــــــــــ فُلانٌ مِنَ الشَّيْءِ: فَزِعَ مِنْهُ وضاقَتْ نَفْسُهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَمْرُو بْنُ كُلْثومٍ (ت: 40ق هـ=583م) يَفْخَرُ:

مَعَاذَ الإِلهِ أَنْ تَنوحَ نِساؤُنا
عَلى هالِكٍ أَوْ أَنْ تَضِجَّ مِنَ القَتْلِ

ديوان عمرو بن كلثوم، تح: ميدان، ص: 320.— المصدر

الإسلامي(1ه‍/622م) إلى (132ه‍/749م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البَطَلْيَوْسِيُّ (ت: 521هـ=1127م) يَذْكُرُ فَضْلَ البُخارِيِّ ومُسْلِمٍ وابْنِ مَعينٍ في تَنْقيحِ الأَحاديثِ:

"ولِلبُخارِيِّ -رَحِمَهُ اللهُ- في هذا البابِ غَناءٌ مَشْكورٌ وسَعْيٌ مَبْرورٌ، وكَذلِكَ لِمُسْلِمٍ، وابْنِ مَعينٍ، فَإِنَّهُمُ انْتَقَدوا الحَديثَ وحَرَّروهُ ونَبَّهوا عَلى ضُعَفاءِ المُحَدِّثينَ والمُتَّهَمينَ بِالكَذِبِ ‌حَتّى ‌ضَجَّ ‌مِنْ ‌ذلِكَ مَنْ كانَ في عَصْرِهِمْ".



د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الذَّهَبِيُّ (ت: 748هـ=1347م) يَذْكُرُ بَعْضَ ما فَعَلَهُ السُّلْطانُ ‌‌سَنَةَ خَمْسٍ وأَرْبَعينَ وأَرْبَعِمِائَةٍ:

"وفيها أَعْلَنَ بِنَيْسابورَ بِلَعْنِ أَبي الحَسَنِ الأَشْعَرِيِّ، ‌فَضَجَّ مِنْ ‌ذلِكَ ‌الشَّيْخُ ‌أَبو ‌القاسِمِ القُشَيْرِيُّ، وصَنَّفَ رِسالَةَ (شِكايَةُ السُّنَّةِ لِما نالَهُمْ مِنَ المِحْنَةِ)".

الذهبي، تاريخ الإسلام، تح: التدمري، ج: 30، ص: 13.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد شَوْقي (ت: 1351هـ=1932م) يَعيبُ عَلى بَعْضِ الحَجيجِ أَكْلَ أَمْوالِ النّاسِ بالباطِلِ:

"وهَلْ عَلِمْتَ أَنَّ اللهَ لا يَقْبَلُ مِنْكَ مالًا، ونَفَقَةُ المُطَلَّقَةِ مِنْ مَطْلٍ مُعَلَّقَةٌ، وذو القُرْبى وراءَكَ جائِعٌ، والوَلَدُ طَريدُ المَدارِسِ ضائِعٌ ... وجارُكَ الضَّعيفُ ‌يَضِجُّ ‌مِنْ حَيْفِكَ".

أحمد شوقي، أسواق الذهب، ص: 87.— المصدر
1.5معنى من معاني «ضَجَّ»

وــــــــــ إِلى اللهِ وغَيْرِهِ: استَغاثَهُ وجَأَرَ إِلَيْهِ بِالدُّعاءِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُخْبِرُ عَنْ بَعْضِ ما رآهُ لَيْلَةَ المِعْراجِ:

«ثُمَّ مَضَيْتُ هُنَيَّةً فَإِذا أَنا بِنِساءٍ مُعَلَّقاتٍ بِثُدِيِّهِنَّ وسَمِعْتُهُنَّ يَضْجِجْنَ إِلى اللهِ، قُلْتُ: يا جِبْريلُ مَنْ هؤُلاءِ؟! قالَ: هؤُلاءِ الزُّناةُ مِنْ أُمَّتِكَ».

القرطبي، التذكرة، تح: الصادق، ج: 1، ص: 404.— المصدر
”وقالَ ابْنُ عَبّاسٍ (ت: 68هـ=687م) يَذْكُرُ نُفورَ المَخْلوقاتِ مِنَ المُرائينَ :

"إِنَّ الأَرْضَ ‌لَتَضِجُّ ‌إِلى ‌رَبِّها مِنَ الَّذينَ يَلْبَسونَ الصّوفَ رياءً".

ابن حجر، لسان الميزان، تح: دائرة المعارف النظامية بالهند، ج: 7، ص: 36.— المصدر

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سَلّامٍ الجُمَحِيُّ (ت: 232هـ=846م) يَتَحَدَّثُ عَنِ البَعيثِ المُجاشِعِيِّ:

"وكانَ البَعيثُ شَاعِرًا فَاخِرَ الكَلامِ، حُرَّ اللَّفْظِ، وقَدْ غَلَبَهُ جَريرٌ وأَخْمَلَهُ. وكانَ قَدْ قاوَمَ جَريرًا في قَصَائِدَ، ‏ثُمَّ ضَجَّ إِلى الفَرَزدَقِ واسْتَغاثَهُ".

ابن سلام، طبقات فحول الشعراء، تح: محمود شاكر، ج: 2، ص: 535.— المصدر

الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)

"(يَجْأَرونَ) يَجْزَعونَ، أَوْ يَسْتَغيثونَ، أَوْ ‌يَضِجّونَ، أَوْ يَصْرُخونَ إِلى اللهِ -تعالى- بِالتَّوْبَةِ فَلا تُقْبَلُ مِنْهُمْ".

العز بن عبد السلام، تفسير القرآن، تح: الوهيبي، ج: 2، ص: 378.— المصدر

7.2معنى من معاني «ضَجاج/ ضِجاج»

وــــــــــ: صَمْغٌ يُؤْكَلُ رَطْبًا، فَإِذا جَفَّ كُتِّلَ وَقُوِّيَ بِالقَلْيِ واسْتُعْمِلَ كَالصّابونِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُمَيْدُ بْنُ ثَوْرٍ (ت: 60هـ=680م) يَتَغَزَّلُ:

أَلا إِنَّما هِنْدُ جِنِّيَّةٌ
وطَعْمُ الضَّجاجِ وطَعْمُ العَسَلْ

ديوان حميد بن ثور، تح: البيطار، ص: 346.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"الضَّجاجُ: صَمْغٌ يُؤْكَلُ رَطْبًا فَإِذا جَفَّ سُحِقَ ثُمَّ كُتِّلَ وَقُوِّيَ بالقَلْيِ ثُمَّ غُسِلَ بِهِ الثَّوْبُ، فَيُنَقّي تَنْقِيَةَ الصّابونِ". (وهذِهِ الدّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: علي حسن، ج: 10، ص: 447.— المصدر
7.3معنى من معاني «ضَجاج/ ضِجاج»

وــــــــــ مِنَ النّاسِ: الشَّديدُ المُشاغَبَةِ والمُشارَّةِ، (ج) ضُجُجٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ لَيْلى الأَخْيَلِيَّةُ (ت: 85هـ=704م) تَرْثي:

كَأَنَّ فَتى الفِتْيانِ تَوْبَةَ لَمْ يَسِرْ
بِنَجْدٍ ولَمْ يَطْلُعْ مَعَ المُتَغَوِّرِ

ولَمْ يَغْلِبِ ‌الخَصْمَ ‌الضِّجاجَ ويَمْلَإِ الـ
ـجِفانَ سَديفًا يَوْمَ نَكْباءَ صَرْصَرِ

ديوان ليلى الأخيلية، تح: واضح الصمد، ص: 47.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"رَجُلٌ ضَجَاجٌ، وقَوْمٌ ضُجُجٌ". (وهذِهِ الدّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم، تح: هنداوي، ج: 7، ص: 176.— المصدر
7.4معنى من معاني «ضَجاج/ ضِجاج»

وــــــــــ: المُشاغَبَةُ والمُشارَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ قُتَيْبَةَ (ت: 276هـ=889م):

"الضَّجاجُ: المُشاغَبَةُ". (وهذِهِ الدّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، الجراثيم، تح: الحميدي، ج: 1، ص: 349.— المصدر
08 —

ضَجَّة

اسم
8.1معنى من معاني «ضَجَّة»

o ضَجَّةٌ إعْلامِيَّةٌ: دِعايَةٌ واسِعَةٌ في الصُّحفِ والتّلفزيونِ عن عَمَلٍ ما، أو عَنْ حادِثَةٍ أو عَنْ شَخْصٍ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد مُختار (ت: 1424ه=2003م):

"ضَجَّةٌ إعْلامِيَّةٌ: دِعايَةٌ واسِعَةٌ في الصُّحفِ والتّلفزيونِ عن عَمَلٍ ما، أو عَنْ حادِثَةٍ أو عَنْ شَخْصٍ".

أحمد مختار، معجم اللغة العربية المعاصرة. ج: 2، ص: 1347.— المصدر
