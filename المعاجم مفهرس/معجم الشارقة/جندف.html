جندف
جُنادِف(اسم) جُنادِفةٌ(فعل) جُنْدُف(اسم)
01 —

جُنادِف

اسم
1.1معنى من معاني «جُنادِف»

*الجُنادِفُ:الجافي الجَسيمُ مِنَ النّاسِ والإِبِلِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بنُ حَجَرٍ (ت: 2 ق ه=620م) يَصِفُ صائِدًا:

أَزَبُّ ظُهورِ السَّاعِدَيْنِ عِظامُهُ
على قَدَرٍ شَثْنُ البَنانِ جُنادِفُ

ديوان أوس بن حجر. تح: محمّد يوسف نجم، ص: 70.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُمَيْدُ بنُ ثَوْرٍ الهِلاليُّ (ت: 30ه=651م):

لَقَدْ تَسَرَّيْتُ إِذا الهَمُّ وَلَجْ

واجْتَمَعَ الهَمُّ هُمومًا، واعْتَلَجْ

جُنادِفَ الـمِرْفَقِ مَبْنِيَّ الثَّبَجْ

ديوان حميد بن ثور. تح: محمد البيطار، ص: 259.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الجُنادِفُ: الجافي الجَسيمُ مِنَ النّاسِ، والإِبِلِ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 1، ص: 266.— المصدر
”وقالَ الشّاعِرُ (180ه=796م) مِنْ شواهِدِ اللّيْثِ بْنِ المُظَفَّرِ:

وقَرَّبوا كُلَّ جُلالٍ دَخْنَسِ

عِنْدَ القِرى جُنادِفٍ عَجَلَّسِ

الأزهري، تهذيب اللغة. تح: عبد السلام سرحان، ج: 7، ص: 661.— المصدر
1.2معنى من معاني «جُنادِف»

وــــــــــ مِنَ النّاسِ: الغَليظُ الخِلْقَةِ القَصيرُ الـمُلَزِّزُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بنُ حَجَرٍ (ت: 2 ق ه=620م) يَصِفُ صائِدًا:

أَزَبُّ ظُهورِ السَّاعِدَيْنِ عِظامُهُ
على قَدَرٍ شَثْنُ البَنانِ جُنادِفُ

ديوان أوس بن حجر. تح: محمّد يوسف نجم، ص: 70.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البَنْدَنيجِيُّ (ت: 284ه=897م):

"الجُنادِفُ: الغَليظُ الـمُجْتَمِعُ الـمُلَزَّزُ مِنَ الرِّجالِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بقيَّةِ العُصورِ)

البندنيجيُّ، التقفية. تح: خليل إبراهيم العطية، ص: 130.— المصدر
1.3معنى من معاني «جُنادِف»

وــــــــــ: الغَليظُ القَصيرُ الرَّقَبَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْرِيُّ (ت: 90ه=709م)

جُنادِفٌ لاحِقٌ بالرَّأْسِ مَنْكِبُهُ
كَأَنَّهُ كَوْدَنٌ يوشي بِكُلّابِ

ديوان الرّاعي النّميريّ. تح: واضح الصّمد، ص: 42.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216ه=831م):

"رَجُلٌ جُنادِفٌ: غَليظٌ قَصيرُ الرَّقَبَةِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بقيَّةِ العُصورِ)

الأزهريّ، تهذيب اللغة، تح: محمد أبو الفضل، ج: 11، ص: 252.— المصدر
1.4معنى من معاني «جُنادِف»

وــــــــــ: الذي إِذا مَشى حَرَّكَ كَتِفَيْهِ، وهُوَ مَشْيُ القِصارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"ويُقالُ: إِنَّ الجُنادِفَ القَصيرُ الذي إِذا مَشى حَرَّكَ كَتِفَيْهِ، وهُوَ مَشْيُ القِصارِ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكّي، ج:2، ص: 1212.— المصدر
02 —

جُنادِفةٌ

فعل
2.1معنى من معاني «جُنادِفةٌ»

*الجُنادِفَةُ:الجافِيَةُ الجَسيمَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الجُنادِفُ: الجافي الجَسيمُ مِنَ النّاسِ، والإِبِلِ، يُقالُ: ناقَةٌ جُنادِفَةٌ، وأَمَةٌ جُنادِفَةٌ، ولا توصَفُ بِهِ الحُرَّةُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 1، ص: 266.— المصدر
2.2معنى من معاني «جُنادِفةٌ»

وــــــــــ مِنَ النّوقِ: السَّمينَةُ القَوِيَّةُ الظَّهْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"ناقَةٌ جُنادِفَةٌ: سَمينَةٌ ظَهيرَةٌ".

ابن عباد، المحيط. تح: محمد حسن آل ياسين، ج: 7، ص: 230.— المصدر
2.3معنى من معاني «جُنادِفةٌ»

وــــــــــ: الـمُحْتَقِرُ لِلأَشْياءِ مِنْ جَفاءِ خُلُقِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395ه=1005م):

"ومِنْ ذَلِكَ قَوْلُهُمْ لِلْجافي (جُنادِفٌ) فالنّونُ فيهِ زائِدَةٌ، والأَصْلُ: الجَدْفُ، وهُوَ احْتِقارُ الشَّيْءِ، يُقالُ: جَدَفَ بِكَذا؛ أَيْ: احْتَقَرَ، فكَأَنَّ الجُنادِفَ الـمُحْتَقِرُ لِلأَشْياءِ، مِنْ جَفائِهِ".

ابن فارس، مقاييس اللغة. تح: عبد السلام هارون، ج: 1، ص: 511.— المصدر
03 —

جُنْدُف

اسم
3.1معنى من معاني «جُنْدُف»

* الجُنْدُفُ:القَصيرُ الـمُلَزَّزُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"والجُنْدُفُ: القَصيرُ الـمُلَزَّزُ".(وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 7، ص: 592.— المصدر
