01 —

حَشْحَشَ

فعل
1.1معنى من معاني «حَشْحَشَ»

* حَشْحَشَ فُلانٌ: وطِئَ الأَرْضَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبيُّ ﷺ (ت: 11 هـ=632م) يُخاطِبُ بِلالًا:

«ما دَخَلْتُ الجَنّةَ قَطُّ إِلّا سَمِعْتُ حَشْحَشَتَكَ أَمامي».

الإمام السيوطي، إسعاف المبطأ برجال الموطأ، ج: 1، ص: 7.— المصدر
1.2معنى من معاني «حَشْحَشَ»

وــــــــــ القَوْمُ: تَحَرَّكوا مُسْرِعينَ ودَخَلَ بَعْضُهُم في بَعْضٍ.

—
غير محدد
” قالَ أبو بَكْرِ بْنُ دُرَيْدٍ (ت: 321ه=933م):

"حَشْحَشَ القَوْمُ حَشْحَشةً: إذا تَحَرَّكوا، ودَخَلَ بَعْضُهُمْ في بَعْضٍ".

السرقسطي، كتاب الأفعال. تح: حسين شرف، ج: 1، ص: 427.— المصدر
1.3معنى من معاني «حَشْحَشَ»

وــــــــــ فُلانٌ الشَّيْءَ: حَرَّكَهُ أَوْ خَضْخَضَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ تَأَبَّطَ شَرًّا (ت: 82 ق هـ=540م) يَفْتَخِرُ بِسُرْعَةِ فِرارِهِ:

وحَشْحَشْتُ مَشْعوفَ النَّجاءِ وراعَني
أُناسٌ بِفَيْفانٍ فَمِزْتُ القَرائِنا

[ورِوايَةُ الدِّيوانِ: حَثْحَثَ]
ابْن سِيدَه، المحْكَم. تح: هنداوي، ج: 6، ص: 366.— المصدر
1.4معنى من معاني «حَشْحَشَ»

وــــــــــ النّارُ الشَّيْءَ: أَحْرَقَتْهُ. (انْظُرْ: ح س ح س، ح ش ش)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابيّ (ت: 231 هـ= 845م):

"يُقالُ: حَسْحَسَتْهُ النّارُ، وحَشْحَشَتْهُ بِمَعْنًى".

الأَزْهَريّ، تَهْذيبُ اللُّغَةِ. تح: درويش، ج: 3، ص: 409.— المصدر
”وقالَ إِبْراهِيمُ الحَضْرَمِيُّ (475 هـ=1082م) يَمْدَحُ أبا الفَضْلِ العَبّاسَ بْنَ مَعْنٍ:

كَمِثْلِكَ حَشْحاشٌ فَحَشْحِشْ لَظَى الوَغَى
أَبا الفَضْلِ حَشْحِشْها أَبا الفَضْلِ هَيِّجِ

ديوانه. تح: بدر اليحمدي، ج: 1، ص: 123.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أَحْمَدُ بْنُ مُحَمَّدٍ الوِرْغيُّ (ت: 1190ه=1776م) يَمْدَحُ قائدًا:

ونادى فَلَبَّتهُ الدَّواعي وأَبْطَأَتْ
نَعَمْ رَيْثَما خَطَّتْ كِتابَ الكَتائِبِ

وحَشْحَشَها جُرْدًا تُخُيِّرْنَ لِلنَّوى
وكومَ ذُرًى مَعْدودَةً لِلنَّوائِبِ

ديوانه. ج: 1، ص87:.— المصدر
1.5معنى من معاني «حَشْحَشَ»

وــــــــــ فُلانٌ اللَّحْمَ: وضَعَهُ عَلى النّارِ حَتّى سُمِعَ لَهُ صَوْتٌ وتَقَبَّضَ. (انظر: ح س ح س)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395 هـ=1005م):

"ومِنْ هَذا حَسْحَسْتُ الشَّيْءَ مِنَ اللَّحْمِ: إِذا جَعَلْتَهُ عَلى الجَمْرَةِ، وحَشْحَشْتُ أَيْضًا".

ابْن فارس، مَقاييس اللّغة. تح: هارون، ج: 2، ص: 9.— المصدر
02 —

تَحَشْحَشَ

فعل
2.1معنى من معاني «تَحَشْحَشَ»

* تَحَشْحَشَ النَّباتُ: يَبِسَ. (انظر: ح ش ش)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَبو دُؤادٍ الإياديُّ (ت: 79ق هـ=545م) يَصِفُ خَيْلًا:

في كُلِّ مَنْزِلةٍ وكُلِّ مُعَرَّسٍ
سَخْلٌ تَناسَلَهُ الزِّجاجُ مِنَ الصَّلا

مُهْرٌ يُؤَبَّنُ هالِكًا أَوْ مُهْرةٌ
كالفِلْقِ سُلَّ مِنَ القَرابِ قَدِ انْحَنى

وكَأَنَّ أَسْلاءَ الجِيادِ شَقائِقٌ
أَوْ عُتْرُفانٌ قَدْ تَحَشْحَشَ لِلبِلى

ديوانه. تح: أنوار الصالحي وآخر، ص: 186.— المصدر
2.2معنى من معاني «تَحَشْحَشَ»

وــــــــــ فُلانٌ: تَحَرَّكَ لِلنُّهوضِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (11 هـ=632م):

"تَزَوَّجْتُ فاطِمَةَ، فَدَخَلَ عَلَيْنا رَسولُ اللهِ ﷺ وعَلَيْنا كِساءٌ أَوْ قَطِيفَةٌ، فَلَمّا رَأَيْناهُ تَحَشْحَشْنا".

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 1، ص: 673.— المصدر
”وقالَ حَسّانُ بْنُ ثابِتٍ (ت: 54 هـ=673م) يَهْجو قَوْمًا:

هَيَّجْتُمُ حَسّانَ عِنْدَ ذَكائِهِ
غَيٌّ لِمَنْ ولَدَ الحِماسُ طَويلُ

إِنَّ الهِجاءَ إِلَيْكُمُ لَبِعِلَّةٍ
فَتَحَشْحَشوا إِنَّ الذَّليلَ ذَليلُ

ديوانه. تح: وليد عرفات، ج: 1، ص: 217.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعيُّ (ت: 216 هـ=831م):

"تَحَشْحَشَ القَوْمُ: إِذا تَحَرَّكوا".

الخَطّابيّ، غَريب الحَديث. تح: العزباوي، ج: 1، ص: 292.— المصدر
”وقالَ الطَّبَريُّ (ت: 310 هـ=922م) يَصِفُ انْتِظارَ النُّعْمانِ بْن مُقَرِّنٍ دُخولَ السّاعاتِ التي كانَ يُحبُّ رَسولُ اللهِ ﷺ القِتالَ فيها:

"وجَعَلَ النُّعْمانُ يَنْتَظِرُ بِالْقِتَالِ إكْمالَ ساعاتٍ كانَتْ أَحَبَّ إلى رَسولِ الله ﷺ في القِتالِ أَنْ يَلْقى فيها العَدُوَّ، وذَلِكَ عِنْدَ الزَّوالِ وتَفَيُّؤِ الأَفْياءِ ومَهَبِّ الرِّياحِ، فَلَمّا كانَ قَريبًا مِنْ تِلْكَ السّاعَةِ، تَحَشْحَشَ النُّعْمانُ، وسارَ في النّاسِ على بِرْذَوْنٍ أَحْوى قَريبٍ مِنَ الأَرْضِ".

الطبري، تاريخ الأمم والملوك. ج: 4، ص: 131.— المصدر
2.3معنى من معاني «تَحَشْحَشَ»

وــــــــــ: هَزَلَ وضَعُفَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ قَيْلَةُ بْنتُ مَخْرَمَةَ (11 هـ=632م) بَعْدَ أَنْ لَحِقَ بِها أَثْوَبُ بْنُ مالِكٍ في طَريقِ وِفادَتِها عَلى النَّبِيِّ ﷺ:

"وقال: أَلْقي إِلَيَّ ابْنَةَ أَخي يا دَفارِ، فَأَلْقَيْتُهَا إِلَيْهِ وكُنْتُ أَعْلَمَ بِهِ مِنْهُم، وقَدْ تَحَشْحَشَ لَهُ القَوْمُ".

ابن طيفور، بلاغات النِّساء. تح: أحمد الألفي، ج: 1، ص: 122.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ طَيْفورٍ (ت: 280ه=893م):

"قال أَبو عَبْدِ اللهِ في قَوْلِها: (تَحَشْحَشَ لَهُ القَوْمُ) إنَّ التَّحَشْحُشَ أَنْ يَهْزَلَ الرَّجُلُ بَعْدَ يُبْسٍ. قال العُقَيْلِيُّ: قَدْ تَحَشْحَشْنا في آخِرِ هذا الشَّهْرِ، يَعْني شَهْرَ رَمَضانَ، أَيْ: يَبِسْنا وهَزَلْنا وقَحِلْنا مِنَ الصِّيامِ".

ابن طيفور، بلاغات النِّساء. تح: أحمد الألفي، ج: 1، ص: 123.— المصدر
”قالَ أَبو عَليٍّ المُسوحيُّ (ت: 261 هـ=875م):

"فَرَأَيْتُ كَأَنَّ سَقْفَ المَسْجِدِ قَدِ انْشَقَّ، وكَأَنَّ جارِيَةً قَدْ نَزَلَتْ عَلَيَّ مِنَ السَّقْفِ عَلَيْها قَميصُ فِضَّةٍ يَتَحَشْحَشُ".

ابن الجوزي، صفة الصّفوة. تح: خالد طرطوسي، ج: 1، ص: 469.— المصدر
2.4معنى من معاني «تَحَشْحَشَ»

وــــــــــ فُلانٌ: تَجَمَّعَ وتَقَبَّضَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللَّيثُ (ت: 180 هـ=796م):

"وتَحَشْحَشَ الرَّجُلُ: إِذا تَجَمَّعَ وتَقَبَّضَ".

السرقسطي، الدلائل في غريب الحديث. تح: محمد القناص، ج: 1، ص: 674.— المصدر
2.5معنى من معاني «تَحَشْحَشَ»

وــــــــــ القَوْمُ: اجْتَمَعوا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سَعْدٍ (ت: 230 هـ=845م) يرْوي حادِثَ مَكْرِ القَوْمِ بِعُبَيْدِ اللهِ بْنِ زِيادٍ لِيَقْتُلوهُ إِذا دَخَلَ عَلَيْهِمْ:

"وتَحَشْحَشَ القَوْمُ في البَيْتِ، فَأَنْكَرَ عُبَيْدُ اللهِ ما رَأَى مِنْهُمْ".

ابن سعد، الطبقات الكبرى. تح: علي عمر، ج: 6، ص: 432.— المصدر
2.6معنى من معاني «تَحَشْحَشَ»

وــــــــــ القَوْمُ: تَفَرَّقوا. (ضِدٌّ).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385 هـ=995م):

"تَحَشْحَشَ القَوْمُ: تَفَرَّقُوا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 295.— المصدر
03 —

حَشْحاش

اسم
3.1معنى من معاني «حَشْحاش»

* الحَشْحاشُ: الشَّيْءُ الكَثيرُ الحَرَكَةِ والتَّداخُلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سَعيدُ بْنُ هاشِمٍ الخالِديُّ (ت: 371 هـ=981م) يَهْجو أبا رِياشٍ بِسُوءِ نَظافَتِهِ:

كَأَنَّما قَمْلُ أَبي رِياشِ

ما بَيْنَ صِئْبانِ قَفاهُ الفاشي

وذا وذا قَدْ لَجَّ في انْتِفاشِ

شُهْدانِجٌ بُدِّدَ في حَشْحاشِ

الثعالبي، يتيمة الدهر. تح: مفيد قميحة، ج: 2، ص: 412.— المصدر
3.2معنى من معاني «حَشْحاش»

وــــــــــ: البَطَلُ المِقْدامُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ إِبْراهِيمُ الحَضْرَمِيُّ (475 هـ=1082م) يَمْدَحُ أبا الفَضْلِ العَبّاسَ بْنَ مَعْنٍ:

كَمِثْلِكَ حَشْحاشٌ فَحَشْحِشْ لَظَى الوَغَى
أَبا الفَضْلِ حَشْحِشْها أَبا الفَضْلِ هَيِّجِ

ديوانه. تح: بدر اليحمدي، ج: 1، ص: 123— المصدر
3.3معنى من معاني «حَشْحاش»

وــــــــــ: الشَّيْءُ الخَفيفُ. (انْظُرْ: ع س ع س).

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ قَيّمِ الجَوْزِيَّةِ (ت: 751 هـ=1350م):

"وحَشْحاشٌ وعَسْعاسٌ، أَيْ: خَفيفٌ".

ابن القيم، التفسير القيم. تح: محمد الفقي، ج: 1، ص: 604.— المصدر
