نأش
نَأَشَ(فعل) انْتَأَشَ(فعل) تَناءَشَ(فعل) نَئيش(اسم) مَنْؤوشَة(اسم)
النظائر الساميّة

احْتَفَظَتِ السّاميّاتُ بهذا الجَذْرِ مِنَ العَرَبيَّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذْرُ (ن أ ش):

في العِبْريَّةِ؛ ومِنْهُ الفِعْلُ: נֵאֵשׁ (ن۪ٮٰـئ۪ٮٰـش) nēʼēš بِمَعْنى: "أجْهَدَ، أنْهَكَ، فَتَكَ، بَطَشَ"، وهُوَ يُوافِقُ الفِعْلَ العَرَبيَّ (نَأَشَ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَأَشَ" ما وَرَدَ في المِدْراشيمِ: "למה קורא אותה וכאב אנוש... שהיא מנאשת את הגוף שהיא נכנסת בו‏"؛ أَيْ: "لماذا يَصِفُهُ بـ "الألَمِ الشَّديدِ"... لأنَّهُ يَنْأَشُ الجَسَدَ الذي يَنالُ مِنْهُ".

(במ"ר: ז', ד')

في السُّرْيانيَّةِ؛ ومِنْهُ الفِعْلُ: ܢܥܰܓ̥ (نْعَغ) nəʽaġ بِمَعْنى: "سَحَبَ بَعيدًا/ باعَدَ"؛ وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَأَشَ) مَبْنًى ومَعْنًى، غَيْرَ أَنَّ الفِعْلَ السُّرْيانيَّ خَضَعَ لإِبْدالِ الهَمْزَةِ عَيْنًا، والشّينِ جيمًا؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَأَشَ": "ܘܣܥܘ ܢܥܓܘ ܥܪ̈ܒܐ‏"؛ أَيْ: "لَقَدْ تَجَرَّؤوا ونَأَشوا الكِباشَ".

(ESHaer 199:22)

01 —

نَأَشَ

فعل
1.1معنى من معاني «نَأَشَ»

* نَأَشَ (بِفَتْحِ الهَمْزَةِ) فُلانٌ، يَنْأَشُ (بِفَتْحِ الهَمْزَةِ)، نَأشًا: أَبْطَأَ، في النُّهوضِ وغَيْرِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110ه=728م):

إِلَيْكَ نَأَشْتُ يَا بْنَ أَبي عَقيلٍ
ودوني الغافُ غافُ قُرى عُمانِ

[هكَذا رَواهُ ابْنُ مَنْظورٍ، ورَواهُ ابْنُ دُرَيْدٍ "إِلَيْكَ رَحَلْتُ" بِلا نِسْبَةٍ، ولَيْسَ في ديوان الفرزدق]
ابن منظور، لسان العرب، ج: 9، ص: 273.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207ه=822م) يُفَسِّرُ قِراءَةَ ﴿‌وَأَنَّىٰ ‌لَهُمُ ٱلتَّنَاؤُشُ مِن مَّكَانِۭ بَعِيدٖ﴾ [سَبَأ:52]:

"وقَرَأَ الأَعْمَشُ وحَمْزَةُ والكِسائِيُّ: ﴿ٱلتَّنَاؤُشُ﴾ بِالهَمْزَةِ يَجْعَلونَهُ مِنْ نَأَشْتُ، وهُوَ البُطْءُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد أبو الفضل، ج: 11، ص: 417.— المصدر
1.2معنى من معاني «نَأَشَ»

وــــــــــ: تَباعَدَ وتَأَخَّرَ، فهُوَ نَئيشٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ثَعْلَبٌ (ت: 291ه=904م):

"والنَّئِيشُ أَيْضًا: البَعيدُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[نَصَّ على الفِعْلِ السَّرَقُسْطِيُّ وأَبو بَكْرٍ الأَنْبارِيُّ]
ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 93.— المصدر
1.3معنى من معاني «نَأَشَ»

وــــــــــ الأَمْرَ، والشَّيْءَ: أَخَّرَهُ وباعَدَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مُحَمَّدٍ الأُمَوِيّ (ت: 203ه=818م):

"نَأَشْتُ الشَّيْءَ أَخَّرْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

إبراهيم الحربي، غريب الحديث. تح: العايد، ج: 2، ص: 883.— المصدر
1.4معنى من معاني «نَأَشَ»

وــــــــــ: طَلَبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ الـمُثَنّى (ت: 209ه=825م):

"نَأَشْتُ: طَلَبْتُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، تفسير غريب القرآن. تح: أحمد صقر، ص: 359. — المصدر
1.5معنى من معاني «نَأَشَ»

وــــــــــ: تَناوَلَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"ونَأَشْتُهُ أَنْأَشُهُ ‌نَأْشًا، إِذا تَناوَلْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ص: 882.— المصدر
1.6معنى من معاني «نَأَشَ»

وــــــــــ: أخَذَهُ في بَطْشٍ، فهُوَ نَؤوشٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"النَّأْشُ: الأَخْذُ والبَطْشُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 384.— المصدر
1.7معنى من معاني «نَأَشَ»

وــــــــــ اللهُ فُلانًا: أَحْياهُ ورَفَعَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"ونَأَشَهُ اللهُ ‌نَأْشًا ‌كنَعَشَهُ أَيْ أَحْياهُ ورَفَعَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 8، ص: 93.— المصدر
02 —

انْتَأَشَ

فعل
2.1معنى من معاني «انْتَأَشَ»

* انْتَأشَ فُلانٌ: تأخَّرَ وتباعَدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مُحَمَّدٍ الأُمَوِيّ (ت: 203ه=818م):

"نَأَشْتُ الشَّيْءَ أَخَّرْتُهُ، وانْتَأَشَ هُوَ: تَأَخَّرَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال. تح: شرف، ج: 3، ص: 232.— المصدر
2.2معنى من معاني «انْتَأَشَ»

وــــــــــ بِغَنَمِهِ: ظَعَنَ بِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وانْتَأَشَ فُلانٌ بِغَنَمِه كَرِعانِ السَّحابِ: أَيْ ظَعَنَ بِها وأتاها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 384.— المصدر
2.3معنى من معاني «انْتَأَشَ»

وــــــــــ الدّينَ وغَيْرَهُ: أقامَهُ وتَدارَكَهُ مِنَ الهَلَكَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”بِهِ رُوِيَ قَوْلُ عائِشَةَ أُمِّ الـمُؤْمِنينَ (ت: 58ه=678م) تَصِفُ أباها:

" فانْتَأَشَ الدّينَ بِنَعْشِهِ إِيّاهُ".

الزبيدي، تاج العروس. تح: حجازي، ج: 17، ص: 397.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ العُقَيْلِيّ (ت: 190ه=806م):

"انْتَأَشَهُ أَي أَدْرَكَهُ نَئيشًا أَيْ بَعْدَما كادَ يَهْلِكُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 258.— المصدر
2.4معنى من معاني «انْتَأَشَ»

وــــــــــ الشَّيْءَ: انْتَزَعَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيّ (ت: 206ه=821م):

"وانْتَأَشَهُ، أَيِ انْتَزَعَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد أبو الفضل، ج: 11، ص: 417.— المصدر
2.5معنى من معاني «انْتَأَشَ»

وــــــــــ فُلانًا: أعْجَلَهُ.

03 —

تَناءَشَ

فعل
3.1معنى من معاني «تَناءَشَ»

* تَناءَشَ فُلانٌ: تأَخَّرَ وأَبْطَأَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11ه=632م) - على قِراءَةِ ابْنِ كَثيرٍ وأبي عَمْرٍو وحَمْزَةَ والكِسائِيِّ وشُعْبَةَ- قالَ اللهُ :

﴿وَقَالُوٓاْ ءَامَنَّا بِهِۦ ‌وَأَنَّىٰ ‌لَهُمُ ٱلتَّنَاؤُشُ مِن مَّكَانِۭ بَعِيدٖ﴾ [سَبَأ: 52]

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسائِيُّ (ت: 189ه=805م):

"‌التَّناؤُشُ: التَّأَخُّرُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الحربي، غريب الحديث. تح: العايد، ج: 2، ص: 883.— المصدر
3.2معنى من معاني «تَناءَشَ»

وــــــــــ القَوْمُ: تنَاولَ بَعْضُهُمْ بَعْضًا بِالرِّماحِ ونَحْوِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207ه=822م):

"وتَناوَشَ القَوْمُ في القِتالِ إِذا تَناوَلَ بَعْضُهُمْ بَعْضًا ولَمْ يَتَدانَوْا كُلَّ التَّداني. وقَدْ يَجوزُ هَمْزُها وهِيَ مِنْ نُشْتُ لانْضِمامِ الواوِ، يَعْني ‌التَّناؤُشَ مِثْلَ قَوْلِهِ: ﴿وَإِذَا ٱلرُّسُلُ ‌أُقِّتَتۡ﴾ [المرسلات: 11] ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[في الـمَطْبوعِ: "يَعْني التَّناوُشَ" والسِّياقَ يَقْتَضي ما أُثْبِتَ]
الفراء، معاني القرآن. تح: النجار، ج: 2، ص: 365.— المصدر
3.3معنى من معاني «تَناءَشَ»

وــــــــــ فُلانٌ الشَّيْءَ: أَخَذَهُ مِنْ بَعيدٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو إِسْحاقَ الثَّعْلَبِيُّ (ت 427ه=1035م):

"يُقالُ: تَناءَشْتُ الشَّيْءَ أَيْ: أَخَذْتُهُ مِنْ بَعيدٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الثعلبي، الكشف والبيان. تح: الوذيناني وآخر، ج: 22، ص: 139.— المصدر
04 —

نَئيش

اسم
4.1معنى من معاني «نَئيش»

* النَّئيشُ: الحَرَكَةُ في إِبْطاءٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ نَهْشَلُ بْنُ حَرِّيّ (ت: 45ه=665م) يُنَدِّمُ مَنْ عَصاهُ:

تَمَنّى نَئيشًا أنْ يَكونَ أَطاعَني
وقَدْ حَدَثَتْ بَعْدَ الأُمورِ أُمورُ

ابن السكيت، الألفاظ. تح: قباوة، ص: 203.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيّ (ت: 206ه=821م):

"ويُقالُ: جاءَ نَئيشًا، أَيْ: بَطيئًا آخِرَ النّاسِ".

ابن السكيت، الألفاظ. تح: قباوة، ص: 203.— المصدر
4.2معنى من معاني «نَئيش»

وــــــــــ: القَليلُ مِنَ الشَّيْءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ البَكْرِيُّ (190ه=806م) مِنْ رُواةِ الشَّيْبانِيِّ:

"ما أَخَذْتُ إِلّا نَئيشًا أَيْ قَليلًا".

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 262.— المصدر
4.3معنى من معاني «نَئيش»

وــــــــــ: البَعيدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ إِبْراهيمُ الحَرْبِيُّ (ت: 285ه=898م):

"والنَّئيشُ: البَعيدُ".

الحربي، غريب الحديث. تح: العايد، ج: 2، ص: 883.— المصدر

ع
