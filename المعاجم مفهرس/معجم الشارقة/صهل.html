المقدّمة
النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ص ه ل):

في العِبْريّة: وَرَدَ الفِعْلُ: צָהַל (صاهَل) ṣāhal بِمَعْنى (صاحَ الفَرَسُ، حَمْحَمَ) يوافِقُ الفِعْلَ العَرَبيَّ (صَهَلَ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (صَهَلَ) ما وَرَدَ في سِفرِ إرميا: "סוּסִים מְיֻזָּנִים... יִצְהָלוּ"، أي: "أحْصِنَةٌ مَعْلُوفَةٌ... تَصْهَلُ".

(יר' 5: 8)

في السُّرْيانيّة: وَرَدَ الفِعْلُ: ܨܗܰܠ (صْهَل) ṣəhal بِمَعْنى (حَمحَمَ الفرسُ) يوافِقُ الفِعْلَ العَرَبيَّ (صَهَلَ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (صَهَلَ) ما وَرَدَ في سِفرِ إرميا: "ܐܰܝܟ݂ ܖ̈ܰܟ݂ܫܳܐ ܐܽܘܫ̈ܢܶܐ ܥܳܗܖ̈ܶܐ ܗܘܰܘ: ܓܒܰܪ ܥܰܠ ܐܰܢ̱ܬܰܬ݂ ܚܰܒܪܶܗ ܨܳܗܠܺܝܢ ܗܘܰܘ"، أي: "صارُوا أحصِنةً مَعْلُوفَةً سائِبَةً، وكانوا يَصهَلونَ كُلُّ واحِدٍ عَلى امْرَأةِ صاحِبِهِ".

(ܐܶܪܰܡܝܳܐ- ܗ: 8)

01 —

صَهَلَ

فعل
1.1معنى من معاني «صَهَلَ»

* صَهَلَ (بِفَتْحِ الهاءِ) الفَرَسُ يَصْهِل ويَصْهَلُ (بِكَسْرِ الهاءِ وفَتْحِها) صَهيلًا، وصُهالًا وتَصْهالًا، وصاهِلَةً: صَوَّتَ، فَهُوَ صاهِلٌ، وهِيَ بِتاءٍ، (ج) صَواهِلُ، وهُوَ أَيْضًا صَهّالٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قَالَ الحارِثُ بْنُ عُبادٍ (ت: 50ق هـ=573م) يَطْلُبُ فَرَسَهُ لِحَرْبِ بَني تَغْلِبَ:

قَرِّبا مَرْبِطَ النَّعامَةِ مِنّي
كُلَّ مُهْرٍ مُصَرْصَرٍ صَهّالِ

ديوان الحارث بن عباد، تح: أبو هلال، ص: 203.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ دُرَيْدُ بْنُ الصِّمَّةِ (ت: 8هـ=630م) يَصِفُ فَرَسًا:

يُجاوِبُ جُرْدًا كَالسَّراحينِ ضُمَّرا
تَرودُ بِأَبْوابِ البُيوتِ وتَصْهَلُ

ديوان دريد بن الصمة، تح: عبد الرسول، ص: 144.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ= 787م):

"الصَّهيلُ: صَوْتُ الخَيْلِ. صَهَلَ يَصْهَلُ صَهيلًا، وفَرَسٌ صَهّالٌ: كَثيرُ الصَّهيلِ".

الخليل، العين، تح المخزومي وآخر، ج: 3، ص: 413.— المصدر
”وقالَ العَكَوَّكُ (ت: 213هـ=828م) يَمْدَحُ:

وزَحوفٍ في صَواهِلِهِ
كَصِياحِ الحَشْرِ في أَمَرِهْ

ديوان علي بن جبلة، تح: عطوان، ص: 69.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ هُتَيْمِلٍ (ت: 696هـ=1297م) يَصِفُ فَرَسَ مَمْدوحِهِ:

شَمْسٌ مِنَ القُبِّ الصَّواهِلِ بُرْجُها
أَسَدٌ مِنَ الأُسْدِ الغَوائِلِ غيلُهُ

ديوان ابن هتيمل، تح: عبد الولي الشميري، ج: 2، ص: 751.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ نِقولا التُّرْك (ت: 1244هـ=1828م) يَمْدَحُ:

وهَزَّ العَواليْ وانْتَضى كُلَّ أَبْيَضٍ
يَحِزُّ لِأَرْقابِ العُداةِ ويَنْحَرُ

عَلى سابِقٍ كَالبَرْقِ في كُرُهٍ لَهُ
صَهيلٌ مَهولٌ خِلْتُهُ الرَّعْدَ يَهْدِرُ

ع

صَهَلا بِشِعْري مُقْرِفَيْنِ فَكُذِّبا

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

وقَد صَهَلَتني مِن خَبيثِ فِعالِكُم

02 —

صَهِلَ

فعل
2.1معنى من معاني «صَهِلَ»

* صَهِلَ (بِكَسْرِ الهاءِ) صَوْتُ فُلانٍ يَصْهَلُ (بِفَتْحِ الهاءِ) صَهَلًا: بُحَّ. (انظر: ص ح ل)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو هُرَيْرَةَ (ت: 59هـ=679م) يُخْبِرُ عَنْ عَلِيِّ بْنِ أَبي طَالِبٍ حِينَ بَعَثَهُ النَّبِيُّ ﷺ إِلى المُشْرِكينَ بِما تَضَمَّنَتْهُ سورَةُ التَّوْبَةِ:

"كُنْتُ مَعَ عَلِيِّ بْنِ أَبي طالِبٍ لَمّا بَعَثَهُ رَسولُ اللهِ ﷺ فَنادى بِأَرْبَعٍ ‌حَتّى ‌صَهِلَ ‌صَوْتُهُ".

ابن حجر، إتحاف المهرة، تح: الناصر، ج: 11، ص: 694.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م):

"وقيلَ: صَهيلُ الفَرَسِ: لِبُحَّةٍ فيهِ، مِنْ قَوْلِهِمْ: في صَوْتِهِ ‌صَهَلٌ وصَحَلٌ، وقَدْ ‌صَهِلَ صَوْتُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، أساس البلاغة، تح: محمد باسل عيون السود، ص: 567.— المصدر

صاهَلَ

فعل
3.1معنى من معاني «صاهَلَ»

* صاهَلَتِ الخَيْلُ: تَجاوَبَتْ بِصَوْتِها.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ المُقْري (ت: 837هـ=1433م) يَصِفُ جَيْشَ مَمْدوحِهِ:

يُصاهِلُ في ظِلِّ الصِّفاحِ جِيادُهُ
وتَزْأَرُ في غابِ الرِّماحِ أُسودُهُ

ديوان ابن المقري، تح: عبد الله الأنصاري، ص: 151.— المصدر

وــــــــــ الخَيْلُ ونَحْوُها الخَيْلَ وغَيْرَها: جاوَبَتْها بِصَوْتِها.

ع

تَمْضي الحُجورُ بِها تُصاهِلُها
مِنْها شُخوصٌ تُحْسَبُ الحُصْنا

07 —

صَهَل

اسم
7.1معنى من معاني «صَهَل»

* الصَّهَلُ: صَوْتُ الفَرَسِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العُلَيْفِ المَكِّيُّ (926هـ=1520م) يَهْجو:

سُمْتُمْ مَقامًا رَفيعًا فَوْقَ رُتْبَتِكُمْ
ولا يُقاسُ نَهِيقُ العَيْرِ ‌بِالصَّهَلِ

العصامي، سمط النجوم العوالي، تح: عادل عبد الموجود وآخر، ج: 4، ص: 335.— المصدر
08 —

صُهْلَة

اسم
8.1معنى من معاني «صُهْلَة»

* الصُّهْلَةُ: بُحَّةٌ في الصَّوْتِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"وفي صَوْتِ فُلانٍ صَهَلٌ ‌وصُهْلَةٌ، مِثْلُ صَحَلٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن دريد، جمهرة اللغة، تح: رمزي بعلبكي، ج: 2، ص: 898.— المصدر
09 —

صَهيل

اسم
9.1معنى من معاني «صَهيل»

* الصَّهيلُ: صَوْتُ الخَيْلِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ سَلامَةُ بْنُ جَنْدَلٍ (ت: 23 ق هـ= 600م) يَصِفُ خَيْلَ قومِهِ مُفْتَخِرًا:

عَلَيْهِنَّ أَوْلادُ المُقاعِسِ قُرَّحًا
عَناجيجُ في حُوٍّ لَهُنَّ صَهيلُ

ديوان سلامة بن جندل، تح: الأحول، ص: 191.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المُزَرِّدُ بْنُ ضِرارٍ (ت: 10هـ=631م) يَصِفُ فَرَسًا:

‌أَجَشُّ صَريحِيٌّ كَأَنَّ صَهيلَهُ
مَزاميرُ شَرْبٍ جاوَبَتْها جَلاجِلُ

ديوان المزرد بن ضرار ، تح: العطية، ص: 35.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"‌الصَّهيلُ: صَوْتُ الخَيْلِ".

الخليل، العين ، تح: المخزومي وآخر، ج: 3، ص: 413.— المصدر
”وقالَ أَبو تَمّامٍ (ت: 231هـ=846م) يَمْدَحُ:

فَفَدى نَفْسَهُ بِكُلِّ شَوارٍ
وصَهيلٍ في أَرْضِهِ ونَهيقِ

التبريزي، شرح ديوان أبي تمام، تح: الأسمر، ج: 1، ص: 456.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ تَيْمِيَّةَ (ت: 728هـ=1328م) يَذْكُرُ القَدْرَ المُشْتَرَكَ بَيْنَ الإِنْسانِ وَالفَرَسِ:

"فَإِنَّ الإِنْسانَ لَهُ تَمْييزٌ ولِلفَرَسِ تَمْييزٌ، ولِهذا صَوْتٌ هُوَ النُّطْقُ، ولِذاكَ صَوْتٌ هُوَ ‌الصَّهيلُ".

ابن تيمية، الانتصار لأهل الأثر، تح: عبد الرحمن بن حسن، ص: 320.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ اللَّطيف فَتْحُ اللهِ (ت: 1260هـ=1844م) يَمْدَحُ:

فَسِلاحُهُ وصَهيلُ طِرْفٍ تَحْتَهُ
بَرْقٌ ورَعْدٌ في العَجاجِ الأَقْتَمِ

ديوان المفتي عبد اللطيف فتح الله، تح: زهير فتح الله، ص: 367.— المصدر
9.2معنى من معاني «صَهيل»

وــــــــــ: الخَيْلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسّانُ بْنُ ثابِتٍ (ت: 54هـ=674م) يَهْجو:

تَمَنّى أَنْ تَكونَ إِلى قُرَيْشٍ
شَبِيهَ البَغْلِ شَبَّهَ بِالصّهيلِ

ديوان حسّان بن ثابت، تح: مهنا، ص: 202.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ صَفِيُّ الدّينِ الحِلِّيُّ (741هـ=1340م) يَمْدَحُ النّاصِرَ مُحَمَّدَ بْنَ قلاوونَ ويَصِفُ جَيْشَهُ:

صافي الفِرِنْدِ حَكى صَباحًا جامِدًا
أَبْدى النَّجيعَ بِهِ شُعاعًا ذائِبا

وكَتيبَةٍ ‌تَذَرُ الصّهيلَ رَواعِدًا
والبيضَ بَرْقًا والعَجاجَ سَحائِبا

ديوان صفي الدين الحلي. ص: 97.— المصدر
10 —

صَواهِل

اسم
10.1معنى من معاني «صَواهِل»

* الصَّواهِلُ: أَصْواتُ الخَيْلِ، مُفْرَدُها صاهِلةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَبو دُؤادٍ الإِيادِيُّ (ت: 79ق هـ=545م) يَصِفُ جَيْشًا:

لَجِبٌ ‌تُسْمَعُ ‌الصَّواهِلُ فيهِ
وحَنينُ اللِّقاحِ والإِرْزامُ

ديوان الأصمعيات، تح: طريفي، ص: 209.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحُطَيْئَةُ (ت: 45هـ=665م) يَمْدَحُ:

يَؤُمُّ العَدُوَّ حَيْثُ كانَ بِجَحْفَلٍ
يُصِمُّ السَّميعَ جَرْسُهُ وصَواهِلُهْ

ديوان الحطيئة، تح: قميحة، ص: 160.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابن نباتة (768هـ=1366م) يَرْثي:

لَهْفي عَلى الخَيْلِ قَدْ وَفَّتْ صَواهِلُها
حَقَّ العَزا فَهْوَ يُشْجيها وتُشْجيهِ

ديوان ابن نباتة المصري، إحياء التراث، ص: 571.— المصدر
10.2معنى من معاني «صَواهِل»

وــــــــ: أَصْواتُ الذُّبابِ وغُنَّتُها في العُشْبِ. (ز)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37هـ=658م) يَصِفُ نَوْعًا مِنَ النَّباتِ:

كَأَنَّ ‌صَواهِلَ ‌ذِبّانِهِ
قُبَيْلَ الصَّباحِ صَهيلُ الحُصُنْ

ديوان ابن مقبل، تح: عزة حسن، ص: 208.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م) يُعَلِّقُ عَلى بَيْتِ ابْنِ مُقْبِلٍ السّابِقِ:

"وجَعَلَ ابْنُ مُقْبِلٍ لِلذِّبّانِ ‌صَواهِلَ في العُشْب يُريدُ بِها غُنَّةَ طَيَرانِها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخر، ج: 6، ص: 111.— المصدر
10.3معنى من معاني «صَواهِل»

وــــــــــ: أَصْواتُ المَساحِي في نُزولِها عَلى الحِجارَةِ. (ز)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو زُبَيدٍ الطّائِيُّ (ت: 62هـ=682م) يَصِفُ وَقْعَ المَساحِي على الحِجارَةِ:

لَهَا ‌صَوَاهِلُ فِي صُمِّ السِّلَامِ كَما
صاحَ القِسِيّاتُ في أَيْدي الصَّياريفِ

أبو زبيد الطائي، شعر أبي زبيد الطائي، تح: نوري القيسي، ص: 119.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م) يُعَلِّقُ عَلى بَيْتِ أَبي زُبَيْدٍ السّابِقِ:

"وجَعَلَ أَبو زُبَيْدٍ لِأَصْواتِ ‌المَساحي الَّتي يُحْفَرُ بِها ‌صَواهِلَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخر، ج: 6، ص: 111.— المصدر
