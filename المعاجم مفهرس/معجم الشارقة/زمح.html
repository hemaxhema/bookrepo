زمح
زَمَّحَ(فعل) زامِح(اسم) زِماح(اسم) زُمّاح(اسم) زُمَّح(اسم) زَوْمَح(اسم)

المعانى الكلّيّة

  1. لية:,سوءُ الخِلْقَةِ والخُلُقِ
01 —

زَمَّحَ

فعل
1.1معنى من معاني «زَمَّحَ»

* زَمَّحَ فُلانٌ: قَتَلَ طائِرَ الزُّمَّاحِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الأَعرابِيِّ (ت: 231هـ=845م):

"وزَمَّحَ الرَّجُلُ إِذا قَتَلَ الزُّمّاحَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 4، ص: 219.— المصدر
02 —

زامِح

اسم
2.1معنى من معاني «زامِح»

* الزّامِحُ: الدُّمَّلُ، وهُوَ الخُرّاجُ يَظْهَرُ فِي جَسَدَ الإِنْسانِ وَغَيرِهِ، وَيَمْتَلِئُ بالصَّديدِ، وَيَتَقَيَّحُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت: 170هـ=787م) يُفَسِّرُ الحِبْنَ:

"وهُوَ الزّامِحُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 491.— المصدر
03 —

زِماح

اسم
3.1معنى من معاني «زِماح»

* الزِّماحُ: الجُموحُ والشُّرودُ. (انْظُرْ: ر م ح)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السَّرَقُسطي (ت: 400هـ=1010م):

"وَيُقالُ: بَرِئْتُ إِلَيكَ مِن الِجماحِ، والطِّماحِ، والزِّماحِ".

السرقسطي، الأفعال. تح: حسين شرف، ج: 2، ص: 290.— المصدر
04 —

زُمّاح

اسم
4.1معنى من معاني «زُمّاح»

* الزُّمّاحُ: طائِرٌ مِنَ الجَوارِحِ الكِبارِ، رِيشُهُ أَطْحَلُ أَوْ أَسْوَدُ، أَبيضُ الزّمِكّى والعُنُقِ، لَهُ مِنْسَرٌ أَعْقَفُ يَهْجِمُ عَلَى حَيواناتِ البَرِّ والبَحْرِ، وَقَدْ يَتَعَرَّضُ لِأَخْذِ الصَّبِيِّ مِنْ مَهْدِهِ، يَكونُ في وَقْتِ الثَّمَرِ، ومِنْ أَنْواعِهِ الزُّماحُ العرَكِيُّ والزُّماحُ الأَبْيَضُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ قيس بن الخَطيم (ت: 2ق هـ=620م) يَصِفُ:

أَعَلى العَهْدِ أَصْبَحَتْ أُمُّ عَمْرٍو
لَيْتَ شِعْري أَمْ عاقَها ‌الزُّمّاحُ

ديوان قيس بن الخطيم. تح: السامرائي وآخر، ص: 78.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت: 170هـ=787م):

"والزُّمّاحُ: طائِرٌ عَظيمٌ".

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 167.— المصدر
”وقالَتِ العربُ (395هـ=1005م) من أمثال أبي هلال العسكري، يضربون المثلَ في شِدَّةِ الشُّؤْمِ والنَّحْسِ:

"أَشْأَمُ مِنَ الزُّمّاحِ".

العسكري، جمهرة الأمثال. تح: عبد السلام وآخر، ج: 1، ص: 456.— المصدر
4.2معنى من معاني «زُمّاح»

وــــــــــ: سَهْمٌ يُجْعَلُ عَلَى رَأْسِهِ طِينٌ كالبُنْدُقَةِ يُرْمَى بِهِ الطَّيرُ. (انْظُرْ: ج م ح)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الراجزُ (321هـ=933م) من شواهد ابن دريد، يَتَشَوَّقُ:

هَلْ يُبْلِغَنِّيهِمْ إِلى الصَّبَاحْ

هَيْقٌ كَأَنَّ رَأْسَهُ زُمَّاحْ

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 529.— المصدر
4.3معنى من معاني «زُمّاح»

وــــــــــ من الرِّجالِ: الضَّعيفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عَبّاد (ت: 385هـ=995م):

"ورَجُلٌ زُمّاحٌ وزُمَّحٌ: ضَعيفٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 3، ص: 18.— المصدر
05 —

زُمَّح

اسم
5.1معنى من معاني «زُمَّح»

* الزُّمَّحُ مِنَ الرِّجالِ: اللَّئيمُ البَخيلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مُتَمِّمُ بن نُوَيْرة (12هـ=633م) يَرْثي أَخاهُ مالِكًا:

بَذُولٌ لِما في ‌رَحْلِهِ غَيْرُ زُمَّحٍ
إِذا أَبْرَزَ الحورَ الرَّوائِعَ جوعُ

المفضل الضبي، المفضليات. تح: شاكر، وآخر، ص: 272.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيبانيُّ (ت: 206هـ=821م):

"الزُّمَّحُ: ‌اللَّئيمُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلام، الغريب المصنف. تح: داوودي، ج: 1، ص: 113.— المصدر
5.2معنى من معاني «زُمَّح»

وــــــــــ: القَصيرُ السَّمْجُ الخِلْقَةِ السَّيِّئُ المَشْؤومُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الكميت بن زيد (ت: 126هـ=744م) يَصِفُ:

ولَمْ يَتَهَجَّمْ لَكَ النّائِباتُ
ولَمْ تَكُ فيها اللِّياسُ الدَّثورا

ولَمْ تَكُ شِهْدارَةَ الأَبْعَدَيْنِ
ولا زُمَّحَ الأَقْرَبَيْنِ الشَّريرا

ديوان الكميت. تح: طريفي، ص: 168.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشاعرُ (206هـ=821م) من شواهد أبي عمرو الشيباني، يَصِفُ:

طَويلُ نِجادِ السَّيْفِ لَيْسَ بِجَأَنَبٍ
ولا كَيْئَةٍ ‌كَزِّ الأَنامِلِ ‌زُمَّحِ

الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 79.— المصدر
5.3معنى من معاني «زُمَّح»

وــــــــــ: الضَّعيفُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الطِّرِمّاح (ت: 125هـ=743م) يُخَوِّفُ ابْنَهُ صَمْصامَةَ مِنْ تَزْويجِ أُمِّهِ بَعْدَ مَوْتِهِ:

إِذا جِئْتَها تَبْكي بَكَتْ وتَذَكَّرَتْ
مَعَ الحُزْنِ صَوْلاتِ امْرِئٍ غَيْرِ ‌زُمَّحِ

ديوان الطّرماح. تح: عزّة حسن، ص: 98.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيبانيُّ (ت: 206هـ=821م):

"الزُّمَّحُ: الضَّعيفُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الهروي، الغريب المصنف. تح: داوودي، ج: 1، ص: 119.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحمد محرم (ت: 1364هـ=1945م) يَذْكُرُ المُسْلِمينَ يَوْمَ بَدْرٍ:

أَخَذوا السِّلاحَ وقَدْ أَغارَ لِأَخْذِهِمْ
جُنْدٌ بِآياتِ الكِتابِ مُسَلَّحُ

فيهِم مِنَ الأَنْصارِ كُلُّ مُشَيَّعٍ
يَمْضي إِذا نَكَصَ اليَراعُ الزُّمَّحُ

أحمد محرم، ديوان مجد الإسلام، ص: 59.— المصدر
5.4معنى من معاني «زُمَّح»

وــــــــــ: الأَسْوَدُ القَبيحُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (ت: 170هـ=787م):

"الزَّوْمَحُ والزُّمَّحُ: الأَسْوَدُ القَبيحُ مِنَ الرِّجالِ".

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 167.— المصدر

* الزَّوْمَحُ مِنَ الرِّجالِ: الأَسْوَدُ القَبيحُ.
