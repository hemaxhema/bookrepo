المعانى الكلّيّة

  1. شَيْءٌ مِنَ الحُسْنِ أَوِ القُبْحِ في خَلْقِ الإنْسانِ وغيْرِه
01 —

تَطَهَّمَ

فعل
1.1معنى من معاني «تَطَهَّمَ»

* تَطَهَّمَ فُلانٌ عن النّاسِ: نَفَرَ مِنْهُمْ، واسْتَوْحَشَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م):

"فُلانٌ يَتَطَهَّمُ عَنّا: أَيْ يَسْتَوْحِشُ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
1.2معنى من معاني «تَطَهَّمَ»

وــــــــــ الطَّعامَ، وعَنْهُ: كَرِهَهُ، ورَبَأَ بِنَفْسِهِ عَنْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"ومِنْهُ يُقالُ: ما لَكَ ‌تَطَهَّمُ عَنْ طَعامِنا، أَيْ: تَرْبَأُ بِنَفْسِكَ عَنْهُ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
02 —

تَطَهُّم

اسم
2.1معنى من معاني «تَطَهُّم»

* التَّطَهُّمُ: الجَمالُ البارِعُ، وتَمامُ الخَلْقِ في الفَرَسِ، والإنْسانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"فَرَسٌ مُطَهَّمٌ بَيِّنُ ‌التَّطَهُّمِ، والتَّطْهيمِ. وكذَلِكَ الإنْسانُ، إِذا كانَ تامَّ الجَمالِ والخَلْقِ ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 927.— المصدر
03 —

تَطْهيم

اسم
3.1معنى من معاني «تَطْهيم»

* التَّطْهيمُ: النِّفارُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ذو الرُّمَّةِ (ت: 117هـ=735م) يَصِفُ جَمالَ الحَبيبَةِ عِنْدَ انْجِلاءِ نِقابِها الأَسْوَدِ شَيْئًا فَشَيْئًا:

تِلْكَ الَّتي أَشْبَهَتْ خَرْقاءَ جَلْوَتُـهَا
يَوْمَ النَّقا بَـهْجَةً مِنْها وتَطْهيمُ

ديوان ذي الرُّمّة. تح: طراد، ص: 139.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"التَّطْهيمُ: النِّفارُ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
3.2معنى من معاني «تَطْهيم»

وــــــــــ: الجَمالُ البارِعُ، وتَمامُ الْخَلْقِ في الفَرَسِ، والإِنْسَانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِمُ بنُ سَلّامٍ (ت: 224هـ=839م):

" والتَّطْهِيم: الجَمالُ ".

أبو عبيد، الغريب المصنف. تح: داوودي، ج: 1، ص: 138.— المصدر
3.3معنى من معاني «تَطْهيم»

وــــــــــ: الضِّخَمُ، والسِّمَنُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ثَعْلَب (ت: 291هـ=904م) يُبَيِّنُ ما فُسِّرَ بِهِ قَوْلُ عَلِيٍّ في وَصْفِ النَّبِيِّ ﷺ: "لَمْ يَكُنْ بِالْـمُطَهَّمِ، وَلا بالـمُكَلْثَمِ":

"وأَمّا مَنْ قالَ: إِنَّ التَّطْهيمَ: الضِّخَمُ، فَقَدْ صَحَّ النَّفْيُ، فَكَأَنَّهُ قالَ: لَمْ يَكُنْ بِالضَّخْمِ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
3.4معنى من معاني «تَطْهيم»

وــــــــــ: انْتِفاخُ الْوَجْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (291هـ=903م، مِن شَواهِدِ ثعْلَب):

ووَجْهٌ فيهِ تَطْهيمُ

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
04 —

طَهْم / طُهْم

اسم
4.1معنى من معاني «طَهْم / طُهْم»

* الطَّهْمُ، والطُّهْمُ: الخَلَقُ القَديمُ البالي.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُيَيُّ بنُ هانِئٍ المَعافِرِيُّ (ت: 128هـ=746م):

"كُنَّا عِنْدَ عَبْدِ اللهِ بْنِ عَمْرِو بْنِ العاصِ، فَسُئِلَ: أَيُّ المَدينَتَيْنِ تُفْتَحُ أَوَّلًا؟ قُسْطَنْطينِيَّةُ أَوْ رومِيَّةُ؟ ‌فَدَعا ‌بِصُنْدوقٍ ‌طُهْمٍ - وَالطُّهْمُ الخَلَقُ - فَأَخْرَجَ مِنْها كِتابًا، فَنَظَرَ فيهِ، ثُمَّ قالَ: كُنّا عِنْدَ رَسولِ اللهِ ﷺ نَكْتُبُ ما قالَ: فَسُئِلَ أَيُّ المَدِينَتَيْنِ تُفْتَحُ أَوَّلًا؟ القُسْطَنْطِينِيَّةُ أَوِ الرّومِيَّةُ؟ فَقالَ رَسولُ اللهِ ﷺ: «مَدِينَةُ هِرَقْلَ تُفْتَحُ أَوَّلًا».".

مستدرك الحاكم. تح: عطا، ج: 4، ص: 598.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللِّحْيانِيُّ (ت: 220هـ=835م):

"وما أَدْري أَيُّ الطَّهْمِ هُوَ؟ وأَيُّ الطُّهْمِ؟ أَيْ: أَيُّ الْخَلَقِ؟ ".

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 253.— المصدر
05 —

طَهْمَة / طَهِمَة

اسم
5.1معنى من معاني «طَهْمَة / طَهِمَة»

* الطَّهْمَةُ، والطَّهِمَةُ: المَرْأَةُ القَليلَةُ لَحْمِ الوَجْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385هـ=995م):

"وَامْرَأَةٌ طَهْمَةٌ:‏ قَليلَةُ لَحْمِ الوَجْهِ ".

ابن عباد، المحيط. تح: آل ياسين، ج: 3، ص: 437.— المصدر
06 —

طُهْمَة

اسم
6.1معنى من معاني «طُهْمَة»

* الطُّهْمَةُ: السُّمْرَةُ تُجاوِزُ إلى السَّوادِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"الطُّهْمَةُ، والصُّحْمَةُ في اللَّوْنِ: أَنْ تُجاوِزَ سُمْرَتَهُ إِلى السَّوادِ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
07 —

مُطَهَّم

اسم
7.1معنى من معاني «مُطَهَّم»

* المُطَهَّمُ: البارِعُ الجَمالِ، التّامُّ الخَلْقِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ جَسّاسُ بنُ مُرَّةَ (ت: 110 ق هـ=514م) يَفْتَخِرُ بِفُرْسانِ قَبيلَتِهِ:

وفِتْيانٍ تَكُرُّ على الأَعادي
على الجُرْدِ الـمُطَهَّمَةِ الذُّكورِ

وتَنْتَظِمُ المَعاهِدَ بِالـمَوالي
وتَنْتَصِفُ الجُنانَ مِنَ القُدورِ

ديوان بني بكر. تح: نبوي، ج: 1، ص: 386.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابنُ مُقْبِلٍ (ت: 37هـ=657م) يَفْتَخِرُ بِقُوَّةِ قَوْمِهِ وخَيْلِهِمْ:

وإنَّ فينا صَبوحًا إنْ أرِبْتَ بِهِ
جَـمْعًا بَـهِيًّا وَآلافًا ثَـمَانينا

ومُقْرَباتٍ عَناجيجًا ‌مُطَهَّمَةً
مِنْ آلِ أعْوَجَ مَلْحوفًا وَمَلْبونا

إذا تَجاوَبْنَ صَعَّدْنَ الصَّهيلَ إلى
صُلْبِ الشُّؤونِ ولَمْ تَصْهَلْ بَراذينا

ديوان ابن مقبل. تح: عزة حسن، ص: 236.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الـمُطَهَّمُ: الفَرَسُ التّامُّ الخَلْقِ، الجَهيرُ الجَمالِ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 22.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ الصَّيْقَلِ (ت: 701ه=1301م) يُخْبِرُ عن شَيْخٍ انْقَلَبَ حالُهُ:

"فَأَصْبَحَ بَعْدَ امْتِطاءِ ‌الـمُطَهَّماتِ، واقْتِناءِ الصُّرَرِ الـمُدَرْهَماتِ، يُصافِحُ المَصائِبَ، ويُكافِحُ سَهْمَ السَّقَمِ الصّائِبَ".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ص: 490.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابن عَجيبَةَ (1224هـ=1809م) يُفَسِّرُ قَوْلَ اللهِ ﷻ: ‌‌﴿وَٱلۡخَيۡلِ ٱلۡمُسَوَّمَةِ﴾ (آل عمران: 14):

"والـمُسَوَّمَةُ: الـمُعَلَّمَةُ، أَوِ الرّاعِيَةُ، أو ‌الـمُطَهَّمَةُ الحِسانُ".

ابن عجيبة، البحر المديد. تح: القرشي، ج: 1، ص: 329.— المصدر
7.2معنى من معاني «مُطَهَّم»

وــــــــــ: الكَريمُ العزيزُ مِنَ الخَيْلِ والبَشَرِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتِ الفارِعَةُ بِنتُ مُعاوِيَةَ (ت: 64ق هـ=559م) تُخاطِبُ تَميمًا، وتَمْتَدِحُ الْقُشَيْرَ وجَعْدَةَ ابْنَيْ كَلْبٍ:

 فَما وَجَدَ الحيّانِ عَمْرٌو ومالكٌ
وعُقْدَةُ بالجَزْعاءِ مِن مُتقَدَّمِ

إِلى ابْنَيْ عَجوزٍ مِنْ سُلَيْمٍ غَريبَةٍ
يُؤَيَّهُ فيهِمْ أَلْفُ طِرْفٍ ‌مُطَهَّمِ

[وَأَيَّهَ بِالْخَيْلِ: صَوَّتَ. الطِّرْفُ: الكَريمُ مِنَ النّاسِ والخَيْلِ ونحوِهما]
الهجري، التعليقات والنوادر. تح: الجاسر، ج: 4، ص: 211.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"وأَمّا الخَيْلُ المُطَهَّمَةُ فَإِنَّها المُقَرَّبَةُ المُكَرَّمَةُ العَزيزَةُ الأَنْفُسِ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
7.3معنى من معاني «مُطَهَّم»

وــــــــــ: النَّحيفُ الجِسْمِ. (ضِدٌّ).

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طُفَيْلٌ الغَنَوِيُّ (ت: 13 ق هـ=609م) يَفْتَخِرُ بِما يَمْتَلِكُ قَوْمُهُ مِنَ الْخَيْلِ الْمُضَمَّرَةِ:

وَفينا رِباط الْخَيْلِ كُلُّ ‌مُطَهَّمٍ
رَجيلٍ كَسِرْحانِ الغَضا الْمُتَأَوِّبِ

ديوان طفيل الغنويّ. تح: أوغلي، ص: 27.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبدُ اللهِ بنُ كَعْبٍ الأَنْصارِيُّ (5هـ=626م) يَتَوَعَّدُ الْمُشْرِكينَ بِهَزيمَةٍ نَكْراءَ:

نُصَبِّحُكُمْ بِكُلِّ أَخي حُروبٍ
وَكُلِّ مُطَهَّمٍ سَلِسِ الْقِيادِ

سيرة ابن هشام. تح: تدمري، ج: 3، ص: 211.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُواسٍ (ت: 198هـ=814م) يَصِفُ كَلْبًا:

أَنْعَتُ كَلْبًا لَيْسَ بِالمَسْبوقِ

مُطَهَّمًا يَجْري على العُروقِ

ديوان أبي نُواس. طبعة الإسكندرية، ص: 466.— المصدر
7.4معنى من معاني «مُطَهَّم»

وــــــــــ: الضَّخْمُ السَّمينُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بنُ أبي طالِبٍ (ت: 40هـ=661م) يَصِفُ النَّبِيَّ ﷺ:

"لَمْ يَكُنْ بِالطَّويلِ المُمَغَّطِ، ولا القَصيرِ المُتَرَدِّدِ، وكانَ رَبْعَةً مِنَ القَوْمِ، ولَمْ يَكُنْ بِالجَعْدِ القَطَطِ، ولا السَّبِطِ، كانَ جَعْدًا رَجِلًا، ولَمْ يَكُنْ بِالمُطَهَّمِ، ولا المُكَلْثَمِ".

سيرة ابن هشام. تح: تدمري، ج: 2، ص: 52.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ثَعْلَب (ت: 291هـ=904م) يُبَيِّنُ ما فُسِّرَ بِهِ قَوْلُ عَلِيٍّ في وَصْفِ النَّبِيِّ ﷺ: "لَمْ يَكُنْ بِالْمُطَهَّمِ، وَلا بالمُكَلْثَمِ":

"المُطَهَّمُ: الضَّخْمُ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر

وــــــــــ: المُنْتَفِخُ الوَجْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ عَلِيِّ بنِ أبي طالِبٍ (ت: 40هـ=661م) يَصِفُ النَّبِيَّ ﷺ:

"لَمْ يَكُنْ بِالطَّويلِ المُمَغَّطِ، ولا القَصيرِ المُتَرَدِّدِ، وكانَ رَبْعَةً مِنَ القَوْمِ، ولَمْ يَكُنْ بِالجَعْدِ القَطَطِ، ولا السَّبِطِ، كانَ جَعْدًا رَجِلًا، ولَمْ يَكُنْ بِالمُطَهَّمِ، ولا المُكَلْثَمِ".

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ثَعْلَب (ت: 291هـ=904م) يُبَيِّنُ ما فُسِّرَ بِهِ قَوْلُ عَلِيٍّ في وَصْفِ النَّبِيِّ ﷺ: "لَمْ يَكُنْ بِالمُطَهَّمِ، وَلا بالمُكَلْثَمِ":

"المُطَهَّمُ: المُنْتَفِخُ الوَجْهِ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحمد أمين (ت: 1373هـ=1954م) يَصِفُ صَديقَهُ:

"وتَعَرَّفْتُ في الإِسْكَنْدَرِيَّةِ بِإِنْسانٍ جَديدٍ، فَإذا هو صَديقي القَديمُ، هو في هَذِهِ المَرَّةِ بَدينٌ بَطينٌ،‌ مُطَهَّمُ الوَجْهِ، رَيّانُ السَّواعِدِ".

أحمد أمين، فيض الخاطر. مكتبة النهضة المصرية، ص: 112.— المصدر
7.6معنى من معاني «مُطَهَّم»

وــــــــــ مِنَ الوُجوهِ: الّذي تَجاوَزَتْ سُمْرَتُهُ إِلى السَّوادِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو سَعيدٍ الضَّريرُ (ت: 234هـ=849م):

"الطُّهْمَةُ، والصُّحْمَةُ في اللَّوْنِ: أَنْ تُجاوِزَ سُـمْرَتَهُ إِلى السَّوادِ. وَجْهٌ مُطَهَّمٌ: إِذا كانَ كَذَلِكَ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 106.— المصدر
