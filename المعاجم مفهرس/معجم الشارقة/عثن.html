النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذْرُ (ع ث ن):

في الآرامِيّةِ: ومِنْهُ الفِعْلُ: תְּנַן/ תְּנֵין (تْنَن/ تْن۪ٮٰن) tənēn /tənan بمَعْنى "دَخَّنَ"؛ وهُوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (عَثَنَ) مَبْنًى ومَعْنًى؛ غَيْرَ أَنَّ الفِعْلَ الآراميَّ خَضَعَ لسُقوطِ حَرْفِ العَيْنِ، ولإِبْدالِ حَرْفِ الثّاءِ شينًا؛ ومن مَجيئِهِ بمَعْنى "عَثَنَ" ما وَرَدَ في تُرْجومِ أونكلوس: "וְטוּרָא דְסִינַי תָּנֵן כּולֵיהּ מִן קֳדָם דְאִיתְגְלֵי עֲלוֹהִי יְיָ בְּאִישָׁתָא"؛ أَيْ: "وطورِ سَيْناءَ يَعْثُنُ كُلُّهُ؛ لأَنَّ الرَّبَّ تَجَلّى عَلَيْهِ بالنّارِ".

(ת''א، שמ': יט, יח)

وفي العِبْريّةِ: ومِنْهُ الفِعْلُ: עָשַׁן (عَاشَن) ʽāšan بمَعْنى "دَخَّنَ"، وهُوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (عَثَنَ) مَبْنًى ومَعْنًى؛ غَيْرَ أَنَّ الفِعْلَ العِبْريَّ خَضَعَ لإِبْدالِ الثّاءِ شينًا؛ ومن مَجيئِهِ بمَعْنى "عَثَنَ" ما وَرَدَ في سِفْرِ الخُروجِ: "וְהַר סִינַי עָשַׁן כֻּלּוֹ"؛ أَيْ: "وَعَثَنَ جَبَلُ سِينَاءَ كُلُّهُ".

(שמ' 19: 18)

في السُّرْيانيّةِ: ومِنْهُ الفِعْلُ: ܬܰܢ (تَن) tan بمَعْنى "دَخَّنَ"، وهُوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (عَثَن) مَبْنًى ومَعْنًى؛ غَيْرَ أَنَّ الفِعْلَ السُّريانيَّ جاءَ مُضاعَفًا بَعْدَ أَنْ سَقَطَتْ مِنْهُ العَيْنُ؛ ومن مَجيئِهِ بمَعْنى "عَثَنَ" ما وَرَدَ في سِفْرِ المَزاميرِ: "ܘܟܳܐܶܐ ܒܛܽܘܖ̈ܶܐ ܘܬܳܢܶܝܢ"؛ أَيْ: "يَمَسُّ الْجِبَالَ فَتَعْثُنُ".

(ܡܰܙܡܽܘܪ̈ܶܐ. ܩܕ: 32)

01 —

عَثَنَ

فعل
1.1معنى من معاني «عَثَنَ»

* عَثَنَتِ (بِفَتْحِ الثّاءِ) النَّارُ ونَحْوُهَا تَعْثُنُ (بِضَمِّ الثّاءِ) عَثْنًا، وعُثانًا، وعُثونًا: دَخَّنَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"العُثانُ: ‏الدُّخانُ. عَثَنَ النَّارُ يَعْثُنُ عَثْنًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 2، ص: 110.— المصدر
1.2معنى من معاني «عَثَنَ»

وــــــــــ الشَّيْءُ: ارْتَفَعَ لَهُ عُثانٌ، وهُوَ كَالدُّخانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القُوطِيَّةِ (ت: 367هـ=977م):

"وعَثَنَ الشَّيْءُ عَثْنًا: ارْتَفَعَ لَهُ عُثانٌ، وهُوَ كَالدُّخانِ".

ابن القوطية، الأفعال. تح: فودة، ص: 189.— المصدر
1.3معنى من معاني «عَثَنَ»

وــــــــــ المَرْأةُ بِالدُّخْنَةِ: اسْتَجْمَرَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرِو بْنُ العَلاءِ (ت: 154هـ=771م):

"يُقالُ عَثَنَتِ المَرْأَةُ بِدُخْنَتِها إِذا اسْتَجْمَرَتْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 330.— المصدر
1.4معنى من معاني «عَثَنَ»

وــــــــــ فُلانٌ بِها: فَسا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"وتَقولُ عَثَنَ بِها: إِذا فَسا".

الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 322.— المصدر
1.5معنى من معاني «عَثَنَ»

وــــــــــ في الجَبَلِ: صَعِدَ فيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكِسائِيُّ (ت: 189هـ=805م):

"عَثَنْتُ في الجَبَلِ وعَفَنْتُ: إِذا صَعِدْتَ فيهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 330.— المصدر
”وقالَ الشّاعِرُ (244هـ=858م، مِنْ شَواهِدِ ابْنِ السِّكّيتِ) يُقْسِمُ أَلّا يَزورَ القَوْمَ ما دامَ للجَبَلِ صاعِدٌ فيهِ:

حَلَفْتُ بِمَنْ أرْسَى ثَبِيرًا مَكَانَهُ
أزُورَكمُ مادَامَ لِلطُّورِ عاثِنُ

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 93.— المصدر
1.6معنى من معاني «عَثَنَ»

وــــــــــ بَيْنَ الجَماعَةِ: خَلَّطَ وأَثارَ الفَسادَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م):

"ويُقالُ: عَثَنَ فُلانٌ بَيْنَنا أَيْ خَلَّطَ وأثارَ الفَسادَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 331.— المصدر
1.7معنى من معاني «عَثَنَ»

وــــــــــ في العَمَلِ يَعْثِنُ وَيعْثُنُ (بكَسْرِ الثّاءِ وضَمِّها) عُثونًا: دَأَبَ واجْتَهَدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"وعَثَنَ في العَمَل يَعْثِنُ وَيعْثُنُ عُثونًا: دَأَبَ واجْتَهَدَ ‏".

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 12.— المصدر
1.8معنى من معاني «عَثَنَ»

وــــــــــ الدُّخانُ الطَّعامَ: خالَطَهُ فَأَفْسَدَهُ. فَهُوَ عَثِنٌ، ومَعْثونٌ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672هـ=1273م):

"العَثْنُ: مَصْدَرُ عُثِنَ الطَّعامُ، فَهُوَ مَعْثونٌ، إِذا تَغَيَّرَ بِالدُّخانِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن مالك، إكمال الإعلام. تح: سعد الغامدي، ج: 2، ص: 408.— المصدر
1.9معنى من معاني «عَثَنَ»

وــــــــــ فُلانٌ الثَّوْبَ بِالطِّيبِ: دَخَّنَهُ عَلَيْهِ حَتّى عَبِقَ بِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م):

"وَعَثَنْتُ الثَّوْبَ بِالطِّيبِ، إِذا دَخَّنْتَهُ عَلَيْهِ حَتّى عَبِقَ بِهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 330.— المصدر
02 —

عَثِنَ

فعل
2.1معنى من معاني «عَثِنَ»

* عَثِنَ (بِكَسْرِ الثّاءِ) البيْتُ ونحوُهُ يَعْثَنُ (بِفَتْحِ الثّاءِ) عَثَنًا: عَبِقَتْ بِهِ رائحَةُ الدُّخْنَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وعَثِنَ البَيْتُ يَعْثَنُ عَثَنًا إِذا عَبِقَ بِهِ ريحُ الدُّخْنَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 110.— المصدر
03 —

عَثَّنَ

فعل
3.1معنى من معاني «عَثَّنَ»

* عَثَّنَتِ المَرْأَةُ ونَحْوُها: بَخَّرَتْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ضابِئٌ البُرْجُمِيُّ (ت: 30هـ=651م) يَهْجو بَني جَرْوَلِ بْنِ نَهْشَلٍ:

فَأُمُّكُمُ لا تَتْرُكوها وكَلْبَكَمْ
فَإِنَّ عُقوقَ الوالِداتِ كَبيرُ

إِذا عَثَّنَتْ مِنْ آخِر اللَّيْلِ دُخْنَةً
يَبيتُ لَهُ فَوْقَ السَّريرِ هَريرُ

الجاحظ، كتاب الحيوان. تح: عيون السود، ج: 1، ص: 246.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وعَثَّنَ يُعَثِّنُ تَعْثينًا، أَيْ دَخَّنَ تَدْخينًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 2، ص: 110.— المصدر
3.2معنى من معاني «عَثَّنَ»

وــــــــــ النّارُ وَنَحْوُها: دَخَّنَتْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَواسُ بْنُ ثابِتٍ (ت: 70هـ=690م) يَهْجو:

أَتَتْكَ العيسُ تَنْفُخُ في بُراها
بِوَفْدٍ مِنَ قُناعَةَ كَالقُرودِ

إِذا النَّيْتونُ عَثَّنَ في لِحاهُمْ
فَلَيْسوا بِالغَطارِفَةِ الأُسودِ

شعر قبيلة كلب حتّى نهاية العصر الأمويّ. تح: هارون آدم يوسف، ص: 85.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"عَثَّنَ يُعَثِّنُ تَعْثينًا، أَيْ: دَخَّنَ تَدْخينًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 2، ص: 110.— المصدر
”وقالَ ابْنُ حَمْديسَ (ت: 527هـ=1133م) يَصِفُ آلَةً حَرْبَيَّةً:

إِذا عَثَّنَتْ فيها التَّنانيرُ خِلْتَها
تُفَتِّحُ لِلْبُركانِ عَنْها مَنافِسا

ديوان ابن حمديس. تح: إحسان عباس، ص: 276.— المصدر
3.3معنى من معاني «عَثَّنَ»

وــــــــــ فُلانٌ عَلَيْنا: أَوْقَعَ التَّخْليطَ بَيْنَنا. (مو).

04 —

عُثان

اسم
4.1معنى من معاني «عُثان»

* العُثانُ: الدُّخانُ. (ج) عَواثِنُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ سُراقَةُ بْنُ مالِكٍ (ت: 24هـ=645م) يُخْبِرُ عَمّا حَدَثَ لِفَرَسِهِ وتَخْليصِ النَّبِيِّ ﷺ إيّاهُ في حَديثِ الهِجْرَةِ:

"فَخَرَجَتْ قَوائِمُها ولَها عُثانٌ".

[يُشَبَّهُ الغُبارُ بِالدُّخانِ وقَدْ يُنْعَتُ بِهِ]
الدارقطني، المؤتلف والمختلف. تح: موفق بن عبد القادر، ج: 3، ص: 1598.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ راشِدٍ (ت :153هـ=770م) يَسْتَرْشِدُ عَنْ معنى العُثانِ:

"قُلْتُ لِأَبي عَمْرِو بْنِ عَلاءٍ: ما العُثانُ؟ فَسَكَتَ ساعَةً ثُمَّ قالَ: هُوَ الدُّخانُ مِنْ غَيْرِ نارٍ".

ابن الجوزي، جامع المسانيد. تح: علي حسين البواب، ج: 3، ص: 21.— المصدر
05 —

عَثْن / عَثَن

اسم
5.1معنى من معاني «عَثْن / عَثَن»

* العَثْنُ، والعَثَنُ: الدُّخانُ.

ع

* العُواثِنُ: الأَسَدُ الكَثيرُ الشَّعْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"العُواثِنُ: الأَسَدُ الكَثيرُ الشَّعْرِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 12.— المصدر
11 —

مُعْثِن

اسم
11.1معنى من معاني «مُعْثِن»

* المُعْثِنُ: الطَّويلُ العُثْنونِ كالمُسْبِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو موسى المَدِينِيُّ (ت: 581هـ=1185م):

"والمُعْثِنُ: الطَّويلُ العُثْنونِ كالمُسْبِلِ".

أبو موسى المديني، المجموع المغيث. تح: العزباوي، ج: 2، ص: 405.— المصدر
12 —

مُعَثَّن

اسم
12.1معنى من معاني «مُعَثَّن»

* المُعَثَّنُ: الرَّجُلُ الضَّخْمُ اللِّحْيَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"ورَجُلٌ مُعَثَّنٌ: ضَخْمُ العُثْنونِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 94.— المصدر
