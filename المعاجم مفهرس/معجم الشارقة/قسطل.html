01 —

قَسْطَلَ

فعل
1.1معنى من معاني «قَسْطَلَ»

* قَسْطَلَ فُلانٌ الشَّيْءَ: لَطَّخَهُ بِالغُبارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ المُتَنَبّي (ت: 354ه=965م) يَهْجو كافورًا مُفْتَخِرًا:

وقَـــــدْ أُرِيَ الخِنْزيــــــرُ أَنّي مَدَحْتُــــهُ
ولَوْ عَلِموا قَدْ كانَ يُهْجى بِما يُطْرا

جَسَرْتُ عَلى دَهْياءَ مِصْرَ فَفُتُّها
ولَـــــــمْ يَكُنِ الدَّهْياءَ إلّا مَنِ اسْتَجْرا

سَأَجْلُبُها أَشْباهَ ما حَمَلَتْهُ مِــــنْ
أَسِنَّتِها جُــــــــــــــــــرْدًا مُقَسْطَلَةً غُبْرا

[الدَّهْياءُ: الدّاهِيَةُ. سَأَجْلُبُها: يَقْصِدُ الخَيْلَ]
أبو العلاء المعري، شرح ديوان أبي الطيب المتنبي. تح: دياب، ج: 2، ص: 443.— المصدر

03 —

قَسْطال

اسم
3.1معنى من معاني «قَسْطال»

* القَسْطالُ: الغُبارُ السّاطِعُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بْنُ حَجَرٍ (2ق.ه=620م) يَرْثي فَضالَةَ بْنَ كَلَدَةَ:

فَلَنِعْـــــمَ رِفْــــــدُ الحَيِّ يَنْتَظِــــــرونَـــــــهُ
ولَنِعْمَ حَشْوُ الدِّرْعِ والسِّرْبالِ

ولَنِعْمَ مَأْوى المُسْتَضيفِ إذا دَعا
والخَيْلُ خارِجَةٌ مِنَ القَسْطالِ

ديوان أوس بن حجر. تح: محمّد يوسف نجم، ص: 108.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَلِيُّ بْنُ عاصِمٍ الأَصْفَهانِيُّ (ت: 220ه=835م) يَمْدَحُ عُبَيْدَ اللهِ بْنَ هِلالٍ:

صَنَعَـــتْ بِأَرْواحِ العُــــــــداةِ سُيوفُهُ
مــــــا كانَ يَصْنَعُ جودُهُ في المالِ

نَفْسي فِـــــداؤُكَ أَيُّ لَيْثِ كَريهَـــــــةٍ
نَدْعـــــــو بِهِ والمُعْلِمــــــــــــونَ: نَزالِ

والخَيْلُ قاصِدَةٌ عَلى قَصْدِ الفَتى
نَحْــــــــوَ الحُتـــــــــــــوفِ كَأَنَّهُنَّ مَتالي

مَــــــــــدَّتْ سَنابِكُها عَلَيْكَ سُرادِقًا
نُسِجَتْ مَضارِبُهُ مِنَ القَسْطالِ

[المُعْلِمُ: الفارِسُ يَضَعُ عَلامَةً يُعْرَفُ بِها وذَلِكَ لِشَجاعَتِهِ. المَتالي: الأُمَّهاتُ يَتْبَعُهُنَّ أَوْلادُهُنَّ. السَّنابِكُ: جَمْعُ سُنْبُكٍ: طَرَفُ الحافِرِ. السُّرادِقُ: كُلُّ ما أَحاطَ بِالشَّيْءِ]

ابن المعتز، طبقات الشعراء. تح: فراج، ج: 1، ص: 356.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ السِّتالِيُّ الخَروصِيُّ (ت: 676ه=1277م) يَمْدَحُ آلَ نَبْهانَ ويَصِفُ شَجاعَتَهُمْ:

تَرْمي بِهِمْ غَمْرَةُ الهَيْجا وتَنْقُلُهُمْ
في مُشْمَعِلٍّ مِنَ القَسْطالِ شِخْتيتِ

مِنْ كُلِّ أَرْوَعَ نَدْبِ القَلْبِ مُنْصَلِتٍ
بِكُلِّ أَجْرَدَ عَضْبِ الغَرْبِ إصْليتِ

[المُشْمَعِلُّ: المُتَفَرِّقُ. الشِّخْتيتُ: الغُبارُ السّاطِعُ، مِنَ الشَّخْتِ: الَّذي هُوَ الضاوي الدَّقيقُ. المُنْصَلِتُ: الماضي في الأُمورِ، وأرادَ: السَّيْفَ. العَضْبُ: القاطِعُ. الغَرْبُ: حَدُّ السَّيْفِ. الإصْليتُ: الماضي في الضَّريبَةِ]
ديوان الستالي. تح: التنوخي، ص: 77.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشِّدياقُ (1291ه=1874م) يَذْكُرُ حادِثَةَ خَلْعِ الإِمْبراطورِ الفِرَنْسِيِّ:

فَلَمّا دَرَتْ باريسُ ذَا الخَطْبَ أَعْوَلَتْ
وضَجَّتْ وباتَتْ في شُجونٍ ووَلْوالِ

وقالَتْ مَنَتْني دَوْلَــــــةٌ قَيْصَرِيَّـــــــةٌ
بِإهْــــــلاكِ أَجْنادي وإِتْلافِ أَمْوالي

وإِنَّ صَلاحي دَوْلَــــــــةٌ جَمْهـورِيَّـةٌ
تُسَدِّدُ أَعْمـــالي وتُصْلِــــــحُ أَحْوالي

فَنادَتْ بِخَلْـــــعِ الإمْبراطـــورِ وابْنِهِ
وثارَتْ لِأَخْـــــذِ الثّأْرِ ثَوْرَةَ قَسْطالِ

لويس شيخو، تاريخ الآداب العربية. ص: 214.— المصدر

ع

ع

05 —

قُسْطالَة

اسم
5.1معنى من معاني «قُسْطالَة»

* القُسْطالَةُ: قَوْسُ قُزَحَ. (انظر: ق س ط ن).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بْنُ يَزيدَ الحِصْنِيُّ (ت: 230ه=845م) يَصِفُ خَيْلَ الحَلْبَةِ:

وأَقْبَلْـــــــــــــنَ في أَمَــــــــــــــدٍ باهِرٍ
كَـــــــــــــــــــــما يُقْبِلُ الوابِلُ المُثْجِمُ

تَوابِعُ فَوْضى ومُرْفَضَّـــــــــــــــــةٌ
كَما ارْفَضَّ مِنْ سِلْكِهِ المِنْظَمُ

أَوِ السِّرْبُ سِرْبُ القَطا راعَهُ
مِـــــــــنَ الجَــــــــــوِّ سَوْذانَقٌ مُظْلِمُ

فَواصِلُ مِــــــــنْ كُلِّ قُسْطالَةٍ
كَـــــــــــأَنَّ عَثانينَـــــــــــــــــــــها العَنْدَمُ

[ارْفَضَّ: سالَ وتَتابَعَ. السَّوْذانَقُ: الصَّقْرُ. العَنْدَمُ: صِباغٌ أَحْمَرُ]

ديوان الحصنيّ. تح: إبراهيم صالح، ص: 67.— المصدر
”وقالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"القُسْطالةُ: قَوْسُ قُزَحَ، وهِيَ القُسْطانَةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 390.— المصدر
06 —

قَسْطَل

اسم
6.1معنى من معاني «قَسْطَل»

* القَسْطَلُ: الغُبارُ السّاطِعُ. وَهُوَ بِتاءٍ أَيْضًا. (ج) قَساطِلُ. (انظر: ق س ط ن، ق ص ط ل، ك س ط ل)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ جُشَمُ بْنُ حُبْرانَ (400ق.ه=234م) يوصي ابْنَيْهِ:

إنْ رُمْتُما السُّؤْدُدَ في النّاسِ فَهُمْ

يَســـــــــــــودُهُمْ مَنْ يَعْتَليهِمْ بِالكَرَمْ

حَتّى أَتى القَسْطَلَ مِنْـــــــها والقُتُمْ

مُفَرِّجُ البَأْساءِ والكَـــــــــــــــــرْبِ المُلِمّ

دعبل الخزاعي، وصايا الملوك. تح: نزار أباظة، ص: 124.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كَعْبُ بْنُ مالِكٍ (3ه=625م) يَرْثي حَمْزَةَ بْنَ عَبْدِ المُطَّلِبِ وشُهَداءَ أُحُدٍ رَضِيَ اللهُ عَنْهُمْ:

وأَشْياعُ أَحْمَـــــــدَ إذْ شايَعوا
عَـــــلى الحَــــــــقِّ ذي النّــــورِ والمَنْهَجِ

فَما بَرِحوا يَضْرِبونَ الكُماةَ
ويَمْضونَ في القَسْطَلِ المُرْهِجِ

[ الكُماةُ: جَمْعُ كَمِيٍّ، وهُوَ: الشُّجاعُ. المُرْهِجُ: المُرْتَفِعُ]
ابن إسحاق، السيرة النبوية. تح: المزيدي، ج: 1، ص: 362.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُخَيلَةَ الحِمّانِيُّ (ت: 145ه=762م) يَصِفُ صَيْدَ نَعامَةٍ:

فَانْصاعَ يَسْعى بِالصَّعيدِ الهايِلِ

يَلْحَــــــــــــــبُ عَنْ ذي مَيْعَةٍ مُعاجِلِ

حَتّى دَنا مِــــــنْ وَهَــــــجِ القَساطِلِ

مِــــــــنْ ذاتِ زِفٍّ ساقِطِ الخَمايِلِ

فَاخْتَلَفا تَحْـــــــــــــــــتَ جَناحِ المايِلِ

بِضَرْبَــــــــــــــــــةٍ حَثيثَةٍ في الصّاعِلِ

مَنْقوشَـــــــــــةِ الرَّقينِ والخَصايِلِ

[ الصَّعيدُ الهايِلُ: التُّرابُ المُجْتَمِعُ العَظيمُ. يَلْحَبُ: يَكْشِفُ. المَيْعَةُ مِنْ كُلِّ شَيْءٍ: أَوَّلُهُ. الزِّفُّ: صِغارُ ريشِ النَّعامِ. الجَناحُ: الطّائِفَةُ مِنَ الشَّيْءِ. اخْتَلَفا: يَقْصِدُ المُهْرَ والنَّعامَ. الصّاعِلُ: النَّعامُ الخَفيفُ]
شعر أبي نخيلة الحماني. تح: عدنان الخطيب، ص: 158.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الظّاهِرُ بيبَرْسُ (ت: 676ه=1277م) يُبَشِّرُ قاضِيَ قُضاةِ الشّامِ شَمْسَ الدّينِ أَحْمَدَ بْنَ خَلِّكانَ بِما حَقَّقَهُ مِنْ نَصْرٍ:

"سَرَّ اللهُ خاطِرَ المَجْلِسِ السّامي، وأَطْلَعَ عَلَيْهِ وُجوهَ البَشائِرِ سَوافِرَ (...) وقَدْ كُنّا لَمّا وَصَلْنا الشّامَ بِالعَزْمِ الَّذي نَفَّرَتْهُ دَواعي الجِهادِ (...) وأَعْقَبْنا ذَلِكَ بِما رَأَيْناهُ أَوْلى بِالتَّقْديمِ وأَحْرى (...) وهِيَ صَفَدٌ الَّتي باءَ بِإِثْمِها حامِلُها عَلى النَّصْرانِيَّةِ (...) كَمِ اسْتُبيحَ بِسَبَبِها لِلإِسْلامِ مِنْ حِمًى، وكَمْ اسْتَرَقَّ الكُفّارُ بِواسِطَتِها مُسْلِمَةً مِنَ الأَحْرارِ ومُسْلِمًا، وكَمْ تَسَرَّبَ مِنْها جَيْشُ الفِرَنْجِ إِلى بِلادِ المُسْلِمينَ، فَحازوا مَغْنَمًا وقَوَّضوا مَعْلَمًا، فَنازَلْناها مُنازَلَةَ اللَّيْلِ بِانْعِقادِ القَساطِلِ، وطالَعْناها مُطالَعَةَ الشَّمْسِ بِبَريقِ المُرْهَفاتِ وأَسِنَّةِ الذَّوابِلِ".

[الذَّوابِلُ: الرِّماحُ]
اليونيني، ذيل مرآة الزمان. ج: 2، ص: 340.— المصدر

11.1معنى من معاني «قَسْطَلانِيَّة»

* القَسْطَلانِيَّة: الحُمْرَةُ تَكونُ في قَوْسِ قُزَحَ. (وانْظُرْ: ق س ط ن).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"والقَسْطَلانِيَّةُ: نُدْأَةُ الشَّفَقِ، أَوْ نُدْأَةُ قَوْسِ قُزَحَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1155.— المصدر

قَسْطَلَة

اسم
12.1معنى من معاني «قَسْطَلَة»

* القَسْطَلَةُ: هَديرُ الإِبِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وجَمَلٌ ذو قَسْطَلَةٍ: أَيْ هَديرٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 84.— المصدر
”وقالَ الشّاعِرُ (466ه=1074م) مِنْ شَواهِدِ ابْنِ سِنانٍ الخَفّاجِيِّ، يفْتَخِرُ بِبَأْسِهِ عَلى عَدوِّهِ:

إنّي عَلى رَأْسِ العَدُوِّ وتَحْتَهُ
لَلُغامُ قَسْطَلَةٍ وحَيَّةُ وادِ

[اللُّغامُ: زَبَدُ أَفْواهِ الإبِلِ]
ابن سنان الخفاجي، سر الفصاحة. تح: إبراهيم شمس الدين، ص: 172.— المصدر
12.2معنى من معاني «قَسْطَلَة»

وــــــــــ: حِسُّ النَّهْرِ وصَوْتُهُ إذا انْثَجَّ مِنْ مَكانٍ بَعيدٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"ونَهْرٌ قِسْطالٌ: إذا كانَتْ لَهُ قَسْطَلَةٌ؛ وهُوَ حِسُّهُ إذا انْثَجَّ مِنْ مَكانٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 84.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَعْروف الأَرْنَؤوط (1355ه=1936م) تاريخُ النَّشْرِ، يَتَحَدَّثُ عَنْ كريسْتِيا وسافو بَطَلَيْ قِصَّتِهِ (صَوْتُ الجَبَلِ):

"جَلَسَ كريستِيا عَلى عَمودٍ رُخامِيٍّ، كانَ جاثِمًا عَلى الأَرْضِ، فَجَلَسَتْ سافو حِيالَهُ وطَفِقا يَنْظُرانِ (...) لَمْ يَجْرُؤْ كريسْتِيا عَلى الهَمْسِ، فَلَقَدْ أَمالَتْهُ الصُّوَرُ البارِعَةُ إِلى غَرَقٍ، وأَنْسَتْهُ ذَلِكَ الأَلَمَ الَّذي تَحَيَّفَهُ (...) بَلْ لَقَدْ أَلْهَتْهُ قَسْطَلَةُ الماءِ في الصَّعيدِ المَهْجورِ عَنْ أولَئِكَ النّاسِ الَّذينَ نَفَروا إلى قِتالِ قَيْصَرَ".

مجلة الرسالة. العدد: 159، 1936م، ص: 1176.— المصدر
