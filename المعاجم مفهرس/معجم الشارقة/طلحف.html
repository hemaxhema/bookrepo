طلحف
طِلْحاف(اسم) طَلَحْف/ طِلْحِف/ طِلَحْف(اسم) طَلَحْفى/ طِلَحْفى(اسم) طِلَحْفِيّ(اسم) طِلْحيف(اسم) طِلَّحْف(اسم)

المعانى الكلّيّة

  1. الشِّدَّةُ
01 —

طِلْحاف

اسم
1.1معنى من معاني «طِلْحاف»

* الطِّلْحافُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (انْظُرْ: ط ل خ ف).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"وَضَرْبٌ ‌طِلْحافٌ وَطِلْخافٌ: شَديدٌ، بِالحاءِ وَالخاءِ ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 1203.— المصدر
02 —

طَلَحْف/ طِلْحِف/ طِلَحْف

اسم
2.1معنى من معاني «طَلَحْف/ طِلْحِف/ طِلَحْف»

* الطَّلَحْفُ، والطِّلْحِفُ، والطِّلَحْفُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (انْظُرْ: ط ل خ ف).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الزِّبْرقانُ بنُ بَدْرٍ (13هـ=634م) وَقَدْ حَمَلَ صَدَقاتِ قَوْمِهِ إِلى أَبي بَكْرٍ، رَضِيَ اللهُ عَنْهُ:

مُسْتَحْقِبو حَلَقِ الماذِي بِخَفْرَتِهِ
ضَرْبٌ ‌طِلَحْفٌ وَطَعْنٌ بَيْنَهُ خَضِدُ

ياقوت الحموي، معجم البلدان. ج: 4، ص: 82.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”أَنْشَدَ ابْنُ الأَعْرابيِّ (ت: 231هـ=845م) يَصِفُ ضَرْبًا شَديدًا:

ضَرْبًا طِلَحْفًا لَيْسَ ‌بالمُحَجْحَجِ

ابن منظور، لسان العرب. تح: أمين عبد الوهاب، وآخر، ج: 3، ص: 55.— المصدر
03 —

طَلَحْفى/ طِلَحْفى

اسم
3.1معنى من معاني «طَلَحْفى/ طِلَحْفى»

* الطَّلَحْفَى، والطِّلَحْفَى مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (انْظُرْ: ط ل خ ف).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"ضَرْبٌ طِلَحْفٌ وطِلَخْفٌ ‌وطِلَحْفَى وطِلَخْفَى: شَدِيدٌ ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ص: 1142.— المصدر
04 —

طِلَحْفِيّ

اسم
4.1معنى من معاني «طِلَحْفِيّ»

* الطِّلَحْفِيُّ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (انْظُرْ: ط ل خ ف).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عليُّ بْنُ أَبي طالِبٍ (ت: 40هـ=661م):

"وَلَقَدْ وَطَّنْتُمْ أَنْفُسَكُمْ عَلى الطَّعْنِ وَالضَّرْبِ ‌‌الطِّلَحْفِيِّ، وَمُبارَزَةِ الأَقْرانِ".

المجلسي، بحار الأنوار. ج: 32، ص: 171.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 384هـ=995م):

"ضَرَبْتُهُ ضَرْبًا طِلَحْفِيًّا؛ أَيْ: شَديدًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 3، ص: 280.— المصدر
05 —

طِلْحيف

اسم
5.1معنى من معاني «طِلْحيف»

ع

الخليل، العين. تح: المخزومي وآخر،ج: 3، ص: 334.— المصدر
