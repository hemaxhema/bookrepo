غضرف
غُضْروف(اسم) غُضْروفِيّات(اسم) غَنْضَرِف(اسم)

المعانى الكلّيّة

  1. العِظامُ الليِّنَةُ
01 —

غُضْروف

اسم
1.1معنى من معاني «غُضْروف»

* الغُضْروفُ: كُلُّ عَظْمٍ لَيِّنٍ رَخْصٍ في أَيِّ مَوْضِعٍ كانَ مِنَ الجِسْمِ. (ج) غَضاريفُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ بَحيرا الرّاهِبُ (ت: 25ق.هـ=597م) يَصِفُ علاماتِ النُّبُوَّةِ في جَسَدِ رَسولِ اللهِ صلى الله عليه وسلم:

"وإِنّي لَأَعْرِفُهُ بِخاتَمِ النُّبُوَّةِ أَسْفَلَ مِنْ غُضْروفِ كَتِفِهِ مِثْلَ التُّفّاحَةِ".

ابن أبي شيبة، الكتاب المصنف. تح: الحوت، ج: 7، ص: 327.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ سَلْمانُ الفارِسيُّ (ت: 36هـ=657م) يَصِفُ رَسولَ اللهِ صلى الله عليه وسلم على لِسانِ أَحَدِ رُهْبانِ العَرَبِ:

"وفيهِ ثَلاثُ آياتٍ: يَأْكُلُ الهَدِيَّةَ، ولا يَأْكُلُ الصَّدَقَةَ، وعِنْدَ غُضْروفِ كَتِفِهِ اليُمْنى خاتَمُ النُّبُوَّةِ، مِثْلُ بَيْضَةِ الحَمامَةِ، لَوْنُها لَوْنُ جِلْدِهِ".

ابن أبي شيبة، المصنف. تح: عوامة، ج: 20، ص: 266.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ:

حُرَّ المُحَــــــــــــــيّا لَيِّنَ الغُضْروفِ

كَأَنَّ تَحْتَ المِرْطِ والشُّفوفِ

رَمْلًا حَبَا مِن عُقَــــــــــدِ الغَــرِيفِ

إلى عِنانَيْ ضَــــــامِرٍ لَـــــــــــــطِـــيفِ

ديوان رؤبة. تح: البروسي، ص: 102.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ خَلِّكانَ (ت:681هـ=1282م) يُتَرْجِمُ لإِسْحاقَ المَوْصِلِيِّ ويَذْكُرُ أَمْرَ أَميرِ المُؤْمِنينَ بِقَطْعِ أُذُنِهِ:

"فَقَطَعَ غُضْروفَ أُذُنِهِ مِنْ خارِجٍ".

ابن خلّكان، وفيات الأعيان. تح: إحسان عباس، ج: 1، ص: 412.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الآلوسيُّ (ت: 1270هـ=1854م) يَذْكُرُ ما يَتَرَتَّبُ عَلى رَأْيِ ابْنِ حَزْم في دَعْواهُ عَودَ الضَّمِيرِ في قولِ اللهِ ﷻ: ﴿أَوۡ لَحۡمَ ‌خِنزِيرٖ ‌فَإِنَّهُۥ رِجۡسٌ﴾ (الأنعام: 145) على كَلِمَةِ لَحْمٍ، لا عَلى الخِنْزيرِ:

"فَيَحِلُّ شَحْمُهُ وغُضْرُوفُه وعَظْمُه وجِلْدُه عِنْدَه لِذَلِك".

الآلوسيّ، روح المعاني. تح: المغربي، ج: 16، ص: 298.— المصدر
1.2معنى من معاني «غُضْروف»

o والانْزِلاقُ الغُضْروفِيُّ (في الطِّبِّ): (انظر: ز ل ق)

1.3معنى من معاني «غُضْروف»

o والغُضْروفُ الحَلْقِيُّ: غُضْروفٌ بِشَكْلِ الخاتَمِ في الـمِنْطَقَةِ السُّفْلى للحَنْجَرَةِ، يَتَّحِدُ بِمَفاصِلَ مَعَ الغُضْروفِ الدَّرَقِيِّ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ مَجَلّةُ البَيانِ لِلْبَرْقُوقيِّ (1337هـ=1918م، تاريخُ الطَّبْعِ):

"وَمِنَ الطُّرُقِ المُسْتَحْسَنَةِ لِإِيقافِها عِنْدَ الأَطْفالِ وَضْعُ طَرَفِ الإِصْبَعِ عَلى الغُضْروفِ الحَلْقِيِّ (البُروزِ المَوْجودِ في العُنُقِ تَحْتَ الذَّقْنِ بِبِضْعِ أَصابِعَ) بِلُطْفٍ وَتَحْريكُهُ قَليلًا".

البرقوقي، مجلة البيان، ج: 39، ص: 32.— المصدر
1.4معنى من معاني «غُضْروف»

o والغُضْروفُ الدَّرَقِيُّ أو التُّرْسِيُّ: أَكْبَرُ غَضاريفِ الحَنْجَرَةِ يُكَوِّنُ نُتوءًا ظاهِرًا عِنْدَ الرَّجُلِ يُسَمّى تُفّاحَةَ آدَمَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سينا (ت: 428هـ=1037م):

"الحَنْجَرَةُ عُضْوٌ غُضْروفِيٌّ خُلِقَ آلَةً للصَّوْتِ، وَهُوَ مُؤَلَّفٌ مِنْ غَضارِيفَ ثَلاثَةٍ: أَحَدُها الغُضْروفُ الَّذي يَنالُهُ الجَسُّ والحِسُّ قُدّامَ الحَلْقِ تَحْتَ الذَّقَنِ، ويُسَمّى الدَّرَقِيَّ والتُّرْسِيَّ؛ إذْ كانَ مُقَعَّرَ الباطِنِ مُحَدَّبَ الظَّهْرِ يُشْبِهُ الدَّرَقَةَ وبَعْضَ التِّرَسَةِ".

ابن سينا، القانون في الطب. تح: الضناوي، ج: 1، ص: 68.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ النَّفيسِ (ت: 687هـ=1288م) يَتَحَدَّثُ عَنْ عَضَلاتِ القَصَبَةِ الهَوائِيَّةِ:

"وقَدْ يُتَوَهَّمُ أَنَّها أَرْبَعُ عَضَلاتٍ، والاثْنَتانِ الأُخْرَيانِ أَصْغَرُ مِنْ هاتَيْنِ، وتَأْتِيانِ مِنَ الأَجْزاءِ السُّفْلِيَّةِ مِنَ الغُضْروفِ الدَّرَقِيِّ، وتَنْتَهِيانِ أَيْضًا عِنْدَ القَصِّ مُحْتَوِيَتَيْنِ على القَصَبَةِ مِنَ الجانِبَيْنِ".

ابن النفيس، شرح تشريح القانون. تح: قطاية، ص: 99.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَحْمود صِدْقي (1303هـ=1886م، تاريخُ الطَّبْعِ) يَذْكُرُ أَرْبِطَةَ الحَنْجَرَةِ:

"وأَمّا الجُزْآن الجانِبِيّانِ فَيَمْتَدّانِ مِنَ القُرونِ الكَبيرَةِ مِنَ العَظْمِ اللّامِيِّ إلى القُرونِ الكَبيرَةِ مِنَ الغُضْروفِ الدَّرَقِيِّ، ويوجَدُ فيهِما أَحْيانًا غُضْروفانِ صَغيرانِ".

محمود صدقي، إرشاد الخواص. ص: 580.— المصدر
1.5معنى من معاني «غُضْروف»

o والغُضْروفُ عَديمُ الاسْمِ: هُوَ غُضْروفٌ مِنْ غَضاريفِ الحَنْجَرَةِ، ويَقَعُ خَلْفَ الغُضْروفِ الدَّرَقِيِّ ويَتَّصِلُ بِهِ بِأَرْبِطَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سينا (ت:428هـ=1037م) يَصِفُ تَشْريحَ الحَنْجَرَةِ:

"والغُضْروفُ الثاني خَلْفَهُ، مُقابِلٌ سَطْحُهُ لِسَطْحِهِ، مُتَّصِلٌ بِهِ بالرِّباطاتِ يَمْنَةً ويَسْرَةً، ومُنْفَصِلٌ عَنْهُ إلى فَوْقُ، ويُسَمّى عَديمَ الاسْمِ".

ابن سينا، أسباب حدوث الحروف. تح: الطيان وآخر، ص: 65.— المصدر
1.6معنى من معاني «غُضْروف»

oوغُضْروفُ الكَتِفِ: رَأْسُ لَوْحِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ التَّنوخيُّ النَّصْرانِيُّ رَسولُ هِرَقْلَ (11هـ=632م) يَصِفُ خاتَمَ النُّبُوَّةِ حِينَ لِقَائِهِ النَّبِيَّ ﷺ:

"فَجُلْتُ في ظَهْرِهِ، فإِذا أَنا بِخاتَمٍ في مَوْضِعِ ‌غُضْروفِ ‌الكَتِفِ مِثْلِ الـمِحْجَمَةِ الضَّخْمَةِ".

البيهقي، دلائل النبوة. تح: عبد المعطي قلعجي، ج: 1، ص: 266.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الطَّبَريُّ (ت: 310هـ=922م) يَذْكُرُ العُضْوَ الذي ضُرِبَتْ بهِ بَقَرَةُ بني إِسْرائيلَ:

"وجائِزٌ أَنْ يَكونَ الّذي أُمِروا أَنْ يَضْرِبوهُ به هُوَ الفَخِذَ، وجائِزٌ أنْ يَكونَ ذَلكَ الذَّنَبَ، ‌وغُضْروفَ ‌الكَتِفِ، وغَيْرَ ذَلِكَ مِن أَبْعاضِها".

الطبري، جامع البيان عن تأويل آي القرآن. تح: عبد الله التركي، ج: 2، ص: 127.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال حَسَنٌ الفَيّوميُّ (ت: 870ه=1466م):

"التَّرْقُوَةُ هي النُّقْرَةُ الـمُنْخَفِضَةُ عِنْدَ ‌غُضْروفِ ‌الكَتِفِ".

حسن الفيومي، فتح القريب المجيب على الترغيب. تح: محمد إسحاق، ج: 9، ص: 137.— المصدر
1.7معنى من معاني «غُضْروف»

o والغُضْروفُ المُتَّكِئُ على المَجْرى: هُوَ المُسَمّى حَديثًا بِلِسانِ المِزْمارِ، مِنْ وَظيفَتِهِ تَسْهيلُ عَمَلِيَّةِ البَلْعِ كَما هُوَ مَعْروفٌ طِبِّيًّا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سينا (ت:428هـ=1037م) يَشْرَحُ تَرْكيبَ الـمَرِيءِ:

"الـمَرِيءُ يَأْخُذُ في الانْبِساطِ إِلَيْهِ ويَنْفُذُ فيه، وخَصوصًا والازْدِرادُ لا يُجامِعُ النَّفَسَ؛ لأَنَّ الازْدِرادَ يُحْوِجُ إلى انْطِباقِ مَجْرى قَصَبَةِ الرِّئَةِ مِنْ فَوْقُ، لِئَلّا يَدْخُلَها الطَّعامُ المارُّ فَوْقَها، ويَكونَ انْطِباقُها بِرُكوبِ الغُضْروفِ المُتَّكِئِ على المَجْرى".

ابن سينا، القانون في الطب. تح: الضناوي، ج: 2، ص: 301.— المصدر
1.8معنى من معاني «غُضْروف»

o والغُضْروفُ المُكَبّي أو الطِّرْجِهالِيُّ: هُوَ غُضْروفٌ مِنْ غَضاريفِ الحَنْجَرَةِ، وسُمِّيَ كذَلِكَ لِأَنَّهُ شَبيهٌ بقَصْعَةٍ مَكْبوبَةٍ عَلَيْها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سينا (ت: 428هـ=1037م) يَصِفُ تَشْريحَ غَضاريفِ الحَنْجَرَةِ:

"والغُضْروفُ الثّالِتُ كَقَصْعَةٍ مَكْبوبَةٍ عَلَيْها، وهُوَ مُنْفَصِلٌ عَنِ الدَّرَقِيِّ، مَرْبوطٌ بالذي لا اسْمَ لَهُ مِنْ خَلْفُ، بِمَفْصِلٍ مُضاعَفٍ يَحْدُثُ مِنْ زائِدَتَيْنِ تَصْعَدانِ مِنَ الذي لا اسْمَ لَهُ، وتَسْتَقِرّانِ في نُقْرَتَيْنِ لَهُ، ويُسَمّى المُكَبّي والطِّرْجِهالِيَّ".

ابن سينا، أسباب حدوث الحروف. تح: الطيان وآخر، ص: 65.— المصدر
1.9معنى من معاني «غُضْروف»

o والغُضْروفُ الهَرَمِيُّ: هُوَ أَحَدُ الغُضْروفَيْنِ المُكَوِّنَيْنِ جُزْءًا رَئيسِيًّا مِنْ غَضاريفِ الحَنْجَرَةِ، ولَهُما وَظيفَةٌ حَرَكِيَّةٌ في الجِهازِ الصَّوْتِيِّ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ كَمال بِشْر (1390هـ=1970م، تاريخُ الطَّبْعِ) يَشْرَحُ كَيْفِيَّةَ نُطْقِ صَوْتِ الهَمْزَةِ:

"ما نُسَمِّيِه اليَوْمَ بالغُضْروفِ الهَرَمِيِّ...، وهُما في الواقِعِ غُضْروفانِ هَرَمِيّانِ، وسُمِّيا كذلِكَ لأَنَّ كُلًّا مِنْهُما يُشْبِهُ الهَرَمَ. وهذانِ الغُضْروفانِ يُكَوِّنانِ جُزْءًا رَئيسِيًّا مِنْ غَضاريفِ الحَنْجَرَةِ، ولَهُما قُدْرَةٌ على الحَرَكَةِ، فَقَدْ يَقْتَرِبُ أَحَدُهُما مِنَ الآخَرِ أَوْ يَبْتَعِدُ عَنْهُ. وقَدْ يَشْتَدُّ الاقْتِرابُ بَيْنَهُما حَتّى تَلْتَقِيَ قِمَّتاهُما فَيَنْسَدُّ فَراغُ الحَنْجَرَةِ. وهذا هُوَ ما يَحْدُثُ بالفِعْلِ عِنْدَ النُّطْقِ بالهَمْزَةِ، حَيْثُ يُؤَدّي هذا الالتِقاءُ إلى انْطِباقِ الوَتَرَيْنِ الصَّوْتِيَّيْنِ انْطِباقًا تامًّا".

كمال بشر، دراسات في علم اللّغة. ص: 81.— المصدر
02 —

غُضْروفِيّات

اسم
2.1معنى من معاني «غُضْروفِيّات»

* الغُضْروفِيّاتُ: أَسْماكٌ فَقارِيّاتٌ فَكِّيَّةٌ بِزَوْجَيْنِ مِنَ الزَّعانِفِ، وفَتْحَتَيْ أَنْفٍ، وحَراشِفَ، وهَيْكَلٍ مِنَ الغَضاريفِ بَدَلًا مِنَ العَظْمِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ زُهَيْرٌ الكَرْمِيُّ (1398هـ=1978م، تاريخُ الطَّبْعِ):

"فَفي بَعْضِ الأَسْماكِ الغُضْروفِيَّةِ مَثَلًا، تَحْتَفِظُ الأُنْثى بِالبَيْضِ في قَنَواتٍ خَاصَّةٍ داخِلَها إلى أَنْ يَفْقِسَ البَيْضُ لِتَخْرُجَ الصِّغارُ قَادِرَةً على الحَرَكَةِ".

غَنْضَرِف

اسم
3.1معنى من معاني «غَنْضَرِف»

* الغَنْضَرِفُ مِنَ النِّساءِ: المَرْأَةُ الضَّخْمَةُ التي لَها خَواصِرُ وبُطونٌ وغُضونٌ.

د

"وامْرَأَةٌ غَنْضَرِفٌ وغَنْضَفيرٌ: إذا كانَتْ ضَخْمَةً لَها خَواصِرُ وبُطونٌ وغُضونٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)
