01 —

غَسا

فعل
1.1معنى من معاني «غَسا»

* غَسا اللَّيْلُ يَغْسو غَسْوًا وغُسُوًّا: أَقْبَلَ بِظَلامِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال عَمْرو بْنُ أحْمَر (ت: 75هـ=695م) يَذْكُرُ تَعَرُّضَهُ لِنَوائِبِ الدَّهْرِ:

فَلَمّا غَسا لَيْلي وأَيْقَنْتُ أَنَّها
هِيَ الأُرَبى جاءَتْ بِأُمِّ حَبَوْكَرَى

ابن سيده، المخصص. تح: خليل جفال، ج: 4، ص: 489.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ حالَهُ:

وَأَعْسِفُ اللَّيْلَ إِذا اللَّيْلُ غَسا

وَاعْرَنْكَسَتْ أَهْوالُهُ وَاعْرَنْكَسا

ديوان رؤبة بن العجاج. تح: البروسي، ص: 32.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال حازمٌ القَرْطاجَنِّي (ت: 684هـ=1285م) يَتَغَزَّلُ:

وَجْهٌ بَدا بِمُشْرِقِ الحُسْنِ بِهِ
بَدْرٌ مُنيرٌ تَحْتَ لَيْلٍ قَدْ غَسا

حورية روّاق، مقصورة حازم القرطاجني. ص: 92.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ شِهابُ الدّينِ الـمَكيُّ (ت: 1274هـ=1858م) يَبْتَهِلُ:

وَاسْـتَجْلِ كَــواكِبَ طَلْعَتِــهِ
إِنْ جَـنَّ دُجــى خَطْــبٍ وَغَسـا

ديوان محمد شهاب الدين، ص: 298.— المصدر
1.2معنى من معاني «غَسا»

وـــــــــــ الشَّيْخُ غُسِيًّا: كَبِرَ وهَرِمَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وشَيْخٌ غاسٍ: طالَ عُمُرُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 433.— المصدر
1.3معنى من معاني «غَسا»

وــــــــــ اللَّيلُ عَلى فُلانٍ: أَظْلَمَ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُحَمَّدُ بنُ حَبيبٍ (ت: 245هـ=860م) يَذْكُرُ حَنْظَلَةَ بنَ عُثْمانَ بنِ فاتِكٍ وقَدْ أَعَدَّ مَكيدَةً:

"وأَرْسَلَ بُنَيَّةً لَهُ حينَ غَسا عَلَيْهِ اللَّيْلُ إِلَيْهِمْ رَجُلًا رَجُلًا".

ابن حبيب، المحبر، ص: 202.— المصدر
1.4معنى من معاني «غَسا»

* وغَسى اللَّيْلُ يَغْسى غَسًا: أَظْلَمَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"غَسِيَ اللَّيْلُ يَغْسى، وغَسا يَغْسو ويَغْسى، وأَغْسى يُغسي، ثَلاث لُغات فَصيحَة ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 846.— المصدر
02 —

غَسِيَ

فعل
2.1معنى من معاني «غَسِيَ»

* غَسِيَ اللَّيْلُ يَغْسى غَسًى: أَقْبَلَ بِظَلامِهِ. (والشّينُ لُغَةٌ فيهِ).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال عَمْرو بْنُ أحْمَر (ت: 75هـ=695م) يَذْكُرُ فَتًى مِقْدامًا:

كَأَنَّ الصَّقْرَ يَقْلِبُ مُقْلَتَيْهِ
إِذا نَفَضَ العُيوبَ وقَدْ خَفينا

كَأَنَّ اللَّيْلَ لا يَغْسى عَلَيْهِ
إِذا زَجَرَ السَّبَنْتاةَ الأَمونا

شعر عمرو بن أحمر الباهلي. تح: حسين عطوان، ص: 163.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعيُّ (ت: 216هـ=831م):

"يُقالُ: غَسا اللَّيْلُ يَغْسُو غُسُوًّا، وَغَسِيَ يَغْسَى ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، الألفاظ. تح: قباوة، ص: 410.— المصدر
03 —

أَغْسَى

فعل
3.1معنى من معاني «أَغْسَى»

* أَغْسى اللَّيْلُ: أَظْلَمَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بنُ غَلْفاءَ (ت: 27ق.هـ=596م) يَهْجو:

هَجَوْا شَرَّ يَرْبوعٍ رِجالًا وَخَيْرَها
نِساءً، إِذا أَغْسى الظّلامُ تُزارُ

ابن منظور، لسان العرب. تح: أمين عبد الوهاب وآخر، ج: 10، ص: 73.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال حَضْرَميُّ بْنُ عامِر (ت: 17هـ=638م) يَرْثي:

أَخي ثِقَةٍ إِذا ما اللَّيْلُ أَغْسى
إِلى تَمْريدِ حَبْلي قَدْ كَفاني

قَطَعْتُ قَرينَتي عَنْهُ فَأَغْنى
غِناهُ فَلَنْ أَراهُ ولَنْ يَراني

الأسود الغندجاني، فرحة الأديب. تح: محمد سلطاني، ص: 201.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"غَسا اللَّيْلُ، وأَغْسى أَصْوَبُ: إِذا أَظْلَمَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 433.— المصدر
”وقال كَيْسانُ بْنُ الـمُعَرّف الهُجَيْميّ (ت: 310هـ=922م):

هَجَوْا شَرَّ يَرْبوعٍ رجالًا وخَيْرَها
نِساءً إِذا أَغْسى الظّلامُ تُزارُ

الزّجّاج، فعلتُ وأفعلتُ. تح: ماجد حسن الذهبي، ص: 175.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال الشّابِّي (ت: 1353هـ=1934م):

أَيُّها الشَّعْبُ أَنْتَ طِفْلٌ صَغيرٌ
لاعِبٌ بِالتُّرابِ واللَّيْلُ مُغْسِ

ديوان أبي القاسم الشّابيّ ورسائله. ص: 118.— المصدر
3.2معنى من معاني «أَغْسَى»

وـــــــــــ فُلانٌ: دَخَلَ فى أَوَّلِ الظُّلْمَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَسْعَديُّ (206هـ=821م) مِنْ رُواةِ الشَّيْبانيِّ:

"قَدْ أغْسَيْتُ: أَمْسَيْتُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 5.— المصدر
3.3معنى من معاني «أَغْسَى»

وـــــــــــ اللَّيْلُ فُلانًا: أَلْبَسَهُ ظَلامَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ أَرْضًا:

وَأَرْضِ جِنٍّ تَحْتَ حَرٍّ سَخْتِ

لَها نِعافٌ ‌كَهَوادي ‌البُخْتِ

يُغْسى عَلى أَلْوانِهِنَّ الكُمْتِ

أَوْطَفُ مِنْ وادِقِ لَيْلٍ هَفْتِ

ديوان رؤبة بن العجاج، البروسي، ص: 24.— المصدر
3.4معنى من معاني «أَغْسَى»

o أَغْسِ مِنَ اللَّيْلِ: لا تَسِرْ أَوَّلَهُ وانْتَظِرْ آخِرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ السِّكّيتِ (ت: 244هـ=858م):

"يَقولُ الرَّجُلُ لِلرَّجُلِ إِذا أَرادَ السَّيْرَ بِاللَّيْلِ: أَغْسِ مِنَ اللَّيْلِ شَيْئًا، وَاسْدُفْ عَنّا مِنَ اللَّيْلِ شَيْئًا ثُمَّ ارْتَحِلْ، أَيْ: حينَ يَمْضي بَعْضُ اللَّيْلِ وَيَخِفُّ عَنّا وَيَبْقى بَعْضُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، الألفاظ. تح: قباوة، ص: 298.— المصدر
04 —

غاسي

اسم
4.1معنى من معاني «غاسي»

* الغاسي مِنَ البَلَحِ: الَّذي يَسْقُطُ بُسْرًا وهُوَ أَخْضَرُ فَيَنْضَجُ بَعْدَ ذَلِكَ، واحِدَتُهُ بِهاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السُّلَميُّ (206هـ=821م) مِنْ رُواةِ الشَّيْبانيِّ:

"الغاسي: الكِمْرُ، الواحِدَةُ غاسِيَة، يُقَطِّعُهُ بُسْرًا ثُمَّ يَنْضَجُ بَعْدَما يُقْطَعُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 17.— المصدر
05 —

غَسا

اسم
5.1معنى من معاني «غَسا»

* الغَسا: ظُلْمَةُ اللَّيْلِ. (ج): غُسْيانٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال الأخْنَسُ بْنُ نَعْجَة الكَلْبيّ (ت: 13ق.هـ=609م):

هَلّا سَأَلْتَ بَني صَعْبٍ بِخُبْرِهِمُ
والحَيَّ مِنْ قاسِطٍ حَيَّ بْنِ قَوّادِ

أَنّي صَبَحْتُ غَداةَ الشّيحِ خَيْلَهُمُ
عِنْدَ الغَسا مِثْلَ سيدِ الأَمْسَحِ الغادي

عبد عون الروضان، موسوعة شعراء العصر الجاهلي. ص: 20.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّاعِرُ (180هـ=796م، مِنْ شَواهِدِ الكَلْبِيِّ) يَفْتَخِرُ بِقَوْمِهِ:

هَلّا سَأَلَتِ بَني سَعْدٍ بِخُبْرِهِمُ
والحَيَّ مِنْ قاسِطٍ حَيَّ بْنِ قَوّادِ

إِنّا صَبَحْنا غَداةَ الرَّوْعِ خَيْلَهُمْ
تَحْتَ الغَسا مِثْلَ سيدِ الأَمْسَحِ الغادي

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 1.— المصدر
06 —

غَساة

اسم
6.1معنى من معاني «غَساة»

* الغَساةُ: البَلَحَةُ الصَّغيرَةُ. (ج) غَسَواتٌ، وغَسَياتٌ، وغَسًا، وغَسًى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ السِّكّيت (ت: 244هـ=858م):

"يُقالُ لِلْبَلَحِ: غَساةٌ، والجَميعُ غَسًى، مِثال نَواةٍ ونَوًى ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: محمد أبو الفضل، ج: 6، ص: 480.— المصدر
6.2معنى من معاني «غَساة»

وــــــــــ: البَلَحُ، واحِدَتُهُ غَساةٌ، (ج) غَسَواتٌ، وغَسَياتٌ، وغَسًى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو حَنيفَةَ الدَّيْنَوَريُّ (ت: 282هـ=895م):

"الغَسا: البَلَحُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 37.— المصدر
6.3معنى من معاني «غَساة»

وــــــــــ: الخَلالَةُ؛ وهِيَ مايَبْقَى في أُصولِ السَّعَفِ مِنَ التَّمْرِ، أَوِ البَلَحَةُ الصَّغيرَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"الغَسا واحِدَتُها غَساةٌ، وهِيَ الخَلالَةُ أَوِ البَلَحَةُ الصَّغيرَةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1072.— المصدر

غَسْو

اسم

* الغَسْوُ: النَّبِقُ، واحِدَتُهُ ‌غَسْوَةٌ.

ع

غُسُوّ

8.1معنى من معاني «غُسُوّ»

ع

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 107.— المصدر
