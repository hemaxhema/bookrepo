01 —

كَنَدَ

فعل
1.1معنى من معاني «كَنَدَ»

* كَنَدَ (بِفَتْحِ النّونِ) فُلانٌ يَكْنُدُ ويَكْنِدُ (بِضَمِّ النّونِ وكَسْرِها) كُنودًا: جَمَعَ، فهُوَ كَنودٌ وكَنّادٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"وَكَنَدَ أَيْضًا:جَمَعَ".

كراع النمل، المنتخب. تح: محمد العمري، ص: 672.— المصدر
1.2معنى من معاني «كَنَدَ»

وــــــــــ: أَساءَ مِلْكَ مَنْ يَمْلِكُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الكَلْبِيُّ (ت: 146هـ=763م):

"الكَنودُ البَخيلُ بِما أُعْطِيَ، الَّذي يَمْنَعُ رِفْدَهُ، ويُجيعُ عَبْدَهُ، ويَأْكُلُ وَحْدَهُ، ولا يُعْطي الباسَةَ تَكونُ في قَوْمِه ولا يَكونُ كَنودًا حَتّى تَكونَ هذِهِ الخِصالُ فيهِ. ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو بكر البيهقي، شعب الإيمان. تح: عبد الحميد حامد ، ج:6، ص: 347.— المصدر
1.3معنى من معاني «كَنَدَ»

وــــــــــ الأَرْضُ: لَمْ تُنْبِتْ

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (الجَدُّ) (ت: 367هـ=977م):

"كَنَدَ كُنودًا كَفَرَ النِّعْمَةَ...، والأَرْضُ: لَمْ تُنْبِتْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، كتاب الأفعال. تح: علي فودة، ص: 226.— المصدر
1.4معنى من معاني «كَنَدَ»

وــــــــــ فلانٌ النِّعْمَةَ كُنودًا وكَنْدًا: كَفَرَها وجَحَدَها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال المُثَقِّبُ العَبْدِيُّ (ت: 36ق.هـ=587م) يَمْدَحُ:

فإِنَّ أَبا قابوسَ عِنْدي بَلاؤُهُ
جَزاءً بِنُعْمى لا يَحِلُّ كُنودُها

ديوان المثقب العبدي. تح: حسن الصيرفي، ص: 147.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القرآنِ الكَريمِ (11هـ=632م) قالَ اللَّهُ ﷻ:

‌﴿إِنَّ ٱلۡإِنسَٰنَ لِرَبِّهِۦ ‌لَكَنُودٞ﴾ ]العاديات: 6[

”وقالَ أَبو زُبَيْدٍ الطّائِيُّ (ت: 62هـ=682م) يَصِفُ أَخْلاقَ رِجالٍ:

إِنْ تَفُتْني فلَمْ أَطِبْ عَنْكَ نَفْسًا
‌غَيْرَ ‌أَنّي ‌أُمْنى ‌بِدَهْرٍ ‌كَنودِ

اليزيدي، الأمالي شعره، جمعية دائرة المعارف، ص: 13.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الكَنودُ: الكَفورُ لِلنِّعْمَة".

الخليل، العين. تح: مهدي المخزومي وآخر، ج: 5، ص: 331.— المصدر
”وقالَ أَبو تَمّامٍ الطّائِيُّ (ت: 231هـ=846م):

خُذْها مُثَقَّفَةَ القَوافي رَبُّها
لِسَوابِغِ النَّعْماءِ غَيْرُ ‌كَنودِ

عبيد الله بن أحمد الميكالي، المنتخل. تح: يحيى الجبوري، ج: 1، ص: 74.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ إِبْراهيمُ المازِنيُّ (ت: 1368هـ=1949م):

يا جاحِدًا فَضْلَ الإِلـ
ـهِ عَلَيْهِ وَيْحَكَ مِنْ كَنُودِ

ديوان المازني، ج: 1، ص: 194.— المصدر
1.5معنى من معاني «كَنَدَ»

وــــــــــ الـمَرْأَةُ الوِصالَ ونَحْوَهُ: أَبَتْهُ وجَحَدَتْهُ، فَهِيَ كَنودٌ وكُنُدٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى الكَبيرُ (ت: 7هـ=628م):

أَحْدِثْ لَها تُحْدِثْ لِوَصْلِكَ إِنَّهُ
كُنُدٌ ‌لِوَصْلِ الزّائِرِ المُعْتادِ

الثعلبي، الكشف والبيان. تح: علي بن عاشور، ج: 10، ص: 272.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِمُ بْنُ سَلاّمٍ (ت: 224هـ=839م):

"امْرَأَةٌ كُنُدٌ: كَفورٌ لِلْمُواصَلَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 3، ص: 425.— المصدر
1.6معنى من معاني «كَنَدَ»

وــــــــــ فُلانٌ الشَّيْءَ كَنْدًا: قَطَعَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى الكَبيرُ (ت: 7هـ=628م) يَمْدَحُ:

فميطي تُميطي بِصُلْبِ الفُؤادِ
وَصولِ حِبَالٍ وكَنّادِها

ديوان الأعشى الكبير. تح: محمود الرضواني، ج: 1، ص: 228.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نَصْرٍ الجَوْهَرِيُّ (ت: 393هـ=1003م):

"وكَنَدَهُ، أَيْ: قَطَعَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: أحمد عطار، ج: 2، ص: 532.— المصدر
1.7معنى من معاني «كَنَدَ»

وــــــــــ أَباهُ: عَقَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رَجُلٌ (ت: 418هـ=1027م) مِنْ شَواهِدِ الوَزيرِ المَغْرِبِيِّ:

"لِأَنَّهُ كَنَدَ أَباهُ".

الوزير الحسين بن علي المغربي، أدب الخواص. ص: 140.— المصدر
02 —

كَنَّدَ

فعل
2.1معنى من معاني «كَنَّدَ»

* كَنَّدَ فُلانٌ عَلَى فُلانٍ:حَقَدَ عَلَيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رَجُلٌ (ت: 418هـ=1027م) مِنْ شَواهِدِ الوَزيرِ المَغْرِبِيِّ:

"‌كَنَّدَ عَلى أَخيهِ فقَتَلَهُ".

الوزير الحسين بن علي المغربي، أدب الخواص. ص: 140.— المصدر
03 —

كِنْدَة

اسم
3.1معنى من معاني «كِنْدَة»

* الكِنْدَةُ: القِطْعَةُ مِنَ الجَبَلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والكِنْدَةُ: القِطْعَةُ مِنَ الجَبَلِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 6، ص: 214.— المصدر
3.2معنى من معاني «كِنْدَة»

o وكِنْدَةُ: إِحْدى قَبائِلِ قَحْطانَ، ويُنْسَبُ إِلَيْها العَديدُ مِنَ الشُّعَراءِ، مِثْلِ امْرِئِ القَيْسِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عُمارَةُ بْنُ عَبيدٍ الهَمْدانيُّ (ت: 202ق.هـ=426م) يَفْتَخِرُ:

ويَوْمَ جُرادٍ لَمْ نَدَعْ لرَبيعَةٍ
وإِخْوَتِها أَنْفًا بِهِ غَيْرَ أَجْدَعا

بضَرْبٍ تَظَلُّ الطَّيْرُ تَقْفو رَشاشَهُ
على الصَّخْرِ حَتَّى تَنْثَني مِنْهُ ظُلَّعا

ودارَتْ على سَبْعينَ مِنْ سَرَواتِهِمْ
رَحى الحَرْبِ مَكْتوفًا بِها ومُدَرَّعا

فأَطْلَقَهُمْ زَيْدٌ رِعايَةَ كِنْدَةٍ
وثَبَّتَهُمْ بِالفَضْلِ مِنْهُ وشَيَّعا

شعر تغلب في الجاهلية. تح: أيمن ميدان، ص: 76.— المصدر
04 —

كَنُود

اسم
4.1معنى من معاني «كَنُود»

* الكَنودُ: الكَفورُ لِلنِّعَمِ والأَفْضالِ الجَحودُ لَها، يَسْتَوي فيه المُذَكَّرُ والمُؤَنَّثُ. فارِسِيَّةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الكَنودُ: الكَفورُ لِلنِّعْمَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: مهدي المخزومي وآخر، ج: 5، ص: 331.— المصدر
”وقالَ الخَوارِزْمِيُّ، أَبو بَكْرٍ (ت: 383هـ=993م) يَذْكُرُ ما جاءَ في القُرْآنِ فضُرِبَتْ بِهِ الأَمْثالُ:

"وفي المَكّارِ الكَنودِ‌ ﵟوَلَا يَحِيقُ ‌ٱلۡمَكۡرُ ‌ٱلسَّيِّئُ إِلَّا بِأَهۡلِهِﵞ [فاطر: 43]".

الخوارزمي، أبو بكر، الأمثال المولدة، المجمع الثقافي، أبو ظبي، ص: 344.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ القَيِّمِ (ت: 751هـ=1324م):

"‌وقَدْ ذَمَّ اللهُ سُبْحانَهُ ‌الكَنودَ وهُوَ الَّذي لا يَشْكُرُ نِعَمَهُ".

ابن القيم، عدة الصابرين، دار ابن كثير، دمشق، ص: 39.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشِّدياق (ت: 1304هـ=1887م):

"قالَ قَدْ فَرَّطْتَ وقَسَطْتَ. قالَ إِنَّكَ ‌كَنودٌ".

الشدياق، الساق على الساق، مكتبة العرب، ج: 1، ص: 56.— المصدر
4.2معنى من معاني «كَنُود»

وــــــــــ مِنَ الأَرَضينَ: القَفْرُ الَّتي لا تُنْبِتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنّى (ت: 209هـ=825م):

"الأَرْضُ الكَنودُ الَّتي لا تُنْبِتُ شَيْئًا ".

أبو عبيدة، مجاز القرآن. تح: فؤاد سزكين، ج: 2، ص: 311.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ اليوسي، نورُ الدّينِ الحَسَنُ بْنُ مَسْعودٍ (ت: 1102هـ=1691م):

"‌ واعْلَمْ أَنَّ بِقاعَ الأَرْضِ كَأَفْرادِ الإِنْسانِ، هِيَ كُلُّها مُشْتَرِكَةٌ في كَوْنِها أَرْضًا وتُرْبَةً، ثُمَّ تَتَفاوَتُ في المَزايا الاخْتِصاصِيَّةِ، إِمّا مِنْ ذاتِهَا بِأَنْ يَجْعَلَها اللهُ مَنْبَتًا لِلْعُشْبِ، وهِيَ أَفْضَلُ مِنَ السَّبْخَةِ، أَوْ مَزْرَعَةً وهِيَ أَفْضَلُ مِنَ ‌الكَنودِ ".

اليوسي، المحاضرات . تح: محمد حجي وآخر، دار ابن كثير، دمشق، ص: 116.— المصدر
4.3معنى من معاني «كَنُود»

وــــــــــ:البَخيلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَنْبارِيِّ (اللُّغَوِيُّ) (ت: 328هـ=940م):

"الكَنودُ: البَخيلُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو بكر الأنباري، المذكر والمؤنث. تح: عضيمة، ج: 2، ص: 63.— المصدر
4.4معنى من معاني «كَنُود»

وــــــــــ: العاصي.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ يَحْيى بْنُ سَلّامٍ (ت: 200هـ=815م):

"الكَنودُ: العاصي ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الماوردي، النكت والعيون. تح: بن عبد الرحيم، ج: 6، ص: 325.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَطّارِ (ت: 724هـ=1324م):

"‌الحَمْدُ لِلهِ الواحدِ المَعْبودِ، الواجِدِ الماجِدِ الوَدودِ، الغَنِيِّ عَنْ طاعَةِ الطّائِعِ ولا يَعْجَلُ عَلى ‌الكَنودِ، القَوِيِّ لا بِكَثْرَةِ الأَعْوانِ والجُنودِ".

ابن العطار، العدة في شرح العمدة. تح: يعقوبي ، ج: 1، ص: 39.— المصدر
