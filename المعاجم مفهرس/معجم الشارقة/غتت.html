01 —

غَتَّ

فعل
1.1معنى من معاني «غَتَّ»

* غَتَّ (كَنَصَرَ) الـماءُ يَغُتُّ غَتًّا: تَدَفَّقَ مِنْ دونِ انْقِطاعٍ مُحْدِثًا صَوْتًا وخَريرًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (ت: 11هـ=632م) يَصِفُ حَوْضَهُ يَوْمَ الْقِيامَةِ:

«أَشَدُّ بَياضًا مِنَ اللَّبَنِ وَأحْلى مِنَ الْعَسَلِ، يَغُتُّ فيهِ ميزابانِ مِنَ الـجَنَّةِ، أَوْ مِدادُهُ مِنَ الـجَنَّةِ، أحَدُهُما مِنْ وَرِقٍ والآخَرُ مِنْ ذَهَبٍ».

هنّاد بن السّريّ، كتاب الزهد. تح: عبد الرحمن الفريوائي، ص: 111.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ إسْحاقَ السَّعْديّ (ت: 350هـ=961م):

"ومَعْنى: يَغُتُّ: يَجْري جَرْيًا، لَهُ صَوْتٌ وخَريرٌ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 12.— المصدر
1.2معنى من معاني «غَتَّ»

وــــــــــ فُلانٌ عَلى فُلانٍ: غَمَّهُ وأَكْرَبَهُ وَأَفْسَدَ عَلَيْهِ رَأْيَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مالِكُ بْنُ أنَس (ت: 179هـ=795م) يُحَذِّرُ أهْلَ الْعِراقِ مِنْ أَنْ يُفْسِدَ عَلَيْهِمْ أَحَدٌ رَأْيَهُمْ:

"يا أهْلَ الْعِراقِ لا يَغُتَّ عَلَيْكُمْ بَعْدَ مُحَمَّدٍ بْنِ إِسْحاقَ أحَدٌ".

ابن عدي، الكامل. تح: عبد الموجود وآخرين، ج: 7، ص: 256.— المصدر
1.3معنى من معاني «غَتَّ»

وــــــــــ فُلانًا: عَصَرَ حَلْقَهُ نَفَسًا أَوْ نَفَسَيْنِ وخَنَقَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال النَّبِيُّ ﷺ (ت: 11هـ= 632م) يَصِفُ فِعْلَ جِبْريلَ -عَلَيْهِ السَّلامُ- حينَ أتاهُ بالوَحْيِ:

«فَأَخَذَنِي، ‌فَغَتَّني حَتَّى بَلَغَ مِنِّي الجَهْدَ».

مصنف عبد الرزاق. تح: مجموعة محققين، ج: 6، ص: 10.— المصدر
”وقالَ عُبَيْدُ بْنُ عُمَيْر (ت: 74هـ=694م) يَرْوي حادِثَةَ نُزولِ أَوَّلِ الْوَحْيِ عَلى النَّبِيِّ:

"فَغَتَّهُ بِنَمَطٍ، فَقالَ: ﴿ٱقۡرَأۡ بِٱسۡمِ رَبِّكَ ٱلَّذِي خَلَقَ﴾".

[بنَمَطٍ؛ أي: بِنَمَطٍ مِنْ ديباجٍ]
التفسير من سنن سعيد بن منصور. تح: سعد آل حميد، ج: 8، ص: 388.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْد الأنْصاريّ (ت: 215هـ=830م):

"غَتَتُّ الرَّجُلَ أَغُتُّهُ غَتًّا: إِذا عَصَرْتُ بِحَلْقِهِ نَفَسًا أَوِ اثْنَيْنِ أَوْ أَكْثَرَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، مستدرك تهذيب اللغة. تح: هارون، ص: 55.— المصدر
”وقالَ ابْنُ الرّوميّ (ت: 283هـ=896م) يَمْدَحُ فَضْلَ آلِ سُلَيْمانَ بْنِ وَهْبٍ:

عَيْبُكُمْ أَنَّكُمْ أُناسٌ كِرامٌ
كُلُّ شَرٍّ بِفَضْلِكُمْ مَغْتوتُ 

ديوان ابن الرّوميّ. ج: 1، ص: 255.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ قَيِّم الجَوزيّة (ت: 751هـ=1350م) يَسْتَحْضِرُ اللَّحْظَةَ الْأولى لِنُزولِ الْوَحْي لَيُبَيِّنَ دَوْرَ الشَّدائِدِ في بُلوغِ الْفَرَجِ:

"كَما كانَ الغَتُّ والغَطُّ مُقَدِّمَةً بَيْنَ يَدَيْ الوَحْيِ وإِعْدادًا لِوُرودِهِ".

ابن القيم، مدارج السالكين. تح: السعودي وآخرين، ج: 1، ص: 3495.— المصدر
1.4معنى من معاني «غَتَّ»

وــــــــــ الْقَوْلَ أَوِ الشُّرْبَ: أَتْبَعَ بَعْضَهُ بَعْضًا دُونَ أَنْ يَأْخُذَ نَفَسًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو ذؤيب الهُذليّ (ت: 27هـ=648م) يَصِفُ شُرْبَهُنَّ الـماءَ عَلى عَجَلٍ:

 شَدَّ الضُّحى‌ فَغَتَتْنَ ‌غَيْرَ ‌بَواضِعٍ
غَتَّ الغَطاطِ مَعًا عَلى إِعْجالِ

الزبيدي، تاج العروس. تح: مصطفى حجازي، ج: 5، ص: 16.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والغَتُّ: أَنْ تُتْبِعَ القَوْلَ القَوْلَ، والشُّرْبَ الشُّرْبَ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 344.— المصدر
1.5معنى من معاني «غَتَّ»

وــــــــــ اللهُ أَوْ فُلانٌ فُلانًا وغَيْرَهُ: غَرَّقَهُ وَغَمَرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الْغَتُّ كَالغَطِّ في الـماءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 344.— المصدر
”وقالَ مُصْعَب الماجِنُ (ت: 250هـ=865م) يَعِظُ:

عَلَيْكَ إِذا لَقيتَ بِحُسْنِ بِشْرٍ
وكُنْ مِنْ أَكْثَرِ الثَّقَلَيْنِ سَمْتًا

ولا تُخْلِ الأَصابِعَ مِنْ عُقودٍ
وغُتَّ النّاسَ بِالآثارِ غَتًّا

الشابشتي، الديارات. تح: كوركيس عواد، ج: 1، ص: 194.— المصدر
1.6معنى من معاني «غَتَّ»

وــــــــــ فُلانٌ الدّابَّةَ وَغَيْرَها: رَكَضَها، وجَهَدَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْد الأنْصاريّ (ت: 215هـ=830م):

"وبَلْتُ الصَّيْدَ: أَلْحَحْتُ عَلَيْهِ في الطَّرَدِ وغَتَتُّهُ".

ابن سيده، المخصص. تح: خليل جفال، ج: 2، ص: 297.— المصدر
1.7معنى من معاني «غَتَّ»

وــــــــــ فُلانٌ وغَيْرُهُ فُلانًا: غَلَبَهُ وقَهَرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ يَموتُ بْنُ المُزَرِّع (ت: 304هـ=916م) يُخاطِبُ ابْنَهُ الْمُهَلْهلَ وَيُخْبِرُهُ بِما فَعَلَ بِهِ الزَّمَنُ:

مُهَلْهلُ قَدْ حَلَبْتُ شُطورَ دَهْرٍ
وَكافَحَني بِها الزَّمَنُ العَفوتُ

وَجارَيْتُ الرِّجالَ بِكُلِّ رَبْعٍ
فَأَذَعْنَ لي الـحُثالَةُ والرُّتوتُ 

فَأَرْجَعَ ما أُجِنُّ عَلَيْهِ قَلْبي
كَريمٌ غَتَّهُ زَمَنٌ ‌غَتوتُ

الخطيب، تاريخ بغداد. تح: مصطفى عطا، ج: 14، ص: 361.— المصدر
1.8معنى من معاني «غَتَّ»

وــــــــــ ضِحْكَتَهُ: أَخْفاهَا وَكَتَمَها بِسَتْرِ فَمِهِ بِيدِهِ أَوْ بِثَوْبِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القُوطيّة (ت: 367هـ=977م):

"وغَتَّهُ غَتًّا(...)، والضَّحِكَ: أَخْفَيْتَهُ بِسَتْرِ الفَمِ".

ابن القوطية، الأفعال. تح: علي فوده، ج: 1، ص: 196.— المصدر
1.9معنى من معاني «غَتَّ»

وـــــــــ فُلانًا بِالشَّيْءِ: كَدَّهُ بِهِ وأَلْقاهُ عَلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ الله بْنُ أبَيّ بْنِ سَلول الـمُنافِقُ (11هـ=632م) يُسيءُ أدَبَهُ مَع النَّبيّ ﷺ:

"يا هذا، إنَّهُ لا أَحْسَنُ مِنْ حَديثِكَ هَذا؛ إنْ كانَ حَقًّا فاجْلِسْ في بَيْتِكَ فَمَنْ جاءَكَ لَهُ فَحَدِّثْهُ إيّاهُ، ومَنْ لَمْ يَأْتِكَ فَلا تَغُتَّهُ بِهِ، ولا تَأْتِهِ في مَجْلِسِهِ بِما يَكْرَهُ مِنْهُ".

ابن هشام، السيرة النبوية. تح: عمر تدمري، ج: 2، ص: 228.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"غَتَّهُ بجُشَمِهِ: أَيْ أَلْقى صَدْرَه عَلَيْه ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 6، ص: 40.— المصدر
”وقالَ أبو بكر الصُّولي (ت: 335هـ=946م) يَرْوي ثَوْرَةَ العامَّةِ عَلى حُكْمِ الرّاضِي بِاللهِ وَصِدامَهُمْ مَعَ جُنْدِهِ:

"وشَغَبَ الْعامَّةُ لِغَلاءِ السِّعْرِ في مَسْجِدِ الرُّصافَةِ، ودَخَلَ الـجُنْدُ في طَلَبِهِمْ إِلى الصَّحْنِ؛ فَصَعِدوا إِلى السُّطوحِ وغَتّوا الفُرْسانَ بِالْحِجارَةِ حَتّى هَرَبوا".

الصولي، أخبار الراضي بالله والمتقي لله. تح: ج هيورث دن، ج: 1، ص: 71.— المصدر

وــــــــــ فُلانًا بِالكَلامِ: بَكَّتَهُ. (مو).

ع

02 —

غُتَّ

فعل
2.1معنى من معاني «غُتَّ»

*غُتَّ فُلانٌ: غُمَّ وحَزِنَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاج (90هـ=709م) يَصِفُ حالَ نَبِيِّ اللهِ يونُسَ وَهُوَ في بَطْنِ الْحوتِ:

وَجَوْشَنُ الْبَحْرِ لَهُ مَبيتُ

يَدْفَعُ عَنْهُ جَوْفَهُ الْمَسْحوتُ

كِلاهُما مُقتَمِسٌ ‌مَغْتوتُ

وَاللَّيْلُ فَوْقَ الْماءِ مُسْتَميتُ

عبد الله بن رؤبة العجاج، ديوان العجاج. تح: عزّة حسن، ج: 1، ص: 408.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمِرُ بْنُ حَمْدَوَيْه (ت: 255هـ=869م):

"غُتَّ فَهُوَ ‌مَغْتوتٌ، وغُمَّ: فَهُوَ مَغْمومٌ  ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، مستدرك تهذيب اللغة. تح: هارون، ص: 56.— المصدر
2.2معنى من معاني «غُتَّ»

وــــــــــ: جُنَّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السَّرَقُسطيّ (ت: 400هـ=1010م):

"وغُتَّ غَتًّا: جُنَّ".

السرقسطي، الأفعال. تح: حسين شرف، ج: 2، ص: 25.— المصدر
