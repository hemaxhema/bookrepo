01 —

قَرْقَمَ

فعل
1.1معنى من معاني «قَرْقَمَ»

* قَرْقَمَ الشَّيْءُ فُلانًا وغَيْرَهُ: أضْعَفَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”

قالَ امْرؤ القَيْس الكِنْديّ (ت: 80ق.ه=544م) يَصِفُ حالَ فَرْخٍ:

‌تُطْعِمُ ‌فَرْخًا ‌لَها ‌صَغيرًا
قَرْقَمَهُ الجوعُ والإِحْثالُ

الأصمعي، كتاب الإبل، تح: حاتم الضامن ج: 1، ص: 69.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأصْمَعيّ (ت: 216ه=831م):

"قُلْتُ لغُلامٍ من العَرَبِ: ما لي أَراكَ ضَعيفًا نَحيفًا وصَغيرَ الجِسْمِ قَليلًا مَهْزولًا؟ قالَ: قَرْقَمَني العِزُّ ".

الجاحظ، البرصان والعرجان. تح: هارون، ص: 41.— المصدر
1.2معنى من معاني «قَرْقَمَ»

وــــــــــ فُلانٌ فُلانًا وغَيْرَهُ: أَساءَ غِذاءَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو زَيْد الأنْصاريّ (ت: 215ه=830م):

"ويُقالُ: ‌قَرْقَمْتُ غِذاءَهُ قَرْقَمَةً: إذا أَسَأْتَ غِذاءَهُ؛ فهُوَ مُقَرْقَمٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو علي القالي، البارع في اللغة. تح: هشام الطعان، ص: 552.— المصدر
”

وقالَ الأصْمَعيّ (ت: 216ه=831م):

"ويُقالُ: أَصابَتِ النّاسَ سَنَةٌ ‌فَقَرْقَمَتِ السِّخالَ".

كتاب الإبل، تح: حاتم الضامن، دار البشائر. ج: 1، ص: 69.— المصدر
02 —

تَقَرْقَمَ

فعل
2.1معنى من معاني «تَقَرْقَمَ»

* ‌تَقَرْقَمَ الوَحْشِيُّ في وَجارِهِ: تَقَبَّضَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابن دُرَيْد (ت: 321ه=933م):

"ويُقالُ: تَجَرْجَمَ الوَحْشِيُّ فِي وَجارِهِ: إذا تَقَبَّضَ فيه، ويُقالُ: تَقَرْقَمَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ص: 1162.— المصدر
03 —

قُرْقِمَ

فعل
3.1معنى من معاني «قُرْقِمَ»

* ‌قُرْقِمَ الغُلامُ: أُسِيءَ غِذاؤُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"قُرْقِمَ الغُلامُ فهو مُقَرْقَمٌ: إذا أُسِيءَ غِذاؤُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 264.— المصدر
3.2معنى من معاني «قُرْقِمَ»

وــــــــــ فُلانٌ وغَيْرُهُ: ضَعُفَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الـمُقَمَّحُ من الفِصالِ: الضَّعيفُ، يُقالُ: إنَّهُ ‌لَمُقَرْقَمٌ مُقَمَّحٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 73.— المصدر
04 —

قِرْقِم

اسم
4.1معنى من معاني «قِرْقِم»

* القِرْقِمُ: حَشَفَةُ الرَّجُلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ مَعْدانُ بن عُبَيْد الطّائيّ (ت: 136ه=754م) يَتَهَتَّكُ في وَصْفِ امْرأةٍ:

مَشْغوفَةٌ بَرَهْزِ حَكِّ القِرْقِمِ

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 419.— المصدر
05 —

قُرْقُمان

اسم
5.1معنى من معاني «قُرْقُمان»

* القُرْقُمانُ: ما يُسَوِّسُ وسْطَ الأخْشابِ العَتيقَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابن البيطار (ت: 646ه=1248م):

"قُرْقُمانٌ: هُوَ الخشَبُ الحِجازيُّ الذي في جَوْفِ المُقْلِ الحِجازيّ والصَّعيديّ، بارِدٌ يابِسٌ يَدْخُلُ في السُّفوفاتِ فَيُقَوّي لَحْمَ اللِّثَةِ والأَسْنانَ ويُنْقِيها ويُبَيِّضُها ".

ابن البيطار، الجامع لمفردات الأدوية والأغذية. ج: 4، ص: 19.— المصدر
06 —

قَرْقَمَة

اسم
6.1معنى من معاني «قَرْقَمَة»

* القَرْقَمَةُ: ضُؤولَةُ الجِسْمِ ودِقَّةُ العِظامِ لتَقارُبِ نَسَبِ الأَبَوَيْن.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابن قُتَيْبَة الدِّينَوَرِيّ (ت: 276ه=889م):

"والقَرْقَمَةُ: صِغَرُ الجِسْمِ للتَّزَوُّجِ فِي القَرابات".

ابن قتيبة: غريب الحديث، تح: عبد الله الجبوري ج: 3، ص: 738.— المصدر
6.2معنى من معاني «قَرْقَمَة»

وــــــــــ: ثِيابُ كَتّانٍ بيضٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابن سِيدَه (ت: 458ه=1066م):

"والقَرْقَمَةُ: ‌ثِيابُ ‌كَتَّانٍ ‌بيضٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 632.— المصدر
07 —

مُقَرْقَم

اسم
7.1معنى من معاني «مُقَرْقَم»

* الـمُقَرْقَمُ: الصَّغيرُ من البَهْمِ السَّيِّئُ الغِذاءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"والـمُقَرْقَم: الصَّغيرُ من البَهْمِ السَّيِّئُ الغِذاءِ".

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 109.— المصدر
7.2معنى من معاني «مُقَرْقَم»

وــــــــــ: البَطيءُ الشَّبابِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ القاسِمُ بن سَلّام (ت: 224ه=839م):

"الـمُقَرْقَمُ: البَطيءُ الشَّبابِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 1، ص: 55.— المصدر
7.3معنى من معاني «مُقَرْقَم»

وــــــــــ: الغِذاءُ السَّيِّئُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو مِسْحَل الأعْرابيّ (ت: 230ه=845م):

"ويُقالُ في الغِذاءِ السَّيِّئِ: غِذاءٌ مُقَرْقَمٌ".

أبو مسحل الأعرابي، كتاب النوادر. تح: عزة حسن، ص: 73.— المصدر
