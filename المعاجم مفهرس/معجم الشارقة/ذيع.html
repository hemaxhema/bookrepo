  1. المع,إِظْهارُ الشَّيْءِ وانْتِشارُهُ
01 —

ذاعَ

فعل
1.1معنى من معاني «ذاعَ»

*ذاعَالخَبَرُ وغَيْرُهُ يَذِيعُ ذَيْعًا، وَذُيوعًا، وَذَيَعانًا، وَذَيْعوعَةً: فَشا وانْتَشَرَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الشَّنْفَرى (ت: 70ق ه=554م) يُعاتِبُ قَوْمَهُ:

وَلي دونَكُمْ أَهْلونَ سِيدٌ عَمَلَّسٌ
وَأَرْقَطُ زُهْلولٌ وَعَرْفاءُ جَيْأَلُ

هُمُ الأَهْلُ لا مُسْتَوْدَعُ السِّرِّ ذائِعٌ
لَدَيْهِمْ وَلا الجاني بِما جَرَّ يُخْذَلُ

ديوان الشنفرى. تح: إميل يعقوب، ص: 59.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ يَزيدُ بْنُ مُعاوِيَةَ (ت: 64ه=683م):

وَما سِرُّ لَيلَى ما حَييتُ بِذائِعِ
وَما عَهْدُ لَيلَى إِنْ تَناءَتْ بِضائِعِ

ديوان يزيد بن معاوية. تح: واضح الصّمد، ص: 50.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=760م) يُوصِي بِضَرورَة حِفْظِ السِّرِّ:

"فَإِذا تَكَلَّمَ بِالسِّرِّ اثْنانِ فَلا بُدَّ مِنْ ثالِثٍ مِنْ جِهَةِ الواحِدِ، أَوْ مِنْ جِهَةِ الآخَرِ، فَإِذا صارَ إِلى الثَّلاثَةِ فَقَدْ شاعَ وَذاعَ حَتّى لا يَسْتَطيعُ صاحِبُهُ أَنْ يَجْحَدَهُ ويُكابِرَ فيهِ".

آثار ابن المقفع، بيروت، ص: 37.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابن الأَبّار (ت: 658هـ=1260م):

إِنْ ضاعَ قَلْبي فَأَيْنَ أَطْلُبُهُ
أَوْ ذاعَ حُبِّي فَأَنْتَ مُوجِبُهُ

ديوان ابن الأبار. تح: الهراس، ص: 59.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ البَرْبِيرُ (ت: 1226ه=1811م) يَذْكُرُ ما فاخَرَتْ بِهِ الأَرْضُ السَّماءَ:

"وقَدْ ساغَ لي أَنْ أُشَنِّفَ الأَسْماع، بِما شاعَ مِنْ مَآثِري وَذاع".

أحمد البربير، المفاخرات والمناظرات. تح: محمد الطيان، ص: 95.— المصدر
1.2معنى من معاني «ذاعَ»

وـــــــــــ النّاسُ: كَثُرُوا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ سَهْلُ بْنُ سَعْدٍ السّاعِدِيُّ (ت: 91ه=712م):

"كانَ رَسُولُ اللهِ ﷺ يَقُومُ إذا خَطَبَ إلى خَشَبَةٍ كانَتْ فِي الـمَسْجِدِ فَلَمّا ذاعَ النّاسُ وكَثُرُوا قِيلَ لَهُ: يا رَسُولَ اللهِ لَوْ جَعَلْتَ مِنبَرًا تُشْرِفُ عَلى النّاسِ مِنهُ؟..".

أبو نعيم الأصبهاني، دلائل النبوة. تح: محمد رواس، ج: 1، ص: 403.— المصدر
1.3معنى من معاني «ذاعَ»

وـــــــــــ الشَّبابُ: ظَهَرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَشْجَع السُّلَمِيُّ (ت: 195ه=811م) يَفخَرُ:

فَإِن يَخلُ دِرعي مِن مَراحِي فَرُبَّما
بَسَطتُ يَمِينِي فِي الصِّبا وَشِمالِي

بِأَلفِ ظِباءٍ طائِعِينَ لِإِمْرَتي
وَعَهْدِ شَبابٍ ذائِعٍ وَجَمالِ

محسن الحسيني، أعيان الشيعة، ج: 5، ص: 269.— المصدر
02 —

ذايَع

فعل
2.1معنى من معاني «ذايَع»

*ذايَعَالشَّيءَ: أَظْهَرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مِهْيارٌ الدَّيْلَمِيُّ (ت: 428ه=1037م) يَمدَحُ:

ماتَ الـمُلُوكُ عَلَى عِصْيانِهِم كَمَدًا
بِهِ فَكاتِمُ داءٍ أَوْ مُذايِعُهُ

ديوان مهيار الديلمي. دار الكتب المصرية، ج: 1، ص: 166.— المصدر
03 —

أَذاعَ

فعل
3.1معنى من معاني «أَذاعَ»

*أَذاعَالشَّيءُ بِالشَّيْءِ: أَفْناهُ وَذَهَبَ بِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ بِشْرُ بْنُ عُلِيقٍ الطّائِيُّ (ت: 86 ق ه=538م) يَصِفُ طَلَلًا طَمَسَتْهُ الرِّياحُ:

خَليلَيَّ عُوجا فانْظُراني لَعَلَّني
أُسائِلُ رَسْمًا قَدْ عَفا وَتَهَدَّما

أَذاعَتْ بِهِ الأَرْواحُ حَتَّى كَأَنَّما
حَسِبْتُ بَقاياهُ كِتابًا مُنَمْنَما

شعر طيئ وأخبارها، جمع: وفاء السنديوني، ص: 306.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو قَيْسِ بْنُ الأَسْلَتِ (ت: 1ه=622م) يُذَكِّرُ قُرَيْشًا بِما كانَ مِنْ حَرْبِ داحس:

وَماءٍ هُريقَ في الضَّلالِ كَأَنَّما
أَذاعَتْ بِهِ رِيحُ الصَّبا والجَنائِبِ

ديوان أبي قيس بن الأسلت. تح: باجوده، ص: 67.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَدِيٍّ العَبْلِيُّ (ت: 145هـ=762م) يَذْكُرُ ما حَلَّ بِقَوْمِهِ:

أُولَئِكَ قَوْمي أَذاعَتْ بِهِمْ
نَوائِبُ مِنْ زَمَنٍ مُتْعِسِ

البلاذري، أنساب الأشراف. تح: سهيل زكّار وآخر، ج: 9، ص: 334.— المصدر
3.2معنى من معاني «أَذاعَ»

وـــــــــــ بِالأَمرِ وَأَذاعَهُ: أَفْشاهُ وَنَشَرَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَنْتَرَةُ (ت: 22 ق ه=601م) يَصِفُ حالَهُ وقَدْ مَرَّتْ عَلَيْهِ عَبْلَةُ:

مَرَّتْ أَوانَ العِيدِ بَيْنَ نَواهِدٍ
مِثْلِ الشُّمُوسِ لِحاظُهُنَّ ظُباءُ

فاغْتالَني سَقَمي الَّذي في باطِني
أَخْفَيتُهُ فَأَذاعَهُ الإخْفاءُ

ديوان عنترة. تح: حمدو طمّاس، ص: 67.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11هـ=632م) قال اللهُ ﷺ:

﴿وَإِذَا جَآءَهُمۡ أَمۡرٞ مِّنَ ٱلۡأَمۡنِ أَوِ ‌ٱلۡخَوۡفِ ‌أَذَاعُواْ ‌بِهِۦۖ وَلَوۡ رَدُّوهُ إِلَى ٱلرَّسُولِ وَإِلَىٰٓ أُوْلِي ٱلۡأَمۡرِ مِنۡهُمۡ لَعَلِمَهُ ٱلَّذِينَ يَسۡتَنۢبِطُونَهُۥ مِنۡهُمۡ﴾ (النساء: 83).

ورُوِيَ في: الأثَرِ (11هـ=632م):

«مَنْ أَذاعَ فاحِشَةً كانَ كَمُبْتَدِئِها، وَمَنْ عَيَّرَ مُؤْمِنًا بِشَيْءٍ لَمْ يَمُتْ حَتَّى يَرْتَكِبَهُ».
الشيخ المفيد، الاختصاص. تح: علي أكبر الغفاري، ص: 224.— المصدر
”وَقالَ عُمَرُ بْنُ الخَطّابِ (ت: 23هـ=644م):

"ثَلاثٌ مِنَ الفَواقِرِ: جارُ مُقامَةٍ، إِنْ رَأى حَسَنَةً سَتَرَها، وَإِنْ رَأى سَيِّئَةً أَذاعَها، وامْرَأَةٌ إِنْ دَخَلَتْ لَسَنَتْكَ، وَإِنْ غِبْتَ عَنْها لَمْ تَأْمَنْها، وسُلْطانٌ إِنْ أَحْسَنْتَ لَمْ يَحْمَدْكَ، وإِنْ أَسَأْتَ قَتَلَكَ".

ابن قتيبة، عيون الأخبار، ج: 4، ص: 4.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142ه=759م) يَذْكُر وَصِيَّةَ أُمِّ الأَسَدِ لابْنِها بِحِفْظِ الأَسْرارِ:

"ولَوْلا ما قالَتِ العُلَماءُ في إذاعَةِ الأَسْرارِ وما فيها مِنَ الإِثْمِ والشَّنارِ، لَذَكَرْتُ لَكَ وأَخْبَرْتُكَ بِما عَلِمْتُ".

آثار ابن المقفع، بيروت، ص: 119.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابن الأَبّار (ت: 658هـ=1260م) يَتَغَزَّلُ:

وَكَمْ لَيالٍ قَطَعْناها بِكاظِمَةٍ
نَجْوى وَشَكْوى بِما يَلْقاهُ مُضْناكِ

كَتَمْتُ مَسْراكِ فيها خَوْفَ عاذِلَةٍ
وَعاذِلٍ فَأَذاعَ المِسْكُ مَسْراكِ

ديوان ابن الأبار. تح: الهراس، ص: 231.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدٌ البَسّام (ت: 1233ه=1817يَصِفَ أَبا نُقْطَةَ محمَّدَ بنَ عامِرٍ وجَيْشَهُ:

"وَذَلِكَ أَنَّ أَبا نُقْطَةَ شَيْخَ عَسِيرٍ نَزَلَ في غَلاقَ سَنَةَ 1804 بِجَيْشٍ عَرَمْرَمٍ مُؤَلَّفٍ مِنَ العُرْبانِ القاطِنَةِ في الجِبالِ، وَأَذاعَ الخَوْفَ والارْتِياعَ في البِلادِ، ونَهَبَ كُلًّا مِنْ مَدينَتَيْ لَحْيَة والحُدَيْدَةَ".

محمد البسام، الدرر المفاخر. تح: سعود العجمي، ج: 1، ص: 460.— المصدر
3.3معنى من معاني «أَذاعَ»

وـــــــــــ القَوْمُ وَغَيْرُهُم بِالماءِ وَأَذاعُوهُ: شَرِبوهُ كُلَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصارِيُّ (ت: 215ه=830م ):

"يُقالُ: أَذاعَ النّاسُ بِما في الحَوْضِ إِذاعَةً إِذا شَرِبوا ما فيهِ، وأَذاعَتْ بِهِ الابِلُ إِذاعَةً إذا شَرِبَتْهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 3، ص: 149.— المصدر
04 —

إذاعَة

اسم
4.1معنى من معاني «إذاعَة»

*الإذاعَةُ: بَثُّ البَرامِجِ الإِذاعِيَّةِ المُتَنَوِّعَةِ بواسِطَةِ الإرْسالِ اللّاسِلْكِيِّ، وهِيَ نَوْعانِ؛ مَسْموعَةٌ بِالمِذْياعِ، وَمَرْئِيَّةٌ بِالتِلِفِزْيونِ أَوْ نَحْوِهِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد الخَضِر حُسَيْن (ت: 1377ه=1958م) يَذْكُرُ ما يَنْبَغي أَنْ نُحاكِيَ فيهِ الغَرْبِيينَ:

"وَإنَّما يَحْسُنُ أَنْ نُحاكيَهُمْ فيما سَبَقونا إِليْهِ مِنَ الأَشْياءِ المادِّيَّةِ؛ كالهاتِفِ والبَرْقِ، والأَسْلِحَةِ، والطّائِراتِ، والغَوّاصاتِ، والإذاعَةِ".

محمد الخضر حسين، الأعمال الكاملة. تح: علي الحسيني، ج: 2، ص: 69.— المصدر
4.2معنى من معاني «إذاعَة»

o مَحَطَّةُ الإذاعَةِ، أَوْ دارُ الإِذاعَةِ: مُؤَسَّسَةٌ مُهَيْكَلَةٌ تَضُمُّ المُذيعينَ والمُخْرِجينَ والتِّقْنِيّينَ الَّذينَ يَعْمَلونَ عَلى البَثِّ الإذاعيِّ لِلمُسْتَمِعينَ أَوِ المُشاهِدينَ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد الخَضِر حُسَيْن (ت: 1357ه=1938م):

"سَأُلْقي -إِنْ شاءَ اللهُ- مُحاضَرَةً مَوْضوعُها: "الشَّجاعَةُ وأَثَرُها في عَظَمَةِ الأُمَّةِ"، وَقَدْ عَيَّنَتْ لَها مَحَطَّةُ الإِذاعَةِ السّاعَةَ التّاسِعَةَ وَنِصْفًا مِنْ مَساءِ يَوْمِ الإِثْنَيْنِ 28 مِنْ هَذا الشَّهْرِ".

محمد الخضر حسين، الأعمال الكاملة. تح: علي الحسيني، ج: 20، ص: 107.— المصدر
05 —

إذاعِيّ

اسم
5.1معنى من معاني «إذاعِيّ»

*الإِذاعِيُّ: وَصْفٌ لِكُلِّ ما يَتَّصِلُ بالإِذاعَةِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مالِكُ بْنُ نبي (ت: 1393ه=1973م) يَتَحَدَّثُ عَنِ الموسيقى العَرَبِيَّةِ الحَديثَةِ:

"وَيَكْفينا في إيضاحِ هَذهِ النُّقْطَةِ أَنْ نَذْكُرَ أَنَّ المُوسيقى العَرَبيَّةَ الحَديثةَ لا تَسْتَمِدُّ وَحْيَها مِنَ الأَساتِذَةِ الكِلاسيكيّينَ في الغَرْبِ، بَلْ إِنَّها تَبْحَثُ عَنْهُ في أَغانيهِ العاطِفيَّةِ وَلا حاجَةَ بِنا إِلى تَحْليلِ بَرْنامَجٍ إِذاعيٍّ".

مالك بن نبي، فكرة الإفريفية الآسيوية، ص: 90.— المصدر
06 —

ذَيْع

اسم
6.1معنى من معاني «ذَيْع»

*الذَّيْعُ:ما فَشا مِنَ الأَمْرِ وانْتَشَرَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جميل بُثَينة (ت: 82هـ=701م) يَذْكُرُ وَصايا مَحبوبَتِهِ:

وَطَرْفَكَ إِمّا جِئْتَنا فاحْفَظَنَّهُ
فَذَيْعُ الهَوى بادٍ لِمَنْ يَتَبَصَّرُ

ديوان جميل بثينة، دار صادر، ص: 62.— المصدر
07 —

مِذْياع

اسم
7.1معنى من معاني «مِذْياع»

*المِذْياعُ مِنَ النّاسِ: الَّذي لا يَكْتُمُ خَبَرًا عَلِمَهُ عَنْ نَفسِهِ أَو غَيْرِهِ خَيْرًا كانَ أوْ شَرًّا. (ج) مَذاييعُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (ت: 40ه=661م) يَصِفُ الأَوْلياءَ:

"لَيْسوا بِالـمَذاييعِ البُذُرِ"

ابن الأثير، النهاية. تح: طاهر الزاوي وآخر، ج: 2، ص: 174.— المصدر
”وقالَ عُمَرُ بْنُ أَبي رَبيعَةَ (ت: 93ه=712م):

يا ابْنَ سُرَيْجٍ لا تُذِعْ سِرَّنا
قَدْ كُنْتَ عِنْدي غَيْرَ مِذْياعِ

ديوان عمر بن أبي ربيعة. تح: فايز محمد، ص: 226.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليلُ بنُ أَحمَدَ (ت: 170هـ=787م):

"والبَذيرُ: مَنْ لا يَسْتَطيعُ أَنْ يُمْسِكَ سِرَّ نَفْسِهِ، وَرَجُلٌ بَذيرٌ وَبَذورٌ: مِذْياعٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 182.— المصدر
7.2معنى من معاني «مِذْياع»

وـــــــــــ الجِهازُ الآليُّ الَّذي يَنْشُرُ الأَخْبارَ والبَرامِجَ الإذاعِيَّةَ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَلِيُّ مَحْمودُ طَهَ (ت: 1368هـ=1949م):

"لَقَدْ أَعْلَنَ الـمِذْياعُ نَبَأَ إغْلاقِ الـمَوانِئِ الإيطالِيَّةِ وانْقِطاعِ الـمُواصَلاتِ بَيْنَ فَرَنْسا وَإيطالِيا".

علي محمود طه، أرواح شاردة، ص: 90.— المصدر
08 —

مُذيع

اسم
8.1معنى من معاني «مُذيع»

*المُذيعُ: مَنْ يَقومُ بِتَقْديمِ الـمَوادِّ الإِذاعيَّةِ والرَّبْطِ بَيْنَ الفِقْراتِ وَقِراءَةِ نَشَراتِ الأَخْبارِ والتَّعْليقاتِ السِّياسِيَّةِ وَوَصْفِ الإِذاعاتِ الخارجيَّةِ، وَهُوَ غَيْرُ مُقَدِّمِ البَرامِجِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد أَمِين (ت: 1373ه=1954م) يَسْرُدُ كَثْرَةَ مَشاغِلِهِ وتَعَدُّدَ وَظائِفِهِ:

"فَأَنا عُضْوٌ في الـمَجْمَعِ اللُّغَوِيِّ وفي مَجْلِسِ دارِ الكُتُبِ وَمَجْلِسِ كُلِّيَّةِ الآدابِ ودارِ العُلومِ، ورَئيسُ لَجْنَةِ التَّأْليفِ والجَماعَةِ الشَّعْبِيَةِ إلخ إلخ، وَمُذِيعٌ فِي الرّادْيُو".

أحمد أمين، حياتي، القاهرة، ص: 229.— المصدر
