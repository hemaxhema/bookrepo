المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السَّاميَّاتُ بِهذا الجَذْرِ مِنَ العَرَبِيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (ح ش ف):

في العِبريّة: وَرَدَ الاسْمُ: חֲשִׁיפָה (حَشِيفَا) ḥašīfā بِمَعْنى (سِيقانُ النَّباتاتِ الجافَّةِ، لُحاءُ الأشْجارِ) ويُوافِقُ الاسْمَ العَرَبيَّ (حَشَفَةٌ) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (حَشَفَةٌ) ما وَرَدَ في التّوسفتا: "מַחְצֶלֶת שֶׁל חֲשִׁיפָה וְשֶׁל גֶּמִי גְּדוֹלָה מְסַכְּכִין בָּהּ"، أيِ: "الحَصائِرُ الكَبيرةُ المَصْنوعةُ مِنْ حَشَفِ النَّباتاتِ وسِيقانِ البَرْدِيِّ يَجوزُ اتِّخاذُها عَرائِشَ".

(תוס': סד' מועד, מסכ' סוכה, פר' א', ה' י')

في الجِعزيّة: وَرَدَ الفِعْلُ: ሐሠፈ/ሐሰፈ (حَسَفَ/حَشَفَ) ḥasafa/ḥašafa بِمَعْنى (جَرَبَ- قَشَرَ- حَكَّ- أُصيبَ بِطَفَحٍ جِلْدِيٍّ) ويُوافِقُ الفِعْلَ العَرَبِيَّ (حَسَفَ) مَبْنًى وفي بَعْضِ مَعانيهِ، ومِنْهُ اشْتُقَّ الاسْمُ ሕሱፍ/ሕሡፍ (حْسُف/حْشُف) ḥəsuf/ḥəšuf بِمَعْنى (أجْرَبُ- مُغَطًّى بِالخُدوشِ) ومِنْ مَجيئِهِ بِمَعْنى (أجْرَبُ) ما وَرَدَ في سِفْرِ اللَّاويينَ: ወዕውረሰ አው ስቡረ አው ምቱረ ልሳን አው ሕሱፈ አው ዕቡቀ አው ዘጽርንእተ ቦቱ ላዕሌሁ ዘንተ ኢያመጽኡ ለእግዚአብሔር። أيِ: " الأعْمى والمَكْسورَ والمَجْروحَ والذي بِهِ بُثورٌ والأجْرَبَ والأكْلَفَ، هَذِهِ لا تُقَرِّبُوها للرَّبِّ".

(ኦሪት ዘሌዋውያን :22:22)

01 —

حَشَفَ

فعل
1.1معنى من معاني «حَشَفَ»

* حَشَفَ (بِفَتْحِ الشّينِ) خِلْفُ النّاقَةِ يَحْشِفُ (بِكَسْرِ الشّينِ) حَشْفًا: تَقَبَّضَ وارْتَفَعَ مِنها اللَّبَنُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وحَشَفَ خِلْفُ النّاقَةِ: إذا ارْتَفَعَ مِنْها اللَّبَنُ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 537.— المصدر
02 —

حَشِفَ

فعل
2.1معنى من معاني «حَشِفَ»

* حَشِفَ (بِكَسْرِ الشّين) الثَّوبُ يَحْشَفُ (بِفَتْحِ الشّينِ) حَشَفًا: بَلِيَ وخَلَقَ، فهو حَشيفٌ، والقِطْعَةُ مِنْه حَشيفَةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مالِكُ بْنُ خالِدٍ الخُناعِيُّ (ت: 86 ق هـ=538م) يَصِفُ صَيّادًا:

حَتّى أُشِبَّ لَها رامٍ بمُحْدَلةٍ
ذو مِرّةٍ بدِوارِ الصَّيْد هَمّاسُ

يُدْني الحَشيفَ عَلَيْها كَيْ يُوارِيَها
ونَفْسَهُ وهْوَ لِلأَطْمارِ لَبّاسُ

ديوان الهُذَلِيِّين. تح: محمّد الشنقيطي، ج: 3، ص: 3.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو ذُؤَيْبٍ الهُذَليّ (ت: 27 هـ=648م) يَصِفُ:

فَراغَ وزَوَّدوهُ ذَاتَ فَرْغٍ
لَها نَفَذٌ كَما قُدَّ الحَشيفُ

دِيوانُ الهُذَلِيِّين. تح: محمّد الشنقيطي، ج: 1، ص: 103.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"والحَشيفُ: الثَّوْبُ الخَلَقُ".

الخَليل، العَيْن. تح: المخزومي وآخر، ج: 3، ص: 96.— المصدر
”وقالَ أَبو العَلاءِ المَعَرِّيُّ (ت: 449ه=1057م) يَنْصَحُ:

"واضْحَكْ في وجْهِ ضَيْفِكَ وإنْ كانَ غَيْرَ حَبيبٍ، فَرُبَّ خُلُقٍ شيفَ، تَحْتَ خَلَقٍ حَشيفٍ، فابْتَسِمْ لِطارِقِكَ مَعَ النُّجومِ المُبْتَسِماتِ".

المَعري، الفصول والغايات. تح: محمود زناتي، ص: 130.— المصدر
2.2معنى من معاني «حَشِفَ»

وـــــــــ التَّمْرُ: يَبِسَ ولَمْ يَعْقِدْ نَواهُ. فَهوَ حَشَفٌ وحَشِفٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرؤُ القَيْسِ (ت: 80 ق هـ=544م) يُشَبِّهُ قُلوبَ الطُّيورِ الرَّطْبَةَ واليابِسَةَ في وكْرِ العُقابِ:

كأنَّ قُلوبَ الطَّيرِ رَطْبًا ويابسًا
لَدى وكْرِها العُنّابُ والحَشَفُ البالي

ديوانه. تح: أنور عليان وآخرين، ج: 1، ص: 359.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ زَيْدُ بْنُ مُهْلهلٍ الطّائيُّ (ت: 9 هـ=631م) يَفْتَخِرُ:

قَتَلْنا بِقَتْلانا مِنَ القَوْمِ عُصْبَةً
كِرامًا ولَمْ نَأْكُلْ بِهِمْ حَشَفَ النَّخْلِ

شعر طيئ وأخبارها. تح: وفاء السنديوني، ج: 1، ص: 566.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"والصِّيصاءُ: ما حَشِفَ مِنَ التَّمْرِ فَلَمْ يَعْقِدْ نَواهُ".

الخَليلُ، العَيْن. تح: المخزومي وآخر، ج: 7، ص: 175.— المصدر
”وقالَ مالِكُ بْنُ أَنَسٍ (ت: 179 هـ=795م):

"‌لا يَصْلُحُ ‌مُدُّ ‌زُبْدٍ ‌ومُدُّ ‌لَبَنٍ، ‌بِمُدَّيْ زُبْدٍ، وَهُوَ مِثْلُ الَّذِي وصَفْنا مِنَ التَّمْرِ الَّذِي يُباعُ صاعَيْنِ مِنْ كَبِيسٍ، وصاعًا مِنْ حَشَفٍ بِثَلاثَةِ آصُعٍ مِنْ عَجْوَةٍ".

مالِك بْن أَنَس، المُوَطَّأ. تح: محمد الأعظمي، ج: 6، ص: 273.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرّافعيُّ (676 هـ=1277م، مِنْ شَواهِدِ النَّووي):

"‌‌هَذا ‌الخِلافُ ‌هُوَ ‌فِيما ‌لا ‌يُغَيِّرُهُ ‌تَجْفِيفُهُ ، ولَوْ جُفِّفَ جاءَ مِنْهُ تَمْرٌ رَديءٌ حَشَفٌ".

النووي، المجموع شرح المهذب. تح: محمد المطيعي، ج: 5، ص: 440.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَدُ بْنُ مُصْطَفى الكُمُشْخانَوي (ت: 1226 هـ=1894م) يَصِفُ النَّبيَّ ﷺ:

"ويُصافِحُ الغَنِيَّ والفَقيرَ، ويَبْدَؤهُما بِالسَّلامِ ، ولا يَحْتَقِرُ ما دُعيَ إِلَيْهِ ولَوْ كانَ حَشَفَ التَّمْرِ".

الكمشخانوي، جامع الأصول في الأولياء- الطرق الصوفية. تح: أديب نصر الله، ج: 1، ص: 266.— المصدر
03 —

حَشِفَ

فعل
3.1معنى من معاني «حَشِفَ»

حَشِفَ (بِكَسْرِ الشِّينِ) خِلْفُ النّاقَةِ يَحْشَفُ (بِفَتْحِ الشِّينِ) حَشَفًا: تَقَبَّضَ وارْتَفَعَ مِنها اللَّبَنُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طَرَفةُ بْنُ العَبْدِ (ت: ه=م) يَصِفُ ناقةً:

تَريعُ إلى صَوْتِ المُهيبِ وتَتَّقي
بِذي خُصَلٍ رَوْعاتِ أَكْلَفَ مُلْبِدِ

فَطَوْرًا بِهِ خَلْفَ الزَّميلِ وتارةً
على حَشِفٍ كالشَّنِّ ذاوٍ مُجَدَّدِ

ديوانه، شرح الأعلم الشنتمري. تح: درية الخطيب وآخر، ص: 30.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321 هـ= 933م):

"الحَشَفُ مِنْ قَوْلِهم: حَشِفَ خِلْفُ النّاقَةِ: ارْتَفَعَ مِنْهُ اللَّبَنُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْن دُرَيْدٍ، جَمْهَرَة اللُّغَة. تح: بعلبكي، ص: 537.— المصدر
3.2معنى من معاني «حَشِفَ»

وــــــــــ التَّمْرُ: صارَ حَشَفًا، أَيْ: رَديئًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367 هـ=977م):

"وحَشِفَ ‌التَّمْرُ: صارَ حَشَفًا، أَيْ: رَديئًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْن القوطِيَّة، كِتاب الأَفْعال. تح: فودة، ج: 1، ص: 43.— المصدر
04 —

أَحْشَفَ

فعل
4.1معنى من معاني «أَحْشَفَ»

* أَحْشَفَ ضَرْعُ النّاقَةِ: نَفِدَ لَبَنُهُ، فَيَبِسَ وتَقَبَّضَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"وقَدْ أَحْشَفَ ضَرْعُ النّاقَةِ: إِذا يَبِسَ وتَقَبَّضَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الخَليل، العَيْن. تح: المخزومي وآخر، ج: 3، ص: 96.— المصدر
4.2معنى من معاني «أَحْشَفَ»

وــــــــــ النَّخْلَةُ: صارَ تَمْرُها حَشَفًا، وهُوَ أَرْدَأُ التَّمْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفارابيُّ (ت: 350 هـ=961م):

"وأَحْشَفَتِ النَّخْلَةُ، مِنَ الحَشَفِ".

الفارابيّ، مُعْجَم ديوان الأَدَب. تح: أحمد مختار عمر، ج: 2، ص: 314.— المصدر
”وقالَ السَّرْخَسِيُّ (ت: 483ه=1090م) يُبَيِّنُ صِحّةَ العَقْدِ بَيْنَ صاحِبِ النَّخْلِ وبَيْنَ العامِلِ:

"وإنْ كانَ تَرَكَ اشْتِراطَ التَّلْقيحِ عَلَيْهِ وقَدِ اشْتَرَطَ ما سِواهُ لَمْ يَجُزْ لِأنَّ تَرْكَ التَّلْقيحِ يَضُرُّهُ عَلى ما بَيَّنّا أَنَّ النَّخيلَ إذا لَمْ يُلَقَّحْ أحْشَفَ التَّمْرُ، فَقَدْ بَقِيَ بَعْضُ العَمَلِ عَلى صاحِبِ النَّخْلِ وهُوَ ما تَحْصُلُ بِهِ صِفَةُ السَّلامَةِ".

السرخسي، المبسوط. ج: 23، ص: 151.— المصدر
4.3معنى من معاني «أَحْشَفَ»

وــــــــــ فُلانٌ الجَفْنَ: أَعْطَبَهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ البُجَيْرَمِيُّ (1211ه=1796م، تاريخ تبييض الكتاب) يُبَيِّنُ أَحْكامَ الدِّية في الجِناياتِ:

"قَوْلُهُ (وفِي إحْشافِ الجَفْنِ) أيْ بِأنْ ضَرَبَهُ بِهِ، وأحْشَفَ جَفْنَهُ، أَيْ: أوْقَفَهُ فَصارَ لا يَتَحَرَّكُ".

البجيرمي، تحفة الحبيب على حاشية الخطيب. ج: 4، ص: 142.— المصدر
05 —

حَشَّفَ

فعل
5.1معنى من معاني «حَشَّفَ»

* حَشَّفَ فُلانٌ الشَّيْءَ: جَفَّفَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعِيُّ (ت: 204ه=819م) يُقَدِّرُ دِياتِ الجُروحِ:

"وإنْ كانَ الضَّرْبُ لَمْ يَجْرَحْ وبَقِيَ مِنهُ شَيْنٌ ،فَهَكَذا أوَّلًا يُؤْخَذُ لِلشَّيْنِ شَيْءٌ إلّا أنْ يَكُونَ شَيْنٌ لا يَذْهَبُ بِحالٍ، أوْ يَنالُ اللَّحْمَ بِما يُحَشِّفُهُ ،أوْ يُفَجِّرُ مِنْهُ شَيْئًا ،أوْ يَجْرَحُهُ . فَإنْ جَرَحَهُ في الرَّأْسِ أوِ الوَجْهِ جُرْحًا دُونَ المُوضِحَةِ ؛ قيلَ لِأهْلِ البَصَرِ بِذَلِكَ : قَدِّرُوا لِذَلِكَ بِقَدْرِهِ مِنَ المُوضِحَةِ واحْتاطُوا".

الشافعي، الأم. تح: رفعت عبد المطلب، ج: 7، ص: 206.— المصدر

وــــــــــ عَيْنَهُ: ضَمَّ جُفونَهُ ونَظَرَ مِنْ خِلالِ هُدْبِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321 هـ= 933م):

"وحَشَّفَ الرَّجُلُ عَيْنَهُ: ضَمَّ جُفونَهُ ونَظَرَ مِنْ خِلالِ هُدْبِها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دُرَيْدٍ، جَمْهَرَة اللّغة. تح: بعلبكي، ص: 537.— المصدر

تَحَشَّفَ

فعل
6.1معنى من معاني «تَحَشَّفَ»

* تَحَشَّفَ فُلانٌ: لَبِسَ الحَشيفَ، وَهُوَ الخَلَقُ مِنَ الثِّيابِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبانُ بْنُ سَعيدِ بْنِ العاصِ الأُمَوِيُّ (8 هـ=629) يُخاطِبُ عُثْمانَ بْنَ عَفّانَ:

"يا بْنَ عَمِّ، أَراكَ مُتَحَشِّفًا، أَسْبِلْ كَما يُسْبِلُ قَومُكَ".

عمر بن شبة، تاريخ المدينة. تح: فهيم شلتوت وآخر، ج: 1، ص: 956.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانيُّ (ت: 650 هـ=1252م):

"وتَحَشَّفَ: لَبِسَ الحَشيفَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في بَقِيَّةِ العُصورِ)

الصَّغانيّ، العُباب الزّاخِر. تح: آل ياسين، ج: 1، ص: 99.— المصدر
6.2معنى من معاني «تَحَشَّفَ»

وــــــــــ فُلانٌ: صارَ هَزيلَ الجِسْمِ رَثَّ الهَيْئَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الحُسَيْنُ بْنُ الضَّحّاكِ الخَليعُ (ت: 250ه=864م) يَهْجو رَجُلًا:

سَكِرَ الشَّيْخُ مُعَرِّفْ
فاسْقِهِ الكَأْسَ وصَرِّفْ

لا يَروعَنْكَ مِنَ المَجْـ
دورِ وجْهٌ مُتَحَشِّفْ

السري الرفّاء، المحب والمحبوب والمشموم والمشروب. تح: ماجد الذهبي، ج: 4، ص: 44.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَديمِ العُقَيْليُّ (ت: 660 هـ=1262م) يَذْكُرُ رَجُلًا:

"وأَمّا صاحِبُ الحِصْنِ فَقالَ لَهُ خَواصُّهُ ونُدَماؤُهُ: عَلى أَيِّ شَيءٍ عَزَمْتَ؟ فَقالَ: عَلى القُعودِ، فَإِنَّ نورَ الدّينِ قَدْ تَحَشَّفَ مِنْ كَثْرَةِ الصَّومِ والصَّلاةِ، فَهُوَ يُلْقي نَفْسَهُ ومَنْ مَعَهُ في المَهالِكِ".

ابن العديم، زبدة الحلب من تاريخ حلب. ج: 1، ص: 346.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1389 هـ=1970م):

"وتَحَشَّفَ: صارَ يابِسَ الجِلْدِ رَثَّ الهَيْئَةِ".

مجمع القاهرة، المعجم الكبير. ج: 5، ص: 371.— المصدر
6.3معنى من معاني «تَحَشَّفَ»

وــــــــــ الشَّيءُ: يَبِسَ وتَقَبَّضَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

تُطْعِمُ الأَيْرَ تِينَةً
قَدْ عَلاها التَّحَشُّفُ

ديوانه. تح: بسج ج: 2، ص: 440.— المصدر
6.4معنى من معاني «تَحَشَّفَ»

وــــــــــ أَوْبارُ الإِبِلِ: طارَتْ عَنْها وتَفَرَّقَتْ.

د

"وتَحَشَّفَتْ أَوْبارُ الإِبِلِ: طارَتْ عَنْها وتَفَرَّقَتْ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في العَصْرِ الحَديثِ)

ابْن مَنْظور، لسان العرب. تح: أمين عبد الوهاب وآخرين، ج: 3، ص: 190.— المصدر
07 —

اسْتَحْشَفَ

فعل
7.1معنى من معاني «اسْتَحْشَفَ»

* اسْتَحْشَفَ العُضْوُ: يَبِسَ وتَقَبَّضَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ نِسْوةٌ مِنْ قُرَيْشٍ(23ه=644م) يُخاطِبْنَ عُمَرَ بْنَ الخَطّابِ في نِسْبةِ طِفْلٍ تَنازَعَ فيهِ رَجُلانِ:

"إنَّ الأَوَّلَ خَلا بِها، وخَلّاها، فَحاضَتْ على الحَمْلِ، فاسْتَحْشَفَ الوَلَدُ، فَظَنَّتْ أَنَّ عِدَّتَها انْقَضَتْ، فَدَخَلَ بِها الثّاني، فانْتَعَشَ المَوْلودُ بِماءِ الثّاني".

ابن القصّار، عيون الأدلة في مسائل الخلاف بين فقهاء الأمصار. تح: عبد الحميد السعودي، ص: 1420.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ إِسْماعيلُ بْنُ يَحْيى المزنيُّ (ت: 264 هـ=878م):

"وفي الأُذُنَيْنِ المُسْتَحْشِفَتَيْنِ بِهِما مِنَ الِاسْتِحْشافِ ما بِاليَدِ مِنْ الشَّلَلِ، وذَلِكَ أَنْ تُحَرَّكا فَلا تَتَحَرَّكا أَوْ تُغْمَزا بِما يُؤْلِمُ فَلا تَأَلَّما".

المزني، مختصر المزني في فروع الشافعية. ج: 1، ص: 322.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوَويُّ (ت: 676 هـ=1277م) يُبَيِّنُ أَحْكامَ القِصاصِ:

"تُقْطَعُ الأُذُنُ الصَّحيحَةُ بِالمُسْتَحْشِفَةِ؟ قَوْلانِ، أَظْهَرَهُما: نَعَمْ، لِبَقاءِ الجَمالِ والمَنْفَعَةِ مِنْ جَمْعِ الصَّوْتِ ورَدِّ الهَوامِّ بِخِلافِ اليَدِ الشَّلّاءِ".

النووي، رَوْضَة الطالِبين. تح: زهير الشاويش، ج: 9، ص: 196.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ أَبي سُلَيْمانَ البَكْريّ (ت: 1408 هـ=1987م) يُبَيِّنُ دِيةَ الجَفْنِ:

"وفيهِ الرُّبْعُ سَواءٌ الأَعْلى والأَسْفَلُ، وفي جَفْنَيْنِ نِصْفُ الدِّيَةِ، وإِنِ اسْتَحْشَفَ فَقَطْ فَفيهِ كَذَلِكَ".

البكري، محمد بن أبي بكر، الاعتناء في الفرق والاستثناء. تح: عادل عبد الموجود وآخر، ج: 1، ص: 971.— المصدر
7.2معنى من معاني «اسْتَحْشَفَ»

وــــــــــ التَّمْرُ ونَحْوُهُ: صارَ حَشَفًا رَديئًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّافِعيُّ (ت: 204 هـ=819م):

"والحِنْطَةُ قَدْ تَفْسُدُ بِما وُصِفَتْ ويَكونُ لَها فَسادٌ بِأَنْ تَكونَ مُسْتَحْشِفَةً".

الشافعي، الأم. تح: محمد النجار، ج: 3، ص: 52.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بْنُ عَبْدِ السَّلامِ (ت: 660ه=م) يُبَيِّنُ حُكْمَ بَيْعِ الرُّطَبِ قَبْلَ جَفافِهِ:

وما لا يُجفَّفُ مِنَ الرُّطَبِ والعِنَبِ، ولَوْ جُفِّفَ لَفَسَدَ واسْتَحْشَفَ، ففيهِ أَرْبَعةُ أَقْوالٍ أَحَدُها لا يُباعُ إلّا في حالِ الجَفافِ".

العز بن عبد السلام، الغاية في اختصار النهاية. تح: إياد الطباع، ج: 3، ص: 198.— المصدر
7.3معنى من معاني «اسْتَحْشَفَ»

وــــــــــ فُلانٌ: لَبِسَ الرَّثَّ مِنَ الثِّيابِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

مُسْتَـرِقُ النَّغْمَةِ مَخْنونُها
مُسْتَحْشِفٌ فـي خِلْقَةِ العَنْكَبِ

ذُو صَلْعَةٍ بَرْصاءَ مَغْسولَةٍ
مِنْ صِيغَةِ المُذْهَبِ والمُشِرَبِ

ديوانه. تح: بسج، ج: 1، ص: 201.— المصدر
7.4معنى من معاني «اسْتَحْشَفَ»

وــــــــــ الضَّرْعُ: جَفَّ لَبَنُهُ فَصارَ مُتَقَلِّصًا ومُتَقَبِّضًا.

ع

7.5معنى من معاني «اسْتَحْشَفَ»

وــــــــــ فُلانٌ العُضْوَ: أَعْطَبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَلِيُّ بْنُ مُحَمَّدٍ الماوَرديُّ (ت: 450ه=1058م) يُبَيِّنُ دِياتِ أَعْضاءِ الإنْسانِ:

"ولَوْ جَنى عَلَيْهِما فاسْتَحْشَفَتا ويَبِسَتا حَتّى لَمْ يَتَحَرَّكا ولَمْ يَأْلَما فَعَلَيْهِ الدِّيَةُ كامِلَةً قَوْلًا واحِدًا، بِخِلافِ الأنْفِ إذا اسْتَحْشَفَهُ أَنَّ عَلَيْهِ حُكومةً في أحَدِ القَوْلَيْنِ، لِأنَّ مَنفَعةَ الأنْفِ باقِيةٌ ومَنفَعةَ الشَّفةِ ذاهِبةٌ، وإنْ تَقَلَّصَتا بِالجِنايةِ حَتّى صارَ كاشِرَ الأسْنانِ".

الماوردي، الحاوي الكبير. تح: علي معوض وآخر، ج: 12، ص: 261.— المصدر

حَشَف

اسم
10.1معنى من معاني «حَشَف»

* الحَشَفُ: التَّمْرُ اليابِسُ الفاسِدُ العَديمُ النَّوى.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرؤُ القَيْسِ (ت: 80 ق هـ=544م) يُشَبِّهُ قُلوبَ الطُّيورِ الرَّطِبَةَ واليابِسَةَ في وكْرِ العُقابِ:

كأنَّ قُلوبَ الطَّيرِ رَطْبًا ويابسًا
لَدى وكْرِها العُنّابُ والحَشَفُ البالي

ديوانه. تح: أنور عليان وآخرين، ج: 1، ص: 359.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النّبِيُّ (ت: 11ه=632م) لَمّا تَصَدَّقَ رَجُلٌ بِتَمْرٍ رَديءٍ يُخاطِبُ أَصْحابَهُ:

«لَوْ شاءَ رَبُّ هَذِهِ الصَّدَقَةِ تَصَدَّقَ بِأَطْيَبَ مِنْ هَذا، ‌إِنَّ ‌رَبَّ ‌هَذِهِ ‌الصَّدَقَةِ ‌يَأْكُلُ ‌الحَشَفَ ‌يَوْمَ ‌القِيامةِ".

”وقالَ سالِمٌ مَوْلى عُمَرَ بْنِ الخَطّابِ (23ه=644م) يُبَيِّنُ كَيْفَ كانَ يُقَدِّمُ التَّمْرَ للفاروقِ:

"كُنْتُ أَحْشِفُ لِعُمَرَ صاعًا مِنْ تَمْرٍ، فَيَأْكُلُهُ بَحَشَفِهِ".

”وقالَ الحُرَيْثُ بْنُ زَيْدِ الخَيْلِ (ت: 60ه=680م):

قَتَلْنا بِقَتْلانا مِنَ القَوْمِ عُصْبةً
كِرامًا ولَمْ نَأْكُلْ بِهِمْ حَشَفَ النَّخْلِ

الجرّاوي، الحماسة المغربية. تح: محمد الداية، ج: 2، ص: 823.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
