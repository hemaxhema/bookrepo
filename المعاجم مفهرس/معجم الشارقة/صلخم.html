01 —

اصْلَخَمَّ

فعل
1.1معنى من معاني «اصْلَخَمَّ»

*اصْلَخَمَّتْ لُجَجُ النَّهْرِ ونَحْوِهِ: تَلاطَمَتْ واصْطَخَبَتْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَمْرُو بْنُ العاصِ (23هـ=644م) يَصِفُ فَيَضانَ النّيلِ في رِسالَةٍ إِلى عُمَرَ بْنِ الخَطَّابِ:

"لَهُ أَيَّامٌ تَسيلُ لَهُ عُيونُ الأَرْضِ وينَابيعُها مَأْمورَةً إِلَيهِ بِذَلِكَ، حَتّى إِذا رَبا وطَما واصْلَخَمَّ لُجَجُهُ، واغْلَوْلَبَ عُبابُهُ، كانَتِ القُرى بِما أَحاطَ بِها كَالرُّبا".

ابن زولاق، فضائل مصر. تح: محمد، ص: 79.— المصدر
1.2معنى من معاني «اصْلَخَمَّ»

وــــــــــ فُلانٌ وغَيْرُهُ: اسْتَكْبَرَ، أَوْ رَفَعَ رَأْسَهُ تَكَبُّرًا. (انْظُرْ: ط ر ه م، ط ل خ م)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ذو الرُّمَّةِ (ت: 117هـ=735م) يَصِفُ حُمُرًا وَحْشِيَّةً:

فَظَلَّـتْ بِمَلْقـى واحِـفٍ جَـرَعَ المِعـى
قِيامــًا يُغالي مُصْـلَخِمًّا أَميرُهـا 

الأزهري، تهذيب اللغة. تح: سرحان، ج: 7، ص: 660.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (136هـ=754م) يَتَهَدَّدُ في سياقِ مَدْحِهِ أَبا العَبّاسِ السَّفّاحَ:

فَأَيُّها الحامِلُ أَنْفًا نَخْشِمُهْ

وكُلَّ صَتْمٍ صامِلٍ مُصَتَّمُهْ

إذا اصْلَخَمَّ لمْ يُرَمْ مُصْلَخْمِمُهْ

شرح ديوان رؤبة لعالم لغوي قديم، تح: عوض الله، ج: 2، ص: 257.— المصدر
1.3معنى من معاني «اصْلَخَمَّ»

وــــــــــ الجَبَلُ ونَحْوُهُ: صَلُبَ وامْتَنَعَ. (انظر: ط ل خ م)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَهْجو:

يَهْوونَ عَنْ أَرْكانِ عِزٍّ أَدْرما

عَنْ صامِلٍ عاسٍ إِذا ما اصْلَخْمَما

ديوان رؤبة. تح: البروسي، ص: 184.— المصدر
1.4معنى من معاني «اصْلَخَمَّ»

وــــــــــ: سَكَتَ غاضِبًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الـمُصْلَخِمُّ: السّاكِتُ الغَضْبانُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 330.— المصدر
1.5معنى من معاني «اصْلَخَمَّ»

وــــــــــ فُلانٌ وغَيرُهُ: انْتَصَبَ قائِمًا دونَ حَرَكَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زِيادٍ الكِلابِيُّ (ت: 200هـ=815م):

"الاصْلِخْمامُ: القائِمُ لا يتَحَرَّك، وهو مُصْلَخِمٌّ".

الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 171.— المصدر
1.6معنى من معاني «اصْلَخَمَّ»

وــــــــــ الشَّيْءُ: اشْتَدَّ سَوادُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عُبَيْدٍ (ت: 224هـ=838م):

"أَسوَدُ دَجوجِيٌّ وخُدارِيٌّ ودامِجٌ ودَيْجورٌ ودَيْجوجٌ ومُصْلَخِمٌّ".

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 203.— المصدر
02 —

صُلاخِم

اسم
2.1معنى من معاني «صُلاخِم»

* الصُّلاخِمُ مِنَ الإِبِلِ: الصُّلْبُ الشَّديدُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو النَّجْمِ العِجْلِيُّ (ت: 130هـ=748م) يَصِفُ قُوَّةَ بَعيرِهِ:

وكاهِلٍ ضَخْمٍ وعُنْقٍ عَرْطَلِ

صُلاخِمٍ مَفصِلُهُ في الـمَفْصِلِ

ديوان أبي النجم العجلي. تح: أَديب، ص: 357.— المصدر
03 —

صِلْخام

اسم
3.1معنى من معاني «صِلْخام»

* الصِّلْخامُ مِنَ الإِبِلِ: الشَّديدُ الجَسيمُ الطَّويلُ. (انْظُرْ: ص ل ق م)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رُؤْبَةُ (120هـ=738م) يَمْدَحُ مَسْلَمَةَ بْنَ عَبْدِ المَلِكِ بْنِ مَرْوانَ:

يَزيدُ لَوْ سُقْتَ بَني خُمامِ

وسُقْتَ أَلْفَي ساحِرٍ أَثّامِ

لاقَيْتَ نَجْمًا نَكِدَ النِّجامِ

في عارِضٍ مِنْ مُضَرَ الصِّلْخامِ

ديوان رؤبة. تح: البروسي، ص: 147.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"بَعِيرٌ صِلْخامٌ: طَويلٌ، وقالوا: الصُّلْبُ الشَّديدُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: أبو الفضل، ج: 6، ص: 71.— المصدر
04 —

صَلْخَم

اسم
4.1معنى من معاني «صَلْخَم»

* الصَّلْخَمُ مِنَ الجِبالِ: الصُّلْبُ الـمَنيعُ. (ج) صَلاخِمُ، وصَلاخِمَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«عُرِضَتِ الأَمانَةُ على الجِبالِ الصُّمِّ الصَّلاخِمِ».

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 314.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م) يَشْرَحُ مُفْرَدَةَ (الصَّلْخَمِ) في قَوْلِ النَّبِيّ ﷺ: «عُرِضَتِ الأَمانَةُ على الجِبالِ الصُّمِّ الصَّلاخِمِ»:

"جَمْعُ صَلْخَمٍ؛ وهُوَ الجَبَلُ الصُّلْبُ الـمَنيعُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 314.— المصدر
4.2معنى من معاني «صَلْخَم»

وــــــــــ مِنَ الإِبِلِ: الشَّديدُ الجَسيمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ هَمّامٍ السَّلوليُّ (65هـ=685م) يَطْلُبُ مِنَ النُّعْمانِ بْنِ بَشيرٍ الأَنْصارِيِّ زِيادَةَ العَطاءِ:

فَإِنَّكَ قَدْ حُمِّلْتَ مِنّا أَمانَةً
بِما عَجَزَتْ عَنْهُ الصَّلاخِمَةُ البُزْلُ

شعر عبد الله بن همّام السّلوليّ. تح: السّراقبي، ص: 90.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"بَعيرٌ صِلَّخْمٌ وصَلْخَمٌ ومُصْلَخِمٌّ: جَسيمٌ ماضٍ شَدِيدٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 159.— المصدر
4.3معنى من معاني «صَلْخَم»

وــــــــــ: الماضي السَّريعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"بَعيرٌ صِلَّخْمٌ وصَلْخَمٌ ومُصْلَخِمٌّ: جَسيمٌ ماضٍ شَدِيدٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 159.— المصدر
05 —

صِلَّخْم

اسم
5.1معنى من معاني «صِلَّخْم»

* الصِلَّخْمُ مِنَ الإِبِلِ: الشَّديدُ الجَسيمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَفْخَرُ:

فَاليَوْمَ أَرْمي بِسَنًا ذي جِسْمِ

بِكُلِّ صَرّافِ الشَّبا صِلَّخْـــمِ

ديوان رؤبة. تح: البروسي، ص: 141.— المصدر
5.2معنى من معاني «صِلَّخْم»

وــــــــــ مِنَ الجِبالِ: الصَّغيرُ الـمَنيعُ أَوِ الصُّلْبُ الـمُمْتَنِعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (170هـ=786م) مِنْ شَواهِدِ الخَليلِ، في بَيْتٍ مُفْرَدٍ:

ورَأْسُ عِزٍّ راسيًا صِلَّخْما

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 330.— المصدر
5.3معنى من معاني «صِلَّخْم»

وــــــــــ مِنَ النّاسِ: العَيّابُ الشَّتّامُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ قُدامَةُ بْنُ جَعْفَرٍ (ت: 337هـ=949م) في بابِ الثَّلْبِ والمُلاحاةِ:

"ورَجُلٌ عَيَّابٌ مُغْتابٌ، سَبّابٌ ... مُتْبَعٌ صِلَّخْمٌ".

ابن جعفر، جواهر الألفاظ. تح: عبد الحميد، ص: 42.— المصدر
5.4معنى من معاني «صِلَّخْم»

وــــــــــ مِنَ الإِبِلِ: الماضي السَّريعُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"بَعيرٌ صِلَّخْمٌ ...: جَسِيمٌ شديدٌ ماضٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، لسان العرب. تح: عبد الوهاب وآخر، ج: 7، ص: 385.— المصدر
06 —

مُصْلَخِمّ

اسم
6.1معنى من معاني «مُصْلَخِمّ»

* المُصْلَخِمُّمِنَ الإِبِلِ: الشَّديدُ الجَسيمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُمَيْدُ بْنُ ثَوْرٍ (ت: 60هـ=680م) يَصِفُ بَعِيرًا في بَيْتٍ مُفْرَدٍ:

عَلَى مُصْلَخِمٍّ ما يَكادُ جَسيمُه
يَمُدُّ بِعِطْفَيهِ الوَضينَ المُسَمَّما

ديوان حميد بن ثور. تح: البيطار، ص: 386.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"بَعيرٌ صِلَّخْمٌ وصَلْخَمٌ ومُصْلَخِمٌّ جَسيمٌ ماضٍ شَدِيدٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 159.— المصدر
