01 —

أَهَّب

فعل
1.1معنى من معاني «أَهَّب»

: * أَهَّبَفُلانٌ لِلأَمْرِ: اسْتَعَدَّ لَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ سيدَه (ت: 458هـ=1066م):

"أَخَذَ لِذَلِكَ الأَمْرِ أُهْبَتَهُ؛ أَيْ: هَيْئَتَهُ وعُدَّتَهُ، وقَدْ أَهَّبَ لَهُ، وَتَأَهَّبَ".

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 362.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ الأَبّارِ (ت: 658هـ=1260م):

أَهْلُهُ لِلْبَينِ يُثْقِلُهُمْ
بَيْنَ تَأْهيبٍ وتَأْهيلِ

ديوان ابن الأبار. تح: الهرّاس، ص: 246.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عاشور (ت: 1355هـ=1936م):

"إِنَّ الأَذانَ شُرِعَ لِمَصْلَحَةِ الإِعْلامِ بِدُخولِ الوَقْتِ، والإِقامَةَ شُرِعَتْ لِإِعْلامٍ بِالدُّخولِ في الصَّلاةِ. والتَّحْضيرُ والتَّصْبيحُ والتَّأْهيبُ مِنَ ذَلِكَ النَّوْعِ".

ابن عاشور، جمهرة المقالات والرسائل،ج: 1، ص: 782.— المصدر
02 —

تَأَهَّب

فعل
2.1معنى من معاني «تَأَهَّب»

: * تَأَهَّبَفُلانٌ لِلأَمْرِ: اسْتَعَدَّ لَهُ وَتَهَيَّأَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال جَسَّاسُ بْنُ مُرَّةَ (ت: 91ق ه=533م) يَحُثُّ أَباهُ عَلى الاسْتِعْدادِ لِلْحرْبِ:

تَأَهَّبْ عَنْكَ أُهْبَةَ ذي كِفاحٍ
فَإِنَّ الأَمْرَ جَلَّ عَنِ التَّلاحي

ديوان بني بكر في الجاهلية. تح: نبوي، ج: 1، ص: 380.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال كَعْبُ بْنُ مالِكٍ (ت: 50هـ=669م) يُخبِرُ عَنْ فِعْلِ رَسولِ اللهِ ﷺ في غَزْوَةِ تَبُوك:

"اسْتَقْبَلَ سَفَرًا بَعيدًا؛ فَجَلَّى لِلْمُسْلِمينَ أَمْرَهُمْ وأَخْبَرَهُمْ بِذَلِكَ لِيَتَأَهَّبوا أُهْبَةَ عَدُوِّهِمْ، وأَخْبَرَهُمْ بِالوَجْهِ الَّذي يُريدُ".

مسند ابن أبي شيبة. تح: العزازي وآخر، ج: 2، ص: 335.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال القاضي أبو يوسُفَ (ت: 182هـ=798م):

"فَإِنَّهُ سافَرَ في حَرٍّ شَديدٍ، وأَرادَ أَنْ يَسْتَقْبِلَ سَفَرًا بَعيدًا؛ فَأَخْبَرَ النّاسَ بِذَلِكَ لِيَتَأَهَّبوا لِعَدُوِّهِمْ".

القاضي أبو يوسف، الخراج، ج: 1، ص: 192.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ عِذاري المَراكُشيُّ (ت: 695هـ=1296م) يُخبِرُ عَنْ زَعيمٍ مِنْ زُعَماءِ النَّصارى:

"فَجَمَعَ جُموعَهُ، ورَتَّبَ أُمورَهُ، وتَأَهَّبَ لِلِقائِهِمْ بِخارِجِ الحَضْرَةِ".

ابن عذاري، البيان المغرب. تح: معروف وآخر، ج: 3، ص: 444.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال البُجَيْرَميُّ (ت: 1221هـ=1806م):

"قَوْلُهُ: (بِقَصْدِ الجِهادِ)؛ أَيْ: بِقَصْدِ التَّأَهُّبِ لِلْجِهادِ".

البجيرمي، تحفة الحبيب، ج: 5، ص: 263.— المصدر
03 —

إِهَاب

اسم
3.1معنى من معاني «إِهَاب»

: * الإِهَابٌ: الجِلْدُ ما لَمْ يُدْبَغْ. (ج) أُهُبٌ، وأَهَبٌ، وآهِبَةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال الشَّنْفَرى (ت: 70ق ه=553م):

وكَفِّ فَتًى لَمْ يَعْرفِ السَّلْخَ قَبْلَها
تَجورُ يَداهُ في الإِهابِ وتَخرُجُ

ديوان الشنفرى. تح: إميل يعقوب، ص: 39.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال النَّبيُ ﷺ (ت: 11هـ = 632م):

«إِذا دُبِغَ الإِهابُ فَقَد طَهُرَ».

سنن أبي داود. تح: الأرناؤوط وآخر، ج: 6، ص: 211.— المصدر
”وقال الـمُنذِرُ بنُ حَسَّانٍ (ت: 40هـ=661م) يَهْجو فارًّا مِنَ الـمَعْرَكَةِ:

وأفَلتنا هَجينُ بَني سُلَيْمٍ
يُفَدّي الـمُهْرَ مِنْ حُبِّ الأيابِ

فَلَوْلا اللّهُ والـمُهْرُ الـمُفَدّى
لَأُبْتَ، وأَنْتَ غِرْبالُ الإِهابِ

ابن سيده، المخصص. تح: جفال، ج: 4، ص: 424— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو نُواسٍ (ت: 198هـ=814م):

أَدْرَجْتُمُ في إِهابِ العَيْرِ جُثَّتَهُ
فَبِئْسَ ما قَدَّمَتْ أَيْديكُمُ لِغَدِ

أبو نواس في نوادره وبعض قصائده. تح: سالم شمس الدين، ج: 1، ص: 138.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ دَقيقِ العيدِ (ت: 702هـ=1302م):

"أَنَّ الـمَقْصودَ تَعْميمُ الطَّهارَةِ فيما يُدْبَغُ مِنَ الأُهُبِ النَّجِسَةِ".

ابن دقيق، شرح الإلمام. تح: خلوف، ج: 2، ص: 445.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال البُجَيْرَميُّ (ت:1221هـ=1806م):

"الإِهابُ بِكَسْرِ الهَمْزَةِ كَكِتابٍ: اسمٌ لِلجِلْدِ قَبْلَ دَبْغِهِ؛ سُمّي بِهِ لَأَنَّهُ أُهْبَةٌ للحَيّ؛ أَيْ: يَنْتَفِعُ بِهِ وبَقاءٌ لِحِمايَةِ جَسَدِهِ".

البجيرمي، تحفة الحبيب، ج: 1، ص: 142.— المصدر
3.2معنى من معاني «إِهَاب»

وــــــــــــ: جِلْدُ الْإِنْسَانِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال عَنْتَرَةُ (ت: 22ق ه=601م):

فَشَكَكْتُ بِالرُّمْحِ الطَّويلِ إِهابَهُ
لَيْسَ الكَريمُ عَلى القَنا بِمُحَرَّمِ

ابن سعيد الأندلسي، نشوة الطرب. تح: نصرت عبد الرحمن، ج: 1، ص: 549.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالت عائِشَةُ بِنْتُ أَبي بَكْرٍ (ت: 58هـ=678م) تَصِفُ أَباها:

"حَقَنَ الدِّماءَ في أُهُبِها".

أم المؤمنين عائشة، تفسيرها. تح: أبو السعود، ج: 1، ص: 105.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أَبو نواسٍ (ت: 198هـ=814م) يَهْجو رَجُلًا:

لقَدْ غَرَّني مِنْ جَعْفَرٍ حُسْنُ بابِهِ
ولَمْ أَدْرِ أَنَّ اللُّؤْمَ حَشْوُ إهابِهِ

ديوان أبي نواس. تح: الحديثي، ج: 1، ص: 62.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال نَجْمُ الدّينِ الغَزّيُّ (ت: 1061هـ=1651م):

"كُلُّ ذَلِكَ مَجْمُوِعٌ في القَلْبِ، وكَأَنَّ المَجْموعَ في إِهابِ الإِنْسانِ خِنْزيرٌ، وكَلْبٌ، وشَيْطانٌ، وحَكَيمٌ".

الغزي، حُسْن التنبه، ج: 10، ص: 437.— المصدر
04 —

أُهْبة

اسم
4.1معنى من معاني «أُهْبة»

: * الأُهْبَةُ: العُدَّةُ. (ج) أُهُباتٌ، وأُهَبٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال المُهَلَهَلُ بْنُ رَبيعَةَ (ت: 100ق ه=525م) يُخاطِبُ نَفْسَهُ:

تَأَهَّبْ عَنْكَ أُهْبَةَ ذي امْتِناعِ
فَإِنَّ الأَمْرَ جَلَّ عَنِ التَّلاحي

ديوان المهلهل. تح: القوال، ص: 126.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الأَعْشى مَيْمونُ (ت: 7هـ=628م) يَمدَحُ:

كَفى قَومَهُ شَيبانَ أَنَّ عَظيمَةً
مَتى تَأتِهِ تُؤخَذْ لَها أُهُباتُها

ديوان الأعشى الكبير. تح: محمد حسين، ص: 87.— المصدر
”وقالَ كَعْبُ بْنُ مالِكٍ (ت: 50هـ=671م) يُخبِرُ عَنِ النَّبيِّ ﷺ في غَزْوَةٍ مِنَ الغَزَواتِ:

"فَجَلّى لِلْمُسْلِمينَ أمْرَهُمْ لِيَتَأهَّبوا أُهْبَةَ غَزْوِهِمْ".

صحيح البخاري. تح: محمد الناصر، ج: 6، ص: 3— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال السَّدوسيُّ (ت: 195هـ=811م) يَشرَحُ مَثَلًا:

"خُذْ أُهْبَةَ الأَمْرِ قَبْلَ أَنْ يَنْزِلَ بِكَ".

السدوسي، الأمثال. تح: عبد التواب، ج: 1، ص: 40.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ العَديمِ (ت: 660هـ=1262م) يَصِفُ خُروجَ ابْنِ حَمْدانَ لِقِتالِ الرَّجّالَةِ:

"كانَ أَبو العَلاءِ في دارِ الخَليفَةِ عَلى غَيْرِ أُهْبَةٍ، فَأَمَرَه بِالخُروجِ إِلَيْهِمْ".

ابن العديم، بغية الطلب. تح: زكار، ج: 9، ص: 4292.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال عَبْدُ اللهِ بْنُ الحاجِ الشَّنْقيطيُّ (ت: 1226هـ=1811م) يَنْصَحُ:

"إِذا لَبِسْتَ دِرْعَ الصّيانَةِ، وجَعَلْتَ عَلى رَأْسِكَ بَيْضَةَ الأَمانَةِ، وأَخَذْتَ بِشِمالِكَ جُنَّةِ التَّقْوى، وهَزَزْتَ بِيَمينِكَ صارِمَ التَّوَكُّلِ والنَّجوى؛ فَقَدْ أَخذْتَ أُهْبَةَ النَّجاةِ مِنَ الـمَهالِكِ".

الشنقيطي، فتح الودود. تح: مأمون أحمد، ج: 1، ص: 374.— المصدر
4.2معنى من معاني «أُهْبة»

0 أُهْبَةُ الحَرْبِ: عُدَّتُها وأَدَواتُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الواقِديُّ (ت: 207هـ=822م) عَنْ اسْتِعْدادِ المُسْلِمينَ لِحَرْبِ الرّومِ:

"ما أَصْبَحَ الصَّباحُ إِلَّا والمُسْلِمونَ قَدْ أَخَذوا أُهْبَةَ الحَرْبِ".

الواقدي، فتوح الشام. تح: عبد اللطيف عبد الرحمن، ج: 2، ص: 133.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ العَديمِ (ت: 660هـ=1262م) يُخبِرُ عَنْ أَبي التُّرْكِ السُّلَمي وشَجاعَتِهِ:

"عَقَدَ بَيْنَهُ وبَيْنَ الْلَّهِ -عَزَّ وَجَلَّ- لا رَكْبَ بَعْدَها إِلى جِهادٍ ولا نَفيرٍ إِلّا بَعْدَ أَنْ يَتَأَهَّبَ أُهْبَةَ الحَرْبِ كَما يَتَأَهَّبُ مَنْ يُبارِزُ عَدُوَّهُ في مَصافِهِ وأَوْقاتِ حَذَرِهِ وخَوْفِهِ".

ابن العديم، بغية الطلب. تح: زكار، ج: 10، ص: 4369.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال عَبْدُ الرَّحْمنِ بْنُ الحَسَنِ البَهْكَليُّ (ت: 1248هـ=1832م) يُخبِرُ عَنِ الشَّريف حُمود:

"أَقْبَلَ الشَّريفُ يَقْصُدُ قَرْيَةَ باجِلَ غَيْرَ مُتَأَهِّبٍ أُهْبَةَ الحَرْبِ".

اليهكلي، نفح العود. تح: العقيلي، ج: 22، ص: 189.— المصدر
4.3معنى من معاني «أُهْبة»

0 أُهْبَةُ النِّكاحِ: الاستِعدَادُ لَهُ والقُدْرَةُ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال البَغَويُّ (ت: 516هـ=1123م):

"يُسْتَحَبُّ لِمَنْ تاقَتْ نَفْسُهُ إِلى النِّكاحِ وَوَجَدَ أُهْبَةَ النِّكاحِ أَنْ يَتَزَوَّجَ".

معالم التنزيل. تح: النمر، وآخرين، ج: 6، ص: 38.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال النَّوَويُّ (ت: 676هـ=1277م) يَتَكَلَّمُ عَنِ التّائِقِ إِلى النِّكاحِ:

"إنْ وَجَدَ أُهْبَةَ النِّكاحِ؛ اسْتُحِبَّ لَهُ".

روضة الطالبين. تح: زهير الشاويش، ج: 7، ص: 18— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال نوَوي الجاوي (ت: 1316هـ=1937م) يَتَحَدَّثُ عَنْ وَقْتِ النِّكاحِ الـمَسْنونِ:

"(لِتائِقٍ) لَهُ بِتَوَقانِهِ للْوَطْءِ ولَو خَصِيًّا، أَوْ مُشْتَغِلًا بِالعِبادَةِ (قادِرٍ) عَلى أُهْبَةِ النِّكاحِ".

نووي الجاوي، نهاية الزين، ص: 298-299.— المصدر
