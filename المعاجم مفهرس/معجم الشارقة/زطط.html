زطط
زَطَّ(فعل) أَزَطّ(اسم) زُطّ(اسم) زُطِّيَّة(اسم)
المقدّمة
النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ز ط ط):

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܙܽܘܼܛܳܝܳܐ (زُطَّايَا) zuṭṭāyā يُطلَقُ على (جَيلٍ مِنَ النّاسِ مِنَ السِّندِ) ويُوافِقُ الاسمَ العَرَبيَّ (زُطّيٌّ) مَبنًى ومَعنًى.

(ܠܟܣܝܩܘܢ ܕܐܘܓܝܢ ܡܢܢܐ: ܙܘܛܝܐ)

المعانى الكلّيّة

  1. جِنْسٌ أَسْوَدُ مِنَ الهِنْدِ والسّودانِ
  2. تَصْويتُ الذُّبابِ
01 —

زَطَّ

فعل
1.1معنى من معاني «زَطَّ»

زَطَّ (كَنَصَرَ) الذُّبابُ يَزُطُّ زَطًّا: صَوَّتَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"وزَطَّ الذُّبابُ: صَوَّتَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 289.— المصدر
02 —

أَزَطّ

اسم
2.1معنى من معاني «أَزَطّ»

* الأَزَطُّ: الـمُسْتَوي الوَجْهِ، (ج) زُطُطٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"الأَزَطُّ: الـمُسْتَوي الوَجْهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الطحاوي، ج: 4، ص: 132.— المصدر
2.2معنى من معاني «أَزَطّ»

وــــــــــ: الـمُعْوَجُّ الفَكِّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"الأَزَطُّ: الـمُعْوَجُّ الفَكِّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الطحاوي، ج: 4، ص: 132.— المصدر
2.3معنى من معاني «أَزَطّ»

وــــــــــ: الخَفيفُ شَعْرِ اللِّحْيَةِ مِنَ العارِضَيْنِ، أَوْ مَنْ لا شَعْرَ عَلى عارِضَيْهِ. (انْظُرْ: ث ط ط)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"الزُّطُطُ والثُّطُطُ: الكَواسِجُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: البردوني، ج: 13، ص: 159.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد فارِس الشِّدْياق (ت: 1304هـ=1887م) يُعَدِّدُ مَحاسِنَ المَرْأَةِ والرَّجُلِ:

"وإِنَّ الـمَرْأَةَ إِذا كانَ في وَجْهِها شَعْرٌ ناعِمٌ أَوْ زَغَبٌ - ولا سِيَّما عَلى شَفَتَيْها - تُسْتَحَبُّ عِنْدَ جَميعِ النّاسِ. فَأَمّا الأَجْرَدُ مِنّا أَوِ السِّناطُ والأَزَطُّ، فمَكْروهٌ عِنْدَ اللهِ والنّاسِ".

أحمد فارس الشدياق، الساق على الساق، طبعة باريس، ص: 168.— المصدر
03 —

زُطّ

اسم
3.1معنى من معاني «زُطّ»

الزُّطُّ:جِنْسٌ أَسْوَدُ مِنْ الهِنْدِ والسّودانِ، والواحِدُ زُطِّيٌّ، وهِيَ بِتاءٍ. (مع).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (ت: 11هـ=632م):

«رَأَيْتُ عيسى وموسى وإِبْراهيمَ، فَأَمّا عيسى فأَحْمَرُ جَعْدٌ عَريضُ الصَّدْرِ، وأَمّا موسى فَآدَمُ جَسيمٌ سَبْطٌ كَأَنَّهُ مِنْ رِجالِ الزُّطِّ».

صحيح البخاري. تح: العطار، ص: 366.— المصدر
”وقالَ عَبْدُ اللهِ بْنُ مَسْعودٍ (ت: 32هـ=653م) يَقُصُّ حينَما أَخَذَهُ النَّبِيُّ ﷺ إِلى بَطْحاءِ مَكَّةَ، وخَطَّ لَهُ خَطًّا، وأَمَرَهُ أَلّا يَتَجاوَزَهُ:

"ثُمَّ انْطَلَقَ رَسولُ اللهِ ﷺ حَتّى لَمْ أَرَهُ، فَبَيْنا أَنا كَذلِكَ، إِذا أَنا بِرِجالٍ كَأَنَّهُمُ الزُّطُّ، شُعورُهُمْ وأَجْسامُهُمْ لا أَرى عَوْرَةً، ولا أَرى بِشَرًا، فَجَعَلوا يَنْتَهونَ إِلى الخَطِّ فَلا يَجوزونَهُ".

الفاكهي، أخبار مكة. تح: دهيش، ج: 4، ص: 24.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الزُّطُّ: جيلٌ مِنَ السّودانِ، والزُّطُّ: أَعْرابُ جَتَّ بالهِنْدِيَّةِ، وهُمْ جيلٌ مِنْ أَهْلِ الهِنْدِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 347.— المصدر
”وقالَ سَيْفُ بْنُ عُمَرَ (ت: 180هـ=796م) يَتَحَدَّثُ عَنِ الفِتْنَةِ الكُبْرى:

"فَرَجَعَ كَعْبٌ، وقَدِ اعْتَدَّ طَلْحَةُ والزُّبَيْرُ فيما بَيْنَ ذَلِكَ بِأَشْياءَ كُلُّها كانَتْ مِمّا يُعْتَدُّ بِهِ، مِنها أنَّ مُحَمَّدَ ابْنَ طَلْحَةَ - وكانَ صاحِبَ صَلاةٍ - قامَ مُقامًا قَريبًا مِنْ عُثْمانَ بْنِ حُنَيْفٍ، فَخَشِيَ بَعْضُ الزُّطِّ والسَّيابِجَةِ أنْ يَكونَ جاءَ لِغَيْرِ ما جاءَ لَهُ، فَنَحَّياهُ".

الطبري، تاريخ الأمم والملوك، دار الكتب العلمية، ج: 3، ص: 17.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شَرَفُ الدّينِ الأَنْصارِيّ (ت: 662هـ=1264م):

والأَحْمَقُ الغِرُّ لا يُصْغي لِمَوْعِظَةٍ
كَالأَقْرَعِ الزُّطِّ لا يَلْوي عَلى مَشَطِهْ

ديوان الصاحب شرف الدين الأنصاري. تح: عمر باشا، ص: 298.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ المُخْتارُ الكُنْتِيُّ الشِّنْقيطِيّ (ت: 1226هـ=1811م):

"وأَمّا حامُ، فَهُوَ أَبو القِبْطِ والحَبَشَةِ والزِّنْجِ والزُّطِّ والنّوبَةِ والهِنْدِ وجَميعِ أَنْواعِ السّودانِ".

المختار الكنتي، فتح الودود. تح: مأمون محمد، ص: 580.— المصدر
3.2معنى من معاني «زُطّ»

وــــــــــ: حُفّاظُ الطُّرُقِ في الهِنْدِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَوارِزْمِيُّ (ت: 387هـ=997م) يُعَدِّدُ طَبَقاتِ النّاسِ بالهِنْدِ:

"الزُّطُّ: هُمْ حُفّاظُ الطُّرُقِ".

الخوارزمي، مفاتيح العلوم. تح: الأبياري، ص: 143.— المصدر
04 —

زُطِّيَّة

اسم
4.1معنى من معاني «زُطِّيَّة»

* الزُّطِّيَّةُ: نَوْعٌ مِنَ الثِّيابِ يُنْسَبُ إِلى الزُّطِّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والزُّطُّ: أَعْرابُ جَتَّ بِالهِنْدِيَّةِ، وهُمْ جيلٌ مِنْ أَهْلِ الهِنْدِ، إِلَيْهِمْ تُنسَبُ الثِّيابُ الزُّطِّيَّةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 347.— المصدر
4.2معنى من معاني «زُطِّيَّة»

وــــــــــ: نَوْعٌ مِنْ حَلْقِ الرَّأْسِ يُنْسَبُ إِلى الزُّطِّ وَيُشْبِهُ الصَّليبَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو موسى المَدينِيّ (ت: 581هـ=1185م):

"في بَعْضِ الأَخْبارِ: "فَحَلَقَ رَأْسَهُ زُطِّيَّةً"، قيلَ: هُوَ مِثْلُ الصَّليبِ، كَأَنَّهُ مِنْ فِعْلِ الزُّطِّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

المديني، المجموع المغيث. تح: العزباوي، ج: 2، ص: 15.— المصدر
