01 —

غَفَقَ

فعل
1.1معنى من معاني «غَفَقَ»

* غَفَقَ (بِفَتْحِ الفاءِ) ‌فُلانٌ يَغْفِقُ (بِكَسْرِ الفاءِ) ‌غَفْقًا: أَكْثَرَ الشَّرابَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو حَنيفةَ الدِّينَوَريُّ (ت: 282هـ=895م):

"وكَذَلِكَ غَفَقَ يَغْفِقُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 3، ص: 205.— المصدر
1.2معنى من معاني «غَفَقَ»

وــــــــــ: خرَجَتْ مِنْهُ ريحٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيبانيُّ (ت: 206هـ=821م):

"غَفَقَ وعَفَقَ: إذا خَرَجَتْ مِنْهُ ريحٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
1.3معنى من معاني «غَفَقَ»

وــــــــــ الإبِلُ: ورَدَتْ كُلَّ ساعَةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الفَرّاءُ (ت: 207هـ=822م):

"‌شَرِبَتِ ‌الإبِلُ ‌غَفَقًا، وهِيَ تَغْفِقُ: إذا شَرِبَتْ مَرَّةً بَعْدَ أُخْرى، وهُوَ الشُّرْبُ الواسِعُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
”وقال الرّاجِز (231هـ=846م) مِن شَواهِدِ ابْنِ الأعْرابيّ، يَصِفُ الإبِلَ بِشُرْبِ الماءِ بَعْدَ أَكْلِ الحامِضِ:

تَرْعى الغَضَى مِنَ جانِبَيْ مُشَفِّقِ

غِبًّا ومَنْ يَرْعَ الحُموضَ يَغْفِقِ

الجوهري، الصحاح. تح: عطار، ج: 3، ص: 1072.— المصدر
1.4معنى من معاني «غَفَقَ»

وــــــــــ: نامَ قَليلًا ثُمَّ اسْتَيْقَظَ ثُمَّ نامَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال كُراعُ النَّمْلِ (ت: 309هـ=921م):

"غَفَقَ الرَّجُلُ غَفْقًا: إذا نامَ قَليلًا ثُمَّ اسْتَيْقَظَ ثُمَّ نامَ".

كراع النمل، المنتخب. تح: العمري، ص: 265.— المصدر
1.5معنى من معاني «غَفَقَ»

وــــــــــ مِنَ اللَّيلِ غَفْقَةً: نامَ نَوْمةً.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"يُقالُ: غَفَقْنا غَفْقَةً مِنَ اللَّيْلِ: إذا نِمْنا نَوْمَةً ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مجمل اللغة. تح: زهير عبد المحسن، ص: 682.— المصدر
1.6معنى من معاني «غَفَقَ»

وــــــــــ الإبِلَ: أَسْرَعَ إيرادَها ثُمَّ كَرَّرَ ذَلِكَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"غَفَقَ إبِلَهُ، وذَلِكَ إذا ‌أَسْرَعَ ‌إيرادَها ثُمَّ كَرَّرَ ذَلِكَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مقاييس اللغة. تح: هارون، ج: 4، ص: 385.— المصدر
1.7معنى من معاني «غَفَقَ»

وــــــــــ الحِمارُ الأتانَ: أَتاها مَرّةً بعْدَ مَرّةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"وغَفَقَ الحِمارُ الأَتانَ: إذا أَتاها مَرَّةً بَعْدَ مَرَّةٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مقاييس اللغة. تح: هارون، ج: 4، ص: 385.— المصدر
1.8معنى من معاني «غَفَقَ»

وــــــــــ فُلانًا بالسَّوْطِ ونَحْوِهِ: ضَرَبَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال سَلَمَةُ بْنُ الأكْوَعِ (ت: 74هـ=694م):

"مَرَّ بي عُمَرُ بنُ الخَطّابِ وأَنا قاعِدٌ في السُّوقِ، وهُوَ مارٌّ لِحاجَةٍ لَهُ، مَعَهُ الدِّرَّةُ، فقالَ هَكَذا يا سَلَمةُ عَنِ الطَّريقِ، فغَفَقَنِي بها فَما أَصابَ إلّا طَرَفُها ثَوْبي".

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 3، ص: 70.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216هـ=831م):

"غَفَقْتُهُ بِالسَّوْطِ، أَغْفِقُهُ، ومَتَنْتُهُ بِالسَّوطِ أَمْتِنُهُ وهُوَ أشَدُّ مِنَ الغَفْقِ".

القاسم بن سلام، السلاح. تح: الضامن، ص: 35.— المصدر
02 —

غَفَّقَ

فعل
2.1معنى من معاني «غَفَّقَ»

* غَفَّقَ فُلانٌ: نامَ وهُوَ يَسْمَعُ حَديثَ القَوْمِ ولا يُحَقِّقُهُ. 

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216هـ=831م):

"‌التَّغْفيقُ النَّوْمُ، وأنتَ تَسْمَعُ حَديثَ القَوْمِ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
2.2معنى من معاني «غَفَّقَ»

وــــــــــ السَّليمَ: عالَجَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال مُلَيحٌ الهُذَلِيُّ (ت: 50هـ=670م) يَصِفُ وُحُوشَ مَفازَةٍ:

وداويّةٍ مَلْساءَ تُمْسي سِهامُها
بِها مِثْلَ عُوّادِ السَّليمِ الـمُغَفَّقِ

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216هـ=831م):

"يُقالُ: غَفِّقُوا السَّليمَ تَغْفيقًا؛ أيْ: عالِجوهُ وسَهِّنوهُ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
03 —

تَغَفَّقَ

فعل
3.1معنى من معاني «تَغَفَّقَ»

* تَغَفَّقَ فُلانٌ الشَّرابَ: أَكْثَرَ مِنْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ الأعْرابِيِّ (ت: 231هـ=845م):

"وإذا أكْثَرَ الشُّرْبَ، فقَدْ تَغَفَّقَ".

الجوهري، الصحاح. تح: عطار، ج: 4، ص: 1537.— المصدر
3.2معنى من معاني «تَغَفَّقَ»

وــــــــــ الماءَ: شَرِبَهُ ساعَةً بَعْدَ أُخْرى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيدَه (ت: 458هـ=1066م):

"‌وتَغَفَّقَ الشَّرابَ: شَرِبَهُ ساعَةً بَعْدَ أُخْرى".

ابن سيده، المحكم. تح: هنداوي، ج: 5، ص: 387.— المصدر
04 —

اغْتَفَقَ

فعل
4.1معنى من معاني «اغْتَفَقَ»

* اغْتَفَقَ الشَّيءُ بِالشَّيءِ: أَحاطَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانِيُّ (ت: 650هـ=1252م):

"كُلُّ شَيْءٍ أَحاطَ بشَيْءٍ، فقَدِ ‌اغْتَفَقَ بِهِ".

الصغاني، التكملة. تح: الأبياري، ج: 5، ص: 130.— المصدر

غَيْفَقَ

فعل
5.1معنى من معاني «غَيْفَقَ»

* غَيْفَقَ فُلانٌ الماءَ ونَحْوَهُ: أراقَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيبانيُّ (ت: 206هـ=821م):

"‌الغَيْفَقَةُ: الإهْراقُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
06 —

غَفْق

اسم
6.1معنى من معاني «غَفْق»

* الغَفْقُ: الهُجومُ على الشَّيْءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الغَفْقُ: الهُجومُ على الشَّيْءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 356.— المصدر
6.2معنى من معاني «غَفْق»

وــــــــــ: الإيابُ مِنَ الغَيْبَةِ فَجْأَةً.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الغَفْقُ: (...)، الإيابُ من الغَيْبِ فَجْأَةً ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 356.— المصدر
6.3معنى من معاني «غَفْق»

وــــــــــ: الغَبَرَةُ أَو القَتَمَةُ تَكُونُ في أَقْطارِ السَّماءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"والغَفْقُ: الغَبَرَةُ أو القَتَمَةُ تَكونُ في أَقْطارِ السَّماءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الاشتقاق. تح: هارون، ص: 485.— المصدر
6.4معنى من معاني «غَفْق»

وــــــــــ: الـمطَرُ الخَفيفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"والغَفْقُ: مَطَرٌ لَيْسَ بالشَّديدِ".

ابن فارس، مجمل اللغة. تح: زهير عبد المحسن، ص: 682.— المصدر
07 —

تَغْفيق

اسم
7.1معنى من معاني «تَغْفيق»

* التَّغْفيقُ: نوْمٌ في أرَقٍ.

ع

”قالَ الأزْهَرِيُّ (ت: 370هـ=981م):

الأزهري، تهذيب اللغة. تح: مرعب، ج: 8، ص: 38.— المصدر
08 —

مَغْفِق

اسم
8.1معنى من معاني «مَغْفِق»

* الـمَغْفِقُ: الـمَرْجِعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال رُؤْبَةُ (ت: 145هـ=762م) يَشْتَكي:

مِنْ بُعْدِ مَغْزايَ وبُعْدِ ‌الـمَغْفِقِ

[قال الجوهريُّ: "الـمَغْفِقُ: الـمَرْجِعُ"]
الجوهري، الصحاح. تح: عطار، ج: 4، ص: 1537.— المصدر
09 —

مُنْغَفَق

اسم
9.1معنى من معاني «مُنْغَفَق»

* الـمُنْغَفَقُ: الـمُنْعَطَفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”وبِهِ فُسِّرَ قَوْلُ رُؤْبَةَ (ت: 145هـ=762م) يَصِفُ هالِكًا:

حَتَّى تَرَدَّى أَرْبَعٌ في الـمُنْغَفَقْ

الجوهري، الصحاح. تح: عطار، ج: 4، ص: 1537.— المصدر
9.2معنى من معاني «مُنْغَفَق»

وــــــــــ: الـمُنْصَرَفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الجَوْهَريّ (ت: 393هـ=1003م):

"والـمُنْغَفَقُ: الـمُنصَرَفُ".

الجوهري، الصحاح. تح: عطار، ج: 4، ص: 1537.— المصدر
