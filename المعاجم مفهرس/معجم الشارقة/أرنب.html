أرنب
مُؤَرْنَب(اسم) مُرْنِب(اسم) أَرْنَب(اسم) أَرْنَبانِيّ(اسم) أَرْنَبِيّة(اسم) مَرْنَبابِيّ(اسم) أَرْنَبةٌ(اسم)
01 —

مُؤَرْنَب

اسم
1.1معنى من معاني «مُؤَرْنَب»

* المُؤَرْنَبُ، أَوْ المُرَنَّبُ مِنَ الأَكْسِيةِ: ما كانَ غَزْلُهُ مَخْلوطًا بِوَبَرِ الأَرْنَبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ = 787م):

"وأمّا المُؤَرْنَبُ؛ فالذي يُخْلَطُ غَزْلُهُ بِوَبَرِ الأَرْنَبِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ.)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 268.— المصدر
1.2معنى من معاني «مُؤَرْنَب»

وــــــــــ الأَرْضُ المُؤَرْنِبةُ: الكَثيرةُ الأَرانِبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِمُ بْنُ سَلّامٍ الهَرَويُّ (ت: 224هـ = 838م):

"أَرْضٌ مُؤَرْنِبةٌ، مِنَ الأَرانِبِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ.)

الأزهري، التهذيب. تح: الإبياري، ج: 15، ص: 212.— المصدر
02 —

مُرْنِب

اسم
2.1معنى من معاني «مُرْنِب»

* المُرْنِبةُ مِنَ الأَراضي: الكَثيرةُ الأَرانِبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللَّيْثُ بْنُ المُظَفَّرِ (ت: 193هـ = 808م):

"أَرْضٌ مُرْنِبةٌ: كَثيرةُ الأَرانِبِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ.)

الأزهري، التهذيب. تح: مخيمر، ج11، ص: 189.— المصدر
03 —

أَرْنَب

اسم
3.1معنى من معاني «أَرْنَب»

* الأَرْنَبُ: حَيَوانٌ بَرّيٌّ، عاشِبٌ، يُفيدُ الإنْسانُ مِنْ لَحْمِهِ وفَرْوِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق هـ = 543م):

أَيا هِنْدُ لا تَنْكِحي بوهةً
عَلَيْهِ عَقيقَتُهُ، أَحْسَبا

مُرَسَّعةٌ بَيْنَ أَرْساغِهِ
بِهِ عَسَمٌ يَبْتَغي أَرْنَبا

أحمد التام الأحمدي، شعراء حمير، مجمع العربية السعيدة، صنعاء. ج2، ص: 26.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبيُّ ﷺ (ت: 11هـ = 632م):

«... فَقالَ: يا بْنَ حَوالةَ! كَيْفَ تَفْعَلُ في فِتْنةٍ تَخْرُجُ في أَطْرافِ الأَرْضِ كَأَنَّها صَياصي بَقَرٍ؟ قُلْتُ: لا أَدْري، ما خارَ اللَّهُ لي ورَسولُهُ، قالَ: وكَيْفَ تَفْعَلُ في أُخْرى تخْرُجُ بَعْدَها، كَأَنَّها الأولى، فيها انْتِفاجةُ أرْنَبٍ؟».

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 28، ص: 214.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142هـ = 759م) في قِصّةِ الأَرْنَبِ والأَسَدِ:

"... فَقالَ الأَسَدُ: انْطَلِقي مَعي فَأَريني مَوْضِعَ هذا الأَسَدِ، فانْطَلَقَتِ الأَرْنَبُ بِهِ إلى جُبٍّ فيهِ ماءٌ غامِرٌ صافٍ، فاطَّلَعَتْ فيهِ، وقالَتْ: هذا المَكانُ".

آثار ابن المقفع، ج: 1، ص: 94.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ العَديمِ (ت: 660هـ = 1262م):

"... تَتْركونَهُ يَقْتُلُ تاجَ الدَّوْلةِ، ويَأْخُذُ الأَمْوالَ، ويَمْضي، فَصاحَ بِالتُّركيّةِ -يَعْني- الأَرْنَبَ الأَرْنَبَ، فَضَرَبوهُ بالسِّهامِ؛ فَقَتَلوهُ".

ابن العديم، بغية الطلب. تح: زكار، ج: 1، ص: 1985.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ حَسَن بن جَمال الرّيكي (ت: 1233هـ = 1817م):

"...وهِيَ دائِمةُ العُشْبِ، وكَثيرةُ الصَّيْدِ مِنَ الغِزْلانِ والأَرانِبِ وطَيْرِ الحُبارِ والنَّعامِ".

الريكي، لمع الشهاب. تح: عبد الله العثيمين، ج: 1، ص: 255.— المصدر
3.2معنى من معاني «أَرْنَب»

وــــــــــ: الكَثيبُ المُلْتَوي مِنَ الرَّمْلِ (ج) أَرانِبُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المُخَبَّلُ السَّعْديُّ (ت: 12هـ = 633م):

فَإنْ أَكُ لاقَيْتُ الدَّهاريرَ مِنْهُما
فَقَدْ أَفْنَيا نُعْمانَ قَبْلُ، وتُبَّعا

كَما قالَ سَعْدٌ، إذْ يَقودُ بِهِ ابْنُهُ
كَبِرْتُ فَجَنِّبْني الأَرانِبَ صَعْصَعا

عبد الحميد المعيني، شعر بني تميم في العصر الجاهلي، ج: 1، ص: 135.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"والأرانبُ: أحْقافٌ مِنَ الرَّمْلِ مُنْحَنِيةٌ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ.)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 233.— المصدر
3.3معنى من معاني «أَرْنَب»

وــــــــــ: ضَرْبٌ مِنَ الحُليِّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبةُ بْنُ العَجّاجِ (ت: 145هـ = 762م):

لِذي الهَوى تَبْلُ بِغَيْرِ تَبْلِ

لِما اكْتَسَتْ مِنْ ضَرْبِ كُلِّ شَكْلِ

صُفْرًا وخُضْرًا كاخْضِرارِ البَقْلِ

وعُلِّقَتْ مِنْ أَرْنَبِ ونَخْلِ

ديوان رؤبة. تح: وليم بن الورد، ج: 3، ص: 130.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ = 1311م):

"والأَرْنَبُ: ضَرْبٌ مِنَ الحُليِّ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، اللسان. تح: أمين عبد الوهاب، ج: 5، ص: 331.— المصدر
3.4معنى من معاني «أَرْنَب»

وــــــــــ: المَلِكُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (321هـ = 932م، مِنْ شَواهِدِ ابْنِ دُرَيْدٍ):

اللهُ يَعْلَمُ، والأَقْوامُ قَدْ عَلِموا
أَنْ لَمْ يَكُنْ لأَبيكُمْ أَرْنَبُ السَّلَفِ

ابن دريد، الجمهرة. تح: بعلبكي، ج: 1، ص: 523.— المصدر
04 —

أَرْنَبانِيّ

اسم
4.1معنى من معاني «أَرْنَبانِيّ»

* الأَرْنَبانِيُّ: الخَزُّ الأَدْكَنُ الشَّديدُ الدُّكْنةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"والأَرْنَبانيُّ: الخَزُّ الأَدْكَنُ الشَّديدُ الدُّكْنةِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 233.— المصدر
05 —

أَرْنَبِيّة

اسم
5.1معنى من معاني «أَرْنَبِيّة»

* الأَرْنَبِيّةُ: أُكْلةٌ شاميّةٌ تُصْنَعُ مِنَ اللَّبَنِ والطَّحينِ وغَيْرِهِما.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَحَمّدُ بْنُ الحَسَنِ البَغْداديُّ (ت: 636هـ = 1239م):

"الأَرْنَبيّةُ: تُعْرَفُ عِنْدَ أَهْلِ الشّامِ بالفاخِتيّةِ، وتُصْنَعُ مِنَ اللَّبَنِ، والطَّحينةِ، وعَصيرِ الرُّمّانِ، والحِمَّصِ، والقَرْنَبيطِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

محمد بن الحسن البغدادي، كتاب الطبيخ. تح: هنداوي، ج: 1 ص: 81.— المصدر
06 —

مَرْنَبابِيّ

اسم
6.1معنى من معاني «مَرْنَبابِيّ»

* كِساءٌ مَرْنَبانِيٌّ: لَوْنُهُ لَوْنُ الأَرْنَبِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ النّابِغةُ الذُّبْيانيُّ (ت: 18ق هـ = 605م):

تَراهُنَّ خَلْفَ القَوْمِ خُزْرًا عُيونُها
جُلوسَ الشُّيوُخِ في ثِيابِ المَرانِبِ

ديوان النابغة الذبياني. تح: أبو الفضل، ج: 1، ص: 43.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مالِكُ بْنُ الرّيبِ المازِنيُّ (ت: 60هـ = 680م):

إذا مُتُّ فاعْتادي القُبورَ، وسَلِّمي
على الرَّمْسِ، أُسْقيتِ السَّحابَ الغَوادِيا

على جَدَثٍ قَدْ جَرَّتِ الرّيحُ فَوْقَهُ
تُرابًا كسَحْقِ المَرْنَبانيِّ هابِيا

ديوان مالك بن الريب. تح: القيسي، ص: 94.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ = 787م):

"... فأَمّا المَرْنَبانيُّ؛ فالذي لَوْنُهُ لَوْنُ الأَرْنَبِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 268.— المصدر
07 —

أَرْنَبةٌ

اسم
7.1معنى من معاني «أَرْنَبةٌ»

* الأَرْنَبةُ: طَرَفُ الأَنْفِ، (ج) أَرانِبُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو قَيْسٍ صَيْفيُّ بْنُ عامِرٍ (ت: 1هـ = 622م):

وأَنْتُمْ، إذا ما حُصِّلَ النّاسُ، جَوْهَرٌ
لَكُمْ سُرّةُ البَطْحاءِ، شُمَّ الأَرانِبِ

ديوان صيفي بن عامر. تح: باجودة، ص: 68.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نواسٍ (ت: 198هـ = 814م) يَصِفُ ثَوْرًا:

يُكَشِّرُ عَنْ نابٍ لَهُ قَرّاصِ

أَرْنَبةً سَوْداءَ كالعَناصي

بِها يُعاطي، وبِها يُعاصي

ديوان أبو نواس. تح: أحمد الغزالي، ص: 641.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أَسْعَدُ بْن إبراهيمَ الشّيبانيُّ (ت: 657هـ = 1259م):

"وكانَتْ لَهُ ناصِيةٌ يَشُدُّها بَيْنَ عَيْنَيْهِ، وكانَ يَضْرِبُ بِلِسانِهِ أَرْنَبةَ أَنْفِهِ، ويُكَنَّى أَبا الوَليدِ".

أسعد الشيباني، المذاكرة في ألقاب الشعراء. شاكر العاشور، ج: 1، ص: 22.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمّد بَهاء الدّين البَيْطارُ (ت: 1328هـ = 1910م):

"ثُمَّ شَمَّرَ عَنْ ساعِدِهِ الأَشَدّ، وضَرَبَ بِلِسانِهِ أَرْنَبةَ أَنْفِهِ".

المفاخرات والمناظرات. تح: الطيان، ص: 62.— المصدر
