جَنادِعُ(اسم) جُنْدُع(اسم) جُنْدُعَة(اسم)
01 —

جَنادِعُ

اسم
1.1معنى من معاني «جَنادِعُ»

*الجَنادِعُ:البَلايا وَالآفاتُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عُبَيْدُ بْنُ عَبْدِ العُزّى السَّلامِيُّ (ت: 82ق هـ=542م) يَفْتَخِرُ بِمكارِمِ أَخْلاقِهِ:

أُعرِضُ عَنْ أَشْياءَ لَوْ شِئْتُ نِلْتُها
حَياءً إِذا ما كان فيها مقادِعُ

وَلا أَدْفَع ابْنَ العَمِّ يَمْشي عَلى شَفًا
وَلَوْ بَلَغَتْني مِنْ ‌أَذاهُ ‌الجنادِعُ

الخالديان محمد وسعيد ابنا هشام، الأشباه والنظائر. تح: السيد محمد يوسف، ج: 1، ص: 76.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في الحَديثِ الشَّريفِ (11هـ=632م) قالَ النَّبِيُّ ﷺ:

«إِنّي أَخافُ عَلَيْكُم الجَنادِعَ وَالمَرَبّاتِ».

الخليل، العين. تح: المخزومي، ج: 2، ص: 316.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"إِنّي أَخافُ عَلَيْكُم الجَنادِعَ وَالمَرَبّاتِ: يَعْني البَلايا والآفاتِ، وَالمَرَبّاتُ: الدّواهي الشَّديدةُ" (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 2، ص: 316.— المصدر
”وَقالَ أَبو العَلاءِ المَعَرِّيُّ (ت: 449هـ=1057م) على لِسانِ الجَمَلِ في رِسَالَةِ "الصّاهِلُ والشّاحِجُ":

"ما أَجْدَرَني بالتَّقْصيرِ عَمّا تَسْأَلُ، ولَكِنّي لا أَحْرِمُك دَأْبًا فيما قُلْتَ، وإِنّما يُكَلِّفُني عَناءً في حاجَتِكَ ما سَبَقَ إِلَيْك بالحِدَّةِ، وفَرَطَ في جَنادِعِ الغِرَّةِ".

أبو العلاء المعرّي، رسالة الصاهل والشّاحج. دار المعارف، ص: 221.— المصدر
1.2معنى من معاني «جَنادِعُ»

وــــــــــ: ما يَسوءُ مِنَ القَوْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وجَنادِعُ الشَّرِّ: أَوائِلُهُ. ومِنَ القَوْلِ ما يَسوءُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: محمد حسن آل ياسين، ج: 2، ص: 229.— المصدر
1.3معنى من معاني «جَنادِعُ»

وــــــــــ مِنْ كُلِّ شَيْءٍ: أَوائِلُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وجَنادِعُ كُلِّ شَيْءٍ: أَوائِلُهُ.

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكّي، ج: 1، ص: 1136.— المصدر
1.4معنى من معاني «جَنادِعُ»

Οوجَنادِعُ الشَّرِّ: أَوائِلُهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُذَيْفَةُ بنُ اليَمانِ (ت: 36ه=657م) في الدَّجَّالِ:

"وأَيْمُ اللهِ؛ لا يَخْرُجُ حَتّى يَكونَ خُروجُهُ أَحَبَّ إلى الـمَرْءِ الـمُسْلِمِ مِنْ بَرْدَةِ الشَّرابِ عَلى الظَّمَإِ، فقالَ عَبْدُ اللهِ: لِمَ؟ للهِ أَبوكَ، فقَالَ حُذَيْفَةُ: مِنْ شِدَّةِ البَلاءِ، وجَنادِعِ الشَّرِّ".

ابن أبي شيبة، الكتاب المصنف. تح: كمال يوسف الحوت، ج: 7، ص: 495.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216ه=831م):

"جاءَتْ جَنادِعُهُ؛ يَعْني حَوادِثَ الدَّهْرِ، وأَوائِلَ شَرِّهِ".

أبو عبيد القاسم بن سلام، الأمثال. تح: عبد المجيد قطامش، ص: 335.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"رَأَيْتُ جَنادِعَ الشَّرِّ؛ أَيْ أَوائِلَهُ، الواحِدَةُ: جُنْدُعَةٌ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، لسان العرب. تح: أمين محمد عبد الوهاب، وآخر، ج: 2، ص: 209.— المصدر
1.5معنى من معاني «جَنادِعُ»

Οوجَنادِعُ الخَمْرِ:الحَبَبُ الّذي يَتَراءى مِنْها عِنْدَ المَزْجِ. الواحِدَةُ جُنْدُعٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى (ت: 7ه=628م) يَصِفُ الخَمْرَ:

وشَمولٍ تَحْسِبُ العَيْنُ إذا
صُفِّقَت جُنْدُعَها نَوْرَ الذُّبَحْ

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 273.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعيُّ (ت: 216ه=831م):

"وجَنادِعُ الخَمْرِ: ما يَنْزو مِنْها إذا مُزِجَت". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، كتاب الألفاظ. تح: فخر الدين قباوة، ص: 272.— المصدر
1.6معنى من معاني «جَنادِعُ»

Οوجَنادِعُ الضَّبِّ:دَوابٌّ أَصْغَرُ مِنَ القُرادِ، تَكونُ عِنْدَ جُحْرِهِ، فإذا بَدَتْ عَلِمَ الحافِرُ أَنَّهُ دَنا مِنْ قَعْرِ الجُحْرِ، أَوْ أَنَّ الضَّبَّ خارِجٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مُؤَرِّجٌ السَّدوسِيُّ (ت: 195ه=811م):

"والجَنادِعُ: دَوابُّ أَمْثالِ الجَداجِدِ، تَكُونُ قَريبًا مِنَ الضَّبِّ، إِذا خَرَجَتْ تِلْكَ، فهُوَ عِنْدَ دُنُوِّهِ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السدوسي، كتاب الأمثال. تح: رمضان عبد التواب، ص: 62.— المصدر
1.7معنى من معاني «جَنادِعُ»

Οوالقَوْمُ جَنادِعُ: إذا كانوا فِرَقًا لا يَجْتَمِعُ رَأْيُهُمْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْريُّ (ت: 90ه=709م) يَفْتَخِرُ:

وأَمّا مُصابُ الغادِياتِ فإِنَّنا
على الهَوْلِ نَرْعاهُ ولَوْ أَنْ نُقارِعا

بِحَيٍّ نُمَيْريٍّ عَلَيْهِ مَهابَةٌ
جَميعٍ إذا كانَ اللِّئامُ جَنادِعا

ديوان الرّاعي النّميريّ. تح: واضح الصّمد، ص: 175.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزْهَرِيُّ (ت: 370ه=981م):

"القَوْمُ جَنادِعُ: إِذا كانوا فِرَقًا لا يَجْتَمِعُ رَأْيُهُمْ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح:عبد الله درويش، ج: 3، ص: 314.— المصدر

Οوذاتُ الجَنادِعِ:الدّاهِيَةُ.

ع

02 —

جُنْدُع

اسم
2.1معنى من معاني «جُنْدُع»

*الجُنْدَعُ، وَالجُنْدُعُ: جُنْدُبٌ أَسْوَدُ لَهُ قَرْنانِ طَوِيلانِ، وهُوَ أَضْخَمُ الجَنادِبِ. (ج) جَنادِعُ. (انْظُرْ: ج د ع).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والجُنْدُعُ: الجُخْدُبُ، وهُوَ شِبْهُ الجَرادَةِ إلَّا أَنَّهُ أَضْخَمُ مِنَ الجَرادَةِ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 1، ص: 266.— المصدر
2.2معنى من معاني «جُنْدُع»

وــــــــــ: الحَنَشُ (الحَيَّةُ).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ الله بنُ عَبَّاسٍ ﷺ (ت: 68ه=688م):

"لَـمّا أَرْسَلَ اللهُ ﷺ عَلى عادٍ الريحَ جَعَلوا يَهْرُبونَ مِنْها، فَتَلَقَّتْهُمُ الجَنادِعُ؛ وهِيَ الحَيّاتُ".

ابن أبي الدنيا، رسائل في الزهد والرقائق والورع. المنتدى الإسلامي، ج: 2، ص: 352.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَوْهَرِيُّ (ت: 350ه=960م):

"والجَنادِعُ: الأَحْناشُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصّحاح. تح: أحمد عبد الغفور عطّار، ص: 711.— المصدر
2.3معنى من معاني «جُنْدُع»

وــــــــــ: الدّاهِيَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وذاتُ الجَنادِعِ: الدّاهِيَةُ، وتُسَمّى الدَّواهي الجَنادِعُ أَيْضًا". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكّي، ج:2، ص: 1136.— المصدر
2.4معنى من معاني «جُنْدُع»

وــــــــــ مِنَ النّاسِ: القَصيرُ. (ج) جَنادِعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244ه=858م):

"والجُنْدُعُ أَيْضًا: القَصيرُ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السّكّيت، كتاب الألفاظ. تح: فخر الدين قباوة، ص: 166.— المصدر
03 —

جُنْدُعَة

اسم
3.1معنى من معاني «جُنْدُعَة»

*الجُنْدُعَةُ:نُفَّاخَةٌ تَرْتَفِعُ فَوْقَ الماءِ مِنَ الـمَطَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"والجُنْدُعَةُ: نُفّاخَةٌ تَرْتَفِعُ فَوْقَ الماءِ مِنَ الـمَطَرِ، وجَمْعُها جَنادِعُ".

ابن عباد، المحيط في اللغة. تح: محمد حسن آل ياسين، ج:2، ص: 229.— المصدر
3.2معنى من معاني «جُنْدُعَة»

وــــــــــ مِنَ النّاسِ: الذي لا خَيْرَ فيهِ ولا غَناءَ عِنْدَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 310ه=921م):

"وجَنادِعُ الرِّجالِ: مَنْ لا خَيْر فِيهِ، ولا غَناءَ عِنْدَهُ؛ الواحِدُ جُنْدُعَةٌ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع النّمل، المنتخب من كلام العرب. تح: محمد العمري، ص: 202.— المصدر
