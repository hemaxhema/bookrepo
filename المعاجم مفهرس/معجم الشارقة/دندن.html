دَنْدَنَ(فعل) دِنْدان(اسم) دِنْدِن(اسم) دَنْدَنَة(اسم) دَنين(اسم)
المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (د ن د ن):

في العِبريّةِ: وَرَدَ الاسْمُ: דַּנְדַּנָּה (دَندَنّا) danadannā بِمَعْنى (نَباتٌ مِنَ الفَصيلةِ السَّرخسيّةِ قيلَ: هُو حَشيشةُ الطِّحالِ) ويُوافِقُ الاسْمَ العَرَبيَّ (دِنْدِن) مَبْنًى ومَعْنًى، ومِنْ مَجيئهِ بِمَعْنى (دِنْدِنٌ) ما وَرَدَ في المشنا: "עִקַּר הַלּוּף שׁוֹטֶה‏, וְעִקַּר הַדַּנְדַּנָּה"، أيْ: "جُذُورُ اللّوفِ البَرِّيِّ، وجُذُورُ الدِّنْدِنِ".

(שביעית: ז', ב')

01 —

دَنْدَنَ

فعل
1.1معنى من معاني «دَنْدَنَ»

* دَنْدَنَ فُلانٌ: تَكَلَّمَ بِصَوْتٍ خَفِيٍّ يُسْمَعُ ولايُفْهَمُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (11هـ=632م) يُبَيِّنُ أنَّهُ يَسْألُ اللهَ الجَنَّةَ، ويَسْتَعيذُ مِنَ النّارِ في دَنْدَنَتِهِ بَعْدَ تَشَهُّدِهِ:

«حَوْلَهَا نُدَنْدِنُ».

مسند أحمد، تح: شعيب الأرناؤوط وآخرين، ج: 25، ص: 234.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (158هـ=774م) مِنْ قَصيدَةٍ يَمْدَحُ فيها أَبا جَعْفَرٍ المَنْصورَ:

والعُودُ دَنْدَنْ دَنْدَنٌ
والطَّبْلُ طَبْطَبْطَبَ لي

إبراهيم النّجّار، شعراء عبّاسيّون منسيّون، ج: 4، ص: 338.— المصدر
”قالَ ابنُ الأثِيرِ (637هـ=1239م) يَتَحَدَّثُ عَنِ المَعاني المُبْتَدَعَةِ عِنْدَ الشُّعَراءِ:

"وهَذا مَعْنًى دَنْدَنَ حَوْلَهُ الشُّعراءُ".

ابن الأثير، المثل السائر، تح: الحوفي، وآخر، ج: 2، ص: 23.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابن قيِّم الجَوزية (ت: 751هـ=1350م) يُبَيِّنُ أنَّ حَقيقَةَ الفَقْرِ تَوَجُّهُ العَبْدِ بجَميعِ أحْوالِهِ إلى اللهِ:

"وهُوَ الغايَةُ التي شَمَّرَ إلَيْها السّالِكونَ، والعِلْمُ الذي أمَّهُ العابِدونَ، ودَنْدَنَ حَوْلَهُ العارِفونَ".

ابن قيم الجوزية، طريق الهجرتين، تح: محمد أجمل، ص: 35.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عامر ياسين (ت: 1418هـ=1997م) يُبَيِّنُ مَنْهَجَ الإمامِ النَّوَوِيِّ في العَمَلِ بالحَديثِ الضَّعيفِ في صالِحِ الأعْمالِ:

"وهذا أمْرٌ قَدْ دَنْدَنَ النَّوَوِيُّ حَوْلَهُ كَثيرًا".

النووي، الأذكار النووية، تح: عامر ياسين، ص: 31.— المصدر
1.2معنى من معاني «دَنْدَنَ»

وــــــــــ النَّحْلُ والذُّبابُ وغيرُهما منَ الحشراتِ، والأشياءِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ برثنُ الصَّموتيُّ (ت: 1ق.هـ=619م) يَهْجو:

فَمِنْهُنَّ النَّمامَةُ أَنْتَ فِيها
تُدَنْدِنُ مِثْلَ دَنْدَنَةِ الذُّبابِ

المستدرك في شعر بني عامر، تح: الوصيفي، ص: 206.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمِر (ت: 255هـ=869م) :

"طَنْطَنَ طَنْطَنَةً، ودَنْدَن دَنْدَنةً: بِمَعْنًى واحِدٍ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: يعقوب عبد النبي، ج: 14، ص: 70.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ السَّيّاب (ت: 1384هـ=1964م) يَتَغَزَّلُ:

"دَنْدَنَ كالأَجْراسِ في
ها كالزَّنابِقِ إِذْ تَعوم".

بدر السياب، شناشيل ابنة الجلبي، ص: 23.— المصدر
1.3معنى من معاني «دَنْدَنَ»

وــــــــــ فُلانٌ: تَرَدَّدَ مَجيئًا وذَهابًا في مَكانٍ واحِدٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَريُّ (ت: 538هـ=1144م):

"دَنْدَنَ الرَّجُلُ: إذا اخْتَلَفَ في مَكانٍ واحِدٍ مَجيئًا وذَهابًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق، تح: البجاوي، وآخر، ج: 1، ص: 441.— المصدر
02 —

دِنْدان

اسم
2.1معنى من معاني «دِنْدان»

* الدِّنْدانُ: الصَّوتُ والكَلامُ الذي يُسْمَعُ ولا يُفْهَمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رؤبةُ (ت: 145هـ=762م) يَصِفُ بِنْتًا تَشْتَكي لِأَبيها عَدَمَ النَّوْمِ:

يا أبَــــــــــــــــتا أرَّقَــــــــــــــــــــــني القِذَانُ

فالنَّـــــــــــــوْمُ لا تَطْعَمُهُ العَيْنانُ

مِنْ وَخْزِ بَرْغوثٍ لَهُ أَسْنانُ

ولِلبَعــــــــــــــــوضِ فَــــــــوقَهُ دِنْدانُ

ديوان رؤبة بن العجاج، تح: وليم البروسيي، ص: 186.— المصدر
03 —

دِنْدِن

اسم
3.1معنى من معاني «دِنْدِن»

* الدِّنْدِنُ: مابَلِيَ واسْوَدَّ مِنْ أُصولِ النَّباتِ والشَّجَرِ.(ج) دَنادِنُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ خُفافُ بنُ ندبةَ (ت: 20هـ=641م) يَفْخَرُ بِكَرَمِ قَوْمِهِ:

أَنْمي إلى مَجْدِ أجْدادٍ لَهُمْ عَدَدٌ
مُذَلِّلينَ لِوَطْءِ الحَقِّ أزْوالِ

والمُطْعِمينَ إذا هَبَّتْ شآمِيَةٌ
تُذْري الهَشيمَ وثَمَّ الدِّنْدِنُ البالي

محمّد بن المبارك، منتهى الطّلب ، تح: طريفي، ج: 1، ص: 136.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصمعيُّ (ت: 216هـ=831م) يَصِفُ أحْوالَ النَّباتِ:

"فَإذا اسْوَدَّ مِنَ القِدَمِ فهُوَ الدِّنْدِنُ".

ابن سلام، الغريب المصنف، تح: داودي، ج: 1، ص: 479.— المصدر
3.2معنى من معاني «دِنْدِن»

وــــــــــ: نَبْتٌ يقالُ له الصِّلِّيّانُ المُحيلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الدِّنْدِنُ:الصِّلِّيانُ المُحيلُ، تَمِيميّةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم، تح: هنداوي، ج: 9، ص: 274.— المصدر
3.3معنى من معاني «دِنْدِن»

وــــــــــ: طَرَفُ الثَّوْبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385هـ=995م) :

"الدَّنادِنُ: أَطْرافُ الثَّوْبِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في عَصْرِ الدّولِ والإِماراتِ)

ابن عباد، المحيط، تح:آل ياسين، ج: 9، ص: 262.— المصدر
3.4معنى من معاني «دِنْدِن»

وــــــــــ: صَوْتُ الذُّبابِ والنَّحْلِ والزَّنابيرِ ونَحْوِهِا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458ه=1066م):

"الدَّنِينُ والدَّنْدَنَةُ والدِّنْدِنُ: صَوْتُ الذُّباب والزَّنَابِير ونحوِهما من هَيْنَمَة الكَلام الَّذِي لَا يُفْهَم". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص، تح: خليل جفال، ج: 2، ص: 359.— المصدر
04 —

دَنْدَنَة

اسم
4.1معنى من معاني «دَنْدَنَة»

* الدَّنْدَنَةُ: الصَّوتُ والكَلامُ الذي يُسْمَعُ ولا يُفهَمُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ برثنُ الصَّموتيُّ (ت: 1ق.هـ=619م) يَهْجو:

فَمِنْهُنَّ النَّمّامَةُ أَنْتَ فيها
تُدَنْدِنُ مِثْلَ دَنْدَنَةِ الذُّبابِ

المستدرك في شعر بني عامر، تح: عبد الرحمن الوصيفي، ص: 206.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (11هـ=632م) يُبَيِّنُ ما يَقولُ في كَلامِهِ الخَفِيِّ:

«وهَلْ تَصيرُ دَنْدَنَتي، ودَنْدَنَةُ مُعاذٍ إلّا أنْ نَسْألَ اللَّهَ الجَنَّةَ، ونَعُوذَ بِهِ مِنَ النّارِ؟».

مسند أحمد، تح: شعيب الأرناؤوط وآخرين، ج: 34، ص: 307.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليلُ بنُ أحمدَ (ت: 170هـ=787م):

"الدَّنْدَنةُ مِنْ هَيْنَمةِ الكَلامِ الّذي لا يُفهَمُ".

الخليل، العين، تح: المخزومي وآخر، ج: 8، ص: 10.— المصدر
”وقالَ الشّاعِرُ (مِنْ شَواهِدِ اللَّيْثِ بْنِ المُظَفَّرِ (180هـ=796م):

كَدَنْدَنَةِ النَّحْلِ في الخَشْرَمِ

الأزهري، تهذيب اللغة، تح: يعقوب عبد النبي، ج: 14، ص: 70.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ رَجَبٍ (ت: 795هـ=1393م) يُبَيِّنُ أنَّ مَحَبَّةَ الأشْياءِ الطَّبيعِيَّةِ لا تَتَنافى مَعَ مَحَبَّةِ اللهِ إذا لَمْ يُؤْثِرْها الشَّخصُ على أمْرِ اللهِ:

"إنَّ هَذِهِ الأقْوالَ كُلَّها فيها دَنْدَنَةٌ على ذِكْرِ المَحَبَّةِ".

ابن رجب الحنبلي، شرح رسالة كلمة الإخلاص، تح: ياسر العسكر، ج: 1، ص: 112.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحمَّدٌ الشّنْقِيطِيُّ (ت: 1302هـ=1885م):

"والمُخْتَلَفُ فيهِ مِنَ الغِناءِ ما عَرِيَ عَنِ الآلاتِ المُلْهِيَةِ غَيْرِ التَّصْفيقِ بالأكُفِّ ونَحْوِهِ مَبْنِيًّا على الثّاني في النَّغَماتِ بالدَّنْدَنَةِ ونَحْوِها".

محمد الشنقيطي، لوامع الدرر، ج: 2، ص: 332.— المصدر
4.2معنى من معاني «دَنْدَنَة»

وــــــــــ: كَثْرَةُ الشِّكايَةِ.(مو)

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ بطرس البستاني (1300هـ=1883م):

"الدَّنْدَنَةُ: صَوْتُ الذُبابِ...، وعِنْدَ المُوَلَّدينَ: كَثْرَةُ الشِّكايَةِ".

بطرس البستاني، محيط المحيط، ج: 1، ص: 294.— المصدر
