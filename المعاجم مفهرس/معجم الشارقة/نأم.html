نَأَمَ(فعل) ناءَمَ(فعل) تَناءَمَ(فعل) تَنَأَّمَ(فعل) تَنْآم(اسم) نَأْمَة(اسم) نَأْمَة/ نَأَمَة(اسم) نَئيم(اسم)
النظائر الساميّة

احْتَفَظَتِ السّاميّاتُ بهذا الجَذْرِ مِنَ العَرَبيَّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذْرُ (ن أ م):

في الآراميَّةِ؛ ومِنْهُ الفِعْلُ: נְהַם (نْهَم) nəham بِمَعْنى: "زَمْجَرَ/ زَأَرَ"؛ وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَأَمَ) مَبْنًى ومَعْنًى، غَيْرَ أَنَّ الفِعْلَ الآراميَّ خَضَعَ لإِبْدالِ الهَمْزَةِ هاءً؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَأَمَ" ما وَرَدَ في تُرْجومِ يوناثانَ: "כְּאַרְיְוָן נָהֲמִין לְמִקְטַל קְטוֹל"؛ أَيْ: "كَأُسودٍ تَنْئِمُ لقَتْلِ الْفَريسَةِ".

(ת''י, יח': כב, כה)

في العِبْريَّةِ؛ ومِنْهُ الاسْمُ: נְאוּם (نْؤوم) nəʼūm بِمَعْنى: "كَلِمَةٍ، قَوْلٍ"، وهُوَ يُوافِقُ الاسْمَ العَرَبيَّ (نَأْمَةً) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَأْمَةٍ" ما وَرَدَ في سِفْرِ التَّكْوينِ: "נְאֻם יְהוָה כִּי יַעַן אֲשֶׁר ... וְלֹא חָשַׂכְתָּ אֶת בִּנְךָ אֶת יְחִידֶךָ, כִּֽי־בָרֵ֣ךְ אֲבָרֶכְךָ"؛ أَيْ: "نَأْمَةُ الرَّبِّ... لَمْ تَبْخَلْ بِوَلَدِكَ وَحيدِكَ، فَبَرَكَةً أُبارِكُكَ".

(בר' 22: 16, 17)

في السُّرْيانيَّةِ؛ ومِنْهُ الفِعْلُ: ܢܰܗܶܡ (نَهّ۪م) nahhem بِمَعْنى: "زَمْجَرَ/ زَأَرَ/ صَوَّتَ"؛ وهُوَ يَتَّفِقُ في أُصولِهِ معَ أُصولِ الفِعْلِ العَربيِّ (نَأَمَ) مَبْنًى ومَعْنًى، غَيْرَ أَنَّ الفِعْلَ السُّرْيانيَّ خَضَعَ لإِبْدالِ الهَمْزَةِ هاءً. (يُنْظَرُ: ن ه م/ ن ح م).

01 —

نَأَمَ

فعل
1.1معنى من معاني «نَأَمَ»

* نَأَمَ (بِفَتْحِ الهَمْزَةِ) الحَيَوانُ وغَيْرُهُ، يَنْئِمُ ويَنْأَمُ (بِكَسْرِ الهَمْزَةِ وفَتْحِها) نَئيمًا: صَوَّتَ بِصَوْتٍ فيهِ ضَعْفٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أبو دُؤادٍ الإِيادِيّ (ت: 79ق.ه=545م) يَصِفُ طَريقًا:

مُذْكِرٌ تَهْلِكُ الـمَقانِبُ فيهِ
يَنْئِمُ البومُ فيهِ كَالـمَحْزونِ

[الـمُذْكِرُ: لا يَقْطَعُهُ إِلّا الفُحولُ. الـمَقانِبُ: الجُيوشُ]

ديوان أبي دؤاد. تح: أنوار الصالحي، وآخر، ص: 175.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"النَّئيمُ: صَوْتٌ فيهِ ضَعْفٌ، وصَوْتُ الهامِ نَئيمٌ، وصَوْتُ الضَّفادِعِ نَئيمٌ. والفِعْلُ: نَأَمَ يَنْئِمُ نَئيمًا ".

الخليل، العين. تح: المخزومي، وآخر، ج: 8، ص: 388.— المصدر
”وقالَ المُتَنبّي (ت: 354ه=965م) يُهَوِّنُ ما اسْتَعْظَمَهُ غَيْرُهُ مِنْ شِعْرِهِ في رِثاءِ أُمِّهِ:

‌يَسْتَكْبِرونَ أُبَيّاتًا نَأَمْتُ بِها
لا تَحْسُدُنَّ عَلى أَنْ يَنْأَمَ الأَسَدا

ابن جني، الفسر. تح: رضا رجب، ج: 1، ص: 976.— المصدر
1.2معنى من معاني «نَأَمَ»

وــــــــــ فُلانٌ: نَطَقَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244ه=858م):

"وسَكَتَ فُلانٌ فَما نَأَمَ بِحَرْفٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، إصلاح المنطق. تح: شاكر وهارون، ص: 431.— المصدر
”وقالَ المُتَنبّي (ت: 354ه=965م) يُهَوِّنُ ما اسْتَعْظَمَهُ غَيْرُهُ مِنْ شِعْرِهِ في رِثاءِ أُمِّهِ:

‌يَسْتَكْثِرونَ أُبَيّاتًا نَأَمْتُ بِها
لا تَحْسُدُنَّ عَلى أَنْ يَنْئِمَ الأَسَدا

المعري، شرح ديوان المتنبي. تح: عبد المجيد دياب، ج: 2، ص: 269.— المصدر
1.3معنى من معاني «نَأَمَ»

وــــــــــ القَوْسُ: صَوَّتَتْ عِنْدَ خُروجِ السَّهْمِ مِنْها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نَصْرٍ الجَوْهَرِيّ (ت: 393ه=1003م):

"ونَأَمَتِ القَوْسُ نَئيمًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: أحمد عطار، ج: 5، ص: 2038.— المصدر
02 —

ناءَمَ

فعل
2.1معنى من معاني «ناءَمَ»

* ناءَمَ فُلانٌ فُلانًا: كَلَّمَهُ بِصَوْتٍ ضَعيفٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المرّارُ العَدَوِيّ (ت: 100ه=719م):

وأَنْ أَلِجَ البَيْتَ مُدْجى الغِطاءِ
أُنائِمُ في البَيْتِ صَوْتًا ضَعيفَا

الزمخشري، أساس البلاغة. تح: عيون السود، ج: 2، ص: 239.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"ونَأَمْتُ إِلَيْهِ نَأْمَةً: أَيْ كَلَّمْتُهُ تَكْليمَةً. وناءَمْتُهُ مُناءَمَةً: مِنَ النَّئيمِ".

ابن عباد، المحيط. تح: آل ياسين. ج: 10، ص: 412.— المصدر
03 —

تَناءَمَ

فعل
3.1معنى من معاني «تَناءَمَ»

* تَناءَمَ فُلانٌ وغَيْرُهُ: خَفَضَ صَوْتَهُ ولَمْ يُبِنْ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كَعْبُ بْنُ زُهَيْرٍ (ت: 26ه=646م) يَصِفُ نِسْوَةً نَزَلْنَ مَكانًا مُنْخَفِضًا:

وباكَرْنَ جَوْفًا تَنْسُجُ الرّيحُ مَتْنَهُ
تَناءَمُ تَكْليمَ المَجوسِ غَرانِقُهْ

[تَناءَمُ: مُخَفَّفُ تَتَناءَمُ، وفاعِلُهُ هُوَ: غَرانِقُهُ، وهِيَ الكَراكِيُّ أَوْ طُيورُ الماءِ؛ أرادَ أنَّهُنَّ نَزَلْنَ بُكْرَةً إلى مُنْخَفَضٍ تَخْتَلِفُ عَلَيْهِ الرّيحُ، وتَتجاوَبُ طُيورُ الماءِ بِهِ كَما يُكَلِّمُ المَجوسُ بَعْضَهُمْ]
ديوان كعب بن زهير. تح: حنا الحتّي، ص: 145.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السُّكَّرِيُّ (ت: 275ه=888م) يَشْرَحُ بَيْتَ كَعْبٍ:

"وقَوْلُهُ: تَناءَمُ، أَرادَ: تَتَناءَمُ، وهُوَ مَأْخوذٌ مِنَ النَّئيمِ وهُوَ صَوْتٌ ضَعيفٌ ".

ديوان كعب بن زهير. تح: حنا الحتّي، ص: 145.— المصدر

تَنَأَّمَ

فعل
4.1معنى من معاني «تَنَأَّمَ»

* تَنَأَّمَتِ الدِّيَكَةُ: صاحَتْ.

ق

”بِهِ رُوِيَ قَوْلُ عَبْدِ الـمَسيحِ بْنِ عَسَلَةَ (ت: 50ق.ه=573م):

وسَماعِ مُدْجِنَةٍ تُعَلِّلُنا
حَتّى نَؤوبَ تَنَؤُّمَ العُجْمِ

ابن سيده، المحكم. تح: هنداوي. ج: 10، ص: 490.— المصدر
ع

"يُريدُ صياحَ الدّيكَةِ كَأَنَّهُ قالَ وَقْتَ تَنَؤُّمِ العُجْمِ، وإِنَّما سَمّى الدِّيَكَةَ عُجْمًا، لِأَنَّ كُلَّ حَيَوانٍ غَيْرَ الإِنْسانِ أَعْجَمُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي. ج: 10، ص: 490.— المصدر

تَنْآم

اسم
5.1معنى من معاني «تَنْآم»

* التَّنْآمُ: الصَّوْتُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ذو الرُّمَّةِ (ت: 117ه= 735م) يَصِفُ ناقَةً:

لانَتْ عَرِيكَتُهَا مِنْ طُولِ ما سَمِعَتْ
بَيْنَ المَفَاوِزِ تَنْآمَ الصَّدى الغَرِدِ

[لانَتْ عَريكَتُها: سَلُسَ قِيادُها. الـمَفاوِزُ: الـفَلَواتُ البَعيدَةُ. الصَّدى: ذَكَرُ البومِ. الغَرِدُ: المُتَطَرِّبُ]

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نَصْرٍ الباهِلِيّ (ت: 231ه=846م) يَشْرَحُ بَيْتَ ذي الرُّمَّةِ:

"وتَنْآمُ﻿ الصَّدى: صَوْتُ الصَّدى ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو نصر الباهلي، ديوان ذي الرمة. ج: 1، ص: 174.— المصدر

ع

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وصَوْتُ الهامِ نَئيمٌ، وصَوْتُ الضَّفادِعِ نَئيمٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، وآخر، ج: 8، ص: 388.— المصدر
8.2معنى من معاني «نَئيم»

وــــــــــ: صَوْتُ القَوْسِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَوْسُ بْنُ حَجَرٍ (ت: 2ق.ه=620م) يَصِفُ قَوْسًا:

إِذا ما تَعاطَوْها سَمِعْتَ لِصَوْتِها
إِذا أَنْبَضوا عَنْها ‌نَئيمًا وأَزْمَلا

ابن قتيبة، الشعر والشعراء. تح: أحمد شاكر، ص: 204.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رَبيعَةُ الضَّبِّيُّ (ت: 16ه=637م):

وبِالكَفِّ زَوْراءُ حِرْميَّةٌ
مِنَ القُضْبِ تُعْقِبُ عَزْفًا نَئيما

المفضل الضبي، المفضليات. ج: 1، ص: 180.— المصدر

"والنَّأْمَةُ والنَّئيمُ: ‌صَوْتُ ‌القَوْسِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 490.— المصدر

وــــــــــ: الصَّوْتُ الخَفيُّ الضَّعيفُ كَالأَنينِ، وهُوَ بِهاءٍ أَيْضًا.

”قالَ الخَليلُ (ت: 170ه=787م):

"النَّئيمُ: صَوْتٌ فيهِ ضَعْفٌ ".

الخليل، العين. تح: المخزومي، وآخر، ج: 8، ص: 388.— المصدر
”وقالَ أَبو العَلاءِ المَعَرِّيُّ (ت: 449ه=1057م):

"إِنَّ في آثارِ الأَوَّلينَ لَمُعْتَبَرًا، فَلْتَعِظْكَ مَنازِلُ القَوْمِ الذّاهِبينَ لا تَسْمَعُ الأُذُنُ لَهُمْ ‌نَئيمًا".

المعري، الفصول والغايات. ص: 442.— المصدر
8.4معنى من معاني «نَئيم»

وــــــــــ: صَوْتُ الفَهْدِ ونَحْوِهِ، وهُوَ صَوْتٌ شَديدٌ شَبيهٌ بِالسُّعالِ.
