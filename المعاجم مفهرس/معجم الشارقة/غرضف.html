غرضف
غُرْضوف(اسم)
01 —

غُرْضوف

اسم
1.1معنى من معاني «غُرْضوف»

* الغُرْضوفُ: عَظْمٌ رَقيقٌ في رَأْسِ الكَتِفِ مِنْ جِهَةِ الظَّهْرِ. لُغَةٌ في الغُضْروفِ. (ج) غَراضيفُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ سَلْمانُ الفارِسِيُّ (ت: 36هـ=657م) يَحْكي -في قِصَّةِ إسْلامِهِ- ما قالَهُ لَهُ راهِبُ عَمّورِيَّةَ حينَ أوْصاهُ باتِّباعِ النَّبِيِّ ﷺ:

"فإنِ اسْتَطَعْتَ أَنْ تَكونَ مَعَهُ فافْعَلْ، فإنَّهُ على الدِّينِ، وأَمارَةُ ذَلِكَ أَنَّ قَوْمَهُ يَقولونَ: ساحِرٌ مَجْنونٌ كاهِنٌ، وأَنَّهُ يَأْكُلُ الهَدِيَّةَ، ولا يَأْكُلُ الصَّدَقَةَ، وأَنَّ عِنْدَ ‌غُرْضوفِ كَتِفَيْهِ خاتَمَ النُّبُوَّةِ".

الطبراني، المعجم الكبير، تح: السلفي، ج: 6، ص: 232.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"النُّغْضُ: ‌غُرْضوفُ ‌الكَتِفِ".

الخليل، العين، تح: المخزومي، ج: 4، ص: 367.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ الأنْدَلُسِيُّ (ت: 672هـ=1273م):

"والكُتْفُ جَمْعُ أَكْتَفَ، وهُوَ المُجْتَمِعُ الكَتِفَيْنِ، والَّذِي انْفَرَجَتْ ‌غَراضيفُ كَتِفَيْهِ".

ابن مالك، إكمال الإعلام، تح: الغامدي، ج: 2، ص: 542.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَد رِضا العامِلِيُّ (ت: 1372هـ=1953م):

"الجائِفُ: عِرْقٌ يَجْري عَلى العَضُدِ إلى ‌غُرْضوفِ ‌الكَتِفِ".

أحمد رضا، متن اللغة، دار مكتبة الحياة، ج: 1، ص: 605.— المصدر
1.2معنى من معاني «غُرْضوف»

وــــــــــ: نَوْعٌ مَرِنٌ مِنَ النَّسيجِ الضّامِّ، في رُباعِيّاتِ الأَطْرافِ، يَتَحَمَّلُ الضَّغْطَ والاحْتِكاكَ، يُغَطِّي نِهاياتِ العِظامِ الطَّويلَةِ عِنْدَ المَفاصِلِ، ويُكَوِّنُ هَيْكَلَ بَعْضِ الأعْضاءِ كطَرَفِ الأنْفِ وصيوانِ الأُذُنِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=709م) يَصِفُ تَقاسيمَ الوَجْهِ عِنْدَ الضَّحِكِ:

يَضْحَكْنَ عَنْ كَالبَرَدِ المُنْهَمِّ

تَحْتَ غَراضيفِ الأُنوفِ الشُّمِّ

الخليل، العين، تح: المخزومي، ج: 8، ص: 497.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"ولِسانُ الزَّورِ: ‌غُرضوفُهُ".

الخليل، العين، تح: المخزومي، ج: 3، ص: 201.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ هُذَيْلٍ الأنْدَلُسِيُّ (ت: 763هـ=1362م):

"وسِمامُ الفَرَسِ قَصَبُ خَياشيمِهِ الَّتي فيها الغَضاريفُ، ويُقالُ ‌الغَراضيفُ، وهُوَ مِنَ المَقْلوبِ".

الأندلُسيّ، حلية الفُرسان، تح: محمد حسن، ص: 54.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَد رِضا العامِلِيُّ (1372هـ=1953م):

"المُهْرَةُ: واحِدَةُ المُهَرِ، ‌وهِيَ غَراضيفُ ‌الضُّلوعِ، أو مَفاصِلُ مُتَلاحِكَةٌ في الصَّدْرِ".

أحمد رضا، متن اللغة. ج: 5، ص: 700.— المصدر
1.3معنى من معاني «غُرْضوف»

وــــــــــ: ما دَقَّ مِنَ العِظامِ في الجِسْمِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأخْطَلُ (ت: 90هـ=708م) يَصِفُ هُزالَ امْرَأَةٍ:

كَأَنَّ غَراضيفَ اسْتِها حَوْلَ أَيْرِهِ
وحَجْمَ تَراقيها سَكاكينُ جازِرِ

[التَّراقي: عِظامٌ في أَعْلى الصَّدْرِ. والجازِرُ: الجَزّارُ]
شعر الأخطل، تح: قباوة، ص: 435.— المصدر
1.4معنى من معاني «غُرْضوف»

وــــــــــ: كُلُّ عَظْمٍ لَيِّنٍ في الجِسْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"‌الغُرْضُوفُ: كُلُّ عَظْمٍ رَخْصٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي، ج: 8، ص: 491.— المصدر
1.5معنى من معاني «غُرْضوف»

وــــــــــ: عَظْمٌ رَقيقٌ في أَسْفَلِ الصَّدْرِ يُشْرِفُ على رَأْسِ المَعِدَةِ، يُعْرَفُ برَهابَةِ الصَّدْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216هـ=831م):

"وفي ‌الصَّدْرِ ‌الرَّهابَةُ، وهِيَ العَظْمُ الرَّقيقُ المُشْرِفُ على رَأْسِ المَعِدَةِ كأنَّهُ غُرْضوفٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأصمعي، خلق الإنسان، تح: هفنر، ص: 217.— المصدر
1.6معنى من معاني «غُرْضوف»

وــــــــــ: طَرَفُ الضِّلَعِ ومُنْتَهاها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنَّى (ت: 209هـ=825م):

"وغَراضيفُ الأَضْلاعِ أَيْضًا مُنْتَهاهُنَّ في القَصِّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

القالي، البارع، تح: الطعان، ص: 452.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزابادِيُّ (817هـ=1414م):

"الدَّأْيُ والدُّئِيُّ والدِّئِيُّ: فِقَرُ الكاهِلِ والظَّهْرِ، أَوْ ‌غَراضيفُ الصَّدْرِ".

الفيروزابادي، القاموس المحيط، تح: العرقسوسي، ص: 1282.— المصدر
1.7معنى من معاني «غُرْضوف»

وــــــــــ (في الفَرَسِ): طَرَفُ الكَتِفِ مِنْ أَعاليهِ، ما دَقَّ عَنْ صَلابَةِ العَظْمِ. وهُما غُرْضوفانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنَّى (ت: 209هـ=825م) يَذْكُرُ أجْزاءَ الفَرَسِ:

"وفيهِ كَتِفاهُ، وفي كَتِفَيْهِ غُرْضوفاهُما، ويُقالُ لَهُ: الغُضْروفُ، أيْضًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

معمر بن المثنى، الخيل. ص: 26.— المصدر
1.8معنى من معاني «غُرْضوف»

وــــــــــ: الحاجِزُ الَّذي بَيْنَ فَتْحَتَيِ الأَنْفِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأنْصارِيُّ (ت: 215هـ=830م):

"وَيُقَال ‌للحاجِزِ الَّذي بَيْنَ المَنْخِرَيْنِ: ‌غُرَضُوفٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: يعقوب عبد النبي، ج: 14، ص: 315.— المصدر
1.9معنى من معاني «غُرْضوف»

وــــــــــ: إحدى الخَشَبَتَيْنِ اللَّتَيْنِ تُشَدّانِ يَمينًا وشِمالًا بَيْنَ واسِطِ الرَّحْلِ وآخِرَتِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانِيُّ (ت: 650هـ=1252م):

"والغُرْضوفانِ: الخَشَبَتانِ اللَّتانِ تُشَدّانِ يَمينًا وشِمالًا بَيْنَ واسِطِ الرَّحْلِ وآخِرَتِهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، العباب - حرف الفاء، تح: آل ياسين، ص: 467.— المصدر
