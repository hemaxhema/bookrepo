المقدّمة
النُّقوش العربيّة:

وَرَدَ الجَذرُ أ ر م ʾrm ʾrfفي عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبيّةِ، مِنْ بَينِها نَقْشٌ قتبانيٌّ يَعودُ تاريخُهُ إلى القَرْنِ الرّابِعِ قَبلَ الميلادِ (RES 4337A/12)، وَرَدَ فيهِ الفِعْلُ أرم ʾrmأي: يُخَزّنُ، فَجاءَ في النَّقشِ: "أ ر م / و- س ش أ م/ ب- ش م ر"، أي "تُخَزَّنُ (البَضائِعُ) وتُباعُ في (سوقِ) شمرَ".

(CSAI; Beeston 1959: 2-16)

النظائر الساميّة

اِحتَفَظَتِ السَّاميَّاتُ بِهذا الجَذرِ مِنَ العَربيّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذرُ (أ ر م)

في الجِعْزيّةِ؛ ومِنهُ الفِعْلُ አረመ (أَرَّمَ) ’arrama بِمَعْنى "اسْتأصَلَ الحَشائِشَ- صَحَّحَ"؛ وهُوَ يَتَّفِقُ مَعَ نَظيرِهِ العَربيِّ مَبْنًى وفي بَعضِ مَعانيهِ مع تَضعيفِ عَينهِ.

(CODG, P.38)

أَرَمَ

فعل
1.1معنى من معاني «أَرَمَ»

* أَرَمَ (بِفَتْحِ الرّاءِ) فُلانٌ على الشَّيْءِ يَأْرِمُ (بِكَسْرِ الرّاءِ)، أَرْمًا: عَضَّ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الجَوْهَريُّ (ت: 393هـ = 1003م):

"أَرَمَ على الشَّيْءِ يَأْرِمُ بِالكَسْرِ، أيْ: عَضَّ عَلَيْهِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ج: 1، ص: 1860.— المصدر
1.2معنى من معاني «أَرَمَ»

وــــــــــ الشَّيْءَ: شَدَّهُ وأَحْكَمَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال عَبْدُاللهِ بْنُ رَبْعيٍّ الحَذْلَميُّ الفَقْعَسيُّ (ت: 13هـ = 634م)

لِلدَّلْوِ في أَيْديهِمُ سَفْحٌ عَجِلْ

صَقْبانِ مَمْشوقانِ مَأْروما الأُصُلْ

ما تبقى من أراجيز الفقعسي. تح: المعيبد، ج: 1، ص: 77.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال رُؤْبةُ بْنُ العجّاجِ (ت: 145هـ = 762م) يَصِفُ راعِيًا دَرَّتْ لَهُ إبِلُهُ مِنَ اللَّبَنِ ما يُقَوّي عِظامَهُ:

إذا الْتَقَتْ أَرْبَعُ أَيْدٍ تَهْجُمُهْ

حَفَّ حَفيفَ الغَيْثِ جادَتْ دِيَمُهْ

يَمْسُدُ أَعْلى لَحْمِهِ ويَأْرِمُهْ

جادَتْ بِمَطْحونٍ لَها لا تَأْجِمُهْ

ديوان رؤبة. تح: وليم بن الورد، ج: 3، ص: 186.— المصدر

9.1معنى من معاني «أَرَم»

* الأَرَمُ والأَرِمُ والِإرَمُ: الحِجارةُ تُنْصَبُ لِتَكونَ عَلامةً لِلطَّريقِ (ج) أُرومٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال المُرَقِّشُ الأَكْبَرُ (ت: 75ق هـ = 550م)َ يَصِفُ ناقةً:

لَمْ تَقْرَأِ القَيْظَ جَنينًا ولا
أَصُرُّها تَحْمِلُ بَهْمَ الغَنَمْ

بَلْ عَزَبَتْ في الشَّوْلِ حتّى نَوَتْ
وسُوِّغَتْ ذا حُبُكٍ كالإرَمْ

ديوان المرقشين، كارين صادر، ج: 1، ص: 74.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال دُرَيْدُ بْنُ الصِّمّةِ (ت: 8هـ = 629م) يَرْثي صَديقَهُ مُعاوِيةَ أَخا الخَنْساءِ:

رَأَيْتُ مَكانَهُ فَعَرَضْتُ بَدْءًا
وأَيُّ مَقيلِ رُزْءٍ يابْنَ بَكْرٍ

إلى إرَمٍ وأَحْجارٍ وصيرٍ
وأَغْصانٍ مِنَ السَّلَماتِ سُمْرِ

ديوان دريد. تح: عبد الرسول، ص: 109.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال البَعيثُ المُجاشِعيُّ (ت: 134هـ = 751م):

قالَ مُشَبِّهًا الأُرومَ بِالرِّجالِ في قِيامِها:

على مَتْنِ عاديٍّ كَأَنَّ أُرومَهُ
رِجالٌ يُتَلّونَ الصَّلاةَ قِيامُ

ديوان البعيث المجاشعي. تح: عدنان محمد، ص: 84.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال البوصيريُّ (ت: 696هـ = 1295م) يَصِفُ زَوالَ مَمالِكِ الأَقْدَمينَ:

وأَيْنَ ما شادَهُ شَدّادُ مِنْ إرَمٍ
وأيَنْ َما ساسَهُ في الفُرْسِ ساسانُ

صديق خان، الإذاعة. تح: الجابي، ج: 1، ص: 236.— المصدر

أُرْمة

اسم
10.1معنى من معاني «أُرْمة»

* أُرْمةُ الرَّأْسِ: حُروفُهُ، (ج) أُرومٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال النَّضْرُ بْنُ شُمَيلٍ المازِنيُّ (ت: 203هـ = 819م):

"‏أُرومُ الرَّأْسِ: حُروفُهُ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

الأزهري، التهذيب. تح: الإبياري، ج: 15، ص: 302.— المصدر
10.2معنى من معاني «أُرْمة»

وــــــــــ: خَيْطٌ مِنَ الصّوفِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"الأُرْمُ: نَحْوُ السُّيورِ مِنَ العِهْنِ، واحِدَتُها: أُرْمةٌ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 290.— المصدر
10.3معنى من معاني «أُرْمة»

وــــــــــ: القَبيلةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"وبَنو فُلانٍ أَشَدُّ أُرْمةً مِنَ العَرَبِ: أيْ: قَبيلةً". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 290.— المصدر
11 —

إرَم

اسم
11.1معنى من معاني «إرَم»

* الإرَمُ: القَبْرُ، (ج) آرامٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال غطيفُ بْنُ عَدِيٍّ الكَلْبيُّ (ت: 86ق هـ = 538م):

كَما سَطا بِالآرامِ عادٌ وبِالـــ
حِجْرِ وأَرْعى لِتُبَّعٍ تَبَعا

ديوان شعراء بني كلب بن وبرة. تح: البيطار، ج: 1، ص: 402.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الزِّبْرِقانُ بْنُ بَدْرٍ التَّميميُّ (ت: 45هـ = 665م) يَعِظُ:

فَإنَّـما النّاسُ يا لِلَّهِ أُمُّهُمُ
أَكائِلُ الطَّيْرِ أَوْ حَشْوٌ لِآرامِ

هُمْ يَهْلَكونَ ويَبْقى بَعْدُ ما صَنَعوا
كَأَنَّ آثارَهُمْ خُطَّتْ بِأَقْلامِ

الجاحظ، البيان والتبيين. تح: هارون، ج: 3، ص: 179.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ مَيّادةَ (ت: 140هـ = 766م):

ومُضَبَّرٍ عِرْدِ الزِّجاجِ كَأَنَّهُ
إرَمٌ لِعادَ مُلَزَّزِ الأَرْكاحِ

ديوان ابن ميادة، مجمع اللغة العربية، ج: 1، ص: 101.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ مَنْظورٍ (ت: 711هـ = 1311م):

"والأُرومُ أَيضًا: الأَعْلامُ، وقيلَ: هِيَ قُبورُ عادٍ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، اللسان. تح: أمين عبد الوهاب، ج: 1، ص: 124.— المصدر
12 —

أَرومة

اسم
12.1معنى من معاني «أَرومة»

* الأَرومةُ والأَرومُ: أَصْلُ كُلِّ شَيْءٍ ومُجْتَمَعُهُ، (ج) أَرومٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال رَبْخةُ بْنُ حارِثٍ الأَزْديُّ (ت: 86ق هـ = 538م) يَصِفُ اسْتَعانةَ حَرْبِ بْنِ حُمامٍ بهِ حينَ أَعْجَزَهُ دَفْعُ دِيّاتِ القَتْلى:

يُنادي بِأَعْلى الصَّوْتِ يا رُبْخُ إنَّني
تَحَمَّلْتُ غُرْمًا مِنْ َثقيلِ غَرامِ

فَنَحْنُ وأَنْتُمْ مِنْ أُرومةِ مالِكٍ
مِنَ النَّبْعِ لا مِنْ جِزْوَعٍ وثُمامِ

سلمة بن مسلم الصحاري، الأنساب. تح: محمد إحسان، ص: 779.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الأَعْشى الكَبيرُ (ت: 7هـ = 628م) يَمْدَحُ قَيْسَ بْنَ مَسْعودٍ:

ما فَوْقَ بَيْتِكَ مِنْ بَيْتٍ عَلِمْتُ بِهِ
وفي أَرومَتِه ما مَنْبِتُ العُودِ

إنّي لَمُمْتَدِحٌ قَيْسًا وسائِلُهُ
خَيْرُ القُيوسِ لَنا قَيْسُ بْنُ مَسْعودِ

ديوان الأعشى الكبير. تح: الرضواني، ص: 143.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو نواسٍ (ت: 198هـ = 814م):

فَرْعٌ تَمَكَّنَ في أَرومِ عِمارةٍ
ثَبْتٌ مَكارِمُها على الأَيّامِ

ديوان أبو نواس. تح: الحديثي، ج: 1، ص: 347.— المصدر

أَيْرَم

اسم
13.1معنى من معاني «أَيْرَم»

0 ما بِها مِنْ أَيْرَمٍ: ما بِها مِنْ أَحَدٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الرّاجِزُ (370هـ = 981م، مِنْ شَواهِدِ الأزهريِّ):

لَمْ تَرْعَ يَوْمًا غَنَما
في الرَّوايا أَيْرَما

الأزهري، التهذيب. تح: الإبياري، ج: 15، ص: 301.— المصدر
14 —

أَيْرَمِيّ

اسم
14.1معنى من معاني «أَيْرَمِيّ»

0 ما بِالدّارِ أَيْرَمِيُّ: ما بِها أَحَدٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ الأَنْباريُّ (ت: 328هـ = 938م):

"وما بِالدّارِ أَيْرَمِيُّ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن الأنباري، الزاهر في معاني كلمات الناس. تح: الضامن، ج: 1، ص: 372.— المصدر
15 —

آرِم

اسم
15.1معنى من معاني «آرِم»

* الآرِمُ: الضِّرْسُ أو النّابُ (ج) أُرَّمٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال العجّاجُ (ت: 90هـ = 708م) يَصِفُ حَنَقَ الأَعْداءِ وغَيظَهُمْ:

وقَدْ وعَظْناها اتِّقاءَ المَأْثَمِ

وحَذَرَ الفَحْشاءِ ما لَمْ تُظْلَمِ

تَقَرُّبًا والأَمْرُ لَمّا يُفْقَمِ

فَجَعَلوا الغايةَ حَرْقَ الأُرَّمِ

ديوان العجاج. تح: عزة حسن، ج: 1، ص: 288.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الرّاجِزُ (170هـ = 786م، مِنْ شَواهِدِ الخَليلِ):

أُخْبِرْتُ أَحْماءَ سُلَيْمى أَنَّما

باتوا غِضابًا يَحْرِقونَ الأُرَّما

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 296.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ مَنْظورٍ (ت: 711هـ = 1311م):

"ويُقالُ: فُلانٌ يَحْرِقُ علَيْكَ الأُرَّمَ، إذا تَغَيَّظَ وحَكَّ أَضْراسَهُ بَعْضَها بِبَعْضٍ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، اللسان. تح: أمين عبد الوهاب، ج: 1، ص: 123.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَليٌّ الجارِمُ (ت: 1369هـ = 1949م) يَتَحَدَّثُ عَنْ سُقوطِ غَرْناطةَ:

"وانْقَضَّ فِرْدينانْدُ على المَرْجِ يَعيثُ فَسادًا، بَيْنَما كانَ مَوْلايَ أبو الحَسَنِ -وقَدْ عادَ إلى مُلْكِهِ- يَحْرِقُ الأُرَّمَ غَيْظًا مِنْ وَراءِ أَسْوارِهِ".

علي الجارم، سلاسل الذهب، ج: 1، ص: 410.— المصدر
