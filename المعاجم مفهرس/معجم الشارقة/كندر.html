01 —

كُنادِر

اسم
1.1معنى من معاني «كُنادِر»

* الكُنادِرُ: رَجُلٌ فيهِ قِصَرٌ وغِلَظٌ مَعَ شِدَّةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"فإِذا كانَ قِصَرٌ وغِلَظٌ مع شِدَّةٍ قيلَ: رَجُلٌ كُلْكُلٌ، وكُلاكِلٌ، وكَوَأْلَلٌ، وجُعْشُمٌ وكُنْدُرٌ وَكُنَيْدِرٌ وكُنادِرٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلاّم، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 100.— المصدر
1.2معنى من معاني «كُنادِر»

وــــــــــ: الحِمارُ الوَحْشِيُّ الغَليظُ والعَظيمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ دابَّتَهُ:

كَأَنَّ تَحْتي كُنْدُرًا كُنادِرا

جَأْبًا قَطوطى يَنْشِجُ المَشاجِرا

[الجَأْبُ: الغَليظُ. قَطوطى: قَصيرُ القَدَمَيْنِ يُقارِبُ الخَطْوَ. يَنْشِجُ المَشاجِرا: يُصَوِّتُ بِالأَشْجارِ]
رؤبة بن العجاج، مجموع أشعار العرب. تح: وليم البروسي، ج: 2، ص: 77.— المصدر
02 —

كِنْدارَة

اسم
2.1معنى من معاني «كِنْدارَة»

* الكِنْدارَةُ: سَمَكَةٌ لَها سَنامٌ كَسَنامِ الجَمَلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=831م):

"الكِندارَةُ: سَمَكَةٌ بَيْنَ البَياضِ والصُّفْرَةِ لَها سَنامٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: حسن آل ياسين، ج: 6، ص: 369.— المصدر
03 —

كَنْدَر

اسم
3.1معنى من معاني «كَنْدَر»

* الكَنْدَرُ: ما غَلُظَ مِنَ الأَرْضِ وارْتَفَعَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"والكَنْدَرُ مِنَ الأَرْضِ: ما غَلُظَ وَارْتَفَعَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم والمحيط الأعظم. تح: هنداوي، ج: 7، ص: 165.— المصدر
04 —

كَنْدَر/ وكُنْدُر

اسم
4.1معنى من معاني «كَنْدَر/ وكُنْدُر»

* الكَنْدَرُ والكُنْدُرُ: ضَرْبٌ مِنْ حِسابِ الرّومِ في النُّجومِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والكُنْدُرُ: ضَرْبٌ مِنْ حِسابِ الرّومِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد، العين. تح: مهدي المخزومي ،وآخر، ج: 5، ص: 429.— المصدر
05 —

كُنْدُر

اسم
5.1معنى من معاني «كُنْدُر»

* الكُنْدُرُ: الحِمارُ الوَحْشِيُّ الغَليظُ والعَظيمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ حِمارَهُ:

كَأَنَّ تَحْتي كُنْدُرًا كُنادِرا

جَأْبًا قَطوطى يَنْشِجُ المَشاجِرا

[الجَأْبُ: الغَليظُ. قَطوطى: قَصيرُ القَدَمَيْنِ يُقارِبُ الخُطى. يَنْشِجُ المَشاجِرا: أَيْ يُصَوِّتُ بِالأَشْجارِ]
رؤبة بن العجاج، مجموع أشعار العرب. تح: وليم البروسي، ج: 2، ص: 77.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الهادي بْنُ مُحَمَّدي (ت: 1319هـ=1901م):

بَعْدَ اشْتِراءِ ‌كُنْدُرِ الأَغْلالِ
أَصْبَحَ في القُيودِ والأَغْلالِ

أحمد الشنقيطي، الوسيط في تراجم أدباء شنقيط. تح: أحمد الشنقيطي، ص: 77.— المصدر
5.2معنى من معاني «كُنْدُر»

وــــــــــ: صَمْغُ شَجَرَةٍ شائِكَةٍ وَرَقُها كالآسِ، لُبانٌ، بُخُر. واحِدَتُهُ كُنْدُرَةٌ. يونانِيَّةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الكُنْدُرُ اسْمٌ لِلْعِلْكِ".

الخليل بن أحمد ،العين. تح: مهدي المخزومي وآخر، ج: 5، ص: 429.— المصدر
”وقالَ جابِرُ بْنُ حَيّانَ (ت: 200هـ=816م) يَشْرَحُ مُكَوِّناتِ بَخورِ زُحَلَ:

"وإِنْ زيدَ فيهِ قَليلٌ مِنَ الكُنْدُرِ والجَوْزَبُوا كانَ جَيِّدًا".

جابر بن حيان، رسائل جابر بن حيان. تح: المزيدي، ص: 381.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ النَّفيسِ (ت: 687هـ=1288م) يُعْطي وَصَفاتٍ طِبِّيَّةً لِلْمَعِدَةِ:

"وقَدْ يَحْتاجونَ مَعَ ذَلِكَ إِلى مِثْلِ الكُنْدُرِ أَوْ المِصْطَكى ونَحْوِ ذَلِكَ".

ابن النفيس، الشامل في الصناعة الطبية. تح: يوسف زيدان، ص: 90.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الطَّحْطاوِيُّ (ت: 1231هـ=1816م) يَذْكُرُ مَكْروهاتِ الصِّيامِ:

"الكُنْدُرُ لِأَنَّهُ يُتَّهَمُ بِالإِفْطارِ بِمَضْغِهِ".

الطحطاوي، شرح الطحطاوي على مراقي الفلاح. تح: نعيم زرزور، ج: 1، ص: 680.— المصدر
5.3معنى من معاني «كُنْدُر»

وــــــــــ: الرَّجُلُ يَكونُ فيهِ قِصَرٌ وغِلَظٌ مَعَ شِدَّةٍ. (ج) وكَنادِرَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"وكُنْدُرٌ وكُنَيْدِرٌ وكُنادِرٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلاّم، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 100.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عُصْفورٍ (ت: 669هـ=1271م) يَذْكُرُ ما جاءَ عَلى وَزْنِ فُعْلُلٍ:

"والصِّفَةُ نَحْوُ: جُرْشُعٌ وكُنْدُرٌ".

ابن عصفور، الممتع الكبير، مكتبة لبنان، ج: 1، ص: 54.— المصدر
06 —

كَنْدَرَة / كُنْدُرَة

اسم
6.1معنى من معاني «كَنْدَرَة / كُنْدُرَة»

* الكَنْدَرَةُ والكُنْدُرَةُ: مِجْثَمُ البازي يُهَيَّأُ لَهُ مِنْ خَشَبٍ أَوْ مَدَرٍ. (ج) كَنادِرُ. (د)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"وكُنْدُرَةُ البازي: مِجْثَمٌ يُهَيَّأُ لَهُ مِنْ خَشَبٍ أَوْ مَدَرٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد ،العين. تح: مهدي المخزومي وآخر، ج: 5، ص: 429.— المصدر
”وقالَ الجاحِظُ (ت: 255هـ=869م) يُقارِنُ بينَ الصَّقْرِ والبازي:

"والبازي أكْرَمُ مِنَ الصَّقْرِ وأشَدُّ وأكْثَرُ ثَمَنًا، وأَجْمَلُ جَمالًا، وأَعْفى صَيْدًا، وأَنْبَلُ نُبْلًا، إِنْ قَبَضَ عَلَيْهِ قَتَلَهُ، وإِنْ لَمْ يُنَحِّ كُنْدُرَتَهُ عَنْ قُرْبِهِ أَوْهَنَ نَفْسَهُ".

الجاحظ، رسائل الجاحظ. تح: عبد السلام هارون، ج: 1، ص: 291.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ شِهابُ الدّينِ النُّوَيْرِيُّ (ت: 733هـ=1333م):

"فأَمَرَ بِجَمْعِ عِدَّةٍ مِنَ البُزاةِ، فجُمِعَتْ وجُعِلَتْ في مَجْلِسِهِ. فعَرَضَ لِبَعْضِها أَيْمٌ، فوَثَبَ عَلَيْهِ؛ فقالَ: مَلِكٌ يَغْضَبُ كَما تَغْضَبُ المُلوكُ. ثُمَّ أَمَرَ بِهِ فنُصِبَ عَلى ‌كُنْدُرَةٍ بَيْنَ يَدَيْهِ".

[ الأَيْمُ: الحَيَّةُ الذَّكَرُ]
شهاب الدين النويري، نهاية الأرب، دار الكتب والوثائق القومية، ج: 10 ص: 188.— المصدر
6.2معنى من معاني «كَنْدَرَة / كُنْدُرَة»

وــــــــــ: ما غَلُظَ مِنَ الأَرْضِ وغيرِها وارْتَفَعَ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"‌والكُنْدُرَةُ مِنَ الأَرْضِ: ما غَلُظَ وارْتَفَعَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب. ج: 5، ص: 153. ج: 5، ص: 153.— المصدر
”وقالَ ابْنُ تَيْمِيَّةَ (ت: 728هـ=1328م) يُبَيِّنُ مُشابَهَةَ الرّافِضَةِ لِلْيَهودِ والنَّصارى:

"‌واليَهودُ يَسْجُدونَ في صَلاةِ الفَجْرِ الكُنْدُرَةَ، وكَذَلِكَ الرّافِضَةُ ".

ابن تيمية، منهاج السنة النبوية. تح: رشاد سالم، ج: 1، ص: 32.— المصدر
07 —

كُنْدَرَة

اسم
7.1معنى من معاني «كُنْدَرَة»

* الكُنْدَرَة: الحِذاءُ. (ج) كَنادِرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو الهِنْدِيِّ (ت: 180هـ=795م) يُخاطبُ الأَخْطَلَ الشّاعِرَ:

إِنْ كُنْتَ نَدْماني أَبا مالِكٍ
فاسْقِ أَبا الهِنْدِيِّ بِالكُنْدَرَهْ

ديوان أبي الهندي. تح: عبد الله الجبوري، ص: 42.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ تَيْمِيَّةَ (ت: 728هـ=1327م) يَذْكُرُ أَحْكامَ الإِحْرامِ:

"فإِنَّ مَنْ لَبِسَ الكُنْدَرَةَ الَّتي لا ساقَ لَها في زَمَنِنا الحاضِرِ فإِحْرامُهُ صَحيحٌ".

ابن تيمية، الاختيارات. تح: الحارثي، وآخرين، ج: 5، ص: 17.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ محمد رشيد رضا (1316هـ=1899م) يَذْكُرُ رَأْيَ بَعْضِ العُلَماءِ في بَعْضِ المُسْتَحْدَثاتِ:

"واخْتُرِعَ الحِذاءُ المُعَرَّفُ ‌بِالكُنْدَرَةِ أَوِ الجَزْمَةِ، فَقامَتْ قِيامَةُ العُلَماءِ عَلى مُحْتَذيها".

مجلّة المنار. العدد: 1 ج: 1، ص: 342.— المصدر
08 —

كُنْدُرَة

اسم
8.1معنى من معاني «كُنْدُرَة»

* الكُنْدُرَةُ: الزَّوْرَقُ أَوِ القارِبُ الصَّغيرُ. (ج) كَنادِرُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ بَطُّوطَةَ (ت: 779هـ=1377م) يَرْوي قِصَّةَ وُصولِهِ إِلى الجَزائِرِ، ويَذْكُرُ أَهْلَها:

"ومِنْ عَوائِدِهِمْ إذا قَدِمَ علَيْهِمْ مَرْكِبٌ أَنْ تَخْرُجَ إِلَيْهِ الكَنادِرُ، وهِيَ القَوارِبُ الصِّغارُ، واحِدُها ‌كُنْدُرَةٌ، بِضَمِّ الكافِ والدّالِ".

ابن بطّوطة، تحفة النّظّار. تح: العريان، ص: 593.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ محمد رشيد رضا (1344هـ=1926م) تاريخُ الطَّبْعِ:

"وفي صَباحِ الأَرْبِعاءِ 7 جُمادى الثّانِيَةِ سَنَةَ 1344 تَلَقّى فَريقٌ مِنْ جُنْدِ الـمُشاةِ ورَهْطٌ مِنْ خَيّالَةِ الحَرَسِ السُّلْطانِيِّ الخاصِّ بِقِيادَةِ سُمُوِّ الأَميرِ عَبْدِ اللهِ أَخي عَظَمَةِ السُّلْطانِ الأَمْرَ بِالمَسيرِ مِنَ المَقَرِّ العالي إِلى ‌الكُنْدُرَةِ لِلانْتِظارِ فيها، لِاسْتِقْبالِ عَظَمَتِهِ حينَ دُخولِهِ إِلَيْها، وقَدْ أُعِدَّ فيها سُرادِقٌ خاصٌّ لِاسْتِقْبالِ الـمُسْتَقْبِلين".

مجلّة المنار. العدد: 1 ج: 2، ص: 673.— المصدر
09 —

كِنْدير

اسم
9.1معنى من معاني «كِنْدير»

* الكِنْدير: الحِمارُ الغَليظُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المُزَرِّدُ الغَطفانِيُّ (ت: 10هـ=631م) يَهْجو مُرَّةَ بْنَ واقِعٍ:

فأَيِّهْ ‌بِكِنْديرٍ حِمارِ ابْنِ واقِعٍ
رَآكَ بِإيرٍ فاشْتَأى مِنْ عُتائِدِ

[أَيِّهْ: اسْتَعِنْ بِهِ وادْعُهُ. حِمارُ ابْنُ واقِعٍ: المَقْصودُ مُرَّة بْنُ واقِعٍ. إيرٌ: جَبَلٌ في أَرْضِ غَطَفانَ. اشْتَأى: سَبَقَ إِلَيْكَ. عُتائِدٌ: هَضَباتٌ لِبَني مُرَّةَ أَسْفَلَ مِنْ إيرٍ]
ديوان المزرد بن ضرار الغطفاني. تح: خليل العطية، ص: 79.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بْنُ بُرْدٍ (ت: 167هـ=784م) يَهْجُو أَبا هِشامٍ عَمْرَو بْنَ عَبْدِ الرَّحْمنِ الباهِلِيَّ:

بَكَيْتَ خِلافَ كِنْديرٍ عَلَيْهِ
وهَلْ يُغْني عَنِ الحَرْبِ البُكاءُ

ديوان بشار بن برد. تح: إحسان عباس ، ص: 17.— المصدر
9.2معنى من معاني «كِنْدير»

وــــــــــ: ضَخْمُ المَحْزِمِ المُمْتَلِئُ مِنَ النّاسِ وغيرِهِمْ، وهُوَ بِتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"الكِنْديرَةُ: الضَّخْمُ ضَخْمٌ مَحْزِمُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو الشيباني، كتاب الجيم. تح: العزباوي، ج: 3، ص: 160.— المصدر
9.3معنى من معاني «كِنْدير»

وــــــــــ مِنَ النّاسِ: القَصيرُ الغَليظُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230هـ=845م):

"رَجُلٌ كِنْديرٌ بَيِّنُ ‌الكِنْديرَةِ، لِلْغَليظِ الخُلُقِ القَصيرِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أَبو مسحل الأعرابي، النوادر.تح: عزة حسن، ج: 1ص: 226. — المصدر
9.4معنى من معاني «كِنْدير»

o وذو كِنْديرَةٍ: غَليظٌ ضَخْمٌ شَديدُ الخَلْقِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَلْقَمَةُ الفَحْلُ (ت: 20ق.هـ=603م):

يَتْبَعْنَ ذا كِنْديرةٍ عَجَنَّسا

الصغاني، التكملة والذيل والصلة. تح: أبو الفضل إبراهيم ، ج: 3، ص: 184.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"إِنَّهُ لَذو كِنْديرَةٍ.". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الأزهري، تهذيب اللغة. تح: تحقيق: مرعب، ج: 10، ص: 233.— المصدر
