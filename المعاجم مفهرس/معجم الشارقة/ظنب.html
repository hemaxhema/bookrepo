01 —

ظِناب

اسم
1.1معنى من معاني «ظِناب»

* الظِّنابُ مِنَ الأرْضِ: الحَزْنُ الشَّديدُ الوَعْرُ، خِلافُ السَّهْلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْنُ بْنُ أبي فُهَيْرَةَ المِرْداسيُّ (ت: 250هـ=864م) يَبْكي عَلى الأَطْلالِ:

تأبَّدَ من جُمْلٍ مَعَارِفُ واسطٍ
فَأَطْلَالُها مِن قُنَّةٍ فَشِعابُها

فَبَطْنُ سَواسٍ فالخِيامُ فَمُنْثَنى
لِواءٍ فذاتُ العُصْلِ قَفْرٌ يَبابُها

فَرَوْضَةُ عَرّامٍ فَهَضْما نُبايِعٍ
فَبَطْنُ رِيامٍ سَهْلُها وظِنَابُها

الهجري، التعليقات والنوادر. تح: الجاسر، ج: 2، ص: 856.— المصدر
02 —

ظِنْب

اسم
2.1معنى من معاني «ظِنْب»

* الظِّنْبُ: أصْلُ الشَّجَرَةِ، أَوْ أَصْلُ العَرْفَجِ إِذا انْسَلَخَ مِنْ وَرَقِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال جُبَيْهاءُ الأشْجَعيُّ (ت: 65هـ=685م) يَصِفُ عَنْزًا أُهْدِيَتْ لَهُ:

ولَوْ أَنَّها طافَتْ بِظِنْبٍ مُعَجَّمٍ
نَفَى الرِّقَّ عَنْهُ جَدْبُهُ فهْوَ كالِحُ

لَجاءَتْ كأَنَّ القَسْوَرَ الـجَوْنَ بَجَّها
عَساليجُهُ والثَّامِرُ الـمُتَناوِحُ

المفضل الضبي، المفضليات. تح: شاكر، وآخر، ج: 1، ص: 168.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ الأعْرابيِّ (ت: 231هـ=845م) -:

"الظِّنْبُ: أَصْلُ الشَّجَرَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد مرعب وآخرين، ج: 14، ص: 390.— المصدر
03 —

ظُنْبَة

اسم
3.1معنى من معاني «ظُنْبَة»

* الظُّنْبَةُ مِنَ السَّهْمِ: عَقَبَةٌ أَوْ عَصَبَةٌ تُلَفُّ عَلَى الرِّيشِ مِمَّا يَلِي فُوقَ السَّهْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو حَنيفَةَ الدِّينَوَريُّ (ت: 282هـ=895م):

"الظُّنْبَةُ: عَقَبَةٌ تُلَفُّ عَلَى أطَرَافِ الرِّيشِ مِمَّا يَلِي الفُوقَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 30.— المصدر
04 —

ظُنْبوب

اسم
4.1معنى من معاني «ظُنْبوب»

* الظُّنْبوبُ: حَرْفُ عَظْمِ السَّاقِ أَوْ قَصَبَتِهِ مِنْ أَمامٍ. (ج) ظَنابيبُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال تأَبَّط شَرًّا (ت: 82ق.هـ=542م) يَصِفُ الصُّعْلوكَ الذي يَبْكي لِفَقْدِ صَداقَتِهِ:

لكنَّما عَوَلِي إِنْ كنْتُ ذا عَوَلٍ
علَى بَصيرٍ بِكَسْبِ الحمدِ سَبّاقِ

سَبّاقِ غاياتِ مَجدٍ في عَشيرَتِهِ
مُرَجِّعِ الصَّوتِ هَدًّا بينَ أَرْفاقِ

عارِي الظَّنابيبِ مُمْتدٍّ نَواشِرُهُ
مِدْلاجِ أَدْهَمَ واهي الماءِ غَسّاقِ

المفضل الضبي، المفضليات. تح: شاكر، وآخر، ج: 1، ص: 29.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال ابْنُ مُقْبِل (ت: 37هـ=658م) يَصِفُ حَثَّهُ ناقَتَهُ عَلى السَّيْرِ:

وقَدْ أَبْعَثُ الوَجْناءَ يَزْجُلُ خُفَّها
وَظيفٌ كَظُنْبوبِ النَّعامَةِ أَرْوَحُ

ديوان ابن مقبل. تح: عزة حسن، ص: 46.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بْنُ أحْمَد (ت: 170هـ=787م) -:

"الظُّنْبوبُ: حَرْفُ السّاقِ اليابِسِ مِنْ قُدُمٍ".

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 165.— المصدر
”وقال أبو نُواس (ت: 198هـ=814م) يَصِفُ مُطارَدَةً بَيْنَ كَلْبٍ وثَعْلَبٍ، حاوَلَ فيها الثَّعْلَبُ الفِرارَ:

وصارَ لِحْياهُ عَلى أَنْسائِهِ
ولَيْسَ يُنْجيهِ عَلى دَهائِهِ

تَنَسُّمُ الْأرْوَاحِ في انْبِرَائِهِ
خَضْخَضَ ظُنْبُوبَيْهِ فِي أمْعَائِهِ

ديوان أبي نُواس. تح: الحديثي، ص: 166.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ الصَّيْقلِ الجَزَريُّ (ت: 701هـ=1302م) يَذْكُرُ سَعْيَهُ في الوُصولِ إلى زاهِدٍ حَقيقِيٍّ:

"ثم إنَي رجوتُ نُشورَ الطَّاعِ، وقَبَضْتُ مَنْشورَ الإِقْطاعِ بالانْقِطاعِ، وأَخَذْتُ أَزْحَلُ عَنِ المَحْدوج، وأَرْحَلُ واحِدًا عَنِ الحُدوجِ، عَلَّ أَنْ أَظْفَر بَعْدَ هِياجِ الخَبَبِ والخُبوبِ، والقِيامِ على ساقِ السَّفَهِ والظُّنْبوبِ بِمَنْ زَهِدَ في الحُطامِ (...) لِأَجْعَلَهُ جَوْنًا لِظُلَمِ المُتَشابِهاتِ، وعَوْنًا لقَدْعِ شَهَواتِ الشُّبُهاتِ".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ج: 1، ص: 584.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال مُحَمَّدُ بْنُ الطُّلْبَة (ت: 1272هـ=1856م) يَصِفُ ظَليمًا:

أزَجَّ مِنَ الزُّعْرِ الظَّنابيبِ مُعْرِسٍ
بِخَرْجاءَ هَوْجاءِ البُرايَةِ عَوْهَجِ

ديوان ابن الطلبة. ج: 1، ص: 173.— المصدر
4.2معنى من معاني «ظُنْبوب»

وــــــــــ: مِسْمَارٌ يَكُونُ فِي جُبَّةِ السِّنانِ حَيْثُ يُرَكَّبُ فِي عَالِيَةِ الرُّمْحِ، وَيُكَنَّى بِقَرْعِ الظُّنْبوبِ عَنْ سُرْعَةِ الاسْتِجابَةِ والاهْتِمامِ بالأَمْرِ، والعَزْمِ عَلَيْهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال سَلامَةُ بْنُ جَنْدَلٍ (ت: 23ق.هـ=599م) يَصِفُ حَالَهَمْ فِي الْإسْرَاعِ إلَى نَجْدَةِ الصَّارِخِ الْمُسْتَغِيثِ:

كُنَّا إِذَا ما أَتانا صارخٌ فَزِعٌ
كانَ الصُّراخُ لهُ قَرْعَ الظَّنابيبِ

المفضل الضبي، المفضليات. تح: شاكر، وآخر، ج: 1، ص: 124.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الفَرَزْدَقُ (ت: 110هـ=728م) يَصِفُ صاحِبًا لَهُ:

قَرَعْتُ ظَنابيبي على الصَّبْرِ بَعْدَهُ
فَقَدْ جَعَلَتْ عَنْهُ الـجَنائِبُ تُصْحِبُ

ديوان الفرزدق. تح: علي فاعور، ص: 67.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بْنُ أحْمَد (ت: 170هـ=787م) -:

"والظُّنْبوبُ: مِسْمارٌ يَكونُ في جُبَّةِ السِّنانِ حَيْثُ يُرَكَّبُ في عالِيَةِ الرُّمْحِ، والجَميعُ الظَّنابيبُ".

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 165.— المصدر
”وقال ابْنُ هانئٍ الأنْدَلُسيُّ (ت: 362هـ=973م) يَمْدَحُ:

وما هُوَ إِلّا أَنْ يُشيرَ بِلَحْظِهِ
فَتَمْخُرَ فُلْكٌ أَوْ تُغِذَّ مَقانيبُ

فلا قارِعٌ إِلّا القَنا السُّمْرُ بالقَنا
إذا قُرِعَتْ لِلْحادِثاتِ الظَّنابيبُ

ديوان ابن هانئ الأندلسي. تح: كرم البستانيّ، ص: 37.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال أَحْمَدُ بنُ عَبْدِ النّورِ (ت: 676هـ=1277م) في رِسالَةٍ كَتَبها إلى رَئيسِ ديوانِ الإِنْشاءِ الصّاحِبِ فَخْرِ الدّينِ إِبْراهيمَ بنِ لُقْمانَ الشَّيْبانِيِّ في زَمَنِ الظّاهِرِ بَيْبَرْس:

"فَإنَّهُ يُقْرَعُ فِي كُلِّ مُسْتنَدٍ إلَيْهِ ظُنْبوبُ الاعْتِزامِ، ويَدُلُّ عَلَيْهِ مَنْجَحُهُ بِالتَّضَمُّنِ والـمُطابَقَةِ والالتِزامِ".

العمري، مسالك الأبصار. تح: كامل الجبوري، وآخر، ج: 13، ص: 196.— المصدر
4.3معنى من معاني «ظُنْبوب»

o وظُنْبوبُ السَّيْفِ: طَرَفُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الطّائِيُّ (ت: 206هـ=821م، من رُواةِ أبي عَمْرٍو الشَّيْبانيِّ):

"ظُنْبوبُ السَّيْفِ: طَرَفُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

أبو عمرو، الجيم. تح: الطحاوي، ج: 2، ص: 224.— المصدر
