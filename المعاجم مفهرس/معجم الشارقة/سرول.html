سرول
سَرْوَلَ(فعل) سُرْوِلَ(فعل) تَسَرْوَلَ(فعل) سِرْوال(اسم)
المقدّمة

النَّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (س ر و ل):

في الآراميّة: وَرَدَ الاسْمُ: סַרְבָּלָא (سَربالا) sarbālā بِمَعْنى (ما يُلبس: قَميصٌ، عباءةٌ، بنطالٌ) يُوافِقُ الاسمَ العَرَبيَّ (سروالٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسْمَ الآراميَّ خَضَعَ لإبدالِ الواوِ باءً، ومِنْ مَجيئِهِ بِمَعْنى (سروالٌ) ما وَرَدَ في سِفرِ دانيالَ: "וּשְׂעַר רֵאשְׁהוֹן לָא הִתְחָרַךְ וְסַרְבָּלֵיהוֹן לָא שְׁנוֹ"، أيْ: "وشعرُ رُؤوسِهِمْ لم يَحترِقْ، وسَراويلُهُمْ لَم تَتَغيّرْ".

(דנ': ג, כז)

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܫܰܪܒ̊ܳܠ (شَربال) šarbāl بِمَعْنى (تُبّانٌ: لباسٌ ذو ياقينِ يَسترُ العَورةَ) يُوافِقُ الاسمَ العَرَبيَّ (سروالٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسْمَ السُّريانيَّ خَضَعَ لإبدالِ السّينِ شينًا والواوِ باءً، ومِنْ مَجيئِهِ بِمَعْنى (سروالٌ) ما وَرَدَ في سِفرِ دانيالَ: "ܒܫܰܖ̈ܒܳܠܰܝܗܽܘܢ. ܘܰܒܦܶܛܫܰܝ̈ܗܽܘܢ. ܘܰܒܢ̈ܰܚܬܰܝܗܽܘܢ ܘܰܒܩܽܘܒ̈ܥܰܝܗܽܘܢ. ܘܰܐܪܡܺܝܘ ܐܶܢܽܘܢ ܒܓܰܘ ܐܰܬܽܘܢܳܐ ܕܢܽܘܪܳܐ ܝܳܩܶܕܬܳܐ"، أي: " ثُمَّ أُوثِقَ هؤُلاءِ الرِّجالُ في سَراويلِهِمْ وأقْمِصَتِهِمْ وأَرْدِيَتِهِمْ ولِباسِهِمْ وأُلْقُوا في وَسَطِ أتُّونِ النّارِ المُتَّقِدةِ".

(ܕܳܢܺܝܐܶܝܠ-ܓ: 21)

في الجِعزيّة: وَرَدَ الاسْمُ: ሠራውር (شَراوْر۪) šarāwər بِمَعْنى (سِروالٌ) يُوافِقُ الاسمَ العَرَبيَّ (سِروالٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسْمَ الجعزيَّ خَضَعَ لإبدالِ السّينِ شينًا واللّامِ راءً، مَعَ تَغيّرٍ في بَعضِ حَركاتِهِ.

(CODG, p.536)

المعانى الكلّيّة

  1. الم,اللِّباسُ مِنَ ساقَيْنِ طَويلَتَيْنِ
  2. الشَّعْرُ أَوِ الرّيشُ عَلى ساقِ الحَيَوانِ أَوِ الطّائِرِ
01 —

سَرْوَلَ

فعل
1.1معنى من معاني «سَرْوَلَ»

* سَرْوَلَ فُلانٌ وغَيْرُهُ: أَبْطَأَ في مَشْيِهِ مِنْ إِعْياءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ (ت: 515هـ=1122م):

"وسَرْوَلَ: أَبْطَأَ في مَشْيِهِ مِنْ إِعْياءٍ".

ابن القطاع، الأفعال. ج: 2، ص: 174.— المصدر
1.2معنى من معاني «سَرْوَلَ»

وــــــــــ: لَبِسَ السِّرْوالَ، فَهُوَ مُسَرْوَلٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«رَحِمَ اللهُ الـمُسَرْوَلاتِ».

الخطيب، المتفق والمفترق. تح: الحامدي، ج: 2، ص: 1124.— المصدر
”وقالَ مُعاوِيَةُ بْنُ أَبي سُفْيانَ (ت: 60ه=680م):

أَمّا قُرَيْشٌ فَأَشْياخٌ مُسَرْوَلَةٌ
واليَثْرِبِيّونَ أَصْحابُ التَّبابينِ

البلاذري، أنساب الأشراف. تح: زكار وآخر، ج: 5، ص: 49.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سيبَوَيْهِ (ت: 180هـ=796م):

"وإِذا حَقَّرْتَ الـمُسَرْوَلَ، فَهُوَ مُسَيْريلٌ".

سيبويه، الكتاب. تح: هارون، ج: 3، ص: 433.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيّ (ت: 1205هـ=1790م):

"سَرْوَلَ فَهُوَ مُسَرْوَلٌ؛ أَي: لَبِسَ السِّرْوالَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزبيدي، تاج العروس. تح: الحلو، ج: 29، ص: 198.— المصدر
1.3معنى من معاني «سَرْوَلَ»

وــــــــــ فُلانًا: أَلبَسَهُ السِّرْوالَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=709م) يَصِفُ مِشْيَةَ المُخْتالِ الفَخورِ:

يَمْشي كَمَشْيِ الـمَرِحِ الفِخّيرِ

سُرْوِلَ في سَراوِلِ الصُّفورِ

ديوان العجاج. تح: عزة حسن، ص: 236.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"وسَرْوَلْتُهُ: أَلْبَسْتُهُ إِيّاهُ؛ ‌فَتَسَرْوَلَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 242.— المصدر
02 —

سُرْوِلَ

فعل
2.1معنى من معاني «سُرْوِلَ»

* سُرْوِلَ الخَيْلُ والبَقَرُ ونَحْوُهُما: جاوَزَ بَياضُ تَحْجيلِهِ العَضُدَيْنِ والفَخِذَيْنِ أَوْ صارَ في قَوائِمِهِ سَوادٌ مِنْ شَعْرٍ أَوْ لَوْنٍ. (ولَمْ يُسْتَعْمَل هذا الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ الـمُشْتَقُّ مِنْهُ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=709م) يَصِفُ الأَطْلالَ وما نَزَلَ بِها مِنْ بَقَرٍ وَحْشِيٍّ:

تَبَدَّلَتْ عَيْنَ النِّعاجِ الخُذَّلِ

وكُلَّ بَرّاقِ الشَّوى مُسَرْوَلِ

ديوان العجاج. تح: عزة حسن، ص: 172.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنّى (ت: 209هـ=825م) يَصِفُ الفَرَسَ الأَبْلَقَ:

"فَإِنْ تَجاوَزَ البَياضُ إِلى العَضُدَيْنِ والفَخِذَيْنِ؛ فَهُوَ أَبْلَقُ ‌مُسَرْوَلٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 92.— المصدر
2.2معنى من معاني «سُرْوِلَ»

وــــــــــ الطَّيْرُ: أَلْبَسَ ريشُهُ ساقَيْهِ. (ولَمْ يُسْتَعْمَل هذا الفِعْلُ بِهذا المَعْنى مِنْ قَبْلُ، واسْتُعْمِلَ الـمُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (206ه=821م) مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانِيّ:

أَباضوا عَلَيْهِمْ هامَهُمْ ثُمَّ أَنْقَفوا
فِراخَ القَطا بيضَ النَّعامِ الـمُسَرْوَلا

الشيباني، الجيم. تح: الأبياري، ج: 1، ص: 78.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَبْدُ اللهِ بْنُ أَحْمَدَ النَّسَفِيُّ (ت: 710ه=1310م):

"ولِلمُحْرِمِ ذَبْحُ شاةٍ وبَقَرَةٍ وبَعيرٍ ودَجاجَةٍ وبَطٍّ أَهْلِيٍّ، وعَلَيْهِ الجَزاءُ بِذَبْحِ حَمامٍ مُسَرْوَلٍ وظَبْيٍ مُسْتَأْنَسٍ".

عبد الله بن أحمد النسفي، كنز الدقائق. تح: بكداش، ص: 241.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُصْطَفى صادِق الرّافِعِيّ (ت: 1356هـ=1937م) يَصِفُ حَمامَةً:

"وكانَتْ كَالعَروسِ ‌مُسَرْوَلَةً، قَدْ غابَتْ ساقاها في الرّيشِ".

الرافعي، وحي القلم. تح: الجويدي، ج: 1، ص: 123.— المصدر
2.3معنى من معاني «سُرْوِلَ»

وــــــــــ فُلانٌ وغَيْرُهُ الشَّيْءَ، وبِهِ: صارَ مِنْهُ بِمَنْزِلَةِ السِّرْوالِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النّابِغَةُ الجَعْدِيُّ (ت: 50هـ=670م) يُشَبِّهُ ناقَتَهُ بِالمَهاةِ الَّتي ظَهَرَ الوَشْمُ في ساقِها بِمَنْزِلَةِ السِّرْوالِ:

كَأَنَّها بَعْدَما جَدَّ النَّجاءُ بِها
بِالشَّيِّطَيْنِ مَهاةٌ سُرْوِلَتْ رَمَلا

ديوان النابغة الجعدي. تح: واضح الصمد، ص: 139.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ جُرَيْجٍ (ت: 150ه=766م):

"ولِجِبْريلَ سِتَّةُ أَجْنِحَةٍ: جَناحٌ بِالـمَشْرِقِ، وجَناحٌ بِالـمَغْرِبِ، وجَناحانِ عَلى عَيْنَيْهِ، وجَناحانِ مِنْهُمْ مَنْ يَقولُ: عَلى ظَهْرِهِ، ومِنْهُمْ مَنْ يَقولُ: مُسَرْوَلًا بِهِما".

السيوطي، الحبائك. تح: محمد السعيد بسيوني، ص: 202.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيُّ (ت: 1205هـ=1791م) يَشْرَحُ بَيْتًا شِعْرِيًّا:

"شَبَّهَ هَذِهِ البَقَرَ البيضَ الـمُسَرْوَلَةَ بِالسَّوادِ بِسَبْيِ الرّومِ؛ لِبَياضِهِمْ ولِباسِهِمُ الأَخْفافَ السّودَ".

الزبيدي، تاج العروس. تح: حجازي، ج:5، ص: 420.— المصدر
03 —

تَسَرْوَلَ

فعل
3.1معنى من معاني «تَسَرْوَلَ»

* تَسَرْوَلَ فُلانٌ: لَبِسَ السِّرْوالَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«يَرْحَمُ اللهُ الـمُتَسَرْوِلاتِ».

مصنف عبد الرزاق. تح: الأعظمي، ج: 3، ص: 132.— المصدر
”وقالَتْ مَشْيَخَةٌ مِنَ الأَنْصارِ (11هـ=632م) يُحَدِّثونَ النَّبِيَّ ﷺ:

"يا رَسولَ اللهِ، إِنَّ أَهْلَ الكِتابِ يَتَسَرْوَلونَ، ولا يَأْتَزِرونَ".

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 36، ص: 623.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عُمَرُ بْنُ حَمْزَةَ العَدَويُّ (ت: 140هـ=758م) يَذْكُرُ:

"أَنَّ ‌سالِمًا ‌اطَّلى مَرَّةً، وتَسَرْوَلَ أُخْرى".

[سالمٌ: هو ابنُ عبدِ الله بنِ عُمَرَ،عَمُّ حمزةَ الراوي عنه]
ابن أبي شيبة، الكتاب المصنف. تح: الحوت، ج: 1، ص: 105.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ البَيْضاوِيُّ (ت: 685هـ=1286م) يُوَفِّقُ بَيْنَ خَبَرِ اسْتِلْقاءِ النَّبِيِّ ﷺ بِالمَسْجَدِ واضِعًا إِحْدى رِجْلَيْهِ عَلى الأُخْرى وحَديثٍ لَهُ يَنْهى فيهِ عَنْ ذَلِكَ:

"‌والتَّوْفيقُ ‌بَيْنَ ‌هَذا ‌الحَديثِ ‌وبَيْنَ ‌ما ‌رَوى ‌جابِرٌ ‌أَنَّهُ ‌قالَ -عَلَيْهِ ‌الصَّلاةُ ‌والسَّلامُ- :«لا ‌يَسْتَلْقِيَنَّ أَحَدُكُمْ، ثُمَّ يَضَعُ إِحْدى رِجْلَيْهِ عَلى الأُخْرى »؛ أَنْ يُقالَ: إِنَّهُ ﷺ لَـمّا فَعَلَ ذَلك كانَ مُتَسَرْوِلًا".

البيضاوي، تحفة الأبرار. تح: طالب وآخرين، ج: 3، ص: 213.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ سَعيدُ بْنُ مُحَمَّد باعِشِن (ت: 1270هـ=1854م) يَذْكُرُ بَعْضَ أَحْكامِ الصَّلاةِ:

"ويُسْتَحَبُّ لِلذَّكَرِ أَنْ يَلْبَسَ أَحْسَنَ ثِيابِهِ ويَتَقَمَّصَ ويَتَعَمَّمَ ويَتَطَيْلَسَ ويَرْتَدِيَ ويَتَّزِرَ أَوْ يَتَسَرْوَلَ".

سعيد باعشن، شرح المقدمة الحضرمية، ص: 264.— المصدر
3.2معنى من معاني «تَسَرْوَلَ»

وــــــــــ بِالشَّيْءِ: اتَّخَذَهُ عَلى هَيْئَةِ السِّرْوالِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَرْطَأَةُ بْنُ كَعْبٍ البَكّاءُ (ت: 15هـ=636م) يَصِفُ أَجْنِحَةَ المَلَكِ إِسْرافيلَ:

"إِنَّ أَقْرَبَ الـمَلائِكَةِ إِلى اللهِ إِسْرافيلُ، ‌ولَهُ ‌أَرْبَعَةُ ‌أَجْنِحَةٍ: جَناحٌ بِالـمَشْرِقِ، وجَناحٌ بِالـمَغْرِبِ، وقَدْ تَسَرْوَلَ بِالثّالِثِ، والرّابِعُ بَيْنَهُ وبَيْنَ اللَّوْحِ الـمَحْفوظِ".

تفسير يحيى بن سلام. تح: هند شلبي، ج: 1، ص: 50.— المصدر
3.3معنى من معاني «تَسَرْوَلَ»

وــــــــــ سَراويلَهُ: لَبِسَها.

”قالَ ابْنُ السِّكّيتِ (ت: 244ه=858م):

"وقَدْ تَسَرْوَلَ سَراويلَهُ".

ابن السكيت، الألفاظ. تح: قباوة، ص: 495.— المصدر
04 —

سِرْوال

اسم
4.1معنى من معاني «سِرْوال»

* السِّرْوالُ: لِباسٌ مِنْ ساقَيْنِ طَويلَتَيْنِ يَسْتُرُ النِّصْفَ الأَسْفَلَ مِنَ الجِسْمِ، وهُوَ بِتاءٍ أَيْضًا، (ج) سَراويلُ. وقيلَ: إنَّ السَّراويلَ مُفْرَدٌ، جَمْعُهُ سَراويلاتٌ، وهذا الأشْهَرُ في الاسْتعمالِ عَنِ العَرَبِ، وهُوَ يُذَكَّرُ ويُؤَنَّثُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ بِشْرُ بْنُ عَمْرٍو (ت: 57ق هـ=566م) حينَ هَمَّ رَجُلٌ لِيَقْتُلَهُ:

"أَجِرْ لي سَراويلي، فَإِنّي لَمْ أَسْتَعْنِ".

ابن سيده، المخصص. تح: جفال، ج: 1، ص: 87.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ المُزَرِّدُ بْنُ ضِرارٍ الغَطَفانِيُّ (ت: 10هـ=631م) يَصِفُ نَصيبَهُ بَعْدَ أَنْ ضنَّ قَوْمُهُ بِمالِهِمْ عَلَيْهِ:

‌فَكانَتْ ‌سَراويلٌ ‌وجُرْدٌ ‌خَميصَةٌ
وخَمْسُ مِئٍ مِنْها قَسِيٌّ وزائِفُ

ديوان المزرد. تح: العطية، ص: 53.— المصدر
”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) ‌‌يَذْكُرُ ما يَلْبَسُهُ المُحْرِمُ مِنَ الثِّيابِ:

«ومَنْ ‌لَمْ ‌يَجِدْ ‌إِزارًا، فَلْيَلْبَسْ سَراويلَ».

مالك بن أنس، الموطأ. تح: الأعظمي، ج: 1، ص: 469.— المصدر

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ العِزُّ بْنُ عَبْدِ السَّلامِ (ت: 660هـ=1262م) يَذْكُرُ ما يَجِبُ عَلى الزَّوْجِ مِنَ الكِسْوَةِ لِأَهْلِهِ:

"فَتَجِبُ في الصَّيْفِ ‌سَراويلُ وقَميصٌ وخِمارٌ".

العز بن عبد السلام، الغاية. تح: الطباع، ج: 6، ص: 209.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ المُخْتارُ الكُنْتِيّ (ت: 1226هـ=1811م) يُوَضِّحُ حُكْمَ ارْتِداءِ الحُبْوَةِ؛ وهِيَ الشَّمْلَةُ المُدارَةُ عَلى الظَّهْرِ والسّاقَيْنِ:

"وإِذا كانَ ﷺ نَهى عَنْها خَوْفَ التَّعَرّي، فَإِذا كانَ تَحْتَها قَميصٌ أَوْ إِزارٌ أَوْ سَراويلُ، انْتَفَتِ الحُرْمَةُ، وبَقِيَ النَّظَرُ في الكَراهَةِ".

المختار الكنتي، فتح الودود. تح: مأمون محمد، ج: 1، ص: 494.— المصدر
4.2معنى من معاني «سِرْوال»

o وسَراويلُ قَيْسٍ: يُضْرَبُ بِها الـمَثَلُ في الطّولِ؛ وقَيْسٌ هُوَ ابْنُ سَعْدِ بْنِ عُبادَةَ الأنْصارِيُّ، كانَ شَديدَ الطّولِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مَنْصورٍ الثَّعالِبِيُّ (ت: 429هـ=1038م):

‌"سَراويلُ قَيْسٍ: يُضْرَبُ مَثَلًا لثَوْبِ الرَّجُلِ الضَّخْمِ الطَّويلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الثعالبي، ثمار القلوب. ص: 601.— المصدر
4.3معنى من معاني «سِرْوال»

o وشمَّرتِ الحرب عن ساقٍ وسروال: اشتدَّ وَطيسُها واسْتَحَرَّ القَتْلُ فيها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الواقِدِيُّ (ت: 207هـ=822م) يَصِفُ هَوْلَ واقِعَةِ فَتْحِ قَلْعَةِ رَأْسِ العَيْنِ:

"‌والشَّمْسُ ‌تَضْجَرُ مِنْها الجُسومُ والنُّفوسُ، والحَرْبُ قَدْ أَخَذَتْ أَمْرًا يَقْطَعُ الآجالَ، وقَدْ شَمَّرَتْ عَنْ ساقٍ وسِرْوالٍ".

الواقدي، فتوح الشام. تح: عبد اللطيف عبد الرحمن، ج: 2، ص: 134.— المصدر
