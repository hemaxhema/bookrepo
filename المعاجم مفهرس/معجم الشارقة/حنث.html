01 —

حَنِثَ

فعل
1.1معنى من معاني «حَنِثَ»

*حَنِثَ (بِكَسْرِ النّونِ) فُلانٌ يَحْنَثُ (بِفَتْحِ النّونِ) حَنَثًا وحِنْثًا: مالَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَريُّ (ت: 370 هـ=981م):

"يُقالَ: قَدْ حَنِثْتَ، أَيْ: مِلْتَ إِلى هَواكَ عَلَيَّ، وقَدْ حَنِثْتَ مَعَ الحَقِّ عَلى هَواكَ".

الأَزْهَريّ، تَهْذيبُ اللُّغَةِ، تح: محمد عوض مرعب، ج: 4، ص: 278.— المصدر
1.2معنى من معاني «حَنِثَ»

وــــــــــ في يَمينِهِ: لَمْ يَبِرَّ فيها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيْدٍ العِبادِيُّ (ت: 35ق ه=588م):

لَيْسَ شَيْءٌ على المَنُونِ بِباقِ
غَيْرُ وَجْهِ المُسَبَّـحِ الخَلّاقِ

فبَريءٌ صَدْرِي مِنَ الظُّلْمِ لِلرَّ
بِّ وحِنْثٍ بمُعْقَدِ المِيثاقِ

ديوان عدي بن زيد. تح: المعيبد، ص: 150.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وفي القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ ﷻ:

﴿وخُذْ بِيَدِكَ ضِغْثًا فاضْرِبْ بِهِ ولَا تَحْنَثْ﴾. (ص: 44).

”وقالَ النَّبيُّ ﷺ (ت: 11 هـ=632م):

«مَنْ حَلَفَ وَاسْتَثْنَى، إِنْ شَاءَ رَجَعَ، وإِنْ شَاءَ تَرَكَ ‌غَيْرُ ‌حَانِثٍ».

ابن ماجه، صحيح سنن ابن ماجه. تح: عبد الباقي، ج: 1، ص: 680.— المصدر
”وقالَ فَرْعانُ بْنُ الأَعْرَفِ (ت 23 هـ=644م) يَذْكُرُ ابْنَهُ العاقَّ:

أَيَظْلِمُني حَقِّي ويَحْنثُ أُلْوَتي؟
وسَوْفَ يُلاقِي رَبَّهُ فَيُحاسِبُهْ

الخليل، العين. تح: المخزومي، ج: 8، ص: 356.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والحِنْثُ، إِذا لَمْ يُبِرَّ بِيَمِينِهِ، وقَدْ ‌حَنِثَ ‌يَحْنَثُ"

الخليل، العين. تح: المخزومي، ج: 3، ص: 206.— المصدر
”وقالَ مالِكُ بْنُ أنَسٍ (ت: 179ه=795م):

"والكَفّارةُ بَعْدَ الحِنْثِ أَحَبُّ إلَيَّ".

مالك بن أنس، المدونة الكبرى. دار الكتب العلمية. ج2، ص: 579.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّارِ (ت: 658ه=1260م) يُتَرْجِمُ لابنِ غَطُّوسٍ:

" وكانَ قَدْ آلَى على نَفْسِهِ أَلّا يَخُطَّ حَرْفًا مِنْ غَيْرِهِ، ولا يَخْلِطَ بِهِ غَيْرَهُ تَقَرُّبًا إِلى اللهِ، وتَنْزِيهًا لِتَنْزِيلِهِ، فَما حَنِثَ، واللهُ أَعْلَمُ".

ابن الأبار، التكملة لكتاب الصلة. تح: بشار عواد ، ج: 2، ص: 296.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد أَمين الزُّلَلِي (ت: 241ه=1826م):

وأُقْسِمُ إنِّي لا أَميلُ عَنِ الهَوى
وما أَنا في هذي الألِيَّةِ حانِثُ

ديوان محمّد أمين الزُّلليّ. تح: محمد العيد الخطراوي، ج: 1، ص 183.— المصدر
02 —

أحْنَثَ

فعل
2.1معنى من معاني «أحْنَثَ»

*أَحْنَثَ فُلانٌ في يَمِينهِ: أَخْلَفَ البِرَّ فِيها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَحْمَدُ بْنُ حَنْبَلٍ (ت: 241ه=855م):

"وسَأَلْتُهُ: إِلى أَيِّ شَيْءٍ تَذْهَبُ في المَشْيِ؟ قالَ: كَفَّارةُ يَمِينٍ، ولا أَحْمِلُهُ على الحِنْثِ، فإِنْ أَحْنَثَ أَطْعَمَ عَشَرَةَ مَساكينَ".

أحمد بن حنبل، مسائل الإمام أحمد رواية ابنه صالح. تح: فضل الرحمن، ج: 1، ص: 133.— المصدر
2.2معنى من معاني «أحْنَثَ»

وــــــــــ فُلانًا: جَعَلَهُ لا يَبَرُّ في يَمِينِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (3هـ=625م) يُخاطِبُ السَّيِّدَةَ عائِشَةَ في شَأْنِ امْرَأَةٍ أقْسَمَتْ عَلَيْها أنْ تَأكُلَ التَّمْرَ الَّذي أهْدَتْها إيَّاهُ:

"أَبِرِّيها، فَإِنَّ الإِثْمَ على المُحْنِثِ".

مسند أحمد. تح: الأرناؤوط، وآخرون، ج: 41، ص: 331.— المصدر
”وقالَتِ الخَنْساءُ (ت: 24ه=645م) تَرْثي أخاها:

حَلَفْتَ على أَهْلِ اللِّواءِ لَيوضَعَنْ
فَما أَحْنَثَتْكَ الخَيْلُ حَتّى أَبَرَّتِ

ديوان الخنساء، تح: لويس شيخو اليسوعي، ص: 20— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ جُرَيْجٍ (150ه=766م):

"أُخْبِرْتُ أَنَّ مَوْلاةً لعائِشَةَ أُمِّ المُؤْمِنينَ أَقْسَمَتْ عَلَيْها في قَديدَةٍ تَأْكُلُها، فأحْنَثَتْها عائِشَةُ فجَعَلَ النَّبِيُّ ﷺ تَكْفيرَ اليَمِينِ على عائِشَةَ".

الصنعاني، المصنف، تح: مركز البحوث وتقنية المعلومات، ج7، ص: 250.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أحْمَدُ بْنُ يَحْيى الوَنْشريسي (ت: 914ه=1508م):

" ... ثُمَّ تَحَيَّلَتِ الزَّوْجَةُ المَذْكُورَةُ على نَفْسِها خِيفَةً مِنْهُ، وارْتَحَلَتْ إِلى فاسٍ المَذْكُورةِ قاصِدَةً بِذَلِكَ إِحْناثَهُ، ومُخالَفَتَهُ فيما أَوْجَبَهُ على نَفْسِهِ بالطَّلاقِ مَتى أَسْكَنَها بالمَدِينةِ المَذْكورَةِ".

الونشريسي، المعيار المعرب. تح: محمد حجي، وآخرين، ج: 4، ص: 509.— المصدر
2.3معنى من معاني «أحْنَثَ»

وــــــــــ الفَقِيهُ فُلانًا: عَدَّهُ ناكثًا في يَمينِهِ وألْزَمَهُ الكَفّارَةَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّافِعِيُّ (ت: 204ه=819م):

" ... فقَدْ يُحْدِثُ على مِثالِها، وعلى خِلافِ مِثالِها، فلَمّا كانَ هكَذا لَمْ أُحْنِثْهُ على سَبَبِ يَمينِهِ، وأُحْنِثُهُ على مَخْرَجِ يَمِينِهِ".

الشافعي، الأم. تح: محمد زهري النجار، ج: 7، ص: 73.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أحْمَدُ بْنُ يَحْيى الوَنْشريسي (ت: 914ه=1508م):

"أَمّا مَسْألَةُ الحالِفِ المَذْكورِ، فَإِنْ كانَ كلَّمَهُ بَعْدَ اسْتِحْقاقِ البَقَرَةِ فقَدْ حَنِثَ بالثَّلاثِ، وإِنْ كانَ لَمْ يُكَلِّمْهُ بَعْدُ، فَفي إِحْناثِهِ بكَلامِهِ، بَعْدَ دَفْعِ البَقَرَةِ، وقَبْلَ اسْتِحْقاقِها، خِلافٌ مَبْنيٌّ على مُراعاةِ اللَّفْظِ، أَوِ المَقْصِدِ".

الونشريسي، المعيار المعرب. تح: محمد حجي، وآخرين، ج: 4، ص 111.— المصدر
03 —

حانَثَ

فعل
3.1معنى من معاني «حانَثَ»

*حانَثَ فُلانٌ: خالَفَ في يَمينِهِ وناقَضَها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُرْوةُ بْنُ أُذَيْنَةَ (ت: 130 هـ=748م) يَصِفُ مَحْبوبَتَهُ:

تُطيعُ إِذا النُّصْحُ يَوْمًا بَدَا
وتَأْبَى مِرارًا فَتَعْصِي حِناثَا

ديوان عُرورة بن أذينة. تح: يحيى الجبوري، ص: 303.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبةُ بْنُ العَجّاجِ (ت: 145ه=762م):

وحَـــلَّ شَدَّ العُقَدِ المُحانِثُ

وعاثَ فينا مُسْتَحِلٌّ عايِثُ

شرح ديوان رؤبة بن العجاج، تح: د. ضاحي عبد الباقي محمد، وآخر، ص: 73.— المصدر
04 —

حَنَّثَ

فعل
4.1معنى من معاني «حَنَّثَ»

*حَنَّثَ فُلانٌ اليَمِينَ: نَقَضَها ولَمْ يَبَرَّ بِها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُدَيُّ بْنُ حاتِمٍ (68ه=687م):

"أَما والله، لَوْلا أَنّي سَمِعْتُ رَسُولَ الله، ﷺ، يَقولُ: مَنْ حَلَفَ على يَمِينٍ، ثُمَّ رَأَى أَتْقَى للهِ مِنْهَا، فَلْيأْتِ التَّقْوَى ما حَنَّثْتُ يَمِيني".

القاضي عياض، إكمال المعلم. تح: يحيى اسماعيل، ج: 5، ص: 411.— المصدر
4.2معنى من معاني «حَنَّثَ»

وــــــــــ فُلَانٌ غَيْرَهُ: جَعَلَهُ يَنْكُثُ في يَمينِهِ ويَنْقُضُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سَحْنونُ (ت: 240ه=855م):

"... وهذا يَدُلُّ على أَنَّ الحَجَّ عِنْدَهُ على التَّراخي، إِذْ لَوْ كانَ عِنْدَهُ على الفَوْرِ لَما شَكَّ في تَعْجِيلِ تَحْنِيثِ الزَّوْجِ في أَوَّلِ العامِ".

مالك بن أنس، المدونة الكبرى، ج: 5، ص: 201.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أحْمَدُ بْنُ مُحَمَّدٍ القُرْطُبِيُّ (ت: 671ه=1273م) يَشْرَحُ حَديثَ رَسُولِ الله ﷺ »واللَّهِ لَأنْ يَلَجَّ أحَدُكُمْ بِيَمِينِهِ في أهْلِهِ، آثَمُ لَهُ عِنْدَ اللَّهِ مِن أنْ يُعْطِي كَفّارَتَهُ التي فَرَضَ اللَّهُ .«

" ... فَإِنْ كانَ شَيْءٌ مِنْ ذَلِك، فَالْأَوْلى بِهِ تَحْنيثُ نَفْسِهِ، وفِعْلُ الكَفَّارَةِ".

القرطبي، الجامع لأحكام القرآن. تح: عبد الله التركي، ج: 8، ص: 150.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ التُّسولِيُّ (ت: 1258ه=1842م):

"قالَ بَعْضُ أَصْحابِنا: إِذا قالَ لامْرَأَتِهِ:‏ أَنْتِ طالِقٌ، أَوْ لِعَبْدِهِ:‏ أَنْتَ حُرٌّ ،إِنْ فَعَلْتِ أَوْ فَعَلْتَ لِقَصْدِ تَحْنيثِهِ لا يَلْزَمُهُ طَلاقٌ، ولا عِتْقٌ".

التسولي، البهجة. تح: شاهين، ج: 1، ص: 570.— المصدر
05 —

تَحَنَّثَ

فعل
5.1معنى من معاني «تَحَنَّثَ»

*تَحَنَّثَ فُلانٌ: تَعَبَّدَ واتَّقى الإثْمَ والوقوعَ فيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَكيمُ بْنُ حِزامٍ (8ه=629م) مُخاطِبًا النَّبيَّ ﷺ يَوْمَ الفَتْحِ:

"يا رَسُولَ اللّهِ، أَرَأَيْتَ أُمُورًا كُنْتُ أَتَحَنَّثُ بِها في الجاهِلِيَّةِ، ما لي فيها مِنْ شَيْءٍ؟ فَقالَ لَهُ رَسُولُ اللهِ ﷺ: أَسْلَمْتَ على ما أَسْلَفْتَ مِنْ خَيْرٍ".

مسند أبي عوانة. تح: أيمن عارف الدمشقي، ج: 1، ص: 72.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الحَكيمُ التِّرْمِذِيُّ (ت: 320ه=932م) يَذْكُرُ حالَ النَّبِيِّ ﷺ قَبْلَ الرِّسالَةِ:

" ... فكانَ يَخْلُو بغارِ حِراءٍ للتَّحَنُّثِ فيهِ إِلى أَنْ أَرْسَلَهُ اللهُ للنّاسِ كافَّةً".

الحكيم الترمذي، ختم الأولياء. تح: عثمان اسماعيل، ج: 1، ص: 283.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرّازِيُّ (ت: 666 هـ=1268م):

"‌تَحَنَّثَ: ‌تَعَبَّدَ وَاعْتَزَلَ الْأَصْنَامَ، مِثْلُ تَحَنَّفَ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الرازي، مختار الصحاح. تح: يوسف الشيخ محمد، ج: 1، ص: 83.— المصدر
06 —

حِنْثٌ

اسم
6.1معنى من معاني «حِنْثٌ»

*الحِنْثُ: ما يُتَحَرَّجُ مِنْهُ مِنْ ذَنْبٍ وغَيْرِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ كُلَيْبُ بْنُ رَبيعَةَ (ت: 134ق هـ=488م):

فظَنُّوا أنَّنِي بالحِنْثِ أوْلى
وأنّي كُنْتُ أوْلى بالنَّجاحِ

التّبريزيّ، شرح ديوان الحماسة لأبي تمّام. تح: غريد الشّيخ وآخرين، ج: 1، ص: 590.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ ﷻ:

﴿وَكَانُواْ يُصِرُّونَ عَلَى ٱلۡحِنثِ ٱلۡعَظِيمِ﴾. (الواقعة:46)



ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"الحِنْثُ: الذَّنْبُ العَظيمُ". (وَهَذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 3، ص: 206.— المصدر
6.2معنى من معاني «حِنْثٌ»

وــــــــــ: الإدراكُ والبُلوغُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11ه=632م):

"ما مِنْ مُسْلِمَيْنِ يَمُوتُ لَهُما ثَلاثَةُ أَوْلادٍ لَمْ يَبْلُغُوا الحِنْثَ إِلّا أَدْخَلَهُما اللَّهُ وإِيّاهُمْ بِفَضْل رَحْمَتِهِ الجَنَّةَ".

مسند أحمد. تح: الأرنؤوط وآخرين، ، ج: 16، ص: 364.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170 هـ= 787م):

"ويُقالَ: بَلَغَ الغُلامُ ‌الحِنْثَ، أَيْ: بَلَغَ مَبْلغًا جَرَى عَلَيْهِ القَلَمُ في المَعْصِيَةِ والطَّاعَةِ". ‌

الخليل، العين. تح: المخزومي، ج: 3، ص: 206.— المصدر
”وقالَ أَبو الحَسَنِ المَدائِنِيُّ (220ه=835م):

" كانَ الوَلِيدُ حِينَ بايَعَ لَهُ أَبوهُ بَعْدَ هِشامٍ ابنَ إِحْدَى عَشْرَةَ سَنَةً، قَالُوا:‏ لَمّا بلَغَ الوَلِيدُ الحِنْثَ نَدِمَ أَبوهُ على تَوْلِيَةِ هِشامٍ".

البلاذري، أنساب الأشراف. تح: سهيل زكّار، وآخر، ج: 9، ص: 3743.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711ه=1311م) يَذْكُرُ قَوْمَ نوحٍ:

"إِنَّ اللهَ أَعْقَمَ رِجالَهم قَبْلَ الطُّوفانِ بِأَرْبَعينَ عامًا، وأَعْقَمَ نِساءَهُمْ فلَمْ يَتَوَلَّدُوا أَرْبَعِينَ عامًا مُنْذُ يَوْمِ دَعا نُوحٌ حتَّى أَدْرَكَ الصَّغِيرُ فَبَلَغَ الحِنْثَ".

ابن منظور، مختصر تاريخ دمشق. ج: 26، ص: 198.— المصدر

7.1معنى من معاني «مُحْنِثٌ»

*المُحْنِثُ: الشَّيءُ المُخْتَلَفُ فيهِ. (وانْظُرُ: ح ل ف)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّاحِبُ ابْنُ عَبَّادٍ (ت: 385 هـ=995م):

"وكُلُّ شَيْءٍ يُخْتَلَفُ فيه عند العَرَبِ فهو ‌مُحْنِثٌ ومُحْلِفٌ".

ابْنُ عَبَّادٍ، المُحيطُ. تح: آل ياسين، ج: 3، ص: 75.— المصدر
