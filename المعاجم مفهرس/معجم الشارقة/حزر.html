النقوش العربيّة

وَرَدَ الجَذْرُ (ح ز ر ح ز ر ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَيْنِها نَقْشٌ سَبَئِيٌّ يَعُودُ تاريخُهُ إلى القَرْنِ الثّاني المِيلادِيِّ (Bajda 1/2)، وَرَدَ فيهِ الاسْمُ (ح ز ر ḥzr ) عَلَمًا على أحَدِ الأفْرادِ، فَجاءَ في النَّقْشِ:"أ ح م د/ ع ب د .../ ب ن/ ح ز ر" أيْ: " أحْمَدُ عَبْدُ... بْنُ حَزَرٍ".

(الحاج، 2017م: 206-207)

النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السَّاميَّاتُ بِهذا الجَذْرِ مِنَ العَرَبِيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (ح ز ر):

في العِبريّة: وَرَدَ الاسْمُ: חֲזֶרֶת (حَز۪ر۪ت) ḥazeret بِمَعْنى (نَوْعٌ مِنَ الخَسِّ أو الفُجْلِ اللّاذِعِ) ويُقارِبُ الاسْمَ العَرَبِيَّ (حَزْرَة) معنًى ويُوافقُهُ مَبْنًى، ومِنْ مَجيئِهِ بِمَعْنى (حزرة) ما وَرَدَ في المشنا: "וְאֵלּוּ יְרָקוֹת שֶׁאָדָם יוֹצֵא בָהֶן יְדֵי חוֹבָתוֹ בַפֶּסַח: בַּחַזֶּרֶת וּבָעֻלְשִׁים..."، أيْ: "وتِلْكَ الخَضْراواتِ التي إذا أكَلَها المَرْءُ في الفِصْحِ بَرِئَتْ ذِمَّتُهُ: الحَزْرةُ والعَلَثُ...".

(משנ': סד' מועד, מסכ' פסחים, פר' ב', משנ' ו')

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܚܶܙܪܳܐ (ح۪زرَا) ḥezrā بِمَعْنى (الخَشْكَرُ/ الدَّقيقُ الخَشِنُ) ويُوافِقُ الاسْمَ العَرَبِيَّ (حازر) مَبْنًى ومَعْنًى.

(ܒܰܪ ܠܶܒ̊ܳܐ: ܚܙܪ)

01 —

حَزَرَ

فعل
1.1معنى من معاني «حَزَرَ»

* حَزَرَ (بفتح الزّاي) اللَّبَنُ ونَحْوُهُ يَحْزُرُ (بضَمِّ الزّاي) حُزورًا: حَمِضَ. فهو حازِرٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ: أبو النَّجْمِ العِجْلِيُّ (ت: 130هـ=748م) يُخاطِبُ امْرَأةً:

وأمِّري الأُنْثى عَلَيْكِ والذَّكَرْ

وارْضَيْ بِإحْلابَةِ وَطْبٍ قَدْ حَزَرْ

العجلي، ديوانه. تح: محمد جمران، ج: 1، ص: 165.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَتِ امْرَأةُ امْرِئ القيسِ (نحو 80ق. ه= 544م) تحكي قِصَّةَ تَمْييزِها بين امرئ القَيْسِ وغُلامِهِ:

"اسْقوهُ لَبنًا حازِرًا (...) فسَقَوْهُ فَشَرِبَ".

الأصبهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 9، ص: 76.— المصدر
”قالَ الخليلُ (ت: 170هـ=878م):

"والصَّريفُ: اللَّبَنُ الحازِرُ الحامِضُ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الخليل، العين. تح: المخزومي، ج: 5، ص: 107.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ البيضاويّ (ت: 685هـ=1286م) يُفَسِّرُ قَوْلَه تَعالى: ﴿وأنْهارٌ مِنْ لَبَنٍ لَمْ يَتَغَيَّرْ طَعْمُهُ﴾ (محمد: 15):

"لَمْ يَصِرْ قارِصًا ولا حازِرًا".

البيضاويّ. أنوار التنزيل. ج: 5، ص: 121.— المصدر
1.2معنى من معاني «حَزَرَ»

وــــــــــ الوجْهُ: عَبَسَ وبَسَرَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الجَنْدَلُ بْنُ المُثَنّى (ت: 90هـ=709م) يخاطب امرَأتَهُ:

قامَتْ تُعَنْظي بِكِ سِمْعَ الحاضِرِ

تَرْمي البَذاءَ بِجَنـانٍ واقِـــرِ

وشِدَّةَ الصَّوْتِ بِوجْــهٍ حازِرِ

ابن السِّكّيت ، كتاب الألفاظ. تح: قباوة، ج: 1، ص: 177.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزهري (ت: 370هـ=981م):

"ووجْهٌ حازِرٌ: عابِسٌ باسِرٌ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 4، ص: 359.— المصدر
1.3معنى من معاني «حَزَرَ»

وــــــــــ المالُ ونَحْوُهُ: زَكا وثَبَتَ ونَما.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن سيده (ت: 458هـ=1066م):

"ولَمْ يُفَسِّرْ حَزَرَ، غَيْرَ أنّي أظُنّهُ زَكا أو ثَبَتَ فنَما". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 3، ص: 220.— المصدر
1.4معنى من معاني «حَزَرَ»

وــــــــــ فُلانٌ الشَّيْءَ يَحْزُرُهُ ويَحْزِرُهُ (بِضَمِّ الزّاي وكَسْرِها) حَزْرًا ومَحْزَرَةً: قَدَّرَهُ بِالحَدْسِ والتَّخْمينِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عبدُ اللهِ بنُ رَواحَةَ (ت: 8هـ=629م) يَرُدُّ عَلى يَهودِ خَيْبَرَ اعْتِراضَهُمْ عَليْه في خَرْصِ نَخْلِهِمْ:

"فَأنا أَلي حَزْرَ النَّخْلِ وأُعْطيكُمْ نِصْفَ الَّذي قُلْتُ، قالوا هَذا الحَقُّ".

سنن أبي داود. تح: شعيب الأرنؤوط وآخر، ج: 5، ص: 287.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن المقفع (ت: 142هــ=759م):

"فقالَ: أنا بائِعٌ ما في هذِهِ الجَرَّةِ بِدينارٍ، فأشْتَري بِالدّينارِ عَشْرَةَ أعْنُزٍ، فيَحْمِلْنَ ويَلِدْنَ لِسِتَّةِ أشْهُرٍ، ثُمَّ حَزَرَ عَلى هذا الحِسابِ لِخَمْسِ سِنينَ، فوجَدَ ذلِكَ أكْثَرَ مِنْ أرْبَعِمِائَةِ عَنْزٍ".

ابن المقفع، كليلة ودمنة. تح: مجموعة محققين، ص: 232.— المصدر
”قالَ القرطبي (ت: 656هـ=1258م) يَذْكُرُ أضْرُبَ الكَهانَةِ عِنْدَ العَرَبِ:

"الثّالِثُ: التَّخْمينُ ‌والحَزْرُ، وهذا يَخْلُقُ اللهُ فيهِ لِبَعْضِ النّاسِ شِدَّةَ قُوَّةٍ".

القرطبي، المُفهم. تح: محيي الدين مستو وآخرين، ج: 5، ص: 633.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ محمد بن سالم المجلسي (ت: 1302ه=1884م) في خَرْصِ النّخْلِ:

"صِفَةُ الخَرْصِ هي أن يَحْزِرَ الخارِصُ كُلَّ نَخْلَةٍ على حِدَتِها".

المجلسي، لوامع الدرر. ج: 3، ص: 347.— المصدر

وــــــــــ العَقْلَ وغَيْرَهُ: اخْتَبَرَهُ وامْتَحَنَهُ.

حازِر

5.1معنى من معاني «حازِر»

[القَعْبُ: القِدْحُ العَظيمُ]

”قالَ ابن قتيبة الدّينَوَريّ (ت: 276ه=889م):

"والزَّوارِحُ: الرَّوابي الصِّغارُ، واحِدُها زَرْوحٌ، والحَزاوِرُ مِثْلُها واحِدَتُها حَزْوَرَةٌ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، الجراثيم. تح: الحميدي، ج: 2، ص: 38.— المصدر
12.2معنى من معاني «حَزْوَرَة»

وــــــــــ: السّوقُ، ومِنْهُ سوقُ مَكَّةَ، وأدْخِلَتْ في المسْجِدِ لمّا زيدَ فيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابن شهاب الزهري (ت: 124هــ=742م) في فَتْحِ مَكّةَ في قومٍ من بني بَكْرٍ:

"وانْهَزَموا وقُتِلوا بِالحَزْوَرَةِ".

الزهري، مرويات الزهري في المغازي. ج: 2، ص: 728.— المصدر

"والحَزاوِرُ والجُباجِبُ الأسْواقُ".

الأزرقي ، تاريخ مكة. تح: رشدي الصالح ج: 2، ص: 294.— المصدر
د

"قالَ بَعْضُ المَكِّيّينَ: بَلْ كانَتِ الحَزْوَرَةُ في مَوْضِعِ السِّقايَةِ".

تقي الدين الفاسي المكي، العقد الثمين. ت: محمد عطا، ص: 55.— المصدر
12.3معنى من معاني «حَزْوَرَة»

وــــــــــ: الأرْضُ ذاتُ الحِجارَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دريد (ت: 321هـ=933م):

"والحَزْوَرَةُ: أرْضٌ ذاتُ حَصًى كِبارٍ ورَمْلٍ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 1179.— المصدر
12.4معنى من معاني «حَزْوَرَة»

وــــــــــ مِنَ النّوقِ:المُذَلَّلَةُ، أو العَظيمةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن عبّاد (ت: 385هــ=995م):

"والحَزْوَرَةُ، ‏مُخَفَّفَةً: ‏النّاقَةُ المُقَتَّلَةُ المُذَلَّلَةُ". (وهذه الدَّلالةُ سَجَّلَتْها المَعاجِمُ في عصر الدول والإمارات)

ابن عبّاد، المحيط. تح: محمد آل ياسين، ج: 3، ص: 6.— المصدر
13 —

حَزَوَّر

اسم
13.1معنى من معاني «حَزَوَّر»

* الحَزَوَّرُ: الرَّجُلُ القَوِيُّ الشَّديدُ. (ج) حَزاوِرَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ: عَنْترَةُ بن الأخْرَس (ت: 90ه=709م) يَرْثي خالِدِ بْنِ يزيدَ بنِ مُعاويَةَ:

لَقَدْ حَلَّقَتْ بِالجَوِّ فَتْخاءُ كاسِرٌ
كَعَنْقاءِ دَمْخٍ حَلَّقَتْ بِالحَزَوَّرِ

عنترة الأخرس، شعر طيئ وأخبارها. تح: وفاء فهمي، ص: 662.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ النّابِغَةُ الذُّبْيانيُّ (ت: 18ق.هـ=605م) يَصِفُ المُتَجَرِّدَةَ زَوْجَ النُّعْمانِ:

وإذا نَزَعْتَ نَزَعْتَ مِنْ مُسْتَحْصَفٍ
نَزْعَ الحَزَوَّرِ بِالرِّشاءِ المُحْصَدِ

البطليوسي، شرح الأشعار الستة الجاهلية. تح: محمد هاشم، ص: 377.— المصدر
”قالَ بشار بن برد (ت: 167هـ=784م):

وقائِلَةٍ: طالَبْتَ سَلْمى حَزَوَّرًا
إلى أنْ خَلَتْ سِنٌّ وزالَ طِلابُ

ديوان بشّار. تح: إحسان عباس، ج: 1، ص: 69.— المصدر
13.2معنى من معاني «حَزَوَّر»

وــــــــــ مِنَ الرِّجالِ: الضَّعيفُ. (ضِدٌّ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأحنفُ بْنُ قيسٍ (72هـ=691م) يَسْتَنْهِضُ هِمَمَ رِجالِهِ لِيُقاتِلوا المُشْرِكينَ:

أحَقُّ مَنْ لَمْ يَكْرَهِ المَنِيَّهْ

حَزَوَّرٌ لَيْسَتْ لَهُ ذُرِّيَّهْ

الطبري، تاريخ الأمم والملوك. ج: 2، ص: 631.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ قُطْرُبُ (ت: 206هــ=821م):

"والحَزَوَّرُ: الضَّعيفُ أيْضًا". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

قطرب، كتاب الأضداد. ص: 145.— المصدر
13.3معنى من معاني «حَزَوَّر»

وــــــــــ مِنَ الغِلْمانِ: الَّذي لَمْ يَحْتَلِمْ وقَدْ راهَقَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَت أسماء بنت أبي بكر(ت: 73هـ=693م) تَذْكُرُ زيدَ بنَ عَمْرٍو الذي كانَ أحَدَ المُوحِّدينَ في الجاهِلِيَّةِ:

"لَقَدْ رَأيَتُهُ وإنّي حَزَوَّرَةٌ وهْو مُسْنِدٌ ظَهْرَهُ إلى الكَعْبَةِ".

البغوي، معجم الصحابة. تح: محمد الأمين الجكني، ج: 2، ص: 447.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن السِّكّيت (ت: 244هـ=858م):

"ويُقالُ لِلْغُلامِ الَّذي كادَ يُدْرِكُ ولَمْ يَفْعَلْ: هو غُلامٌ حَزَوَّرٌ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السِّكّيت ، ترتيب إصلاح المنطق. تح: محمد حسن بكائي، ص: 257.— المصدر
13.4معنى من معاني «حَزَوَّر»

وــــــــــ: الفَرَسُ المُتَقاعِسُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ: الكميت (ت: 126هـ=744م) يَصِفُ حَمالَةً احْتَمَلَها:

ولَمْ أكُ عِنْدَ مَحْمِلِها أزوحًا
كَما يَتَقاعَسُ الفَرَسُ الحَزَوَّرْ

[الأزوحُ: المُتَخَلِّفُ. الثَّقيلُ الذي يَزْحَرُ عِنْدَ الحَمْلِ.تَقاعَسَ: تَأخَّرَ ولَمْ يَتَقَدَّمْ]
الكميت، ديوانه. تح: نبيل طريفي، ص: 141.— المصدر
13.5معنى من معاني «حَزَوَّر»

وــــــــــ من الأقْدامِ: القَصِيرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عمرو الشيباني (ت: 206هـ=821م):

"إنَّهُ لَحَزَوَّرُ القَدَمِ؛ أيْ قَصيرُ القَدَمِ".

الشيباني، الجيم. تح: الإبياري، ج: 1، ص: 144.— المصدر

حَزيرَة

اسم
14.1معنى من معاني «حَزيرَة»

* الحَزيرَةُ مِنَ المالِ: خِيارُهُ. (ج) حَزائِرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاضي عَبد الجبار (ت: 415ه=1025م) يذكرُ أبا عَلِيٍّ الجُبّائِيَّ:

"وقَدْ بَيَّنَ أبو عَلِىٍّ، رَحِمَهُ اللهُ، الحالَ في أهْلِ الخَيالِ والحَزائِرِ والمِهَنِ".

القاضي عبد الجبار، المغني في أبواب التوحيد. تح: إبراهيم، ج: 12، ص: 343.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابن منظور (ت: 711هـ=1311م):

"وحَزْرَةُ المالِ: خِيارُهُ، وبِها سُمِّيَ الرَّجُلُ وحَزيرَتُهُ كَذلِكَ". (وهَذِهِ الدِّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، لسان العرب. تح: أمين محمد عبد الوهاب، وآخر، ج: 3، ص: 150.— المصدر
14.2معنى من معاني «حَزيرَة»

وــــــــــ: حَساءٌ منَ الطَّحينِ مَعَ اللَّحْمِ أو السَّمْنِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن سيده (ت: 458هــ=1066م):

"والصُّلَيْفاءُ كالغُرَيْراءِ عَلى لَوْنِها وفيها بَياضٌ وسَوادٌ والسُّرَيْطاءُ حَساءٌ كالحَزيرَةِ".

ابن سيده، المُخَصّص. تح: خليل جفال، ج: 5، ص: 49.— المصدر

الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
