علكس
عَلْكَسَ(فعل) اعْلَنْكَسَ(فعل) اعْلُنْكِسَ(فعل) عِلَّكْس(اسم) مُعْلَنْكِس(اسم)

المعانى الكلّيّة

  1. الاجْتِماعُ، والتَّراكُبُ
01 —

عَلْكَسَ

فعل
1.1معنى من معاني «عَلْكَسَ»

* عَلْكَسَ العُشْبُ وغَيْرُهُ: كَثُرَ، واجْتَمَعَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال اللَّيْثُ بنُ المُظَفَّرِ (ت: 180هـ=796م):

"والمـُعَلْكِسُ والمـُعْلَنْكِسُ مِنَ اليَبيسِ: ما كَثُرَ واجْتَمَعَ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: عبد الحليم النجار، ج: 3، ص: 302.— المصدر
02 —

اعْلَنْكَسَ

فعل
2.1معنى من معاني «اعْلَنْكَسَ»

* اعْلَنْكَسَ الشَّعْرُ: اشْتَدَّ سَوادُهُ، وكَثُرَ. (انْظُرْ: ع ر ك س)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال رَجُلٌ مِنْ طَيِّئٍ (نحو: 25 ق هـ= 598م) يَصِفُ شَعْرَ امْرَأَةٍ:

مُعْلَنْكِسٌ شَبَّ لَها لَوْنَها
كَما يَشُبُّ البَدْرَ لَوْنُ الظَّلامْ

شعر طيئ وأخبارها في الجاهلية والإسلام، تح: وفاء السنديوني، ص: 710.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاج (ت: 90هـ=709م) يَتَغَزَّلُ:

ولَهْوَةَ اللّاهي ولَوْ تَنَطَّسا

أَزْمانَ غَرّاءُ تَروقُ العُنَّسا

بِفاحِمٍ دُوِّيَ ‌حَتّى ‌اعْلَنْكَسا

ديوان العجاج، تح: عزّة حسن، ص: 158.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"اعْلَنْكَسَ الشَّعْرُ: إِذا اشْتَدَّ سَوادُهُ وكَثُرَ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 304.— المصدر
2.2معنى من معاني «اعْلَنْكَسَ»

وــــــــــ اللَّيْلُ ونَحْوُهُ: اشْتَدَّ ظَلامُهُ. (انْظُرْ: ع ر ك س).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال أبو زُبَيْدٍ الطّائِيُّ (ت: 62هـ=682م) يَصِفُ الأَسَدَ:

"وإِذا أَتى ‌اللَّيْلُ و‌اعْلَنْكَسَ تَبَوَّأَ وتَجَسَّسَ".

الراغب الأصفهاني، محاضرات الأدباء، ج: 3، ص: 667.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابنُ السِّكّيتِ (ت: 244هـ=858م):

"ويُقالُ: ‏لَيْلَةٌ مُعْلَنْكِسَةٌ، ولَيْلَةٌ طِلْمِساءُ، وطِرْمِساءُ مِثْلُها: وهِيَ المـُظْلِمَةُ التي لا تَرى فيها نَجْمًا ولا مَنارًا".

ابن السكيت، الألفاظ، تح: قباوة، ص: 305.— المصدر
”وقالَ جَعْفَرُ بنُ مَنْصورِ اليَمَنِ الإِسْماعيلِيُّ (ت: 380هـ= 990م) يَصِفُ فِعْلَ بَعْضِ الفِرَقِ والمَذاهِبِ المُخالِفَةِ لِهَدْيِ النَّبِيِّ ﷺ:

"وخالَفوا ونَقَضوا عَهْدَهُمُ افْتِراءً عَلى اللهِ، وتَكْذيبًا لِقَوْلِ رَسولِ اللهِ، وزَعَموا أَنَّهُمْ باخْتِلافِهِمْ مُصيبونَ، واحْتَجّوا فيهِ، وأَكَّدوا الأَخْبارَ بِهِ، فلَمّا اعْلَنْكَسَ الظَّلامُ، وتَغَطّى النّورُ، وتَغَيَّرَ الإِسْلامُ، تاهوا في حَيْرَةِ الجَهْلِ والطُّغْيانِ".

جعفر بن منصور اليمن، سرائر وأسرار النّطقاء، تح: مصطفى غالب، ص: 243، 244.— المصدر
2.3معنى من معاني «اعْلَنْكَسَ»

وــــــــ اللَّحْمُ، والرَّمْلُ، ونَحْوُهُما: تَراكَمَ بَعْضُهُ عَلى بَعْضٍ، واجْتَمَعَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الفَرَزْدَقُ (ت: 110هـ=728م) يَهْجو جَريرًا واصِفًا إِيّاهُ بامْرَأَةٍ:

هَدَرْتُ لَمّا تَلَقَّتْني بِجونَتِها
وخَشْخَشَتْ لي حَفيفَ الرّيحِ في العُشَرِ

ثُمَّ اتَّقَتْني بِجَهْمٍ لا سِلاحَ لَهُ
كَمِنْخِرِ الثَّوْرِ مَعْكوسًا مِنَ البَقَرِ

‌‌مُعْلَنْكِسِ ‌الكَيْنِ مَجْلومٍ مَشافِرُهُ
ذي ساعِدَيْنِ يُسَمّى دارَةَ القَمَرِ

البغدادي، خزانة الأدب، تح: هارون، ج: 7، ص: 540.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"‌والمـُعْلَنْكِسُ: المـُتَراكِمُ مِنَ الرَّمْلِ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 304.— المصدر
2.4معنى من معاني «اعْلَنْكَسَ»

وـــــــــ القَوْمُ ونَحْوُهُمْ: زادوا، وكَثُروا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال بَشّارُ بنُ بُرْدٍ (151هـ=767م) يَمْدَحُ شَجاعَةَ عُقْبَةَ بْنِ سَلْمٍ والي البَصْرَةِ:

وابْنُ حَكيمٍ إِذْ أَتاكَ يَرْدي

في العَدَدِ المـُعْلَنكِسِ الأَعَدِّ

راحَ بِحَدٍّ وغَدا بِحَدِّ

ديوان بشار، تح: إحسان عباس، ص: 217.— المصدر
2.5معنى من معاني «اعْلَنْكَسَ»

وــــــــــ فُلانٌ وغَيْرُهُ: أَقامَ بالمـَكانِ ونَحْوِهِ، واتَّخَذهُ سَكَنًا لَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بنُ أحمدَ (ت: 170هـ= 787م):

"ورَجُلٌ مُعْلَنْكِسٌ: إذا كانَ مُقيمًا بالبَلَدِ، ويُقالُ: ما لَهُ قَدِ اعْلَنْكَسَ، وقَوْمٌ مُعْلَنْكِسونَ: مُقيمونَ بالبَلَدِ".

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 304.— المصدر
”وقالَ الرّاجِزُ (170هـ=787م، مِنْ شَواهِدِ الخَليلِ بنِ أحمدَ) يَصِفُ تَيْسًا:

يا رُبَّ تَيْسٍ قَهَوانٍ قَهْوَسِ

سيقَتْ لَهُ في ‌نَشَرٍ ‌مُعْلَنْكِسِ

مُطْبِقَةَ الغَضِّ كَعَيْنِ الأَشْوَسِ

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 304.— المصدر
2.6معنى من معاني «اعْلَنْكَسَ»

وــــــــــ العُشْبُ ونَحْوُهُ: كَثُرَ، والْتَفَّ، وتَراكَبَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ بنُ أحمدَ (ت: 170هـ=787م):

"‌والمـُعْلَنْكِسُ مِنَ اليَبيسِ: ما كَثُرَ واجْتَمَعَ‌".

الخليل، العين، تح: المخزومي وآخر، ج: 2، ص: 304.— المصدر
”وقال الرّاجِزُ (206هـ=821م، مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانِيِّ) يَصِفُ ثِمارًا:

حَتّى دَعاها عُنْفوانٌ مورِسُ

وحانِطٌ مِنَ الغَضا مُعْلَنْكِسُ

أبو عمرو الشيباني، الجيم، تح: الأبياري، ج: 1، ص: 186.— المصدر
2.7معنى من معاني «اعْلَنْكَسَ»

وــــــــــ الشَّيْءُ: تَرَدَّدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الجَوْهَرِيُّ (ت: 393هـ=1003م):

"ويُقالُ: اعْلَنْكَسَ الشَّيْءُ: إِذا تَرَدَّدَ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح، تح: عطار، ج: 3، ص: 952.— المصدر
2.8معنى من معاني «اعْلَنْكَسَ»

وــــــــــ فُلانٌ وغَيْرُهُ بالمـَكانِ ونَحْوِه، وفيهِ: أَقامَ بِهِ، واتَّخَذَهُ سَكَنًا لَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق هـ=544م) يَصِفُ أَسَدًا:

أَجْوَفُ جافٍ جاهِلٌ مُصَدَّرُ

مُعْلَنْكِسُ الغابَةِ جَأْبٌ جَيْفَرُ

ديوان امرئ القيس، تح: عليان وآخر، ص: 716.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"وأنا مُعْلَنْكِسٌ بِمَوْضِعِ كَذا وكَذا، أَيْ مُقيمٌ بِهِ".

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 2، ص: 1217.— المصدر
2.9معنى من معاني «اعْلَنْكَسَ»

وــــــــــ الإِبِلُ وغَيْرُها في المـَكانِ: اجْتَمَعَتْ فيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سيدَهْ (ت: 458هـ=1066م):

"واعْلَنْكَسَتِ الإِبِلُ في المَوْضِعِ: اجْتَمَعَتْ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم، تح: هنداوي، ج: 2، ص: 417.— المصدر
03 —

اعْلُنْكِسَ

فعل
3.1معنى من معاني «اعْلُنْكِسَ»

* اعْلُنْكِسَ الوادي: كَثُرَ شَجَرُهُ والْتَفَّ. (ولَمْ يُسْتَعْمَلْ هَذا الفِعْلُ مِنْ قَبْلُ، واسْتُعْمِلَ المـُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230هـ= 845م):

"ويُقالُ: هُوَ في مُعْلَنْكَسِ الوادي...: إذا كَثُرَ شَجَرُهُ والْتَفَّ".

أبو مسحل الأعرابي، النوادر، تح: عزة حسن، ص: 395.— المصدر
04 —

عِلَّكْس

اسم
4.1معنى من معاني «عِلَّكْس»

* العِلَّكْسُ مِنَ الشَّعْرِ ونَحْوِهِ: الغَزيرُ، الشَّديدُ السَّوادِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ بنُ العَجّاجِ (ت: 145هـ=762م) يَتَغَزَّلُ:

بَعْدَ غُدافِ جَثْلَةٍ عِلَّكْسِ

ومِشْيَةٍ هَذَّ العَتيقِ الوَهْسِ

ثابت بن أبي ثابت، خلق الإنسان، تح: فراج، ج: 1، ص: 64.— المصدر
05 —

مُعْلَنْكِس

اسم
5.1معنى من معاني «مُعْلَنْكِس»

* المُعْلَنْكِسُ: الأَسَدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ خالَوَيْه (ت: 370هـ=980م) يُعَدِّدُ أَسْماءَ الأَسَدِ:

"والأَزْهَرُ، والأَزْأَرُ، والأَجْوَفُ، والجافي، والجاهِلُ، والمـُعْلَنْكِسُ".

ابن خالويه، أسماء الأسد، تح: الدرويش، ص: 15.— المصدر
