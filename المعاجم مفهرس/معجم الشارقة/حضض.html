حضض
حَضَّ(فعل) حاضَّ(فعل) حَضَّضَ(فعل) احْتَضَّ(فعل) تَحاضَّ(فعل) تَحْضيض(اسم) حَضَض / حُضَضُ / حُضُض(اسم) حُضِّيّ(اسم) حَضَوْضاة(اسم) حَضَوْضى(اسم) حَضيض(اسم) حَضيضَةُ(اسم)
المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السَّاميَّاتُ بِهذا الجَذْرِ مِنَ العَرَبِيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (ح ض ض):

في العِبريّة: وَرَدَ الفِعْلُ: חִצֵּץ (حِصّ۪ٮٰـص) ḥiṣṣēṣ بِمَعْنى (حَثَّ، حَرَّضَ) ويُوافِقُ الفِعْلَ العَرَبِيَّ (حَضَّ، حَضَّضَ) مَبْنًى ومَعْنًى غَيرَ أنَّ الفِعْلَ العِبْرِيَّ خَضَعَ لِإبْدالِ الضَّادِ صادًا، ومِنْ مَجيئِهِ بِمَعْنى (حَضَّ) ما وَرَدَ في سِفْرِ القُضاةِ: "מִקּוֹל מְחַצְצִים בֵּין מַשְׁאַבִּים"، أيْ: "فَوْقَ صَوْتِ مَنْ يَحُضُّونَ عِنْدَ الأحْواضِ".

(שופ' 5: 11)

المعانى الكلّيّة

  1. الحَثُّ
  2. المُنْخَفِضُ مِنَ الأرْضِ
01 —

حَضَّ

فعل
1.1معنى من معاني «حَضَّ»

* حَضَّ (كَنَصَرَ) فُلانٌ فُلانًا على الشَّيْءِ يَحُضُّ (بِضَمِّ الضّادِ) حَضًّا، وحُضًّا، وحِضّيضى، وحُضِّيضى: حَرَّضَهُ وحَثَّهُ عَلَيْهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عُمَيْرَةُ التَّغْلِبِيُّ (ت: 60ق.هـ=563م):

فَإنَّ الحُرَّ يَأْنَفُ في خَلاءٍ
وإنْ حَضَّ الجَماعَةَ أنْ يُهانا

مجمع اللغة العربية بالشارقة، منتخبات شعرية من العصر الجاهلي، ص: 10.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وفي القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ :

﴿ولا يَحُضُّ على طَعامِ المِسْكينِ﴾. (الحاقة: 34)

”وقالَ أبو الدَّرْداءِ (ت: 32ه = 653م):

"حُضّي على طَعامِ المِسْكينِ يا أُمَّ الدَّرْداءِ".

ابن زنجويه، الأموال، تح: شاكر فياض، ج: 2، ص: 763.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ موسى بْنُ عُقْبَةَ القُرَشِيُّ (ت: 141ه=758م):

"زَعَموا أنَّ عُيَيْنَةَ بْنَ بَدْرٍ أبى عَلَيْهِمْ، وحَضَّ على مَنْعِهِمْ".

موسى بن عقبة، المغازي، تح: محمد باقشيش، ص: 292.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ إبْراهيمُ بْنُ الحاجِّ النُّمَيْرِيُّ (ت: 768ه=1367م) يَمْدَحُ:

وأصْبَحْتَ في الأمْلاكِ أَمْجَدَ أوْحَدا
تَحُضُّ على دينِ النَّدى أَيَّما حَضِّ

ديوانه، تح: عبد الحميد الهرامة، ص: 173.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمَنِ بْنُ أحْمَدَ البَهْكَليّ (ت: 1248هـ=1832م)

"عَيَّنَ على عِقالِ البَدِّ القِيامِ التّامِّ بتَحَصُّلِ الأعْوانِ والأُجراءِ، وحَضَّهُمُ الشَّريفُ على المُسارَعَةِ بِإنْجازِهِ لِما يَتَوَقَّعُهُ مِنْ حُصولِ حاصِلٍ مِنَ الإمامِ".

البهكلي، نفح العود في سيرة دولة الشريف حمود. تح: محمد العقيلي، ص: 228.— المصدر
02 —

حاضَّ

فعل
2.1معنى من معاني «حاضَّ»

* حاضَّ فُلانٌ فُلانًا: حَرَّضَ كُلٌّ مِنْهُما الآخَرَ وحَثَّهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَلِيُّ بْنُ مُحَمَّدٍ الخازِنُ (ت: 741ه=1341م):

"ولا يَحاضّونَ مَعْناه: ولا يَحُضُّ بَعْضُهُمْ بَعْضًا على ذَلِكَ".

الخازن، لباب التأويل، تح: عبد السلام شاهين، ج: 1، ص: 427.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ عَلِيٍّ الشَّوْكانِيُّ (ت: 1250هـ=1834م):

"تُحاضّون، بِضَمِّ التّاءِ: مِنَ الحَضِّ، وهُوَ الحَثُّ".

[يُفَسِّرُ قَوْلَهُ ﷻ: ﴿ولا تُحاضّونَ على طَعامِ المِسْكينِ﴾ (الفجر: 18)، وهِيَ قِراءَةٌ شاذَّةٌ]
الشوكاني، فتح القدير، ج: 1، ص: 1621.— المصدر
03 —

حَضَّضَ

فعل
3.1معنى من معاني «حَضَّضَ»

* حَضَّضَ فُلانًا على الشَّيْءِ: بالَغَ في تَحْريضِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ القَعْقاعُ بْنُ عَمْرٍو (ت: 40ه=660م):

حَضَّضَ قَومي مَضرَحِيُّ بْنُ يَعمُرٍ
فَلِلَّهِ قَومي حينَ هَزّوا العَوالِيا

رياض عبد الله ليلا، معارك حاسمة في تاريخ الإسلام، ص: 356.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216ه=831م):

"لَكِنَّ سَعْدَ بْنَ مالِكٍ حَضَّضَهُ بِقَصيدَةٍ".

الأصمعيات، تح: أحمد شاكر وآخر، ص: 17.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَلِيُّ بْنُ أبي الفَرَجِ البَصْرِيُّ (ت: 659ه=1261م):

"كانَ ابْنُ مِلْقَطٍ هُوَ الذي حَضَّضَهُ على ذَلِكَ لِما فَعَلَهُ سُوَيْدُ بْنُ رَبيعَةَ مِنْ بَني دارِمٍ بِأخي عَمْرِو بْنِ هِنْدٍ".

ابن أبي الفرج، الحماسة البصرية، تح: عادل سليمان، ط 1، ص: 65.— المصدر
04 —

احْتَضَّ

فعل
4.1معنى من معاني «احْتَضَّ»

* احْتَضَّ فُلانٌ نَفْسَهُ لِفُلانٍ: اسْتَزادَها عَطاءً. (وانْظُرْ: ب ض ض).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو تُرابٍ اللُّغَوِيُّ (ت: 280ه=893م):

"احْتَضَضْتُ نَفْسِي لِفُلانٍ، وابْتَضَضْتُها: إذا اسْتَزَدْتَها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد الحليم النجار، ج: 3، ص: 398.— المصدر
4.2معنى من معاني «احْتَضَّ»

وــــــــــ مِنْ فُلانٍ شَيْئًا: أخَذَهُ مِنْهُ قَسْرًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"احْتَضَضْتُ مِنْ فُلانٍ شَيْئًا: أخَذَتُهُ مِنْهُ قَسْرًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط، تح: محمد آل ياسين، ج 2، ص: 297.— المصدر
05 —

تَحاضَّ

فعل
5.1معنى من معاني «تَحاضَّ»

* تَحاضَّ القَوْمُ على الشَّيْءِ: حَثَّ بَعْضُهُمْ بَعْضًا عَلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وفي القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ :

﴿ولا تَحاضّونَ على طَعامِ المِسْكين﴾. (الفجر: 18)



ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاء (ت: 207هـ=822م):

"تَحاضُّونَ: يَحُضُّ بَعْضُكُمْ بَعْضًا".

الفرّاء، معاني القرآن، تح: محمد النجار، ج: 3، ص: 261.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ جُزَيٍّ الكَلْبِيُّ (ت: 741ه=1340م) يُفَسِّرُ قَوْلَهُ : ﴿ولا تَحاضّونَ على طَعامِ المِسْكينِ﴾ (الفجر: 18):

"تَحاضُّونَ، بِفَتْحِ الحاءِ وأَلِفٍ بَعْدَها، بِمَعْنى: لا يَحُضُّ بَعْضُكُمْ بَعْضًا".

ابن جزي، التسهيل لعلوم التنزيل، تح: محمد هاشم، ج: 2، ص: 571.— المصدر
06 —

تَحْضيض

اسم
6.1معنى من معاني «تَحْضيض»

* التَّحْضيضُ (عِنْدَ النُّحاةِ): طَلَبُ الشَّيْءِ بِعُنْفٍ. وأدَواتُهُ: هَلّا، وألّا، وألا، ولَوْلا، ولَوْما.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سيبَوَيْهِ (ت: 180ه=797م):

"لأنَّ هَذِهِ الحُروفَ جُعِلَ فيها مَعْنى التَّحْضيضِ، واسْتِدْعاءِ الفِعْلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السيرافي، شرح كتاب سيبويه، تح: أحمد مهدلي وآخرين، ج: 1، ص: 405.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ الطّائِيُّ (ت: 672هـ=1273م):

وبِهِما التَّحضيضَ مِزْ وهَلّا

أَلّا أَلا وأوْليَنْها الفِعْلا

شرح ابن الناظم على ألفية ابن مالك، تح: محمد عيون السود، ص: 510.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابْنُ عَجيبَةَ (ت: 1224هـ=1809م):

"لَوْلا: إذا دَخَلَتْ على الماضي أفادَتِ التَّوْبيخَ، وإذا دَخَلَتْ على المُسْتَقْبَلِ أفادَتِ ‌التَّحْضيضَ".

ابن عجيبة، البحر المديد، تح: حسن عبّاس زكي، 1، ص: 460.— المصدر
07 —

حَضَض / حُضَضُ / حُضُض

اسم
7.1معنى من معاني «حَضَض / حُضَضُ / حُضُض»

* الحَضَضُ، والحُضَضُ، والحُضُضُ: الشَّيْءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"ما عِنْدَهُ حَضَضٌ، ولا بَضَضٌ، أيْ: شَيْءٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط، تح: محمد آل ياسين، ج: 2، ص: 297.— المصدر
7.2معنى من معاني «حَضَض / حُضَضُ / حُضُض»

وــــــــــ: دَواءٌ كانَ يُعْقَدُ مِنْ أبْوالِ الإبِلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الحُضُضُ: دَواءٌ يُتَّخَدُ مِنْ أبْوالِ الإبِلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخَليلُ، العَيْن. تح: المخزومي، ج: 3، ص: 13.— المصدر
7.3معنى من معاني «حَضَض / حُضَضُ / حُضُض»

وــــــــــ: داءٌ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711ه=1311م):

"الحُضَضُ، والحُضُضُ، بِفَتْحِ الضّادِ الأُولى وضَمِّها: داءٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، لسان العرب، تح: أمين عبد الوهاب وآخر، ج: 3، ص: 219.— المصدر
7.4معنى من معاني «حَضَض / حُضَضُ / حُضُض»

وــــــــــ: كُحْلُ الخَوْلانِ، وهُوَ عُصارة شَجَرَةٍ شَوْكِيَّةٍ ثَمَرَتُها تُشْبِهُ الفُلْفُلَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مُحَمَّدُ بْنُ سيرينَ (ت: 110ه=728م) يَصِفُ عِلاجًا لِرَجُلٍ أصابَهُ سُلاقٌ في شَفَتَيْهِ:

"لا بَأْسَ بِالحُضَضِ".

ابن أبي شيبة، المصنف، تح: كمال الحوت، ج: 2، ص: 305.— المصدر
7.5معنى من معاني «حَضَض / حُضَضُ / حُضُض»

وــــــــــ: صَمْغٌ مِنْ نَحْوِ الصَّنَوْبَرِ والمُرِّ وما أَشْبَهَهُما مِمّا لَهُ ثَمَرٌ، كَالْفُلْفُلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787):

"الحُضَضُ، بِالضّاد والظّاءِ: وهُوَ صَمْغٌ مُرٌّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: رمزي بعلبكي، ص: 1004.— المصدر
7.6معنى من معاني «حَضَض / حُضَضُ / حُضُض»

وــــــــــ: عُصارَةُ الشَّجَرِ الـمُرِّ كالصَّبِرِ ونَحْوِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ(ت: 458ه=1066م):

"الحُضُضُ، والحُضَضُ: عُصارَةُ الشَّجَرِ الـمُرِّ كالصَّبِرِ ونَحْوِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 490.— المصدر

حُضِّيّ

[الوَأْبُ مِنَ الحَوافِرِ: الشَّديدُ الصُّلْبُ]— المصدر

11 —

حَضيض

اسم
11.1معنى من معاني «حَضيض»

* الحَضيضُ: قَرارُ الأرْضِ عند سَفْحِ الجَبَلِ. (ج) أَحِضَّةٌ، وحُضُضٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُثْمانُ بْنُ عَفّانَ (ت: 35ه=656م):

"هَل تَعْلَمونَ أَنَّ رَسولَ اللهِ ﷺ كانَ على ثَبيرِ مَكَّةَ ومَعَهُ أبو بَكْرٍ وعُمَرُ، رَضِيَ اللهُ تَعالى عَنْهُما، وأنا، فَتَحَرَّكَ الجَبَلُ حَتّى تَساقَطَتْ حِجارَتُهُ بِالحَضيضِ، فَرَكَضَهُ بِرِجْلِهِ، فقالَ: اسْكُنْ ثَبيرُ، فَإنّما عَلَيْكَ نَبِيٌّ وصِدّيقٌ وشَهيدانِ".

العيني، عمدة القاري، تح: عبد الله عمر، ج: 14، ص: 101.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَهْ (ت: 458هـ=1066م):

"الحَضيضُ: قَرارُ الأرْضِ عِنْدَ سَفْحِ الجَبَلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 490.— المصدر

وــــــــــ: ما سَفُلَ مَنْزِلًا أوْ مَنْزِلَةً.

فَلَمّا أَجَنَّ الشَّمْسَ عَنِّي غُؤورُها
نَزَلْتُ إلَيْهِ قائِمًا بِالحَضيضِ

ديوانه وملحقاته، تح: أنور عليان وآخر، ص: 465.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) لِمَنْ أهْدى إلَيْهِ هَدِيَّةً، ولَمْ يَجِدْ شَيْئًا يَضَعُها عَلَيْهِ:

"ضَعْهُ بِالحَضيضِ، فَإنّما أنا عَبْدٌ آكُلُ كَما يَأْكُلُ العَبْدُ".
