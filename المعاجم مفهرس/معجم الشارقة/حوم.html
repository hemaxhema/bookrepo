  1. الدَّورانُ
01 —

حامَ

فعل
1.1معنى من معاني «حامَ»

* حامَتِ الإِبِلُ وغَيْرُهَا تَحومُ حَوْمًا وحَوَمانًا: عَطِشَتْ. (ج) حوائِمُ، وحُوَمٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُرَقَّمُ الذُّهْلِيُّ (ت: 108ق.ه=517م):

فلَرُبَّ باكٍ مِنْ بَني
ذُهْلٍ وقاعِدَةٍ وقاشِمِ

ومُشَقِّقاتٍ لِلجُيو
بِ عَلَيَّ كالبَقَرِ الحوائِمِ

الحسن بن بشر الآمدي، المؤتلف والمختلف في أسماء الشعراء وكناهم، تح: ف. كرنكو، ص: 128.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«اللَّهُمَّ ارْحَمْ بَهَائِمَنا الحائِمَةَ، والأَنْعامَ السّائِمَةَ، والأَطْفالَ المُحْثَلَةَ».

أبو سليمان الخطابي، غريب الحديث، تح: عبد الكريم العزباوي، ج: 1، ص: 336.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170ه=787م):

"والحَوائِمُ: الإبلُ ‌العِطَاشُ جِدًّا، وكلُّ عَطْشَانَ حَائِمٌ، وَهَامَةٌ حَائِمَةٌ، أَيْ: عَطِشَ دِماغُها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل بن أحمد، العين، تح: المخزومي، ج: 3، ص: 314.— المصدر
1.2معنى من معاني «حامَ»

وــــــــــ: الطّائِرُ وغَيْرُهُ عَلى الشَّيْءِ وحَوْلَهُ: دارَ وطَافَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مَالِكُ بْنُ حُرَيْمٍ الهَمَذانِيّ (ت: 86ق.ه=537م) يَصِفُ جَيْشَ عَدُوِّهِمْ بَعْدَ قَتْلِهِمْ:

تَحومُ الطَّيْرُ فَوْقَهُمُ وجالَتْ
عَلى خَوْلانَ بِالأَسَلِ الحِدادِ

شعر همدان وأخبارها، تح: حسن أبو ياسين، ص: 85.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ اللهِ بْنُ رواحَةَ (ت: 8ه=629م):

غَدَاةَ ثَوَى أَبُو جَهْلٍ صَرِيعًا
عَلَيْهِ الطَّيْرُ حَائِمَةٌ تَجولُ

ديوان عبد الله بن رواحة، تح: وليد قصاب، ص: 133.— المصدر
”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«أَلا لِكُلّ مَلِكٍ حِمًى وإِنّ حِمى اللهِ مَحارِمُهُ، مَنْ حامَ حَوْلَ الحِمى يوشِكُ أَنْ يَقَعَ فيهِ».

سحنون، المدونة الكبرى، دار الكتب العلمية، ج: 5، ص: 391.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ جميلُ بُثَيْنَةَ (ت: 167ه=784م):

فَما صادياتٌ حُمْنَ يَوْمًا وَلَيْلَةً
عَلى الماء يُحْيينَ العِصِيَّ حَراني

بشار بن برد، المختار من شعر بشار، تح: محمد العلوي، ص: 54.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّارِ (ت: 658ه=1260م):

أُلامُ عَلى لَيِّ العِنانِ إِلى اللِّوى
وأُعْذَلُ في حَوْمِ الجَنانِ عَلى الحِمى

ديوان ابن الأبار، تح: عبد السّلام الهرّاس، ص: 282.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الكُمُشْخانَوي (ت: 1311ه=1894م):

"والرّوحُ الأَعْظَمُ هِيَ الرّوحُ الإنْسانِيَّةُ مَظْهَرُ الذَّاتِ الإِلهِيَّةِ مِنْ حَيْثُ رُبوبِيَّتُها؛ لِذَلِكَ لا يَحومُ حَوْلَها ولا يَرومُ وَصْلَها دائِمٌ ولا يَعْلَمُ كُنْهَها إلّا اللهُ، وهُوَ العَقْلُ الأَوَّلُ".

الكمشخانوي، جامع الأصول في الأولياء- الطرق الصوفية، تح: أديب نصر الله، ص: 332.— المصدر
1.3معنى من معاني «حامَ»

وــــــــــ فُلانٌ عَلى قَرابَتِهِ: عَطَفَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وقالَ عُمَرُ بْنُ الخَطّابِ (ت: 23ه=644م):

"ما وَلِيَ أحدٌ إلّا حامَ عَلى قَرابَتِهِ، وقَرى في عَيْبَتِه، ولَنْ يَلِىَ النّاسَ كَقُرَشِىّ عَضَّ عَلى ناجِذِهِ".

ابن قتيبة، غريب الحديث، تح: نعيم زرزور، ج: 1، ص: 324.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عُمَرُ بْنُ شَبَّةَ النُّمَيريُّ (ت: 262ه=876م):

"ولَيْسَ أَحَدٌ وَلِيَ مِنْ أضْمَرِ القَبائِلِ شَيْئًا مِنْ أَضْمَرِ النّاسِ إِلّا حامَ عَلى قَرابَتِهِ".

عمر بن شبة، تاريخ المدينة، تح: فهيم شلتوت، السيد حبيب محمود أحمد، ج: 3، ص: 887.— المصدر
1.4معنى من معاني «حامَ»

وــــــــــ الشّيْءَ حَوْمًا، وحَوَمانًا، وحِيامًا، وحُؤومًا، وتَحْوامًا: طَلَبَهُ وأَمَّهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَرِيرٌ (ت: 110هـ=728م):

لَمْ أَرَ مِثْلَكِ يا أُمامَ خَليلًا
أَنْأى بِحاجَتِنا وأَحْسَنَ قيلَا

لَوْ شِئْتِ قَدْ نَقَعَ الفُؤادُ بِمَشْرَبٍ
يَدَعُ الحَوائِمَ لا يَجِدْنَ غَليلَا

شرح ديوان جرير، تح: محمد الصاوي، ص: 453.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ النَّديمِ المَوْصِلِيُّ (ت: 235ه=850م):

يا سَرْحةَ المَاءِ قَدْ سُدَّتْ مَوارِدُه
أَما إليْكِ طَريقٌ غَيْرُ مَسْدودِ

لِحائمٍ حامَ حتّى لا حَراكَ بِهِ
مُحَلإٍ عَنْ سَبيلِ المَاءِ مَطْرودِ

الجاحظ، التاج في أخلاق الملوك، تح: أحمد زكي باشا، ص: 21.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عَبْدِ المَلِكِ (ت: 703ه=1303م) يَذْكُرُ تَرْجَمَةً لِأَبي عَبْدِ اللهِ الشَّوّاشِ:

"كانَ أَديبًا ذا حَظٍّ مِنَ الكِتابَةِ وقَرْضِ الشِّعْرِ، واخْتَصَّ وقْتَهُ وبَعْدَهُ بِبَراعَةِ الخَطِّ، فكانَ أَنيقَ الوِراقَةِ رائِقَها، وتَوارَثَ النّاسُ التَّنافسَ فيما كَتَبَ إِلى اليَوْمِ، وَكَمْ حامَ كَثيرٌ مِنَ الوَرّاقِينَ عَلى سُلوكِ طَريقَتِهِ فلَمْ يُدْركوها".

ابن عبد الملك المراكشي، الذيل والتكملة، تح: إحسان عباس وآخرين، ج: 4، ص: 500.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)

بَحْرٌ خِضَمٌّ حَوْلَهُ مَنْ حامَ رُدْ
دَ بِهِ قَريرَ العَيْنِ مِنْ إِكْرامِهِ

أحمد البربير، المفاخرات والمناظرات، تح: محمد حسان الطيان، ص: 112.— المصدر

حَوَّمَ

فعل
2.1معنى من معاني «حَوَّمَ»

* حَوَّمَ عَلى الشَّيْءِ وحَوْلَهُ: حامَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عيسى بْنُ فاتِكٍ الخَطِّيُّ (ت: 70ه=690م) يَرثي قَتْلاهُ مِنَ الخَوارِجِ:

مَضَوا قَتْلًا و تَمْزيقًا وصَلْبًا
تُحَوِّمُ حَوْلَهُمْ طَيْرٌ وُقوعُ

شعر الخوارج، تح: إحسان عباس، ص: 56.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

قُلوبًا عَهِدْنا طَيْرَها وهْيَ وُقَّعُ

الآمديّ، الموازنة بين شعر أبي تمام والبحتري، تح: السيد أحمد صقر، ج: 2، ص: 83.— المصدر

"وتُوُفّيَ بِمُرْسِيَّةَ يَوْمَ الجُمُعَةِ الرَّابِع لِرَمضانَ سَنَةَ ثَلاثٍ وثَلاثينَ وخَمْسِ مِئَةٍ وقَدْ حَوَّمَ عَلى التِّسْعينَ".
