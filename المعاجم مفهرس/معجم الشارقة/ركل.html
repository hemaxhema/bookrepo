النقوش العربيّة

وَرَدَ الجَذْرُ (ر ك ل ر ك ل ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَينِها نَقْشٌ سَبَئِيٌّ رُبَّما يَعُودُ تاريخُهُ إلى النِّصفِ الأوَّلِ مِنَ الألْفِ الأوَّلِ قَبْلَ الميلادِ (B-L Nashq ? = Demirjian 1/13)، وَرَدَ فيهِ الفِعْلُ (ركل rkl) أيْ: ( شارَكَ، خَرَجَ في رِحلةٍ أو قافِلةٍ تجاريّةٍ)، فَجاءَ في النَّقْشِ: "و يوم/ ركل/ و مصر/ عد/ ددن/ و غزت"، أيْ: " وعِندَما شارَكَ وانطلَقَ في رِحلةٍ تِجاريّةٍ إلى دادانَ وغزّةَ".

(Robin and de Maigret, 2009: 82- 84; http://dasi.cnr.it/)

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ر ك ل):

في الآراميّة: وَرَدَ الاسْمُ: רַכָּלָא (رَكّالا) rakkālā بِمَعْنى: (بقّالٌ/ بائِعٌ مُتَجوِّلٌ)، ويوافِقُ الاسمَ العَرَبيَّ (ركّال) لفظًا ومَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى (ركّال): "אבנצרף 1 זי השכחו גניב ביד רכליא"، أيْ: "حَجَرُ صبغٍ وجَدوهُ مَسروقًا بِحَوزةِ الرّكّالينَ".

(TAD A4.3R.4)

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܪܰܟܳ̊ܠܳܐ (رَكّالا) rakkālā بِمَعْنى: (بقَالٌ/ عَطّارٌ)، ويوافِقُ الاسمَ العَرَبيَّ (ركّالٌ) لفظًا ومَبْنًى ومَعْنًى.

(R. Payne Smith's Thesaurus Syriacus. P. 3918)

01 —

رَكَلَ

فعل
1.1معنى من معاني «رَكَلَ»

* رَكَلَ (بِفَتْحِ الكافِ) فُلانٌ الشَّيْءَ يَرْكُلُهُ (بِضَمِّ الكافِ) رَكْلًا: ضَرَبَهُ بِرِجْلٍ واحِدَةٍ، وقيلَ: رَفَسَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتِ الخِرْنِقُ بِنْتُ بَدْر (نحو: 50ق.هـ=573م) تَهْجو عَبْدَ عَمْروٍ حينَ أَنْشَدَ المَلِكَ شِعْرَ أَخيها طَرَفَةَ بْنِ العَبْدِ:

هُمُ رَكَلوكَ لِلوَرِكَيْنِ رَكْلًا
ولَوْ سَأَلوكَ أَعْطَيْتَ البُروكا

أَلا ثَكِلَتْكَ أُمُّكَ عَبْدَ عَمْرٍو
أَبَا الخَزَياتِ واخَيْتَ المُلوكا

أبو زيد القرشي، جمهرة أشعار العرب. تح: علي البجادي، ص: 94.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدَةُ بْنُ الطَّبيبِ (ت: 25هـ=646م) يَصِفُ:

وقَلَّ ما في أساقي القَوْمَ فانْجَرَدوا
وفي الأَداوَى بَقيَّاتٌ صَلَاصِيلُ

والعِيسُ تُدْلَكُ دَلْكًا عَنْ ذَخائِرِها
يُنْحَزْنَ مِنْهُنَّ مَحْجونٌ ومَرْكولُ

المفضل الضبي، المفضليات. تح: شاكر وآخر، ص: 137.— المصدر
”وقالَ عُثْمانُ بْنُ عَفّان (ت: 35هـ=656م) يُخاطِبُ مَنْ حَوْلَهُ، وهُوَ مَحْصورٌ:

"أَنْشُدُ بِاللَّهِ مَنْ شَهِدَ رَسولَ اللَّهِ ﷺ يَوْمَ حِراءَ؛ إِذِ اهْتَزَّ الجَبَلُ، فرَكَلَهُ بِقَدَمِهِ، ثُمَّ قالَ: «اسْكُنْ حِراءُ، لَيْسَ عَلَيْكَ إِلَّا نَبِيٌّ، أَوْ صِدِّيقٌ، أَوْ شَهيدٌ»، وأَنا مَعَهُ".

مسند أحمد. تح: الأرناؤوط وآخرين، ج: 1، ص: 478.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أحْمَد (ت: 170هـ=787م):

"الرَّكْلُ: الضَّرْبُ برِجْلٍ واحِدَةٍ".

الخليل، العين. تح: هنداوي، ج: 2، ص: 147.— المصدر
”وقالَ الجاحِظُ (ت: 255هـ=869م) يَصِفُ فَهْمَ الفِيَلَةِ:

"وأَنَّ فيلًا ‌مِنْها ‌رَكَلَهُ برِجْلِهِ رَكْلَةً صَكَّ بِها الحائِطَ حَتَّى خيفَ عَلَيْهِ مِنْها".

الجاحظ، كتاب الحيوان. تح: عيون السود، ج: 7، ص: 54.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ أبو شامة الـمَقْدسيّ (ت: 665هـ=1267م) يَذْكُرُ و اقِعَةَ القَبْضِ عَلى شَمْسِ الدّينِ عَليِّ بْنِ الدّايَةِ:

"وحُمِلَ إِلى بَيْنَ يَدَيِ الـمَلِكِ الصّالِحِ فاسْتَقْبَلَهُ أَحَدُ مَماليكِ نورِ الدّينِ المَعْروفِ بِالجُفَيْنَةِ، فرَكَلَهُ بِرِجْلِهِ رَكْلَةً دَحاهُ بِها عَلى وَجْهِهِ".

أبو شامة، الروضتين في أخبار الدولتين. تح: إبراهيم الزيبق، ج: 2، ص: 331.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أحْمَد شَوْقي (ت: 1350هـ=1932م) عَلى لِسانِ الأَميرَةِ تَتيناسَ في مَسْرَحِيَّةِ قِمْبيزِ:

بِالصَّفْعِ أَجْزيهِ وبِالرَّكْلِ أَوْ
كُنْتُ أُرِيهِ النَّجْمَ في الظُّهْرِ

أحمد شوقي، الأعمال الكاملة - المسرحيات. تح: سعد درويش، ج: 1، ص: 393.— المصدر

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو خِراشٍ الهُذَليّ (ت: 14هـ=636م) يَذْكُرُ رَجُلًا قُتِلَ غَدْرًا وهو أعْزَلُ:

دعا قَوْمَهُ لـمَّا اْستُحِلّ حَرامُهُ
ومِنْ دُونِهِمْ عَرْضُ الأَعِقَّةِ فالرَّمْلُ

ولَوْ سَمِعوا مِنْهُمْ دُعاءً يَروعُهُمْ
إِذًا لَأَتَتْهُ الخَيْلُ أَعْيُنُها قُبْلُ

شَواحِيَ يَمْريهِنَّ بِالقَوْمِ والقَنا
فُروعُ السِّياطِ والأَعِنَّةُ والرَّكْلُ

إذًا لأتاه كلُّ شاكٍ سِلاحُهُ
يُعانِشُ يومَ البأسِ ساعِدُه جَدْلُ

[قُبْلُ: إِقْبالُ إِحْدى الحَدَقَتَيْنِ عَلى الأُخْرى. شَواحي: فاتِحاتٌ أَفْواهَها. يَمْريهِنَّ: يُخْرِجُ ما عِنْدَهُنَّ]
ديوان الهذليين. تح: الشنقيطي، ج: 2، ص: 165.— المصدر

فعل
2.1معنى من معاني «راكَلَ»

* راكَلَ فُلانٌ وغَيْرُهُ الشَّيْءَ: غالَبَهُ في الضَّرْبِ والرَّكْلِ.

يُمَشِّينَ وَسْطَ الدَّارِ في كُلِّ مُنْعَلِ

[الـمِرْطُ: ثَوْبٌ تَلْبَسُهُ الـمَرْأَةُ قَدْ لُبِسَ قَبْلَ ذَلِكَ]

5.1معنى من معاني «تَراكَلَ»

* تَراكَلَ القَوْمُ: ضَرَبَ بَعْضُهُمْ بَعْضًا بِالأَرْجُلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفارابِيُّ (ت: 350هـ=961م):

"وتَراكَلوا: من الرَّكْلِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الفارابي، معجم ديوان الأدب. تح: أحمد مختار، ج: 2، ص: 471.— المصدر
06 —

تَرَكَّلَ

فعل
6.1معنى من معاني «تَرَكَّلَ»

* تَرَكَّلَ فُلانٌ بِالمِسْحاةِ ونَحْوِها، أَوْ عَلَيْها: ضَرَبَها بِإِحْدى رِجْلَيْهِ، وتَوَرَّكَ عَلَيْها لِيُدْخِلَها في الأَرْضِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأخْطَلُ (ت: 90هـ=708م) يَصِفُ خَمْرَةً حادَّةَ المِزاجِ:

رَبَتْ ورَبا في كَرْمِها ابْنُ مَدينَةٍ
يَظَلُّ عَلى مِسْحاتِهِ يَتَرَكَّلُ

[ابْنُ مدينَةٍ: أيْ: حاذِقٌ في صُنْعِ الخَمْرِ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أحْمَد (ت: 170هـ=787م):

"والتَّرَكُّلُ: كَفِعْلِ الحافِرِ بِالمِسْحاةِ حينَ يَتَركَّلُ عَلَيْها بِرِجْلِهِ".

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 353.— المصدر
”وقالَ أبو عبد الرحمن العُتْبيّ (ت: 228هـ=842م) يَذْكُرُ وادِيًا خِصْبًا مَرَّ بِهِ:

"هَجَمْتُ عَلى بَطْنٍ بَيْنَ جَبَلَيْنِ، فَلَمْ أَرَ وادِيًا أَخْصَبَ مِنْهُ، وإِذا رِجالٌ يَتَرَكَّلونَ عَلى مَساحيهِمْ".

الجاحظ، كتاب الحيوان. تح: عيون السود، ج: 3، ص: 61.— المصدر
07 —

رِكال

اسم
7.1معنى من معاني «رِكال»

* الرِّكالُ: الضَّرْبُ بِالرِّجْلِ أَوِ الرَّفْسُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو النَّجْمِ العِجْليّ (ت: 130هـ=748م) يَصِفُ حالَ ناقَتِهِ مِنَ التَّعَبِ والإِجْهادِ:

عَنْ عَتَبِ الأَرْضِ وعَنْ أَدْحالِها

مُحَرَّضُ اللَّحْيَيْنِ مِنْ رِكالِها

[العَتَبُ: الشِّدَّةُ، دَحْلُ الأَرْضِ: حُفَرُها، الحَرَضُ: الإِعْياءُ والمَرَضُ]
الشيباني، كتاب الجيم. تح: الأبياري، ج: 2، ص: 300.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو دُلامَة (ت: 161هـ=778م) يَصِفُ بَغْلَتَهُ مُعَدِّدًا عُيوبَها:

وعُقَّالٍ يُلازِمُها شَديدٍ
ومِنْ هَدْمِ المَعالفِ والرِّكالِ

أبو دُلامة، ديوانه. تح: إميل بديع يعقوب، ج: 1، ص: 99.— المصدر

رَكْل

اسم
8.1معنى من معاني «رَكْل»

* الرَّكْلُ: الكُرّاثُ وهُوَ عُشْبَةٌ طَيِّبَةُ الرّائِحَةِ تُتَّخَذُ لِلطَّعامِ (لُغَةِ عَبْدِ القَيْسِ).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأعْرابيّ (ت: 231هـ=845م):

"‌الرَّكلُ: الطِّيطانُ، وهُوَ ‌الكرَّاثُ".

الأزهري، تهذيب اللغة. تح: علي حسن هلالي، ج: 10، ص: 188.— المصدر
”وقالَ الشّاعِرُ (286هـ=898م)، مِنْ شَواهِدِ المُبَرِّدِ، يَمْدَحُ:

أَلا حَبَّذا الأَحْساءُ طِيبُ تُرابِها
ورَكْلٌ بِها غادٍ عَلَيْنا ورائِحُ

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 801.— المصدر
8.2معنى من معاني «رَكْل»

وــــــــــ: لَعِبُ الشُّبّانِ وتَدافُعُهُمْ بِالأَرْجُلِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عِيسى بْنُ قُتَيْبَة (ت: 501هـ=1107م):

"والرَّكْلُ أَيْضًا: لَعِبُ الشُّبّانِ وتَدافُعُهُمْ بِالأَرْجُلِ".

ابن قتيبة، الألفاظ المغربة بالألقاب المعربة. ج: 1، ص: 313.— المصدر
09 —

رَكْلَة

اسم
9.1معنى من معاني «رَكْلَة»

* الرَّكْلَةُ: الرَّفْسَةُ أَوِ الرَّكْضَةُ بِرِجْلٍ واحِدَةٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَبْدُ المَلِكِ بْنُ مَروانَ (ت: 86هـ=705م) يُوَبِّخُ الحَجّاجَ حينَ أساءَ لِلصَّحابيّ الجَليلِ أنَسِ بْنِ مالِكٍ رَضِيَ الله عَنْهُ:

"يا بْنَ الـمُسْتَفْرِمَةِ بِعَجْمِ الزَّبيبِ، واللهِ لَقَدْ هَمَمْتُ أنْ أَرْكُلَكَ ‌رَكْلَةً تَهْوي بها في نارِ جَهَنَّمَ، قاتَلَكَ الله".

الجاحظ، البيان والتبيين. مكتبة الهلال، ج: 1، ص: 306.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"وَهَبوا لِلكِنانِيِّ الـمُغَنّي خابِيَةً فارِغَةً (...)، فلَمْ يَحْمِلْها، فكانَ يَرْكُلُها رَكْلَةً، فتَتَدَحْرَجُ وتَدورُ بِمَبْلَغِ حَمِيَّةِ الرَّكْلَةِ".

الجاحظ، البخلاء. دار الهلال، ج: 1، ص: 200.— المصدر
د

"وحُمِلَ إِلى بَيْنَ يَدَيِ الـمَلِكِ الصّالِحِ فاسْتَقْبَلَهُ أَحَدُ مَماليكِ نورِ الدّينِ الـمَعْروفِ بِالـجُفَيْنَةِ، فرَكَلَهُ بِرِجْلِهِ رَكْلَةً دَحاهُ بِها عَلى وَجْهِهِ".

مجلة الرسالة، العدد: 100، 1935م، ص: 60.— المصدر
9.2معنى من معاني «رَكْلَة»
