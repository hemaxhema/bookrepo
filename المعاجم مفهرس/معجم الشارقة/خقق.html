المقدّمة

النّظائِرُ السّامِيَّةُ:

احْتَفَظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فَقَدْ وَرَدَ الجَذْرُ (خ ق ق):

في العِبْريّة: وَرَدَ الاسْمُ: מְזִקָּה (مْزِقّا) məziqqā بِمَعْنى (غَديرٌ، تِرعةٌ، قَناةٌ) ويُوافِقُ الاسْمَ العَرَبيَّ (خَقّ) مَعْنًى ومَبْنًى غَيْرَ أنَّ الاسْمَ العِبْرِيَّ خَضَعَ لِإبْدالِ الخاءِ زايًا، ومِنْ مَجيئِهِ بِمَعْنى (خَقّ) ما وَرَدَ في مَخْطوطاتِ قَمرانَ: "מזיקות שרוו מהנחל הגדול" أيْ: "أَخْقاقٌ تُروَى مِن النَّهرِ الكَبيرِ".

(מגילת הנחושת: עמ' 10, שו' 3- 4.)

المعانى الكلّيّة

  1. صَوْتُ الفَرْجِ والحَياءِ عِنْدَ الجِماعِ
  2. الشَقُّ في الأَرْضِ
01 —

خَقَّ

فعل
1.1معنى من معاني «خَقَّ»

*خَقَّتِ (كَضَرَبَ) الأُنْثى مِنَ الدَّوابِ وَالإنْسانِ تَخِقُّ (بِكَسْرِ الخاءِ) خَقيقًا: صَوَّتَ حَياؤُها عِنْدَ الجِماعِ مِنَ الهُزالِ والاسْتِرْخاءِ. (انْظُرْ: غ ق ق).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرُ (ت: 110هـ=728م) يَذْكُرُ نساءَ كُلَيبٍ:

تَخِقُّ الكُلَيْبِيّاتُ تَحْتَ رِجالِهِم
كَما خَقَّ في جَوْفِ الصَّراةِ الضَّفادِعُ

أبو عُبيدة مَعمر، شَرْح نقائِضِ جرير والفَرْزدق، تح: محمد حور وآخر، ج2، ص830.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخليلُ (ت: 170هـ=787م) :

" يُقَالُ لِقُنْبِ الدَّابَّةِ إذَا زَعَقَ: خَقَّ ... والأَتانُ تَخِقُّ خَقيقًا، وقَدْ ‌خَقَّتْ، وهُوَ صَوْتُ حَيائِها مِنَ الهُزالِ والاسْتِرْخاءِ عِنْدَ المُجامَعَةِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 131.— المصدر
1.2معنى من معاني «خَقَّ»

وـــــ القارُ والقِدْرُ ونَحْوُهُما: غَلى، فَسُمِعَ لَهُ صَوْتٌ. (انْظُرْ: غ ق ق).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م) :

"خَقَّ القِدْرُ وما أَشْبَهَهُ خَقًّا وخَقيقًا؛ إذا غَلى فَسَمِعْتَ لَهُ صَوْتًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 106.— المصدر
1.3معنى من معاني «خَقَّ»

وــــــــ الأَرْضُ: تَشَقَّقَت بِكَثْرَةِ المَطَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ القُوطِيَّة (ت: 367هـ=977م) :

"خَقَّ فَرْجُ كُلِّ أُنْثى خَقيقًا: صَوَّتَ، والأَرْضُ خَقًّا: تَشَقَّقَتْ بِكَثْرَةِ المَطَرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال، تح: فودة، ص: 39.— المصدر
1.4معنى من معاني «خَقَّ»

وــــــــــ البَكَرَةُ: اتَّسَعَ خَرْقُها عَنِ المِحْوَرِ، أَو اتَّسَعَتِ النّعامَةُ عَن مَوضِع طَرَفِها مِنَ الزُّرْنوقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابنُ عَبّاد (ت: 385هـ=995م) :

"وأَخَقَّتِ البَكَرَةُ وخَقَّتْ: اتَّسَعَ خَرْقُها عَنِ المِحْوَرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة، تح: آل ياسين، ج: 4، ص: 149.— المصدر
1.5معنى من معاني «خَقَّ»

وـــــــــ السَّيْلُ وغَيْرُهُ الأَرْضَ، وفيها: حَفَرَ فيها حَفْرًا عَميقًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال النّضرُ بنُ شُمَيْل (ت: 203هـ=819م) :

"خَقَّ السَّيْلُ في الأَرْضِ خَقًّا، إِذا حَفَرَ فِيها حَفْرًا عَميقًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخرين، ج: 6، ص: 541.— المصدر

أَخَقَّ

فعل
2.1معنى من معاني «أَخَقَّ»

*أخقَّ الفَرْجُ: صَوَّتَ مِنْ شِدَّةِ الجِماعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخليلُ (ت: 170هـ=787م) :

"وإخْقاقُ الأَحْراحِ: صَوْتُها عِنْدَ النَّحْجِ، وهُوَ شِدَّةُ المُجامَعَةِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 131.— المصدر

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"يُقالُ: اسْتَخَقَّ الفَرَسُ ‌وأَخَقَّ وامْتَخَضَ: إذا اسْتَرْخى سُرْمُه، يُقالُ ذَلِكَ في الذَّكَرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة، تح: آل ياسين، ج: 4، ص: 149.— المصدر
2.4معنى من معاني «أَخَقَّ»

وـــــــ السَّيْلُ في الأَرْضِ: حَفَرَ فيها حُفَرًا وأَخاديدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزهريُّ (ت: 370هـ=981م) :

"يُقالُ: خَدَّ السَّيْلُ فيها خَدًّا وأَخَقَّ فيها خَقًّا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخرين، ج:6، ص: 541.— المصدر

04 —

أُخْقُوق

اسم
4.1معنى من معاني «أُخْقُوق»

* الأُخْقُوقُ: الشَّقُّ في الأَرْضِ المُتَفَقِّرَةِ أَوْ في مُنْفَرَجِ الجَبَلِ، وقَدْ يَكونُ قَدْرَ ما يَخْتَفي فيهِ الرَّجُلُ أَوِ الدّابَّةُ. (ج) أَخاقيقُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امرؤُ القيسِ السَّكُوني (ت: 27ق هـ=595م) يَصِفُ مَوْضِعَينِ لهُ مَعَهُما ذِكْرَياتٌ:

ودونَهُما مِنْ تَلْعِ بُسْيانَ فاللِّوى
أَخاقيقُ فيها صِلِّيانٌ وحَنْظَلُ

ابن المبارك، مُنتهى الطّلب من أشعار العرب، تح: طريفي، ج: 8، ص: 345.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عبدُ الله بن عباس (ت: 68هـ=688م) يَحْكِي:

"أَنَّ رَجُلًا كانَ واقِفًا مَعَ النَّبيِّ ﷺ، فَوَقَصَتْ بِهِ ناقَتُهُ في أَخاقيقِ جِرْذانٍ".

القاسم بن سلام، غريب الحديث. تح: دار الكتب العلمية، ج: 1، ص: 65.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّرْقِيُّ بنُ القُطاميّ (ت: 155هـ=772م) يَحْكي ما حَدَثَ معَهُ هو وَرَجُل مِنْ تَغلبَ:

"وَنَحْنُ نَمْشي في أَرْضٍ ذاتِ أَخاقِيقَ وَصُدُوعٍ، إِذْ هَجَمْنا عَلى حَيَّةٍ مُنكَرةٍ فَسَاوَرَتْنا...".

الجاحظ، البيان والتبيين، تح: هارون، ج:3 ، ص:46.— المصدر
05 —

إخْقيق

اسم
5.1معنى من معاني «إخْقيق»

* الإِخْقيقُ: الشَّقُّ في الأَرْضِ، أَوْ في مُنْفَرَجِ الجَبَلِ، وقَدْ يَكونُ قَدْرَ ما يَخْتَفي فيهِ الرَّجُلُ أَوِ الدّابَّةُ. (ج) أَخاقيقُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ فارس (ت: 395هـ=1005م) :

"والأُخْقوقُ: هَزْمٌ في الأَرْضِ، وقالَ قَوْمٌ: هو الإخْقيقُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مجمل اللغة، تح: عبد المحسن، ج: 1، ص: 276.— المصدر
06 —

خِقاق

اسم
6.1معنى من معاني «خِقاق»

* الخِقاقُ: صَوْتٌ يَكونُ في حَياءِ الأُنْثى مِنَ الخَيْلِ، بِسَبَبِ رَخاوَةِ خِلْقَتِها وارْتِفاعِ مُلْتَقاها، فإذا تَحَرَّكَتْ العَنَقَ أَوْ غَيْرَهُ احْتَشَتْ رَحِمَها الرَّيحُ، فَصَوَّتَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بنُ المُثنّى (ت: 209هـ=825م) يَذْكُرُ بعضَ عُيوبِ الخَيْلِ:

"وأمَّا ‌الخِقاقُ فَصَوْتٌ يَكونُ في ظَبْيَةِ الأُنْثى مِنْ رَخاوَةِ خَلْقِها وارْتِفاعِ مُلْتَقاها وانْحِدارِ حُجْزَتِها، فإذا تَحَرَّكَتْ العَنَقَ أَوْ غَيْرَه احْتَشَتْ رَحِمَها الرَّيحُ، فَصَوَّتَتْ لِذلِكَ".

معمر بن المثنى، الخيل، تح: دائرة المعارف العثمانية، ص: 52.— المصدر
07 —

خَقّ، خُقّ

اسم
7.1معنى من معاني «خَقّ، خُقّ»

*الخَقُّ والخُقُّ: الشَّقُّ في الأَرْضِ، أو ما انْخَفَضَ مِنْها. (ج) أَخاقِقُ، وخُقوقٌ، (جج) أَخاقيقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحُسَيْنُ بنُ عَليٍّ (ت: 61هـ=680م) يُبَرِّرُ أَخْذَهُ عَشْرَةَ أحْمالٍ من إِبِلٍ لِمُعاوِيَةَ قدْ مَرَّتْ عَلَيْهِ:

"هَذَا مِنْ خُقِّي وَلُقِّي، لِي أَكْثَرُ مِنْهُ".

أبو عُبيد البكري، شرح كتاب الأمثال، تح: إحسان عباس وآخر، ص: 69.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الرّياشيُّ (ت: 257هـ=871م) يُعقِّبُ على خَبَرِ ابنِ عَبّاسٍ في الرّجُل الذي وَقَصَتْ بِهِ ناقَتُهُ في أَخاقيقِ جِرْذانٍ:

"إنَّما هي أَخاقيقُ... واحِدُها: ‌خَقٌّ، وهُوَ: الجُحْرُ. ثُمَّ تُجْمَعُ، فيُقالُ: ‌أَخْقاقٌ ‌وخُقوقُ، ثُمَّ تُجْمَعُ ‌أَخْقاقٌ، فيُقالُ: أَخاقيقُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، إصلاح غلط أبي عبيد. تح: عبد الله الجبوري، ص: 76.— المصدر
7.2معنى من معاني «خَقّ، خُقّ»

وــــــــ: الغَديرُ إذا يَبِسَ وتَشَقَّقَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْد (ت: 321هـ=932م):

"والخَقُّ: الغَديرُ إذا جَفَّ وتَقَلْفَعَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ج: 1، ص: 106.— المصدر
”وَقالَ الرّاجِزُ (321هـ=932م مِنْ شَواهِدِ ابْنِ دُرَيْد) يَصِفُ نِساءً:

كأنَّما يَمْشِينَ في خَقٍّ يَبَسْ

08 —

خَقّاقَة

اسم
8.1معنى من معاني «خَقّاقَة»

*الخَقّاقَةُ: الاِسْتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسمُ بن سلاّم (ت: 224هـ=838م) :

"الخَقّاقَةُ: الاِسْتُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال، تح: حسين شرف، ج: 1، ص: 441.— المصدر
8.2معنى من معاني «خَقّاقَة»

وــــــــ مِنَ الأُتُنِ وَالنِّساءِ: الواسِعَةُ الفَرْجِ أَوِ الدُّبُرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سِيده (ت: 458هـ=1066م) :

"والخَقُوقُ ‌والخَقَّاقَةُ مِنَ الأُتُنِ والنِّساءِ: الواسِعَةُ الدُّبُرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم، تح: هنداوي، ج: 4، ص: 493.— المصدر
09 —

خَقَقَة

اسم
9.1معنى من معاني «خَقَقَة»

*‌الخَقَقَةُ: الآبارُ المُتَلاحِماتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الأعرابيّ (ت: 231هـ=845م) :

"الخَقَقَةُ: الرَّكَوات المُتَلاحِماتُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخرين، ج:6، ص: 542.— المصدر
9.2معنى من معاني «خَقَقَة»

وـــــــــ: الشُّقوقُ الضَّيِّقَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الأعرابيّ (ت: 231هـ=845م) :

"الخَقَقَةُ أَيْضًا: الشُّقوقُ الضَّيِّقَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة، تح: خفاجي وآخرين، ج:6، ص: 542.— المصدر
10 —

خَقوق

اسم
10.1معنى من معاني «خَقوق»

* الخَقوقُ مِنَ الأُتُنِ والنِّساءِ: الواسِعَةُ الفَرْجِ أَوِ الدُّبُرِ. ويُقالُ في السِّبابِ: يا ابْنَ ‌الخَقوقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُخيلةَ السعديُّ (ت: 145هـ=762م) في الخَقوقِ مِنَ النِّساء:

لَو نِكْتَ مِنْهُنَّ خَقوقًا عَرْدا

سَمِعْتَ رِزًّا ودَوِيًّا إدّا

شعر أبي نخيلة الحماني، تح: عدنان الخطيب، ص: 95.— المصدر
