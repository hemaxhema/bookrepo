01 —

رَغَثَ

فعل
1.1معنى من معاني «رَغَثَ»

* رَغَثَ (بِفَتْحِ الغَيْنِ) الجَدْيُ وغَيْرُهُ يَرْغَثُ (بِفَتْحِ الغَيْنِ) أُمَّهُ رَغْثًا: رَضِعَها. فَهِيَ مَرْغوثَةٌ، ورَغوثٌ، ورَغوثَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو هُرَيْرَةَ (ت: 59هـ=679م) يَصِفُ حالَ النَّاسِ مَعَ الدُّنيا بَعْدَ مَوْتِ رَسُولِ اللهِ ﷺ:

"فَقَدْ ذَهَبَ رَسُولُ اللهِ ﷺ ... وأَنْتُمْ تَرْغَثونَها".

صحيح البخاري. تح: محمد الناصر، ج: 9، ص: 91، (7273).— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بَشّارُ بْنُ بُرْدٍ (161هـ=777م) يَذُمُّ عَجْرَدًا، ويُعَرِّضُ بأُمِّهِ:

وعَجْردٌ يَنْزو عَلى أُمِّهِ
خِنْزيرَةٌ يَرْغَثُها راغِثُ

كَأَنَّهُ حينَ تَصَدّى لَهُ
طالِبُ عُرْفٍ أَسَدٌ شارِثُ

ديوان بشار بن برد. تح: إحسان عباس، ص: 163.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672هـ=1273م):

"ومَرْغوثَةٌ: أَيْ مَرْضوعَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن مالك، شرح الكافية الشافية. تح: هريدي، ج: 4، ص: 1740.— المصدر
1.2معنى من معاني «رَغَثَ»

وــــــــــ فُلانٌ فُلانًا: اسْتَنْفَدَ ما عِنْدَهُ بِكَثْرَةِ السُّؤالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَمْدَحُ الحارِثَ بْنَ سُلَيْمٍ الهُجَيْمِيَّ:

فَساقَكَ اللهُ إلَيْنا الباعِثُ

فَما يَني يَرْغَثُ مِنْكَ الرَّاغِثُ

خَــــيْــــرًا فَــــراجي عِـــــدَّةٍ وشــــابِثُ

أَرْضُكَ لا جَدْبٌ ولا مَخابِثُ

ساحاتُ سَهْلٍ سَهْلَةٌ دَمايِثُ

ديوان رؤبة بن العجاج. تح: وليم البروسي، ص: 29.— المصدر
1.3معنى من معاني «رَغَثَ»

وــــــــــ: طَعَنَهُ في رُغَثائِهِ، وكَرَّرَ عَلَيْهِ الطَّعْنَ مَرَّةً بَعْدَ أُخْرى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّياشِيُّ (ت: 257هـ=871م):

"الرُّغَثاءُ مِنَ الإِنْسانِ: مَغْرِزُ ثَدْيِهِ، ويُقالُ: رَغَثَهُ يَرْغَثُهُ ‌رَغْثًا: إذا ‌طَعَنَهُ في ذَلِكَ المَوْضِعِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو علي القالي، المقصور والممدود. تح: أحمد هريدي، ص: 485.— المصدر
02 —

رُغِثَ

فعل
2.1معنى من معاني «رُغِثَ»

* رُغِثَفُلانٌ: نَفِدَ ما عِنْدَهُ بِكَثْرَةِ سُؤالِهِ، فَهُوَ مَرْغوثٌ. (ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَلِيٌّ الأَحْمَرُ (ت: 194هـ=810م):

"رُغِثَ الرَّجُلُ فَهُوَ مَرْغوثٌ: إذا كَثُرَ عَلَيْهِ السُّؤالُ حَتّى يَنْفَدَ ما عِنْدَه". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: أحمد عطار، ج: 1، ص: 283.— المصدر
2.2معنى من معاني «رُغِثَ»

وــــــــــ المَرْأَةُ: اشْتَكَتْ رُغَثاءَها. وهِيَ عَصَبةُ الثَّدْيِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانِيُّ (ت: 650هـ=1252م):

"رُغِثَتِ المَرْأةُ -عَلى ما لَمْ يُسَمَّ فاعِلُهُ- تُرْغَثُ رَغْثًا: إذا اشْتَكَتْ رُغَثاءَها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل. تح: الطحاوي، ج: 1، ص: 365.— المصدر
03 —

أَرْغَثَ

فعل
3.1معنى من معاني «أَرْغَثَ»

* أَرْغَثَ فُلانٌ فُلانًا: طَعَنَهُ في رُغَثائِهِ، وكَرَّرَ عَلَيْهِ الطَّعْنَ مَرَّةً بَعْدَ أُخْرى.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتِ الخَنْساءُ (ت: 10ق هـ=612م): تَرْثي أَخاها صَخْرًا:

شَدَدْتَ عِصابَ الحَرْبِ إذْ هِيَ مانِعٌ
فَأَلْقَتْ بِرِجْلَيْها مَرِيًّا ودَرَّتِ

وكانَ أبو حسّانَ صَخْرٌ يَصُدُّها
ويُرْغِثُها بالرُّمْحِ حتّى أَقَرَّتِ

المبرد، التعازي والمراثي. تح: خليل المنصور، ص: 72.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّجّاجُ (ت: 311هـ=923م):

"رَغَثْتُ الرَّجُلَ بالرُّمْحِ وأرْغَثْتُهُ: إذا طَعَنْتَهُ بِه مَرَّةً بَعْدَ مَرَّةٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل. تح: الطحاوي، ج: 1، ص: 365.— المصدر
3.2معنى من معاني «أَرْغَثَ»

وــــــــــ الأُمُّ وَلَدَها: أَرْضَعَتْهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37هـ=658م) يَفْخَرُ:

يا بِنْتَ آلِ شِهابٍ هَلْ عَلِمْتِ إذا
أَمْسى المَراغِثُ في أَعْناقِها خَضَعُ

أَنّـــي أُتـــمِّـــمُ أَيْســــــــــاري بِـــــــــذي أَوَدٍ
مِنْ فَرْعِ شَيْحاطَ صافٍ ليطُهُ قَرِعُ

يَحْدو قَنابِلَهُمْ شُعْثٌ مَقادِمُهُمْ
بيضُ الوُجوهِ مَغاليقُ الضُّحى خُلُعُ

ديوان ابن مقبل. تح: عزة حسن، ص: 137.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بِشْرُ بْنُ المُعْتَمِرِ(ت: 210ه=825م) يَذْكُرُ آياتِ اللهِ في صُنوفِ خَلْقِهِ:

وإلْقَةٌ ‌تُرْغِثُ رُبّاحَها
والسَّهْلُ والنَّوْفَلُ والنَّضْرُ

وهِقْلَةٌ تَرْتاعُ مِنْ ظِلِّها
لَها عِرارٌ ولَها زَمْرُ

تَلْتَهِمُ المَرْوَ عَلى شَهْوَةٍ
أَحَبُّ شَيْءٍ عِنْدَها الجَمْرُ

الجاحظ، الحيوان. تح: عيون السود، ج: 4، ص: 483.— المصدر
04 —

ارْتَغَثَ

فعل
4.1معنى من معاني «ارْتَغَثَ»

* ارْتَغَثَالجَدْيُ وغَيْرُهُ أُمَّهُ: رَضِعَها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ أُذَيْنَةَ اللَّيْثِيُّ (ت: 130هـ=748م) يَتَغَزَّلُ:

وما ذِكْرُ سُعدى وقَدْ باعَدَتْ
وعادَ قوى الحَبْلِ مِنْها رِماثا

كَـــــــأَنَّ القَـــلائِـــــدَ في جيدِها
إلى حَيْثُ يَعْقِدُ مِنْها الرِّعاثا

مِنَ الدُّرِّ يَحْفِلُ ياقــــــــــوتَـــــهُ
كَجَمْرِ الغَضـــــــا يَتَلَظَّى مُجاثا

على ظَبْيةٍ مُغْزِلٍ أَشْرَفَتْ
لِخِشْفٍ لَها لَمْ يُلِحْها ارْتِغاثا

محمّد بن المبارك، منتهى الطّلب. تح: طريفي، ج: 3، ص: 151.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (216هـ=831م، مِنْ شَواهِدِ الأَصْمَعِيِّ) يَذْكُرُ مَكانًا، ويَطْلُبُ مِنَ اللهِ ألَّا يُصيبَهُ المَطَرُ:

احْرِمْهُ كُلَّ رَزَمانيٍّ مُلِثّْ

ودَعَقاتِ الدُّرَآنِ المُنْدَلِثْ

حَتَّى يُرَى في يابِسِ الثَّرْيـــاءِ حُثّْ

يَعْجَزُ عَنْ رِيِّ الطُلَيِّ المُرْتَغِثْ

الصغاني، التكملة والذيل. تح: الطحاوي، ج: 1، ص: 356.— المصدر
05 —

رَغاث/ رُغاث

اسم
5.1معنى من معاني «رَغاث/ رُغاث»

* الرَّغاثُ، والرُّغاثُ مِنَ الأَرْضِ: الَّتي لا تَسيلُ إلّا مِنْ مَطَرٍ كَثيرٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"أَرْضٌ رَغاثٌ: لا تَسيلُ إِلّا مِنْ مَطَرٍ كَثيرٍ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 5، ص: 57.— المصدر
06 —

رِغاث

اسم
6.1معنى من معاني «رِغاث»

* الرِّغاثُ: الرَّضاعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (142ه=760م) حينَ شَكا لَهُ سُلَيْمانُ بْنُ عَليٍّ العَبّاسيُّ سوءَ حالِهِ مَعَ النِّساءِ:

"بأَبي أَنْتَ، لَيْسَ ذَاكَ السِّنُّ، إنَّما ذَلِكَ لِطولِ ‌الرِّغاثِ".

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 2، ص: 297.— المصدر
6.2معنى من معاني «رِغاث»

وـــــــــ : النَّعْجَةُ حينَ تَفْطِمُ ولَدَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"والرَّغوثُ: النَّعْجَةُ حينَ تَفْطِمُ ولَدَها، وهِيَ الرِّغاثُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 9.— المصدر
07 —

رَغَثاء/ رُغَثاء

اسم
7.1معنى من معاني «رَغَثاء/ رُغَثاء»

* الرَّغَثاءُ والرُّغَثاءُ: عِرْقٌ في الثَّدْيِ يَدِرُّ اللَّبَنَ. أَوْ عَصَبَةٌ تَحْتَهُ. وقيلَ: عَصَبَةٌ بَيْنَ الثَّنْدُوَةِ والمَنْكِبِ بِجانِبِ الصَّدْرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والرُّغَثاوانِ: بَضْعَتانِ بَيْنَ الثَّنْدُوَةِ والمَنْكِبِ بِجانِبَيِ الصَّدْرِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 400.— المصدر
7.2معنى من معاني «رَغَثاء/ رُغَثاء»

وــــــــــ: أَصْلُ الضَّرْعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"والرُّغَثاءُ: أَصْلُ الضَّرْعِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 421.— المصدر
7.3معنى من معاني «رَغَثاء/ رُغَثاء»

وــــــــــ: سَوادُ حَلَمَةِ الثَّدْيِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَزْهَرِيُّ (ت: 370هـ=981م):

"الرُّغَثاوانِ: سَوادُ حَلَمَةِ الثَّدْيَيْنِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 90.— المصدر
08 —

رَغوث

اسم
8.1معنى من معاني «رَغوث»

* الرَّغوثُ: المُرْضِعَةُ مِنَ النِّساءِ وغَيْرِها. (ج) رِغاثٌ، ورَغائِثُ، ورَواغِثُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتْ هِنْدُ بِنْتُ الخُسِّ (ت: 202ق هـ=426م):

"وأَجْمَلُ النِّساءِ الفَخْمَةُ الأَسيلَةُ، وأَقْبَحُ النِّساءِ الجَهْمَةُ القَفْرَةُ، وآكَلُ الدَّوابِّ الرَّغوثُ".

أحمد زكي صفوت، جمهرة خطب العرب. المكتبة العلمية، ج: 1، ص: 72.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْرِيُّ (ت: 90هـ=709م) يَصِفُ ناقَةً:

فَجاءَتْ إلَيْنا والدُّجَى مُرْجَحِنَّةٌ
رَغوثُ شِتاءٍ قَدْ تَقوَّبَ عودُها

ديوان الرّاعي النّميريّ. تح: واضح الصّمد، ص: 113.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"كُلُّ مُرْضِعَةٍ رَغوثٌ، تَرْغَثُ وَلَدَها أَيْ تُرْضِعُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 400.— المصدر
”وقالَ أَعْرابِيٌّ (255هـ=869م، مِنْ شَواهِدِ الجاحِظِ) يُجيبُ حينَ سُئِلَ" لِمَ صارَتِ اللَّبُؤَةُ أَنْزَقَ، وعَلَى اللَّحْمِ أَحْرَصَ؟":

"هِيَ الرَّغوثُ".

الجاحظ، البيان والتبيين. تح: هارون، ج: 3، ص: 212.— المصدر
8.2معنى من معاني «رَغوث»

وــــــــــ: وَلَدُ المُرْضِعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ قُطْرُبٌ (ت: 206هـ=822م):

"والوَلَدُ أَيْضًا رَغوثٌ. والمَعْنَى أَنَّهُ راغِثٌ لَها. فَصارَ رَغوثٌ لِلمَفْعولِ والفاعِلِ".

قطرب، كتاب الأضداد. ج: 1، ص: 83.— المصدر
8.3معنى من معاني «رَغوث»

وــــــــــ: النَّعْجَةُ حينَ تَفْطِمُ ولَدَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"والرَّغوثُ: النَّعْجَةُ حينَ تَفْطِمُ ولَدَها، وهِيَ الرِّغاثُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو الشيباني، الجيم. تح: الطحاوي، ج: 2، ص: 9.— المصدر
8.4معنى من معاني «رَغوث»

وــــــــــ مِنَ البَراذينِ: الَّتي لا تَكادُ تَرْفَعُ ‌رَأْسَها ‌مِنَ ‌المَعْلَفِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"بِرْذَوْنَةٌ رَغوثٌ لا تَرْفَعُ ‌رَأْسَها ‌مِنَ ‌المَعْلَفِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 109.— المصدر

اسم
9.1معنى من معاني «مُرْغَث»

* المُرْغَثُ، والمُرَغَّثُ: مَوْضِعُ الخاتَمِ مِنَ الإِصْبَعِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والمُرْغَثُ: مَوْضِعُ الخاتَمِ مِنَ الإِصْبَع". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 5، ص: 57.— المصدر
