01 —

دَهْمَقَ

فعل
1.1معنى من معاني «دَهْمَقَ»

* دَهْمَقَ فُلانٌ: أسْرَعَ في الـمَشْيِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ صاحِبُ نَوادِرِ الأَعْرابِ ( 370هـ=981م، منْ مَصادِرِ الأَزْهَرِيِّ):

"دَغمَشْتُ في الـمَشْي ودَهْمَقْتُ ودَمْشَقْتُ، أيْ: أسْرَعْتُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 229.— المصدر
1.2معنى من معاني «دَهْمَقَ»

وــــــــــ: جادَ وَكَرُمَ. (وَهذا الفِعْلُ لَمْ يُسْتَعْمَلْ بِهَذَه الدِّلالَةِ مِنْ قَبْلُ واسْتُعْمِلَ الـمُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458هـ=1066م):

"الدَّهْمَقَةُ: الكَيْسُ".

ابن سيده، المحكم. تح: هنداوي، ج: 4، ص: 457.— المصدر
1.3معنى من معاني «دَهْمَقَ»

وــــــــــ الطَّعامَ والطَّحينَ وغَيْرَهُما: لَيَّنَهُ ورَقَّقَهُ وجَوَّدَهُ. (انظر: د ه ق ن، د ه ن ق)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عمرُ بْنُ الخَطُّابِ (ت: 23هـ=644م) يَلومُ قَوْمًا مِنْ أهْلِ العِراقِ أنِ اسْتَمْتَعوا بكُلِّ طَيِّباتِ الدُّنْيا ولَمْ يُبْقوا للآخِرَةِ شَيْئًا:

"ما هذا يا أهْلَ العِراقِ؟ لَوْ شِئْتُ أنْ يُدَهْمَقَ لي كما يُدَهْمَقُ لَكُمْ لَفَعَلْتُ، ولَكِنّا نَسْتَبْقي مِنْ دُنْيانا ما نَجِدُهُ فِي آخِرَتِنا".

ابن أبي شيبة، الكتاب المصنف. تح: الحوت، ج: 7، ص: 97.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"دَهْمِقْ طَحينَكَ، أَيْ دَقِّقْهُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 111.— المصدر
”قالَ الرّاجِزُ(190هـ=806م مِنْ شَواهِدِ الفَقْعَسَيِّ):

إذا أرَدْتَ عَمَلًا سوقِيّا

مُدَهْمَقًا فادْعُ لَهُ سِلْمِيّا

الأزهري، تهذيب اللغة. تح: خفاجي، وآخر، ج: 6،ص: 500.— المصدر
1.4معنى من معاني «دَهْمَقَ»

وــــــــــ فُلانٌ الوَتَرَ: لَيَّنَهُ، وَفَتَلَهُ جَيِّدًا، فَجاءَ بهِ مُسْتَوِيًا مِنْ أَوَّلِهِ إلى آخِرِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّوّاسُ الكِلابِيُّ (ت: 50هـ=670م) يَصِفُ الوَتَرَ:

دَهْمَقَهُ الفاتِلُ بَيْنَ الكَفَّيْنْ

فَهْوَ أَمينٌ مَتْنُهُ يُرْضي العَيْنْ

الصغاني، التكملة. تح: الإبياري، ج: 5، ص: 57.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"دَهْمَقَ الفاتِلُ الوَتَرَ، إذا جاءَ بِهِ مُسْتَوِيًا مِنْ أَوَّلِهِ إلى آخِرِهِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الإبياري، ج: 5، ص: 57.— المصدر
”وَقالَ الرّاجِزُ ( 352 هـ =963م، مِنْ شَواهِدِ ابْنِ سِمعانَ):

كَأَنَّ رِزَّ الوَتَرِ الـمُدَهْمَقِ

إِذَا مَطَاهَا هَزَمٌ مِنْ فُرَّقِ

الأزهري، تهذيب اللغة. تح: خفاجي، وآخر، ج: 6،ص: 500.— المصدر
1.5معنى من معاني «دَهْمَقَ»

وــــــــــ فُلانٌ الشَّيْءَ وغَيْرَهُ: أفْسَدَهُ وجَعَلَهُ رَديئًا. (ضِدٌّ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَقْعَسِيُّ (ت: 190هـ=806م):

"والـمُدَهْمَقُ: الّذي لَمْ يُجَوَّدْ".

الأزهري، تهذيب اللغة. تح: خفاجي، وآخر، ج: 6،ص: 500.— المصدر
1.6معنى من معاني «دَهْمَقَ»

وــــــــــ فُلانٌ الشَّيْءَ: قَطَعَهُ وكَسَرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"دَهَقْتُ الشَّيْءَ: كَسَرْتُه وقَطَّعْتُهُ...ودَهْمَقْتُهُ بِزيادَةِ الـميمِ مِثْلُهُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ج: 4، ص: 1478— المصدر
1.7معنى من معاني «دَهْمَقَ»

وــــــــــ الكَلامَ: أجادَهُ وأَحْسَنَ بَيانَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمِرُ (ت: 255هـ=869م) يَصِفُ جَوْدَةَ بَيانِ الشاعِرِ مُدْرِكِ الفَقْعَسِيِّ:

" يُسَمَّى مُدَهْمِقًا لبَيانِ لِسانِهِ وجَوْدَةِ شِعْرِهِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: خفاجي، وآخر، ج: 6، ص: 501.— المصدر
1.8معنى من معاني «دَهْمَقَ»

وــــــــــ السِّهامَ: جَعَلَها مُسْتَويَةَ الـمَتْنِ نَقِيَّةً مِنَ العُيوبِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ ( 351 هـ =962م، مِنْ شَواهِدِ أبي الطّيّبِ اللُّغويِّ):

لَقَدْ رَبَطْنَا لِلْجِيَادِ السُّبَّقِ

وَرْدًا كَقِدْحِ النَّبْعَةِ الـمُدَهْمَقِ

أبو الطّيّب اللُّغويّ كتاب الأضداد. تح: عزة حسن، ص: 183— المصدر
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"الـمُدَهْمَقُ مِنَ القِداحِ: النَّقِيُّ مِنَ العُيوبِ الـمُسْتَوي الـمَتْنِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 106.— المصدر
1.9معنى من معاني «دَهْمَقَ»

وــــــــــ: شَقَّقَها وشَرَطَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والـمُدَهْمَقُ مِنَ القِداحِ:... الـمُشَقَّقُ".

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 106.— المصدر
1.10معنى من معاني «دَهْمَقَ»

وــــــــــ الكِتابَ: حَبَّرَهُ وجَوَّدَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"كِتابَةٌ مُدَهْمَقَةٌ: لَطيفَةٌ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 106.— المصدر
02 —

دُهامِق

اسم
2.1معنى من معاني «دُهامِق»

* الدُّهامِقُ: التُّرابُ اللَّيِّنُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ خَلَفُ بْنُ خَليفَةَ (ت: 130هـ=748م) يَصِفُ الرِّمالَ:

ومَعْرِضٌ مِنَ الكَثيبِ ناطِقُ

جَوْنٌ رَوابي تُرْبِهِ دُهامِقُ

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 110.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الدُّهامِقُ: التُّرابُ اللّيِّنُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 110.— المصدر
2.2معنى من معاني «دُهامِق»

وــــــــــ مِنَ الأراضي: اللَيِّنَةُ الرَّقيقَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"الدُّهامِقِ، أيِ الأرْضُ اللَّيِّنَةُ الرَّقيقَةُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 111.— المصدر
03 —

دَهاميق

اسم
3.1معنى من معاني «دَهاميق»

* الدَّهاميقُ مِنَ الأراضي: اللَّيِّنَةُ الدَّقيقَةُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ (ت: 711هـ=1311م):

"أرْضٌ دَهاميقُ: لَيِّنَةٌ دَقيقَةٌ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في العَصْرِ الحَديثِ)

ابن منظور، لسان العرب. تح: عبد الوهاب، وآخر، ج: 4، ص: 432.— المصدر
04 —

دَهْمَقَة

اسم
4.1معنى من معاني «دَهْمَقَة»

* الدَّهْمَقَةُ: لِينُ الطَّعامِ وطِيبُهُ وَرِقَّتُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"الدَّهْمَقَةُ: لِينُ الطَّعامِ وطِيبُهُ وَرِقَّتُهُ ".

ابن بري، التنبيه والإيضاح. تح: عبد الجواد، ج: 3، ص: 471.— المصدر
05 —

مُدَهْمَق

اسم
5.1معنى من معاني «مُدَهْمَق»

* الـمُدَهْمَقُ مِنَ الطَّعَامِ وَنَحْوِهِ: الَجيِّدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَقْعَسِيُّ (ت: 190هـ=806م):

"والـمُدَهْمَقُ: الَجيِّدُ مِنَ الطَّعَامِ".

الأزهري، تهذيب اللغة. تح: خفاجي، وآخر، ج: 6،ص: 500.— المصدر
5.2معنى من معاني «مُدَهْمَق»

وــــــــــ مِنَ الرَّمْلِ وَنَحْوِهِ: الدَّقيقُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو تُرابٍ ( 280هـ=893م):

"الـمُدَهْمَقُ: الرَّمْلُ الدَّقيقُ، التُّرابُ أيْضًا إذا كانَ دَقيقًا كالـمَنْخولِ، فَهْوَ مُدَهْمَقٌ ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

عبد الواحد بن علي، كتاب الأضداد. ج: 1، ص: 184.— المصدر
