قرضم
قَرْضَمَ(فعل) قُراضِم(اسم) قِرْضِم(اسم) قِرْضِميَّة(اسم)
النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذرُ (ق ر ض م):

في الآراميّة؛ ومنه الفِعْلُ:קַרְסֵם (قَرس۪ٮٰم)qarsēm بمَعْنى "قَضَمَ/ قَرَضَ/ قَطَعَ"؛ وهُوَ يَتَّفِقُ مع الفِعْلِ العَرَبيِّ (قَرْضَمَ) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الفِعْلَ الآراميَّ خَضَعَ لإبْدال الضَّادِ سينًا؛ يُنْظَرُ (ق ر ص م).

في العِبْرِيَّةِ؛ ومنه الفِعْلُ: כִּרְסֵם (كِرس۪ٮٰـم)kirsēm بمَعْنى "قَرَضَ، قَضَمَ، قَرَمَ"، وهُوَ يُوافِقُ الفِعْلَ العَرَبيَّ (قَرْضَمَ) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الفِعْلَ العِبْريَّ خَضَعَ لإبْدالِ القافِ والضَّادِ كافًا وسينًا على المُوالاةِ؛ ومن مَجيئِهِ بمَعْنى "قَرْضَمَ" ما وَرَدَ في سِفْرِ المَزامير: "יְכַרְסְמֶנָּה חֲזִיר מִיָּעַר‏"؛ أَيْ "يُقَرْضِمُهَا خِنْزيرُ الوَعْرِ".

(תה'80: 14)

في السُّرْيانِيَّةِ؛ ومنه الفِعْلُ الكِلْداني:ܩܰܪܣܶܡ (قَرس۪م)qarsem بمَعْنى "قَطَعَ"، وهُوَ يَتَّفِقُ مع الفِعْلِ العَرَبيِّ (قَرْضَمَ) مَبْنًى ومَعْنًى، غَيرَ أَنَّ الفِعْلَ السُّرياني خَضَعَ لإبْدالِ الضَّادِ سينًا؛ يُنْظَرُ: (ق ر ص م).

المعانى الكلّيّة

  1. القَطْعُ
  2. الكَسْرُ

01 —

قَرْضَمَ

فعل
1.1معنى من معاني «قَرْضَمَ»

* قَرْضَمَ فُلانٌ الشّيءَ: قَطَّعَهُ. (انظر: ق ر ض ب).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال أبو زَيْد الأنْصاريّ (ت: 215ه=830م):

"قَرْضَمْتُ اللَّحْمَ: قَطَّعْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 1، ص: 422.— المصدر
1.2معنى من معاني «قَرْضَمَ»

وــــــــــ كَسَرَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال الأصْمَعيّ (ت: 216ه=831م):

"قَرْضَمْتُهُ قَرْضَمَةً: كَسَرْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو علي القالي، الأمالي. ج: 2، ص: 32.— المصدر
1.3معنى من معاني «قَرْضَمَ»

ــــــــــ: أَخَذَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن دُرَيْد (ت: 321ه=933م):

"فُلانٌ يُقَرْضِمُ كُلَّ شيْء؛ أي: يأخُذُه ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ص: 1182.— المصدر
02 —

قُراضِم

اسم
2.1معنى من معاني «قُراضِم»

* القُراضِمُ من الرِّجالِ: الذي يُقَرْضِمُ كُلَّ شيء.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن سِيدَه (ت: 458ه=1066م):

"وَرَجُلٌ قِرْضَمٌ وقُراضِمٌ: ‌يُقَرْضِمُ ‌كُلَّ ‌شَيْء ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 4، ص: 36.— المصدر

قِرْضِم

اسم
3.1معنى من معاني «قِرْضِم»

* القِرْضِمُ من الرِّجالِ: الذي يُقَرْضِمُ كُلَّ شيء.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن دُرَيْد (ت: 321ه=933م):

"ورَجُلٌ قِرْضِمٌ وقُراضِمٌ: الذي يُقَرْضِمُ كُلَّ شيء ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

3.2معنى من معاني «قِرْضِم»

وــــــــــ: النّاقَةُ الضّخْمَةُ الثَّقيلَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن سِيدَه (ت: 458ه=1066م):

"وقِرْضِمٌ: ضَخْمَةٌ ثَقيلَةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: خليل جفال، ج: 5، ص: 113.— المصدر
3.3معنى من معاني «قِرْضِم»

وــــــــــ: قِشْرُ الرُّمّانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن سِيدَه (ت: 458ه=1066م):

"والقِرْضِمُ: قِشْرُ الرُّمّانِ وهو يُدْبَغُ به ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 6، ص: 607.— المصدر
04 —

قِرْضِميَّة

اسم
4.1معنى من معاني «قِرْضِميَّة»

* القِرْضِميّةُ من الإِبِلِ: نوعٌ يُنْسَبُ إلى قَبيلَةٍ عَرَبيّةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قال ابن دُرَيْد (ت: 321ه=933م):

"وقِرْضِمٌ: اسْمُ قَبيلَةٍ إليهم تُنْسَبُ الإِبِلُ القِرْضِميَّةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 2، ص: 1153.— المصدر
