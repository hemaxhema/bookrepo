ضَغَبَ(فعل) اسْتَضْغَبَ(فعل) ضُغاب(اسم) ضَغابَة(اسم) ضَغيب(اسم) مَضْغَبَة / مُضْغَبَة / مُضْغِبَة(اسم)

المعانى الكلّيّة

  1. الصَّوْتُ
01 —

ضَغَبَ

فعل
1.1معنى من معاني «ضَغَبَ»

* ضَغَبَ (بفَتْحِ الغَيْنِ) الذِّئْبُ، والأَرْنَبُ، ونَحْوُهُما يَضْغَبُ (بفَتْحِ الغَيْنِ) ضُغابًا، وضَغْبًا، وضَغيبًا: صاحَ فمَدَّ صَوْتَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الشَّمَرْدلُ بن شُريكٍ (ت: 80ه=700م) يَصِفُ غِلَظَ صَوْتِ صَيْدٍ:

تَحْتَ جَديدِ الأَرْضِ أَوْ تُرابِهِ

مِنْ كُلِّ شَحّاجِ الضُّحى ضَغّابِهِ

أبو الفرج الأصبهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 13، ص: 254.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والسِّنَّوْرُ يَضْغَبُ، وهو أَنْ يَصيحَ فيَمُدَّ صَوْتَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 369.— المصدر
”وقالَ أبو نُواسٍ (ت: 198ه=814م) يَتَحَدَّثُ عَنْ رَمْقِه طَريدَةَ صَيْدٍ:

في النَّظرِ مِنْ حِمْلاقِه المَقْلوبِ
على رِفِلٍّ بالضُّحى ضَغوبِ

[الرِّفِلُّ: الغَزالُ والطَّيْرُ، يَعني أنّه يَتَبَخْتَرُ في مِشْيَتِه]
ديوان أبي نُواس. تح: الحديثي، ص: 177.— المصدر
1.2معنى من معاني «ضَغَبَ»

وــــــــــ الشَّيْءُ: صَوَّتَ مِن ضَخامَتِهِ وارْتِخائِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110ه=728م) يَهْجو التَّيْمَ ويُشَهِّرُ بعُيوبِ خِلْقَتِهم:

لِكُلِّ التَّيْمِ سَحْبَلَةٌ ‌ضَغوبٌ
تُقاسِمُ نِصْفَ مِعْدَتِهِ الشَّرابا

شرح ديوان جرير. تح: الصاوي، ص: 24.— المصدر
1.3معنى من معاني «ضَغَبَ»

وــــــــــ الأَرْنَبُ: تَضَوَّرَتْ عِنْدَ الأَخْذِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والضَّغيبُ: تَضَوُّرُ الأَرْنَبِ عِنْدَ الأَخْذِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 369.— المصدر
”وقالَ المَعَرّي (ت: 449ه=1057م) في الشَّجاعَةِ:

فلَمْ يُبالِ باللُّوامِ واللَّغْبْ

تَسْمَعُ لِلثَّعْلَبِ فيها كالضَّغْبْ

المعرّي، شروح سقط الزند. تح: السقا وآخرين، ج: 1، ص: 1869.— المصدر
1.4معنى من معاني «ضَغَبَ»

وــــــــــ الرَّجُلُ ضَغَبانًا: تَكَلَّمَ فلَمْ يَسْكُتْ؛ فَهو ضَغّابٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو الخَليلِ الكَلْبِيُّ (ت: 180ه=796م):

"رَجُلٌ ضَغّابٌ، لِلَّذي يَتَكَلَّمُ فَلا يَسْكُتُ ولا يُفْهَمُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو، الجيم. تح: الطحاوي، ج: 2، ص: 201.— المصدر
”وقالَ الشّاعِرُ (206ه=821م) مِنْ شَواهِدِ أبي عَمْرٍو الشَّيْبانِيِّ، يَكُفُّ قَوْمَهُ عَنْ رَجُلٍ مُمِلٍّ:

أُنَهْنِهُ قَوْمي عَنْ صَحابَةِ خالِدٍ
أُشَيِّمَ ضَغّابًا يَصيحُ إِلى الجَنْبِ

أبو عمرو، الجيم. تح: الطحاوي، ج: 2، ص: 201.— المصدر
1.5معنى من معاني «ضَغَبَ»

وــــــــــ: اخْتَبَأَ لِيُفْزِعَ شَخْصًا بصَوْتٍ مُخيفٍ، كصَوْتِ الأَسَدِ ونَحْوِه؛ فهو ضاغِبٌ. (وانظر: ض غ ث)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"الضّاغِبُ: الرَّجُلُ يَخْتَبئُ في الخَمَرِ فَيُفْزِعُ الإِنْسانَ بصَوْتٍ مِثْلِ صَوْتِ السِّباعِ أَوْ صَوْتِ الوَحْشِ، فيُقالُ: ضَغَبَ؛ فَهو ضاغِبٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمود، ج: 8، ص: 18.— المصدر
”وقالَ الشّاعِرُ (255ه=869م) مِنْ شَواهِدِ الجاحِظِ، يَتَحَدَّثُ عَنْ شَخْصٍ اخْتَبَأَ، وتَسَتَّرَ ليُخيفَ غَيْرَه:

يا أَيُّها الضّاغِبُ بالغُمْلولْ

إِنَّكَ غولٌ وَلَدَتْكَ غولْ

[الغُمْلولُ: كُلُ ما يَسْتُر ويُخْفي الشَّيْءَ مِنْ شَجَرٍ وغيرِه]
الجاحظ، كتاب الحيوان. تح: عيون السود، ج: 6، ص: 404.— المصدر
1.6معنى من معاني «ضَغَبَ»

وــــــــــ: اشْتَهى أَكْلَ الضَّغابيسِ؛ فهو ضَغْبٌ، وضَغِبٌ، وهي بتاءٍ. (انظر: ض غ ب س)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَتِ امْرَأَةٌ (216ه=831م) مِنْ شَواهِدِ الأَصْمَعِيِّ:

"طَعامُنا الحارُّ والقارُّ، وإِنْ ذُكِرَتِ الضَّغابيسُ فإِنّي ضَغِبَةٌ".

[والضَّغابيسُ صِغارُ القِثّاءِ]— المصدر
1.7معنى من معاني «ضَغَبَ»

وــــــــــ امْرَأَتَهُ: نَكَحَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385ه=995م):

"ويُقالُ للرَّجُلِ إذا نَكَحَ امْرَأَتَه: قَدْ ضَغَبَها". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 556.— المصدر
02 —

اسْتَضْغَبَ

فعل
2.1معنى من معاني «اسْتَضْغَبَ»

* اسْتَضْغَبَ الفَرَسَ: فَزَّعَها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

تَأْبى بدِرَّتِها إِذا ما اسْتُضْغِبَتْ
إِلّا الحَميمَ فإِنَّه يَتَبَضَّعُ

تفسير الطبري. تح: محمود شاكر، ج: 11، ص: 449.— المصدر
03 —

ضُغاب

اسم
3.1معنى من معاني «ضُغاب»

* الضُّغابُ: صَوْتُ الأَرْنَبِ أو الذِّئْبِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110ه=728م) يَصِفُ حالَ نِسْوَةٍ تَعَرَّضْنَ لِلسَّبْيِ:

يُناطِحْنَ الأَواخِرَ مُرْدَفاتٍ
وتَسْمَعُ مِنْ أَسافِلِها ضُغابَا

أبو عبيدة، النقائض. تح: حور وآخر، ج: 1، ص: 645.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بنُ المُثَنّى (ت: 209ه=825م):

"الضُّغابُ، والضَّغيبُ: صَوْتُ الأَرْنَبِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

أبو عبيدة، النقائض. تح: حور وآخر، ج: 1، ص: 645.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ مالِكٍ (ص: الأَلْفِيَّةِ) (ت: 672ه=1273م):

وهافِتٌ مَصْدَرُه هُفاةُ
مُوازِنُ الضُّغابِ والنُّعابِ

ابن مالك، الإعلام بمثلث الكلام. ص: 208.— المصدر
04 —

ضَغابَة

اسم
4.1معنى من معاني «ضَغابَة»

* الضَّغابَةُ مِنَ الإِبِلِ: الضَّعيفَةُ الهَزيلَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأنْصارِيُّ (ت: 215ه=830م):

"يُقالُ لنُفايَةِ المالِ وضَعْفانِهِ: ضَغاثَةٌ مِنَ الإِبِلِ، ‌وضَغابَةٌ، وغُثايَةٌ، وغُثاثَةٌ، وقُثاثَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمود، ج: 8، ص: 6.— المصدر
05 —

ضَغيب

اسم
5.1معنى من معاني «ضَغيب»

* الضَّغيبُ: صَوْتُ الأَرْنَبِ أو الذِّئْبِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طَرَفَةُ (ت: 60ق ه=564م) يَصِفُ جُبْنَ قَوْمٍ وفَزَعَهم:

إِذا جَلَسوا خَيَّلْتَ تَحْتَ ثِيابِهِمْ
خَرانِقَ توفي بالضَّغيبِ لها نَذْرا

[الخَرانِقُ: صِغارُ الأَرانِبِ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بنُ المُثَنّى (ت: 209ه=825م):

"الضُّغابُ، والضَّغيبُ: صَوْتُ الأَرْنَبِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيدة، النقائض. تح: حور وآخر، ج: 1، ص: 645.— المصدر
5.2معنى من معاني «ضَغيب»

وــــــــــ: صَوْتُ تَقَلْقُلِ قَضيبِ الفَرَسِ في جِرابِه يُسْمَعُ في جَرْيِه.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110ه=728م) يَهْجو:

كَما في خُصى تَيْمٍ ضَغيبٌ كَأَنَّه
ضُغاءُ جِراءٍ في قَراميصِها كُدْرِ

شرح ديوان جرير. تح: الصاوي، ج: 1، ص: 211.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن سيده (458هـ=1066م):

"والضَّغيبُ: صَوْتُ تَقَلْقُلِ الجُرْذانِ في قُنُبِ الفَرَسِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 5، ص: 414.— المصدر
5.3معنى من معاني «ضَغيب»

وــــــــــ: صَوْتُ اللَّبَنِ في القِرْبَةِ ونَحْوِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (291ه=904م) مِنْ شَواهِدِ ثَعْلَبٍ، يُشَبِّهُ صَوْتَ اللَّبَنِ في وِعائِهِ بصَوْتِ الأَرْنَبِ:

كَأَنَّ ضَغيبَ المَحْضِ في حاوِيائِه
مَعَ التَّمْرِ أَحْيانًا ضَغيبُ الأَرانِبِ

مجالس ثعلب. تح: هارون، ص: 70.— المصدر
06 —

مَضْغَبَة / مُضْغَبَة / مُضْغِبَة

اسم
6.1معنى من معاني «مَضْغَبَة / مُضْغَبَة / مُضْغِبَة»

* المَضْغَبَةُ، والمُضْغَبَةُ،والمُضْغِبَةُ مِنَ الأَراضِي: الّتي يَكْثُرُ بها الضَّغابيسُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ الدّينَوَرِيُّ (ت: 282ه=895م):

"وإِذا كانَتِ الأَرْضُ كَثيرَةَ الضَّغابيسِ، قيلَ: أَرْضٌ مَضْغَبَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 3، ص: 283.— المصدر
