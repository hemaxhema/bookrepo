قنزع
قَنازِع(اسم) قُنْزُع(اسم) قَنْزَعَة/قُنْزَعَة/قُنْزُعَة/قِنْزِعَة(اسم) قَنْزَعَة/قُنْزُعَة(اسم) قُنْزُعَة(اسم) مُقَنْزَعَة(اسم)
النظائر الساميّة

احْتَفَظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَرَبِيَّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذْرُ (ق ن ز ع):

في الجِعْزِيَّةِ؛ ومِنْهُ الاسْمُ: ቍንዛዕት (ݣُـْنزَاعت) k'wǝnzāʽt بِمَعْنى "خُصْلَةُ شَعْرٍ- جَديلَةٌ- ضَفيرَةٌ"، وهُوَ يَتَّفِقُ مَعَ الاسْمِ العَرَبِيِّ (قَنْزَعَةٌ) بِمَعْنى (خُصْلَةُ شَعْرٍ- جَديلَةٌ) مَبْنًى، وفي بَعْضِ مَعانيهِ، غَيْرَ أَنَّ الاسْمَ الجِعْزِيَّ خَضَعَ لِإِبْدالِ القافِ كافًا دَفْعِيَّةً؛ ومِنْ مَجيئِهِ بِمَعْنى "خُصْلَةُ شَعْرٍ" ما وَرَدَ في سِفْرِ القُضاةِ: ወይቤላ እመ ፀፈርክዮን ለሰብዑ ቈናዝዕየ ዘርእስየ ወተከልክዮን በመትከል ውስተ አረፍት ወእደክም እንከ ከመ አሐዱ እምነ እጓለ እመሕያው። أَيْ: "فَقالَ لَها: إِنْ ضَفَرْتَ خُصْلاتِ شَعْري السَّبْعَ بِمِغْزَلٍ، وثَبَّتَها بِوَتَدٍ، فَإِنّي أُصْبِحُ ضَعيفًا كَأَيِّ واحِدٍ مِنَ النّاسِ".

(ኦሪት ዘመሳፍንት: 16:14)

01 —

قَنازِع

اسم
1.1معنى من معاني «قَنازِع»

* القَنازِعُ: الفُحْشُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيْدٍ (ت: 35ق.ه=588م) يَمْدَحُ نَفْسَهُ:

‌فلَمْ ‌أَجْتَعِلْ ‌فيما ‌أَتَيْتُ مَلامَةً
أَتَيْتُ الجَمالَ واجْتَنَبْتُ القَنازِعا

ديوان عدي بن زيد. تح: المعيبد، ص: 145.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309ه=921م):

"والقَنازِعُ: الفُحْشُ ".

كراع النمل، المنتخب. تح: العمري، ص: 197.— المصدر
1.2معنى من معاني «قَنازِع»

وــــــــــ: أَنْ يُؤْخَذَ الشَّعْرُ، ويُتْرَكَ مِنْهُ مَواضِعُ لا تُؤْخَذُ. واحِدُها قُنْزُعَةٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”رَوى عَبْدُ اللهِ بْنُ عُمَرَ (ت: 74ه=694م):

"أَنَّ النَّبيَّ ﷺ ‌نَهى عَنِ ‌القَنازِعِ".

أبو عوانة، المسند الصحيح. الجامعة الإسلامية، ج: 17، ص: 128.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعيُّ (ت: 216ه=831م):

"القَنازِعُ واحِدُها قُنْزُعَةٌ، وهُوَ أَنْ يُؤْخَذَ الشَّعْرُ، ويُتْرَكَ مِنْهُ مَواضِعُ لا تُؤْخَذُ ".

الأزهري، تهذيب اللغة. تح: مرعب، ج: 3، ص: 182.— المصدر
1.3معنى من معاني «قَنازِع»

وــــــــــ: الدَّواهي.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابيِّ (ت: 231ه=845م):

"القَنازِعُ: الدَّواهي ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 3، ص: 182.— المصدر
1.4معنى من معاني «قَنازِع»

وــــــــــ: صِغارُ القَوْمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"والقَنارِعُ: صِغارُ النّاسِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 404.— المصدر
02 —

قُنْزُع

اسم
2.1معنى من معاني «قُنْزُع»

* القُنْزُعُ: الشَّعْرُ حَوْلَ الرَّأْسِ. (ج) قَنازِعُ وقُنْزُعاتٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ زُهَيْرُ بْنُ أَبي سُلمى (ت: 13ق.ه=609م) يَصِفُ رَجُلًا طالَ اعْتِمامُهُ في السَّفَرِ:

وأَشْعَثَ قَدْ طارَتْ قَنازِعُ رَأْسِهِ
دَعَوْتُ عَلى طولِ الكَرى ودَعاني

ابن دقيق العيد، شرح الإلمام. تح: العبد الله، ج: 3، ص: 367.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رَسولُ اللهِ ﷺ (ت: 11ه=632م) لِأُمِّ سُلَيْمٍ:

«خَضِّلي قَنازِعَكِ».

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 1، ص: 306.— المصدر
”وقالَ عَبْدُ اللهِ بْنُ عُمَرَ (ت: 74ه=694م) لَمّا سُئِلَ عَن رَجُلٍ أَهَلَّ بِعُمْرَةٍ وقَدْ لَبَّدَ وهُوَ يُريدُ الحَجَّ:

"خُذْ مِنْ ‌قَنازِعِ رَأْسِكِ، أَوْ مِمّا يَشْرُفُ مِنْهُ".

القاسم بن سلام، غريب الحديث. تح: خان، ج: 4، ص: 273.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"والقُنْزُعُ واحِدُ القَنازِعِ - قَنازِعِ الرَّأْسِ - وهُوَ الشَّعْرُ المُجْتَمِعُ في نَواحي الرَّأْسِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ص: 1154.— المصدر
2.2معنى من معاني «قُنْزُع»

وــــــــــ: الرّيشُ المُجْتَمِعُ عَلى رَأْسِ الطَّيْرِ ونَحْوِهِ. وهُوَ بِتاءٍ. (ج) قَنازِعُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ذو الرُّمَّةِ (ت: 117ه=735م) يَصِفُ القَطا وفِراخَها:

يَنُؤْنَ ولَمْ يُكْسَيْنَ إِلّا قَنازِعا
مِنَ الرّيشِ تَنْواءَ الفِصالِ الهَزائِلِ

[يَنُؤْنَ: يَنْهَضْنَ]
ديوان ذي الرمة. تح: طراد، ص: 464.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو السَّمْحِ الكِلابيُّ (ت: 185ه=801م) مِنْ رُواةِ الشَّيبانيِّ:

"العِفْرِيَةُ: عِفْرِيَةُ الدّيكِ ‌وقُنْزُعَتُهُ".

[عِفْرِيَةُ الدّيكِ: ريشُ عُنُقِهِ]
الشيباني، الجيم. تح: الأبياري، ج: 2، ص: 257.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ اللهِ البَسّامُ (1423ه=2002م):

"‌الهُدْهُدُ: ‌جِنْسُ طَيْرٍ ‌ِمنَ ‌الجَواثِمِ الرَّقيقاتِ المَناقيرِ، لَهُ قُنْزُعَةٌ عَلى رَأْسِهِ".

البسام، توضيح الأحكام. ج: 7، ص: 15.— المصدر
2.3معنى من معاني «قُنْزُع»

وــــــــــ: ضَعيفُ القَوْمِ. (ج) قَنازِعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309ه=921م):

"وقَنازِعُ النّاسِ: أَقْماؤُهُمْ وضُعَفاؤُهُمْ، واحِدُهُمْ قُنْزُعٌ".

كراع النمل، المنتخب. تح: العمري، ص: 197.— المصدر
03 —

قَنْزَعَة/قُنْزَعَة/قُنْزُعَة/قِنْزِعَة

اسم
3.1معنى من معاني «قَنْزَعَة/قُنْزَعَة/قُنْزُعَة/قِنْزِعَة»

* القَنْزَعَةُ والقُنْزَعَةُ والقُنْزُعَةُ والقِنْزِعَةُ: الشَّعْرُ حَوْلَ الرَّأْسِ. (ج) قَنازِعُ وقُنْزُعاتٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حُمَيْدٌ الأَرْقَطُ (ت: 95ه=714م) يَصِفُ الصَّلَعَ:

كَأَنَّ طَسًّا بَيْنَ قُنْزُعاتِهِ

مَرْتًا تَزِلُّ الكَفُّ عَنْ صَفاتِهِ

ابن منظور، لسان العرب. تح: اليازجي وآخرين، ج: 6، ص: 122.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمِرٌ الهَرَويُّ (ت: 255ه=869م):

"القَنازِعُ مِنَ الشَّعْرِ: ما يَبْقى في نَواحي الرَّأْسِ مُتَفَرِّقًا. واحِدُها قُنْزُعَةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 3، ص: 182.— المصدر
04 —

قَنْزَعَة/قُنْزُعَة

اسم
4.1معنى من معاني «قَنْزَعَة/قُنْزُعَة»

* القَنْزَعَةُ والقُنْزُعَةُ: ما تَتَّخِذُهُ المَرْأَةُ عَلى رَأْسِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"‌القَنْزَعة ‌والقُنْزُعَةُ: الَّتي تَتَّخِذُها المَرْأَةُ عَلى رَأْسِها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 292.— المصدر
4.2معنى من معاني «قَنْزَعَة/قُنْزُعَة»

وــــــــــ: الخُصْلَةُ مِنَ الشَّعْرِ الَّتي تُتْرَكُ عَلى رَأْسِ الصَّبِيِّ. (ج) قَنازِعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والقَنْزَعَةُ: الخُصْلَةُ مِنَ الشَّعْرِ الَّتي تُتْرَكُ عَلى رَأْسِ الصَّبيِّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 292.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ خُلود العموش (ت: 1431ه=2010م):

"أَمّا تَرْكُ القُنْزُعَةِ - وهِيَ الخُصْلَةُ مِنَ الشَّعْرِ على رَأْسِ الصَّبيِّ – فلِتَدْليلِهِ، ولِيُرى جَمالُهُ".

العموش، أشعار ترقيص الأطفال. الجامعة الهاشمية، ص: 123.— المصدر

قُنْزُعَة

اسم
5.1معنى من معاني «قُنْزُعَة»

* القُنْزُعَةُ: بَقايا الشَّجَرِ والعُشْبِ. (ج) قَنازِعُ. (ز)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ذو الرُّمَّةِ (ت: 117ه=735م) يَصِفُ أَرْضًا مُقْفِرَةً:

سَباريتَ إِلّا أَنْ يَرى مُتأَمِّلٌ
قَنازِعَ إِسْنامٍ بِها وتَغامِ

[السَّباريتُ: أَرْضٌ لا زَرْعَ فيها ولا نَبْتَ. الإسْنامُ: الشَّجَرُ، واحِدَتُها إِسْنامَةٌ]

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"‌والقَنازِعُ: الدَّواهي. وهُوَ مِنَ الكَلَأِ: القِطَعُ المُعَرّاةُ، الواحِدَةُ: قُنْزُعَةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)
