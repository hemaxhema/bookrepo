في الفينيقيَّةِ؛ ومِنْهُ الفِعْلُ: skn(نكس) nksبِمَعْنى: "قَضى/ أَفْنى"، وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَكَشَ) مَبْنًى ومَعْنًى، غَيْرَ أَنَّ الفِعْلَ الفينيقيَّ خَضَعَ لإِبْدالِ الشّينِ سينًا؛ ومِنْ مَجيئِهِ بِمَعْنى: "فَنِيَ" بصيغَةِ الاسْمِ ما وَرَدَ في نَقْشِ قَرْطاجَ: " xskn. Nb. Lob"؛ أَيْ: "أفْنَيْتُ ابْنَ بَعْلٍ".

(DNWSI: p 731)

في الآراميَّةِ؛ ومِنْهُ الفِعْلُ: נַכֵּישׁ (نَكّ۪ٮٰش) nakkēš بِمَعْنى: "اجْتَثَّ/ اسْتَأْصَلَ"؛ وهُوَ يَتَّفِقُ في أُصولِهِ معَ أُصولِ الفِعْلِ العَربيِّ (نَكَشَ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَكَشَ" ما وَرَدَ في التَّلْمودِ البابِليِّ: "בְּאַתְרָא דְּלָא מְנַכְּשִׁי וַאֲזַל אִיהוּ וְנַכֵּישׁ"؛ أَيْ: "في المَوْضِعِ الّذي لا تَنْكُشُ [الأَعْشابَ الضّارَّةَ] فيهِ، اذْهَبْ وانْكُشْـ[ـها]".

(ב''מ: קג, ב)

في العِبْريَّةِ؛ ومِنْهُ الفِعْلُ: נִכֵּשׁ (نِكّ۪ٮٰـش) nikkēš بِمَعْنى: "عَزَقَ الأَرْضَ، قَلَّبَها وأَزالَ أَعْشابَها الضّارَّةَ"، وهُوَ يُوافِقُ الفِعْلَ العَرَبيَّ (نَكَشَ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَكَشَ" ما وَرَدَ في التّوسْفتا: "פּוֹעֲלִים שֶׁהָיוּ מְנַכְּשִׁין בַּשָּׂדֶה וּמְעַדְּרִין בַּשָּׂדֶה לֹא יֹאכְלוּ מִן הַזַּיִת‏"؛ أَيْ: "والعُمّالُ الّذينَ كانوا يَنْكُشونَ الحَقْلَ وَيَعْزِقونَهُ لا يَأْكُلونَ مِنْ أَشْجارِ الزَّيْتونِ".

(תוס': מעשרות, ב', י"ג)

في السُّرْيانيَّةِ؛ ومِنْهُ الفِعْلُ: ܢܟ̥ܰܫ (نْخَش) nəẖaš بِمَعْنى: "اسْتَأْصَلَ"، وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَكَشَ) مَبْنًى ومَعْنًى.

(ܠܟܣܝܩܘܢ ܕܐܘܓܝܢ ܡܢܢܐ: ܢܟܫ)

01 —

نَكَشَ

فعل
1.1معنى من معاني «نَكَشَ»

* نَكَشَ (بِفَتْحِ الكافِ) فُلانٌ الشَّيْءَ ومِنْهُ يَنْكُشُ ويَنْكِشُ (بِضَمِّ الكافِ وكَسْرِها) نَكْشًا ونُكْشانًا: أَفْناهُ فأَتى عَلَيْهِ وفَرَغَ مِنْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"النَّكْشُ: شِبْهُ الأَتِيِّ عَلى الشَّيْءِ، والفَرَاغِ مِنْهُ. نَكَشْتُهُ ونَكَشْتُ مِنْهُ؛ أَيْ: أَتَيْتُ عَلَيْهِ، وفَرَغْتُ مِنْهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 297.— المصدر
”وقال ابْنُ الرُّومِيِّ (ت: 283ه=896م) يَذْكُرُ قَصيدَةً هَجا بِها:

رُوَيْدًا تَزُرْكَ عَلى رِسْلِها
وتَجْري كَعَهْدِكَ لَمْ تُنْكَشِ

ديوان ابن الرومي. تح: أحمد بسج، ج: 2، ص: 248.— المصدر
1.2معنى من معاني «نَكَشَ»

وــــــــــ الأَمْرَ، وفيهِ: بَحَثَ فيهِ ونَقَّبَ عَنْهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُرْتَضى الزَّبيدِيُّ (ت: 1205ه=1790م):

"النَّكْشُ: ‌البَحْثُ في الأُمورِ، والنَّقْبُ عَنْها ". (وَهَذِهِ الدَّلالَةُ سَجَّلَتْها المعَاجِمُ في العَصْرِ الحَدِيثِ)

الزبيدي، تاج العروس. تح: حجازي وآخرين، ج: 17، ص: 429.— المصدر
1.3معنى من معاني «نَكَشَ»

وــــــــــ الأَرْضَ: حَفَرَها وقَلَّبَها بِعودٍ ونَحْوِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيّ بْنُ أَبي طالبٍ (ت: 40ه=661م):

"كُنَّا مَعَ رَسولِ اللهِ ﷺ في جَنازَةٍ فَجَعَلَ ‌يَنْكُشُ ‌الأَرْضَ بِعودٍ".

الصالحي، سبل الهدى. تح: عبد الموجود وآخر، ج: 7، ص: 136.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ الرَّحْمنِ العَدَوِيُّ (590ه=1194م) يَذْكُرُ أَماراتِ مُرْتَكِبِ الجَرائِمِ:

"يَكْثُرُ تَثاؤُبُهُ، ويَعْرَقُ جَبينُهُ، أَوْ يَفْتِلُ هُدْبَ ثِيابِهِ ويَبْعَثُ بِها، أَوْ ‌يَنْكُشُ ‌الأَرْضَ بِإِبْهامِ رِجْلِهِ".

العدوي، المنهج المسلوك في سياسة الملوك. تح: علي الموسى، ج: 1، ص: 592.— المصدر
1.4معنى من معاني «نَكَشَ»

وــــــــــ البِئْرَ: اسْتَقى مِنْها حَتَّى نَزَفَها وذَهَبَ ماؤُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ قُتَيْبَةَ الدِّينَوَرِيُّ (ت: 276ه=889م):

"هَذِهِ ‌بِئْرٌ ما ‌تُنْكَشُ؛ أَيْ: ما تُنْزَحُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 2، ص: 106.— المصدر
1.5معنى من معاني «نَكَشَ»

وــــــــــ البِئْرَ ونَحْوَها: أَخْرَجَ ما فيها مِنَ الماءِ والطِّينِ ونَحْوِهما.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"ونَكَشْتُ الرَّكيَّةَ أَنْكُشُها نَكْشًا، إِذا أَخْرَجْتُ ما فيهَا مِنَ الحَمْأَةِ والطِّينِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 878.— المصدر
1.6معنى من معاني «نَكَشَ»

وــــــــــ الطَّعامَ ونَحْوَهُ: أَفْناهُ وقَضى عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبَّادٍ (ت: 385ه=995م):

"ونَكَشْنا ماءَ الرَّكِيَّةِ وطَعامَ فُلانٍ: أَيْ نَزَحْناها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 165.— المصدر
02 —

نُكِش

فعل
2.1معنى من معاني «نُكِش»

* نُكِشَ فُلانٌ: نَفِدَ ما عِنْدَهُ مِنْ كَثْرَةِ سُؤالِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مِسْحَلٍ الأَعْرَابيُّ (ت: 230ه=848م):

" يُقالُ: قَدْ شُفِهَ الرَّجُلُ، ووُكِظَ، ورُغِثَ، ونُكِدَ، ونُجِفَ، وجُلِذَ، وثُمِدَ، وثُمِلَ، ‌ونُكِشَ ... وذَلِكَ بِمَعْنى كُدَّ بِالمَسْأَلَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو مسحل، النوادر. تح: عزة حسن، ج: 1، ص: 70.— المصدر
2.2معنى من معاني «نُكِش»

o ورَجُلٌ بَحْرٌ لا يُنْكَشُ: كَريمٌ لا يَنْفَدُ كَرَمُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنَّى (ت: 209ه=825م):

"يُقالُ: هو ‌بَحْرٌ ‌لا ‌يُنْكَشُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

معمر بن المثنى، شرح النقائض. تح: حوّر وآخر، ج: 1، ص: 163.— المصدر
2.3معنى من معاني «نُكِش»

o وفَرَسٌ بَحْرٌ لا يُنْكَشُ حُضُرُهُ: تُقالُ لِلفَرَسِ واسِعِ الجَرْيِ كَثيرِ العَدْوِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَر بْنُ المُثَنَّى (ت: 209ه=825م):

" يُقَالُ لِلفَرَسِ الجَوادِ: إِنَّهُ ‌لَبَحْرٌ ‌لَا ‌يُنْكَشُ حُضُرُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب وآخر، ج: 5، ص: 68.— المصدر
2.4معنى من معاني «نُكِش»

o وعِنْدَهُ شَجاعَةٌ لا تُنْكَشُ: مِقْدامٌ لا يَهابُ أَحَدًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رَجُلٌ مِنْ قُرَيْشٍ (ت: 224ه=838م) مِنْ شَواهِدِ أَبي عُبَيْدٍ، يَصِفُ عَلِيَّ بْنَ أَبي طالِبٍ:

"عِنْدهُ ‌شَجاعَةٌ ‌لا ‌تُنْكشُ".

الأزهري، تهذيب اللغة. تح: هلالي، ج: 10، ص: 26.— المصدر
2.5معنى من معاني «نُكِش»

o ولُمْعَةٌ ما تُنْكَشُ: أَيْ ما تُسْتَأْصَلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبَّادٍ (ت: 385ه=995م):

"لُمْعَةٌ ‌ما ‌تُنْكَشُ: أَيْ ما تُسْتَأْصَلُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 165.— المصدر
03 —

انْتَكَشَ

فعل
3.1معنى من معاني «انْتَكَشَ»

* انْتَكَشَ فُلانٌ البِئْرَ ونَحْوَها: أَخْرَجَ ما فيها مِنَ الماءِ والطّينِ ونَحْوِهِما.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاغانِيُّ (ت: 650ه=1252م):

"انْتَكَشَ ‌ماءَ ‌البِئْرِ؛ أَيْ نَكَشَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: محمد أبو الفضل، ج: 3، ص: 520.— المصدر
04 —

اسْتَنْكَشَ

فعل
4.1معنى من معاني «اسْتَنْكَشَ»

* اسْتَنْكَشَ فُلانٌ وغَيْرُهُ: اسْتَنْهَدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"واسْتَنْكَشَ، أَيِ: اسْتَنْهَدَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 5، ص: 297.— المصدر
4.2معنى من معاني «اسْتَنْكَشَ»

وــــــــــ فُلانٌ البِئْرَ ونَحْوَها: سَعى إِلى إِخْراجِ ما فيها مِنَ الماءِ ونَحْوِهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَخْطَلُ (ت: 90ه=708م):

رَفيعُ الـمُنَى لا يَسْتَقِلُّ بِـهَمِّهِ
سَؤومٌ ولا ‌مُسْتَنْكِشُ البَحْرِ ناضِبُهْ

ديوان الأخطل. تح: مهديّ ناصر الدّين، ص: 39.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبَّادٍ (ت: 385ه=995م):

"واسْتَنْكَشَ ماءَ البِئْرِ: بِمَعْنى نَكَشَهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 165.— المصدر
05 —

مِنْكاش

اسم
5.1معنى من معاني «مِنْكاش»

* المِنْكاشُ: أَداةٌ يُنْكَشُ بِها، (ج) مَناكيشُ. (لُغَةٌ في: مِنْقاش).

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال مُرْتَضى الزَّبِيدي (ت: 1205ه=1790م):

"‌والمِنْكاشُ: المِنْقاشُ، لُغَيَّةٌ".

الزبيدي، تاج العروس. تح: حجازي وآخرين، ج: 17، ص: 429.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد كُرْد عَلِيّ (1330ه=1912م) يَذْكُرُ مَكْتَبَةَ أَحْمَد زَكي باشا:

"ولَكِنَّكَ إِذا أَتَيْتَ الخِزانَةَ الزَّكِيَّةَ؛ تَجِدْ قِسْمًا عَظيمًا مِنْها، وتَقْرَأْ فيها المُعْجِبَ المُطْرِبَ مِمَّا يَدُلُّ عَلى حالَةِ البِلادِ في تِلْكَ الأَيّامِ، وأَنْتَ لَوِ الْتَمَسْتَهُ ‌بِالمِنْكاشِ في تَضاعيفِ الكُتُبِ أَوْ بِـمُساءَلَةِ الشُّيوخِ؛ لا تَصِلْ إِلَيْهِ أَلْبَتَّةَ .هَذا مِثالٌ واحِدٌ مِنْ أَمْثِلَةٍ كَثيرَةٍ".

محمد كرد علي، مجلة المقتبس، العدد 79. ، ص: 27.— المصدر
