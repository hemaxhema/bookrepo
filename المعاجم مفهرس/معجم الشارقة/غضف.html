01 —

غَضَفَ

فعل
1.1معنى من معاني «غَضَفَ»

* غَضَفَتْ (بِفَتْحِ الضّادِ) الأُتُنُ ونَحْوُها تَغْضِفُ (بِكَسْرِ الضّادِ) غَضْفًا: أَخَذَ الجَرْيَ أَخْذًا بِغَيْرِ حِسابٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أُمَيَّةُ بْنُ أَبي عائِذٍ (ت: 85هـ= 704م) يَصِفُ حِمارًا وَحْشِيًّا وأُتُنَهُ:

يَغُضُّ ويَغْضِفْنَ مِنْ رَيِّقٍ
كَشُؤْبوبِ ذي بَرَدٍ وانْسِحالِ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 504. — المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السُّكَّرِيُّ (ت: 275هـ= 888م) يَشْرَحُ البَيْتَ السّابِقَ:

"يَغْضِفْنَ غَضْفًا: يُريدُ الأُتُنَ يَأْخُذْنَ أَخْذًا مِنَ الجَرْيِ بِغَيْرِ حِسابٍ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 504. — المصدر
1.2معنى من معاني «غَضَفَ»

وــــــــــ فُلانٌ: نَعِمَ بالُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"والغاضِفُ: النّاعِمُ البالِ، ويُقالُ: غَضَفَ يَغْضِفُ غُضوفًا ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 368. — المصدر
”وقالَ الشّاعِرُ (356هـ= 967م، مِنْ شَواهِدِ أَبي عَلِيٍّ القاليِّ) يَمْدَحُ:

كَمِ اليَوْمَ مَغْبُوطٌ بخَيْرِكَ بائِسٌ
وآخَرُ لَمْ يُغْبَطْ بخَيْرِكَ غاضِفُ

القالي، البارع في اللغة. تح: الطعان، ص: 261. — المصدر
1.3معنى من معاني «غَضَفَ»

وــــــــــ الكَلْبُ وغَيْرُهُ: ضَرَطَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ= 831م):

"غَضَفَ: إِذا ضَرَطَ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القالي، البارع في اللغة. تح: الطعان، ص: 262. — المصدر
1.4معنى من معاني «غَضَفَ»

وــــــــــ العَيْشُ غُضوفًا: تَوَسَّعَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القوطِيَّةِ (ت: 367هـ= 977م):

"وغَضَفَ العَيْشُ غُضوفًا: تَوَسَّعَ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: فودة، ص: 28. — المصدر
1.5معنى من معاني «غَضَفَ»

وـــــــــــ فُلانٌ مِنَ الطَّعامِ ونَحْوِهِ غَضْفًا: أَخَذَ وغَرَفَ مِنْهُ في سَمْحٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو سَعيدٍ السُّكَّرِيُّ (ت: 275هـ= 888م):

"يُقالُ: غَضَفَ فُلانٌ مِنْ طَعامٍ لَيِّنٍ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 504. — المصدر
1.6معنى من معاني «غَضَفَ»

وــــــــــ الكَلْبُ أُذُنَهُ، غَضْفًا، وغَضْفانًا، وغَضَفانًا: أَرْخاها وكَسَرَها، وقَدْ يَكونُ ذَلِكَ مِنْ ريحٍ أَصابَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأنصارِيُّ (ت: 215هـ= 830م):

"وقالوا: غَضَفَ الكَلْبُ أُذُنَهُ أَشَدَّ الغَضَفانِ: إِذا لَوَتْها الرّيحُ مِنْ رِقَّتِها، أَوْ لَواها هُوَ أَشَدَّ الغَضْفِ أَيْضًا ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو زيد الأنصاري، النوادر. تح: محمد عبد القادر، ص: 544.— المصدر
1.7معنى من معاني «غَضَفَ»

وــــــــــ فُلانٌ العودَ ونَحْوَهُ: كَسَرَهُ أَوْ لَواهُ فَلَمْ يُنْعِمْ كَسْرَهُ. (انظر: ف ض غ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244هـ= 858م):

"وغَضَفْتُ أَغْضِفُ غَضْفًا، وخَضَدْتُ أَخْضِدُ خَضْدًا، وغَرَضْتُ أَغْرِضُ غَرْضًا؛ فَهؤُلاءِ الثَّلاثُ: الكَسْرُ الَّذي لَمْ يَبِنْ، مِنْ رَطْبٍ أَوْ يابِسٍ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، الألفاظ. تح: قباوة، ص: 18.— المصدر
02 —

غَضِفَ

فعل
2.1معنى من معاني «غَضِفَ»

* غَضِفَ (بِكَسْرِ الضّادِ) اللَّيْلُ ونَحْوُهُ يَغْضَفُ (بِفَتْحِ الضّادِ) غَضَفًا: اشْتَدَّ سَوادُهُ وظُلْمَتُهُ، فَهُوَ أَغْضَفُ، وهِيَ غَضْفاءُ. (ج) غُضْفٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال عُبَيْدُ بنُ عَبْدِ العُزّى السَّلامِيُّ (ت: 82 ق هـ= 542م) يَصِفُ ارْتِحالَهُ في صَحْراءَ موحِشَةٍ:

دَعاني بِها داعي رَميمٍ وبَيْنَنا
بَهيمُ الحَواشي ذو أَهاويلَ أَغْضَفُ

محمد بن المبارك، منتهى الطلب. تح: طريفي، ج: 8، ص: 283.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ= 787م):

"ولَيْلٌ أَغْضَفُ: تُشَبَّهُ ظُلْمَتُهُ بِالغُبارِ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 368.— المصدر
”وقالَ أَبو تَمّامٍ (ت: 231هـ= 845م) يَمْدَحُ:

فَمَتى تَعَثَّرَ بِالرِّفاقِ ذَكَرْتَهُ
فَيَمُرُّ تَحْتِيَ قِطْعَ لَيْلٍ أَغْضَفِ

التبريزي، شرح ديوان أبي تمام. تح: الأسمر، ج: 1، ص: 436.— المصدر
2.2معنى من معاني «غَضِفَ»

وــــــــــ الكَلْبُ وغَيْرُهُ: طالَتْ أُذُنُهُ، واسْتَرْخَتْ، وتَكَسَّرَتْ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق هـ=544م) يَذْكُرُ ثَوْرًا وَحْشِيًّا تُطارِدُهُ كِلابُ صَيْدٍ:

حَتّى إِذا قالَ نَالَتْهُ سَوابِقُها
غُضْفٌ جَواهِلُ في أَشْعارِها زَبَبُ

ديوان امرئ القيس. تح: عليان وآخر، ج: 2، ص: 711. — المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى (ت: 7هـ= 628م) يَصِفُ كِلابًا تُطارِدُ ثَوْرًا وَحْشِيًّا:

وتَلَتْهُ غُضْفٌ طَوارِدُ كالنَّحْـ
ـلِ مَغاريثُ هَمُّهُنَّ اللَّحاقُ

ديوان الأعشى. تح: محمد حسين، ص: 213.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بنُ أَحْمَدَ (ت: 170هـ= 787م):

"وانْغَضَفَتْ أُذُنُه، أَيِ: اسْتَرْخَتْ مِنْ غَيْرِ خِلْقَةٍ، وغَضِفَتْ: إذا كانَتْ خِلْقَةً، وكِلابٌ غُضْفٌ: مُسْتَرْخِيةُ الآذانِ، يُقالُ: أُذُنٌ غَضْفاءُ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 368.— المصدر
”وقالَ أَبو نُواسٍ (ت: 198هـ= 813هـ) يَصِفُ كِلابَ صَيْدٍ:

يَدْعو إِلى الصَّيْدِ أَلا قُلتَ انطَلِقْ

بِأَكْلُبٍ غُضْفٍ صَحيحاتِ الحَدَقْ

ديوان أبي نواس. تح: واصف، ص: 218.— المصدر
2.3معنى من معاني «غَضِفَ»

وــــــــــ السَّهْمُ: غَلُظَ ريشُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَيَّةَ النُّمَيْرِيُّ (ت: 183هـ= 799م) يَفْخَرُ:

فَلَمّا رَمَيْناهُمْ بِكُلِّ مُؤَزَّرٍ
بغُضْفٍ تَخيَّرْنَ الظُّهارَ الخَوافِيا

عَلى كُلِّ عِجْزٍ مِنْ رَكوضٍ تَرى لَها
هِجارًا يُقاسي طائِفًا مُتَعادِيا

مَشَيْنا إِلَيْهِمْ في الحَديدِ كَأَنَّنا
قَياسِرُ لاقَتْ بالعَنِيّةِ طالِيا

محمد بن المبارك، منتهى الطلب. تح: طريفي، ج: 7، ص: 175.— المصدر
2.4معنى من معاني «غَضِفَ»

وــــــــــ البِئْرُ: انْهَدَمَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَسْعَدِيُّ (206هـ= 821م، مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانِيِّ):

"غَضِفَتِ القَليبُ مِنْ كَثْرَةِ مائِها، وهُوَ أَنْ تَنْهَدِمَ".

أبو عمرو الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 5.— المصدر
2.5معنى من معاني «غَضِفَ»

وــــــــــ السَّنَةُ ونَحْوُها: أَخْصَبَتْ، واتَّسَعَ عَيْشُها ونَعِمَ ورَغُدَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ= 846م):

"سَنَةٌ غَضْفاءُ، وغَلْفاءُ، إِذا كانَتْ مُخْصِبَةً، وعَيْشٌ أَغْضَفُ وأَغلَفُ: رَغْدٌ واسِعٌ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 16.— المصدر
03 —

أَغْضَفَ

فعل
3.1معنى من معاني «أَغْضَفَ»

* أَغْضَفَ المكانُ وغَيْرُهُ: كَثُرَ خَيْرُهُ، مِنَ الأَنْعامِ وغَيْرِها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أُحَيْحَةُ بْنُ الجُلاحِ (ت: 129ق هـ=496م) يَصِفُ اتِّساعَ عَيْشِهِ:

إِذا جُمادى مَنَعَتْ قَطْرَها
زانَ جَناني عَطَنٌ مُغْضِفُ

ديوان أحيحة بن الجلاح. تح: باجودة، ص: 68. — المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْزُ بنُ سَوادَةَ (255هـ= 869م، مِنْ رُواةِ شَمِرٍ):

"وعَطَنٌ مُغْضِفٌ: إِذا كَثُرَ نَعَمُهُ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 15. — المصدر
3.2معنى من معاني «أَغْضَفَ»

وــــــــــ الثَّمَرَةُ: تَدَلَّتْ واقْتَرَبَ إِدْراكُها ونُضْجُها، ولَم يَبْدُ صَلاحُها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عُمَرُ بْنُ الخَطّابِ (ت: 23هـ=644م) يَذْكُرُ بابًا مِنْ أَبْوابِ الرِّبا:

"الثَّمَرَةُ تُباعُ وهِيَ مُغْضِفَةٌ".

ابن الجوزي، غريب الحديث. تح: قلعجي، ج: 2، ص: 158.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ شَمِرٌ (ت: 255هـ= 869م):

"ثَمَرَةٌ مُغْضِفَةٌ: إِذا تَقارَبَتْ مِنَ الإِدْراكِ ولَمّا تُدْرِكْ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 13. — المصدر
3.3معنى من معاني «أَغْضَفَ»

وــــــــ النَّخْلَةُ: كَثُرَ سَعَفُها، وساءَ ثَمَرُها، فَهِيَ مُغْضِفٌ، ومُغْضِفةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"ونَخْلَةٌ مُغْضِفٌ: كَثُرَ سَعَفُها، وساءَ ثَمَرُها ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 368. — المصدر

وــــــــــ السَّماءُ: أَخَالتْ لِلمَطَرِ، وذلِكَ إِذا لَبِسَها الغَيْمُ.

ع

"الغَضَفَةُ: القَطاةُ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 1159. — المصدر

وــــــــــ: الأَكَمَةُ الصَّغيرَةُ البَيْضاءُ. (ج) غِضافٌ.

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَبّادٍ (ت: 385هـ=995م):

"والغِضافُ مِنَ الإِكامِ -واحِدُها غَضَفَةٌ- هِيَ الإِكامُ الصَّغيرَةُ البَيْضاءُ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: آل ياسين، ج: 4، ص: 554.— المصدر
