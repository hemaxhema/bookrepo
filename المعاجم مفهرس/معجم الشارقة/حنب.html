حنب
حَنِبَ(فعل) حَنَّبَ(فعل) تَحَنَّبَ(فعل) حَنْبَة(اسم) حُنْبوب(اسم)
النقوش العربيّة

وَرَدَ الجَذْرُ (ح ن ب ح ن ب ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ الجَنوبِيّةِ، مِنْ بَيْنِها نَقْشٌ سَبَئِيٌّ يَعُودُ تاريخُهُ إلى القَرْنِ الثّالِثِ المِيلادِيِّ (ZI 20)، وَرَدَ فيهِ الاسْمُ (ح ن ب ḥnb ) لَقَبًا لِصاحِبِ مَنْصِبٍ يَتْبَعُ المَلِكَ)، فَجاءَ في النَّقْشِ: "ح ن ب/ م ل ك ن" أيْ: "حَنْبُ المَلِكِ".

(Beeston and Others, 1982: 69)

المعانى الكلّيّة

  1. الاعْوِجاجُ والانْحِناءُ في الشَّيْءِ
  2. الإحكامُ والتَّقويسُ
01 —

حَنِبَ

فعل
1.1معنى من معاني «حَنِبَ»

* حَنِبَ ( بِكَسْرِ النّونِ) الشَّيءُ يَحْنَبُ (بِفَتْحِ النّونِ) حَنَبًا: اعْوَجَّ وانْحَنى.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَبْدُ اللهِ بْنُ رَواحَةَ (5ق.ه=617م) يَفْتَخِرُ بِقَوْمِهِ:

نَحْنُ حُماةُ الأَطَامِ في سالِفِ الــــ
ـــدَّهْرِ وقِدْمًا سُقْناكُمُ حَنَبا

ديوان عبد الله بن رواحة. تح: وليد قصاب، ج: 1، ص: 126.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رُؤْبةُ بْنُ العَجّاجِ (83ه=762م) يَمْدَحُ خِنْدِفًا وقَيْسًا:

ضَرْبٌ وطَعْنٌ بالقَنا نَحيسُ

مُعْتَرِضٌ أَوْ مِسْعَرٌ دَعُوسُ

يَحْنَبُ عن حَنَقِها تُيــُـــــوسُ

ديوان رؤبة، تح: وليم البروسيي، ج: 3، ص: 77.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170ه=787م):

"الحَنَبُ: اعْوَجاجٌ في السَّاقَيْنِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 250.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ سِوارٍ الدِّمَشْقِيُّ (ت: 677ه=1278م) يَذْكُرُ حالَهُ:

ما حيلَتي قَرَّبوني مِنْ مَحَبَّتِهِمْ
وحالَ دونَهُمُ التَّقْريبُ والحَنَبُ

ديوان ابن سوار الدمشقي. تح: محمد أديب الجادر، ج: 1، ص: 584.— المصدر
1.2معنى من معاني «حَنِبَ»

وــــــــــ في الفَخِّ ونَحْوِهِ: عَلِقَ فيهِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ مُحَمَّدٌ الجَنَدِيُّ (ت: 732ه=1331م):

"فَنَزَلَتِ الهِرَّةُ وأَرادَتْ أَنْ تَخْرُجَ مِنْ طاقَةٍ فيهِ فحَنِبَتْ فيها".

محمد الجندي، السلوك في طبقات العلماء والملوك. تح: محمد بن علي، ص: 471.— المصدر
02 —

حَنَّبَ

فعل
2.1معنى من معاني «حَنَّبَ»

* حَنَّبَ فُلانٌ وغَيْرُهُ: اِعْوَجَّ وانْحَنى.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق.ه=543م) يَصِفُ حَمْلَ غُلامٍ عَلى فَرَسٍ بَعْدَ تَعَبٍ:

فَأَلْقَيْتُ في فيهِ اللِّجامَ وفُتْنَني
وقالَ صِحابي قَدْ شَأَوْنَكَ فَاطْلُبِ

فَلأْيًا بِلَأْيٍ ما حَمَلْنــــــــــــا غُـــــــلامَنا
على ظَهْرِ مَحْبوكِ السَّراةِ مُحَنَّبِ

ديوان امرئ القيس، تح: عليان، ص 391.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ خُفافُ بْنُ نُدْبَةَ (ت: 20ه=641م) يَصِفُ فَرَسَهُ:

رَبِذُ الخِلافِ إِذا اتْلَأَبَّ ورِجْلُهُ
في وَقْعِها ولَحاقِها تَحْنيبُ

ديوان الأصمعيات، تح: محمد نبيل طريفي، ج: 1، ص: 33.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ أبي حَفْصَةَ (ت: 182ه=798م) يَصِفُ الخَيْلَ:

جَلَبَ الجِيادَ مِنَ العِراقِ عَوابِسًا
قُبَّ البُطُونِ يُقَدْنَ بِالأَرْسانِ

جُرْدًا مُحَنَّبَةً تُعاضِـــــــــــــدُ في السُّرى
بِالبيـــــــــــــــــــــــــدِ كُلَّ شِمِلَّةٍ مِذْعانِ

شعر مروان بن أبي حفصة. تح: حسين عطوان ، ج: 1، ص: 107.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مَنْظورٍ(ت: 711ه=1311م):

"الحَنَبُ، والتَّحْنيبُ: احْديدابٌ في وَظيفَيْ يَدَيِ الفَرَسِ، ولَيْسَ ذَلِكَ بِالاعْوِجاجِ الشَّدِيدِ".

ابن منظور، لسان العرب، ج: 3، ص: 351.— المصدر
”وقالَ ابْنُ هُذَيْلٍ الأنْدَلُسِيُّ (ت: 763=1362م) يَذْكُرُ صِفاتِ الفَرَسِ الجَيِّدِ:

"عَريضُ الكَتِفَيْنِ طَويلُهُما غامِضُ أَعاليهِما شَديدُ الصَّدْرِ لَطيفُ الزَّوْرِ شَديدُ تَحْنيبِ السّاقَيْنِ طَويلُ الذِّراعَيْنِ".

ابن هُذيل الأندلسي، حِلْيَةُ الفُرْسانِ وشِعارُ الشُّجْعانِ، تح: مُحمّد عَبْدُ الغني حَسَن، ص: 126 .— المصدر
2.2معنى من معاني «حَنَّبَ»

وــــــــــ لِفُلانٍ وغَيْرِهِ: نَصَبَ له شَرَكًا.

—
غير محدد
”قالَ هارون بْنُ زَكريا الهَجَرِيُّ (ت300ه=912م):

‌"والحنبُ: بِلُغَتِهِمْ: الشَرَكُ يُصْطادُ بِهِ الطَّيْرُ (...)، وحَنِّبْ لَهُ، أَيْ: انْصُبْ لَهُ، عَسى يَقَعُ في الحَنَبِ".

الهجري، التعليقات والنوادر. تح: حمد الجاسر، ج: 3، ص: 1093. — المصدر
2.3معنى من معاني «حَنَّبَ»

وــــــــــ البِناءَ ونَحْوَهُ : أَحْكَمَهُ وقَوَّسَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ثَعْلَب (ت: 291 هـ=904م):

"وقَدْ ‌حَنَّبَ ‌فُلانٌ ‌أَزَجًا مُحْكَمًا، أَيْ: بَناهُ مُحْكَمًا فحَناهُ".

الأزهريّ، تهذيب اللّغة، تح: محمد عوض مرعب، ج: 5، ص: 75.— المصدر
”وقالَ أَبو حيَّانَ التَّوحيديّ (ت: 414 هـ=1023م):

"ولا أَرَى في بُيُوتِكُمْ ودُورِكُمْ بُيوتًا قَدْ غُشِّيَتْ سُقوفُها بِالسَّاجِ...أَوْ حِيرِيٌّ بِكُمَّيْنِ، فيهِ إِيوانٌ مُحَنَّبٌ بَيْنَ حَنَبَتَيْنِ"

أبو حيَّان التوحيدي، الرسالة البغدادية، تح: عبود الشالجي، ص: 134.— المصدر
2.4معنى من معاني «حَنَّبَ»

وــــــــــ الكِبَرُ فُلانًا: نَكَّسَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَمْرو بْنُ قَمِيئَةَ(ت: 85 ق هـ=539م) يَذْكُرُ حالَهُ:

شَكَوْتُ إِلَيْهِ أَنَّني ذو جَلالَةٍ
وأَنِّي كَبيرٌ ذو عِيالٍ مُحَنَّبُ

ديوان عمرو بن قميئة. تح: خالد العطية، ص: 66.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170 هـ= 787م):

"ورجلٌ ‌مُحَنَّبٌ، أَيْ: شيخٌ مُنْحَنٍ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 251.— المصدر
”وقالَ الشَّاعِرُ (170هـ=786م، مِنْ شَواهِدِ الخَليلِ) يَصِفُ مُسِنًّا ابْتُليَ بِالعاهاتِ:

يَظَلُّ نَصْبًا لِرَيبِ الدَّهْرِ يَقْذِفُهُ
قَذْفَ المُحَنَّبِ بِالعاهاتِ والسَّقَمِ

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 251.— المصدر
03 —

تَحَنَّبَ

فعل
3.1معنى من معاني «تَحَنَّبَ»

* تحَنَّبَ فُلانٌ وغَيْرُهُ: تَقَوَّسَ وانْحَنى.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ صَخْرٌ الهُذَليُّ (ت: 12ق.ه=611م) يُشَبِّهُ وَعِلًا بِشَيْخٍ هَرِمٍ:

أُتيحَ لَهُ يَوْمًا وقَدْ طالَ عُمْرُهُ
جَرِيمَةُ شَيْخٍ قَدْ تَحَنَّبَ ساغِبِ

[جَرِيمَةُ القَوْمِ : كاسِبُهُمْ ، والجَرِيمَةُ مِنَ الرِّجَالِ : الكاسِبُ]
ابن قتيبة، المعاني الكبير. تح: سالم الكرنكوي، ج: 2، ص: 729.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ سَاعِدَةُ بْنُ جُؤَيَّةَ (ت: 10ه=631م) يَصِفُ حُسْنَ صَوْتِ امْرَأَةٍ:

وَلَوَ انَّها ضَحِكَتْ فَتُسْمِعَ نَغْمَها
رَعِشَ المَفاصِلِ صُلْبُهُ مُتَحَنِّبُ

السّكّريّ، شرح أشعار الهذليّين. تح: عبد السّتار أحمد فراج، ج: 3، ص: 1338.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الرُّوميّ (ت: 283هـ=896م) يَصِفُ فَرَسًا:

واشْتَدَّ عِلْباءاهُ وانْقَوَسا
دونَ العِذارِ فَضاقَ بُرْقُعُهُ

وتَحَنَّبَتْ ساقاهُ وانْشَنَجَتْ
أَنْساؤُهُ فَقَمَصْنَ أَكْرُعُهُ

ديوان ابن الرّوميّ. تح: حسين نصار، ج: 4، ص: 1540.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ بَدْرُ الدِّينِ العينِيُّ مِنْ شَوَاهِدِهِ (ت:855ه=1451م):

تَرْمي الأَمَاعِيزُ بِمُجْمَرَاتٍ
وَأَرْجُلٍ رَوْحٍ مُحَنَّباتِ

بدر الدّين العيني، المقاصد النّحويّة. تح: عليّ فاخر وآخرين. ص: 184.— المصدر
3.2معنى من معاني «تَحَنَّبَ»

وــــــــــ على غَيْرِهِ: تَحَنَّنَ وتَعَطَّفَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّاحِبُ ابْنُ عَبَّادٍ (ت: 385 هـ=995م):

"وفُلانٌ ‌مُتَحَنِّبٌ ‌عَلى فُلانٍ، بِمَعْنى: مُتَحَنِّنٍ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، المحيط. تح: آل ياسين، ج: 3، ص: 126.— المصدر
04 —

حَنْبَة

اسم
4.1معنى من معاني «حَنْبَة»

* الحَنْبَةُ: القَوْسُ ونَحْوُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو حيَّانَ التَّوحيديّ (ت: 414هـ=1023م):

"ولا أَرَى في بُيُوتِكُمْ ودُورِكُمْ بُيوتًا قَدْ غُشِّيَتْ سُقوفُها بِالسَّاجِ (...)، أَوْ حِيرِيٌّ بِكُمَّيْنِ ، فيهِ إِيوانٌ مُحَنَّبٌ بَيْنَ حَنَبَتَيْنِ".

أبو حيَّان التوحيدي، الرسالة البغدادية، تح: عبود الشالجي، ص: 134.— المصدر
05 —

حُنْبوب

اسم
5.1معنى من معاني «حُنْبوب»

* الحُنْبوبُ مِنَ الأَشْياءِ: الأَسْوَدُ الشَّدِيدُ السَّوادِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّاحِبُ ابْنُ عَبَّادٍ (ت: 385 هـ=995م):

"ويُقالُ: أَسْودُ حُنْبوبٌ، وجَمْعُهُ: حَنابيبُ، مِثْلُ: حُلْكوكٍ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْنُ عَبَّادٍ، المُحيطُ. تح: محمد آل ياسين، ج: 3، ص: 126.— المصدر
