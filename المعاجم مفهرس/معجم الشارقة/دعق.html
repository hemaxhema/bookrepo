01 —

دَعَقَ

فعل
1.1معنى من معاني «دَعَقَ»

*دَعَقَتِ(بِفَتْحِ العَيْنِ) الخَيْلُ وَغَيْرُها في الوَحَلِ وغَيْرِهِ تَدْعَقُ (بِفَتْحِ العَيْنِ) دَعْقًا: وَطِئَتْ فيهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عليُّ بنُ أَبي طالبٍ (ت: 40هـ=661م) يَذْكُرُ فِتْنَةً:

"حَتّى تَدْعَقَ الخَيْلُ في الدِّماءِ".

ابن الأثير، النهاية.، تح: الحلبي، ج: 1، ص: 306.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأثير ِ(ت: 606هـ=1210م) يَشْرَحُ مَعْنى "دَعَقَ" في حَديثِ عَليٍّ:

"حَتّى تَدْعَقَ الخَيْلُ في الدِّماءِ"؛ أَيْ: تَطَأَ فيهِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن الأثير، النهاية.، تح: الحلبي، ج: 1، ص: 306.— المصدر
1.2معنى من معاني «دَعَقَ»

وــــــــــــ فلانٌ الخَيْلَ وَنَحْوَها: دَفَعَها في الغارَةِ عَلَى العَدُوِّ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ أَبُو الحَوَّاسِ الخُزَيميُّ (نَحْوَ: 43ق هـ=580م) يَذْكُرُ صَبْرَ قَبيلَةِ سُلَيْمٍ ضِدَّ قَبيلَةِ عامِرٍ:

صَبَرَتْ سُلَيْمٌ يَوْمَ دَعْقَةِ عامِرٍ
صَبْرَ الكِرامِ ويالَهُ مِنْ مَدْعَقِ

الهجري، التعليقات والنوادر. تح: الجاسر، ج: 2، ص: 613.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"دَعَقَ الخَيْلَ يَدْعَقُها دَعْقًا، إِذا دَفَعَها في الغارَةِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 1، ص: 206.— المصدر
”وقالَ رَجُلٌ (نَحْوَ: 321هـ=933م، مِنْ شَواهِدِ ابنِ دُرَيْدٍ) يصفُ يَوْمَ شَرٍّ:

"إِنَّهُ يَوْمُ دَعْقٍ وشَوْبٍ".

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 1، ص: 351.— المصدر
1.3معنى من معاني «دَعَقَ»

وــــــــــــ الدَّوابُّ ونَحْوُها الأَرْضَ وغَيْرَها، وفيها: داسَتْها دَوْسًا شَديدًا حَتّى أَثَّرَتْ فيها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

" حَتَّى تَدْعَقَ الخُيولُ في نَواحِر أَرْضِهِمْ".

ابن الأثير، النهاية.، تح: الزواوي وآخر، ج: 5، ص: 27.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ= 787م):

"دَعَقَتِ الدَّوابُّ في الأَرْضِ لِشِدَّةِ الوَطْءِ حَتَّى تَصيرَ فيها آثارٌ مِنْ دَعْقِها".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين، تح: المخزومي وآخر، ج:، ص:.— المصدر
1.4معنى من معاني «دَعَقَ»

وــــــــــــ فُلانٌ وَغَيْرُهُ الإِبِلَ وَنَحْوَها: طَرَدَها طَرْدًا شَديدًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ لبيدٌ (ت: 41هـ=662م) يَمْدَحُ قَوْمَهُ ويُشيدُ بثَباتِهِمْ عِنْدَ الفَزَعِ:

في جَميعٍ حافِظي عَوْراتِهِمْ
لا يَهُمّونَ بِأدْعاقِ الشَّلَلْ

ابن قتيبة، المعاني الكبير، تح: الكرنكوي وآخر، ج: 2، ص: 885.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395ه=1005م):

"شَلَّ إبِلَهُ شَلاًّ دَعْقًا، إذا طرَدَها".

ابن فارس، مقاييس اللغة، تح: هارون، ج: 2، ص: 281.— المصدر
1.5معنى من معاني «دَعَقَ»

وــــــــــــ الإِبِلُ الحَوْضَ: وَرَدَتْهُ فازْدَحَمَتْ عَلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مُلَيْحٌ الهُذَليُّ (ت: 50هـ=670م) يَفْخَرُ بِكَثْرَةِ خَيْلِهِ:

لَوْ وَرَدوا ذا اللُّجَجِ العَوامِقِ

لَشَرِبوهُنَّ بِوِرْدٍ داعِقٍ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 1055.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو نُخَيْلَةَ السَّعْديُّ (ت: 145هـ=762م) يَنْصَحُ الـمَنْصورَ بِعَقْدِ البَيْعَةِ لابْنِهِ الـمَهْديِّ:

فَلَوْ سَمِعْنا قَوْلَكَ امْدُدِ امْدُدِ

كانَتْ لَنا كَدَعْقَةِ الوِرْدِ الصَّدي

شعر أبي نخيلة الحماني. تح: الخطيب، ص: 110.— المصدر
1.6معنى من معاني «دَعَقَ»

وــــــــــــ النّاسُ وغَيرُهُم الطَّريقَ: وَطِئوهُ وأَكْثَروا الـمَشْيَ فيهِ. فالمـَفْعولُ مَدْعوقٌ، ودَعْقٌ، ودَعِقٌ. (انْظُر: د ع ك)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الزَّفَيانُ (ت: 75هـ=694م) يَصِفُ نوقًا:

يَرْكَبْنَ نيرَيْ لاحِبٍ مَدْعوقِ

شرح ديوان الزفيان. تح: الأطرم، ص: 119.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَذْكُرُ حُمُرًا وَحْشيَّةً:

في رَسْمِ آثارٍ ومِدْعاسٍ دَعِقْ

الزّبيدي، تاج العروس. تح: حجازي، ج: 25، ص: 288.— المصدر
”وقالَ أَبو عَمْرٍو الشَّيْبانيُّ (ت: 206هـ=821م):

"دَعَقَ النّاسُ الطَّريقَ، إِذا أَكْثَروا الاخْتِلافَ فيهِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عمرو، كتاب الجيم. تح: الإبياري، ج: 1، ص: 260.— المصدر

يَعْكُفْنَ ضَحْوًا في مَكانِسِهِ

”قالَ اليُونِينِيُّ (ت: 726 هـ=1326م):

"وَكانَ يَموتُ مِنْ عَسْكَرِهِ في كُلِّ مَنْزِلَةٍ جَماعَةٌ كَثيرَةٌ وَتُدْعَقُ خُيولٌ كَثيرَةٌ، فَعَدِمَ مِنَ الرِّجالِ والخُيولِ ما لا يُحْصَى كَثْرَةً"

احْرِمْهُ كُلَّ رَزَمانيٍّ مُلِثّْ

ودَعَقاتِ الدُّرَآنِ الـمُنْدَلِثْ

1.9معنى من معاني «دَعَقَ»

وــــــــــــ فُلانٌ وَغَيْرُهُ الشَّيْءَ، وعَلَيْهِ: دَقَّهُ، أو دَقَّ عَلَيْهِ. (انْظُر: د ق ق)

أَدْعَقَ

4.1معنى من معاني «دَعْق»

*الدَّعْقُ:الجَماعَةُ مِنَ النّاسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والدَّعْقُ: الجَماعَةُ مِنَ النّاسِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 1، ص: 152.— المصدر

دَعْقَة

*الدَّعْقَةُ: الحَمْلَةُ في القِتالِ ونَحْوِه.

ق

”

صَبْرَ الكِرامِ ويالَهُ مِنْ مَدْعَقِ

الهجري، التعليقات والنوادر. تح: الجاسر، ج: 2، ص: 613.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"والدَّعْقَةُ:... الحَمْلَةُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 1، ص: 152.— المصدر

وــــــــــــ: الجَماعَةُ مِنَ الخَيْلِ أَوِ الإِبِلِ. (انْظُر: د ع ك)

س

لَجوجٌ غَداةَ الفَوْتِ حَتّى كَأَنَّهُ
حِصانٌ يُلاقي دَعْقَةَ الخَيْلِ أَبْلَقُ

الجمحي، طبقات فحول الشعراء. تح: محمود شاكر، ج: 2، ص: 623.— المصدر

العباسي(133ه‍/750م) إلى (656ه‍/1258م)

الجوهري، الصحاح. تح: عطار، ص: 1474.— المصدر

07 —

مَدْعَق

اسم
7.1معنى من معاني «مَدْعَق»

*الـمَدْعَقُ: مَوْضِعُ دَعْقِ الدَّوابِّ بِالأَرْضِ. (ج) مَداعِقُ، ومَداعيقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مَعْقِلٌ الهُذَليُّ (ت: 30هـ=650م) يَصِفُ سَيْفَهُ:

كَأَنَّ عَروضَيْهِ مَحَجَّةُ أَبْقُرٍ
لَهُنَّ إِذا ما رُحْنَ فيها مَداعِقُ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 3، ص: 1319.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللَّيْثُ (ت: 180هـ=796م):

"الـمَدْعَقُ: مَوْضِعُ دَعْقِ الدَّوابِّ التُّرابَ بِالأَرْضِ".(وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الزّبيدي، تاج العروس. تح: حجازي، ج: 25، ص: 290.— المصدر
7.2معنى من معاني «مَدْعَق»

وـــــــــــ: مَفْجَرُ الماءِ. (ج) مَداعِقُ، ومَداعيقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رُؤْبَةُ (نَحْوَ: 132ه=750م)، مِنْ قَصيدَةٍ يَمْدَحُ فيها مَرْوانَ بْنَ مُحَمَّدٍ)يَذْكُرُ سَيْلًا:

يَضْرِبُ عِبْرَيْهِ ويَغْشى الـمَدْعَقا

ديوان رؤبة. تح: وليم البروسيي، ص: 115.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)

وَشَـتّانَ مـا بَيـن الخَميسَـيْنِ فـي الـوَغَى
إذا الحَــربُ جَــدَّتْ فــي مَضـيقٍ وَمَـدْعَقِ

ديوان عبد المطلب، تح: الإبياري وآخر، ص: 169.— المصدر

08 —

مَداعيق

اسم
8.1معنى من معاني «مَداعيق»

*الـمَداعيقُ:الخَيْلُ الـمُتَقَدِّمَةُ التي تَدوسُ القَوْمَ في الغاراتِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجَوْهَريُّ (ت: 393هـ=1003م):

"وخَيْلٌ مَداعيقُ: تَدوسُ القَوْمَ في الغاراتِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في عَصْرِ الدّولِ والإِماراتِ)

الجوهري، الصحاح. تح: أحمد عطار، ص: 1474.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَبو الفَضْلِ الوليدُ(ت: 1366هـ=1947م)يذكرُ الخيلَ المعَدَّةَ للحربِ:

مَناصِلُ حَبْسٍ أو مَقانِبُ غَزْوَةٍ
مَداعيقُ تَعْدو تحتَ فُرْسانِ جَمْرَةِ

ديوان أبي الفضل الوليد، ص: 367.— المصدر
