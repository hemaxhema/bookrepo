النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (س م ق):

في الفينيقيّة: وَرَدَ الاسْمُ: qmt (صمق) ṣmq بِمَعْنى (نَوعٌ مِنَ الشَّجرِ والنَّباتِ) يُوافِقُ الاسمَ العَرَبيَّ (سمّاقٌ) مَبنًى ومَعنًى، غَيرَ أنَّ الاسمَ الفينيقيَّ خَضَعَ لإبدالِ السّينِ صادًا، ومِنْ مَجيئِهِ بِمَعْنى (سمّاقٌ) ما وَرَدَ في نَقشِ ليبيا:" rsa. Rkk. Qmt"، أي: "عَشرةُ مَكاييل مِنَ السُّمّاقِ".

(DNWSI: p 970)

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܣܽܘ̣ܡܳܩܳܐ (سُوماقا) sūmāqā بِمَعْنى (شَجرةٌ تُستَعمَلُ أوراقُها دِباغًا) يُوافِقُ الاسمَ العَرَبيَّ (سُمّاقٌ) مَبنًى ومَعنًى.

(ܠܟܣܝܩܘܢ ܕܐܘܓܝܢ ܡܢܢܐ: ܣܡܩ)

01 —

سَمَقَ

فعل
1.1معنى من معاني «سَمَقَ»

* سَمَقَ (بِفَتْحِ الميمِ) الشَّجَرُ وغَيْرُهُ، يَسْمُقُ (بِضَمِّ الميمِ) سَمْقًا، وسُموقًا: طالَ وارْتَفَعَ، فهُوَ سامِقٌ، (ج) سُمَّقٌ، وسَوامِقُ، وهُوَ أيضًا سَميقٌ، (ج) سِماقٌ، وأَسْمِقَة. وهِيَ بِتاءٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال أُحَيْحَةُ بنُ الجُلاحِ (ت: 129ق.هـ=496م) يَصِفُ نَخْلًا:

إذا جُمادى مَنَعَتْ دَرَّها
زانَ جَنابي عَطَنٌ مُعْصِفُ

في مُشْرِفٍ يَسْمُقُ جَبّارُهُ
بِحافَتَيْهِ الشّوعُ والغِرْيَفُ

أبو العلاء المعرّي، اللامع العزيزي. تح: المولويّ، ص: 745.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال عَدِيُّ بنُ وَداعٍ (ت: 11هـ=632م) يَصِفُ فَرَسَهُ:

وقَدْ أَغْدو بمُنْشَقٍّ نَساهُ
جَوادٍ في المَحَثَّة والنِّزاقِ

لِغَيْثٍ يَجْنُبُ الرُّوّادُ عَنْهُ
يُباري الرّيحَ بالعُشْبِ السِّماقِ

محمّد بن المبارك، منتهى الطّلب. تح: طريفيّ، ج: 8، ص: 315.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ خَيْلًا:

ظَلَّتْ بَأرْضٍ سَامِقٍ أَعْشابُها

مِنَ الرَّبيعِ صَخِبٍ ذُبابُها

ديوان رؤبة. تح: البروسي، ص: 19.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ فَضْلِ اللهِ العُمَرِيُّ (ت: 749هـ=1348م) يُتَرْجِمُ لأَبي عَبْدِ اللهِ مُحمَّدِ بْنِ البَيْنِ:

"سَمَقَ في الأدَبِ شَجَرُهُ".

العمريّ، مسالك الأبصار. تح: الجبوريّ، ج: 17، ص: 324.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال مُحَمَّدُ بْنُ الطُّلْبَة الشِّنْقيطِيُّ (ت: 1272هـ=1856م) يَذْكُرُ غارَةً للمُسْلِمينَ عَلى هوازِنَ:

كَمْ غادَروا ثَمَّ مِنْ بَطْلٍ تُلاعِبُهُ
عُرْجُ الضِّباعِ وعانٍ حالَفَ الرِّبَقا

وغادَرَ العونَ والأَبْكارَ في جَرَدٍ
مِنْ قَيْسِ عَيْلانَ في الفَرْعِ الّذي سَمَقا

ديوان محمّد بن الطّلبة. ص: 299.— المصدر
1.2معنى من معاني «سَمَقَ»

وــــــــــ  فُلانٌ في الأَمْرِ: عَلا شَأْنُهُ فيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابنُ قُتَيْبَةَ (ت: 276هـ=889م) يَذْكُرُ بَعْضَ ما يُسْقِطُ عَدالَةَ الرّاوي:

"وقَدْ يَكونُ الرّجُلُ مَتْبوعًا في الأدَبِ قَدْ سَمَقَ فيهِ وأخَذ بالحَظِّ الأوفَرِ مِنْهُ، إِلّا أَنَّهُ أَغْفَلَ شَيْئًا مِنَ الحَليلِ؛ كانَ أَوْلى بِهِ مِنْ بَعْضِ ما حَفِظَ فَلَحِقَتْهُ فيهِ النّقيصَةُ، وتَرْجِعُ عَلَيْهِ مِنْهُ الهُجْنَةُ".

ابن قتيبة، المعارف. تح: ثروت عكاشة، ص: 157.— المصدر
02 —

سامَقَ

فعل
2.1معنى من معاني «سامَقَ»

* سامَقَ فُلانٌ لِفُلانٍ: طاوَلَهُ واسْتَكْبَرَ عَلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عَساكر (ت: 571هـ=1175م) يَذْكُرُ خِيانَةَ أَميرِ جُيوشِ مِصْرَ للأَميرِ قُطْبِ الدّينِ مَودودِ بْنِ زِنْكي في قِتالِ الرّومِ:

"واسْتَجاشَ بِجَيْشِ العَدُوِّ طَلَبًا لِلْبَقاءِ في السُّمُوِّ، ثُمَّ وَجَّهَ إِلَيْهِ بَعْدَ ذَلِكَ جَيْشًا آخَرَ، فَأَصَرَّ عَلى المُسامَقَةِ لَهُ وكابَرَ، واسْتَنْجَدَ بِالعَدُوِّ-خَذَلَهُ اللهُ- فَأَنْجَدوهُ".

ابن عساكر، تاريخ دمشق. تح: العمروي، ج: 57، ص: 122.— المصدر
2.2معنى من معاني «سامَقَ»

وــــــــــ  فُلانٌ وغَيْرُهُ الشَّيْءَ: بَلَغَهُ طولًا أَوْ فاقَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو طالِبٍ مُحَمَّدُ بنُ عَبْدِ اللهِ الدِّمِشْقِيُّ (460هـ= 1067م) يُجيبُ عَلِيَّ بنَ الحَسَنِ الباخَرْزِيَّ، ويَمْدَحُهُ:

رَعاكَ اللهُ مِنْ يَقِظٍ أَريبٍ
يُسامِقُ كُلَّ مَنْقَبَةٍ مُطيفِ

الباخَرْزيّ، دمية القصر. تح: ألتونجيّ، ج: 1، ص: 212.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال السّيّد عمر الخطيب (1368هـ=1948م) تاريخُ الطَّبْعِ، يَتَحَدَّثُ عنْ مَكانَةِ الزّعيمِ السّوري فارِس الخوريّ:

"وفي نَظَري: أَنَّ فارِسَ الخوري لا يَطْمَعُ إلى أَبْعَدَ مِنْ هَذا، وهُوَ قانِعٌ بِما أَصابَ مِنْ صيتٍ ذائِعٍ واسْمٍ لامِعٍ وشُهْرَةٍ تُسامِقُ الفَرْقَدَيْنِ".

مجلّة الرّسالة، مج: 16، العدد: 806، ص: 1417.— المصدر
03 —

تَسامَقَ

فعل
3.1معنى من معاني «تَسامَقَ»

* تَسامَقَ الشَّيْءُ في المـَكانِ ونَحْوِهِ: عَلا وارْتَفَعَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى (ت: 7هـ=628م) يَصِفُ ثَغْرَ مَحْبوبَتِهِ:

كَذُرى مُنَوِّرِ أُقْحُوا
نٍ قَدْ تَسامَقَ في قَرارِهْ

ديوان الأعشى. تح: محمّد حسين، ص: 153.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ زاهرٌ الألمعيُّ(1389هـ=1969م) تاريخُ الإِلْقاءِ، يَفْتَخِرُ بشُهداءِ فِلَسْطينَ:

فِدائيي فِلِسْطينَ امتطَيْتُمْ
خُطوبًا في مَلاحِمِها صِلابا

وفَتْحٌ قَدْ تَسامَقَ في ذُراها
ليوثٌ تَجْعَلُ المَيْدانَ غابا

زاهر الألمعيّ، الألمعيّات، ص: 42.— المصدر
04 —

سُماق

اسم
4.1معنى من معاني «سُماق»

* السُّماقُ مِنْ كُلِّ شَيْءٍ: الخالِصُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ القُلاخ بن حَزْنٍ (ت: 86هـ=705م) يَصِفُ رَجُلًا سَريعَ الغَضَبِ:

يَغْضَبُ إِنْ قالَ الغُرابُ غاقِ

أَبْعَدَكُنَّ اللهُ مِنْ نِياقِ

إِنْ لَمْ تُنَجّيْنَ مِنَ الوَثاقِ

بِأَرْبَعٍ مِنْ كَذِبٍ سُماقِ

أبو زيد الأنصاريّ، النّوادر. تح: محمّد عبد القادر، ص: 348.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو زَيْدٍ الأَنْصارِيُّ (ت: 215هـ=830م):

"والسُّماقُ: الخالِصُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو زيد الأنصاريّ، النّوادر. تح: محمّد عبد القادر، ص: 348.— المصدر

سِمِقّ

اسم

* السِّمِقُّ مِنَ الرِّجالِ: الطَّويلُ. (انْظُرْ: ش م ق)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

06 —

سُمّاق

اسم
6.1معنى من معاني «سُمّاق»

* السُّمّاقُ: شَجَرٌ مِنَ الفَصيلَةِ البُطْمِيَّةِ، تُسْتَعْمَلُ أَوْراقُهُ دِباغًا وبُذورُهُ تابِلًا، وثَمَرُهُ أَحْمَرُ كالعَناقيدِ، يَنْبُتُ في المُرْتَفِعاتِ والجِبالِ. واحِدَتُهُ بِتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نُواسٍ (ت: 198هـ=814م) يَهجو صديقًا لَهُ تَرَكَ صُحْبَتَهُ، وصارَ طَبّاخًا لَدى أَحَدِ أُمَراءِ فارِسٍ:

حاسِرُ كَفَّيْكَ عَلى هاوُنٍ
لِدَقِّ ثومٍ أَوْ لِسُمّاقِ

ديوان أبي نُواس. تح: الحديثيّ، ص: 443.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابنُ العَديمِ (ت: 660هـ=1262م) يَصِفُ أَشْجَار جَبَلِ السُّمّاقِ بِحَلَبَ:

"وكَذَلِك أَشْجارُهُ، فَإِنَّها قَدْ عَمَّتِ الجِبالَ والبِقاعَ والأَوْدِيَةَ والتِّلاعَ، مِنَ التّينِ والعِنَبِ... والسُّمّاقِ، وإنَّما عُرِفَ بِجَبَلِ السُّمّاقِ لِكَثْرَتِهِ فيهِ، وسُمّاقُهُ أَجْوَدُ مِنْ غَيْرِهِ".

ابن العديم، بغية الطّلب. تح: زكّار، ج: 1، ص: 423.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ رشيد رضا (1352هـ=1933م) تاريخُ الطَّبْعِ، يَتَحَدّثُ عَنْ نفْسِهِ:

"فَكُنْتُ أُجاهِدُ نَفْسي عَلى طَريقَةِ الصّوفِيَّةِ بِتَرْكِ أَطْيَبِ الطَّعامِ؛ اكْتِفاءً بِقَليلٍ مِنَ الزَّعْتَرِ مَعَ المِلْحِ ‌والسُّمّاقِ".

مجلّة المنار، العدد: 5، ص: 354.— المصدر
07 —

سُمّاقِيّ

اسم
7.1معنى من معاني «سُمّاقِيّ»

* السُّمّاقيُّ: دواءٌ لِعِلاجِ الحُمّى وغَيْرِها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ رَبَن الطَّبَرِيُّ (ت: 247هـ=861م) يَصِفُ بَعْضَ أَدْوِيَةِ الحُمّى:

"السُّمّاقِيُّ نافِعٌ مِنَ الحُمّى".

ابن رَبَنٍ الطّبَرِيّ، فردوس الحكمة. تح: الصّديقيّ، ص: 473.— المصدر
7.2معنى من معاني «سُمّاقِيّ»

وــــــــــ: ضَرْبٌ مِنَ العِنَبِ يَنْبُتُ شَجَرُهُ بالأَهْوازِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ الفَقيهِ الهَمْدانِيُّ (ت: 365هـ=976م) يَذْكُرُ كَثْرَةَ أَصْنَافِ الكَرْمِ:

"كَثْرَةُ فُنونُهِ واخْتِلافُ أَنْواعِهِ لا يُدْرَكُ كالسُّونايا، والخُمْرِيِّ بِطُسّوجِ قُطْرُبُّلَ... والسُّمّاقِيِّ بالأَهْوازِ".

ابن الفقيه الهمدانيّ، البلدان. تح: يوسف الهادي، ص: 173.— المصدر
7.3معنى من معاني «سُمّاقِيّ»

وــــــــــ مِنَ الرُّخامِ ونَحْوِهِ: ما يَكونُ بِلَوْنِ السُّمَّاقِ الأَحْمَرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو مَنْصورٍ الثَّعالِبيُّ (ت: 430هـ=1039م) يَذْكُرُ آنِيَةَ الرّاحِ:

"إِنَّهُ كانَ في الزَّمَنِ القَديمِ امْرَأَةٌ في العِراقِ، تَعْمَدُ إِلى الصّينيِّ الأَبْيَضِ فتُصَيِّرُ مِنْهُ الشَّمْشِيَّ ‌والأَسْوَدَ والسُّمّاقِيَّ والأَخْضَرَ حتّى لا يَشُكَّ ناظِرٌ في أَنَّهُ كانَ كذَلِك في الأَصْلِ".

الغزوليّ، مطالع البدور. ص: 131.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الغزوليُّ (ت: 815هـ=1413م) يَذْكُرُ أَرْدَأَ أَلْوانِ الياقوتِ:

"أَرْدَأُ الأَلْوانِ الأَحْمَرُ الوَرْدِيُّ الذي يَضْرِبُ إِلى البَياضِ، ‌والسُّمّاقيُّ الّذي يَضْرِبُ إِلى السَّوادِ".

الغزوليّ، مطالع البدور. تح: التّجانيّ سعيد، ج: 2، ص: 231.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الجبرتيّ (ت: 1237هـ=1822م) يَذْكُرُ ما فَعَلَهُ بَعْضُ الإِفْرَنْجِ في مِصْرَ ورَغْبَتَهُمْ في البَحْثِ عَنِ الآثارِ:

"وأحْضَروا قِطَعَ أَحْجارٍ عَلَيْها نُقوشٌ وأَقلامٌ وتَصاويرُ ونَواويسُ مِنْ رُخامٍ أَبْيَضَ، وتَماثيلَ آدَمِيَّةً مِنَ ‌الحَجَرِ ‌السُّمَّاقِيِّ الأَسْوَدِ المُنَقَّطِ الذي لا يَعْمَلُ فيهِ الحَديدُ".

الجبرتيّ، عجائب الآثار. تح: عبد الرّحيم عبد الرّحمن ج: 3، ص: 571.— المصدر
08 —

سُمّاقِيّة

اسم
8.1معنى من معاني «سُمّاقِيّة»

* السُّمَّاقِّيَّةُ: طَعامٌ يُقَطَّعُ عَلَيْهِ اللَّحْمُ السَّمينُ، ويُعَرَّقُ بِزَيْتِ السِّمْسِمِ والبَصَلِ ونَحْوِهما، ويُضافُ إِلَيْهِ بَعْدَ نُضْجِهِ السُّمّاقُ المَنْقوعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو بكرٍ الرّازيُّ (ت: 311هـ=924م) يَصِفُ طَريقَةَ عِلاجِ الإِسْهالِ:

"وإنْ كانَ الإسْهالُ قَوِيًّا جُعِلَتْ أَغْذِيَتُهُ مُقَوِّيَةً لِلْمَعِدَةِ كالزّيرِباجِ، والنّارباجِ، والسُّمّاقِيَّةِ ".

الرّازيّ، الحاوي. تح: محمّد إسماعيل، ص: 959.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الغزوليُّ (ت: 815هـ=1413م) يَذْكُرُ عُيوبَ الأَطْعِمَةِ المَشْهورَةِ:

"السُّمَّاقِيَّةُ: بارِدَةٌ يابِسَةٌ أَيْضًا".

الغزوليُّ، مطالع البدور. تح: التّجانيّ سعيد، ج: 2، ص: 87.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مجمع القاهرة (1424ه=2003م) تاريخُ الطَّبْعِ:

"وتُعَدُّ السُّمَّاقِيَّةُ بِتَقْطيعِ اللَّحْمِ السَّمينِ وتَقْطيعِهِ مَخْلوطًا بالشَّيْرَجِ، والبَصَلِ".

مجمع القاهرة، معجم الأطعمة والأشربة، ص: 36.— المصدر
09 —

سَموق

اسم
9.1معنى من معاني «سَموق»

* السَّمُوقُ (وقَدْ تُشَدَّدُ ميمُهُ): شَجَرٌ مِنَ الفَصيلَةِ البُطْمِيَّةِ، تُسْتَعْمَلُ أَوْراقُهُ دِباغًا وبُذورُهُ تابِلًا، وثَمَرُهُ أَحْمَرُ كالعَناقيدِ، يَنْبُتُ في المُرْتَفَعاتِ والجِبالِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ عبّادٍ (ت: 385هـ=995م):

"والسَّموقُ والسُّمّاقُ: مِنَ التَّوابِلِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عبّاد، المحيط. تح: آل ياسين، ج: 5، ص: 302.— المصدر
10 —

سَميق

اسم
10.1معنى من معاني «سَميق»

* السَّميقُ: خَشَبَةٌ تُحيطُ بعُنُقِ الثَّوْرِ مِنَ النّيرِ كالطَّوْقِ، وهُما سَميقانِ. (ج) أَسْمِقَةٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ (ت: 170هـ=787م):

"والسَّميقانِ في النّيرِ: عُودانِ قَدْ لوقِيَ بَيْنَ طَرَفَيْهِما تَحْتَ غَبْغَبِ الثَّوْرِ شُدّا بِخَيْطٍ، وتُجْمَعُ: أَسْمِقَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزوميّ وآخر، ج: 5، ص: 88.— المصدر
10.2معنى من معاني «سَميق»

وــــــــــ: إِحْدى خَشَباتٍ تَدْخُلُ في الآلَةِ الّتي يُنْقَلُ عَلَيْها الطّوبُ اللَّبِنُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الخَليلُ (ت: 170هـ=787م):

"والسَّميقانِ: ‏خَشَباتٌ يُدْخَلْنَ في الآلَةِ التي يُنْقَلُ عَلَيْها اللَّبِنُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزوميّ وآخر، ج: 5، ص: 88.— المصدر
