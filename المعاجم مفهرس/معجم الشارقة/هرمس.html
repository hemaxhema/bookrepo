هرمس
هَرْمَسَ(فعل) هُرامِس(اسم) هِرْماس(اسم) هِرْمَوْس(اسم) هِرْميس(اسم) هُرَيْمِسَة(اسم)

المعانى الكلّيّة

  1. الجُرْأَةُ والشِّدَّةُ
  2. التَّكَلُّمُ والصَّخَبُ
01 —

هَرْمَسَ

فعل
1.1معنى من معاني «هَرْمَسَ»

* هَرْمَسَ النّاسُ: تَكَلَّموا وضَجّوا وصَخَبوا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الفَرّاءُ النَّحْوِيُّ (ت: 207ه=822م):

"وهَرْمَسَةُ النّاسِ: كَلامُهُمْ وضَجيجُهُمْ وصَخَبُهُمْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: أبو الفضل إبراهيم، ج: 3، ص: 447.— المصدر
1.2معنى من معاني «هَرْمَسَ»

وــــــــــ فُلانٌ: عَبَسَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"الهَرْمَسَةُ: العُبوسُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 127.— المصدر
02 —

هُرامِس

اسم
2.1معنى من معاني «هُرامِس»

* الهُرامِسُ: الأَسَدُ الجَريءُ الشَّديدُ العادي عَلى النّاسِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الكِسائيُّ (ت: 189ه=805م):

"أَسَدٌ هِرْماسٌ، وهُرامِسٌ: وهُوَ الجَريءُ الشَّديدُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 277.— المصدر
03 —

هِرْماس

اسم
3.1معنى من معاني «هِرْماس»

* الهِرْماسُ: الأَسَدُ الجَريءُ الشَّديدُ العادي عَلى النّاسِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”

قالَ مالِكٌ الخُناعِيُّ (ت: 86ق.ه=538م) يَصِفُ أَسَدًا:

صَعْبُ البَديهَةِ مَسْنونٌ أَظافِرُهُ
مُواثِبٌ أَهْرَتُ الشِّدْقَيْنِ هِرْماسُ

نشوان الحميري، شمس العلوم. تح: حسين العمري وآخرين، ج: 10 ص: 6918.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ رُؤْبَةُ (ت: 145ه=762م) يَصِفُ أَسَدًا:

ووَقْعُ نابَيْهِ مِجَدٌّ فَأْآسْ

يَغْدو بِأَشْبَالٍ أَبوها الهِرْماسْ

ديوان رؤبة بن العجاج. تح: البروسي، ص: 67.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ ابْنُ الصَّيْقَلِ الجَزْرِيُّ (ت: 701ه=1301م) في إِحْدى مَقاماتِهِ:

"ومَضَيْتُ إِلى المِهْراسِ مُضِيَّ الهِرْماسِ".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ص: 214.— المصدر
3.2معنى من معاني «هِرْماس»

وــــــــــ: وَلَدُ النِّمِرِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"والهِرْماسُ: وَلَدُ النَّمِرِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: مرعب، ج: 6، ص: 277.— المصدر
04 —

هِرْمَوْس

اسم
4.1معنى من معاني «هِرْمَوْس»

* الهِرْمَوْسُ: الرَّجُلُ الصُّلْبُ الرَّأْيِ الدّاهِيَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206ه=821م):

"والهِرْمَوْسُ: الصُّلْبُ الرَّأْيِ المُجَرِّبُ الدّاهِيَةُ، الكَبيرُ مِنَ الرِّجالِ، وهُوَ المُنَجَّذُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 318.— المصدر

ابن عباد، المحيط. تح: آل ياسين، ج: 4، ص: 127.— المصدر
