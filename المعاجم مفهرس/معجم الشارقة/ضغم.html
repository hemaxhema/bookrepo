01 —

ضَغَمَ

فعل
1.1معنى من معاني «ضَغَمَ»

* ضَغَمَ (بفَتْحِ الغَيْنِ) الحَيَوانُ يَضْغَمُ (بفَتْحِ الغَيْنِ) الشَّيْءَ، وعَنْهُ، وبهِ ضَغْمًا، وضَغَمًا: عَضَّهُ بمِلْءِ فيهِ عَضًّا شَديدًا دونَ النَّهْشِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مُغلِّسُ بنُ لَقيطٍ (ت: 22ق هـ=600م) يَصِفُ عَضَّةً شَديدَةً:

وقَدْ جَعَلَتْ نَفْسي تَطيبُ لِضَغْمَةٍ
لِضَغْمِهِماها يَقْرَعُ العَظْمَ نابُها

سيبويه، الكتاب. تح: هارون، ج: 2، ص: 365.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رجُلٌ مِنَ المُسْلِمينَ (23هـ=643م) يُخْبِرُ عُمَرَ بْنَ الخَطّابِ حينَ سَأَلَ ما أَغْرى أَبا زُبَيْدٍ بِصِفَةِ الأَسَدِ؟:

"إِنَّهُ واللهِ يا أَميرَ المُؤْمنينَ ضَغَمَهُ ضَغْمَةً عَلى شاطِئِ الفُراتِ فَجَرَّأَهُ".

ديوان كعب بن زهير، تح: الحتّي، ص: 39.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ= 787م):

"الضَّغْمُ: ‏ عَضٌّ مِن غَيْرِ نَهْشٍ".

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 370.— المصدر
”وقالَ ابنُ الرّومِيِّ (ت: 283هـ=896م) يَصِفُ أسَدًا:

أَتاحَتْ لهُ الأَحْداثُ مِنْهُنَّ قِرْنَهُ
كِفاحًا فلَمْ يَكْدَحْ بِظُفْرٍ ولا ضَغَمْ

ديوان ابن الرّوميّ، تح: بسج، ج: 3، ص: 292.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ فَضْلِ الله العُمَريّ (ت: 749هـ=1349م) يَصِفُ الرّاعِيَ أَحَدَ رِجالاتِ العَرَبِ:

"وما زالَ يُقالُ: إِنَّهُ ‏ فَحْلُ مُضَرَ، حَتىّ ضَغَمَهُ جَريرُ بْنُ الخَطَفى عِنْدَما ظَهَرَ، وتُوَفِّيَ مُسِنًّا".

ابن فضل الله العمري، مسالك الأبصار، الجبوري وآخر، ج: 14، ص: 320.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد يوسُف الكانْدَهْلَويّ (ت: 1384هـ=1965م) يَتَحَدَّثُ عَنْ عُتْبَةَ بنِ عبدِ العُزّى:

"فلَقَدْ عَدا عَلَيْهِ الأَسَدُ من بَيْنِ القَوْمِ، فضَغَمَهُ ضَغْمَةً فقَتَلَهُ".

الكاندهلويّ، حياة الصّحابة، تح: بشّار، ج: 1، ص: 329.— المصدر
1.2معنى من معاني «ضَغَمَ»

وــــــــــفُلانٌ الشَّيْءَ، وبه: مَلَأَ فَمَهُ مِمّا يُؤْكَلُ أَوْ يُعَضُّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأنْصاريُّ (ت: 215هـ=830م):

" ضَغَمْتُ به ضَغْمًا، وهُوَ أَنْ تَمْلَأَ فَمَكَ مِمّا أَهْوَيْتَ قَصْدَهُ مِمّا يُؤْكَلُ أَوْ يُعَضُّ".

السرقسطي، الأفعال، تح: شرف، ج: 2، ص: 210.— المصدر
1.3معنى من معاني «ضَغَمَ»

وــــــــــالفَقْرُ فُلانًا: أصابَهُ ونالَ مِنْهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتِ امْرَأَةٌ (23هـ=643م) تَسْأَلُ عُمَرَ بْنَ الخَطّابِ:

"فَهَلْ مِن ناصِرٍ يَجْبُر، أو داعٍ يُشْكَر، أَعاذَكُمُ اللهُ مِنْ جَوْحِ الدَّهْرِ، ‌وضَغْمِ ‌الفَقْر".

الخطابي، غريب الحديث، تح: الغرباوي، ج: 2، ص: 77.— المصدر
02 —

أَضْغَمَ

فعل
2.1معنى من معاني «أَضْغَمَ»

* أَضْغَمَ الفَمُ: كَثُرَ لُعابُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ القُوطِيَّةِ (ت: 367هـ=977م):

"وأَضْغَمَ الفَمُ: كَثُرَ لُعابُهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال، تح: فودة، ص: 88.— المصدر
2.2معنى من معاني «أَضْغَمَ»

وــــــــــفلانٌ فُلانًا: عَضَّهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عبدُ المَلِكِ بْنُ مَرْوانَ (ت: 86هـ=705م) يُوَبِّخُ الحَجّاجَ حينَما أَغْلَظَ القَوْلَ لأَنَسِ بْنِ مالِكٍ:

"لَقَدْ هَمَمْتُ أنْ أُضْغِمَكَ ضَغْمَةً كبَعْضِ ضَغَماتِ اللُّيوثِ للثَّعالِبِ".

الزبير بن بكار، الأخبار الموفقيات، تح: العاني، ص: 274.— المصدر
03 —

ضاغَمَ

فعل
3.1معنى من معاني «ضاغَمَ»

* ضاغَمَ فُلانٌ فُلانًا: عَضَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّريفُ المُرْتَضى (ت: 436هـ=1044م) يَفْتَخِرُ ويُعَرِّضُ بأَعْدائِهِ:

ضاغَمْتُمونا جَهْلَةً وإنَّما
ضاغَمْتُمُ مَن كانَ مِنكُمْ أَضْغَما

ديوان الشريف المرتضى، تح: ألتونجي، ج: 3، ص: 234.— المصدر
04 —

ضِغام

اسم
4.1معنى من معاني «ضِغام»

* الضِّغامُ: العَضُّ الشَّديدُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ رُؤْبَةُ (121هـ= 738م) يَصِفُ أَسَدًا، مِنْ قَصيدَةٍ يَمْدَحُ بها مَسْلَمَةَ بْنَ عبدِ المَلِكِ:

يَسُنُّ أَنْيابَ شَبا الضِّغامِ

أَرْأَسَ شَدَّاخٍ عَلى اللِّكامِ

ديوان رؤبة، تح: البروسي، ص: 148.— المصدر
05 —

ضُغامَة

اسم
5.1معنى من معاني «ضُغامَة»

* الضُّغامَةُ:كُلُّ ما يُعَضُّ ثُمَّ يُلْفَظُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن دُريْد (ت: 321هـ=933م):

"والضُّغامَةُ: كُلُّ ما ضَغَمْتَهُ ولَفَظْتَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة، تح: بعلبكي، ص: 906.— المصدر
06 —

ضَغْمَة

اسم
6.1معنى من معاني «ضَغْمَة»

* الضَّغْمَةُ: العَضَّةُ الشَّديدَةُ بِمِلْءِ الفَمِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مُغلِّسُ بنُ لَقيطٍ (ت: 22ق هـ=600م) يَتَأَسَّفُ على عَداوَةِ أَخَوَيْهِ مُدْرِكٍ ومُرَّةَ:

وقَدْ جَعَلَتْ نَفْسي تَطيبُ لِضَغْمَةٍ
لِضَغْمِهِماها يَقْرَعُ العَظْمَ نابُها

سيبويه، الكتاب، تح: هارون، ج: 2، ص: 365.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عبدُ المَلِكِ بْنُ مَرْوانَ (ت: 86هـ=705م) يُوَبِّخُ الحَجّاجَ حينَما أَغْلَظَ القَوْلَ لأَنَسِ بْنِ مالِكٍ:

"لَقَدْ هَمَمْتُ أنْ أُضْغِمَكَ ضَغْمَةً كبَعْضِ ضَغَماتِ اللُّيوثِ للثَّعالِبِ".

الزبير بن بكار، الأخبار الموفقيات، تح: العاني، ص: 274.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجاحِظُ (ت: 255هـ=869م) يَصِفُ كَيْفَ يَجِبُ أَنْ يُعامِلَ المَرْءُ الحَيوانَ إِنْ هُوَ كَرِهَهُ:

"فسَواءٌ عَلَيْكَ أَكانَ المَكْروهُ سَبُعًا وثَّابًا، أَوْ كانَ مَرَضًا قاتِلًا، وعَلى أنّكَ تَدْري لَعَلَّ النزْعَ والعَلَزَ والحَشْرَجَةَ أَنْ يَكونَ أَشَدَّ من لَدْغِ حَيَّةٍ، وضَغْمَةِ سَبُعٍ".

الجاحظ، الحيوان، تح: هارون، ج: 3، ص: 302.— المصدر
6.2معنى من معاني «ضَغْمَة»

وــــــــــ: ما يَحْمِلُهُ المَرْءُ تَحْتَ إِبْطِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الثَّعالبيُّ (ت: 430هـ=1039م):

"الضَّغْمَةُ: ما حَمَلْتَهُ تحْتَ إِبْطِكَ".

الثعالبي، فقه اللغة وأسرار العربية، تح: مراد، ص: 146.— المصدر
07 —

ضَيْغَم

اسم
7.1معنى من معاني «ضَيْغَم»

* الضَّيْغَمُ: الأَسَدُ. (ج) ضَياغِمُ، وضَياغِمَةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ نَوْفَلُ بْنُ سَعْدٍ (ت: 300ق هـ=330م) يَفْتَخِرُ بمآثِرِ قَوْمِهِ مِنْ حِمْيَرٍ:

مَنْ ذا مِنَ النّاسِ لَهُ ما لنا
مِنْ عارِبٍ في النّاسِ أَوْ أَعْجَمي

سارَ بِنا الرّائِشُ في جَحْفَلٍ
مِثْلِ مَفيضِ السَّيْلِ كالأَنْجُمِ

يَوْمًا لأَرْضِ الهِنْدِ يَسْمو لها
تَجْري بهِ الأَمْواجُ كالضَّيْغَمِ

ابن هشام الحميري، التيجان. ص: 89.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ كَعْبُ بنُ زُهَيْرٍ (8هـ=629م) يَمْدَحُ النَّبِيَّ ﷺ ويُشَبِّهُ هَيْبَتَهُ بهَيْبَةِ الأَسَدِ:

لَذاكَ أَهْيَبُ عِنْدي إِذْ أُكَلِّمُهُ
وقيلَ إِنَّكَ مَسْبورٌ ومَسْؤولُ

مِنْ ضَيْغَمٍ مِنْ ضِراءِ الأُسْدِ مَخْدَرُهُ
بِبَطْنِ عَثَّرَ غيلٌ دونَهُ غيلُ

ديوان كعب بن زهير، تح: الحتّي، ص: 39.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ= 787م):

"والضَّيْغَمُ: الأسَدُ".

الخليل، العين، تح: المخزومي وآخر، ج: 4، ص: 370.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابنُ هُتَيْمِلٍ (ت: 696هـ=1297م) يَمْدَحُ الملكَ المُظَفَّرَ بْنَ عُمَرَ الرَّسوليَّ:

فالخِضْرِمُ الأَحْوى وَإِنْ هاجَ لا
يُرْكَبُ والضَّيْغَمُ إِنْ هِيجَ هاجْ

ديوان ابن هتيمل، تح: الشميري، ص: 181.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ ابنُ سَنَدٍ البَصْريّ (ت: 1242هـ=1826م) يَرُدُّ عَلى مَنْ أَنْكَرَ فَضْلَ أَبي بَكْرٍ الصِّديقِ:

أَتَنْسى سُيوفًا في حَنيفَةَ أُورِدَتْ
لِتُورِدَ أَنْهارَ الهُدى كُلَّ حائِمِ

وتُورِدَ أَحْوَاضَ الرَّدَى كُلَّ ضَيْغَمٍ
بغَيْرِ الظُّبا والسُّمْرِ غَيْرَ مُسالِمِ

ابن سند البصري، ديوان الصارم القرضاب. تح: هنداوي، ص: 207.— المصدر
7.2معنى من معاني «ضَيْغَم»

وــــــــــــ مِنَ السِّباعِ: الواسِعُ الأَشْداقِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ جابرُ بْنُ حُنَيٍّ (ت: 60ق هـ=563م) يَفْخَرُ:

يَرى النّاسُ مِنّا جِلْدَ أَسْوَدَ سالِخٍ
وفَرْوَةَ ضِرْغامٍ من الأُسْدِ ضَيْغَمِ

الضبي، المفضليات، تح: أحمد شاكر وآخر، ص: 212.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ نافعُ بْنُ خَليفَةَ الغَنَويُّ (ت: 65هـ=685م) يُجيبُ الرّاعيَ النُّمَيْريَّ ويَفْتَخِرُ بِقَوْمِهِ:

ويَوْمًا عَلى أَهْوى رَطِئْنا وُجوهَهُمْ
فباءوا عَلى رُغْمٍ لنا بالمَظالِمِ

فَكَكْنا أَخانا بالمِئينَ وأَسْلَموا
أَخاهُمْ بمُعْتَبْسِ السِّباعِ الضّياغِمِ

البلاذري، أنساب الأشراف. تح: زكّار وآخر، ج: 13، ص: 261.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو حاتِمٍ السِّجِسْتانِيُّ (ت:250ه= 864م):

"قيلَ ‌للأَسَدِ: ‌ضَيْغَمٌ وضَيْغَمِيٌّ، وهُوَ الواسِعُ الأَشْداقِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السَّرَقُسْطيّ، الأفعال، تح: حسين شرف، ج: 2، ص: 210.— المصدر
7.3معنى من معاني «ضَيْغَم»

وــــــــــــ: الكَثيرُ العَضِّ الشَّديدُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنَّى (ت: 209هـ=825م):

"‏الضَّيْغَمُ: الّذي يَعَضُّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن فارس، مقاييس اللغة، تح: هارون، ج: 3، ص: 364.— المصدر
08 —

ضَيْغَمِيّ

اسم
8.1معنى من معاني «ضَيْغَمِيّ»

* الضَّيْغَمِيُّ مِنَ الأُسودِ: الواسِعُ الأَشْداقِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110هـ=728م) يَهْجو جَريرًا ويَتَوَعَّدُهُ:

غَرَّ كُلَيْبًا إِذِ اصْفَرَّتْ مَعالِقُها
بِضَيْغَمِيٍّ كَرِيهِ الوَجْهِ والأَثَرِ

ديوان الفرزدق. تح: فاعور، ص: 262.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أحَدُ الشعراءِ (197هـ=812م) يَذْكُرُ هُروبَ عَبْدِ اللهِ بْنِ خازِمٍ مِنْ بَغْدادَ إِلى المَدائِنِ في وَقْعَةِ بابِ الشّماسِيّةِ:

وما جَبُنَ ابنُ خازِمَ مِنْ رَعاعٍ
وأَوْباشِ الطَّغامِ مِنَ الأَنامِ

وَلَكِنْ خافَ ‌صَوْلَةَ ضَيْغَمِيٍّ
هَصورِ الشَّدِّ مَشْهورِ العُرامِ

الطبري، تاريخ الرسل والملوك. ج: 8، ص: 467.— المصدر
8.2معنى من معاني «ضَيْغَمِيّ»

وــــــــــــ: الكَثيرُ العَضِّ الشَّديدُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ (ت: 145هـ=762م) يَصِفُ بَعِيرًا:

أَشْدَقُ يَفْتَرُّ افْتِرارَ الأَفْوَهِ

عَنْ عَصِلاتِ ‌الضَّيْغَمِيِّ الأَجْبَهِ

شرح ديوان رؤبة، تح: عبد الوهاب، ج: 2، ص: 69، 70.— المصدر
