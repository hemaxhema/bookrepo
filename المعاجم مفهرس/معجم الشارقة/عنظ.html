عَنْظى(فعل) عَنْظاء(اسم) عُنْظوان(اسم) عِنْظِــيان(اسم)
النظائر الساميّة

احْتَفظَتِ السّامِيّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ؛ فقَدْ وَرَدَ الجَذْرُ (ع ن ظ):

في الجِعزيّةِ؛ ومنه الاسْمُ: አንጼዋ/አንጸዋ (أَنسَوَا/أَنس۪وَا) ʼans'ewā/ʼans'awā بمَعْنَى"فَأْرٌ- ابْنُ عِرْسٍ- ذُبابَةٌ"، وهو يَتَّفِقُ مَعَ الاسْمِ العَرَبيِّ (عُنْظُوَانٌ) مَبْنًى وفي بَعْضِ مَعَانِيهِ، غَيْرَ أَنَّ الاسْمَ الجِعْزِيَّ خَضَعَ لإِبْدالِ العَيْنِ هَمْزَةً والظَّاءِ سينًا دَفْعِيَّةً؛ ومن مَجيئِهِ بمَعْنَى (ابْنُ عِرْسٍ) ما وَرَدَ في سِفْرِ اللَّاويِّينَ: ወእምነ ዘሕያውኒ ዘይትሐወስ ውስተ ምድር ዝንቱ ዘርኩስ ለክሙ አንጸዋ ወጽቍጽቌን ወኀገጽ፡፡ أي:"وهذا هُوَ النَّجِسُ لَكُمْ من الدَّبِيبِ الَّذي يَدِبُّ على الأَرْضِ، ابْنُ عِرْسٍ والفَأْرُ والضَّبُّ على أَجْناسِهِ".

(ኦሪት ዘሌዋውያን: 11: 30)

01 —

عَنْظى

فعل
1.1معنى من معاني «عَنْظى»

*عَنْظى فُلانٌ بِفُلانٍ: شَتَمَهُ وأَسْمَعَ النّاسَ عُيوبَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَنْدَلُ الطُّهَوِيُّ (ت: 90هـ=709م) يَصِفُ زَوْجَتَهُ التي قامَتْ تُسَمِّعُ بهِ وتَفْضَحُهُ:

حَتَّى إِذَا أَجْرَسَ كُلُّ طائِرٍ

قامَتْ ‌تُعَنْظي بِكَ سَمْعَ الحاضِرِ

ابن السكيت، إصلاح المنطق. تح: أحمد شاكر وآخر، ص: 83.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ السِّكّيتِ (ت: 244هـ=858م):

"قَدْ ‌عَنْظى ‌بهِ، وخَنْذى بهِ، وحَنْظى بهِ، وخَنْظى بهِ، إِذَا نَدَّدَ بهِ، وأَسْمَعَهُ المَكْروهَ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، ترتيب إصلاح المنطق. تح: بكائي، ص: 223.— المصدر
02 —

عَنْظاء

اسم
2.1معنى من معاني «عَنْظاء»

*العَنْظاءُ: الهَضَبَةُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ناظِرُ الجَيْشِ (ت: 778هـ=1377م) يَصِفُ أمَّ الصَبِيَّيْنِ:

أُمّ الصَّبِيَّيْنِ ما يُدْريكِ أَنْ رُبَّما
عَنْظاءُ قَلَّتَها شَمَّاءُ قَرْواخُ

ناظر الجيش، شرح التسهيل. تح: فاخر وآخرين، ج: 6، ص: 3017.— المصدر
03 —

عُنْظوان

اسم
3.1معنى من معاني «عُنْظوان»

*العُنْظُوانُ: أَوَّلُ الشَّبابِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابنُ الغُرَيْزَةِ كُثَيِّر بنُ عَبْدِ اللهِ النَّهْشَلِيُّ (23ه=644م) يَرثي أَصْحابَهُ يَوْمَ الطالقانِ:

وأَيَّ فَتًى دَعَوْتَ وَقَدْ تَوَلَّتْ
بِهِنَّ الخَيْلُ ‌ذَاتُ ‌العُنْظُوانِ

الصفدي، الوافي بالوفيات. تح: الأرناؤوط وآخر، ج: 24، ص: 246— المصدر
3.2معنى من معاني «عُنْظوان»

وــــــــــمِنَ النّاسِ: الفاحِشُ المُسَمِّعُ. وهي بِتاءٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْرِيُّ (ت: 90هـ=709م):

تَرَكْنَ رِجالَ ‌العُنْظُوانِ تَنوبُهُمْ
ضِباعُ خُفافٍ مِنْ وَراءِ الأَباتِرِ

ديوان الرّاعي النميري. تح: الصّمد، ص: 146.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاسِمُ بنُ سَلّامٍ (ت: 224هـ=839م):

"العُنْظُوانُ: ‏الفاحِشُ مِنَ الرِّجالِ، والمَرْأَةُ عُنْظُوانَةٌ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، الغريب المصنف. تح: داوودي، ج: 1، ص: 113.— المصدر
3.3معنى من معاني «عُنْظوان»

وــــــــــ: نَباتٌ مِنَ الحَمْضِ، ضَخْمٌ أَغْبَرُ، يُسْتَظَلُّ بِهِ، وإذا اسْتَكْثَرَ مِنْهُ البَعيرُ وَجِعَ بَطْنُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"العُنْظُوانُ: نبَاتٌ إِذَا اسْتَكْثَرَ مِنْهُ البَعيرُ وَجِعَ بَطْنُهُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 87.— المصدر
”وقالَ الرّاجِزُ (170هـ=786م، مِنْ شَواهِدِ الخَليلِ) يَصِفُ إبِلًا:

‌حَرَّقَها ‌وارِسُ عُنْظُوانِ

فَاليَوْمُ مِنْها يَوْمُ أَرْوَنانِ

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 87.— المصدر
3.4معنى من معاني «عُنْظوان»

وـــــــــ: الجَرادُ. وواحِدَتُهُ بِتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"‌العُنْظُوانَةُ: الجَرادَةُ الأُنْثى، والجَمْعُ العُنْظُواناتُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 87.— المصدر
”وقالَ أَعْرابِيٌّ (231هـ=845م، مِنْ شَواهِدِ ابنِ الأَعْرابِيِّ) يَصِفُ جَرادَةً:

ومِنْ ‌عُنْظُوانٍ ‌صَعْبَةٍ شَمَّرِيَّةٍ
تَخُبُّ برجْلَيْها أَمامَ الرَّكائِبِ

الجاحظ، الحيوان. تح: عيون السود، ج: 6، ص: 440.— المصدر
3.5معنى من معاني «عُنْظوان»

وـــــــــــ مِنَ الرِّجالِ: الطَّويلُ المُضْطَرِبُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن دُرَيْدٍ (ت: 321هـ=933م):

"ورَجُلٌ عُنْظُوانٌ: طَويلٌ مُضْطَرِبٌ".

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 3، ص: 1236.— المصدر
3.6معنى من معاني «عُنْظوان»

وـــــــــــ: السّاخِرُ المُغْري.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابن سيدَه (ت: 458هـ=1066م):

"العُنْظُوانُ، والعِنْظيانُ: الشِّرّيرُ المُسَمِّعُ. وقيلَ: هو ‌السّاخِرُ ‌المُغْري(وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

ابن سيده، المحكم. تح: هنداوي، ج: 2، ص: 66.— المصدر
04 —

عِنْظِــيان

اسم
4.1معنى من معاني «عِنْظِــيان»

* العِنْظِيانُ: الجافي.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو عَمْرٍو الشَّيْبانيُّ (ت: 206هـ=821م):

"العِنْظِيانُ: الجافي". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السيرافي، شرح كتاب سيبويه. تح: مهدليوآخرين، ج: 5، ص: 153.— المصدر
4.2معنى من معاني «عِنْظِــيان»

وــــــــــ: الفاحِشُ البَذيءُ المُسَمِّعُ. وهي بِتاءٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ كُراعُ النَّمْلِ (ت: 309هـ=921م):

"والعُنْظُوانُ، والحُنْظُوانُ، ‌والعِنْظِيانُ، والحِنْظِيانُ، والحِنْذِيانُ كُلُّهُ: الفاحِشُ ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

كراع، المنتخب. تح: العمري،ص: 202.— المصدر
4.3معنى من معاني «عِنْظِــيان»

وـــــــــ: النّاعِمُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السِّيرافِيُّ (ت: 368هـ=979م):

"العِنْظِيانُ: ‏النّاعِمُ. ويُقالُ: هو أَوَّلُ الشَّبابِ".

السيرافي، شرح كتاب سيبويه. تح: مهدلي وآخرين، ج: 5، ص: 153.— المصدر
