01 —

ضاحَ

فعل
1.1معنى من معاني «ضاحَ»

* ضاحَتِ البِلادُ تَضيحُ ضَيْحًا: خَلَتْ مِنَ النَّباتِ والرَّعْيِ. (انظر: ض ح و-ي)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) في دُعاءِ الاسْتِسْقاءِ:

«اللَّهُمَّ ضاحَتْ بِلادُنا واغْبَرَّتْ أَرْضُنا وهامَتْ دَوابُّنا، اللَّهُمَّ ارْحَمْ بَهائِمَنا الحائِمَةَ، والأَنْعامَ السّائِمَةَ، والأَطْفالَ الْمُحْثَلَةَ».

الخطابي، غريب الحديث. تح: الغرباوي، ج: 1، ص: 336.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ سِيْدَهْ (ت: 458هـ=1066م):

"وضاحَتِ البِلادُ: خَلَتْ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 3، ص: 420.— المصدر
1.2معنى من معاني «ضاحَ»

وــــــــــ العَظْمُ ضَيْحًا، ضُيوحًا، وضَيَحانًا: انْكَشَفَ وتَحَرَّكَ مِنَ الضَّعْفِ والهُزالِ. (انظر: ض ي ج)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (231هـ=845م):

"ضاحَتْ عِظامُهُ؛ إِذا تَحَرَّكَتْ مِنَ الهُزالِ، وبَرَزَتْ حَتّى يَرى النَاظِرُ حَجْمَها ضَيْحًا، وضُيوحًا وضَيَحانًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 333.— المصدر
”وقالَ الرّاجِزُ، مِنْ شَواهِدِ ابْنِ الأَعْرابِيِّ (231هـ=846م) يَصِفُ حالَهُ عِنْدَ الشَّيْبِ:

إِمّا تَرَيْني كالعَريشِ الـمَضْروجْ

ضاحَتْ عِظامي عن لَقًى مَفْروجْ

فَقَدْ شَهِدْتُ اللَّهْوَ غَيْرَ التَّزْليجْ

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 333.— المصدر
1.3معنى من معاني «ضاحَ»

وــــــــــ فُلانٌ اللَّبَنَ ضَيْحًا: مَزَجَهُ بالماءِ حَتّى رَقَّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"والضَّيْحُ: مصدر ضِحْتُ اللَّبَنَ ضَيْحًا؛ إذا مَزَجْتُهُ بالماءِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ص: 549.— المصدر
02 —

ضِيح

فعل
2.1معنى من معاني «ضِيح»

* ضِيحَ (بِالبِناءِ لَلْمَجْهولِ) العَيْشُ: كُدِّرَ ونُغِّصَ ومُزِجَ بِالأَلَمِ. (لَمْ يُسْتَعْمَلْ هَذا الفِعْلُ بِهذا المـَعنى مِنْ قَبْلُ، واسْتُعْمِلَ المـُشْتَقُّ مِنْهُ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاءُ (ت: 207هـ=822م):

"عَيْشٌ مَضْيوحٌ؛ أيْ: مَمْذوقٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الأبياري، ج: 2، ص: 69.— المصدر
03 —

أَضاحَ

فعل
3.1معنى من معاني «أَضاحَ»

* أَضاحَ حَمْلُ الدَّوْمِ: نَضِجَ وحانَ لَهُ أَنْ يُؤْكَلَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصَّغانِيُّ (ت:650هـ=1252م):

"والضَّيْحُ، أيضًا: الـمُقْلُ إذا نَضِجَ، وقَدْ ‌أَضاحَ؛ أيْ: حانَ له أَنْ يُؤْكَلَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة. تح: الأبياري، ج: 2، ص: 69.— المصدر
04 —

ضَيَّحَ

فعل
4.1معنى من معاني «ضَيَّحَ»

* ضَيَّحَ اللَّبَنُ: كَثُرَ مَزْجُهُ بالماءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ نَشْوانُ الحِمْيَرِيُّ (ت: 573هـ=1178م):

"ضَيَّحَ ‌اللَّبَنُ: إِذا كَثُرَ مَزْجُهُ بالماءِ".

نشوان، شمس العلوم. تح: العمري وآخرين، ج: 6، ص: 4030.— المصدر
4.2معنى من معاني «ضَيَّحَ»

وــــــــــفُلانٌ فُلانًا: سَقاهُ لَبَنًا مَمْزوجًا بِالماءِ. (انْظُر: ض و ح)

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ ثَعْلَبَةُ بْنُ عَمْرٍو (ت: 27ق.هـ=595م) يُخاطِبُ ابْنَتَهُ أسْماءَ شاكِيًا قَوْمَهُ:

وأَهْلَكَ مُهْرَ أَبيكِ الدِّوا
ءُ ليْس له منْ طَعامٍ نَصيبُ 

خَلا أَنَّهُم كُلَّما أَوْرَدُوا
يُضَيّحُ قَعْبًا عَلَيْهِ ذَنوبُ

ديوان بني بكر في الجاهلية. تح: نبوي، ص: 376.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (286هـ=899م) مِنْ شَواهِدِ المُبَرِّدِ، يَصِفُ عَنْزًا:

إِنّي رَأَيْتُ العَنْزَ تَمْنَعُ رَبَّها
مِنْ أَنْ ‌يُضَيِّحَ جارَها بالسِّنْبِس

المبرد، الفاضل. تح: الميمني، ص: 80.— المصدر
4.3معنى من معاني «ضَيَّحَ»

وــــــــــاللَّبَنَ: مَزَجَهُ بالماءِ وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"يقالُ: ضَيَّحْتُهُ فَتَضَيَّحَ، ولا يُسَمّى ضَياحًا إلّا اللَّبَنُ".

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 267.— المصدر
”وقالَ أَبو نُواسٍ (ت: 198هـ=814م) يَصِفُ صُقورَ الصَّيْدِ:

لا صَيْدَ إلّا بالصُّقورِ اللُّمَّحِ

كُلِّ قُطامِيٍّ بَعيدِ المَطْرَحِ

يَجْلو حِجاجِي مُقْلَةٍ لم تُجْرَحِ

لم تَغْذُهُ باللَّبَنِ الـمُضَيَّحِ

ديوان أبي نُواس. تح: بهجت الحديثي، ص: 185.— المصدر
4.4معنى من معاني «ضَيَّحَ»

o الصَّيْفَ ضَيَّحْتِ اللَّبَنَ: (انْظُرْ: ص ي ف).

05 —

تَضَيَّحَ

فعل
5.1معنى من معاني «تَضَيَّحَ»

* تَضَيَّحَ فُلانٌ: تَأَخَّرَ في الورودِ فَلَمْ يَجِدْ إِلّا البَقِيَّةَ الكَدِرَةَ مِن الماءِ. (لَمْ يُسْتَعْمَلْ هَذا الفِعْلُ بِهذا المـَعنى مِنْ قَبْلُ، واسْتُعْمِلَ المـُشْتَقُّ مِنْهُ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

«مَنْ اعْتَذَرَ إلَيْهِ أَخوهُ مِنْ ذَنْبٍ فرَدَّهُ، لم يَرِدْ عَلَيَّ الحَوْضَ إِلّا مُتَضَيِّحًا».

الأزهري، تهذيب اللغة. تح: درويش، ج: 5، ص: 160.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو الهَيْثَم الرّازي (ت: 276هـ=890م) يَشْرَحُ معنى المُتَضَيِّحِ في الحديثِ النَّبَوِيِّ:

"إلّا مُتَضَيِّحًا... هُوَ الّذِي يَجيءُ آخِرَ النّاس فِي الوِرْدِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: نصار، ج: 6، ص: 572.— المصدر
5.2معنى من معاني «تَضَيَّحَ»

وــــــــــاللَّبَنُ: امْتَزَجَ بالماءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"يُقالُ: ضَيَّحْتُهُ فَتَضَيَّحَ، ولا يُسَمّى ضَياحًا إلّا اللَّبَنُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 267.— المصدر
5.3معنى من معاني «تَضَيَّحَ»

وــــــــــفُلانٌ: شَرِبَ اللَّبَنَ المـَمْزوجَ بِالماءِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م):

"والتَّضَيُّحُ: شُرْبُ ‌الضَّياحِ. يُقالُ: ضَيَّحْتُهُ فتَضَيَّحَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 351.— المصدر
06 —

ضاحَة

اسم
6.1معنى من معاني «ضاحَة»

* الضّاحَةُ: البَصَرُ أو العَيْنَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"والضّاحَةُ البَصَرُ، ما أَجْوَدَ ضاحَتَهُ".(وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 3، ص: 156.— المصدر
07 —

ضياح

اسم
7.1معنى من معاني «ضياح»

* الضَّياحُ: اللَّبَنُ ونَحْوُهُ الـمَمْزوجُ بِالكَثيرِ مِن الماءِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مالِكُ بْنُ الحارِثِ (ت: 80ق.هـ=544م) يَهْجو:

رَأَيْتُ مَعاشرًا يُثْنى عَلَيْهِمْ
إذا شَبِعوا وأَوْجُهُهُمْ قِباحُ

يَظَلُّ الـمُصْرِمونَ لَهُمْ سُجودًا
ولَوْ لَمْ يُسْقَ عِنْدَهُمُ ضَياحُ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 1، ص: 314.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) لعمّارِ بْنِ ياسِرٍ:

«لَنْ تَموتَ حَتّى تَقْتُلَكَ الفِئَةُ الباغِيَةُ، تَشْرَبُ شَرْبَةَ ‌ضَياحٍ تَكونُ آخِرَ رِزْقِكَ مِنَ الدُّنْيا».

الحاكم، المستدرك. تح: عبد القادر، ج: 3، ص: 442.— المصدر
”وقالَ عَمّارُ بْنُ ياسِرٍ (ت: 37هـ=658م):

"إِنَّ حَبيبي حَدَّثَني أَنَّ آخِرَ مَشْرَبي مِنَ الدُّنْيا ‌ضَياحُ لَبَنٍ حَتّى أَرِدَ عَلَيْهِ الحَوْضَ".

البزّار، البحر الزخَّار. تح: زين الله، ج: 4، ص: 259.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الضَّياحُ: اللَّبَنُ الخاثِرُ يُصَبُّ فيه الماءُ، ثُمَّ يُجْدَحُ".

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 267.— المصدر
”وقالَ القاسِمُ بْنُ يُوسُفَ (ت: 220ه= 835م) يَرْثي عَنْزًا لَهُ:

كمْ شَرِبْنا مَحْضًا لَها وضَياحا
وحَقينًا مُخَمَّرًا في السِّقاءِ

الصولي، الأوراق، ج:1، ص: 165.— المصدر
08 —

ضَيْح

اسم
8.1معنى من معاني «ضَيْح»

* الضَّيْحُ: اللَّبَنُ ونَحْوُهُ الـمَمْزوجُ بِالكَثيرِ مِنَ الماءِ. القِطْعَةُ مِنْهُ بِتاءٍ. (ج) أَضْياحٌ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى (ت: 7ه= 628م) يَصِفُ ماءً كَدِرًا:

كَأَنَّ عَصيرَ الضَّيْحِ في سَدَيانِهِ
دَفونًا وأَسْدامًا طَويلًا دُثورُها

ديوان الأعشى. تح: محمد حسين، ص: 373.— المصدر
”وقالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) يُخاطِبُ عَمّارَ بنَ ياسِرٍ:

«إِنَّ آخِرَ زادِكَ مِنَ الدُّنْيا ‌ضَيْحٌ مِنْ لَبَنٍ».

البلاذري، أنساب الأشراف. تح: زكّار وآخر، ج: 1، ص: 195.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ بِلالُ بْنُ جَريرٍ (ت:140هـ=757م) يَهْجو بَخيلًا:

تَناوَمْتَ نِصْفَ اللَّيْلِ ثُمَّ أتَيْتَنا
بقَعْبَيْنِ مِنْ ‌ضَيْحٍ وما كِدْتَ تَفْعَلُ

أبو تمّام، الوحشيات. تح: الميمني، ص: 225.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ المَقْريزِيُّ (ت: 845هـ=1441م) يصِفُ حالَ هلالِ بْنِ أُمَيَّةَ حينَ هَجَرَهُ المُسْلمونَ:

"امْتَنَعَ عَنِ الطَّعامِ، وواصَلَ اليَوْمَيْنِ والثَلاثَةَ ما يَذوقُ طَعامًا، إلّا أَنْ يَشْرَبَ الشَّرْبَةَ مِنَ الماءِ أو ‌الضّيحَ مِنَ اللَّبَنِ".

المقريزي، إمتاع الأسماع. تح: النميسي، ج: 2، ص: 81.— المصدر
ح

”قالَ أَحْمَدُ فارِسٍ الشِّدْياقُ (ت: 1304هـ=1887م) يَذْكُرُ الإِفْرنْجَ:
