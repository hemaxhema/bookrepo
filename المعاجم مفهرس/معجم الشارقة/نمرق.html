01 —

نَمْرُقَة / نُمْرُقَة / نِمْرِقَة

اسم
1.1معنى من معاني «نَمْرُقَة / نُمْرُقَة / نِمْرِقَة»

* النَّمْرُقَةُ، والنُّمْرُقَةُ، والنِّمْرِقَةُ: الوِسادَةُ، وقَدْ تُحْذَفُ تاؤُهُ. (ج) نَمارِقُ. (مع) مِنْ أَصْلٍ فارِسِيٍّ، هُوَ نرمَ (nāmra)، يُطْلَقُ عَلى كُلِّ شَيْءٍ لَيِّنٍ ناعِمٍ ولَطيفٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَتْ كَرْمَةُ البَكْرِيَّةُ (ت: 125ق.ه=500م) تُحَرِّضُ الرِّجالَ وتُهَيِّجُهُمْ في الحَرْبِ

نَحْنُ بَناتِ طارِقْ

نَمْشي عَلى النَّمارِقْ

مَشْيَ القُطِيِّ البارِقْ

بشير يموت، شاعرات العرب. ص: 42.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ هِنْدُ بِنْتُ عُتْبَةَ (3ه=625م) تُحَرِّضُ الرِّجالَ يَوْمَ أُحُدٍ:

نَحْنُ بَناتُ طارِقْ

إنْ تُقْبِلوا نُعانِقْ

ونَفْرِشُ النَّمارِقْ

أَوْ تُدْبِروا نُفارِقْ

فِراقَ غَيْرِ وامِقْ

ابن إسحاق، السيرة النبوية. تح: أحمد المزيدي، ج: 1، ص: 335.— المصدر
”وفي القُرْآنِ الكَريمِ (11ه=632م) قالَ اللهُ يُخْبِرُ عَنْ نَعيمِ الجَنَّةِ:

﴿وَنَمَارِقُ مَصۡفُوفَةٞ﴾ [الغاشية: 15]

”وقالَ النَّبِيُّ ﷺ (ت: 11ه=632م) يَسْأَلُ عائِشَةَ وقَدِ اشْتَرَتْ لَهُ وِسادَةً عَلَيْها تَصاويرُ:

«ما بالُ هَذِهِ النُّمْرُقَةِ؟!».

البخاريّ، الصحيح. تح: صدقي العطّار، ص: 500.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سَديفُ بْنُ إِسْماعيلَ (ت: 146ه=763م) يُثْني عَلى السَّفَّاحِ العَبَّاسِيِّ وَيُحَرِّضُهُ عَلى بَني أُمَيَّةَ:

واقْبَلَنْ أَيُّها الخَليفَةُ نُصْحي
واحْتِياطي لِأَمْرِكُمْ واحْتِراسي

فلَقَدْ ساءَني وساءَ سُوائي
قُرْبُهُمْ مِنْ نَمارِقٍ وكَراسي

شعر سديف بن ميمون. تح: رضوان العبود، ص: 24.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الرَّسْعَنِيُّ (ت: 661ه=1263م) يُفَسِّرُ قَوْلَهُ تَعالى: ﴿‌وَأَعۡتَدَتۡ ‌لَهُنَّ مُتَّكَـٔٗا﴾ [يوسف:31]:

"أَيْ: أَعَدَّتْ وهَيَّأَتْ لَهُنَّ مَجْلِسًا يَتَّكِئْنَ عَلَيْهِ مِنَ النَّمارِقِ والفُرُشِ كَعادَةِ المُتْرَفينَ".

الرّسعني، رموز الكنوز. تح: عبد الملك دهيش، ج: 3، ص: 324.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّد عَبْدُه (ت: 1323ه=1905م) يَذْكُرُ نَعيمَ الجَنَّةِ:

"ثُمَّ في الجَنَّةِ غَيْرَ السُّرُرِ الَّتي توضَعُ على جَوانِبِ العُيونِ (نَمارِقُ مَصْفوفَةٌ)، والنَّمارِقُ: جَمْعُ نُمْرُقَةٍ -بِضَمِّ النُّونِ وكَسْرِها- وهِيَ الوِسادَةُ المُسَمَّاةُ في عُرْفِ العامَّةِ مَسْنَدًا ومِخَدَّةً. وسَواءٌ كانَتْ هذهِ النَّمارِقُ مَصْفوفَةً فَوْقَ الأَسِرَّةِ أَوْ في جَوانِبِ المَساكِنِ".

محمد عبده، الأعمال الكاملة. تح: عمارة، ج: 5، ص: 382.— المصدر
1.2معنى من معاني «نَمْرُقَة / نُمْرُقَة / نِمْرِقَة»

وــــــــــ: البِساطُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”وبِهِ فُسِّرَ قَوْلُ كَرْمَةَ البَكْرِيَّةِ (ت: 125ق.ه=500م) تُحَرِّضُ الرِّجالَ وتُهَيِّجُهُمْ في الحَرْبِ:

نَحْنُ بَناتِ طارِقْ

نَمْشي عَلى النَّمارِقْ

مَشْيَ القُطِيِّ البارِقْ

بشير يموت، شاعرات العرب. ص: 42.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ هِنْدٍ بِنْتِ عُتْبَةَ (3ه=625م) تُحَرِّضُ الرِّجالَ يَوْمَ أُحُدٍ:

نَحْنُ بَناتُ طارِقْ

إِنْ تُقْبِلوا نُعانِقْ

ونَفْرِشُ النَّمارِقْ

أَوْ تُدْبِرُوا نُفارِقْ

فِراقَ غَيْرِ وامِقْ

ابن إسحاق، السيرة النبوية. تح: أحمد المزيدي، ج: 1، ص: 335.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ مَكِّيٍّ الصَّقَلِّيُّ (ت: 501ه=1107م):

"ويَقولونَ لِلبِساطِ: ‌نُمْرُقَةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن مكي، تثقيف اللسان. تح: مصطفى عطا، ص: 169.— المصدر
1.3معنى من معاني «نَمْرُقَة / نُمْرُقَة / نِمْرِقَة»

وــــــــــ: الطِّنْفِسَةُ الَّتي تُلْقى فَوْقَ الرَّحْلِ أَوْ تَحْتَهُ، يَفْتَرِشُها الرَّاكِبُ كَالمِرْفَقَةِ، غَيْرَ أَنَّ مُؤَخَّرَها أَعْظَمُ مِنْ مُقَدَّمِها، ولَها أَرْبَعَةُ سُيورٍ تُشَدُّ بِآخِرَةِ الرَّحْلِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق.ه=544م) يَصِفُ حِمارًا وَحْشِيًا:

كَأَنِّي ورَحْلِي والقُرَابَ ‌ونُمْرُقي
عَلى ظَهْرِ عَيْرٍ وارِدِ الخَبِرَاتِ

ديوان امرئ القيس وملحقاته. تح: أنور عليان وآخرين، ج: 1، ص: 585.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ الأَعْشى (ت: 7ه=628م) يَصِفُ ناقَةً:

وكورٍ عِلافِيٍّ وقِطْعٍ ونُمْرُقٍ
ووَجْناءَ مِرْقالِ الهَواجِرِ عَيْهَمِ

ديوان الأعشى. تح: محمود الرضواني، ج: 1، ص: 312.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرَّاجِزُ (ت: 209ه=824م) مِنْ شَواهِدِ مَعْمَرِ بْنِ المُثَنَّى، يَصِفُ الإِبِلَ:

تَضِجُّ مِنْ أَسْتاهِها النَّمارِقُ

مَفارِشُ الرِّحالِ والأَيانِقُ

الأزهري، تهذيب اللغة. تح: هارون، ج: 9، ص: 418.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ المُرَحَّلِ (ت: 699ه=1300م) يَذْكُرُ مَعْنى الطِّنْفِسَةِ واللُّغاتِ الوارِدَةَ فيها:

وهَذِهِ مِنْ فَوْقِ رَحْلي طِنْفِسَهْ
‌نُمْرُقَةٌ وقيلَ فيها طَنْفَسَهْ

ابن المرحل، متن موطأة الفصيح. تح: عبد الله الحكمي، ص: 154.— المصدر
02 —

نِمْرِقَة

اسم
2.1معنى من معاني «نِمْرِقَة»

* النِّمْرِقَةُ مِنَ السَّحابِ: ما كانَ بَيْنَهُ فُتوقٌ. (ز)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ عَبَّادٍ (ت: 385ه=995م):

"وما عَلى السَّماءِ ‌نِمْرِقَةٌ: وهِيَ مِنَ السَّحابِ ما كانَ بَيْنَه خُلوصٌ؛ أي: فُتوقٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 6، ص: 109.— المصدر
