01 —

حاس

فعل
1.1معنى من معاني «حاس»

* حاسَ فُلانٌ يَحيسُ حَيْسًا: أَطْعَمَ الـحَيْسَ، وَهُوَ الطَّعامُ الـمُتَّخَذُ مِنَ الأَقِطِ والتَّمْرِ والسَّمْنِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ اللِّحْيانيّ (ت: 220هـ=835م) :

"أَتَيْتُ بني فُلانٍ فخَبَزوا وحاسوا وأَقَطوا؛ أَي: أَطْعَموني ذلِك". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزبيدي، تاج العروس. تح: الطحاوي، ج: 19، ص: 134.— المصدر
1.2معنى من معاني «حاس»

وــــــــــ حَيْسُ القَوْمِ: دَنا هَلاكُهُمْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاء (207هـ=822م) :

"قد حاسَ حَيْسُهُمْ: إِذا دَنا هَلاكُهُمْ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن منظور، اللسان. تح: أمين محمد، وآخر، ج: 3، ص: 392.— المصدر
1.3معنى من معاني «حاس»

وــــــــــ الـحَيْسَ: خَلَطَهُ، أَو اتَّخَذَهُ طَعامًا.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عامِرُ بْنُ جُوَيْن (نحو: 74ق.هـ=550م) :

وإذا الشَّدائِدُ بِالشَّدائِدِ مَرَّةً
أشْجَتْكُمُ فأَنا الأَحَبُّ الأَقْرَبُوإذا تَكونُ عَظيمَةٌ أُدعى لَها
وإذا يُحاسُ الحَيْسُ يُدْعى جُنْدَبُ

البحتري، الحماسة.، تح: محمد إبراهيم حور، وآخر، ص: 177.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَراهيديُّ (ت: 170هـ=787م):

"الـحَيسُ: خلْطُ الأَقِطِ بِالتَّمْرِ، يُعْجَنُ كالـخَميرَةِ. حِسْتُهُ حَيْسًا". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 1، ص: 378.— المصدر
1.4معنى من معاني «حاس»

وــــــــــ الشَّرابَ: خَلَطَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ مُسْلِمُ بْنُ عُقْبَةَ (ت: 63هـ=683م) حينَ أُتيَ بِمَعْقِلِ بْنِ سِنان:

"حيسوا لَهُ شرْبَةً مِنْ سَويقِ اللَّوْزِ الَّذي زَوَّدَنا بِهِ أَميرُ الـمُؤْمِنينَ".

ابن قتيبة، الإمامة والسياسة. تح: علي شيري، ص: 237.— المصدر
1.5معنى من معاني «حاس»

وــــــــــ الأَمْرَ: لَمْ يُحْكِمْهُ، واضْطَرَبَ فيهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشَّاعِرُ (255ه=869م، من شَواهِدِ شَمِرِ بْنِ حَمْدَوَيْهِ) يُقَبِّحُ صَنيعَ امْرَأَةٍ:

تَعيبينَ أَمْرًا ثُمَّ تَأْتينَ مِثْلَهُ
لَقَدْ حاسَ هذا الأَمْرَ عِنْدَكِ حائِسُ

الأزهري، التهذيب. تح: درويش، ج: 5، ص: 172.— المصدر
1.6معنى من معاني «حاس»

وــــــــــ الحَبْلَ: فَتَلَهُ ولَمْ يُـحْكِمْهُ. (يَمَنِيَّةٌ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْد (321هـ=933م) :

"وأهْلُ اليَمَنِ يَقولونَ: حِسْتُ الـحَبْلَ أَحيسُهُ حَيْسًا: إِذا فَتَلْتَهُ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، الجمهرة. تح: بعلبكي، ج: 1، ص: 1049.— المصدر
02 —

حيس

فعل
2.1معنى من معاني «حيس»

* حيسَ أَوَّلُ الأُمَّةِ عَلَى آخِرِها: دَنَتْ قِيامَتُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ حُسين بن رُسْتم الأَيْلِيّ (ت: 145هـ=762م) :

"إنَّ هَذِهِ الأُمَّةَ حيسَ أَوَّلُها عَلى آخِرِها، ونودِيَ فيها بالرَّحيلِ"

الفسوي، المعرفة والتاريخ. تح: أكرم العمري، ج: 1، ص: 699.— المصدر
2.2معنى من معاني «حيس»

وــــــــــ حَيْسُ القَوْمِ: دنا هَلاكُهُمْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَرّاء النَّحْوي (ت: 207هـ=822م):

"يُقالُ: قَدْ حِيسَ حَيْسُهُمْ، إذا دَنا هَلاكُهُمْ" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، التهذيب. تح: درويش، ج: 5، ص: 172.— المصدر
2.3معنى من معاني «حيس»

وــــــــــ الوَلَدُ: أَحاطَتْ بِهِ الإِماءُ مِنْ جَوانِبِ نَسَبِهِ. فَهوَ مَحْيوسٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو عُبَيْدٍ القاسِمُ بْنُ سَلّامٍ (ت: 224ه=839م):

"الـهَجينُ: الذي وَلَدَتْهُ أمَةٌ؛ فإنْ وَلَدَتْهُ أمَتانِ أو ثَلاثٌ فهو الـمُكَرْكَسُ، فإنْ أحْدَقَتْ ‌به ‌الإماءُ ‌من كُلِّ وَجْهٍ فهو مَـحْيُوسٌ، وذلِكَ لأنَّهُ يُشبَّه بالـحَيْس" (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو عبيد، الغريب المصنف. تح: داوودي، ج: 1، ص: 403.— المصدر
2.4معنى من معاني «حيس»

وــــــــــ الأَمْرُ حَيْسًا: خُلِطَ، أَوْ فُرِغَ مِنْهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الشَّاعِرُ(ت231هـ=845م، مِنْ شَواهِدِ ابْنِ الأَعْرابِيّ) يهجو سَجاحَ مُدَّعِيَةَ النُّبُوَّةِ:

عَصَتْ سَجاحِ شَبَثًا وقَيْسا
ولَقِيَتْ مِنَ النِّكاحِ وَيْسا
قَدْ حيسَ هَذا الدّينُ عِنْدي حَيْسا

محمود أبو الخير، ديوان حروب الردة. ص: 628.— المصدر
03 —

حَيَّس

فعل
3.1معنى من معاني «حَيَّس»

* حَيَّسَ فُلانٌ الأَقِطَ: خَلَطَهُ بالتَّمْرِ ونَـحْوِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخليل (170هـ=787م) :

"وحيَّسْتُه تَـحْييسًا". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: هنداوي، ج: 1، ص: 378.— المصدر
04 —

حَيْس

اسم
4.1معنى من معاني «حَيْس»

* الـحَيْسُ: الطَّعامُ الـمَخْلوطُ مِنَ الأَقِطِ والتَّمْرِ والسَّمْنِ. واحِدُهُ: حَيْسَةٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ النَّبِيُّ ﷺ (نحو: 3ق ه=618م) لخَديجَة:

"أَمَعَكِ حِلابٌ فيهِ حَيْسٌ؟".

الفاكهي، أخبار مكة.، تح: دهيش، ج: 4، ص: 94.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ عائِشَةُ بِنْتُ أَبي بَكْرٍ الصِّدِّيقِ (نحو: 11ه=632م) تُجيبُ النَّبِيَّ صَلّى اللهُ عَلَيْهِ وَسَلَّمَ حينَ سَأَلَها "هَلْ عِنْدَكُمْ طَعامٌ؟":

"إنّا قَدْ أُهْدِيَ إِلَيْنا حَيْسٌ".

الطبراني، المعجم الأوسط. تح: طارق بن عوض الله وآخر، ج: 7، ص: 233.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الفَراهيديُّ (ت: 170هـ=787م) :

"الـحَيْسُ: خَلْطُ الأَقِطِ بِالتَّمْرِ، يُعْجَنُ كالـخَميرَةِ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 3، ص: 273.— المصدر
”وقالَ هارونُ الرَّشيد (ت: 193ه=809م):

"فُطِمْتُ على الـحَيْسِ والـمَوْزِ".

ابن دريد، الجمهرة. تح: بعلبكي، ج: 1، ص: 537.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النَّوويّ (ت: 676هـ=1277م) يُعَلِّقُ عَلى حَديثٍ نَبَويّ:

"وقَدْ سَبَقَ هذا في البابِ قَبْلَهُ، وسَبَقَ هُناكَ بَيانُ الـحَيْسِ".

النووي، شرح صحيح مسلم، ج: 9، ص: 327.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الـمُبارَكْفوريّ (ت: 1353هـ=1934م) يُعَقِّبُ عَلى تَوْفيقِ الحافِظِ ابْنِ حَجَرٍ بَيْنَ خَبَرَيْنَ ذَكَرا أَنَّ النَّبِيَّ ﷺ أَوْلَمَ عَلى صَفِيَّةَ بَنْتِ حُيَيٍّ بِحَيْسٍ أَوْ بِسَويقٍ وتَمْرٍ بِقَوْلِهِ: "ولا مُخالَفَةَ بَيْنَهُما":

"يَعْني بَيْنَ هذِهِ الرِّوايَةِ وبَيْنَ الرِّوايَةِ الَّتي فيها ذِكْرُ الـحَيْسِ؛ لِأَنَّ هذِهِ مِنْ أَجْزاءِ الـحَيْسِ".

المباركفوري، تحفة الأحوذي.، ج: 4، ص: 185.— المصدر
4.2معنى من معاني «حَيْس»

وــــــــــ مِنَ الأمورِ: الرَّديءُ غَيْرُ الـمُحْكَمِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزهري (370هـ=981م):

"إِنَّ هذَا الأَمْرَ حَيْسٌ؛ أَي: لَيْسَ بِمُحْكَمٍ، وهو رَديءٌ". (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، التهذيب. تح: درويش، ج: 5، ص: 172.— المصدر
4.3معنى من معاني «حَيْس»

وــــــــــ: الـخَيْرُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"وما لَهُ قَلَّ حَيْسُهُ؛ أَيْ: خَيْرُهُ".

ابن فارس، متخير الألفاظ. تح: هلال ناجي، ص: 65.— المصدر
4.4معنى من معاني «حَيْس»

وــــــــــ مِنَ النّاسِ: الـخَسيسُ الوَضيعُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالِكٍ (ت: 672هـ=1273م) يُعَدِّدُ المُفْرَداتِ الدّالّةَ عَلى الخِسَّةِ وَالضَّعَةِ:

"وغْدٌ، دَنيءٌ، حَيْسٌ، نِكْسٌ، نَذْلٌ، مَهينٌ".

ابن مالك، الألفاظ المختلفة في المعاني المؤتلفة. تح: عواد، ج: 1، ص: 192.— المصدر
05 —

حَيوس

اسم
5.1معنى من معاني «حَيوس»

*الـحَيوسُ مِنَ الرِّجالِ: الَّذي يُكْثِرُ صُنْعَ الـحَيْسِ (ج) حُيُسٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الرّاجِزُ (206ه=821م، مِنْ شَواهِدِ أَبي عَمْرٍو الشَّيْبانِي):

لا تَعْلَقي بحَجْحَجٍ حَيوسِ
ضَيِّعَةٍ ذِراعُهُ يَبوسِ

الأزهري، التهذيب. تح: درويش، ج: 3، ص: 391.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ مالك (672هـ=1273م) :

"والـحُيُسُ: جَمْعُ حَيوسٍ: وهُوَ الَّذي يَحيسُ كَثيرًا؛ أَيْ: يَصْنَعُ الـحَيْسَ، وهُوَ تَـمْرٌ وأقِطٌ يُعْجَنُ بالسَّمْنِ".

ابن مالك، إكمال الإعلام بتثليث الكلام. تح: الغامدي، ج: 1، ص: 169.— المصدر
5.2معنى من معاني «حَيوس»

وــــــــــ: القَتّالُ. (لُغَةٌ فِي حَؤوس، عَنِ ابْنِ الأَعْرابِيِّ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأعْرابيّ (ت: 231هـ=845م):

"ورَجُلٌ حَيوسٌ: قَتّالٌ. لُغَةٌ في حَؤوسٍ" (وهذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 3، ص: 424.— المصدر
