01 —

داخَ

فعل
1.1معنى من معاني «داخَ»

* داخَ فُلانٌ وَغَيْرُهُ يَدوخُ دَوْخًا: ذَلَّ وَخَضَعَ، فَهُوَ دائِخٌ. (ج) دوَّخٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الْأَعْشى الكَبيرُ (ت: 12ق هـ=610م) في قَصيدَةٍ هَجا فيها الحارِثَ بْنَ وَعْلَةَ:

وَبيضٌ كَأَمْثالِ العَقيقِ صَوارِمٌ
تُصانُ لِيَومِ الدَّوْخِ فينا وَتُخْشَبُ

ديوان الأعشى، تح: محمود الرضواني، ج: 2، ص: 49.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ وفْدُ ثَقيفٍ (ت: 9ه=630م) يَتَحَدّثونَ عَنِ النَّبِيِّ ﷺ:

"وَاللَّهِ ما لَنا بِهِ طاقَةٌ، وَقَدْ ‌داخَ ‌لَهُ ‌العَرَبُ كُلُّها، فَارْجِعوا إِلَيْهِ، فَأَعْطوهُ ما سَأَلَ، وَصالِحوهُ عَلَيْهِ".

ابن القيم، زاد المعاد، تح: حمدي عبد المجيد، ج: 3، ص: 523.— المصدر
”وقالَ العَجّاجُ (ت: 90هـ=709م) يَفْخَرُ بِقَوْمِهِ ويَهْجو أَعْداءَهُمْ:

لِفَحْلِنَا إِنْ سَرَّهُ التَّنَوُّخُ

قَاعَ، وَإِنْ يُتْرَكْ فَشَوْلٌ دُوَّخُ

ديوان العجاج، تح: عزة حسن، ص: 402.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (ت: 170هـ=787م) مِنْ شَواهِدِ الخَليلِ:

حَتّى يَدوخَ لَنا مَنْ كانَ عادانا

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 295.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَبْدُ اللهِ اليافِعِيُّ (ت: 768هـ=1367م) يَفْخَرُ بِالـمَذْهَبِ الأَشْعَرِيِّ ويُعَرِّضُ بِالـمُعْتَزِلَةِ:

إِذا ما رَمى مِنْ مَنْجَنيقِ اعْتِزالِهِمْ
بِجُلْمودِ عَقْلٍ لِلتّجادُلِ دائِخُ

عبد الله اليافعي، مرهم العلل المعضلة في الرد عَلى أئمة المعتزلة، تح: محمود محمد نصار، ص: 42.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَجْمَعُ القاهِرَةِ (1427هـ=2006م، تاريخُ الطَّبْعِ):

"وَيُقالُ: داخَ فُلانٌ لِفُلانٍ".

مجمع القاهرة، المعجم الكبير، ج: 7، ص: 646.— المصدر
1.2معنى من معاني «داخَ»

وــــــــــ فُلانٌ: أُصيبَ بِالدُّوارِ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ نِعْمَةَ النّابُلْسِيُّ (ت: 697ه=1297م) يُفَسِّرُ حُلْمًا:

"وَرُبَّما دَلَّ الصُّعودُ عَلى الجَبَلِ وَالأَماكِنِ العالِيَةِ عَلى أَمْراضٍ تَحْدُثُ بِالرَّأْسِ أَوْ بِالعَيْنِ، خُصوصًا إِنْ كانَ واقِفًا عَلى مَكانٍ ضَيِّقٍ؛ لِأَنَّ الصّاعِدَ عَلى مِثْلِ ذَلِكَ ‌يَدوخُ رَأْسُهُ، وَيُخَيَّلُ إِلَيْهِ أَنَّ الأَرْضَ تَدورُ بِهِ ...".

شهاب الدين النويري، نهاية الأرب في فنون الأدب، دار الكتب والوثائق القومية، ج: 33، ص: 290.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ النّعيميُّ (1417هـ=1997م، تاريخُ الطَّبْعِ) يُتَرْجِمُ عَنْ رينْهارْت دوزي:

"قامَتْ عَلَيْهِ نَفْسُهُ: أَصابَهُ ‌دُوارٌ، ‌داخَ".

دوزي، تكملة المعاجم العربية، تر: النعيمي، ج: 8، ص: 417.— المصدر
”وَقالَ مُحَمَّدُ الـمَنْسِي قَنْديل (1433هـ=2012م، تاريخُ الطَّبْعِ) يَحْكي عَلى لِسانِ أَحَدِ شُخوصِ رِوايَتِهِ:

"وَفي اليَوْمِ التّالي كُنْتُ دائِخَةً، وَأَنا أَسيرُ فَوْقَ مُنْحَدَرٍ مِنَ الأَرْضِ نَحْوَ مَقابِرِ البَرِّ الغَرْبِيِّ، تَعَثَّرَتْ خُطُواتي، وَأَوْشَكْتُ عَلى السُّقوطِ، لَوْلا أَنَّ يَدًا قَوِيَّةً امْتَدَّتْ وَأَمْسَكَتْ بي".

محمد المنسي قنديل، أنا عشقت، دار الشروق، ص: 228.— المصدر
1.3معنى من معاني «داخَ»

وــــــــــ اللَّيْلُ: أَظْلَمَ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفَيْروزْآبادِيُّ (ت: 817هـ=1414م):

"وَلَيْلٌ دائِخٌ: مُظْلِمٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في العَصْرِ الحَديثِ)

الفيروزآبادي، القاموس المحيط، تح: العرقسوسي، ص: 250.— المصدر
1.4معنى من معاني «داخَ»

وــــــــــ البِلادَ: قَهَرَها وَاسْتَوْلى عَلَيْها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَمْرُو بْنُ النُّعْمانِ الخَولانِيُّ (ت: 320هـ=932م):

أَبُونا الَّذي داخَ العِراقَ بِخَيْلِهِ
وَدانَتْ لَهُ ما بَيْنَ حِمْصٍ وَفارِسِ

الهمذاني، كتاب الإكليل، تح: أنستاس ماري الكرملي، ج: 8، ص: 97.— المصدر
”وقالَ الجَوهَريُّ (ت: 393هـ= 1003م):

"داخَ البلادَ يَدوخها: قَهَرَها واسْتَوْلى عَلَى أَهْلِها". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح، تح: عطار، ج: 1، ص: 421.— المصدر
1.5معنى من معاني «داخَ»

وــــــــــ العَدُوَّ: أَذَلَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القُوطِيَّةِ (ت: 367هـ=977م):

"وَداخَ ... العَدُوَّ: أَذَلَّهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال، تح: علي فودة، ج: 1، ص: 278.— المصدر
1.6معنى من معاني «داخَ»

وــــــــــ الإِبِلَ: ساقَها، أو جَمَعَها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطَّاعِ (ت: 515هـ=1122م):

"وَداخ الإِبِلَ دَوْخًا: ساقَها، وَأَيْضًا جَمَعَها". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القطاع، الأفعال، ج: 3، ص: 375.— المصدر
02 —

أَداخَ

فعل
2.1معنى من معاني «أَداخَ»

* أَداخَفُلانٌ البِلادَ: قَهَرَها وَاسْتَوْلى عَلَيْها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ قُسُّ بْنُ ساعِدَةَ الإِيادِيُّ (ت: 23ق هـ=600م) يَعِظُ:

"أَيْنَ الصَّعْبُ ذو القَرْنَيْن؟! جَمَعَ الثَّقَلَيْن، وَأَداخَ الخافِقَيْن، وَعَمَّرَ أَلْفَيْن، لَمْ تَكُنِ الدُّنْيا عِنْدَهُ إِلَّا كَلَحْظَةِ عَيْن، مَنْ لَمْ يَتَّعِظِ اتَّعَظَ بِها".

عبد الملك بن هشام الحميري، التيجان في ملوك حمير، ص: 126.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ عَذاري الـمَرّاكُشِيُّ (ت: 695 هـ=1296م) يَتَحَدَّثُ عَمّا فَعَلَهُ العَبّاسُ بْنُ الفَضْلِ في سَنَةِ 237ه:

"وَفيها أَغْزى العَبَّاسُ بِصِقِلِيَّةَ أَرْضَ الرُّومِ، فَغَنِمَ غَنائِمَ عَظيمَةً، وَسَبى سَبْيًا كَثيرًا، وَأَداخَ بِلادَهُمْ".

ابن عذاري المراكشي، البيان المغرب، تح: بشار عواد معروف وآخر، ج: 1، ص: 152.— المصدر
2.2معنى من معاني «أَداخَ»

وــــــــــ العَدُوَّ وَغَيْرَهُ: أَذَلَّهُ وَأَخْضَعَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ وَفْدُ ثَقيفٍ (9ه=630م) يُخَوِّفونَ قَوْمَهُمْ من قِتالِ رَسولِ اللهِ ﷺ:

"أَتَيْنا رَجُلًا فَظًّا غَليظًا يَأْخُذُ مِنْ أَمْرِهِ ما يَشاءُ، قَدْ ظَهَرَ بِالسَّيْفِ، وَأَداخَ العَرَبَ، وَدانَ لَهُ النّاسُ، فَعَرَضَ عَلَيْنا أُمورًا شِدَادًا".

البيهقي، دلائل النّبوة، تح: عبد المعطي قلعجي، ج: 5، ص: 302.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ (ت: 515ه=1122م):

"داخَ البِلادَ دَوْخًا: وَطِئَها، وَالعَدُوَّ: أذَلَّهُ، وَأَداخَهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القطاع، الأفعال، ج: 1، ص: 373.— المصدر
03 —

دَوَّخَ

فعل
3.1معنى من معاني «دَوَّخَ»

* دَوَّخَ فُلانٌ فُلانًا وَغَيْرَهُ: ذَلَّلَهُ وَأَخْضَعَهُ. (وانظر: د ي خ).

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عُبَيْدُ بْنُ عَبْدِ العُزّى السَّلامِيُّ (ت: 84ق ه=540م):

إِذا ما نَزَلنا بَلْدَةً دُوِّخَتْ لَنا
فَكُنّا عَلى أَرْبابِها بِالـمُخَيَّرِ

محمد بن المبارك، منتهى الطلب، تح: محمد نبيل طريفي، ج: 8، ص: 292.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتِ الخَنْساءُ (ت: 24هـ=645م) تَرْثي صَخْرًا، وَتَذْكُرُ الحَرْبَ:

وَكانَ أَبو حَسّانَ صَخْرٌ سَما لَها
فَدَوَّخَها بِالخَيْلِ حَتَّى أَقَرَّتِ

ديوان الخنساء بشرح ثعلب، تح: أنور أبو سويلم، ص: 195.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤْبَةُ بْنُ العَجّاجِ (ت: 145هـ=762م) يَتَحَدَّثُ عَنْ إِبِلِهِ:

وَالبُزْلَ قَدْ دَوَّخْتُها بِالكَبْحِ

خَواضِعًا مِنْ صادِماتِ الرَّنْحِ

ديوان رؤبة بن العجاج، تح: وليم البروسيي، ج: 1، ص: 37.— المصدر
3.2معنى من معاني «دَوَّخَ»

وــــــــــ البِلادَ وَالأَعْداءَ: قَهَرَها وَأَخْضَعَها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ النّابِغَةُ (ت: 18ق هـ=602م) يَمْدَحُ:

فَدَوَّخْتَ البِلادَ بِكُلِّ قَصْرٍ
تَجَلَّلَ خَنْدَقًا مِنْهُ وَحامِ

ديوان النّابغة الذبيانيّ، تح: أبو الفضل إِبراهيم، ص: 256.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ زَيْدُ الخَيْرِ (ت: 9ه=630م) يُعَرِّفُ عُمَرَ بْنَ الخَطّابِ بطيِّئٍ ومُلوكِها:

"وَمِنّا الأَسَدُ الرَّهيصُ؛ سَيِّدُ بَني جَديلَة، وَمُدَوِّخُ كُلِّ قَبِيلَة، قاتِلُ عَنْتَرةَ فَارِسِ بَني عَبْس، وَمُكَشِّفُ كُلِّ لَبْس".

أبو الفرج الأصفهاني، كتاب الأغاني، تح: إحسان عباس وآخرَينِ، ج: 17، ص: 182.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الـمُقَفَّعِ (ت: 142هـ= 759م):

"لَمْ يَخْرُجِ الـمُلْكُ مِنْ قَوْمٍ إِلّا بَقِيَتْ فيهِمْ بَقِيَّةٌ يَتَوَثَّبونَ بِها، ثُمَّ كانَ ذَلِكَ التَّوَثُّبُ هُوَ سَبَبَ اسْتِئْصالِهِمْ وَتَدْويخِهِمْ".

أحمد زكي صفوت، جمهرة رسائل العرب، المكتبة العلمية، ج: 3، ص: 42.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الأَبّارِ (ت: 658هـ= 1260م) يَمْدَحُ:

وَدَوَّخَ أَصْقاعَ الشِّقاقِ وَسَكْنَها
بِصَيَّابَةٍ قُعْسٍ وَعَسَّالَةٍ قُعْصِ

ديوان ابن الأبار، تح: الهرّاس، ص: 351.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشَّوْكانِيُّ (ت: 1250هـ= 1834م) يتحدث عَنْ أَحْمَدَ بْنِ الحَسَنِ:

"وَكانَ مُجاهِدًا، وَيَبْعَثُ بِهِ الإِمامُ إِلى الأَقْطارِ النّائِيَةِ لِلْغَزْوِ، فَيَظْفَرُ وَيَعودُ، وَقَدْ دَوَّخَ ما بَعَثَهُ إِلَيْهِ".

الشوكاني، البدر الطالع، ج: 1، ص: 148.— المصدر
3.3معنى من معاني «دَوَّخَ»

وــــــــــ البِلادَ: جالَ فيها فَعَرَفَها، وَكَشَفَ طُرُقَها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الرّاعي النُّمَيْرِيُّ (ت: 90ه=708م):

فَها أَنا قَدْ طَوَّفْتُ شَرْقًا وَمَغْرِبًا
وَأُبْتُ وَقَدْ دَوَّخْتُ كُلَّ مَكانِ

يَجوزُ في الإِرْبَةِ حِذْقَ الحاذِقِ

مُدَوَّخٍ لِقُلَلِ الشَّواهِقِ

الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ (ت: 671هـ=1273م):

"وَيُرْوى أَنَّ سَبَأً دَوَّخَ الأَرْضَ بِأَجْنادِهِ شَرْقًا وَغَرْبًا، وَسَهْلًا وَجَبَلًا، وَقَتَلَ سادَةً، وَسَبى - وَبِهِ سُمِّيَ سَبَأً - وَدانَ لَهُ الخَلْقُ، فَلَمَّا رَأَى ذَلِكَ انْفَرَدَ عَنْ أَصْحابِهِ ثَلاثَةَ أَيَّامٍ".

القرطبي، الجامع لأحكام القرآن، تح: عبد الله التركي، ج: 13، ص: 83.— المصدر
3.4معنى من معاني «دَوَّخَ»

وــــــــــ الوَجَعُ وَغَيْرُهُ رَأْسَهُ: أَصابَهُ بِالدُّوارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 209هـ=824م):

"وَإنَّما سُمِّيَتِ الخَمْرُ قَرْقَفًا؛ لِأَنَّ مَنْ شَرِبَها قَرْقَفَتْهُ، فَأَدارَتْهُ وَأَسْكَرَتهُ، فَهُوَ مُدَوَّخٌ مِنَ السُّكْرِ".

شرح نقائض جرير والفرزدق، تح: محمد حور وآخر، ج: 2 ، ص: 711، 712.— المصدر
3.5معنى من معاني «دَوَّخَ»

وــــــــــ فُلانٌ الباطِلَ: فَنَّدَهُ وَأَضْعَفَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو تَمَّامٍ (ت: 231ه=846م) يَمْدَحُ:

بَعْدَما أَصْلَتَ الوُشاةُ سُيوفًا
قَطَعَتْ فِيَّ، وَهْيَ غَيْرُ حِدَادِ

مِنْ أَحاديثَ حِينَ دَوَّخْتَها بالرْ
رَأْيِ كانَتْ ضَعيفَةَ الإِسْنَادِ

الأعلم الشّنتمريّ، شرح ديوان أبي تمّام، تح: إِبراهيم نادن، ج: 2، ص: 146.— المصدر
3.6معنى من معاني «دَوَّخَ»

وــــــــــ الحَرُّ فُلانًا: أَضْعَفَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م):

"وَدَوَّخَني الحَرُّ: أَضْعَفَني". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، أساس البلاغة، تح: عيون السود، ج: 1، ص: 301.— المصدر
04 —

تَدَوَّخَ

فعل
4.1معنى من معاني «تَدَوَّخَ»

* تَدَوَّخَ فُلانٌ: أُصيبَ بِالدُّوارِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجُوَيْنِيُّ (ت: 478هـ=1085م) يُعَلِّقُ عَلى بَعْضِ أَحْكامِ الحَيْضِ لِلْمَرْأَةِ النّاسِيَةِ:

"وَهَذا لا يُدْرِكُهُ إِلّا مُوَفَّقٌ سَديدُ الفِكْرِ، وَالبَليدُ لا يَزْدادُ بِالإطْنابِ في البَيانِ إِلّا دَهْشَةً وَتَدَوُّخًا".

الجوينيّ، نهاية المطلب في دراية المذهب، تح: عبد العظيم محمود الديب، ج: 1، ص: 398.— المصدر
05 —

دَوْخَة

فعل
5.1معنى من معاني «دَوْخَة»

* الدَّوْخَةُ: الدُّوَارُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«اخْتَضِبوا بِالحِنّاءِ؛ فَإِنَّهُ طَيِّبُ الرّيحِ يُسَكِّنُ ‌الدَّوْخَةَ».

مسند أبي يعلي، تح: حسين أسد، ج: 6، ص: 305.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عِمادُ الدّينِ الأصفهانيُّ (ت: 597هـ=1201م) يَحْكي:

"فَما تَوَسَّطَ نَهارُنا، إِلّا وَقَدِ انْكَسَرَ بِالخَمْرِ خمارُنا، وَنَشَأَتْ نَشْوَةُ النُّفوس، وَتَلاشَتْ ‌الدَّوْخَةُ مِنَ الرُّؤوس".

عماد الدين الأصفهانيُّ، خريدة القصر، تح: شكري فيصل، ج: 1، ص: 366.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ النُّوَيْرِيُّ (ت: 733ه=1333م) يَصِفُ مَرَضَ أَحَدِ الأُمَراءِ:

"فَجَمَعَ جَماعَةً مِنَ الأَطِبّاءِ لِذلِكَ، وَقالوا: إِنَّهُ حَدَثَ مِنْ ريحٍ مُجاوِرٍ لِلْكَبِدِ، وَعولِجَ مِنْهُ، وَبَرِئَ، ثُمَّ عاوَدَهُ، وَحَصَلَ لَهُ دَوْخَةٌ فى دِماغِهِ".

النويري، نهاية الأرب، ج: 33، ص: 290.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مُحَمَّدُ بْنُ أَحْمَدَ الدُّسوقِيُّ (ت: 1230هـ=1815م):

"(قَوْلُهُ ضَرَرًا)؛ أَيْ: مِنْ إِغْماءٍ، أَوْ حُدوثِ مَرَضٍ، أَوْ زِيادَتِهِ، أَوْ تَأَخُّرِ بُرْءٍ، أَوْ حُصولِ ‌دَوْخَةٍ".
