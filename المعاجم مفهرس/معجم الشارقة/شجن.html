النَّظائِرُ السّامِيَّةُ:

01 —

شَجَنَ

فعل
1.1معنى من معاني «شَجَنَ»

* شَجَنَتِ (بِفَتْحِ الجِيمِ) الحَمامَةُ تَشْجُنُ (بِضَمِّ الجِيمِ) شُجونًا: نَاحَتْ وتَحَزَّنَتْ بِتَرْدِيدِ صَوْتِهَا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والحَمامَةُ تَشْجُنُ شُجونًا إِذا ناحَتْ وتَحَزَّنَتْ".

الخليل، العين، تح: المخزومي، ج: 6، ص: 36.— المصدر
”وقالَ ابْنُ عَبْدِ رَبِّهِ (ت: 328هـ=940م) يَصِفُ حَالَهُ:

وإِنَّ ارْتياحي مِنْ بُكاءِ حَمَامَةٍ
كَذي شَجَنٍ داويْتَهُ بِشُجُونٍ

كأَنَّ حَمامَ الأَيْكِ حينَ تَجاوبَتْ
حَزينٌ بَكى من رَحْمِةٍ لِحَزينِ

ديوان ابن عبد ربّه، تح: محمد الداية، ص: 165.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الخَطيبِ (ت: 776هـ=1374م) يَصِفُ حالَهُ:

"فَراجَعَني التَّذكَارُ والتَّهمامُ، وطَاوَعَني شُجونًا يَتَعاطاهُ الحَمامُ".

ابن الخطيب، الإحاطة، تح: يوسف طويل، ج: 4، ص: 251.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَتْ وِصال تَقَّة (1438هـ=2017م) تاريخُ الطَّبْعِ، تُخاطِبُ القارِئَ:

"أَتَسْمَعُ نَقيقَ ذَلِكَ الضِّفْدَعِ في الغَديرِ، وقَدْ حَمَلَ سُكونُ الفَجْرِ تَسبيحَهُ، أَسَمِعْتَ أَزيزَ النَّحْلَةِ، وهَديلَ الحَمَامَةِ فَوْقَ السَّطحِ تَشْجُنُ".

وصال تقة، فيء الرّوح زخات حبر وياسمين، ص: 10.— المصدر
1.2معنى من معاني «شَجَنَ»

وــــــــــ فُلانٌ بَيْنَ الشَّيئَيْنِ: خَلَطَ بَينَهُما. (مو)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ سَلَمَةُ بْنُ مُسْلِمٍ الصُّحارِيُّ (ت: 450هـ=1058م):

"يُقالُ شَجَنْتَ بَيْني وبَيْنَهُ: أَيْ خَلَطْتَ".

سلمة بن مسلم الصحاري، كتاب الإبانة، تح: عبد الكريم خليفة، وآخرين، ج: 3، ص: 288.— المصدر
1.3معنى من معاني «شَجَنَ»

وــــــــــالحَاجَةُ فُلانًا: حَبَسَتْه وعَوَّقَتْهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ بْنُ أَبانَ (ت: 90ق.هـ=534م) يُؤَكِّدُ ابْتِعادَ أَبْناءِ عَصْرِهِ مِنَ الشُّعَراءِ عَنْ إِتْيانِ مَغامِضِ الكَلامِ في القَصِيدَةِ:

"أَدرَكنا الأَوَّلِينَ من العَرَبِ لا يَحفِلونَ بِمَجيءِ ذَلِكَ، ولا أَدري ما شَجَنَ عَنهُ، فَأمّا أَنا وطَبَقَتي فَكُنّا نَمُرُّ في البَيتِ حَتّى نَأْتيَ آخِرَهُ، فإِذا فَنيَ وقارَبَ، تَبَيَّنَ أَمْرُهُ لِلسّامِعِ".

أبو العلاء المعرّي، رسالة الغفران، تح: كامل الكيلاني، ص: 327.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصارِيُّ (ت: 215هـ=830م):

"وقد شَجَنَتْني الحاجَةُ حَيْثُ كانَتْ تَشْجُنُنِي شَجْنًا إِذا حَبَسَتْكَ".

الأزهري، تهذيب اللغة، تح: علي حسن هلالي، ج: 10، ص: 538.— المصدر
”وقالَ المَعَرّيُّ (ت: 449هـ=1057م) يَذْكُرُ كَيْفَ حالَتْ شَواغِلُ الحَياةِ بَيْنَهُ وبَيْنَ العِلْمِ:

"كُنتُ في غَيسانَ الشَّبِيبَةِ أَوَدُّ أَنَّني مِنْ أَهْلِ العِلْمِ فشَجَنَتني عَنْهُ شَواجِنُ غادَرَتني مِثْلَ الكُرَةِ وهِيَ المَحاجِنُ".

السيوطي، الأشباه والنظائر، وضع حواشيه: غريد الشيخ، ج: 1، ص: 193.— المصدر
1.4معنى من معاني «شَجَنَ»

وــــــــــ الأَمْرُ فُلانًا: أَحْزَنَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ناسٌ مِنَ المُنافِقينَ (11هـ=632م) يُخاطِبونَ النَّبِيَّ ﷺ:

"فَذَرْنَا نَبكي شُجونًا، ثُمَّ نَأْتَمِرُ لِأَمْرِكَ".

تفسير مقاتل، تح: عبد الله شحاتة، ج: 4، ص: 276.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142هـ=760م) على لِسانِ الفَرَسِ يُخاطِبُ البَغْلَ:

"إِنَّكَ لَعَالِمٌ بِالعِرَابِ، فَمِنْ أَيْنَ لَكَ ذَلِكَ، والأَيَّامُ لَكَ شَاجِنَةٌ، ونُوَبُهَا عِنْدَكَ راجِنَة؟".

أبو العلاء المعرّي، رسالة الصاهل والشاحج، ص: 98.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الحاجِّ النُّمَيْرِيُّ (ت: 768هـ=1367م) يَذْكُرُ أَهْلَ بِجايةَ:

"فَأَشْبَهوا الخَنْساءَ لا شُجونًا وذِكْرًا، لَكِنْ في كَونِهِمْ يُذَكِّرُهُمْ طُلُوعُ الفَجْرِ وغُروبُهَا صَخْرًا".

ابن الحاج النميري، فيض العباب، تح: محمد بن شقرون، ص: 278.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ قاسِمُ الحلّاقُ الدِّمَشْقِيُّ (ت: 1284هـ=1867م):

يا طَيِّبًا فاقَتِ العُلْيا أَماكِنَهُ
والطِّيبُ مِنْ طَيْبَةَ الفَيحا مَعادِنُهُ

إِنّي اسْتَجَرْتُ وقَلْبي هاجَ شاجِنُهُ

محمد جميل الشطي، أعيان دمشق، تح: عز الدين النجار، وآخر، ص: 228.— المصدر
1.5معنى من معاني «شَجَنَ»

وــــــــــ الأَمْرُ، أَوِ الشَّيْءُ فُلانًا شَجْنًا، وشُجونًا: أَهَمَّهُ وشَغَلَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"الشَّجَنُ: الهَمُّ والحُزْنُ، شَجَنَني الأَمْرُ فَشَجِنْتُ".

ابن عباد، المحيط في اللغة، تح: محمد آل ياسين، ج: 6، ص: 431.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ عَلِيُّ بْنُ الحَسَنِ الخَزْرَجِيُّ (689هـ=1291م) يَروي فَاجِعَتَهُ في الشَّيْخِ مَحَمَّد عَبدِ الوَهّاب حينَ عَادَتْ بَغلةُ الشَّيخِ ولَم يَعُدْ مَعَهَا، مِمَّا اضطَرّهُ إِلى زِيارَةِ أَهلِهِ لِلاِطمِئْنانِ عَلَيهِ:

"وسِرتُ سَيرًا حَثيثًا، فَوَصَلتُ إلى بَلَدِهِ مُسرِعًا، فَرَأَيتُ أَهلَهُ في حُزنٍ شَدِيدٍ فَزادَني ذَلِكَ شَجَنًا".

الخزرجي، طراز أعلام الزمن، تح: الحبشي، ج: 2، ص: 811.— المصدر
02 —

شَجِنَ

فعل
2.1معنى من معاني «شَجِنَ»

* شجِنَ (بِكَسْرِ الجِيمِ) فُلانٌ وغَيْرُهُ يَشْجَنُ (بِفَتْحِ الجِيمِ) شَجَنًا: حَزِنَ، فَهُوَ شَجِنٌ، وهِيَ بِتاءٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80ق.هـ=544م) يَصِفُ حالَهُ:

خَيالٌ هاجَ لي شَجَنا
فَبِتُّ مُكابِدًا حَزَنا

ديوان امرئ القيس وملحقاته، تح: أنور عليان، وآخرين، ص: 754.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ فاطِمَةُ الزَّهْراءُ (11هـ=632م) تَبكي أَباها مُحَمَّدًا رَسولَ اللهِ ﷺ:

فَإِذا بَكَتْ قُمْرِيَّةٌ في لَيْلِها
شَجَنًا على غُصْنٍ بَكَيتُ صباحِيا

ديوان فاطمة الزهراء، تح: محمد عبد الرحيم، ص: 131.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170هـ=787م):

"والأَشْجانُ: الأَحْزانُ، جَمْعُ شَجَنٍ، والفِعْلُ مِنْهُ شَجِنْتُ؛ أَيْ: صارَ الشَّجَنُ فيَّ".

الخليل، العين، تح: المخزومي، ج: 6، ص: 36.— المصدر

"فَخَرَجَتْ بِجَمِيعِ مَالِها وحالِها ونِسائِها ورِجَالِها وأَسفَاطِها وأَعدالِها والصَّنادِيقِ بِأقفالِها، وتَبِعَها مَنْ لَمْ يَكُنْ مِنْ أَتبَاعِها فَراحَتْ فَرحى وإِنْ كانَت مِنْ شَجَنِها قَرْحَى".

أبو شامة، الروضتين ، تح: الزيبق، ج: 3، ص: 343.— المصدر

إِنَّ لِي في الحُدُوجِ لَوْ كُنْتَ تَدْرِي

شَجُنَ

فعل
3.1معنى من معاني «شَجُنَ»

* شَجُنَ (بِضَمِّ الجِيمِ) فُلانٌ يَشْجُنُ (بِضَمِّ الجِيمِ) شَجَنًا وشُجونًا: حَزِنَ، فهو شاجِنٌ وشَجينٌ.

هيَّجْتَ لي شَجَنًا لَعَمْرُكَ شاجِنا

04 —

أَشْجَنَ

فعل
4.1معنى من معاني «أَشْجَنَ»

* أَشْجَنَ الشَّجَرُ: تَشابَكَتْ فُروعُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو حاتِمٍ السِّجِسْتانِيُّ (ت: 255هـ=868م):

"ويُقالُ: أَشْجَنَ الكَرْمُ، وذَلِكَ أَنَّ الشُّجْنَةَ، وقَدْ يُقَالُ بِكَسْرِ الشِّينِ، وهي الشُّعْبَةُ مِنَ العُنْقودِ تُدْرَكُ كُلُّهَا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيّةِ العُصورِ)

أبو علي القالي، البارع في اللغة، تح: هشام الطعان، ص: 613.— المصدر
4.2معنى من معاني «أَشْجَنَ»

وــــــــــ الأَرْضُ: كَثُرَتْ فيها الأَودِيَةُ. (مو)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"يُقالُ: أَشْجَنَتِ الأَرْضُ، إذا كَثُرَتِ الشَّواجِنُ فِيها، وهي الأَودِيَةُ".

4.3معنى من معاني «أَشْجَنَ»

وــــــــــ الأَمْرُ أَوِ الشَّيءُ فُلانًا: أَحْزَنَهُ.
