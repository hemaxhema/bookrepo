النظائر الساميّة

اِحتَفَظَتِ السَّاميَّاتُ بِهذا الجَذرِ مِنَ العَربيّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذرُ (أ ر ن):

في الأكّدية؛ ومِنْهُ الاسْمُ: 𒀀 𒊏 𒉡 (أران) arān بِمَعْنى "نَعْش" أو "تابوت"، وهُوَ يتَّفقُ مَعَ الاسْمِ العَرَبيِّ (إران) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى "تابوت": "𒉌𒌓 𒀀 𒊏 𒉡 𒆠 𒋫 𒊍 𒇷 𒅋 𒋾 𒋗 𒆍 𒋗 𒀝 𒉡 𒊌 𒈠"، أيْ "أغلقتُ بابَ التّابوتِ؛ مَكان راحَتِهِ".

(AHw, p. 65)

وفي الأوجاريتيّةِ؛ ومِنْهُ الاسْمُ: arn (أرن) ʼarn بِمَعْنى "نَعْش" أو "تابوت"، وهُوَ يتَّفقُ مَعَ الاسْمِ العَرَبيِّ (إران)؛ ومِنْ مَجيئِهِ بِمَعْنى "تابوت": "arn.w mznm"، أي " تابوتٌ وموازينُ"

(DULAT, p. 100)

وفي الفينيقيّةِ؛ ومِنْهُ الاسْمُ: NRA (أرن) ʼrn بِمَعْنى "تابوت"، وهُوَ يتَّفقُ مَعَ الاسْمِ العَرَبيِّ (إران) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى "تابوت" ما وَرَدَ في نَقْشِ أحيرامَMRHA . Nb . LOb xa . LOP . z . NRA""، أَيِ "التّابوتُ الذي صَنَعَهُ إت بعل بنُ أحيرامَ".

(DNWSI: p 109)

وفي الآراميّةِ؛ ومِنْهُ الاسْمُ אֲרוֹנָא (أر۫ۈنا) ʼarōnā بِمَعْنى "تابوت"، وهُوَ يتَّفقُ مَعَ الاسْمِ العَرَبيِّ (إران) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى "تابوت" ما وَرَدَ في تَرجومِ أونكلوسَ "כָּפֻרְתָּא עַל אֲרוֹנָא"، أي "الغِطاءُ على التّابوتِ".

(ת"א، שמ'כה,כ).

وفي العِبْريّةِ؛ ومِنْهُ الاسْمُ: אָרוֹן (آر۫ۈن) ʼārōn بِمَعْنى "تابوت"، وهُوَ يُوافِقُ الاسْمَ العَربيَّ (إران) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى "إران" ما وَرَدَ في سِفْرِ التّكْوينِ: "וַיּישֶֹם בָּאָרֹון בְּמִצְרַיִם"، أي "ووَضَعوهُ في الإرانِ بِمِصْرَ".

(בר' 50: 26)

في السُّريانيّةِ؛ ومِنْهُ الاسْمُ: ܐܳܪܽܘܿܢܳܐ (آر۫ۈنا) ʼārōnā بِمَعْنى "تابوت"، وهُوَ يتَّفقُ مَعَ الاسْمِ العَرَبيِّ (إران) مَبْنًى ومَعْنًى، ومِنْ مَجيئِهِ بِمَعْنى "تابوت" ما وَرَدَ في سِفْرِ صُموئيلَ الأوّلِ: "ܘܰܚܙܰܐܗܝ ܠܳܐܪܽܘܢܳܐ"، أيْ "ورَأَوا التّابوتَ".

(ܐ.ܫܡܽܘܐܶܝܠ-ܘ: 13)

01 —

أَرَنَ

فعل
1.1معنى من معاني «أَرَنَ»

* أَرَنَ (بِفَتْحِ الرّاءِ) الشَّيْءُ يَأْرِنُ (بِكَسْرِ الرّاءِ) أُرونًا: اِنْكَمَشَ ودَنا بَعْضُهُ إلى بَعْضٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعيُّ (ت: 216ه=831م):

"أَرَنَ يَأْرِنُ أُرونًا تَقَبَّضَ ودَنا بَعْضُهُ إلى بَعْضٍ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

النحاس، صناعة الكُتّاب. تح: بدر أحمد، ج: 1، ص: 85.— المصدر
1.2معنى من معاني «أَرَنَ»

وــــــــــ فُلانٌ فُلانًا: كَدَمَهُ وعَضَّهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"وقيلَ: أَرَنَهُ أَيْ: كَدَمَهُ وعَضَّهُ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 253.— المصدر
02 —

أَرِنَ

فعل
2.1معنى من معاني «أَرِنَ»

* أَرِنَ (بِكَسْرِ الرّاءِ) الحِصانُ يَأْرَنُ (بِفَتْحِ الرّاءِ) أَرَنًا وإرانًا وأَرينًا أَيْ: نَشِطَ فَهُوَ أَرِنٌ وأَرونٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ مَطْرودُ بْنُ كَعْبٍ الخُزاعيُّ (ت: 66ق هـ =560م):

كَمْ وهَبوا مِنْ طِمِرٍّ سابِحٍ أَرِنٍ
ومِنْ طِمِرةِ نَهْبٍ في طِمِرَّاتِ

الكلاعي، الاكتفا في مغازي المصطفى والثلاثة الخلفا. تح: محمد عطا، ج: 1، ص: 99.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الأَعْشى الكَبيرُ (ت: 7ه=628م) يَصِفُ فَرَسَهُ بِالنَّشاطِ:

تَراهُ إذا ما غَدا صَحْبُهُ
بِهِ جانِبَيْهِ كَشاةِ الأَرَنْ

ديوان الأعشى الكبير. تح: محمد حسين، ص: 21.— المصدر
03 —

آرَنَ

فعل
3.1معنى من معاني «آرَنَ»

* آرَنَ الثَّوْرُ البَقَرةَ مُؤارَنةً وإرانًا: طَلَبَها.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَديُّ بْنُ زَيْدٍ العِباديُّ (ت: 35ق هـ=587م) يَصِفُ مَرَحَ ثيرانٍ:

وإرانُ الثِّيرانِ حَوْلَ نِعاجٍ
مُطْفِلاتٍ يَحْمينَ بِالأَوْراقِ

ديوان عدي بن زيد العبادي، جمع: محمد الهاشمي، ص: 117.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الطِّرمّاح (ت: 125ه=742م) يَصِفُ ناقَتَهُ:

تُطارِدُ بِالقيِّ السَّرابَ كَما قَلا
طَريدَتَهُ ثَورُ الصَّريمِ المُؤارِنِ

ديوان الطرماح. تح: عزة حسن، ص: 274.— المصدر
3.2معنى من معاني «آرَنَ»

وــــــــــ فُلانٌ فُلانًا مُؤارَنةً: باراهُ في السَّيْرِ وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّاد (ت: 385هـ = 995م):

"والمُؤارَنةُ: المُباراةُ في السَّيْرِ وغَيْرِهِ، آرَنَهُ يُؤارِنُهُ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 254.— المصدر
04 —

إران

اسم
4.1معنى من معاني «إران»

* الإرانُ: سَريرُ المَيِّتِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ امْرُؤُ القَيْسِ (ت: 80 ق هـ =543م) يُشَبِّهُ صَلابةَ ناقَتِهِ وقُوَّتَها بِأَلْواحِ التّابوتِ:

وعَنْسٍ كَأَلْواحِ الإرانِ نَسَأْتُها
على لاحِبٍ كالبُرْدِ ذي الحِبَراتِ

ديوان امرئ القيس بشرح السكري. تح: عليان، ص: 588.— المصدر
4.2معنى من معاني «إران»

وــــــــــ: مَأْوى الدّابَّةِ الوَحْشيةِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَديُّ بْنُ زَيْدٍ العِباديُّ (ت: 35ق هـ=587م) يَذْكُرُ مُحاوَلَتَهُ اصْطِيادَ بَقَرةٍ وَحْشيَّةٍ في كِناسِها:

طَلَبْتُ بِها شاةَ الإرانِ غُدَيَّةً
مَرابِيَ سُفْعًا قَدْ حَنَوْنَ لأَطْفالِ

ابن قتيبة، كتاب المعاني الكبير في أبيات المعاني، ص: 89.— المصدر
4.3معنى من معاني «إران»

وــــــــــ: الثَّوْرُ الوَحْشيُّ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ابْنُ مُقْبِلٍ (ت: 37ه=658م) يَفَتَخِرُ بِقُوَّتِهِ:

وكَمْ مِنْ إرانٍ قدْ سَلَبْتُ مَقيلَهُ
إذا ضَنَّ بِالوَحْشِ العِتاقِ مَعاقِلُهْ

ابن ميمون، منتهى الطلب. تح: طريفي، ج: 1، ص: 334.— المصدر
4.4معنى من معاني «إران»

وــــــــــ: السَّيْفُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النّابِغةُ الجَعْديُّ (ت: 50ه=670م) يَصِفُ سَيْفًا قَدْ ظَهَرَ بَعْضُهُ مِنَ جَفْنِهِ:

كالإرانِ المَنْقوفِ مِنْ خَلَلِ الجَفْنِ

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 252.— المصدر
05 —

أُرانَى

اسم
5.1معنى من معاني «أُرانَى»

* الأُرانَى: أُصولُ ثَمَرِ الضَّعةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو عَمْرٍو الشَّيْبانيُّ (ت: 206ه=821م):

"البوصُ: ثَمَرُ الأُرانَى".

الشيباني، الجيم. تح: الإبياري، ج: 1، ص: 94— المصدر
5.2معنى من معاني «أُرانَى»

وــــــــــ: حَبُّ بَقْلٍ يُطْرَحُ في اللَّبَنِ فَيُجَبِّنُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"والأُرانَى: جَناةُ الضَّعةِ؛ وهِيَ بَيْضاءُ فيها ثَمَرةٌ سَوْداءُ تُؤْكَلُ. وقيلَ: حَبُّ بَقْلٍ يُطْرَحُ في اللَّبَنِ فيُجَبِّنُهُ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 253.— المصدر
06 —

أَرانِية

اسم
6.1معنى من معاني «أَرانِية»

* الأَرانِيةُ والأُرانِيةُ: ما يَطولُ ساقُهُ مِنْ شَجَرِ الحَمْض وغَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابنُ سِيْدَهْ (ت: 458ه=1066م):

"والأُرانِيةُ كُلُّ ما يَطولُ ساقُه مِنْ شَجَرِ الحَمْضِ وغَيْرِهِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 280.— المصدر
07 —

أَرَن

اسم
7.1معنى من معاني «أَرَن»

* الأَرَنُ: البَطَرُ. (ج) آرانٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

أُرْنة

اسم
8.1معنى من معاني «أُرْنة»

* أُرْنةُ الـحِرْباءِ: مَوْضِعُهُ مِنَ العودِ إذا انْتَصَبَ عَلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَمْرُو بْنُ أَحْمَرَ الباهِليُّ (ت: 75ه=694م) يَصِفُ حِرْباءَ على مَوْضِعِها الذي انْتَصَبَتْ عَلَيْهِ:

وتَقَنَّعَ الحِرْباءُ أُرْنَتَهُ
مُتَشاوِسًا لِوَريدِهِ نَقْرُ

ديوان عمرو بن أحمر الباهلي. تح: حسين عطوان، ص: 88.— المصدر
8.2معنى من معاني «أُرْنة»

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 280.— المصدر
8.3معنى من معاني «أُرْنة»

وــــــــــ: الجُبْنُ الرَّطْبُ. (ج) أَرَنٌ.

ع

”قالَ حَنّا بِك الأَسْعَد (ت: 1315هـ =1897م) يَهْجو لِئامًا:

يُديرُ لِآلِ العَرْضِ كاساتِ نَشْوةٍ
ولكِنَّ مَشْروبَ اللِّئامِ أَرونُ

ديوان حنا بك الأسعد، ص: 58.— المصدر
10 —

أَرِين

اسم
10.1معنى من معاني «أَرِين»

* الأَرِينُ: الهَدَرُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو مُحَمَّدٍ الفَقْعَسيُّ (ت: 13ه=634م):

مَتى يُنازِعْهُنَّ في ‌الأَرِينِ
يُصْرَعْنَ أَو يُعْطينَ بِالماعونِ

الشيباني، الجيم. تح: الأبياري، ج: 1، ص: 75.— المصدر
10.2معنى من معاني «أَرِين»

وــــــــــ: المَكانُ والمَرْتَعُ لِلإبِلِ

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385هـ = 995م):

"والأَرِينُ: المَكانُ والمَرْتَعُ لِلإبِلِ". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 254.— المصدر
11 —

أُرَيْن

اسم
11.1معنى من معاني «أُرَيْن»

* الأُرَيْنُ: حَبُّ بَقْلٍ يُطْرَحُ في اللَّبَنِ فَيُجَبِّنُهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ الفيروزَآباديّ (ت: 817ه=1414م):

"وحَبُّ بَقْلٍ يُطْرَحُ في اللَّبَنِ فَيُجَبِّنُهُ كالأُرانى، كَحُبارى وزُبَيْرٍ (أُرَيْن)". (وهذِهِ الدَّلالةُ سَجَّلَتْها المَعاجِمُ في بَقيّةِ العُصورِ)

الفيروزَآباديّ، القاموس. تح: العرقسوسي، ص: 1175.— المصدر
12 —

أَرِينة

اسم
12.1معنى من معاني «أَرِينة»

* الأَرِينةُ: نَباتٌ عَريضُ الوَرَقِ يُشْبِهُ الخِطْميَّ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عبيدُ السَلَميُّ السَّعْديُّ (ت: 23ه=643م):

"وفي حَديثِ اسْتِسْقاءِ عُمَرَ: حتَّى رَأَيْتُ الأَرِينةَ تَأْكُلُها صِغارُ الإبِلِ. قالَ: شَمِرُ: الأَرِينةُ: نَبْتٌ".

أبو عبيد، الغريبين غريبي القرآن والحديث. تح: الطناحي، ج: 1، ص: 40.— المصدر
13 —

مِئْران

اسم
13.1معنى من معاني «مِئْران»

* المِئْرانُ: مَأْوى البَقَرِ مِنَ الشَّجَرِ. (ج) مآرِينُ ومَيارينُ ومَآرِنُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ لَيلى الأَخْيَليّة (ت: 80ه=698م) تَصِفُ عِنايةَ فِتْيانِ الخَليفةِ بِناقَتِها:

يُطيفُ بِها فِتْيانُهُ كُلَّ لَيْلةٍ
بِنيرَيْنِ مِئْرانُ الجِبالِ وريفُ

ديوان ليلى الأخيلية. تح: واضح الصمد، ص: 63.— المصدر
