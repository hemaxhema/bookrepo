01 —

قَناذِع

اسم
1.1معنى من معاني «قَناذِع»

* القَناذِعُ: القَبيحُ الفاحِشُ مِنَ الكَلامِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيْدٍ (ت: 35ق.ه=588م) يَفْتَخِرُ بِنَفْسِهِ:

فَلَمْ أَحْتَمِلْ فيما أَتَيْتُ مَلامَةً
أَتَيْتُ الجَمالَ واجْتَنَبْتُ القَنَاذِعا

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 128.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَدْهَمُ بْنُ أَبي الزَّعْراءِ (129ه=747م) يُخاطِبُ بَني خَيْبَريٍّ بَعْدَ أنْ تَزَوَّجَ عَبْدُ اللهِ بْنُ مُدْلِجٍ الطَّائِيُّ امْرَأَةً فأَبَتْ أنْ تُنْزِلَهُ عِنْدَها:

بَني خَيْبَرِيٍّ نَهْنِهُوا عَنْ قَناذِعٍ
أَتَتْ مِنْ لَدُنْكُمْ وانْظُروا ما شُؤونُها

[نَهْنِهُوا: كُفّوا]
التّبريزيّ، شرح الحماسة. تح: غريد الشّيخ، ص: 880.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"القَنازِعُ والقَناذِعُ: القَبيحُ مِنَ الكَلامِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 3، ص: 285.— المصدر
1.2معنى من معاني «قَناذِع»

وــــــــــ: الدَّواهي.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَدِيُّ بْنُ زَيْدٍ (ت: 35ق.ه=588م) يَعِظُ:

﻿ومَنْ لا يُوَرِّعْ نَفْسَهُ تَتْبَعِ الخَنا
ومَنْ يَتْبَعِ الجَرْباءَ يَغْشَ القَناذِعا

ديوان عدي بن زيد العبادي. تح: المعيبد، ص: 147.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ أَدْهَمَ بْنِ أَبي الزَّعْراءِ (129ه=747م) يُخاطِبُ بَني خَيْبَريٍّ بَعْدَ أنْ تَزَوَّجَ عَبْدُ اللهِ بْنُ مُدْلِجٍ الطَّائِيُّ امْرَأَةً فأَبَتْ أنْ تُنْزِلَهُ عِنْدَها:

بَني خَيْبَرِيٍّ نَهْنِهُوا عَنْ قَناذِعٍ
أَتَتْ مِنْ لَدُنْكُمْ وانْظُروا ما شُؤونُها

التّبريزيّ، شرح الحماسة. تح: غريد الشّيخ، ص: 880.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ (ت: 215ه=830م):

"والقَناذِعُ: الدَّواهي ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

المبرد، الفاضل، ص: 80.— المصدر
1.3معنى من معاني «قَناذِع»

وــــــــــ: الخَنا والفُحْشُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ فُسِّرَ قَوْلُ أَدْهَمَ بْنِ أَبي الزَّعْراءِ (129ه=747م) يُخاطِبُ بَني خَيْبَريٍّ بَعْدَ أنْ تَزَوَّجَ عَبْدُ اللهِ بْنُ مُدْلِجٍ الطَّائِيُّ امْرَأَةً فأَبَتْ أنْ تُنْزِلَهُ عِنْدَها:

بَني خَيْبَرِيٍّ نَهْنِهُوا عَنْ قَناذِعٍ
أَتَتْ مِنْ لَدُنْكُمْ وانْظُروا ما شُؤونُها

التّبريزيّ، شرح الحماسة. تح: غريد الشّيخ، ص: 880.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"‌والقَناذِعُ: قَبيحُ الكَلامِ، ‌والفُحْشُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 393.— المصدر
1.4معنى من معاني «قَناذِع»

وــــــــــ: العَنْكَبوتُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ بَرِّيٍّ (ت: 582ه=1187م):

"والقَناذِعُ هو العَنْكَبوتُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن بري، حاشية على درّة الغواص. تح: فرغلي، ص: 750.— المصدر
02 —

قُنْذَع / قُنْذُع

اسم
2.1معنى من معاني «قُنْذَع / قُنْذُع»

* القُنْذَعُ والقُنْذُعُ: الصَّغيرُ الحَقيرُ مِنَ الرِّجالِ. لُغَةٌ في القُنْزُعِ. (ج) قَناذِعُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ جَريرٌ (ت: 110ه=728م) يَهْجو بَني ضَبَّةَ:

وَجَدْنا بَيْتَ ضَبَّةَ في مَعَدٍّ
كبَيْتِ الضَّبِّ لَيْسَ لَهُ سَواري

وَجَدْناهُمْ قَناذِعَ مُلْزِقاتٍ
بِلا نَبْعٍ نَبَتْنَ وَلا نُضارِ

ديوان جرير. تح: البستانيّ، ص: 148.— المصدر
2.2معنى من معاني «قُنْذَع / قُنْذُع»

وــــــــــ مِنَ الرِّجالِ: الدَّيُّوثُ الَّذي لا يَغارُ على أَهْلِهِ. (مع)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ وَهْبُ بْنُ مُنَبِّهٍ (ت: 114ه=732م) في الرَّجُلِ الّذي لا يُنْكِرُ عَمَلَ السّوءِ عَلى أَهْلِهِ:

"فَلَوْ رَأى الرِّجالَ مَعَ امْرَأَتِهِ تُنْكَحُ لَمْ يَرَ ذلِكَ قَبيحًا، فَذلِكَ القُنْذُعُ الدَّيّوثُ لا يَنْظُرُ اللهُ إِلَيْهِ".

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 2، ص: 240.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"القُنْذَعُ والقُنْذُعُ، بالفَتْحِ والضَّمِّ: الدَّيّوثُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 2، ص: 296.— المصدر
”وقالَ أَبو مَنْصورٍ الثَّعالِبيُّ (ت: 430ه=1039م):

"إِذا كانَ يُغْضي عَلى ما يَسْمَعُ مِنْ هَناتِ أَهْلِهِ فهُوَ دَيّوثٌ، فإذا كانَ يُغْضي على ما يَرى مِنْها فهُوَ قُنْذُعٌ، فإذا زادَتْ جَفْلَتُهُ وعُدِمَتْ غَيْرَتُهُ فهو طَسيعٌ وطَزيعٌ".

الثعالبي، فقه اللغة وأسرار العربية. تح: يحيى مراد، ص: 121.— المصدر
03 —

قُنْذُعَة

اسم
3.1معنى من معاني «قُنْذُعَة»

* القُنْذُعَةُ: الخُصْلَةُ مِنَ الشَّعَرِ وما يَبْقى مِنْهُ مُفَرَّقًا في نواحيهِ، أَوْ هِيَ ما ارْتَفَعَ مِنْهُ وطالَ. لُغَةٌ في القُنْزُعَةِ، أو تَصْحيفٌ مِنْها. (ج) قُنْذُعاتٌ، وقَناذِعُ. (انْظُرْ: ق ن ز ع)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو أَيُّوبَ الأَنْصارِيُّ (ت: 52ه=672م):

"ما مِنْ مُسْلِمٍ يَمْرَضُ في سَبيلِ اللهِ إلّا حَطَّ اللهُ عَنْهُ خَطاياهُ ولَوْ بَلَغَتْ قُنْذُعَةَ رَأْسِهِ".

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 3، ص: 286.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538ه=1143م) يَشْرَحُ قَوْلَ أَبي أَيُّوبَ: "ولَوْ بَلَغَتْ قُنْذُعَةَ رَأْسِهِ":

"ولَوْ بَلَغَتْ ‌قُنْذُعَةَ رَأْسِهِ؛ هِيَ القُنْزُعَةُ، واحِدَةُ قَنازِعِ الرَّأْسِ، وهِيَ ما يَبْقى مِنَ الشَّعْرِ مُفَرَّقا في نواحيهِ. وهما لُغَتانِ كالزُّعافِ والذُّعافِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق. تح: البجاوي وآخر، ج: 3، ص: 230.— المصدر
3.2معنى من معاني «قُنْذُعَة»

وــــــــــ : ما يَقْرُصُ مِثْل العَقارِبِ ونَحوِها. وهُوَ بِتاءٍ أَيْضًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230ه=845م):

"يُقالُ: أَتَتْني جَنادِعُ فُلانٍ، وقَنادِعُهُ، وقَناذِعُهُ، وعَقارِبُهُ، وزَنابِرُهُ. ومَعْناهُ: قَوارِصُهُ. والواحِدُ: قُنْدُعٌ، وجُنْدُعٌ. وفي القَناذِعِ: قُنْذُعَةٌ، وقُنْدُعَةٌ".

أبو مسحل، النوادر. تح: عزة حسن، ص: 361.— المصدر
04 —

قُنْذوع

اسم
4.1معنى من معاني «قُنْذوع»

* القُنْذُوعُ: الدَّيُّوثُ الَّذي لا يَغارُ على أهْلِهِ. (مع)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"والقُنْذُعُ والقُنْذوعُ والقُنْذَعُ: الدَّيُّوثُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المخصص. تح: جفال، ج: 4، ص: 224.— المصدر
