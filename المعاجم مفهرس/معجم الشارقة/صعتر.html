صعتر
صَعْتَرَ(فعل) صَعْتَر(اسم) صَعْتَرِيّ(اسم) صَعْتَرِيَّة(اسم) صُعَيْتِرَة(اسم)
المقدّمة
النّظائِرُ السّامِيَّةُ:

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ص ع ت ر):

في الآراميّة: وَرَدَ الاسْمُ: צִתְרֵי (صِثر۪ٮٰ) ṣiṯrē بِمَعْنى (ضَربٌ مِنَ النَّباتِ) يوافِقُ الاسمَ العَرَبيَّ (صَعْتَرٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسمَ الآراميَّ خَضَعَ لسقوطِ حَرفِ العَينِ، ومِنْ مَجيئِهِ بِمَعْنى (صَعترٌ) ما وَرَدَ في التَّلمودِ الأورشليميِّ: "אייכל שמואל לרב מי צתרי הדין טוויה"، أي: "أطْعَمَ صَموئيلُ راڤَ لَحمًا مَشويًّا مَعَ ماءِ الصَّعترِ"

(ירוש'، תרו': נב, א)

في العِبْريّة: وَرَدَ الاسْمُ المُعبرَنُ: צַתְרָה (صَترا) ṣatrā بِمَعْنى (نَوعٌ مِنَ النَّباتاتِ العُشبيّةِ) يوافِقُ الاسمَ العَرَبيَّ (صَعْتَرٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسمَ المُعبرَنَ خَضَعَ لِسُقوطِ العَينِ. ووَرَدَ اللَّفظُ بِصيغةِ צֶתֶר (ص۪ت۪ر) في كِتاباتٍ طِبيّةٍ عِبريّةٍ.

(אסף הרופא כ"י מינכן, 33)

في السُّرْيانيّة: وَرَدَ الاسْمُ: ܨܰܬ̥ܪܳܐ/ ܨܶܬ̊ܪܳܐ، (صَثرا/ ص۪ترا) ṣiṯrē بِمَعْنى (نَباتٌ عُشبيٌّ مِنْ فَصيلةِ الشَّفويّاتِ) يوافِقُ الاسمَ العَرَبيَّ (صَعْتَرٌ) مَبْنًى ومَعْنًى غَيْرَ أنَّ الاسمَ السُّريانيَّ سَقَطَتْ مِنهُ العَينُ، ومِنْ مَجيئِهِ بِمَعْنى (صَعْتَرٌ) : "ܠܥܣ ܩܠܝܠ ܨܬܪ̈ܐ"، أي: "امضَغْ قَليلًا مِنَ الصَّعتَرِ".

(LSt.66:4)

ومِنهُ قَولُ مار يَعقوبَ: "ܥܩܰܪ ܡܶܢ ܐܰܪܥܳܐ ܨܶܬ̊ܪܳܐ ܘܚܳܓܬܳܐ"، أي: "اقتَلَعَ مِنَ الأرْضِ الصَّعترَ والقَرَظَ".

(ܒܰܪ ܠܶܒܳܐ: ܨܬܪ)

المعانى الكلّيّة

  1. ضَرْبٌ مِنَ النَّباتِ
  2. الشِّدَّةُ
  3. الخُبْثُ
  4. الكَرَمُ والشَّجاعَةُ
01 —

صَعْتَرَ

فعل
1.1معنى من معاني «صَعْتَرَ»

* صَعْتَرَ النَّحْلُ: رَعى الصَّعْتَرَ؛ وهُوَ ضَرْبٌ مِنَ البُقولِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَابْنُ عَبّادٍ (ت: 385هـ=995م):

"وصَعْتَرَ النَّحْلُ: رَعى الصَّعْتَرَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 243.— المصدر
1.2معنى من معاني «صَعْتَرَ»

وــــــــــ فُلانٌالشَّيْءَ: زَيَّنَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَحُسَيْنُ بْنُ الضَّحّاكِ (ت: 250هـ=864م) يَصِفُ غُلامًا:

 يا مَنْ رَأى مِثْلَ شادِنٍ خَنِثٍ
يَصولُ في خِدْرِهِ بِزُوَّرِهِ 

يَسْحَبُ ذَيْلَ القَميصِ صَعْتَرَهُ
ووارِداتٍ مِن هُدْبِ مِئْزَرِهِ

الأصفهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 7، ص: 146.— المصدر
02 —

صَعْتَر

اسم
2.1معنى من معاني «صَعْتَر»

* الصَّعْتَرُ مِنَ الرِّجالِ وغَيْرِهِمْ: الشَّديدُ، (ج) صَعاتِرُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَقَيْسُ بْنُ خُوَيْلِدٍ الهُذَلِيُّ (ت: 74ق هـ=549م) يَصِفُ مَوْضِعًا:

أَرى حُثُنًا أَمْسى ذَليلًا كَأَنَّهُ
تُراثٌ وخَلّاهُ الصِّعابُ الصَّعاتِرُ

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 606.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَأَبُو سَعيدٍ السُّكَّرِيُّ (ت: 275هـ=888م):

"الصَّعاتِرُ: الشِّدادُ مِنَ الرِّجالِ، واحِدُهُمْ: صَعْتَرٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السكري، شرح أشعار الهذليين. تح: فراج، ج: 2، ص: 606.— المصدر
2.2معنى من معاني «صَعْتَر»

وــــــــــ: ضَرْبٌ مِنَ البُقولِ، واحِدَتُهُ: صَعْتَرَةٌ. (انْظُرْ: ز ع ت ر ، س ع ت ر )

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَأبو ‌الطَّمَحَانِ القَيْنيُّ (ت: 23ق هـ=650م) يُخاطِبُ ناقَتَهُ:

أَسَرَّكِ لَوْ أَنّا بِجَنْبَيْ عُنَيْزَةٍ
وحَمْضٍ وضُمْرانِ الجَنابِ وصَعْتَرِ

الأصفهاني، الأغاني. تح: إحسان عباس وآخرين، ج: 13، ص: 10.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”رُوِيَ عَنِ: النَّبِيِّ ﷺ (ت: 11هـ=632م) أَنَّهُ قالَ:

«تَبَخَّروا في بُيوتِكُمْ بِاللِّبانِ والـمُرِّ والصَّعْتَرِ».

العسقلاني، المطالب العالية. تح: سمير العمران، ج: 11، ص: 47.— المصدر
”وقالَرَجُلٌ مِنَ الـمُسْلمينَ (36هـ=656م) يُخَاطِبُ صاحِبَهُ وهُمْ ضَيفُ سَلْمانَ الفارِسِيِّ:

"لَوْ كانَ في مِلْحِنا صَعْتَرٌ".

الطبراني، المعجم الكبير. تح: السلفي، ج: 6، ص: 235.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَالخَليلُ (ت: 170هـ=787م):

"الصَّعْتَرُ: ضَرْبٌ مِنَ البُقولِ".

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 338.— المصدر
”وقالَالشّافِعِيُّ (ت: 204ه=819م) يَذْكُرُ بَعْضَ المُعامَلاتِ المُتَعَلِّقَةِ بِالعَسَلِ:

"ولَوْ قالَ: عَسَلُ بُرٍّ، أَوْ قالَ: عَسَلُ صَعْتَرٍ...ووَصَفَ لَوْنَهُ وبَلَدَهُ فَأَتاهُ بِاللَّوْنِ والْبَلَدِ وبِغَيْرِ الصِّنْفِ الَّذي شَرَطَ لَهُ؛ أَدْنَى أَوْ أَرْفَعَ، لَمْ يَكُنْ عَلَيْهِ أَخْذُهُ".

الشافعي، الأم. تح: رفعت فوزي، ج:4، ص: 215— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَيوسفُ بْنُ عُمَرَ التُّرْكُمانِيُّ (ت: 694هـ=1295م) يَتَحَدَّثُ عَنْ كَيْفِيَّةِ تَناوُلِ وَصْفَةٍ طِبِّيَّةٍ:

"وأَمّا الـمَبْرودونَ، فَيَأْكُلونَها بِالصَّعْتَرِ والزَّيْتِ".

ابن التركماني، المعتمد في الأدوية المفردة، ج: 1، ص: 205.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَمُصْطَفى السُّيوطِيُّ (ت: 1243هـ=1827م) يُعَدِّدُ ما يُكالُ:

"وَباقي ثَمَرِ النَّخْلِ...مَكيلٌ...والـمِلْحُ والأُشْنانُ والأَبازيرُ والصَّعْتَرُ وما أَشْبَهَها".

مصطفى السيوطي، مطالب أولي النهى، ج: 3، ص: 170.— المصدر
03 —

صَعْتَرِيّ

اسم
3.1معنى من معاني «صَعْتَرِيّ»

* الصَّعْتَرِيُّ مِنَ الرِّجالِ: الشّاطِرُ؛ الَّذي عُرِفَ بِالخُبْثِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَالخَليلُ (ت: 170هـ=787م):

"والصَّعْتَرِيُّ:‏ الشّاطِرُ".

الخليل، العين. تح: المخزومي وآخر، ج: 2، ص: 338.— المصدر
”وقالَالجاحِظُ (ت: 255هـ=869م) يَصِفُ رَجُلًا:

"وكانَ صَعْتَرِيًّا صاحِبَ نَيْزَكِيَّةٍ وتَخَلُّعٍ".

الجاحظ، البرصان والعرجان. تح: هارون، ص: 326.— المصدر
3.2معنى من معاني «صَعْتَرِيّ»

وــــــــــ: الظَّريفُ؛ الحاذِقُ الذَّكِيُّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَالنَّضْرُ بْنُ شُمَيْلٍ (ت: 203هـ=819م):

"فَإِذا كانَ عَبِقًا لَبِقًا، فَهُوَ صَعْتَرِيٌّ".

الثعالبي، فقه اللغة. تح: يحيى مراد، ص: 124.— المصدر
3.3معنى من معاني «صَعْتَرِيّ»

وــــــــــ: الكَريمُ الشُّجاعُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَأَبو عَمْرٍو الشَّيبانِيُّ (ت: 206هـ=821م):

"ورَجُلٌ صَعْتَرِيٌّ، لا غَيْرَ: إِذا كانَ فَتًى كَريمًا شُجاعًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: درويش، ج: 3، ص: 330.— المصدر
3.4معنى من معاني «صَعْتَرِيّ»

وــــــــــ: ضَرْبٌ طَيِّبُ الرّائِحَةِ مِنْ نَباتِ الحَبَقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَابْنُ وَكيعٍ التِّنِّسيُّ (ت: 393هـ=1003م) يَصِفُ نَوْعًا مِنَ الحَبَقِ:

صَعْتَرِيٌّ أَدَقُّ مِنْ أَرْجُلِ النَّمْـ
ـلِ وأَذْكى مِنْ نَفْحَةِ الزَّعْفَرانِ

النويري، نهاية الأرب. تح: قميحة وآخرين، ج: 11، ص: 166.— المصدر
04 —

صَعْتَرِيَّة

اسم
4.1معنى من معاني «صَعْتَرِيَّة»

* الصَّعْتَرِيَّةُ: التَّخابُثُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَأَبو عَبّادٍ النُمَيْرِيُّ (ت: 218هـ=833م):

"‏كُنْتُ وأَنا غُلامٌ أَشْتَهي الصَّعْتَرِيَّةَ، والـمُواثَبَةَ، والتَّكاتُفَ، والتَّشالَّ".

الجاحظ، البرصان والعرجان. تح: هارون، ص: 327.— المصدر
4.2معنى من معاني «صَعْتَرِيَّة»

وــــــــــ: غُدَّةٌ توجَدُ في أَعْلى الصَّدْرِ خَلْفَ عَظْمَةِ القَصِّ، تُساعِدُ في تَمايُزِ خَلايا الدَّمِ البَيْضاءِ وتَخَصُّصِها.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَمُحَمَّد صِدْقي (1333هـ=1915م):

"وفي الحَيَواناتِ الَّتي تَسْبُتُ تَبْقى فيها هَذِهِ الغُدَّةُ الصَّعْتَرِيَّةُ مُدَّةَ الحَياةِ كُلِّها".

محمد رشيد رضا، مجلة المنار، ج: 18، ص: 433.— المصدر
05 —

صُعَيْتِرَة

اسم
5.1معنى من معاني «صُعَيْتِرَة»

* الصُّعَيْتِرَةُ: نَباتٌ عُشْبِيٌّ يُتَداوى بِهِ، يُعْرَفُ بِالأَفْتيمُونِ الأَنْدَلُسِيِّ أَوِ الإِقْريطِيِّ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَيُوسُفُ بْنُ عَبْدِ الهادي المَقْدِسِيُّ (ت: 909هـ=1503م):

"صُعَيْتِرَةٌ: هُوَ الأَفْتيمونُ الإِقْريطِيُّ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

ابن عبد الهادي، زبد العلوم. تح: الموجان، ج: 2، ص: 504 .— المصدر
