النقوش العربيّة

وَرَدَ الجَذْرُ (ذ ن ن ذ ن ن ) في النُّقوشِ العَربيّةِ الجَنوبيّةِ، مِن بَينِها نَقشٌ سَبَئِيٌّ غَيرُ مُؤرَّخٍ ((RES 2734 R)، وَرَدَ فيهِ الاسْمُ ( ذننم ḏnnm ) عَلَمًا على أحدِ الأفرادِ، فجاءَ في النَّقْشِ: "ذننم/ بن/ وتر"، أيْ: "ذننم بن وتر".

(Ryckmans, G., 1930)

01 —

ذَنَّ

فعل
1.1معنى من معاني «ذَنَّ»

* ذَنَّ (كضَرَبَ) الأنْفُ ونَحْوُهُ يَذِنُّ (بكَسْرِ الذّالِ) ذَنًّا، وذَنينًا: سالَ مِنْهُ الذَّنينُ. (انْظُرْ: ذ م م)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الشَّمّاخُ (ت: 22هـ=643م) يَصِفُ أتانًا:

وإنْ ضُرِبَتْ على العِلّاتِ حَطَّتْ
إلَيْكَ حِطاطَ ‌هادِيَةٍ ‌شَنونِ

تُوائِلُ مِن مِصَكٍّ أنْصَبَتْهُ
حَوالِبُ أسْهَرَيْهِ بِـالذَّنينِ

[تُوائِلُ: تَعْدُو هَرَبًا، المِصَكُّ: الحِمارُ المُغْتَلِمُ، الحَوالِبُ: ما يَتَحَلَّبُ إلَى ذَكَرِهِ مِنَ المَنِيِّ، والأسْهَرانِ: عِرْقانِ يَجْري فِيهِما ماءُ الفَحْلِ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 177.— المصدر
”وقالَ ابْنُ قُتَيْبَةَ (ت: 276هـ=889م):

"وكانَ المَريضُ يَخافُ أن يُفْسِدَ على النّاسِ طَعامَهُمْ بأُمورٍ قَدْ تَعْتَري مَعَ المَرَضِ، مِنْ رائِحَةٍ تَتَغَيَّرُ، أو جُرْحٍ يَبِضُّ، أو أَنْفٍ ‌يَذِنُّ".

الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)

"كانَتِ العَرَبُ ومَنْ بالمَدينَةِ قَبْلَ البَعْثِ تَجْتَنِبُ الأكْلَ مَعَ أهْلِ هَذِهِ الأعْذارِ؛ لِمَكانِ جَوَلانِ يَدِ الأعْمى، وانْبِساطِ جِلْسَةِ الأعْرَجِ، وعَدَمِ خُلُوِّ المَريضِ مِنْ رائِحَةٍ تُؤْذي، أوْ جُرْحٍ يَبِضُّ، أوْ أنْفٍ ‌يَذِنُّ، فَنَزَلَتْ".

الألوسي، روح المعاني. تح: حبّوش وآخرين، ج: 18، ص: 477.— المصدر

8.1معنى من معاني «ذُنانة»

* الذُّنانَةُ: بَقِيَّةُ الدَّيْنِ والعِدَةُ تَبْقى للرَّجُلِ عِنْدَ غَيْرِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأنْصارِيُّ (ت: 215هـ=830م):

"الذُّنانَةُ: بَقِيَّةُ الشَّيْء مِنَ الدَّيْنِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلام، الغريب المصنف. تح: صفوان داوودي، ج: 1، ص: 216.— المصدر
8.2معنى من معاني «ذُنانة»

وــــــــــ: بَقِيَّةُ الشَّيْءِ الهالِكِ الضَّعيفِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو نَصْرٍ الجَوْهَرِيُّ (393هـ=1003م):

"والذُّنانَةُ بَقِيَّةُ الشَّيْءِ الهالِكِ الضَّعيفِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

الجوهري، الصحاح. تح: عطار، ج: 5، ص: 2120.— المصدر
8.3معنى من معاني «ذُنانة»

وــــــــــ: الحاجَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"والذُّنانَةُ: الحاجَةُ والطَّلِبَةُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 59.— المصدر
09 —

ذُنانِيّ

اسم
9.1معنى من معاني «ذُنانِيّ»

* الذُّنانِيُّ: الرَّجُلُ الكَثيرُ الذُّنانِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"الذّنانِيُّ: الرَّجُلُ الذي يَسيلُ مِنْهُ الذُّنانُ".

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 59.— المصدر
10 —

ذُنْذُن

اسم
10.1معنى من معاني «ذُنْذُن»

* الذُّنْذُنُ والذِّنْذِنُ: أسْفَلُ القَميصِ الطَّويلِ. (ج) ذَناذِنُ. (وانْظُرْ: ذ ن ذ ن / ذ ل ل)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ السِّكّيتِ (ت: 244هـ=858م):

"ذَلاذِلُ القَميصِ وذَناذِنُهُ لأسافِلِهِ، الواحِدُ ذُلْذُلٌ وذُنْذُنُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن السكيت، الإبدال. تح: هفنر، ص: 9.— المصدر
”وقالَ الشّاعِرُ (255هـ=869م، مِنْ شَواهِدِ شَمِرِ بْنِ حَمْدَوَيْهِ):

لَوْ أنَّ سَلْمى وَرَدَتْ ذا أَلْجافْ

لَقَــصَّرَتْ ‌ذَناذِنَ الثَّـــــــــــــوْبِ الضّافْ

الأزهري، تهذيب اللغة. الدار المصرية، ج:11، ص: 86.— المصدر
11 —

ذَنَن

اسم
11.1معنى من معاني «ذَنَن»

* الذَّنَنُ: القَذَرُ والثُّفْلُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ حَسّانُ بْنُ ثابِتٍ (ت: 54هـ=674م) يَتَغَزَّلُ:

زادَتْ هُمومٌ فَماءُ العَيْنِ مُنْحَدِرُ
سَحًّا إذا حَفَلَتْهُ عَبْرَةٌ دِرَرُ

وَجْدًا بِشَمّاءَ إذْ شَمّاءُ بَهْكَنَةٌ
هَيْفاءُ لا ذَنَنٌ فيها ولا خَوَرُ

السهيلي، الروض الأنف. تح: عبد الرحمن الوكيل، ج: 7، ص: 289.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السُّهَيْلِيُّ (ت: 581هـ=1185م):

"والذَّنَنُ، محرَّكةً: القَذَرُ والثُّفْلُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في عَصْرِ الدّولِ والإِماراتِ)

السهيلي، الروض الأنف. تح: عبد الرحمن الوكيل، ج: 7، ص: 289.— المصدر
12 —

ذَنين

اسم
12.1معنى من معاني «ذَنين»

* الذَّنينُ: الذُّنانُ، وهُوَ المُخاطُ الرَّقيقُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الحادِرةُ الغَطَفانِيُّ (ت: 5هـ=626م) يُشَبِّهُ ذَنينَ الخَيْلِ ببَيْضِ النَّمْلِ:

وتَرى الذَّنينَ على مَراسِنِهِمْ
يَوْمَ الهِياجِ كمازِنِ النَّمْلِ

[ المازِنُ: بَيْضُ النَّمْلِ]
الفارابي، ديوان الأدب. تح: أحمد مختار، ج: 1، ص: 362.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَلِيٌّ الأحْمَرُ (ت: 194هـ=810م):

"والأذَنُّ: الَّذي يَسيلُ مَنْخِراهُ جَميعًا، ويُقال لذلِكَ الّذي يَسيلُ مِنْهُ الذَّنينُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في العَصْرِ الحَديثِ)

القاسم بن سلام، الغريب المصنف. تح: داوودي، ص: 83.— المصدر
”وقالَ أبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م) يَشْرَحُ قوْلَ الشَّماخِ (حَوالِبُ أسْهَرَيْهِ بِـالذَّنينِ):

"أسْهَراهُ: ذَكَرُهُ وأنْفُهُ، وحَوالِبُهُما عُروقٌ تَمُدُّ ‌الذَّنينَ من الأنْفِ، والمَذْيَ مِنْ قَضيبِهِ".

الأزهري، تهذيب اللغة. الدار المصرية، ج: 5، ص: 87.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ الصَّيْقَلِ الجَزَرِيُّ (ت: 701هـ=1302م) يَصِفُ بُكاءَ صاحِبِهِ:

"وصاحِبي يَمْزُجُ بدَرِّ دَمْعِهِ الذَّنينَ".

ابن الصيقل، المقامات الزينية. تح: الصالحي، ص: 393.— المصدر
12.2معنى من معاني «ذَنين»

وــــــــــ: ماءُ الفَحْلِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

تُوائِلُ مِن مِصَكٍّ أنْصَبَتْهُ
حَوالِبُ أسْهَرَيْهِ بِـالذَّنينِ

[ تُوائِلُ: تَعْدُو هَرَبًا، المِصَكُّ: الحِمارُ المُغْتَلِمُ، الحَوالِبُ: ما يَتَحَلَّبُ إلَى ذَكَرِهِ مِنَ المَنِيِّ، والأسْهَرانِ: عِرْقانِ يَجْري فيهِما ماءُ الفَحْلِ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ مَعْمَرُ بْنُ المُثَنّى (ت: 209هـ=825م) في الحِصانِ:

"فإذا قَطَرَ مِنْهُ ماءٌ صافٍ لَيْسَ بالماءِ الأعْظَمِ قيلَ لهُ الذَّنينُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

معمر بن المثنى، الخيل. دائرة المعارف العثمانية، ص: 39.— المصدر
13 —

ذُنَيْناء

اسم
13.1معنى من معاني «ذُنَيْناء»

* الذُّنَيْناءُ: ما يَخْرُجُ مِنَ القَمْحِ ونَحْوِهِ مِنَ الطَّعامِ فيُرْمى بِهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو حَنيفَةَ الدِّينَوَرِيُّ (ت: 282هـ=895م):

"في الطَّعامِ ذُنَيْناءُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتها المعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 52.— المصدر
14 —

ذَنينَة

اسم
14.1معنى من معاني «ذَنينَة»

* الذَنينَةُ: البَقِيَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الصّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"‌وذَنينَةٌ مِنْ خَيْرٍ وذُنانَةٌ وذانَّةٌ؛ أيْ: بَقِيَّةٌ".

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 59.— المصدر
