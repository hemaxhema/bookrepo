النظائر الساميّة

احْتَفَظَتِ السّاميّاتُ بهذا الجَذْرِ مِنَ العَرَبيَّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذْرُ (ن ه ق):

في الأوجاريتيَّةِ؛ ومِنْهُ الاسْمُ المُؤَنَّثُ: nhqt (نَهقة) بِمَعْنى: "صَوْتِ الحِمارِ"، وهُوَ يَتَّفِقُ مَعَ الاسْمِ العَرَبيِّ (نَهيقٍ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَهيقٍ": l ql nhqt ;mrh ؛ أَيْ: "بِسبَبِ ضَجيجِ نَهيقِ حِمارِهِ".

(DULAT, p. 617)

في الآراميَّةِ؛ ومِنْهُ الفِعْلُ: נַהֵק (نَه۪ٮٰق) nahēq بِمَعْنى: "صَوَّتَ (الحِمارُ)"، وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَهَقَ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَهَقَ" ما وَرَدَ في التَّلْمودِ الأورشَليميِّ: "אזלת וקמת לה על תרעא דמרה שריית מנהקת"؛ أَيْ: "ذَهَبَتِ (الأَتانُ) ووقَفَتْ عِنْدَ بابِ صاحِبِها وبَدَأَتْ تَنْهَقُ".

(ירוש'، שק': כא, ב)

في العِبْريَّةِ؛ ومِنْهُ الفِعْلُ: נָהַק (ناهَق) nāhaq بِمَعْنى: "نَعَرَ الحِمارُ، خارَ"، وهُوَ يُوافِقُ الفِعْلَ العَرَبيَّ (نَهَقَ) مَبْنًى ومَعْنًى؛ ومِنْ مَجيئِهِ بِمَعْنى: "نَهَقَ" ما وَرَدَ في سِفْرِ أَيّوبَ: "בֵּין שִׂיחִים יִנְהָקוּ‏"؛ أَيْ: "بَيْنَ الأَشْجارِ يَنْهَقونَ".

(איוב 30: 7)

في السُّرْيانيَّةِ؛ ومِنْهُ الفِعْلُ: ܢܗܰܓ̥ (نْهَغ) nəhaġ بِمَعْنى: "صَوَّتَ/ أَنَّ/ تَأَوَّهَ"، وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَهَقَ) مَبْنًى ويُدانيهِ مَعْنًى، غَيْرَ أَنَّ الفِعْلَ السُّرْيانيَّ خَضَعَ لإِبْدالِ القافِ جيمًا؛ ومِنْ مَجيئِهِ بِمَعْنى: "تَأَوَّهَ": "ܥܠ ܚܝܠܗ ܒܘܟܪܐ ܢܿܗܓ‏"؛ أَيْ: "يَتَأَوَّهُ على قُوَّتِهِ البِكْرِ".

(IA2 122:v299)

في الجِعْزيَّةِ؛ ومِنْهُ الفِعْلُ: ንህቀ (نْهݣَ) nǝhak'a بِمَعْنى: "نَهَقَ"، وهوَ يَتَّفِقُ مَعَ الفِعْلِ العَرَبيِّ (نَهَقَ) مَبْنًى وفي بَعْضِ مَعانيهِ، غَيْرَ أَنَّ الفِعْلَ الجِعْزيَّ خَضَعَ لإِبْدالِ القافِ كافًا دَفْعيَّةً.

(CODG, p.394)

01 —

نَهَقَ

فعل
1.1معنى من معاني «نَهَقَ»

* نَهَقَ (بفَتْحِ الهاءِ) الحِمارُ يَنْهُقُ ويَنْهِقُ ويَنْهَقُ (بِضَمِّ الهاءِ وكَسْرِها وفَتْحِها) نَهْقًا ونُهاقًا ونَهيقًا وتَنْهاقًا: صَوَّتَ صَوْتًا مُنْكَرًا.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عُرْوةُ بْنُ الوَرْدِ (ت: 15ق.ه=608م) يَسْخَرُ مِنَ يَهودَ:

وقالوا اجْثُ وانْهَقْ لا تَضُرُّكَ خَيْبرٌ
وذلِكَ مِنْ دينِ اليَهودِ وَلوعُ

لَعَمْري لَئِنْ عَشَّرْتُ مِنْ خَشْيةِ الرَّدى
نُـهاقَ الـحَميرِ إِنّني لَـجَزوعُ

[عَشَّرَ: يُقالُ عَشَّرَ الحِمارُ،بِـمَعْنَى كَرَّرَ النَّهِيقَ في طَلَقٍ وَاحِدٍ]
الفيروزآبادي، المغانم المطابة، ج: 2، ص: 821.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال النَّبِيّ ﷺ (ت: 11ه=632م):

«إِذا نَهَقَ الحِمارُ فَتَعَوَّذوا بِاللَّهِ مِنَ الشَّيْطانِ الرَّجيمِ».

الطبراني، الدعاء. تح: عطا، ج: 1، ص: 557.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ رُؤبَةُ بْنُ العَجّاجِ ‏ (ت: 145ه=762م) يَصِفُ حِمارًا:

حَشْرَجَ في الجَوْفِ سَحيلًا أوْ شهَقْ

حَتّى يُقالَ ناهِقٌ وما نَـهَقْ

ديوان رؤبة بن العجاج. تح: البروسي، ص: 106.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال شِهابُ الدّينِ القَرافي (ت: 684ه=1285م):

"فَلَوْ ضَحِكَ أَوْ نَهِقَ كَالحَميرِ أَوْ نَعَقَ كَالغِرْبانِ ونَحْوِهِ، قالَ صاحِبُ الطِّرازِ: تَبْطُلُ صَلاتُهُ".

القرافي، الذخيرة. تح: أعراب، ج: 2، ص: 140.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال ابْنُ عَجيبةَ (ت: 1224ه=1809م) يُفَسِّرُ قَوْلَ اللهِ ﷻ ﴿وَٱنظُرۡ إِلَى ٱلۡعِظَامِ كَيۡفَ نُنشِزُهَا﴾

[البقرة: 259]:

"نَظَرَ إلى العِظامِ، فقامَ كُلُّ عَظْمٍ إلى مَوْضِعِهِ، ثُمَّ كُسِيَ لَحْمًا وجِلْدًا، ‌وجَعَلَ ‌يَنْهَقُ".

ابن عجيبة، البحر المديد. تح: القرشي، 1، 292.— المصدر
1.2معنى من معاني «نَهَقَ»

وــــــــــ فُلانٌ: رَفَعَ صَوْتَه مُقَلِّدًا صَوْتَ الحِمارِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالتْ عائِشَةُ أمُّ المُؤمِنينَ (ت: 58ه=678م):

"كانَتِ المَدينةُ وَبيئةً، وكانوا يَقولونَ إِذا قَدِمَ الرَّجُلُ أَرْضًا وَبيئةً، وَضَعَ يَدَهُ خَلْفَ أُذُنِهِ ونَهَقَ مِثْلَ الحِمارِ عَشَّرَ لَمْ يُصِبْهُ وَباؤُها".

أبو إسحاق، غريب الحديث. تح: العايد، ج: 1، ص: 155.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الجاحظ (ت: 255ه=869م):

"كانَ أَبو دَيّوبةَ مَوْلى آلِ زِيادٍ يَنْهَقُ بِبابِ الكَرْخِ بِحَضْرةِ المَكّارينَ، فَلا يَبْقى حِمارٌ مَريضٌ، ولا هَرِمٌ، ولا حَسيرٌ مُتْعَبٌ إلّا نَهَقَ".

الزمخشري، ربيع الأبرار. ج: 3، ص: 143.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ سَعيدٍ المَغْرِبيّ (ت: 685ه=1286م):

"أَعْجَبُها أَنَّ ابْنَ الصّابُونِيّ شَاعِرَ إِشْبيلِيَةَ كانَ يُلَقَّبُ بالحِمارِ وَيَحْرَدُ، فَلاجَجَهُ يَوْمًا في مَسْألةٍ فَقالَ لَهُ كَذا هِيَ يا حِمارُ يا حِمارُ إِلى أَنْ تَدَرَّجَ حَتّى قالَ: يا مِلْءَ السَّماواتِ والأَرْضِ حَميرًا، ثُمَّ جَعَلَ إِصْبَعَيْهِ في أُذُنَيْهِ، وزَحَفَ إِلى أَذْيالِ الحُصُرِ وهُوَ يَنْهَقُ كالحِمارِ".

ابن سعيد المغربي، المغرب. تح: شوقي ضيف، ج: 2، ص: 130.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ شِهابُ الدّينِ الآلوسي (ت: 1270ه=1854م) يُبَيِّنُ بَعْضَ الأوْهامِ:

"مِثْلُهُ ما شاعَ بَيْنَ الجَهَلةِ مِنْ أنَّ نَهيقَ الحِمارِ لَعْنٌ لِلشّيعةِ الّذينَ لا يَزالونَ يَنْهَقونَ بِسَبِّ الصَّحابةِ، ومِثْلُ هذا مِنَ الخُرافاتِ الّتي يَمُجُّها السَّمْعُ ما عَدا سَمْعَ طَويلِ الأُذُنَيْنِ".

الآلوسيّ، روح المعاني. تح: حبّوش، ج: 21، ص: 69.— المصدر
02 —

تَناهَقَ

فعل
2.1معنى من معاني «تَناهَقَ»

* تَناهَقَتِ الحُمُرُ: نَهَقَتْ مَعًا، أَوْ بَعْضُها إِثْرَ بَعْضٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال أَوْسُ بْنُ حَجَرٍ (ت: 2ق.ه=620م) يَهْجو قَوْمًا:

تَناهَقونَ إذا اخْضَرّتْ نِعالُكُمُ
وفي الحَفيظَة أبْرامٌ مَضاجيرُ

[النِّعالُ: الأَرْضُ لأنَّها تُنْعَلُ وتوطَأُ. اخْضَرَّتْ: أخْصَبَتْ. الحَفيظةُ: الغَضَبُ]
ديوان أوس بن حجر. تح: نجم، ص: 45.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”رُوِيَ أنَ النَبِيَّ ﷺ (ت: 11ه=632م) قالَ:

«إِيّاكُمْ والسَّمَرَ بَعْدَ العِشاءِ الآخِرةِ. وإِذا تَناهَقَتِ الحُمُرُ مِنَ اللَّيْلِ، فَاسْتَعيذوا بِاللهِ مِنَ الشَّيْطانِ».

عبد الرزاق، المصنَّف. تح: الأعظمي، ج: 1، ص: 563.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الشّاعِرُ (207ه=823م) مِنْ شَواهِدِ الفَرّاءِ، يَهْجو قَوْمًا:

قَوْمٌ إذا اخْضَرَّتْ نِعالُهُمُ
يَتَناهَقون تَناهُقَ الحُمُرِ

الأزهري، تهذيب اللغة. تح: درويش، ج: 2، ص: 398.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ المُجاوِرِ الشَّيْبانِيّ (ت: 690ه=1291م):

"فبَيْنَما القَوْمُ فيما هُمْ عَلَيْهِ مِنْ أَحْوالِهم. الرِّجالُ تَحْرُثُ، والنِّساءُ تَغْزِلُ والحَميرُ تَتَناهَقُ، والكِلابُ تَتَنابَحُ إِذِ ارْتَفَعُوا مِنَ الأرْضِ إلى الجَوِّ".

ابن المجاور، تاريخ المستبصر. تح: ممدوح، ص: 107.— المصدر
03 —

ناهِق

اسم
3.1معنى من معاني «ناهِق»

* النّاهِقُ: مَخارِجُ النَّهيقِ مِنَ الحِمارِ ونَحْوِهِ مِنَ الدَّوابِّ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّمِرُ بْنُ تَوْلَبٍ (ت: 13ه=634م) يَصِفُ حِمارًا يَتَرَبَّصُ بِهِ صائدٌ:

فراقَبَهُ وَهْوَ في قُتْرةٍ
وما كانَ يَرْهَبُ أَنْ يُكْلَما

فأَرْسَلَ سَهْمًا لَهُ أَهْزَعًا
فشَكَّ نَواهِقَهُ والفَما

[القُتْرَةُ: بَيْتٌ كَالخُصِّ يَسْتَتِرُ بِهِ الصَّائِدُ عِنْدَ تَصَيُّدِهِ.يُكْلَمُ: يُجْرَحُ. الأَهْزَعُ: السَّهْمُ الّذي يَضِنُّ بِهِ الرَّامِي لِنَفاسَتِهِ]
شعر النّمر بن تولب العكليّ. تح: طريفي، ص: 119.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"النَّواهِقُ مِنَ الخَيْلِ والحُمُرِ حَيْثُ يَخْرُجُ النُّهاقُ مِنْ حَلْقِهِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)

غَداةَ سَعى أبو بَكْرٍ إلَيْهِمْ
كَما يَسْعَى لِمَوْتَتِهِ جُلالُ

أراحَ على نَواهِقِها عَلِيًّا
ومَجَّ لَهُنَّ مُهْجَتَهُ حِبالُ

[جُلالٌ: بَعِيرٌ عَظِيمٌ. حِبالٌ: أَخُو طُلَيْحَةَ بْنِ خُوَيْلِدٍ]

”قالَ مُحمد بْنُ عبد القادر الجَزائريّ (ت: 1331ه=1913م) يُبَيِّنُ ما يُسْتَحْسَنُ مِنَ الخَيْلِ:

"مِنْها أَن تَكونَ أَسيلةَ الخَدِّ، ونَواهِقُها عارِيةً مِنَ اللَّحْمِ

9-2: وــــــــــ: عُروقٌ تُحيطُ بِأنْفِ الدّابّةِ. الواحِدةُ ناهِقةٌ.

(ع) (133ه‍/750م) إلى (656ه‍/1258م)".

ابنُ عبد القادر الجَزائريّ، نخبة عقد الأجياد. ص: 130.— المصدر
”قالَ الخَليلُ (ت: 170ه=787م):

"نَواهِقُ الدّابّةِ: عُروقٌ اكْتَنَفَتْ خَياشيمَها. الواحِدةُ: ناهِقةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 3، ص: 369.— المصدر
