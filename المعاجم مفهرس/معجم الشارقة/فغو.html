فغو
فَغا(فعل) فَغِيَ(فعل) أَفْغى(فعل) فاغِيَة(اسم) فَغا(اسم) فَغى(اسم) فَغْو(اسم) فَغْوَة(اسم)
المقدّمة

الـمَعاني الكُلِّيَّةُ:

1. الإزْهارُ. 2. اليُبْسُ. 3. صِنْفٌ مِنَ الطّيبِ. 4. الفَسادُ.

01 —

فَغا

فعل
1.1معنى من معاني «فَغا»

* فَغَتِ (كدعا) الشَّجَرةُ تَفْغو فَغْوًا: أزْهَرَتْ وتَفَتَّحَ نَوْرُها.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ الحَسَنُ البَصْرِيُّ (ت: 110هـ=728م) يُجيبُ مَنْ سَأَلَهُ عَنِ السَّلَفِ في الزَّعْفَرانِ:

"إذا فَغا".

ابن قتيبة، غريب الحديث. تح: عبد الله الجبوري، ج: 1، ص: 299.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابنُ دُرَيْدٍ (ت: 321هـ=933م):

"أفْغى الشَّجْرُ يُفْغي إفْغاءً، وفَغا يَفْغو فَغْوًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: رمزي بعلبكي، ج: 2، ص: 959.— المصدر
”

وقالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م) يَشْرَحُ قَوْلَ الحَسَنِ (إذا فَغا):

"يُرِيدُ: إِذا انْتَشَرَتْ رَائِحَتُهُ، مِنْ فَغَتِ الرَّائِحةُ فَغْوًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الزمخشري، الفائق في غريب الحديث. تح: علي محمد البجاوي، وآخر، ج: 3، ص: 131.— المصدر
1.2معنى من معاني «فَغا»

وــــــــــ الشَّيْءُ: فَشا وانْتَشَرَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ حِتْرِشٌ الأعْرابِيُّ (ت: 255هـ=869م):

"هَذِهِ كَلِمةٌ فاغِيةٌ فينا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 206.— المصدر
1.3معنى من معاني «فَغا»

وــــــــــ الزَّرْعُ: يَبِسَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"فَغا الزَّرْعُ: يَبِسَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط في اللغة. تح: محمد حسن آل ياسين، ج: 5، ص: 139.— المصدر
1.4معنى من معاني «فَغا»

وــــــــــ فُلانٌ الدُّهْنَ: مَزَجَهُ بِزَهْرِ الحِنّاءِ لِيَشْتَدَّ ويَغْلُظَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"الفاغِيةُ: نَوْرُ الحِنّاءِ، ودُهْنٌ مَفْغُوٌّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، كتاب العين. تح: المخزومي وآخر، ج: 8، ص: 452.— المصدر
1.5معنى من معاني «فَغا»

وــــــــــ فُلانٌ جِسْمَهُ: ‌طَيَّبَهُ ‌بِالفاغِيةِ، وهِيَ الرّائِحةُ الطَّيِّبةُ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”

جاءَ في المُعْجَمِ الوَسيطِ مَجْمَعِ القاهِرةِ (ت: 1379هـ=1960م، تاريخُ الطَّبْعِ):

"فغا الشَّيْءُ: فَشا وفاحَتْ رائِحَتُهُ، و: جِسْمَهُ: ‌طَيَّبَهُ ‌بِالفاغِيةِ".

إبراهيم مصطفى وآخرون، المعجم الوسيط، ج: 2، ص: 696.— المصدر
02 —

فَغِيَ

فعل
2.1معنى من معاني «فَغِيَ»

* فَغِيَ (بِكَسْرِ الغَيْنِ) التَّمْرُ يَفْغى (بِفَتْحِ الغَيْنِ) فَغًى: حَشَفَ، أيْ: يَبِسَ وتَقَبَّضَ. فَهُوَ فَغًى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ وَلّادٍ التَّميمِيُّ (ت: 332هـ=944م):

"أفْغى البُسْرُ، وفَغِيَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن ولاد، المقصور والممدود. تح: بولس برونله، ص: 96.— المصدر
2.2معنى من معاني «فَغِيَ»

وــــــــــ فُلانٌ: مالَ فَمُهُ خِلْقةً. فَهُوَ أَفْغى، وهِيَ فَغْواءُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ القَطّاعِ الصِّقِلِّيُّ (ت: 515هـ=1121م):

"فَغِيَ الرَّجُلُ فَغًى: مالَ فَمُهُ".

ابن القَطَّاع الصقلي، كتاب الأفعال. عالم الكتب،ج: 2، ص: 487.— المصدر
03 —

أَفْغى

فعل
3.1معنى من معاني «أَفْغى»

* أفْغَتِ الشَّجَرَةُ: أزْهَرَتْ وتَفَتَّحَ نَوْرُها.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"أَفْغَتِ الشَّجَرةُ: إذا ‌أخْرَجَتْ ‌فاغِيَتَها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، كتاب العين. تح: المخزومي وآخر، ج: 8، ص: 452.— المصدر
3.2معنى من معاني «أَفْغى»

وــــــــــ: النَّخْلةُ: غَطَّتْها القِشْرةُ التي تُسَمّى قَفَنْدورًا، أيْ: وَقَعَ عَلَيْها الغُبارُ وفَسَدَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأصْمَعِيُّ (ت: 216هـ=831م):

"إذا غَلُظَتِ التَّمْرةُ وصارَ فيها مِثْلُ أَجْنِحةِ الجَرادِ فَذَلِكَ الفَغا، مَقْصورٌ. وقَدْ أَفْغَتِ النَّخْلةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 206.— المصدر
3.3معنى من معاني «أَفْغى»

وــــــــــ فُلانٌ: دامَ على أكْلِ الفَغا، وهُوَ البُسْرُ الفاسِدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابنُ الأَعْرابيِّ (ت: 231هـ=845م):

"أفْغى: إذا دامَ على أكْلِ الفَغا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل والصلة. تح: محمد أبو الفضل إبراهيم، ج: 6، ص: 487.— المصدر
3.4معنى من معاني «أَفْغى»

وــــــــــ: افْتَقَرَ بَعْدَ غِنًى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابنُ الأَعْرابيِّ (ت: 231هـ=845م):

"أَفْغى الرَّجُلُ: إذا افْتَقَرَ بَعْدَ غِنًى ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل والصلة. تح: محمد أبو الفضل إبراهيم، ج: 6، ص: 487.— المصدر
”

وقالَ أَبو بَكْرٍ الصَّنَوْبريُّ (ت: 334هـ=945م):

فَإذا جَنى المَعْروفِ في غُصْنِ امْرِئٍ
أفْغى أمِنْتُ على جَناهُ مِنَ الفَغا

ديوان الصّنوبريّ. تح: إحسان عبّاس، ج: 1، ص: 302.— المصدر
3.5معنى من معاني «أَفْغى»

وــــــــــ: سَمُجَ بَعْدَ حُسْنٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابنُ الأَعْرابيِّ (ت: 231هـ=845م):

"أفْغى: إذا سَمُجَ بَعْدَ حُسْنٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل والصلة. تح: محمد أبو الفضل إبراهيم، ج: 6، ص: 487.— المصدر
3.6معنى من معاني «أَفْغى»

وــــــــــ: عَصى بَعْدَ طاعةٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابنُ الأَعْرابيِّ (ت: 231هـ=845م):

"أفْغى: إذا عَصى بَعْدَ طاعةٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الصغاني، التكملة والذيل والصلة. تح: محمد أبو الفضل إبراهيم، ج: 6، ص: 487.— المصدر
3.7معنى من معاني «أَفْغى»

وــــــــــ النَّجْمُ: صارَ على رَأْسِ الرَّجُلِ، فَرَفَعَ إلَيْهِ رَأْسَهُ وفَغَرَ فاهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أبو مِسْحَلٍ الأعْرابِيُّ (ت: 230هـ=845م):

"قَدْ أفْغى النَّجْمُ: إذا ‌صارَ ‌على ‌رَأْسِ الرَّجُلِ، فَرَفَعَ إلَيْهِ رَأْسَهُ، فَغَرَ فاهُ".

أبو مسحل الأعرابي، كتاب النوادر. تح: عزّة حسن، ص: 121.— المصدر
3.8معنى من معاني «أَفْغى»

وــــــــــ فُلانٌ غَيْرَهُ: أسْمَعَهُ ما يُغْضِبُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ بَعْضُ العَرَبِ (370هـ=981م، مِنْ شَواهِدِ الأزْهَرِيِّ):

"ما الذي أَفْغَاكَ؟".

[قالَ الأَزْهَرِيُّ: أَفْغَاكَ: أَغْضَبكَ وأَوْرَمَكَ.]
الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 206.— المصدر
3.9معنى من معاني «أَفْغى»

وــــــــــ الطَّعامَ: نَقّاهُ مِنَ الأذى.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ القَطّاعِ الصِّقَلِّيُّ (ت: 515هـ=1121م):

"أفْغَيْتُ الطَّعامَ: نَقَّيْتُهُ مِنَ الفَغا".

ابن القطاع، كتاب الأفعال. عالم الكتب، ج: 2، ص: 487.— المصدر
04 —

فاغِيَة

اسم
4.1معنى من معاني «فاغِيَة»

* الفاغِيةُ: زَهْرُ الحِنّاءِ. (انْظُرْ: ف ع و-ي)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م):

«سَيِّدُ الإدامِ في الدُّنْيا والآخِرَةِ اللَّحْمُ، سَيِّدُ الشَّرابِ في الدُّنْيا والآخِرَةِ الماءُ، وسَيِّدُ الرَّياحينِ في الدُّنْيا والآخِرَةِ الفاغِيةُ».

البيهقي، الجامع لشعب الإيمان. تح: مختار الندوي، ج: 8، ص: 186.— المصدر
”

وقالَ أنَسُ بْنُ مالِكٍ (ت: 93هـ=712م) يُخْبِرُ عَمّا يُحِبُّهُ النَّبِيُّ ﷺ مِنَ الطَّعامِ والطِّيبِ:

"كانَتْ تُعْجِبُهُ الفاغِيةُ، وكانَ أَعْجَبُ الطَّعَامِ إِلَيْهِ الدُّبّاءَ".

ابن الجوزي، جامع المسانيد. تح: علي حسين البواب، ج: 1، ص: 155.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"الفاغِيةُ: نَوْرُ الحِنّاءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، كتاب العين. تح: المخزومي وآخر، ج: 8، ص: 452.— المصدر
”

وقالَ ابْنُ ماسَوَيْهِ (ت: 243هـ=857م) يَذْكُرُ طَريقَةَ صُنْعِ دُهْنِ الفاغِيَةِ:

"تَأْخُذُ مِنْ دُهْنِ الحَلِّ الطَّرِيِّ غَيْرِ المَمْلوحِ ثَلاثَةَ أرْطالٍ، فَاجْعَلْها في طَنْجيرٍ أو قِدْرِ حِجارَةٍ، وخُذْ لِذَلِكَ مِنْ فاغِيةِ الحِنّاءِ وقُلوبِهِ زِنةَ مَنَوَيْنِ فَألْقِهِ فيهِ مَفْروكًا".

النويري، نهاية الأرب. تح: مفيد قميحة وآخرين، ج: 12، ص: 68.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”

قالَ شِهابُ الدّينِ النُّوَيْرِيُّ (ت: 733هـ=1333م) يُخْبِرُ عَنْ كَيْفيّةِ صُنْعِ دُهْنِ التُّفّاحِ:

" تَأْخُذُ مِنْ قُشورِ التُّفّاحِ الشّامِيِّ البالِغِ الطَّرِيِّ رَطْلًا فَتُلْقيهِ في المِياهِ وتَغْليها عَلَيْهِ، ثُمَّ تَمْرُسُهُ مَرْسًا جَيِّدًا، وأنْزِلْهُ عَنِ النّارِ، ثُمَّ أَلْقِ فيهِ أوقِيّةً مِنْ فاغِيَةِ الحِنّاءِ".

ابن سيده، المحكم والمحيط الأعظم. تح: عبدالحميد هنداوي، ج: 6، ص: 65.— المصدر

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385هـ=995م):

"الفَغْوةُ في كُلِّ شَجَرةٍ‏: التَّنْويرُ نَفْسُهُ".

ابن عباد، المحيط في اللغة. تح: محمد حسن آل ياسين، ج: 5، ص: 138.— المصدر
