01 —

فِلَزّ/ فِلِزّ/ فُلُزّ

اسم
1.1معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

* الفِلَزُّ، والفِلِزُّ، والفُلُزُّ: نُحَاسٌ أَبْيَضُ، تُعْمَلُ مِنْهُ القُدورُ العِظامُ المُفْرَغَةُ وغَيْرُها. (يونانِيَةٌ)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ جَريرٌ (ت: 110هـ=728م) يَصِفُ السَّيْفَ:

فقُلْتُ لِلقَوْمِ لا تُدْنوا فِلِزَّكُمُ
مِنْ قَاطِعٍ طَبَقَ الأَعْنَاقِ مَسْمومِ

الزمخشري، أساس البلاغة. تح: عيون السود، ج: 2، ص: 35.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"الفِلِزُّ، والفُلُزُّ: نُحَاسٌ أَبْيَضٌ، يُجْعَلُ مِنْهُ قُدورٌ عِظَامٌ مُفْرَغَةٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 368.— المصدر
1.2معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: مَا يَقْذِفْهُ الكيرُ مِنْ خَبَثِ الحَديدِ ونَحْوِهِ مِمَّا يُذابُ مِنْ جَوَاهِرِ الأَرْضِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ رُؤْبَةُ بْنُ العَجّاجِ (ت: 145هـ=762م):

‌أَجْرَدَ ‌أَو ‌جَعْدَ ‌اليَدَيْنِ ‌جِبْزِ

كَأَنَّمَا صُوِّرَ مِنْ فِلِزِّ

[الجَبِيز: الخُبْزُ اليَابِسُ]
ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 821.— المصدر
1.3معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: جَواهِرُ الأَرْضِ مِنَ الذَّهَبِ والفِضَّةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"والمُهْلُ: ‌الفِلِزُّ؛ وهوَ جَوَاهِرُ الأَرْضِ مِنَ الذَّهَبِ والفِضَّةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 57.— المصدر
1.4معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: الحِجَارَةُ. 

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"الفِلِزُّ: الحِجَارَةُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 368.— المصدر
1.5معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ مِنَ الرِّجالِ: الغَليظُ الشَّديدُ الصَّلْبُ. 

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170هـ=787م):

"وَرَجُلٌ فِلِزُّ: غَليظٌ شَديدٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 7، ص: 368.— المصدر
1.6معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: البَخيلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبَّادٍ (ت: 385هـ=999م):

"وكَذِلِكَ الفُلُزُّ؛ وهو البَخيلُ أَيْضًا ".

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 299.— المصدر
1.7معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: الضَّريبَةُ الّتي يُجَرَّبُ عَلَيْها السُّيوفُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبَّادٍ (ت: 385هـ=999م):

"وكَذِلِكَ الفُلُزُّ؛ وهو البَخِيلُ أَيْضًا. والضَّرِيبَةُ الَّتي ‌يُجَرَّبُ ‌عَلَيْهَا ‌السيُوْفُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 2، ص: 299.— المصدر
1.8معنى من معاني «فِلَزّ/ فِلِزّ/ فُلُزّ»

وــــــــــ: عُنْصُرٌ كيمْيَاويٌّ يَتَمَيَّزُ بِالبَريقِ المَعْدنيِّ والقَابِليَّةِ لِتَوْصيلِ الحَرَارَةِ والكَهْرَبَاءِ.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”

قالَ أَحْمَد مُخْتار عُمَر (ت: 1424هـ=2003م):

"فِلِزُّ: عُنْصُرٌ كيمْيَاويٌّ يَتَمَيَّزُ بِالبَريقِ المَعْدنيِّ والقَابِليَّةِ لِتَوْصيلِ الحَرَارَةِ والكَهْرَبَاءِ".

أحمد مختار عمر، معجم اللغة العربية المعاصرة. ج: 3، ص: 1739.— المصدر
