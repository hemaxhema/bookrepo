  1. القَهْرُ والاطِّهادُ
01 —

ضامَ

فعل
1.1معنى من معاني «ضامَ»

* ضامَ فُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ يَضيمُ ضَيْمًا: ظَلَمَهُ باعْتِداءٍ وقَهْرٍ، أو بانْتِقاصٍ وسَلْبِ حَقٍّ، أو بإهانَةٍ وإذْلالٍ. فهُوَ ضائمٌ، وهِيَ بتاءٍ، (ج) ضامَةٌ. والمَفْعولُ مَضيمٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُهَلْهِلُ (135ق.هـ=490م) يَرْثي أخاهُ كُلَيْبًا:

على أَنْ لَيْسَ عَدْلًا مِنْ كُلَيْبٍ
إذا ما ضيمَ جارُ المُسْتَجيرِ

ديوان المهلهل، تح: القوّال، ص: 40.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيُّ ﷺ (ت: 11هـ=632م) وكانَ في لَيْلَةٍ مُقْمِرَةٍ، يُبَشِّرُ الصَّحابَةَ ﷺ بِرُؤْيَةِ اللهِ ﷺ يَوْمَ القِيامَةِ:

«إنَّكُمْ سَتَرَوْنَ رَبَّكُمْ كَما تَرَوْنَ هَذا القَمَرَ، ‌لا ‌تُضامونَ في رُؤْيَتِهِ».

[لا تُضامونَ في رُؤْيَتِهِ: لا تُظْلَمونَ في ذلِكَ، ولا يَلْحَقُمْ ذُلُّ أو قَهْرٌ. ورُوِيَ الحَديثُ بلَفْظِ: لا تَضامُّونَ؛ أي: لا تَتَزاحَمونَ]
البخاري، الصحيح. تح: الناصر، ج: 9، ص: 127.— المصدر
”وقالَ مالِكُ بْنُ نُوَيْرَةَ (ت: 12هـ=633م) يَذْكُرُ سُرْعَةَ فَرَسِهِ العُنابِ:

تَدارَكَ إرْخاءُ العُنابِ وشَدُّهُ
لَبونَ ابْنِ حُبَّى وهْوَ أسْفانُ كامِدُ

تَدارَكَهُ مَنْ لا يُضامُ حَريمُهُ
ولا هوَ رِعْديدٌ لَدى الحَرْبِ هامِدُ

الغندجاني، أسماء خيل العرب. تح: سلطاني، ص: 170.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّعِ (ت: 142هـ=760م):

"يا نَفْسُ لا تَمَلّي مِنْ عِيادَةِ المَرْضى ومُداواتِهِمْ، واعتَبِري كَيْفَ يَجْهَدُ الرَّجُلُ أنْ يُفَرِّجَ عَنْ مَضيمٍ واحِدٍ كُرْبَةً واحِدَةً ويَستَنْقِذُهُ مِنْها رَجاءَ الأجْرِ".

ابن المقفع، كليلة ودمنة. ص: 69.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ هُتَيْمِلٍ (ت: 696هـ=1297م) يُثْني على مَمْدوحِهِ بالكَرَمِ والشَّهامَةِ:

فَدَيتُكَ ما لِوَفْرِكَ كُلَّ يَومٍ
يُضامُ وما لِجارِكَ لا يُضامُ

[الوَفْرُ: المالُ]
ديوان ابن هتيمل. تح: الشميري، ج: 2، ص: 862.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ مَحْمودُ مَقْديش (ت: 1228هـ=1813م) يَذْكُرُ المُعِزَّ لِدينِ اللهِ ابْنَ باديسَ:

"وجاءَ بِأَنْصارِهِ، فَكانوا بِحَيْثُ يَسْمَعونَ نَئيمَهُ، ويَمْنَعونَهُ مِمَّنْ عَساهُ أنْ يَكيدَهُ أوْ يَضيمَهُ، حَتّى بَلَغَ المهْدِيَّةَ".

مقديش، نزهة الأنظار. تح: الزواري وآخر، ج: 1، ص: 375.— المصدر
1.2معنى من معاني «ضامَ»

وــــــــــ فُلانًا حَقَّهُ: نَقَصَهُ إيّاهُ أو مَنَعَهُ مِنْهُ وقَهَرَهُ عَلَيْهِ.

ق

وآخُذُ للْمَوْلى إذا ‌ضيمَ ‌حَقَّهُ
مِنَ الأَعْيَطِ الآبي إذا ما تمَنَّعا

[الأعْيَطُ: المُتَمَنِّعُ]
الأصمعي، الأصمعيات. تح: شاكر وهارون، ص: 64.— المصدر
ع

”قالَ الخَليلُ (ت: 170هـ=787م):

"الضَّيْمُ: الانْتِقِاصُ، ويُقالُ: ما ضِمْتُ أَحَدًا، ولا ضُمْتُ؛ أي: ما ضامَني أَحَدٌ (...) وضامَهُ في الأمْرِ، ‌وضامَهُ حَقَّهُ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 7، ص: 72.— المصدر
02 —

أَضامَ

فعل
2.1معنى من معاني «أَضامَ»

* أضامَ فُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ: ظَلَمَهُ باعْتِداءٍ وقَهْرٍ، أو بانْتِقاصٍ وسَلْبِ حَقٍّ، أو بإهانَةٍ وإذْلالٍ. لُغَةٌ قَليلَةٌ في: ضامَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ الفَرَزْدَقُ (ت: 110هـ=728م) يَمْدَحُ قَبيلَةَ خَنْدِفٍ ويُدافِعُ عَنْها:

فما تُقْبِلُ الأَحْياءُ مِنْ حُبِّ خِنْدِفٍ

ولَكِنَّ أَطرافَ العَوالي تَصورُها

بِحَقّي أُضيمُ العالِمينَ بِخِندِفٍ
وقَدْ قَهَرَ الأَحْياءَ مِنّا قَهورُها

ديوان الفرزدق. تح: الفاعور، ص: 197.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عُبَيْدُ اللهِ بْنُ عَبْدِ اللهِ بْنِ طاهِرٍ (ت: 300هـ=912م) يَذْكُرُ أَرَقَهُ:

أرِقْتُ وما لَيْلُ ‌المُضامِ بنائِمْ
وقَدْ تَرْقُدُ العَيْنانِ والقَلْبُ ساهِرُ

المرزباني، الموشَّح، تح: شمس الدين، ص: 397.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ أبي المَحاسِنِ الطِّيبيُّ (ت: 710هـ=1310م) يَعِظُ:

لا تَغْبِطَنَّ مُضامًا عَيْشُهُ رَغَدٌ
واغْبِطْ أَبِيًّا وإنْ أَوْدى بِهِ الظَّلَفُ

العمري، مسالك الأبصار. تح: الجبوري، ج: 16، ص: 218.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عُمَرُ بْنُ مُحَمَدٍ اليافيُّ (ت: 1233هـ=1818م) وقَدْ زارَ المَسْجِدَ النَّبَوِيَّ:

حاشَ للهِ أَنْ أَكونَ مُضامًا
بَعْدَ ما جِئتُ لِلمَقامِ العَلِيِّ

وَأَتَيتُ الحِمى بِظَنٍّ جَميلٍ
وسُلوكٍ على الصِّراطِ السَّوِيِّ

ديوان عمر اليافي. ص: 8.— المصدر
03 —

ضَيَّمَ

فعل
3.1معنى من معاني «ضَيَّمَ»

* ضَيَّمَفُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ: بالَغَ في ظُلْمِهِ، باعْتِداءٍ وقَهْرٍ، أو بانْتِقاصٍ وسَلْبِ حَقٍّ، أو بإهانَةٍ وإذْلالٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ طُفَيْلٌ الغَنَوِيُّ (ت: 13ق.هـ=609م) يَصِفُ خَيْلًا:

وسَلهَبَةٌ تَنْضو الجِيادَ كأنَّها
رَداةٌ تَدَلَّتْ مِنْ فُروعِ يَلَمْلَمِ

فَذلِكَ أَحْياها وكُلُّ مُعَمَّمٍ
أريبٍ بِمَنعْ الضَّيْفِ غَيْرِ مُضَيَّمِ

[سَلهَبَةٌ: طَويلَةٌ. تَنْضو: تُجاوِزُ. رَداةٌ: صَخْرَةٌ. المُعَمَّمُ. السَّيِّدُ]— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ العَجّاجُ (ت: 90هـ=709م) في رَجُلٍ حُبِسَ وقُيِّدَ:

فإنْ يَكُنْ لاقى حُيَيًّا بالأَمَمْ

ما إنْ يَفُضُّ الصَّخْرَ مِنْ جُولِ العَلَمْ

فَلَــمْ يَعِشْ مُضَيَّمًــا ولَـــمْ يُضَـــــــــــمْ

بالأَخْذِ والأخْذُ لَهُ ثَأْرُ العِيَمْ

[الأمَمُ: الشَّيْءُ اليَسيرُ. جولُ العَلَمِ: ناحِيَةُ الجَبَلِ. العِيَمُ: خِيارُ الشِّيْءِ]
ديوان العجاج. تح: عزّة حسن، ص: 274.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأصْمَعِيُّ (ت: 216هـ=831م) يَشْرَحُ قَوْلَ طُفَيْلٍ الغَنَوِيِّ: (أريبٍ بِمَنعْ الضَّيْفِ غَيْرِ مُضَيَّمِ):

"المُضَيَّمُ: الَّذي لا يُضامُ".

ديوان طفيل الغنويّ. تح: أوغلي، ص: 110.— المصدر
”وقالَ ابْنُ الرُّومِيِّ (ت: 283هـ=896م) يَمْدَحُ رَجُلًا:

هُوَ المَرْءُ أمّا مالُهُ فمُحَلَّلٌ
لِعافٍ وأمّا جارُهُ فمُحَرَّمُ

لجيرانِهِ مِنْهُ مَحَلٌّ مُمَنَّعُ
يَضيمُ بِهِ الدَّهْرَ الذَّليلُ المُضَيَّمُ

ديوان ابن الرومي. تح: بسج، ج: 3، ص: 217.— المصدر
04 —

اِسْتَضامَ

فعل
4.1معنى من معاني «اِسْتَضامَ»

* اسْتَضامَفُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ: ظَلَمَهُ باعْتِداءٍ وقَهْرٍ، أو بانْتِقاصٍ وسَلْبِ حَقٍّ، أو بإهانَةٍ وإذْلالٍ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ حاتِمٌ الطّائِيُّ (ت: 46ق.هـ=577م) يَرْفُضُ الذُّلَّ والإهانَةَ:

يُسَائِلُني النُّعْمانُ كَيْ يَسْتَزِلَّني
وهَيْهاتَ لي أَنْ أُسْتَضامَ فأُصْرَعا

ديوان حاتم الطائي. ص: 117.— المصدر
س

"ما أضاعَكَ مَنْ سَوَّدَكَ، ولا خابَ فيكَ أمَلُ مَنْ أمَّلَكَ، أَنْتَ مَلِكٌ لا تُسْتَضامُ، تَمْنَعُ حِماكَ وتَصونُ رَعِيَّتَكَ".

محمد كرد علي، خطط الشام. ج: 1، ص: 133.— المصدر

فهُمُ شَتَّى قَتيلٌ صَريعٌ
دَمُهُ فيهِمْ حِذارًا يُطَلُّ

وأَسيرٌ في طِمارٍ عَلَيْهِ
مِنْ حَديدِ القَيْنِ كَبْلٌ وغُلُّ

الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”رُوِيَ أنَّ النَّبِيَّ ﷺ (8هـ=629م) قالَ في كِتابِهِ إلى قَوْمِ لُكَيْزِ بْنِ عَبْدِ القَيْسِ:

"وأهْلُ البَحْرَيْنِ خُفَراؤُهُ مِنَ الضَّيْمِ، وأعْوانُهُ على الظّالِمِ، وأنْصارُهُ فِي المَلاحِمِ".

ابن سعد، الطبقات. تح: علي محمد عمر، ج: 1، ص: 245.— المصدر
”وقالَ عَبْدُ اللهِ بْنُ عَنَمَةَ الضَّبِّيُّ (ت: 15هـ=636م) يَفْخَرُ:

إنْ تَسْأَلوا الحَقَّ نُعْطي الحَقَّ سائِلَهُ
والدِّرْعُ مُحْقَبَةٌ والسَّيْفُ مَقْروبُ

أَوْ تَأْنَفون فَإنّا مَعْشَرٌ أُنُفٌ
لا نُطْعَمُ الضَّيْمَ إنَّ السُّمَّ مَشْروبُ 

ابن أبي الحديد، شرح نهج البلاغة. تح: النمري، ج: 1، ص: 926.— المصدر

”قالَ ابْنُ عَرَفَةَ الدُّسوقِيُّ (ت: 1230هـ=1815م) يَشْرَحُ قَوْلَ مَعْنٍ المُزَنِّيِّ: (وَيَرْكَبُ حَدَّ السَّيْفِ ‌مِنْ ‌أَنْ ‌تَضيمَهُ … إذا لَمْ يَكُنْ عَنْ صَفْحَةُ السَّيْفِ مَزْحَلُ):

"والمَعنى: ويَرْكَبُ الأُمورَ الشّاقَّةَ الَّتي تُؤَثِّرُ فيهِ تَأْثيرَ السَّيْفِ؛ مَخافَةَ أنْ يَلْحَقَهُ ‌الضَّيْمُ والعارُ، مَتى لَمْ يَجِدْ عَنْ رُكوبِها بُعْدًا".

07 —

ضِيم

اسم
7.1معنى من معاني «ضِيم»

* الضِّيمُ: النّاحِيَةُ مِنَ الجَبَلِ أَوِ الأَكَمَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ ساعِدَةُ بْنُ جُؤَيَّةَ (ت: 11هـ=632م) يَصِفُ خَلِيَّةَ عَسَلٍ:

وما ضَرَبٌ بَيْضاءُ يَسْقي دَبوبَها
دُفاقٌ وعَرْوانُ الكَراثِ فَضيمُها

أُتِيحَ لها شَثْنُ البَنانِ مُكدَّمٌ
أَخو حُزَنٍ قد وَقَّرَتْهُ كُلوُمُها

[الضَّرَبُ: العَسَلُ الأبْيَضُ. الدَّبُوبُ: الغَوْرُ. دُفاقٌ وعَرْوانُ: مَوْضِعانِ. الكَراثُ: نَوْعُ شَجَرٍ. الشَّثْنُ البَنانِ: خَشِنُ الأصابِعِ. المكدَّمُ: مُتَآكِلُ الأظْفارِ. الحُزَنَ: المكانُ الغَليظُ. وفُسِّرَ الضِّيمُ في البَيْتِ أيْضًا بأنَّهُ اسْمُ وادٍ بعَيْنِهِ]— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْد (321هـ=933م):

"والضّيمُ: ‏ناحِيَةٌ من الجَبَلِ أَو الأكَمَةِ، تَقولُ: قَعَدْتُ في ضيمِ الأكَمَةِ وفي ضيمِ الجَبَلِ؛ أَيْ: في ناحِيَتِهِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 2، ص: 912.— المصدر
