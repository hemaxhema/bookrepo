  1. الغَلَبةُ والعُلُوّ
01 —

قَهَرَ

فعل
1.1معنى من معاني «قَهَرَ»

* قَهَرَ (بِفَتْحِ الهاءِ) فُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ يَقْهَرُ (بفَتْحِ الهاءِ) قَهْرًا: غَلَبَهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ المُهَلْهِلُ التَّغْلبيّ (ت: 100ق.ه=525م) يَتَوَعَّدُ مَنْ قَتَلَ أَخاهُ كُلَيْبًا:

قَتَلوا كُلَيْبًا ثُمَّ قالوا إِرْبَعوا
كَذَبوا ورَبِّ الحِلِّ والإحْرامِ

حتّى نُبيدَ قَبيلَةً وقَبيلَةً
قَهْرًا ‌ونَفْلِقَ ‌بالسُّيوفِ الهامِ

[ورَبَعَ عنه: كَفَّ عنه، واربَعُوا معناه: كُفُّوا عنهم، وهمزتُهُ وصلٌ، وقُطِعَتْ همزتُهُ في البيتِ لضَرُورةِ الوَزْنِ]
الأصمعي، الأصمعيات. تح: مجموعة محققين، ص: 156.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِيّ ﷺ (ت: 11ه=632م) يَحْكي ما تَقولُهُ يَأْجوجُ ومَأْجوجُ:

«فيَقولونَ: قَهَرْنَا أَهْلَ الأَرْضِ وعَلَوْنا أَهْلَ السَّماءِ».

أحمد بن حنبل، المسند. تح: الأرنؤوط وآخرين، ج: 16، ص: 369.— المصدر
”في القُرْآنِ الكَريمِ (11ه=632م) قال اللهُ ﷻ:

﴿فَأَمَّا ‌ٱلۡيَتِيمَ ‌فَلَا ‌تَقۡهَرۡ﴾ [الضحى: 9]

[الطَّبريّ: فَلا تَظْلِمْهُ، فَتَذْهَبَ بِحَقِّهِ، اسْتِضْعافًا مِنْكَ لَهُ]
”وقالَ ابْنُ عَبّاس (ت: 68ه=688م) يَذْكُرُ سَبَبِ نُزولِ قَوْلِه تَعالى: ﴿وَمَن لَّمۡ يَحۡكُم ‌بِمَآ ‌أَنزَلَ ‌ٱللَّهُ فَأُوْلَٰٓئِكَ هُمُ ٱلۡكَٰفِرُونَ﴾ [المائدة: 44]: 

"أَنْزَلَها اللهُ في الطّائفَتَيْنِ مِنَ اليَهودِ، وكانَتْ إحْداهُما قَدْ ‌قَهَرَتِ الأُخْرى في الـجاهِلِيَّةِ".

أحمد بن حنبل، المسند. تح: الأرنؤوط وآخرين، ج: 4، ص: 88.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ المُقَفَّع (ت: 142ه=760م) مُتَرْجِمًا قَوْلَ الإسْكَنْدَرِ لِمَلِكِ الهِنْدِ في دَعْوَتِهِ لَهُ إلى المُبارَزَةِ:

"فأيُّنا ‌قَهَرَ صاحِبَه فَهُوَ الأَسْعَدُ".

ابن المقفع، كليلة ودمنة. ص: 13.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال العِزُّ بن عَبْد السّلام (ت: 660ه=1262م):

"اليَدُ يُعَبَّرُ بِها عن ‌القَهْرِ والاسْتيلاءِ، كَقَوْلِهِ: قُلْ لِمَنْ في أَيْديكُمْ مِنَ الأَسْرى؛ أَيْ: في ‌قَهْرِكُمْ واسْتيلائكُمْ ".

العز، الإمام في بيان أدلة الأحكام. تح: ابن غربية. ج: 1، ص: 242.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال صالِـح الفُلّانيُّ (ت: 1218ه=1803م) يَنْعى على المُقَلِّدَةِ تَرْكَهُمْ بَعْضَ الأَحاديثِ الصَّحيحَةِ تَقْليدًا:

"والمَقْصودُ أَنَّ التَّقْليدَ حَكَمَ عَلَيْكُمْ بذلك وقادَكُمْ إلَيْهِ ‌قَهْرًا، ولَو حَكَّمْتُمُ الدَّليلَ على التَّقْليدِ لَمْ تَقَعوا في مِثْلِ هذا ".

الفلاني، إيقاظ همم أولي الأبصار. ج: 1، ص: 142.— المصدر
1.2معنى من معاني «قَهَرَ»

وــــــــــ قُهْرَةً: أَخَذَهُ غَلَبَةً واضْطِرارًا مِنْ غَيْرِ رِضاهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"يُقالُ: أَخَذَهُمْ قَهْرًا؛ أي: مِنْ غَيْرِ رِضاهُمْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، ج: 3، ص: 365.— المصدر
1.3معنى من معاني «قَهَرَ»

وــــــــــ فُلانٌ وغَيْرُهُ اللَّحْمَ: وَضَعَهُ على النّارِ فَسالَ ماؤهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأزْهَريّ (ت: 370ه=981م):

"‌قَهَرْنا اللَّحْمَ ‌نَقْهَرُه: وذَلِكَ أَوَّلَ ما تَأْخُذُ فيهِ النّارُ فيَسيلُ ماؤهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 5، ص: 395.— المصدر
1.4معنى من معاني «قَهَرَ»

o وقَهَرَ غَرائِزَهُ: سَيْطَرَ عَلَيْها.

ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَحْمَد مُخْتار عُمَر (ت: 1424ه=2003م):

"قَهَرَ ‌غَرائِزَهُ: سَيْطَرَ عَلَيْها".

أحمد مختار عمر، معجم اللغة العربية المعاصرة. ج:3، ص: 1866.— المصدر
02 —

قُهِرَ

فعل
2.1معنى من معاني «قُهِرَ»

* قُهِرَ اللَّحْمُ: أخَذَتْهُ النّارُ فَسالَ ماؤهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال مُضرِّسُ بن رِبْعيّ (ت: 110ه=728م):

فَلَمّا أَنْ تَلَهْوَجْنا شِواءً
بِهِ اللَّهَبانُ مَقْهورًا ضَبيحًا

الزمخشري، أساس البلاغة. تح: عيون السود، ج: 2، ص: 112.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الفارابيُّ (ت: 350ه=961م):

"يُقال: ‌قُهِرَ اللَّحمُ: إذا أخَذَتْهُ النّارُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الفارابي، معجم ديوان الأدب. تح: أحمد مختار عمر، ج: 2، ص: 206.— المصدر
”وقالَتِ العَرَبُ (538ه=1143م) مِنْ شَواهِدِ الزَّمَخْشَريّ:

"أَطْعَمَنا خُبْزَةً بلَحْمٍ مَقْهورٍ، وشَحْمٍ مَصْهورٍ".

الزمخشري، أساس البلاغة. تح: عيون السود، ج: 2، ص: 112.— المصدر
2.2معنى من معاني «قُهِرَ»

وــــــــــ الإنْسانُ: غُلِبَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القَطّاعِ الصَّقَلِّيُّ (ت: 515ه=1121م):

"قُهِرَ الإنسانُ: غُلِبَ".

ابن القطاع، الأفعال. تح: محمد السورتي وغيره. ج: 3، ص: 13.— المصدر
03 —

أَقْهَرَ

فعل
3.1معنى من معاني «أَقْهَرَ»

* أَقْهرَ فُلانٌ وغَيْرُهُ: صارَ هُوَ أوْ أَصْحابُهُ في ذِلَّةٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال الـمُخَبَّلُ السَّعْديّ (ت: 20ه=641م) يَهْجو الزِّبْرِقانَ بنَ بَدْرٍ:

تَمَنّى حُصَيْنٌ أَنْ يَسودَ جِذاعَهُ
فأَمْسى حُصَيْنٌ قَدْ أَذَلَّ وأَقْهَرا

[حُصَيْنٌ: اسمُ الزِّبْرِقان. وجِذاعُه: قَوْمُه مِنْ تَميمٍ]
الزجاج، معاني القرآن. تح: شلبي. ج: 4، ص: 309.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال القاسم بن سلاّم (ت: 224ه=839م) يَشْرَحُ رِوايَةَ الأَصْمَعِيِّ لِبَيْتِ المُخَبَّلِ السَّعْدِيِّ:

"يَرْويهِ الأَصْمَعِيُّ: قَدْ أَذَلَّ وأَقْهَرا؛ أَيْ: قَدْ صارَ أَصْحابُهُ أَذِلّاءَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلاّم، الغريب المصنف. تح: داوودي، ج: 2، ص: 37.— المصدر
3.2معنى من معاني «أَقْهَرَ»

وــــــــــ فُلانٌ وغُيْرُهُ فُلانًا وغَيْرَهُ: وَجَدَه مَقْهُورًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”وبِهِ رُوي ما قال الـمُخَبَّلُ السَّعدي (ت: 20ه=641م) يَهْجو الزِّبْرِقانَ بنَ بَدْرٍ:

تَمَنّى حُصَيْنٌ أَنْ يَسودَ جِذاعَهُ
فَأَمْسى حُصَيْنٌ قَدْ أُذِلَّ وأُقْهِرا

[في التَّهْذيبِ بَعْدَ إِنْشادِهِ البَيْتَ: على ما لَمْ يُسَمَّ فاعلُهُ؛ أي: وُجِدَ كذلك ]
القاسم بن سلاّم، الغريب المصنف. تح: صفوان داوودي، ج: 2، ص: 37.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الكِسائيّ (ت: 189ه=805م):

"أَقْهَرْناهُ: وَجَدْناهُ مَقْهورًا ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

القاسم بن سلاّم، الغريب المصنف. تح: صفوان داوودي، ج: 2، ص: 37.— المصدر
3.3معنى من معاني «أَقْهَرَ»

وــــــــــ: غَلَبَهُ واضْطَرَّهُ إلى ما لا يَرْضاهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال وَكيعُ بن الجَرّاح (ت: 197ه=812م):

"استُعْمِلَ ابنُ هُبَيْرَةَ على الكوفَةِ، فَدَعا مَنْصورَ بنَ الـمُعْتَمِرِ، فَأَرادَ أَنْ يَسْتَقْضِيَهُ، فَأَبى، فَقالَ: إنْ فَعَلْتَ، وإلّا صَنَعْتُ بِكَ ‌وصَنَعْتُ. ‌فأَقْهَرَهُ".

خالد الرباط وسيد عزت، الجامع لعلوم الإمام أحمد. ج: 19، ص: 227.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ تَغْري بَرْدي (ت: 874ه=1470م):

"وكانَ مِنْ خَبَرِ الـمَلِكِ الظّاهِرِ بَرْقوق: أنَّ مِنْطاشَ لَمّا وَثَبَ على الأَميرِ ‌وأَقْهَرَ الأَتابِكَ يَلْبُغا الناصِريَّ وحَبَسَهُ، وحَبَسَ عِدَّةً مِنْ أَكابِرِ الأُمَراءِ؛ عاجَلَ في أَمْرِ الـمَلِكِ الظّاهِرِ بَرْقوقَ بِأَنْ بَعَثَ إليه شَخْصًا يُعْرَفُ بالشِّهابِ البَريدِيّ".

ابن تغري بردي، النجوم الزاهرة. تح: شمس الدين ، ج: 11، ص: 287.— المصدر
04 —

قاهَرَ

فعل
4.1معنى من معاني «قاهَرَ»

* قاهَرَ فُلانٌ وغَيْرُهُ فُلانًا وغَيْرَهُ: غالَبَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال مُـحَمَّدُ بن حَبيب الهاشِميّ (ت: 245ه=860م):

"الـمُقاسَرَةُ: ‌الـمُقاهَرَةُ، والقَسْرُ: القَهْرُ".

محمد بن حبيب، شرح ديوان جرير. تح: طه. ج: 1، ص: 128.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ المُطَهَّر الحِلِّي (ت: 726ه=1325م) يَحْتَجُّ لِمَذْهَبِ الشّيعَةِ في القَدَرِ:

"فَكَيْفَ تُكَلِّفُني الإيمانَ ولا قُدْرَةَ لي عَلَيْهِ؟ بَلْ خَلَقَ فِيَّ الكُفْرَ، وأَنا لا أَتَمَكَّنُ مِنْ ‌مُقاهَرَةِ اللهِ تَعالى".

[نَصُّ كَلامِ الحِلِّي في: منهاج الكرامة، ص: 40. تح: عبد الرحيم مبارك]
ابن تيمية، منهاج السنة النبوية. تح: محمد رشاد سالم، ج: 3، ص: 54.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال عَبِدُ النّاصِر ميلاد (1426ه=2005م) تاريخُ الطَّبْع، يُبَيِّنُ حِكْمَةَ مَشْروعِيَّةِ البَيْعِ:

"الإنْسانُ بِمُفْرَدِهِ لا يَسْتَطيعُ تَوْفيرَ حاجِيّاتِهِ مِنَ الغِذاءِ والكِساءِ وغَيْرِها، وقد يَجْنَحُ بِمُقْتَضى حاجَتِهِ الـمُلِحَّةِ إلى أَخْذِ ما في يَدِ الغَيْرِ عَنْ طَريقِ الـمُغالَبَةِ أو ‌الـمُقاهَرَةِ، أو يَلْجَأُ إلى السُّؤالِ وتَكَفُّفِ أَيْدي النّاسِ، وفي ذلك مِنَ الـمَفاسِدِ العِظامِ ما لا يَخْفى".

عبد الناصر ميلاد، البيوع المحرمة والمنهي عنها. ج: 1، ص: 22.— المصدر

”قالتْ بَنو أَسَد (ت: 2ق.ه=621م) في تَلْبِيَتِهِمْ في الجاهِلِيَّةِ:

رَبَّنَا أَقْبَلَتْ بَنو أَسَدْ

الواحِدُ ‌القَهّارُ والرَّبُّ الصَمَدْ

لا نَعْبُدُ الأَصْنامَ حَتّى تَجْتَهِدْ

قطرب، كتاب الأزمنة وتلبية الجاهلية. تح: حداد، ص: 119.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11ه=632م) قال اللهُ ﷻ:

﴿لِّمَنِ ٱلۡمُلۡكُ ٱلۡيَوۡمَۖ لِلَّهِ ٱلۡوَٰحِدِ ‌ٱلۡقَهَّارِ﴾ [غافر: 16]

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو إِسْحاقَ الزَّجّاجُ (ت: 311هـ=923م) يُعَدِّدُ أَسْماءَ اللهِ الحُسْنى ويُفَسِّرُها:

"القَهّارُ: القَهْرُ في وَضْعِ العَرَبيَّةِ الرّياضَةُ والتَّذْليلُ؛ يُقالُ: قَهَرَ فُلانٌ النّاقَةَ؛ إِذا راضَها وذَلَّلَها ...".

الزجاج، تفسير أسماء الله الحسنى. تح: أحمد الدقاق، ص: 38.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ القَيِّمِ (ت:751ه=1350م) يَرُدُّ أَقوالَ بَعْضِ الفِرَقِ المُنْحَرِفَةِ:

"يَسْتَحيلُ أَنْ يَكونَ لِمَنْ لَهُ المَثَلُ الأَعْلى مِثْلٌ أَوْ نَظيرٌ، وهذا بُرْهانٌ قاطِعٌ ... ونَظيرُ هذا: القَهْرُ المُطْلَقُ مَعَ الوَحْدَةِ فَإِنَّهُما مُتَلازِمانِ، فَلا يَكُونُ القَهّارُ إِلّا واحِدًا؛ إِذْ لَوْ كانَ مَعَهُ كُفُؤٌ لَهُ فَإِنْ لَمْ يَقْهَرْهُ لَمْ يَكُنْ قَهّارًا عَلى الإِطْلاقِ، وإِنْ قَهَرَهُ لَمْ يَكُنْ كُفُؤًا، وكانَ القَهّارُ واحِدًا ...".

ابن القيم، الصواعق المرسلة. تح: علي الدخيل الله، ج: 3، ص: 1032.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ عَبْدُ الرَّحْمنِ السّعْديُّ (ت: 1376ه=1957م) يُفَسِّرُ:

"والقَهّارُ لا يَكونُ إِلّا واحِدًا، وذلِكَ يَنْفي الشَّرِكَةَ لَهُ مِنْ كُلِّ وَجْهٍ".

السعدي، تيسير الكريم الرحمن. تح: اللويحق، ص: 719.— المصدر

قَهيرَة

اسم
14.1معنى من معاني «قَهيرَة»

* القَهيرَةُ: اللَّبَنُ الخالِصُ يُلْقى فيهِ الحِجارَةُ الـمَحْمِيَّةُ، فإذا غَلى ذُرَّ عَلَيْهِ الدَّقيقُ وخُلِطَ بِهِ، ثُمَّ أُكِلَ. (انظر: ف ه ر).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الطّائيّ (ت: 190ه=806م) من شُيوخ أبي عَمْرو:

"‌القَهيرَةُ: مَحْضٌ يُلْقى فيهِ الرَّضْفُ، فإذا غَلى ذُرَّ عَلَيْهِ الدَّقيقُ وسِيْطَ بِهِ، ثُمَّ أُكِلَ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[الرَّضْفُ: الحِجارَةُ المَحْمِيَّةُ. سيط: خُلِطَ. ‌الـمَحْضُ: اللَّبَنُ الخالِصُ الذي لَمْ يُخالِطْهُ الـمَاءُ حُلْوًا كانَ أو حامِضًا]
الأزهري، تهذيب اللغة. تح: عبد الله درويش، ج: 5، ص: 395.— المصدر
