  1. الـمَنْعُ والشُّحُّ
01 —

بَخِل

فعل
1.1معنى من معاني «بَخِل»

* بَخِلَ(بِكَسْرِ الخاءِ) فُلانٌ يَبْخَلُ (بِفَتْحِ الخاءِ) بَخْلًا، وبَخَلًا، وبُخُلًا، وبُخْلًا: مَنَعَ فَضْلَهُ وكَرَمَهُ. فهُوَ باخِلٌ، وبَخيلٌ، وبَخالٌ وبَخّالٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال خُزَيْمةُ القُضاعِيُّ (ت: 382ق.ه=241م):

فَتاةٌ كَأَنَّ رُضابَ العَبيرِ
بِفيها، يُعَــــــلُّ بِهِ الزَّنْجَبيلُ

قَتَلْتُ أَباها على حُبِّها
فَتَبْخَلُ إنْ بَخِلَتْ أَوْ تُنيلُ

ابن حمدون، التذكرة الحمدونية. تح: إحسان عباس، ج: 7، ص: 361.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11ه=632م) قال اللهُ :

﴿ومَنْ يبْخَلْ فَإنَّما يَبْخَلُ عَنْ نَفْسِهِ﴾. (محمّد: 38)

”وقال النَّبِيُّﷺ (ت: 11ه= 632م):

«اللهُمَّ إِنّي أَعوذُ بِكَ مِنَ الكَسَلِ، وأَعوذُ بِكَ مِنَ الجُبْنِ، وأَعوذُ بِكَ مِنَ الهَرَمِ، وأَعوذُ بِكَ مِنَ البُخْلِ».

صحيح البخاري. تح: نصّار، ج 4، ص: 56— المصدر
”وقال عُرْوةُ بْنُ حِزامٍ (ت: 30ه=650م):

ولا تُذْكَرُ الأَهْواءُ إلّا ذَكَرْتُها
ولا البُخْلُ إلّا قُلْتُ سَوْفَ تُثيبُ

ديوان عروة بن حزام. تح: أَحْمد عكيدي، ص:155.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال البَعيثُ المُجاشِعِيُّ (ت: 134ه=751م):

أَلا أَصْبَحَتْ خَنْساءُ جاذِمةَ الوَصْلِ
وضَنَّتْ عَلَيْنا والضَّنِينُ مِنَ البُخْلِ

أبو علي الفارسي، الحجة للقراء السبعة. تح: قهوجي، ج: 7، ص:165.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ سِوار(ت: 677ه=1278م):

مَنَنْتُمْ بِما لا يَمْلِكُ الحَرْفُ وَصْفَهُ
وجُدْتُمْ بِما جودُ الوَرى عِنْدَهُ بُخْلُ

ديوان ابن سوار. تح: محمّد أديب الجادر، ص: 363.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال مُحَمَّدُ بْنُ الطُّلْبةِ اليَعْقوبِيُّ (ت: 1272ه=1856م):

فَلا تَبْخَلَنَّ ولا تَعْتَبَنْ
ولا تَغْدِرَنْ أُفِّ لِلغادِرِ

ديوان اليعقوبي. تح: ابن ابوه، ص: 231.— المصدر
02 —

بَخُل

فعل
2.1معنى من معاني «بَخُل»

* بَخُلَ(بِضَمِّ الخاءِ) فُلانٌ يَبْخُلُ (بِضَمِّ الخاءِ) بُخْلًا، وبُخَلًا، وبُخُلًا، وبَخَلًا: بَخِلَ. فهُوَ بَخيلٌ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال عَنْتَرةُ (ت: 22ق ه=600م):

فَسَطا عَلَيَّ الدَّهْرُ سَطْوةَ غادِرٍ
والدَّهْرُ يَبْخُلُ تارةً ويَجودُ

شرح ديوان عنترة للتبريزيّ. تح: مجيد طراد، ص: 56.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَتْ لَيْلى الأَخيلِيّةُ (ت: 80ه=698م):

وكُنْتَ الـمُرْتَجى وبِكَ اسْتَغاثَتْ
لِتُنْعِشَها؛ إذا بَخُلَ السَّحابُ

ديوان ليلى الأخيلية. تح: واضح الصمد، ج: 1، ص: 178.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال الفيْروزابادِيُّ (ت: 817ه= 1414م):

"وبَخِلَ كَفَرِحَ وكَرُمَ بُخْلًا". (وهَذِهِ الدِّلالةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَدِيثِ)

الفَيْروزابادي، القاموس. تح: العرقسوسي، ص: 965.— المصدر
03 —

أَبْخَل

فعل
3.1معنى من معاني «أَبْخَل»

* أَبْخَلَفُلانٌ فُلانًا: وَجَدَهُ بَخيلًا.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال عَمْرُو بْنُ مَعْدِ يْكَرِبَ (ت: 21ه=642م):

"قاتَلْناكُمْ فَما أَجْبَناكُمْ، وسَأَلْناكُمْ فَما أَبْخَلْناكُمْ".

ابن قتيبة، أدب الكاتب. تح: علي فاعور، ج: 1، ص: 292.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الشّاعِرُ (458ه=1066م، مِن شَواهِدِ ابنِ سِيْدَهْ):

ولا مُعِدَّ بُخْلهِ عَنْ إبْخالْ

ابن سيده، المحكم. تح: هنداوي، ج: 5، ص: 211.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ مَنْظورٍ (ت: 711ه=1311م):

"وأَبْخَلَهُ: وَجَدَهُ بَخيلًا".

ابن منظور، اللسان. تح: أمين عبد الوهاب وآخر، ج: 1، ص: 332.— المصدر
04 —

بَخَّل

فعل
4.1معنى من معاني «بَخَّل»

*بَخَّلَفُلانٌ فُلانًا: نَسَبَهُ إلى البُخْلِ ورَماهُ بِهِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قال عَبْدُ المُطَّلِبِ بْنُ هاشِمٍ (ت: 45ق.ه=578م):

"اللَّهُمَّ سادَّ الخَلّةِ وكاشِفَ الكُرْبةِ، أَنْتَ عالِمٌ غَيْرُ مُعَلَّمٍ، مَسْؤولٌ غَيْرُ مُبَخَّلٍ (...)، اللَّهُمَّ وأَمْطِرْ عَلَيْنا غَيْثًا مَريعًا مُغْدِقًا".

ابن حبيب، المنمّق في أخبار قريش. تح: خورشيد فاروق، ص: 146.— المصدر
س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”‌‌قال الحَزينُ الدّيلِيُّ (ت: 90ه=709م):

إلَيْكَ ابْنَ عُثْمانَ بْنِ عُثْمانَ عاصِمَ بْـ
ـنَ عمرو سَرَتْ عَنْسي فخابَ سُراها

فَقَدْ صادَفَتْ كَــــــــزَّ اليَـــــــــدَيْنِ مُبَخَّــــــلًا
جَبــــــــانًا إذا مــــا الحَـــــرْبُ شُـــبَّ لَــــــظاها

بَـخــــيلًا بِمـــــا في رَحْلِــــــــــهِ غَيْرَ أَنَّـــــــــــهُ
إذا مـــــــــا خَلَــــــــــتْ عِــــــــرْسُ الخَليـــــلِ أَتـــاهـــــــــا

الأصبهاني، الأغاني. تح: الطويل، ج: 15، ص: 329.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ الرّومِيِّ (ت: 283ه=896م):

فَكَمِ اجْتُدِيتَ فَما وُجِدْتَ مُبَخَّلًا
وكَمِ امْتُحِنْتَ فَما وُجِدْتَ جَزوعا

ديوانه. تح: أحمد بسج، ج: 2، ص: 92.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال الوَطْواطُ (ت: 718ه=1318م):

"وكانَ مُحمَّدُ بْنُ يَحْيى بْنِ خالِدٍ مُبَخّلًا بالنِّسْبَةِ لِأبيهِ وأَخَوَيْهِ جَعْفَرٍ والفَضْلِ".

الوطواط، غرر الخصائص الواضحة. تح: شمس الدين، ج: 1، ص: 37.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال مُحَمَّد عَبْد الخالِقِ عُضيمة (ت: 1392ه=1972م):

"وكَذَلِكَ إنْ ذَمَمتَهُ ووَصَفْتَهُ بِالضّيقِ؛ قُلْتَ:‏ سأَلْناهُ وكان إنْسانًا، وتُزْوي وَجْهَكَ وتُقَطِّبُهُ، فَيُغْني ذَلِكَ عَنْ قَوْلِكَ: إنْسانًا لَئيمًا أَوْ لَحِزًا أَوْ مُبَخَّلًا، أَوْ نَحْوُ ذَلِكَ".

محمد عضيمة، دراسات لأسلوب القرآن الكريم. ج: 10، ص: 178.— المصدر
4.2معنى من معاني «بَخَّل»

وــــــــــــ: حَمَلَهُ على البُخْلِ ودَعاهُ إلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال النَّبِيُّ ﷺ (ت: 11ه= 632م):

«واللهِ إِنَّكُمْ ‌لَتُبَخِّلونَ وتُجَبِّنونَ وتُجَهِّلونَ».

مسند ابن راهويه، تح: عبد الغفور البلوشي، ج: 5، ص: 76.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ المُقَفَّعِ (ت: 142ه=759م):

"اِعْلَمْ أَنَّ الـسُّلْطانَ يَقْبَلُ مِنَ الوُزَراء التَّبْخيلَ، ويَعُدُّهُ مِنْهمُ شَفَقةً ونَظَرًا، ويَحْمَدُهُمْ عَلَيْهِ؛ فَإنْ كانَ جَوادًا وكُنْتَ مُبَخَّلًا؛ شِنْتَ صاحِبَكَ بِفَسادِ مُروءَتِهِ، وإنْ كُنْتَ مُسْخِيًا؛ لَمْ تأْمَنْ إضْرارَ ذَلِكَ بِمَنْزِلَتِكَ عِنْدَهُ".

ابن المقفع، الأدب الكبير والأدب الصغير، ص: 94.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال الزَّبيدِيُّ (ت: 1205ه=1790م)

"وبَخَّلَهُ تَبْخيلًا: رماهُ بهِ أَوْ نَسَبَهُ إلَيْهِ، أوْ جَعَلَهُ بَخيلًا".

الزبيديّ، التاج. تح: الطناجيّ، ج: 28، ص: 63.— المصدر
4.3معنى من معاني «بَخَّل»

وــــــــــــ فُلانٌ الشْيْءَ: تَوَقَّعَهُ.

د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ شَرَفٍ (ت: 656ه=1258م):

يَرى البَلِيدُ البَلايا بَعْدَما نَزَلَتْ
وذُو الذَّكاءِ يَرى الأَشْياءَ تَبْخيلا

اليوسي، زهر الأكم في الأمثال والحكم. تح: محمّد حجي، ج: 2، ص: 247.— المصدر
05 —

تَبَخَّل

فعل
5.1معنى من معاني «تَبَخَّل»

*تَبَخَّلَ فُلانٌ: بَخِلَ وتَكَلَّفَ البُخْلَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال رَبيعةُ بْنُ مَقْرومٍ (ت: 16ه=631م):

هَلْ نُكْرِمُ الأَضْيافَ إنْ نَزَلوا بِنا
ونَسودُ بِالـمَعْروفِ غَيرَ تَبَخُّلِ

شعر ربيعة بن مقروم. تح: تماضر عبد القادر وآخر، ج: 1، ص:198.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال أبُو مُحَلِّمٍ الشَّيْبانِيُّ (ت: 245ه=860م):

أَعْطى الَّذي أَعْطى أبوهُ قَبْلَهُ
وتَبَخَّلَتْ أَبْناءُ مَنْ يَتَبَخَّلُ

التادلي، الحماسة المغربية. تح: الداية، ج: 1، ص: 168.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال أَحْمَدُ عَبْد اللهِ الدّامِغ (1423ه=2003م):

ومَنْ تِلْكُمُ الحَسْناءُ يَقْصُرُ عِنْدَها
طَويلٌ ويَسْخو عِنْدَها الـمُتَبَخِّلُ

أحمد الدّامغ، الأدب المثمّن. ج: 3، ص: 186.— المصدر
06 —

اسْتَبْخَل

فعل
6.1معنى من معاني «اسْتَبْخَل»

*اسْتَبْخَلَفُلانٌ فُلانًا: عَدَّهُ بَخيلًا.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال الشَّريفُ الرَّضِيُّ (ت: 406ه=1015م) يُفَسِّرُ قَوْلَهُ تَعالى: ﷺبَلْ يَداهُ مَبْسوطَتانِ يُنْفِقُ كَيْفَ يَشاءُﷺ (المائدة: 64):

"وهَذِهِ اسْتِعارةٌ، ومَعْناها: أنَّ اليَهودَ أخْرَجوا هَذا القَوْلَ مَخْرَجَ الاسْتِبْخالِ لِلهِ سُبْحانَهُ".

الشريف الرضيّ، تلخيص البيان في مجازات القرآن، ص: 32.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال أَحْمَد مُخْتار عُمَر (ت: 1424ه=2003م):

"اسْتَبْخَلَ الشَّخْصَ: عَدَّهُ بَخيلًا".

أحمد مختار عمر، معجم اللغة العربية المعاصرة. ج: 1، ص: 166.— المصدر

ع

08 —

مَبْخَلَة

اسم
8.1معنى من معاني «مَبْخَلَة»

*الـمَبْخَلةُ: ما يَحْمِلُ على البُخْلِ ويَدْعو إلَيْهِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قال النَّبِيُّﷺ (ت: 11ه= 632م) يَصِفُ الأَبْناءَ:

«إنَّهُمْ لَثَمَرةُ القُلوبِ، وقُرّةُ الأَعْيُنِ، وإنَّهُمْ مَعَ ذَلِكَ لَمَجْبَنةٌ، مَبْخَلةٌ، مَحْزَنةٌ».

وكيع بن الجراح، كتاب الزهد. تح: الفريوائي، ص: 418.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قال ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"والـمَبْخَلةُ: الشَّيْءُ الّذي يَحْمِلُكَ على البُخْلِ".

ابن دريد، الجمهرة. تح: بعلبكي، ص: 292.— المصدر
