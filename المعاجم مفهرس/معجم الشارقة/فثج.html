01 —

فَثَجَ

فعل
1.1معنى من معاني «فَثَجَ»

* فَثَجَ (بِفَتْحِ الثّاءِ) الشَّيْءُ يَفْثُجُ (بِضَمِّ الثّاءِ) فَثْجًا: قَلَّ ونَقَصَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"فَثَجَ: إِذا نَقَصَ في كُلِّ شَيْءٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد أبو الفضل، ج: 11، ص: 24.— المصدر
1.2معنى من معاني «فَثَجَ»

وــــــــــ النّاقَةُ: سَمِنَتْ وعَظُمَتْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ القوطِيَّةِ (ت: 367هـ=977م):

"وفَثَجَتِ النّاقَةُ فَثْجًا: سَمِنَتْ وعَظُمَتْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: فودة، ص: 143.— المصدر
1.3معنى من معاني «فَثَجَ»

وــــــــــ: حالَتْ فَلَمْ تَحْمِلْ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ فارِسٍ (ت: 395هـ=1005م):

"ويُقالُ: فَثَجَتِ النّاقَةُ: إِذا حالَتْ، فَلَمْ تَحْمِلْ ".

ابن فارس، مقاييس اللغة. تح: هارون، ج: 4، ص: 474.— المصدر
1.4معنى من معاني «فَثَجَ»

وــــــــــ فُلانٌ: أَثْقَلَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّغانِيُّ (ت: 650هـ=1252م):

"فَثَجَ، وفَثَّجَ: أَثْقَلَ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الصغاني، الشوارد. تح: مصطفى حجازي، ص: 170.— المصدر
1.5معنى من معاني «فَثَجَ»

وــــــــــ الماءَ السّاخِنَ بِالبارِدِ: كَسَرَ حَرَّهُ بِإِضافَتِهِ إِلَيْهِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابِيِّ (ت: 231هـ=845م):

"فَثَجْتُ الماءَ الحارَّ بِالبارِدِ: إِذا كَسَرْتَ حَرَّهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: محمد أبو الفضل، ج: 11، ص: 24.— المصدر
02 —

فُثِجَ

فعل
2.1معنى من معاني «فُثِجَ»

* فُثِجَ البَحْرُ ونَحْوُهُ: غارَ ماؤُهُ، أَوْ نُزِحَ ماؤُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عُبَيْدَةَ مَعْمَرُ بْنُ المُثَنّى (ت: 209هـ=824م):

"يُقالُ: هُوَ بَحْرٌ لا يُنْكَشُ، ولا يُفْثَجُ، ولا يُؤْبى، ولا يُغَضْغَضُ، ولا يُغَرَّضُ، ولا يُنْكَفُ، ولا يُنْزَحُ بِمَعْنًى واحِدٍ ".

أبو عبيدة، شرح نقائض جرير الفرزدق. تح: حوّر وآخر، ص: 163 .— المصدر
2.2معنى من معاني «فُثِجَ»

وــــــــــ الرَّجُلُ: ذُعِرَ عِنْدَما فوجِئَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"ورَجُلٌ مَفْثوجٌ: إِذا ذُعِرَ عِنْدَما يُفْجَأُ".

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 77.— المصدر
03 —

أَفْثَجَ

فعل
3.1معنى من معاني «أَفْثَجَ»

* أَفْثَجَ فُلانٌ: انْقَطَعَ نَفَسُهُ إِعْياءً. (انْظُرْ: ف ث ء)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الكِسائِيُّ (ت: 189هـ=805م):

"عَدا الرَّجُلُ حَتّى أَفْثَجَ، وأَفْثَأَ، وذَلِكَ إِذا أَعْيا وانْبَهَرَ ".

الأزهري، تهذيب اللغة. تح: محمد أبو الفضل، ج: 11، ص: 24.— المصدر
”

وقالَ أَحَدُ شُيوخِ بَني سَعْدٍ (209هـ=824م، مِنْ شَواهِدِ أَبي عُبَيْدَةَ) يُخْبِرُ عَنْ رَجَزِهِ عِنْدَ بَني جَعْدَةَ:

"أَرْجُزُ بِهِمْ يَوْمًا إِلى اللَّيْلِ لا أُفْثِجُ ".

الجاحظ، البيان والتبيين. تح: هارون، ج: 4، ص: 34. — المصدر
3.2معنى من معاني «أَفْثَجَ»

وــــــــــ عَنْ فُلانٍ: تَرَكَهُ وشَأْنَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّاحِبُ بْنُ عَبّادٍ (ت: 385هـ =995م):

"وأَفْثَجَ عَنّي: تَرَكَني وخَلّى عَنّي (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 77.— المصدر
04 —

أُفْثِجَ

فعل
4.1معنى من معاني «أُفْثِجَ»

* أُفْثِجَ عَلى فُلانٍ: انْقَطَعَ نَفَسُهُ إِعْياءً مِنْ شِدَّةِ العَدْوِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو مِسْحَلٍ الأَعْرابِيُّ (ت: 230هـ=845م):

"يُقالُ: عَدا فُلانٌ حَتّى أَفْثَجَ، ‌وأُفْثِجَ عَلَيْهِ، وأَفْثَأَ، (...)، ومَعْناهُ: حَتّى انْقَطَعَ".

أبو مسحل، النوادر. تح: عزّة حسن، ص: 98.— المصدر
4.2معنى من معاني «أُفْثِجَ»

o بَحْرٌ لا يُفْثَجُ: (انْظُرْ: ب ح ر)

4.3معنى من معاني «أُفْثِجَ»

o ماءٌ لا يُفْثَجُ: (انْظُرْ: م و ء)

05 —

فَثَّجَ

فعل
5.1معنى من معاني «فَثَّجَ»

* فَثَّجَ الأَمْرُ فُلانًا: أَثْقَلَهُ، أَوْ أَعْياهُ وأَتْعَبَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ الأَحْمَرُ بْنُ شُجاعٍ (ت: 60هـ=679م) يَصِفُ رَجُلًا مُتْعَبًا:

مَرَّتْ صَحابَتُهُ عَنْهُ وغادَرَهُ
نَوْمٌ فَأْيقَظَهُ ذُعْرٌ ‌وتَفْثيجُ

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 30.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَمْرٍو الشَّيْبانِيُّ (ت: 206هـ=821م):

"فَثَّجَني هَذا الأَمْرُ؛ أَيْ: أَثْقَلَني (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 30.— المصدر
5.2معنى من معاني «فَثَّجَ»

 وــــــــــ فُلانٌ وغَيرُهُ فُلانًا مِنْ فُلانٍ: رَدَّهُ عَنْهُ وكَفَّهُ. (انْظُرْ: ف ث أ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

"وفَثَّجَني الخَجَلُ مِنْهُ؛ أَيْ: فَثَأَني ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 77.— المصدر
06 —

تَفَثَّجَ

فعل
6.1معنى من معاني «تَفَثَّجَ»

* تَفَثَّجَ الأَمْرُ فُلانًا: أَثْقَلَهُ، أَوْ أَعْياهُ وأَتْعَبَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ الأَحْمَرُ بْنُ شُجاعٍ (ت: 60هـ=679م) في بَيْتٍ مُفْرَدٍ:

كَأَنَّ هادِيَهُ مِمّا تَفَثَّجَهُ
إِذا تَكَلَّمَ في الإِدْلاجِ مَوْلوجُ

الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 303.— المصدر
07 —

فاثِج

اسم
7.1معنى من معاني «فاثِج»

* الفاثِجُ مِنَ النّوقِ: الفَتِيَّةُ الحامِلُ. (ج) فَواثِجُ. (انْظُرْ: ف س ج)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ هِمْيانُ السَّعْدِيُّ (ت: 101هـ=720م) يَصِفُ ناقَةً:

يَظَلُّ يَدْعو نيبَها الضَّماعِجا

والبَكَراتِ اللُّحَّقَ الفَواثِجا

الأصمعي، الإبل. تح: حاتم الضامن، ص: 150.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"والفاثِجُ: الفَتِيَّةُ الحامِلُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأصمعي، الإبل. تح: حاتم الضامن، ص: 105.— المصدر
7.2معنى من معاني «فاثِج»

وــــــــــ: الفَتِيَّةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الفاثِجُ: النّاقَةُ الفَتِيَّةُ".

ابن فارس، مجمل اللغة. تح: زهير عبد المحسن، ج: 1، ص: 712.— المصدر
7.3معنى من معاني «فاثِج»

وــــــــــ: السَّريعَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ مَعْمَرُ بْنُ المُثَنّى (ت: 209هـ=824م):

"‌الفاثِجُ مِنَ الإِبِلِ (...)، وهِيَ أَيْضًا السَّريعَةُ (وهَذِهِ الدَّلالَةُ سَجَّلَتْها الـمَعاجِمُ في عَصْرِ الدُّوَلِ والإِماراتِ)".

ابن القطاع، كتاب الأفعال. تح: دار الكتب، ج: 2، ص: 260.— المصدر
7.4معنى من معاني «فاثِج»

وــــــــــ: العَظيمَةُ أَوِ العَظيمَةُ السَّنامِ. (انْظُرْ: ف س ج)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأَصْمَعِيُّ (ت: 216هـ=831م):

"الفاسِجُ ‌والفاثِجُ: العَظيمَةُ مِنَ الإِبِلِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: هلالي، ج: 10، ص: 596.— المصدر
7.5معنى من معاني «فاثِج»

وــــــــــ: السَّمينَةُ الحائِلُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"نَاقَةٌ فاثِجٌ وفاسِجٌ أَيْضًا: سَمينَةٌ حائِلٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن دريد، جمهرة اللغة. تح: بعلبكي، ج: 1، ص: 414.— المصدر

مُفْثِج

اسم
8.1معنى من معاني «مُفْثِج»

* المُفْثِجُ مِنَ الأُمورِ: ما يُقْضى سَريعًا.

ع

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 76.— المصدر
09 —

فَثْجَة

اسم
9.1معنى من معاني «فَثْجَة»

* الفَثْجَةُ: الامْتِلاءُ أَوِ التُّخْمَةُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الصَّاحِبُ بْنُ عَبّادٍ (ت: 385هـ=995م):

ابن عباد، المحيط. تح: آل ياسين، ج: 7، ص: 77.— المصدر
