01 —

الْغانَّ

فعل
1.1معنى من معاني «الْغانَّ»

* الْغانَّ النَّباتُ: طالَ والْتَفَّ، فهو مُلْغانٌّ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ اللَّيْثُ الكِنانِيُّ (ت: 180ه=796م):

"يُقالُ: الْغانَّ النَّباتُ فَهُوَ مُلْغانٌّ: إِذا الْتَفَّ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 135.— المصدر
02 —

لَغْن

اسم
2.1معنى من معاني «لَغْن»

* اللَّغْنُ: شِرَّةُ الشَّبابِ؛ أَيْ نَشاطُهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"واللَّغْنُ: شِرَّةُ الشَّبابِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[الشِّرَّةُ: النَّشاطُ]
ابن عباد، المحيط. تح: آل ياسين، ج: 5، ص: 83.— المصدر
03 —

لَغْن/ لَغَن/ لُغْن

اسم
3.1معنى من معاني «لَغْن/ لَغَن/ لُغْن»

* اللَّغْنُ والَلَّغَنُ واللُّغْنُ: ما تَعَلَّقَ مِنْ لَحْمِ اللَّحْيَيْنِ، وقيلَ: لَحَماتٌ تَكونُ عِنْدَ اللَّهاةِ. (ج) أَلْغانٌ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ قُتَيْبَةَ (ت: 276ه=889م):

"‌اللُّغْنُ: ما تَعَلَّقَ مِنْ لَحْمِ اللَّحْيَيْنِ. يُقَال: لُغْنٌ ولَغانينُ. (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

3-2: وــــــــــ: الوَتَرَةُ الّتي عِنْدَ باطِنِ الأُذُنِ إذا اسْتَقاءَ الإِنْسانُ تَمَدَّدَتْ.

(ع) (133ه‍/750م) إلى (656ه‍/1258م)". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورََََََََََُُُُِِِِِِّّْْْْ)

ابن قتيبة، غريب الحديث. تح: الجبوري، ج: 3، ص: 759.— المصدر
”

قالَ ابْنُ سيدَه (ت: 458ه=1066م):

"اللَّغْنُ: الوَتَرَةُ الّتي عِنْدَ باطِنِ الأُذُنِ إِذا اسْتَقاءَ الإِنْسانُ تَمَدَّدَتْ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابْنُ سيدَه، المحكم. تح: هنداوي، ج: 5، ص: 527.— المصدر
04 —

لُغْن

اسم
4.1معنى من معاني «لُغْن»

o جاءَ فُلانٌ بِلُغْنِ غَيْرِهِ: إِذا أَنْكَرْتَ ما تَكَلَّمَ بِهِ مِنَ اللُّغَةِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”

قالَ رَجُلٌ مِنَ المُسْلِمينَ (11ه=632م) يَذْكُرُ مَوْقِفًا:

"إِنَّ رَجُلًا قالَ لِآخَرَ: إِنَّكَ لَتُفْتي بِلُغْنِ ضالٍّ مُضِلٍّ ‌".

ابن قتيبة الدينوري، غريب الحديث. عبد الله الجبوري، ج: 3، ص: 759.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الأَزْهَرِيُّ (ت: 370ه=981م):

"ويُقالُ: جاءَ فُلانٌ بِلُغْنِ غَيْرِهِ، إِذا أَنْكَرْتَ ما تَكَلَّمَ بِهِ مِنَ اللُّغَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 134.— المصدر

06 —

لُغْنون

اسم
6.1معنى من معاني «لُغْنون»

* اللُّغْنونُ: لَحَماتٌ تَكونُ عِنْدَ اللَّهَواتِ. (ج) اللَّغانينُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ الخَليلُ (ت: 170ه=787م):

"اللُّغْنونُ واللَّغانينُ مِنْ نَواحي اللَّهاةِ، مُشْرِفٌ عَلى الحَلْقِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 378.— المصدر
”

وقالَ الشَّاعِرُ (206ه=822م) مِنْ شَواهِدِ أَبي عَمْرٍو الشّيبانِيِّ، يَصِفُ:

يَرُدُّ عَجْعاجَهُ والجَوْفُ مُحْتَدِمٌ
سَحْماءُ قَدْ عَجَزَتْ عَنْها ‌الَّلغانينُ

[عَجْعاجَهُ: صِياحَهُ، سَحْماءُ: سَوْداءُ]
الشيباني، الجيم. تح: العزباوي، ج: 3، ص: 216.— المصدر
6.2معنى من معاني «لُغْنون»

وــــــــــ: لَحْمٌ بَيْنَ النَّكَفَتَيْنِ واللِّسانِ مِنْ باطِنٍ. وهُوَ لُغَةٌ في اللُّغْدودِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ اللَّيْثُ الكِنانِيُّ (ت: 180ه=796م):

"واللُّغانينُ لَحْمٌ بَيْنَ النَّكَفَتَيْنِ وَاللِّسانِ مِنْ باطِنٍ، ويُقالُ لَها مِنْ ظاهِرٍ: لَغاديدُ واحِدُها لُغْدودٌ ووَدَجٌ، ولُغْنونٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 73.— المصدر
6.3معنى من معاني «لُغْنون»

وــــــــــ: الخَيْشومُ، وهُوَ أَقْصى الأَنْفِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ ابْنُ الأَعْرابِيِّ (ت: 231ه=845م):

"واللُّغْنونُ أَيْضًا: الخَيْشومُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 5، ص: 527.— المصدر
6.4معنى من معاني «لُغْنون»

وــــــــــ: مَجارِي الأَوْداجِ والعُروقِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ أَبو عَلِيٍّ الهَجَريُّ (ت: 300ه=912م):

"وهِيَ اللَّغانينُ واحِدُها لُغْنونٌ، وهِيَ مَجارِي الأَوْداجِ والعُروقِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الهجري، التعليقات والنوادر. تح: الجاسر، ج: 1، ص: 1241.— المصدر
6.5معنى من معاني «لُغْنون»

وــــــــــ: مَدْخَلُ الأَوْدِيَةِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ المَيْدانِيُّ (ت: 518ه=1124م):

"اللُّغْنونُ مَدْخَلُ الأَوْدِيَةِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الميداني، مجمع الأمثال. تح: محيي الدين عبد الحميد، ج: 1، ص: 178.— المصدر
6.6معنى من معاني «لُغْنون»

o وجَهِلَ مِنْ لَغانينَ سُبُلاتٍ: مَثَلٌ يُضْرَبُ لِمَنْ يَجْهَلُ مَشَقَّةَ الدُّخولِ في مَضايِقِ الأُمورِ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”

قالَ رَجُلٌ ( 45ق.ه=578م) يَذْكُرُ مَثَلًا للمَلِكِ عَمْرِو بْنِ هِنْدٍ:

"جَهِلَ مِنْ لَغانينَ سُبُلَاتٍ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[السُبُلاتُ: جَمْعُ سَبيلٍ]
الميداني، مجمع الأمثال. تح: محيي الدين عبد الحميد، ج: 1، ص: 178.— المصدر
07 —

مُلْغانَّة

اسم
7.1معنى من معاني «مُلْغانَّة»

* أَرْضٌ مُلْغانَّةٌ: كَثيرَةُ الكَلَإِ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”

قالَ نَهْشَلُ بْنُ زَيْدٍ (ت: 153ه=770م):

"أَرْضٌ مُلْغانَّةٌ، والْغَيْنانُها كَثْرَةُ كَلَئِها ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الأزهري، تهذيب اللغة. تح: عبد العظيم محمود، ج: 8، ص: 135.— المصدر
