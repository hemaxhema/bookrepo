النقوش العربيّة

وَرَدَ الجَذْرُ (ر ف ث ر ف ث ) في عَدَدٍ مِنَ النُّقوشِ العَربيّةِ، مِنْ بَينِها نَقْشٌ سَبَئِيٌّ يَعُودُ تاريخُهُ إلى النِّصفِ الأوَّلِ مِنَ الألْفِ الأوَّلِ قَبْلَ الميلادِ (RES 3943/1)، وَرَدَ فيهِ الاسْمُ ( يرفث Yrfṯ ) عَلَمًا على إحْدى القَبائِلِ، فَجاءَ في النَّقْشِ: "....شعبن/ يرفث/ ل المقه/ و ل سبأ"، أيْ: "....الشَّعبُ يَرفثُ (للمَعبودِ) إلمقه ولِسَبأٍ".

(العتيبي، 2006م: 19- 21)

احْتَفظَتِ السّاميّاتُ بِهذا الجَذْرِ مِنَ العَربيّةِ الأُمِّ، فقَدْ وَرَدَ الجَذرُ (ر ف ث):

في الأكَّديّة: وَرَدَ الفِعْلُ: 𒊏 𒋫 𒀊 (رتف) ratap بِمَعْنى (أرادَ الاستِمتاعَ بِالمَرأةِ)، ويوافِقُ الفِعْلَ العَرَبيَّ (رَفَثَ) مَبْنًى ومَعْنًى، غَيْرَ أنَّ الفِعْلَ الأكَّديَّ خَضَعَ لإبدالِ الثّاءِ والتّاءِ، ولِلقَلبِ المَكانيِّ، ومِنْ مَجيئِهِ بِمَعْنى (أرادَ الاستِمتاعَ بِالمرأةِ): 𒋳 𒈠 𒂊 𒊑 𒌅 𒈠 𒅈 𒍝 𒀜 𒈠 𒊭𒌗𒌓𒌋𒌋𒌋 𒐈 𒄰 𒅕 𒌅 𒁍 𒌋𒌋 𒅆 𒅅 𒊒 𒁍 𒌋𒌋 𒅆 ، أيْ: "إذا ما كانَتِ المَرأةُ الحاملُ في شَهرِها الثَّالثِ مَريضةً، وأرادَ الرَّجلُ الاستِمتاعَ وعاشَرَها ..".

(CAD/ R, p. 217)

01 —

رَفَثَ

فعل
1.1معنى من معاني «رَفَثَ»

* رَفَثَ (بِفَتْحِ الفاءِ) يَرْفُثُ، ويَرْفِثُ(بِضَمِّ الفاءِ، وكَسْرِها) رَفْثًا، ورَفَثًا، ورُفوثًا: تَكَلَّمَ كَلامًا فيهِ فُحْشٌ وبَذاءَةٌ. (وضَمُّ المُضارِعِ هُوَ اللُّغَةُ العُلْيا)

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النَّبِي ﷺ (ت: 11هـ=632م):

«‌‌والصِّيامُ جُنَّةٌ، وإِذا كانَ يَوْمُ صَوْمِ أَحَدِكُمْ، فَلا ‌يَرْفُثْ ولا يَصْخَبْ، فَإِنْ سَابَّهُ أَحَدٌ أوْ قاتَلَهُ فَلْيَقُلْ: إنِّي امْرُؤٌ صائِمٌ».

صحيح البخاري. تح: محمد الناصر، ج: 3، ص: 26، (1904).— المصدر
”وقالَ خالِدُ بْنُ سَعيدِ بْنِ العاصِ (11هـ=632م) يَسْأَلُ أَبا بَكْرٍ رَضِيَ اللهُ عَنْهُ عَمّا سَمِعَهُ فَأَغْضَبَهُ:

"‏يا أَبا بَكْرٍ، أَلا تَسْمَعُ هَذِهِ ما تَجْهَرُ بِهِ عِنْدَ رَسولِ اللهِ؟ (...) ما تَرى هَذِهِ تَرْفُثُ عِنْدَ رَسولِ اللْهِ ﷺ؟".

مسند أحمد. تح: شعيب الأرناؤوط وآخرين، ج: 40، ص: 117.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"أَعوذُ باللهِ مِنْ كُلِّ نافِثٍ ورافِثٍ، وشَرِّ كُلِّ تالِجٍ ووالِجٍ".

الخليل، العين. تح: المخزومي وآخر، ج: 6، ص: 182.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ دَقيقِ العيدِ (ت: 702هـ=1302م) يَشْرَحُ حُكْمَ الرَّفَثِ:

"إِنِّي امْرُؤٌ صائِمٌ، وَكُلُّ امْرِئٍ صائِمٍ لا ‌يَرْفُثُ، فَأَنا لا ‌أَرْفُثُ".

ابن دقيق العيد، شرح الإلمام بأحاديث الأحكام. تح: محمد العبد الله، ج: 3، ص: 212.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ الشَّوْكانِيُّ (ت: 1250هـ=1834م) يُبَيِّنُ حُكْمَ المُحْرِمِ:

"ولا يَأْخُذُ مِنْ شَعَرِهِ أَوْ بَشَرِهِ إلّا لِعُذْرٍ، ولا ‌يَرْفُثُ ولا يَفْسُقُ ولا يُجادِلُ ولا يَنْكِحُ".

الشوكاني، الدراري المضية. دار الكتب العلمية، ج: 2، ص: 187.— المصدر
1.2معنى من معاني «رَفَثَ»

وــــــــــ الرَّجُلُ المَرْأَةَ، وبِها، وإِلَيْها، ومَعَها: أَفْضى إلَيْها بالجِماعِ ودَواعِيهِ مِنْ قَوْلٍ وفِعْلٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”في القُرْآنِ الكَريمِ (11هـ=632م) قالَ اللهُ ﷻ:

﴿أُحِلَّ لَكُمۡ لَيۡلَةَ ٱلصِّيَامِ ‌ٱلرَّفَثُ إِلَىٰ نِسَآئِكُمۡۚ هُنَّ لِبَاسٞ لَّكُمۡ وَأَنتُمۡ لِبَاسٞ لَّهُنَّۗ﴾ )البقرة: 187(.
وبِهِ فُسِّرَ قَوْلُ: النَّبِي ﷺ (10ه=631م):
« مَنْ حَجَّ هذا البَيْتَ فَلَمْ يَرْفُثْ، ولَمْ يَفْسُقْ، رَجَعَ كَيَوْمِ وَلَدَتْهُ أُمُّهُ».

صحيح البخاري. تح: محمد الناصر، ج: 3، ص: 11، (1819).— المصدر
”وقالَ عَبْدُ اللهِ بْنُ عُمَرَ (ت: 74هـ=694م) حينَ سُئِلَ "ما لِلصّائِمِ مِنِ امْرَأَتِهِ؟":

"لا يُقَبِّلُ، ولا يَلْمِسُ، ولا يَرْفُثُ".

الدارقطني، المؤتلف والمختلف. تح: موفق بن عبد القادر، ج: 2، ص: 1015.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ=787م):

"الرَّفَثُ: الجِماعُ، رَفَثَ إلَيْها وتَرَفَّثَ، وهَذِهِ كِنايَةٌ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي وآخر، ج: 8، ص: 220.— المصدر
”قال الشّاعِرُ (537هـ=1143م، مِنْ شَواهِدِ النَّسَفِيِّ):

فبَاتوا ‌يَرْفُثونَ وباتَ مِنّا
رِجالٌ في سِلاحِهِمُ رُكوبا

النسفي، التيسير في التفسير. تح: ماهر حبوش وآخرين، ج: 3، ص: 130.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قال ابْنُ تَيْمِيّةَ (ت: 728هـ=1327م) يُفَسِّرُ مَعْنى الرَّفَثِ فِي سِيَاقِ آياتِ الحَجِّ في سُورَةِ البَقَرَةِ:

"جَعَلَ الوَعْدَ بِالمَغْفِرَةِ لِمَنْ لَمْ ‌يَرْفُثْ وَلَمْ يَفْسُقْ. فَالمَنْهِيُّ عَنْهُ المُحَرَّمُ في الآيَةِ: هُوَ «‌الرَّفَثُ»، وهُوَ الجِماعُ ودَواعيهِ قَوْلَا وَفِعْلًا".

ابن تيمية، المستدرك على مجموع الفتاوى. تح: محمد بن قاسم، ج: 3، ص: 187.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قال ابن عَجيبة (ت: 1224هـ=1809م) فِي تَفْسيرِ الرَّفَثِ فِي آيَاتِ الحَجِّ مِنْ سُورَةِ البَقَرَةِ:

"هَذا كُلُّهُ لِمَنِ اتَّقَى اللهَ في حَجِّهِ، فَلَمْ ‌يَرْفُثْ، وَلَمْ يَفْسُقْ، فَإِنَّهُ يَخْرُجُ مِنْ ذُنوبِهِ كَيَوْمِ وَلَدَتْهُ أُمُّهُ".

ابن عجيبة، البحر المديد. تح: القرشي رسلان، ج: 1، ص: 233.— المصدر
02 —

رَفِثَ

فعل
2.1معنى من معاني «رَفِثَ»

* رَفِثَ (بِكَسْرِ الفاءِ) فُلانٌ يَرْفَثُ(بِفَتْحِ الفاءِ) رَفَثًا: تَكَلَّمَ كَلامًا فيهِ فُحْشٌ وبَذاءَةٌ. (لُغَةٌ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو زَيْدٍ الأَنْصارِيُّ (ت: 215هـ=830م):

"ويُقالُ قَدْ رَفِثَ كَلامُ الرَّجُلِ يَرْفَثُ رَفَثًا، وهَذِرَ يَهْذَرُ هَذْرًا". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو زيد الأنصاري، النوادر في اللغة. تح: محمد أحمد، ص: 555.— المصدر
2.2معنى من معاني «رَفِثَ»

وــــــــــ الرَّجُلُ المَرْأَةَ وبِها وإِلَيْها: أَفْضى إِلَيْها بِالجِماعِ ودَواعِيهِ مِنْ قَوْلٍ وفِعْلٍ. (لُغَةٌ)

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القُوطِيَّةِ (ت: 367هـ=977م):

"ورَفِثَ رَفَثًا، وأَرْفَثَ: غَشِيَ النِّساءَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: علي فودة، ص: 97.— المصدر

رَفُثَ

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)

"ورَفُثَ: أَفْحَشَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن سيده، المحكم. تح: هنداوي، ج: 10، ص: 141.— المصدر
04 —

أَرْفَثَ

فعل
4.1معنى من معاني «أَرْفَثَ»

* أرْفَثَ الرَّجُلُ : أَفْحَشَ في قَوْلِهِ وأَبْذَأَ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أَبو العالِيَةِ الرِّياحِيُّ (68ه=688م) يُرَاجِعُ ابْنَ عَبّاسٍ فِيمَا قَالَهُ وَهُوَ مُحْرِمٌ:

"أَرْفَثْتَ وأَنْتَ مُحْرِمٌ".

شهاب الدين الرملي، شرح سنن أبي داود. تح: خالد الرباط وآخرين، ج: 3، ص: 425.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ عَبْدُ اللهِ بْنُ عُمَرَ العُمَرِيُّ (ت: 171هـ=788م) يَسْتَغْرِبُ سُلوكَ امْرَأَةٍ في الحَجِّ:

"خَرجْتُ حاجًّا، فَرَأَيْتُ امْرَأَةً جَميلَةً تَتَكَلَّمُ بِكَلامٍ أرْفَثَتْ فيهِ، فَأَدْنَيْتُ ناقَتي مِنْها، ثُمَّ قُلْتُ لَها: يا أَمَةَ اللهِ، أَلَسْتِ حاجَّةً! أَما تَخافينَ اللهَ!".

ديوان العرجيّ. تح: سجيع الجبيلي، ص: 360.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ القُرْطُبِيُّ المُفَسِّرُ (ت: 671هـ=1273م):

"يُقالُ: رَفَثَ وأَرْفَثَ: إذا تَكَلَّمَ بالقَبيحِ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

القرطبي، الجامع لأحكام القرآن. تح: عبد الله التركي، ج: 3، ص: 188.— المصدر
4.2معنى من معاني «أَرْفَثَ»

وــــــــــ الرَّجُلُ بِالمَرْأَةِ: أَفْضى إِلَيْها بِالجِماعِ ودَواعِيهِ مِنْ قَوْلٍ وفِعْلٍ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ القُوطِيَّةِ (ت: 367هـ=977م):

"وأَرْفَثَ: غَشِيَ النِّساءَ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن القوطية، الأفعال. تح: علي فودة، ص: 97.— المصدر
”وقالَ أبو الحَسَنِ اللَّخْمِيُّ (ت: 478هـ=1085م):

"وما لَمْ يَحْصُلْ كَمالُ تَحَلُّلِهِ فَقَدْ أَرْفَثَ في الحَجِّ، فَيَجِبُ فَسادُهُ، ولأَنَّها حالٌ مَمْنوعٌ فيها الوَطْءُ لِبَقاءِ الإحْرامِ".

أبو الحسن اللخميّ، التبصرة. تح: أحمد نجيب، ج: 3، ص: 1222.— المصدر
05 —

رافَثَ

فعل
5.1معنى من معاني «رافَثَ»

* رافَثَ فُلانٌ فُلانًا: بادَلَهُ في الكَلامَ فُحْشًا بِفُحْشٍ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو أُمامَةَ الباهِلِيُّ (ت: 86هـ=705م) يَحْكي قِصَّةَ تَوْبَةِ المَرْأَةِ البَذيئَةِ:

"كانَتْ امْرَأَةٌ تُرافِثُ الرِّجالَ وكانَتْ بَذيئَةً، فَمَرَّتْ بالنَّبِيِّ ﷺ وهُوَ يَأْكُلُ ثَريدًا عَلى طَرَيانٍ، قَالَتْ: انْظُروا إلَيْهِ يَجْلِسُ كَما يَجْلِسُ العَبْدُ، ويَأْكُلُ كَما يَأْكُلُ العَبْدُ (...)".

الطبراني، المعجم الكبير. تح: حمدي السلفي، ج: 8، ص: 236.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيُّ (ت: 538هـ=1144م):

"رَفَثَ في كَلامِهِ، وأَرْفَثَ، وتَرَفَّثَ: أَفْحَشَ (...) ورافَثَ صاحِبَهُ مُرافَثَةً". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في العَصْرِ الحَديثِ)

الزمخشري، أساس البلاغة. تح: عيون السود، ج: 1، ص: 367.— المصدر
د
الدول والإمارات(657ه‍/1259م) إلى (1213ه‍/1798م)
”قالَ ابْنُ دَقيقِ العيدِ (ت: 702هـ=1302م):

"ما هَذِهِ مُنافَثَةٌ، إنَّما هِيَ مُرافَثَةٌ".
