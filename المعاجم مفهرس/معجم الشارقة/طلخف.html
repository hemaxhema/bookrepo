طلخف
طِلْخاف(اسم) طَلَخْف/ طِلَخْف(اسم) طَلَخْفى/ طِلَخْفى(اسم) طِلْخيف(اسم) طِلَّخْف(اسم)

المعانى الكلّيّة

  1. الشِّدَّةُ
01 —

طِلْخاف

اسم
1.1معنى من معاني «طِلْخاف»

* الطِّلْخَافُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (وَانْظُرْ: ط ل ح ف).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قَالَ ابْنُ دُرَيْدٍ (ت: 321هـ=933م):

"وَضَرْبٌ طِلْحافٌ وطِلْخافٌ: شَديدٌ، بِالحاءِ وَالخاءِ ".

ابن دريد، جمهرة اللغة. تح: ر بعلبكي، ص: 1203.— المصدر
02 —

طَلَخْف/ طِلَخْف

اسم
2.1معنى من معاني «طَلَخْف/ طِلَخْف»

* الطَّلَخْفُ، والطِّلَخْفُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (وَانْظُرْ: ط ل ح ف).

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قَالَ حسَّانُ بْنُ ثابِتٍ (3هـ=624م) يُصَوِّرُ شِدَّةَ المُسْلِمينَ وقُوَّتَهُمْ يَوْمَ أُحُدٍ:

أَقَمْنا لَكُمْ ضَرْبًا طِلَخْفًا مُنَكِّلًا
وحُزْناكُمُ بالطَّعْنِ مِنْ كُلِّ جانِبِ

ابن منظور، لسان العرب. تح: أمين عبد الوهاب، وآخر، ج: 8، ص: 134.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ بْنُ أَحْمَدَ (ت: 170هـ= 787م):

"الطِّلَخْفُ: الطَّعْنُ الشَّديدُ ".

الخليل، العين. تح: المخزومي وآخر، ج: 4، ص: 333.— المصدر
”وقالَ ابْنُ الرّوميِّ (ت: 283هـ=896م) يَصِفُ قَدَحًا أهْدَاهُ إلى عَليِّ بنِ يَحْيَى المُنَجِّمِ مادِحًا:

يَتَّقي نَفْحَةَ اللِّسانِ وَيَغْشَى
كُلَّ طَعْنٍ وَكُلَّ ضَرْبٍ طِلَخْفِ

ديوان ابن الرّوميّ. تح: بسج، ج: 2، ص: 401.— المصدر
03 —

طَلَخْفى/ طِلَخْفى

اسم
3.1معنى من معاني «طَلَخْفى/ طِلَخْفى»

* الطَّلَخْفَى، وَالطِّلَخْفَى مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ. (وَانْظُرْ: ط ل ح ف).

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ السّيرافِيُّ (ت: 368ه=978م):

"والطَّلَخْفَى: الشَّديدُ مِنَ الطَّعْنِ ".

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 56.— المصدر
04 —

طِلْخيف

اسم
4.1معنى من معاني «طِلْخيف»

* الطِّلْخِيفُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ القاليُّ (ت: 356هـ=967م):

"ضَرْبٌ ‌طَلَخْفٌ وطِلَخْفٌ وطِلَّخْفٌ وطِلَخْفَى وطِلْخِيفٌ وطِلْخافٌ: شَدِيدٌ ".

ابن سيده، المخصص. تح: جفال، ج: 4، ص: 495.— المصدر
05 —

طِلَّخْف

اسم
5.1معنى من معاني «طِلَّخْف»

* الطِّلَّخْفُ مِنَ الضَّرْبِ وَالجوعِ وَنَحْوِهِما: الشَّديدُ.

ع

"الطِّلَّخْفُ ...: الشَّديدُ مِنَ الطَّعْنِ ".

ابن سيده، المخصص. تح: جفال، ج: 2، ص: 56.— المصدر
