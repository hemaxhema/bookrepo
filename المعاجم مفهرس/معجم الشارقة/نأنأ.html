نأنأ
نَأْنَأَ(فعل) تَنَأْنَأَ(فعل) مُنَأْنَأ(اسم) نَأْناء(اسم) نَأْنَأ/ نُؤْنُؤ(اسم) نَأْنَأَة(اسم) نُؤْنوء(اسم)
النظائر الساميّة

احْتَفَظَتِ السّاميّاتُ بهذا الجَذْرِ مِنَ العَرَبيَّةِ الأُمِّ؛ فَقَدْ وَرَدَ الجَذْرُ (ن أ ن أ):

في العِبْريَّةِ؛ ومِنْهُ الاسْمُ: מְנֻעֲנָע (مْنُعَناع) mənuʽanāʽ بِمَعْنى: "غَيْرِ مُحْكَمٍ، غَيْرِ مُبْرَمٍ، ضَعيفٍ مُتَراخٍ"، وهُوَ يُوافِقُ الاسْمَ العَرَبيَّ (مُنَأْنَأٌ) مَبْنًى ومَعْنًى، غَيْرَ أَنَّ الاسْمَ العِبْريَّ خَضَعَ لإِبْدالِ صَوْتَيِ الهَمْزَةِ عَيْنًا؛ ومِنْ مَجيئِهِ بِمَعْنى: "مُنَأْنَأٍ" ما وَرَدَ في التَّلْمودِ البابِليِّ: "וְהַמְקוּשָּׁרִין וְהַמְנוּעֲנָעִין, בְּכׇל מָקוֹם אֲסוּרִין"؛ أَيْ: "وسَواءٌ كانَتْ[َجْنِحَةُ الطَيْرِ] مُحْكَمَةَ الرَّبْطِ أو مُنَأْنَأَةً، فَهِيَ مُحَرَّمَةٌ أَنّى كانَتْ".

(תלמ"ב: ביצה, כ"ה, א')

المعانى الكلّيّة

  1. الضَّعْفُ والجُبْنُ
  2. النَّهْيُ والكَفُّ
  3. التَّرَفُّعُ
01 —

نَأْنَأَ

فعل
1.1معنى من معاني «نَأْنَأَ»

* نَأْنَأَ فُلانٌ: ضَعُفَ وعَجِزَ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ الأَفْوَهُ الأَوْدِيُّ (ت: 54ق.ه=569م) يُفاخِرُ بِبَني أَوْدٍ وقَبيلَتِهِ الكُبْرى مَذْحِجٍ:

قَدْ أَحْسَنَتْ أَوْدٌ وما نَأْنَأَتْ
مَذْحِجُ في ضَرْبِ الكُلى والرُّؤوسْ

ديوان الأفوه الأودي. تح: محمد التونجي، ص: 84.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"النَّأْنَأَةُ: الضَّعْفُ والعَجْزُ في الأَمْرِ ... ونَأْنَأَ هُوَ نَأْنَأَةً، والنِّساءُ نَأْنَأْنَ، فإِذا أَمَرْتَهُنَّ قُلْتَ: نَأْنِئْنَ ".

الخليل، العين. تح: المخزومي، ج: 8، ص: 395.— المصدر
”وقالَ ابْنُ مُنيرٍ الطَّرابُلُسِيّ (544ه=1149م) يَمْدَحُ نورَ الدّينِ مَحْمودَ بْنَ عِمادِ الدّينِ زَنْكي حينَ اسْتَوْلى سَنَةَ 544ه عَلى حِصْنِ أَفامِيَةَ قُرْبَ حَلَبَ، في حَرْبِهِ ضِدَّ الصَّليبِيّينَ:

أَنْشَرْتَ يا مَحْمودُ مِلّةَ أَحْمَدٍ
مِنْ بعد ما شَمَلَ البِلى أَبْشارَها

إِنْ جانَأَتْ عَدَلَ السِّنانُ قَوامَها
أَوْ نَأْنَأَتْ كانَ الحُسامُ جِبارَها

أبو شامة المقدسي، الروضتين. تح: إبراهيم الزيبق، ج: 1، ص: 218.— المصدر
1.2معنى من معاني «نَأْنَأَ»

وــــــــــ: تَرَفَّعَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"نَأْنَأْتُ: تَرَفَّعْتُ ".

ابن عباد، المحيط. تح: آل ياسين، ج: 10، ص: 421.— المصدر
1.3معنى من معاني «نَأْنَأَ»

وــــــــــ عَنِ الأَمْرِ: جَبُنَ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"نَأْنَأْتُ عَنِ الأَمْرِ: كِعْتُ عَنْهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

[كاعَ: جَبُنَ]
ابن عباد: المحيط في اللغة. تح: آل ياسين، ج: 10، ص: 421.— المصدر
1.4معنى من معاني «نَأْنَأَ»

وــــــــــ الرَّأْيَ ونَحْوَهُ، وفيهِ، وعَنْهُ: ضَعُفَ وخَلَّطَ فيهِ ولَمْ يُبْرِمْهُ.

ق
قبل الإسلام(-1000ه‍/378م) إلى (-1ه‍/621م)
”قالَ عَبْدُ هِنْدِ بْنُ زَيْدٍ التَّغْلِبِيّ (50ق.ه=574م) يَنْصَحُ قَوْمَهُ:

فَلا أَسْمَعَنْ مِنْكُمْ بِأَمْرٍ مُنَأْنَأٍ
ضَعيفٍ ولا تَسْمَعْ بِهِ هامَتي بَعْدي

أبو تمام، الوحشيات. تح: عبد العزيز الميمني، ص: 31.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أَبو زَيْدٍ الأَنْصارِيّ (ت: 215ه=830م):

"وتَقولُ: نَأْنَأْتُ رَأْيي نَأْنَأَةً، إِذا خَلَّطْتَ فيهِ تَخْليطًا فلَمْ تُبْرِمْهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

أبو زيد، الهمز. تح: لويس شيخو، ص: 6.— المصدر
1.5معنى من معاني «نَأْنَأَ»

وــــــــــ فُلانًا: نَهاهُ عَنِ أمْرٍ وكَفَّهُ عَنْهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ النّابِغَةُ الجَعْدِيُّ (ت: 50ه=670م) مُفْتَخِرًا:

ونَهْنَهْتُهُ حَتّى لَبِسْتُ مُفاضَةً
مُضاعَفَةً كالنِّهْيِ ريحَ وأُمْطِرا

وجَمَّعْتُ بَزِّي فَوْقَهُ ودَفَعْتُهُ
ونَأْنَأْتُ مِنْهُ خَشْيَةً أَنْ يُكَسَّرا

[النِّهْيُ: غَدِيرُ الماءِ]
ديوان النابغة الجعدي. تح: واضح الصمد، ص: 82.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"نَأْنَأْتُ الرَّجُلَ: نَهْنَهْتُهُ عَمّا يُريدُ وكَفَفْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، وآخر، ج: 8، ص: 395.— المصدر
1.6معنى من معاني «نَأْنَأَ»

وــــــــــ: أَمْهَلَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ أبو مُحَمَّدٍ الأُمَوِيّ (ت: 203ه=818م):

"نَأْنَأْتُهُ: أَمْهَلْتُهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

السرقسطي، الأفعال. تح: شرف، ج: 3، ص: 243.— المصدر
1.7معنى من معاني «نَأْنَأَ»

وــــــــــ وَلَدَهُ: أَحْسَنَ غِذاءَهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ عَبّادٍ (ت: 385ه=995م):

"ونَأْنَأْتُ وَلَدي: غَذَوْتُهُمْ أَحْسَنَ الغِذاءِ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

ابن عباد، المحيط. تح: آل ياسين ج: 10، ص: 421.— المصدر

تَنَأْنَأَ

فعل
2.1معنى من معاني «تَنَأْنَأَ»

* تَنَأْنَأَ فُلانٌ: ضَعُفَ واسْتَرْخى ولَمْ يَحْزِمْ أَمْرَهُ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ عَلِيُّ بْنُ أَبي طالِبٍ (36ه=656م) يُؤَنِّبُ سُلَيْمانَ بْنَ صُرَدٍ لِتَخَلُّفِهِ عَنْ يَوْمِ الجَمَلِ:

"تَنَأْنَأْتَ، وتَرَبَّعْتَ، وتَراخَيْتَ فكَيْفَ رَأَيْتَ اللهَ صَنَعَ".

القاسم بن سلام، غريب الحديث. تح: شرف، ج: 4، ص: 110.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الخَليلُ (ت: 170ه=787م):

"وتَنَأْنَأْتُ أَنا: إِذا ضَعُفْتُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

الخليل، العين. تح: المخزومي، وآخر، ج: 8، ص: 395.— المصدر
2.2معنى من معاني «تَنَأْنَأَ»

وــــــــــ عَنِ الأَمْرِ: ضَعُفَ عَنْهُ.

ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ ابْنُ دُرَيْدٍ (ت: 321ه=933م):

"وتَنَأْنَأْتُ عَنِ الأَمْرِ: ضَعُفْتُ عَنْهُ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في بَقِيَّةِ العُصورِ)

06 —

نَأْنَأَة

اسم
6.1معنى من معاني «نَأْنَأَة»

* النَّأْنَأَةُ: بَدْءُ الإِسْلامِ.

س
الإسلامي(1ه‍/622م) إلى (132ه‍/749م)
”قالَ أبو بَكْرٍ الصِّدّيقُ (ت: 13ه=634م) يُبَشِّرُ مَنْ ماتَ عَلى الإِسْلامِ والإِسْلامُ ضَعيفٌ بَعْدُ:

"طُوبى لِمَنْ ماتَ في النَّأْنَأةِ".

القاسم بن سلام، غريب الحديث. تح: شرف، ج: 4، ص: 109.— المصدر
ع
العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الأَصْمَعِيّ (ت: 216ه=831م):

"هِيَ النَّأْنَأَةُ -مَهْموزَةً- ومَعْناها: أَوَّلُ الإِسْلامِ، وإِنَّما سُمِّيَ بِذَلِكَ لِأَنَّهُ كانَ قَبْلَ أَنْ يَقْوى الإِسْلامُ ويَكْثُرَ أَهْلُهُ وناصِرُهُ، فهُوَ عِنْدَ النّاسِ ضَعيفٌ ".

القاسم بن سلام، غريب الحديث. تح: شرف، ج: 4، ص: 110.— المصدر
”قالَ الشَّهْرَسْتانيّ (ت: 548ه=1153م) يَحْتَجُّ لِإِمامَةِ عَلِيٍّ بَعْدَ النَّبِيِّ ﷺ:

"وأَمّا تَصْريحاتُهُ فمِثْلُ ما جَرى في نَأْنَأَةِ الإِسْلامِ ... ".

الشهرستاني، الملل والنحل. تح: أحمد فهمي، ص: 163.— المصدر
ح
الحديث(1214ه‍/1799م) إلى (1500ه‍/2076م)
”قالَ أَنِسْتاس الكَرْمَلِيّ (1358ه=1939م) يَتَحَدّثُ عَنْ أَهْلِ السَّراةِ:

"يُنافِحونَ في دينِهِمْ، ويَفْتَحونَ الفُتوحاتِ مَعَ كِبارِ القادةِ مُنْذُ نَأْنَأَةِ الإِسْلامِ".

أنستاس كرملي، مجلة الرسالة. ج: 334، ص: 2186.— المصدر

العباسي(133ه‍/750م) إلى (656ه‍/1258م)
”قالَ الزَّمَخْشَرِيّ (ت: 538ه=1144م):

"رَجُلٌ نَأْنَأَةٌ ونَأْناءٌ ونُؤْنوءٌ: ضَعيفٌ عاجِزٌ ". (وهَذِهِ الدَّلالَةُ سَجَّلَتْها المَعاجِمُ في عَصْرِ الدُّوَلِ والإماراتِ)

الزمخشري، الفائق. تح: علي محمد البجاوي، وآخر، ج: 3، ص: 399.— المصدر
