تَأْثَام[مَصْدَر]

قبل 100ق.هـ=525م

تَأْثَامُ الشَّخْصِ: كَثْرَةُ اقْتِرَافِهِ الإِثْمَ.

"إِنِّي -وَاللهِ- مَا أُحْسِنُ تَكْذَابَكَ وَتَأْثَامَكَ "
نَهْشَل بن دارم الدّارميّ
أمثال العرب: المفضَّل الضَّبّيّ (ت، 178هـ)، قدم له وعلق عليه: إحسان عبّاس، دار الرائد العربي، بيروت، ط2، (1403هـ/ 1983م)، ‏66‏.

إِثْم[اسْم]

قبل 100ق.هـ=525م

الإِثْمُ : الذَّنْبُ.

"إِلَى صَلَاةٍ وَصِيَام، وَصِلَةِ أَرْحَام، وَكَسْرِ أَصْنَام، وَتَعْطِيلِ أَزْلَام، وَاجْتِنَابِ آثَام "
عُفيراء الحِمْيريّة الكاهنة
حياة الحيوان الكبرى: الدَّميري، تح: إبراهيم صالح، دار البشائر، دمشق، 2005م، ‏1/‏709‏.

إِثْم[مَصْدَر]

ن309هـ=921م

الإِثْمُ فِي المَشْيِ: الإِبْطَاءُ فِيهِ.

"يُقَالُ: إِنَّ النَّاقَةَ لَتَأْثَمُ المَشْيَ إِثْمًا "
كُراع النَّمْل الهُنائيّ الأزديّ
المُجَرَّد في غريب كلام العرب ولُغاتها: كُراع النَّمْل الهُنائيّ الأزديّ (ت، 309هـ)، تحقيق: محمّد بن أحمد العمري، مكتبة الثقافة الدينية، القاهرة، ط1، (1428هـ/ 2007م)، ‏63‏.

آثِم[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الآثِمُ : مُرْتَكِبُ الإِثْمِ.

لَعَمْرُكَ، مَا سَعْدٌ بِخُلَّةِ آثِمٍ وَلَا نَأْنَأٍ يَوْمَ الحِفَاظِ، وَلَا حَصِرْ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏112‏.
قبل 13ق.هـ=609م

الآثِمُ مِنَ الدَّوَابِّ: المُبْطِئُ فِي السَّيْرِ.

تَنْفِي الأَوَاثِمَ عَنْ سَوَاءِ سَبِيلِهَا شَرَكَ الأَحِزَّةِ، وَهْيَ غَيْرُ شَمُوسِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏69‏.

آثِم[صِفَةٌ مُشَبَّهَة]

ن11هـ=632م

الآثِمُ مِنَ الأَيْمَانِ وَنَحْوِهَا: الكَاذِبُ الفَاجِرُ، الَّذِي يَسْتَحِقُّ صَاحِبُهُ الذَّمَّ وَالعُقُوبَةَ. [لفظ إسلامي]

"مَا مِنْ عَبْدٍ أَوْ أَمَةٍ يَحْلِفُ عِنْدَ هَذَا المِنْبَرِ عَلَى يَمِينٍ آثِمَةٍ ، وَلَوْ عَلَى سِوَاكٍ رَطْبٍ إِلَّا وَجَبَتْ لَهُ النَّارُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏14/‏99،،‏(رقم الحديث: 8362).

أَثَام[اسْم]

قبل 50ق.هـ=573م

الأَثَامُ : الخَمْرَةُ.

وَمُضْمَرٍ قَدْ أَلَكَتْ لِجَامَهْ وَقَطَعَتْ مِنْ جَرْيِهِ حِزَامَهْ فَسِيقَ جَرْيَ الوَعْلِ وَالنَّعَامَهْ وَرُبَّ زِقٍّ شَرِبَتْ أَثَامَهْ
الشَّنْفَرَى الأزديّ
شرح شعر الشَّنفرى الأزدي: محاسن بن إسماعيل الحلبي، تحقيق وتعليق: خالد عبد الرؤوف الجبر، دار الينابيع، عمان، ط1، (2004م)، ‏51‏.
قبل 13ق.هـ=609م

الأَثَامُ : جَزَاءُ الإِثْمِ.

جَزَى اللهُ ابْنَ عُرْوَةَ حِينَ أَمْسَى عَقُوقًا، وَالعُقُوقُ لَهُ أَثَامُ
بَلْعاء بن قيس اليَعْمَريّ الكِنانيّ
شعر بَلعاء بن قيس الكنانيّ: تحقيق: أحمد بن محمّد علي بن عبيد الهنداسي، مجلة العرب، الرياض، س34، ج(1، 2)، (1419هـ/ 1998م)، ‏81‏.
قبل 13ق.هـ=609م

الأَثَامُ : الإِثْمُ.

فَيَا لِلّٰهِ، لَمْ أَكْسِبْ أَثَامًا وَلَمْ أَهْتِكْ لِذِي رَحِمٍ حِجَابَا
الحارث بن ظالم المُرّيّ الغَطَفانيّ
الحارث بن ظالم المري الوافي الفاتك: تحقيق: عادل جاسم البياتي، مجلة كلية الآداب، بغداد، ع15، (1972م)، ‏372‏.

مَأْثَم[مَصْدَرٌ مِيمِيّ]

قبل 50ق.هـ=573م

المَأْثَمُ : الوُقُوعُ فِي الإِثْمِ.

وَكَائِنْ، أَزَرْنَا المَوْتَ، مِنْ ذِي تَحِيَّةٍ إِذَا مَا ازْدَرَانَا، أَوْ أَسَفَّ لِمَأْثَمِ
جابر بن حُنيّ التَّغْلِبيّ
شعراء تغلب في الجاهلية أخبارهم وأشعارهم: صنعة: علي أبو زيد، المجلس الوطني للثقافة والفنون والآداب، الكويت، ط1، (1421هـ/ 2000م)، ‏2/‏216‏.

مُؤْثَم[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المُؤْثَمُ : المَطْلُوبُ بِدَمٍ.

"مَنْ يَأْخُذُ مِنِّي هَذَيْنِ، وَأَعْقِدُ حِلْفِي إِلَيْهِ؟ فَإِنِّي مُؤْثَمٌ "
مسعود بن غافل الهذليّ
كتاب المنمّق في أخبار قُرَيش: ابن حَبيب البغداديّ (ت، 245هـ)، صححه وعلق عليه: خورشيد أحمد فاروق، عالم الكتب، بيروت، ط1، (1405هـ/ 1985م)، ‏245‏.
ن90هـ=709م

المُؤْثَمُ : المَنْسُوبُ إِلَيْهِ الإِثْمُ.

بَلْ قُلْتُ بَعْضَ القَوْلِ غَيْرَ مُؤْثَمِ لَيَقْذِفَنَّ خَابِرٌ إِلَى عَمِ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏1/‏468‏.
ن428هـ=1037م

المُؤْثَمُ فِيهِ مِنَ الأُمُورِ: الَّذِي وَقَعَ فِيهِ الإِثْمُ.

"المُرَادُ بِهِ المُؤَاخَذَةُ بِإِيجَابِ الكَفَّارَةِ دُونَ المَأْثَمِ، بِدَلَالَةِ أَنَّ الكَفَّارَةَ تَتَعَلَّقُ بِالمُؤْثَمِ فِيهِ، وَبِمَا يَجِبُ فِيهِ الحِنْثُ؛ إِذَا حَلَفَ لَا يُصَلِّي، أَوْ لَيَقْتُلَنَّ فُلَانًا"
أبو الحسين القُدوريّ البغداديّ
التّجريد: أبو الحسين القُدوريّ البغداديّ (ت، 428هـ)، دراسة وتحقيق: محمّد أحمد سراج، علي جمعة محمد، دار السلام، القاهرة، ط1، (1424هـ/ 2004م)، ‏12/‏6401‏.
ن478هـ=1085م

المُؤْثَمُ مِنَ الأَشْخَاصِ: المُجَازَى بِالإِثْمِ.

"فَإِنَّهُ مُؤْثَمٌ إِنْ صَلَّى بَعْضَ العَصْرِ بَعْدَ الغُرُوبِ"
عليّ بن محمّد اللَّخْميّ
التّبصرة: عليّ بن محمّد اللَّخْميّ (ت، 478هـ)، دراسة وتحقيق: أحمد عبد الكريم نجيب، وزارة الأوقاف والشؤون الإسلامية، الدّوحة، ط1، د.ت. ‏2/‏595‏.

أَثَّمَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

أَثَّمَ الشَّخْصَ: نَسَبَهُ إِلَى اقْتِرَافِ الإِثْمِ.

بُرْجٌ يُؤَثِّمُنِي وَيَكْفُرُ نِعْمَتِي صَمِّي، لِمَا قَالَ الكَفِيلُ، صَمَامِ
الحُصين بن الحُمام الفَزاريّ المُرّيّ
الحصين بن الحمام المرّي الفارس الشاعر (سيرته وشعره): جمع وتحقيق: شريف علاونة، جامعة البتراء، عمان، (2002م)، ‏100‏.
ن68هـ=687م

أَثَّمَهُ : أَوْقَعَهُ فِي الإِثْمِ.

"كَرِهْتُ أَنْ أُؤَثِّمَكُمْ ، فَتَجِيئُونَ تَدُوسُونَ الطِّينَ إِلَى رُكَبِكُمْ"
ابن عبّاس
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏1/‏222‏.

أَثِمَ[لَازِم]

قبل 13ق.هـ=609م

أَثِمَ الشَّخْصُ: وَقَعَ فِي الإِثْمِ.

حَلَفْتُ فَلَمْ أَتْرُكْ لِنَفْسِكَ رِيبَةً وَهَلْ يَأْثَمَنْ ذُو أُمَّةٍ، وَهْوَ طَائِعُ؟
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏166‏.
ن110هـ=728م

أَثِمَتِ اليَمِينُ: حَنِثَ فِيهَا صَاحِبُهَا.

حَلَفْتُ، وَمَنْ يَأْثَمْ، فَإِنَّ يَمِينَهُ إِذَا أَثِمَتْ ، لَاقِيهِ مِنْهَا عَذَابُهَا
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏1/‏87‏.

تَأْثِيم[مَصْدَر]

ن370هـ=980م

تَأْثِيمُ الشَّخْصِ: نِسْبَةُ الإِثْمِ إِلَيْهِ.

"إِنْ كَانَ الِاخْتِلَافُ فِي شَيْءٍ جَرَوْا فِيهِ إِلَى تَأْثِيمِ بَعْضِهِمْ بَعْضًا، وَلَمْ يُسَوِّغُوا اجْتِهَادَ الرَّأْيِ فِيهِ، فَإِنَّ الطَّائِفَةَ البَاقِيَةَ يَكُونُ إِجْمَاعُهَا حُجَّةً"
أبو بكر الجَصّاص الرّازيّ
أصول الفقه المسمّى بالفصول في الأصول: أبو بكر الجَصّاص الرّازيّ (ت، 370هـ)، دراسة وتحقيق: عجيل جاسم النشمي، وزارة الأوقاف والشّؤون الإسلاميّة، الكويت، ط2، (1414هـ/ 1994م)، ‏3/‏311‏.

إِثْمِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن1270هـ=1853م

الإِثْمِيَّةُ : الِاتِّصَافُ بِالإِثْمِ.

" وَإِثْمِيَّةُ بَعْضِ الظَّنِّ، كَيْفَ تَكُونُ عِلَّةَ الِاجْتِنَابِ عَنْ الكَثِيرِ، وَلَا اسْتِلْزَامَ بِيْنَهُمَا؟ فَقُلْتُ:... إِذَا عُرِفَ أَنَّ كُلَّ إِثْمٍ يَجِبُ الِاجْتِنَابُ عَنْهُ، ثُمَّ ظُنَّ أَنَّ هَذَا الظَّنَّ إِثْمٌ، فَيُجْتَنَبُ عَنْ جَمِيعِ الظَّنِّ"
أبو الثناء شهاب الدين محمود الألوسي
غرائب الاغتراب ونزهة الألباب: أبو الثناء شهاب الدين محمود الألوسي (ت، 1270هـ)، تقديم: عباس العزاوي، شرحه ووضع حواشيه وفهارسه: إبراهيم شمس الدين، الدار العربية للموسوعات، بيروت، ط1، (1436ه/ 2015م)، ‏400‏.

تَأْثِيمِيّ[اسْمٌ مَنْسُوب]

1427هـ=2005م

التَّأْثِيمِيُّ : المَنْسُوبُ إِلَى التَّأْثِيمِ.

"وَبِالرَّغْمِ مِنْ أَنَّ السِّيَاقَ المُبَاشِرَ، لَا يَبْدُو أَنَّهُ سِيَاقٌ تَأْثِيمِيٌّ ... فَإِنَّ الآيَاتِ التَّالِيَةَ، وَلَا سِيَّمَا الآيَةُ (14)، تَتَحَدَّثُ عَنِ الخَرَابِ"
سعيد الغانمي
العرب والغصن الذهبي: إعادة بناء الأسطورة العربية، لياروسلاف ستيتكيفيتش: سعيد الغانمي ، ترجمة: سعيد الغانمي، المركز الثقافي العربي، الدار البيضاء، بيروت، ط1، (2005م)، ‏163‏.

مُؤَاثَم[اسْمُ مَفْعُول]

1430هـ=2008م

المُؤَاثَمُ : المُجَازَى بِالإِثْمِ.

"لَا بُدَّ أَنْ نَتَذَكَّرَ بِأَنَّ كُلَّ صَائِمٍ لِيَوْمِ عَاشُورَاءَ، لَا يَصُومُهُ إِلَّا طَلَبًا لِلثَّوَابِ، خَاصَّةً وَأَنَّهُ لَيْسَ مِنَ الصَّوْمِ الوَاجِبِ، المُؤَاثَمِ تَارِكُهُ"
أحمد يحيى

إِثْمِيّ[اسْمٌ مَنْسُوب]

1430هـ=2008م

الإِثْمِيُّ : المَنْسُوبُ إِلَى الإِثْمِ.

"التَّجْرِيدُ مِنَ الجِنْسِيَّةِ المَغْرِبِيَّةِ: زَوَالُ الجِنْسِيَّةِ بِالتَّجْرِيدِ يُفْرَضُ عَلَى سَبِيلِ الجَزَاءِ عَلَى الشَّخْصِ الَّذِي تَنُمُّ بَعْضُ تَصَرُّفَاتِهِ عَنْ عَدَمِ الوَلَاءِ لِوَطَنِهِ، أَوْ عَنْ دَعَارَةٍ إِثْمِيَّةٍ "
بو برويطة

تَأَثُّمِيّ[اسْمٌ مَنْسُوب]

1433هـ=2011م

التَّأَثُّمِيُّ : المَنْسُوبُ إِلَى التَّأَثُّمِ.

"لَوَجَدْنَا أَنَّ آلِيَّةَ الِاسْتِبْطَانِ التَّأَثُّمِيِّ لِهَزِيمَةِ حُزَيْرَانَ 1967، قَدِ اشْتَغَلَتْ عَلَى مُسْتَوًى أَعْمَقَ بِكَثِيرٍ، وَأَخْفَى بِكَثِيرٍ، عَلَى نَحْوٍ لَا يُلْحَظُ عَادَةً، إِلَّا فِي سَيْرُورَاتِ النُّكُوصِ النَّفْسِيِّ"
جورج طرابيشي
