لَذْلَاذ[صِفَة]

ن150هـ=767م

اللَّذْلَاذُ : الخَفِيفُ السَّرِيعُ.

قَطَّاعِ أَقْرَانِ القَطَا هَذَّاذِ وَكُلِّ ذَبٍّ أَكْحَلِ المَقَاذِي أَعْيَسَ مِلْسَاسِ النَّدَى مِلْجَاذِ لِكُلِّ عَيَّالِ الضُّحَى لَذْلَاذِ
عمرو بن حَميل الأسديّ
أراجيز المقلّين: تحقيق: محمّد يحيى زين الدّين، مجلّة مجمع اللُّغة العربيّة، دمشق، مج57، ج(1، 2)، 1982م، مج57، ج(1، 2)، 1982م، ج3، ج4، مج59، ج2، 1984م، ج4، مج68، ج2، 1993م، مج70، ج2، ‏438‏.

لَذْلَاذ[اسْم]

638هـ=1240م

اللَّذْلَاذُ : الذِّئْبُ.

وَاللَّيْثُ قَضْقَاضًا أَحَقُّ بِجَاذِبٍ يَحْمِيهِ مِنْ ذِئْبِ الغَضَا لَذْلَاذَا
ابن الأبّار القُضاعيّ
ديوان ابن الأبّار: ابن الأبّار القُضاعيّ (ت، 658هـ)، قراءة وتعليق: عبد السَّلام الهرّاس، وزارة الأَوقاف والشؤون الإسلامية، الرباط، (1420ه/ 1999م)، ‏195‏.

لَذْلَذَة[مَصْدَر]

390هـ=1000م

اللَّذْلَذَةُ : الخِفَّةُ وَالإِسْرَاعُ.

"وَلَذْلَذَ الرَّجُلُ فِي مَشْيِهِ لَذْلَذَةً "
ابن الحدّاد المَعافريّ السَّرَقُسْطيّ
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏2/‏476‏.

لَذْلَذَ[لَازِم]

390هـ=1000م

لَذْلَذَ الشَّخْصُ فِي المَشْيِ وَنَحْوِهِ: خَفَّ وَأَسْرَعَ.

" لَذْلَذَ الرَّجُلُ فِي مَشْيِهِ لَذْلَذَةً "
ابن الحدّاد المَعافريّ السَّرَقُسْطيّ
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏2/‏476‏.
