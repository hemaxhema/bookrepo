ثَمَد[اسْم]

قبل 150ق.هـ=476م

الثَّمَدُ : الحُفْرَةُ يَجْتَمِعُ فِيهَا المَاءُ القَلِيلُ.

أَلِمَالِكٍ طِيبُ البِلَادِ، وَرَعْيُهَا وَلِيَ الثِّمَادُ ، وَرَعْيُهُنَّ المُجْدِبُ؟
منسوب إلى هُنَيّ بن أحمر الكِنانيّ
الشُّعراء الجاهليون الأوائل: تحقيق: عادل الفريجات، دار المشرق، بيروت، ط2، (2008م)، ‏470‏.

ثَمَد[اسْمُ جِنْس]

قبل 13ق.هـ=609م

الثَّمَدُ : المَاءُ القَلِيلُ الَّذِي لَا مَادَّ لَهُ.

القَائِدُ الخَيْلَ تَرْدِي فِي أَعِنَّتِهَا وِرْدَ القَطَا، هَجَّرَتْ، ظِمْئًا، إِلَى الثَّمَدِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏59‏.

ثَمَاد[مَصْدَر]

قبل 150ق.هـ=476م

ثَمَادُ الفَرَسِ وَنَحْوِهِ: سِمَنُهُ مَعَ شِدَّةِ حَرَكَتِهِ.

"أُحِبُّ كُلَّ حَيِيَّةِ الفُؤَاد، سَبُوحٍ جَوَاد، سَلِسَةِ القِيَاد، شَدِيدَةِ الِاعْتِمَاد، فِي الدَّفْعِ وَالِاشْتِدَاد، ذَاتِ هِبَابٍ وَثَمَاد "
منسوب إلى جمعة/خمعة بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏60‏.

ثَمُود[اسْمُ جِنْس]

قبل 50ق.هـ=573م

ثَمُودٌ : قَوْمٌ مِنَ العَرَبِ البَائِدَةِ، اتَّصَفُوا بالشِّدَّةِ وَالقُوَّةِ.

سِيرِي إِيَادُ قَدْ رَأَيْنَا عَجَبَا لَا أَصْلُكُمْ مِنَّا فَسَامِي الطَّلَبَا دَارَ ثَمُودٍ إِذْ رَأَيْتِ السَّبَبَا
عامر بن الظَّرِب العَدْوانيّ
الشُّعراء الجاهليون الأوائل: تحقيق: عادل الفريجات، دار المشرق، بيروت، ط2، (2008م)، ‏189‏.

إِثْمِد[اسْمُ جِنْس]

قبل 50ق.هـ=573م

الإِثْمِدُ : حَجَرٌ يُكْتَحَلُ بِهِ.

وَإِذَا تَبَسَّمَ، قُلْتَ: شَوْكُ سَيَالَةٍ، أَوْ أُقْحُوَانُ صَرِيمَةٍ، مَعْهُودُ
رَيَّانُ، رُكِّبَ فِي نُخَالَةِ إِثْمِدٍ خُضْرٍ، تُزَيِّنُهُ غَدَائِرُ سُودُ
كعب بن الرُّواع الأسديّ
ديوان بني أسد أشعار الجاهليين والمخضرمين: جمع وتحقيق دراسة: مُحمّد علي دقَّة، دار صادر، بيروت، ط1، (1999م)، ‏2/‏175‏.

مَثْمُود[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المَثْمُودُ مِنَ الرِّجَالِ: المُسْتَنْزَفُ مَاؤُهُ لِكَثْرَةِ الجِمَاعِ.

وَإِذَا الوَصْلُ لَمْ يُوَاتِيكَ إِلَّا نَكَدًا، أَوْ مُقَلَّلًا، مَخْفُورَا
أَوْ كَمَاءِ المَثْمُودِ ، بَعْدَ جِمَامٍ زَرِمِ الدَّمْعِ، لَا يَؤُوبُ نَزُورَا
عَديّ بن زيد العِباديّ
ديوان عَديّ بن زيد العِباديّ: حققه وجمعه: محمّد جبار المعيبد، شركة دار الجمهورية، بغداد، (1385هـ/ 1965م)، ‏63‏.
64هـ=684م

المَثْمُودُ مِنَ المِيَاهِ: المَنْزُوحُ لِكَثْرَةِ وُرَّادِهِ.

كَأَنَّهَا قَارِبٌ، أَقْرَى حَلَائِلَهُ ذَاتَ السَّلَاسِلِ، حَتَّى أَيْبَسَ العُودُ
فَظَلَّ مُرْتَبِئًا، وَالأُخْذُ قَدْ حَمِيَتْ وَظَنَّ أَنَّ سَبِيلَ الأُخْذِ مَثْمُودُ
الأخطل
شعر الأخطل أبي مالك غياث بن غوث التغلبي: صنعة: السُّكّري، رواية: أبي جعفر محمّد بن حبيب، تحقيق: فخر الدين قباوة، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط4، (1416هـ/ 1996م)، ‏80‏.
198هـ=813م

المَثْمُودُ مِنَ النَّاسِ: المَجْرُوحُ المُشْرِفُ عَلَى الهَلَاكِ.

حَتَّى أَجَابُوكَ؛ مِنْ مُسْتَأمِنٍ حَذِرٍ رَاجٍ، وَمُنْتَظِرٍ حَتْفًا، وَمَثْمُودِ
صَريع الغواني
شرح ديوان صريع الغواني: صَريع الغواني (ت، 208هـ)، عني بتحقيقه والتعليق عليه: سامي الدهّان، دار المعارف، القاهرة، ط3، (1985م)، ‏3/‏169‏.
360هـ=971م

المَثْمُودُ مِنَ النَّاسِ: المُسْتَهْلَكُ مَالُهُ بِالعَطَاءِ لِسَائِلِيهِ.

رُبَّ عَقْلٍ، حَازِمٍ، مَحْدُودِ وَأَخِي حُمْقَةٍ، بِهَا، مَجْدُودِ
وَبَطِيءٍ، عَنْ حَظِّهِ، ذِي ثَرَاءٍ وَحَرِيصٍ، عَلَى الغِنَى، مَثْمُودِ
من شواهد حمزة بن الحسن الأصفهانيّ
كتاب الأمثال الصّادرة عن بيوت الشِّعر: حمزة بن الحسن الأصفهانيّ (ت، 360هـ)، تحقيق: أحمد بن محمّد الضبيب، دار المدار الإسلامي، بيروت، ط1، (2009م)، ‏616‏.
360هـ=971م

المَثْمُودُ مِنَ النَّاسِ: الخَلِيُّ البَطْنِ.

أَوْغَلُ فِي التَّطْفِيلِ مِنْ مَثْمُودِ أَلْزَمُ لِلشِّوَاءِ مِنْ سَفُّودِ
من شواهد حمزة بن الحسن الأصفهانيّ
الدّرّة الفاخرة في الأمثال السّائرة: حمزة بن الحسن الأصفهانيّ (ت، 360هـ)، حققه وقدم له ووضع حواشيه وفهارسه: عبد المجيد قطامش، دار المعارف، القاهرة، (1972م)، ‏426‏.

ثَمَدَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

ثَمَدَ القَوْمَ: سَأَلَهُمُ العَطَاءَ فِي إِلْحَاحٍ حَتَّى كَادَ مَالُهُمُ يَنْفَدُ.

إِذَا نَزَلُوا ذَا ضَرْغَدٍ فَعُتَائِدًا يُغَنِّيهِمُ، فِيهَا نَقِيقُ الضَّفَادِعِ
قُعُودًا لَدَى أَبْيَاتِهِمْ يَثْمِدُونَهَا رَمَى اللهُ فِي تِلْكَ الأَكُفِّ الكَوَانِعِ
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏172‏-‏ 173‏.
276هـ=889م

ثَمَدَ المَاءَ وَنَحْوَهُ: اسْتَنْزَفَهُ، وَاسْتَنْفَدَهُ.

خَلِيلَيَّ، مَا بَعْدَ الشَّبَابِ رَزِيَّةٌ يُجَمُّ لَهَا مَاءُ الشُّؤُونِ، وَيُعْتَدُ
فَلَا تَلْحَيَا، إِنْ فَاضَ دَمْعٌ، لِفَقْدِهِ فَقَلَّ، لَهُ، بَحْرٌ، مِنَ الدَّمْعِ، يُثْمَدُ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏2/‏584‏-‏ 585‏.

ثِمَاد[اسْم]

قبل 13ق.هـ=609م

الثِّمَادُ : المَاءُ القَلِيلُ الَّذِي لَا مَادَّ لَهُ.

"لَقَفَاكَ أَحْسَنُ مِنْ وَجْهِهِ...، وَلَثِمَادُكَ أَغْزَرُ مِنْ غَدِيرِهِ"
قيس بن رفاعة الواقفيّ الأوسيّ
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏1/‏307‏.

ثَمِيد[صِفَةٌ مُشَبَّهَة]

67هـ=686م

الثَّمِيدُ مِنَ المَاءِ وَنَحْوِهِ: القَلِيلُ.

ثُمَّ حَمْرَا، لَمْ يُرِدْهَا ظَنَّهَا غَيْثًا ثَمِيدَا
من رواية عُبيد بن شَريّة الجُرْهميّ
أخبار عبيد بن شَريَّة الجرهميّ في أخبار اليمن وأشعارها وأنسابها على الوفاء والكمال والحمد لله على كل حال: عُبيد بن شَريَّة الجُرْهميّ (ت، 67هـ)- ضمن: كتاب التيجان في ملوك حِمْيَر: وهب بن منبه (ت، 114هـ)، رواية: أبي محمد عبد الملك بن هشام عن أسد بن موسى عن أبي إدريس بن سنان عن جده لأمه وهب بن منبه: تحقيق ونشر: مركز الدراسات والأبحاث اليمنية، صنعاء، ط1، (1347هـ)، ‏356‏.

ثَمْد[مَصْدَر]

110هـ=728م

الثَّمْدُ : المَنْحُ عَطَاءً وَوَصْلًا.

كَأَنْ لَمْ تَخِدْ بِالوَصْلِ، يَا هِنْدُ، بَيْنَنَا جَلَنْبَاةُ أَسْفَارٍ، كَجَنْدَلَةِ الصَّمْدِ
بَلَى، ثُمَّ لَمْ نَمْلِكْ مَقَادِيرَ، سُدِّيَتْ لَنَا مِنْ كَدَا هِنْدٍ، عَلَى قِلَّةِ الثَّمْدِ
الطِّرمّاح
ديوان الطِّرمَّاح: عني بتحقيقه: عِزّة حسن، دار الشرق العربي، بيروت- حلب، ط2، (1414هـ/ 1994م)، ‏130‏.

اسْتَثْمَدَهُ : أَلَحَّ فِي طَلَبِ مَعْرُوفِهِ.

"ثُمَّ إِنِّي الآنَ، وَقَدْ مَضَى لَهُ سِنُونَ، أُعَانُّ الخَاطِرَ وَأَسْتَثْمِدُهُ ، وَأُفَانِيهِ وَأَتَوَدَّدُهُ، عَلَى أَنْ يَسْمَحَ لِي بِمَا كَانَ أَرَانِيهِ"
أبو الفتح ابن جنّي
الخصائص: أبو الفتح ابن جنّي (ت، 392هـ)، تحقيق: محمّد علي النجار، دار الكتب المصرية- المكتبة العلمية، القاهرة، (1371هـ/ 1952م)، ‏1/‏207‏.

أَثْمَدَ[مُتَعَدٍّ]

ن487هـ=1094م

أَثْمَدَ الشَّيْءَ: أَخَذَ مِنْهُ حَتَّى نَفِدَ.

"وَقَدْ أَثْمَدَتِ الخُطُوبُ يَنْبُوعِي، وَمَحَتِ الحَادِثَاتُ رُبُوعِي"
القاضي أمير الدّولة العسقلانيّ
الذخيرة في محاسن أهل الجزيرة: ابن بسّام الشّنْتَرِيني (ت، 542هـ)، تحقيق: إحسان عبّاس، دار الثَّقافة، بيروت، ط1، (1417ه/ 1997م)، ‏4/‏622‏.

إِثْمَاد[مَصْدَر]

529هـ=1134م

الإِثْمَادُ : اسْتِخْرِاجُ المَاءِ القَلِيلِ.

وَوُدِّي لَكَ الغَمْرُ النَّمِيرُ وَإِنْ عَدَتْ مَوَدَّاتُ قَوْمٍ وَهْيَ كَدْرٌ وَإِثْمَادُ
ابن مَسْعَدة العامريّ الغَرْناطيّ
قلائد العقيان ومحاسن الأعيان: ابن خاقان (ت، 529هـ)، حققه وعلق عليه: حسين يُوسف خريوش، مكتبة المنار، الزَّرقاء، ط1، (1409ه/ 1989م)، ‏578‏.

أَثْمَدِيّ[اسْمٌ مَنْسُوب]

ن593هـ=1197م

الأَثْمَدِيُّ : المَنْسُوبُ إِلَى الأَثْمَدِ.

وَتَجْعَلُ مَا قَدْ كَانَ لَبَّدَهُ النَّدَى هَبَاءً كَمَنْخُولٍ مِنَ الكُحْلِ أَثْمَدِيّ
ابن أرفع رأس
شذور الذهب ديوان ابن أرفع رأس الجياني الأندلسي شاعر الحكماء وحكيم الشعراء: ابن أرفع رأس (ت، 593هـ)، حققه وقدمه وعلق عليه: الهواري غزالي، دار الكتب العلمية، بيروت، ط1، (1439ه/ 2018م)، ‏140‏.

ثُمَيْد[اسْمٌ مُصَغَّر]

ن621هـ=1224م

الثُّمَيْدُ : تَصْغِيرُ الثَّمَدِ.

"مَاءٌ يُقَالُ لَهُ لَحِيظٌ، وَهُوَ ثُمَيْدٌ إِزَاءَ الخَذِيفَةِ"
ياقوت الحَمَويّ
معجم البلدان: ياقوت الحَمَويّ (ت، 626هـ)، دار صادر، بيروت، (1397ه/ 1977م)، ‏2/‏349‏.

إِثِّمَاد[مَصْدَر]

ن830هـ=1426م

الإِثِّمَادُ : الِاكْتِحَالُ بِالثَّمَدِ.

"قَائِمَةٌ بِالزِّينَةِ، مُولَعَةٌ بِالنَّظَافَةِ، لَا تُظْهِرُ لِزَوْجِهَا مَا يَكْرَهُ، مُتَوَلِّعَةٌ بِالطِّيبِ، مُوَاظِبَةٌ عَلَى الإِثِّمَادِ "
أبو عبد الله محمد بن محمد النفزاوي
كتاب الروض العاطر في نزهة الخاطر: أبو عبد الله محمد بن محمد النفزاوي (ت، 830هـ)، حققه وعلق عليه وقدم له وصنع فهارسه: فرج الحوار، الدار المتوسطية، تونس، ط1، (1439ه/ 2018م)، ‏160‏.

ثِمَادَة[مَصْدَر]

1016هـ=1607م

الثِّمَادَةُ : المَاءُ القَلِيلُ المُجْتَمِعُ فِي حُفْرَةٍ عَقِبَ المَطَرِ وَنَحْوِهِ.

"وَلَكِنَّهَا جَمْرَةُ لَيْل، وَأَثَرُ ثِمَادَةٍ لَا سَيْل، وَبِنَاءٌ عَلَى شَفا، وَعِلَّةٌ قَرِيبَةُ الشِّفا"
أحمد أفَندي بن شاهِين القبرسي الشاهيني
كراسة في تراجم ستة من أعيان ولاية دمشق وخمسة من أخبارها في القرن الحادي عشر الهجري السابع عشر الميلادي: الحسن بن محمد البوريني (ت، 1024هـ)، تحقيق ودراسة: مشهور عبد الرّحمن الحبازي، مجلة تراثيات، القاهرة، ع10، (1428ه/ 2007م)، (155- 203)، ‏194‏.

مُسْتَثْمِد[اسْمُ فَاعِل]

1077هـ=1666م

المُسْتَثْمِدُ : نَابِثُ التُّرَابِ لِيَسْتَخْرِجَ القَلِيلَ مِنَ المَاءِ.

يُلْجِى إِلَى مُخِّ العَرَاقِيبِ الطَّوَى وَيَجِيءُ فَقْدُ العِدِّ لِلْمُسْتَثْمِدِ
الحسن بن مسعود اليوسي
ديوان اليوسي: الحسن بن مسعود اليوسي (ت، 1102هـ)، جمع وتحقيق: عبد الجواد السقاط، تقديم: عباس الجراري، مطبعة دار المناهل، الرباط، (2016م)، ‏141‏.
