مَكْرُوه[اسْم]

قبل 150ق.هـ=476م

المَكْرُوهُ : الشِّدَّةُ، وَالمَشَقَّةُ.

فَقَلِّدُوا أَمْرَكُمْ، لِلّٰهِ دَرُّكُمُ! رَحْبَ الذِّرَاعِ، بِأَمْرِ الحَرْبِ مُضْطَلِعَا
لَا مُتْرَفًا إِنْ رَخَاءُ العَيْشِ سَاعَدَهُ وَلَا إِذَا عَضَّ مَكْرُوهٌ بِهِ؛ خَشَعَا
لَقيط بن يَعْمَر الإياديّ
ديوان لقيط بن يَعمر الإيادي: رواية: ابن الكلبيّ (ت، 204هـ)، شرح وتحقيق: محمّد ألتونجي، دار صادر، بيروت، ط1، (1998م)، ‏86‏-‏ 87‏.
ن2هـ=623م

المَكْرُوهُ : الشَّرُّ.

وَإِقْدَامِي عَلَى المَكْرُوهِ نَفْسِي وَضَرْبِي هَامَةَ البَطَلِ المُشِيحِ
ابن الإِطْنابة الخَزْرجيّ
كتاب الأضداد: أبو حاتم السِّجِسْتانيّ (ت، 255هـ)- ضمن: ثلاثة كتب الأضداد: الأصمعي وآخرون: نشرها: أوغست هفنر، المطبعة الكاثوليكية، بيروت، (1912م)، ‏125‏.

مَكْرُوه[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المَكْرُوهُ : المُبْغَضُ المَمْقُوتُ.

الحَرْبُ أَوَّلُ مَا تَكُونُ فَتِيَّةً تَبْدُو بِزِينَتِهَا لِكُلِّ جَهُولِ
حَتَّى إِذَا اسْتَعَرَتْ، وَشَبَّ ضِرَامُهَا، عَادَتْ عَجُوزًا غَيْرَ ذَاتِ خَلِيلِ
شَمْطَاءَ، جَزَّتْ رَأْسَهَا، وَتَنَكَّرَتْ مَكْرُوهَةً لِلشَّمِّ وَالتَّقْبِيلِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏353‏.
2ق.هـ=620م

المَكْرُوهُ مِنَ الأُمُورِ: المُسْتَقْبَحُ المَنْهِيُّ عَنْهُ.

قرآن كريم
قرآن كريم، ‏[ الإِسْرَاء : 37- 38 ].
ن150هـ=767م

المَكْرُوهُ مِنَ الأَعْمَالِ: مَا يُثَابُ تَارِكُهُ، وَلَا يُعَاقَبُ فَاعِلُهُ. [أصول الفقه]

"أَرَأَيْتَ رَجُلًا مُسْلِمًا أَمَرَ نَصْرَانِيَّا أَوْ يَهُودِيًّا، فَذَبَحَ لَهُ أُضْحِيَّتَهُ، هَلْ يُجْزِيهِ؟ قَالَ: نَعَمْ، يُجْزِيهِ، وَهُوَ مَكْرُوهٌ "
أبو حنيفة النّعمان
الأصل: محمّد بن الحسن الشَّيبانيّ (ت، 189هـ)، تحقيق ودراسة: محمّد بوينوكالن، وزارة الأوقاف والشؤون الإسلامية، الدوحة، دار ابن حزم، بيروت، ط1، (1433هـ/ 2012م)، ‏5/‏411‏.
ن179هـ=795م

المَكْرُوهُ : المُسْتَقْبَحُ عُرْفًا، المُحَرَّمُ شَرْعًا. [أصول الفقه]

"قَالَ مُوسَى: إِذَا قَالَ الرَّجُلُ: هَلَكَ النَّاسُ، فَهُوَ أَهْلَكُهُمْ. قَالَ أَبُو دَاوُدَ: قَالَ مَالِكٌ: إِذَا قَالَ ذَلِكَ تَحَزُّنًا لِمَا يَرَى فِي النَّاسِ... فَلَا أَرَى بِهِ بَأْسًا، وَإِذَا قَالَ ذَلِكَ عُجْبًا بِنَفْسِهِ، وَتَصَاغُرًا لِلنَّاسِ؛ فَهُوَ المَكْرُوهُ الَّذِي نُهِيَ عَنْهُ"
مالك بن أنس الأصبحيّ الحِمْيريّ
سنن أبي داود: أبو داود السِّجِسْتانيّ (ت، 275هـ)، حكم على أحاديثه وآثاره وعلّق عليه: محمّد ناصر الدين الألبانيّ، اعتنى به: مشهور حسن سلمان، مكتبة المعارف، الرياض، ط2، (1424هـ)، ‏901،،‏(رقم الحديث: 4983).
200هـ=815م

المَكْرُوهُ مِنَ الأَشْخَاصِ: المَغْصُوبُ المُجْبَرُ.

"وَاكْتَفَى بِالشَّرْطِ فِي عِشْرَتِهَا بِالمَعْرُوفِ، لَا أَنَّهُ أَبَاحَ أَنْ يُعَاشِرَ مَكْرُوهَةً بِغَيْرِ المَعْرُوفِ"
الشّافعيّ
الأمّ: الشّافعيّ (ت، 204هـ)، تحقيق وتخريج: رفعت فوزي عبد المطلب، دار الوفاء، المنصورة، ط1، (1422هـ/ 2001م)، ‏6/‏290‏.

مَكْرُوه[مَصْدَر]

ن10هـ=631م

مَكْرُوهُ الشَّخْصِ: شِدَّتُهُ، وَبَأْسُهُ.

أَضْرِبُ فِي العَوْرَةِ مَا فِيَّ إِنْ أُخْضِمْتُ، أَوْ أُقْضِمْتُ، لَمْ أَأْتَلِ
ذَلِكَ مَكْرُوهِي وَرَوْغِي، فَإِنْ أُحْمَلْ عَلَى الثِّقْلَةِ؛ لَا أَثْقُلِ
عَديّ بن وَداع الأزديّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏8/‏308‏.

كَرِيهَة[اسْم]

قبل 150ق.هـ=476م

الكَرِيهَةُ : النَّازِلَةُ الشَّدِيدَةُ.

أَمِنَ السَّوِيَّةِ أَنْ إِذَا اسْتَغْنَيْتُمُ وَأَمِنْتُمُ؛ فَأَنَا البَعِيدُ الأَخْيَبُ
وَإِذَا تَكُونُ كَرِيهَةٌ ؛ أُدْعَى لَهَا وَإِذَا يُحَاسُ الحَيْسُ؛ يُدْعَى جُنْدُبُ؟
منسوب إلى هُنَيّ بن أحمر الكِنانيّ
الشُّعراء الجاهليون الأوائل: تحقيق: عادل الفريجات، دار المشرق، بيروت، ط2، (2008م)، ‏468‏.
قبل 100ق.هـ=525م

الكَرِيهَةُ : الحَرْبُ الحَامِيَةُ.

حَتَّى نَرَى أَوْصَالَهُمْ، وَجَمَاجِمًا مِنْهُمْ، عَلَيْهَا الخَامِعَاتُ وُقُوعَا
وَالخَيْلَ تَقْتَحِمُ الغُبَارَ عَوَابِسًا يَوْمَ الكَرِيهَةِ ، مَا يُرِدْنَ رُجُوعَا
مُهَلْهِل بن ربيعة التَّغْلِبيّ
ديوان المُهَلْهِل: شرح وتحقيق: أنطوان محسن القوَّال، دار الجيل، بيروت، ط1، (1415هـ/ 1995م)، ‏50‏-‏ 51‏.
قبل 50ق.هـ=573م

كَرِيهَةُ السَّيْفِ: ضَرْبَتُهُ القَاصِمَةُ.

فِي كَفِّهِ مُرْهَفُ الغِرَارِ، إِذَا أَهْوَى بِهِ مِنْ كَرِيهَةٍ ؛ رَسَبَا
السَّموءل بن عُريض/غُريض الغَسّانيّ الأزديّ
ديوانا عُروة بن الورد والسَّموأل: تحقيق: كرم البستاني، دار صادر، بيروت، د.ت. ‏76‏.
قبل 13ق.هـ=609م

كَرِيهَةُ الفَرَسِ وَنَحْوِهِ: بَادِرَتُهُ فِي الشِّدَّةِ.

وَلَقَدْ أَذْعَرُ السُّرُوبَ بِطِرْفٍ مِثْلِ شَاةِ الإِرَانِ، غَيْرِ مُذَالِ
غَيْرِ أَقْنَى، وَلَا أَصَكَّ، وَلَكِنْ مِرْجَمٌ، ذُو كَرِيهَةٍ وَنِقَالِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏109‏.
ن27هـ=648م

كَرِيهَةُ السَّيْفِ: مَا تَقَعُ عَلَيْهِ ضَرْبَتُهُ مِنَ الأَعْضَاءِ.

يَتَنَاهَبَانِ المَجْدَ، كُلٌّ وَاثِقٌ بِبَلَائِهِ، وَاليَوْمُ يَوْمٌ أَشْنَعُ
وَكِلَاهُمَا مُتَوَشِّحٌ ذَا رَوْنَقٍ عَضْبًا، إِذَا مَسَّ الكَرِيهَةَ يَقْطَعُ
أبو ذؤيب الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏1/‏38‏.
ن216هـ=831م

ذُو الكَرِيهَةِ : السَّيْفُ المَضَاءُ.

"وَمِنْهَا ذُو الكَرِيهَةِ ، وَهُوَ الَّذِي يَمْضِي عَلَى الضَّرَائِبِ الشِّدَادِ"
الأصمعيّ
كتاب السّلاح: الأصمعيّ (ت، 216هـ)، حققه وقدم له: محمّد جبار المعيبد، مجلة المورد، بغداد، مج16، ع2، (1407هـ/ 1987م)، ‏78‏.

كَرْه[مَصْدَر]

قبل 150ق.هـ=476م

الكَرْهُ : الغَصْبُ، وَالإِجْبَارُ.

لَقَدْ أَيْقَنَتْ نَفْسُ الفَتَى غَيْرَ بَاطِلٍ وَإِنْ عَاشَ حِينًا أَنَّهُ سَوْفَ يَهْلِكُ
وَيَشْرَبُ بِالكَأْسِ الذُّعَافِ شَرَابُهَا وَيَرْكَبُ حَدَّ المَوْتِ كَرْهًا وَيَسْلُكُ
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏63‏-‏64‏.

جَازَتْ مُلِمَّاتُ الزَّمَانِ حُدُودَهَا وَاسْتَفْرَغَتْ أَيَّامُهَا مَجْهُودَهَا
وَقَضَتْ عَلَيْنَا بِالمَنُونِ، فَعَوَّضَتْ بِالكَرْهِ مِنْ بِيضِ اللَّيَالِي سُودَهَا

ن145هـ=762م

الكَرْهُ مِنَ الإِبِلِ: الشَّدِيدُ الرَّأْسِ.

يَقْتَصِلُ القَصْلَ بِنَابٍ حُدَّادْ كَرْهِ الحِجَاجَيْنِ، شَدِيدِ الأَرْآدْ فِي رَأْسِهِ مُرْتَهِشَاتُ الأَحْيَادْ
