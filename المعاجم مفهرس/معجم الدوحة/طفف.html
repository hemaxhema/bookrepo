أَطَفَّ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

أَطَفَّ الشَّيْءَ: رَفَعَهُ وَهَيَّأَهُ.

وَلَمَّا طَفَا طَافٍ عَلَيْهِ وَقَدْ طَفَا طَفِيفٌ أَطَفَّ الطَّبْلَ بِالرَّعْدِ مُسْقِفُ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏328‏.
قبل 13ق.هـ=609م

أَطَفَّ الشَّيْءَ: أَبْرَزَهُ وَقَرَّبَهُ.

أَطَفَّ لِأَنْفِهِ المُوسَى قَصِيرٌ لِيَجْدَعَهُ، وَكَانَ بِهِ ضَنِينَا
فَأَهْوَاهُ لِمَارِنِهِ فَأَضْحَى طِلَابَ الوِتْرِ مَجْدُوعًا مَشِينًا
عَديّ بن زيد العِباديّ
ديوان عَديّ بن زيد العِباديّ: حققه وجمعه: محمّد جبار المعيبد، شركة دار الجمهورية، بغداد، (1385هـ/ 1965م)، ‏183‏.
200هـ=815م

أَطَفَّ القَدَحَ وَنَحْوَهُ: نَقَّصَهُ وَبَخَّسَهُ.

"مَلَأْتُ القَدَحَ... وَأَطْفَفْتُهُ "
أبو مِسْحَل الأعرابيّ
كتاب النّوادر: أبو مِسْحَل الأعرابيّ (ت، 230هـ)، عني بتحقيقه: عزة حسن، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏1/‏33‏.

أَطَفَّ[مُتَعَدٍّ بِالحَرْف]

ن2ق.هـ=620م

أَطَفَّ لَهُ: فَطِنَ لَهُ وَأَرَادَ خِدَاعَهُ.

أَطَفَّ لَهَا شَثْنُ البَنَانِ، جُنَادِفُ
أوس بن حَجَر المازنيّ التَّميميّ
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏3/‏248‏.
ن175هـ=791م

أَطَفَّ لَهُ: دَنَا مِنْهُ وَتَهَيَّأَ لَهُ.

أَطَفَّ لَهَا عَبَاقِيَةٌ سَرَنْدَى جَرِيءُ الصَّدْرِ، مُنْبَسِطُ اليَمِينِ
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏340‏.

طَفَّفَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

طَفَّفَ الكَيْلَ وَنَحْوَهُ: بَخَّسَهُ وَنَقَّصَهُ.

لَنَا صَاعٌ إِذَا كِلْنَا خُصُومًا نُطَفِّفُهَا ، وَنُوفِي لِلْوَفِيِّ
جُندُب بن ضمرة اللّيثيّ الكنانيّ
الجليس الصّالح الكافي والأنيس الناصح الشافي: أبو الفرج ابن طَرار الجَريريّ النَّهْروانيّ (ت، 390هـ)، دراسة وتحقيق: إحسان عبّاس، محمّد مرسي الخولي، عالم الكتب، بيروت، ط1، (1413هـ/ 1993م)، ‏2/‏276‏.
ن36هـ=656م

طَفَّفَهُ : زَادَ عَلَيْهِ.

"وَفِي حَدِيثِ حُذَيْفَةَ (أَنَّهُ اسْتَسْقَى دِهْقَانًا فَأَتَاهُ بِقَدَحٍ فِضَّةً فَحَذَفَهُ بِهِ، فَنَكَّسَ الدِّهْقَانُ، وَطَفَّفَهُ القَدَحُ)"
حذيفة بن اليَمان العبسيّ
النهاية في غريب الحديث والأثر: ابن الأثير أبو السعادات (ت، 606هـ)، تحقيق: محمود محمد الطناحي، طاهر أحمد الزاوي، المكتبة الإسلامية، بيروت، ط1، (1383ه/ 1963م)، ‏3/‏129‏.
ن73هـ=692م

طَفَّفَ الجِدَارَ وَنَحْوَهُ: عَلَاهُ وَارْتَفَعَ عَنْهُ.

"فَكُنْتُ فَارِسًا يَوْمَئِذٍ، فَسَبَقْتُ النَّاسَ؛ طَفَّفَ بِي الفَرَسُ مَسْجِدَ بَنِي زُرَيْقٍ"
ابن عمر
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏8/‏68‏-‏ 69،،‏(رقم الحديث: 4487).

طَفَّفَ[لَازِم]

ن36هـ=656م

طَفَّفَ فِي الصَّلَاةِ: خَفَّفَ مِنْهَا بِمَا لَا يُخِلُّ بِصِحَّتِهَا.

"عَنْ حُذَيْفَةَ أَنَّهُ رَأَى رَجُلًا يُصَلِّي فَطَفَّفَ "
حذيفة بن اليَمان العبسيّ
كتاب السنن الكبرى: النّسائيّ (ت، 303هـ)، قدم له: عبد الله بن عبد المحسن التركي، أشرف عليه: شعيب الأرنؤوط، حققه وخرج أحاديثه: حسن عبد المنعم شلبي، مؤسسة الرسالة، بيروت، (2001م)، ‏2/‏85‏.

طَفَّفَ[مُتَعَدٍّ بِالحَرْف]

ن557هـ=1162م

طَفَّفَ إِلَيْهِ: دَنَا مِنْهُ وَاقْتَرَبَ.

"أَقَلُّ مُدَّتِهَا عِشْرُونَ يَوْمًا أَوْ أَحَدٌ وَعِشْرُونَ يَوْمًا، وَرُبَّمَا زَادَتْ إِلَى الأَرْبَعِينَ، وَرُبَّمَا طَفَّفَتْ إِلَى السِّتِّينَ"
ابن زُهْر
كتاب التيسير في المداواة والتدبير: ابن زُهْر (ت، 557هـ)، تحقيق: ميشيل خوري، تقديم: محيي الدين صابر، دار الفكر، دمشق، ط1، (1403ه/ 1983م)، ‏2/‏395‏.
ن776هـ=1374م

طَفَّفَ فِي المَسِيرِ وَنَحْوِهِ: تَدَافَعَ فِيهِ وَخَفَّ.

"بَيْنَمَا أَنَا فِي بَعْضِ الطُّرُقِ، وَقَدْ وَصَلَتِ الهَاجِرَة، وَتَبَرَّجَتِ المَفَازَةُ الفَاجِرَة ... وَالطِّيَّةُ تُطَفِّفُ فِي المَسِير ... وَالمَاءُ بِمَنْزِلَةِ الإِكْسِير، إِذْ رُفِعَتْ لِي عَلَى البُعْدِ سَرْحَةٌ فِي فَرِيدَةٍ عَنِ اللَّذَّات، وَالوَشَائِجِ المُوَلَّدَات"
لسان الدِّين ابن الخطيب
ريحانة الكتّاب ونجعة المنتاب: لسان الدِّين ابن الخطيب (ت، 776هـ)، حققه ووضع مقدمته وشواهده: محمّد عبد الله عنان، مكتبة الخانجي- مطبعة العربية الحديثة، القاهرة، ط1، (1400ه/ 1890م- 1401ه/ 1981م)، ‏2/‏270‏.

طَفَاف[اسْم]

قبل 13ق.هـ=609م

الطَّفَافُ مِنَ المِكْيَالِ وَنَحْوِهِ: مَا نَقَصَ مِنْهُ.

لَنَا صَاعٌ، إِذَا كِلْنَا، طَفَافٌ نُطَفِّفُهَا، وَنُوفِي لِلْوَفِيِّ
جُندُب بن ضمرة اللّيثيّ الكنانيّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: محمد باسل عيون السود، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏1/‏607‏.
ن175هـ=791م

الطَّفَافُ : مَا زَادَ عَلَى المِكْيَالِ وَنَحْوِهِ.

" وَالطَّفَافُ : مَا فَوْقَ المِكْيَالِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏406‏.

اسْتَطَفَّ[لَازِم]

قبل 13ق.هـ=609م

اسْتَطَفَّ الشَّيْءُ: عَلَا وَارْتَفَعَ.

قَدْ عُرِّيَتْ حِقْبَةً حَتَّى اسْتَطَفَّ لَهَا كِتْرٌ، كَحَافَةِ كِيرِ القَيْنِ مَلْمُومُ
علقمة الفَحْل
ديوان علقمة الفحل: شرح: الأعلم الشَّنْتَمَري (ت، 476هـ)، حقّقه: لطفي الصقال، درية الخطيب، راجعه: فخر الدين قباوة، دار الكتاب العربي- مطبعة الأصيل، حلب، ط1، (1389هـ/ 1969م)، ‏54‏.
73هـ=692م

اسْتَطَفَّ الشَّيْءُ: دَنَا وَتَهَيَّأَ.

لَا يَسْتَطِفُّ لَهُ مَالٌ فَيَتْرُكَهُ وَلَا يَقُولُ لِمَا يُعْطَاهُ: يَكْفِينِي
عبد الله بن هَمّام السَّلوليّ
شعر عبد الله بن همّام السَّلُولي: جمع وتحقيق ودراسة: وليد محمّد السَّراقبي، مركز جمعة الماجد للثقافة والتراث، دبي، ط1، (1417هـ/ 1996م)، ‏107‏.

اسْتَطَفَّ[مُتَعَدٍّ]

ن68هـ=687م

اسْتَطَفَّ الشَّيْءَ: اسْتَكْمَلَهُ وَاسْتَوْفَاهُ.

"إِنِّي لَأُحِبُّ أَنْ أَتَزَيَّنَ لِلْمَرْأَةِ كَمَا أُحِبُّ أَنْ تَزَيَّنَ لِي... وَمَا أُحِبُّ أَنْ تَسْتَطِفَّ جَمِيعَ حَقٍّ لِي عَلَيْهَا"
ابن عبّاس
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏15/‏119،،‏(رقم الحديث: 14844).

طَفّ[اسْم]

قبل 13ق.هـ=609م

طَفُّ المَكَانِ وَنَحْوِهِ: جَانِبُهُ وَنَاحِيَتُهُ.

فَإِيَّاكُمُ وَالطَّفَّ ، لَا تَقْرَبُنَّهُ، وَلَا المَاءَ، إِنَّ المَاءَ لِلْقَوْدِ وَاصِلُ
قيس بن مسعود الشَّيْبانيّ
كتاب الأمثال: مؤرِّج السَّدوسيّ (ت، 195هـ)، حققه وقدم له ووضع فهارسه: رمضان عبد التوّاب، دار النهضة العربية، بيروت، (1983م)، ‏84‏.
ن11هـ=632م

طَفُّ الصَّاعِ وَنَحْوِهِ: الجُزْءُ الَّذِي لَمْ يَمْتَلِئْ مِنْهُ.

"إِنَّ أَنْسَابَكُمْ هَذِهِ لَيْسَتْ بِسِبَابٍ عَلَى أَحَدٍ، وَإِنَّمَا أَنْتُمْ وَلَدُ آدَمَ، طَفُّ الصَّاعِ لَمْ تَمْلَؤُوهُ، لَيْسَ لِأَحَدٍ فَضْلٌ عَلَى أَحَدٍ إِلَّا بِالدِّينِ أَوْ عَمَلٍ صَالِحٍ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏28/‏548،،‏(رقم الحديث: 17313).

طَفِيف[اسْمُ مَفْعُول]

ن449هـ=1057م

الطَّفِيفُ مِنَ الكَيْلِ وَنَحْوِهِ: المَبْخُوسُ المَنْقُوصُ.

فَإِنْ أُعْطَ القَلِيلَ يَكُنْ هَنِيئًا يَجِيءُ المُسْتَمِيحَ بِغَيْرِ شَفِّ
وَلَوْ كَانَ الكَثِيرُ لَقَلَّ عِنْدِي وَأَهْوِنْ بِالطَّفِيفِ المُسْتَطَفِّ!
أبو العلاء المَعَرّيّ
شرح اللّزوميّات: نظم: أبي العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: منير المدني وآخرين، إشراف ومراجعة: حسين نصار، الهيئة المصرية العامة للكتاب، القاهرة، (1992- 1994م)، ‏3/‏404‏.

"الخَرِيطَةُ المُتَخَيَّلَةُ لِفِلَسْطِينَ الجَدِيدَةِ مُضْطَرِبَةٌ فِي الوَعْيِ السِّيَاسِيِّ الفِلَسْطِينِيِّ؛ وَمِنْ هُنَا ذَلِكَ القَلَقُ النَّاجِمُ عَنْ عَدَمِ التَّحْدِيدِ فِي تَعَابِيرَ مِثْلِ (تَعْدِيلَاتٍ طَفِيفِيَّةٍ ) عَلَى حُدُودِ 1967"
أحمد جميل عزم وآخرون
قضية فلسطين ومستقبل المشروع الوطني الفلسطيني: أحمد جميل عزم وآخرون ، المركز العربي للأبحاث ودراسة السياسات، بيروت، ط1، ج1، (2015م)، ج2، (2016م)، ‏2/‏389‏.
