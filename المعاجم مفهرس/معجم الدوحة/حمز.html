مَحْمُوز[اسْمُ مَفْعُول]

ن15هـ=636م

المَحْمُوزُ مِنَ النِّصَالِ: الصُّلْبُ المَسْنُونُ المُحَدُّ.

فَهَيَّجَهَا وَانْشَامَ نَقْعًا كَأَنَّهُ إِذَا لَفَّهَا، ثُمَّ اسْتَمَرَّ، سَحِيلُ
مُنِيبًا وَقَدْ أَمْسَى، تَقَدَّمَ وِرْدَهَا أُقَيْدِرُ مَحْمُوزُ القِطَاعِ نَذِيلُ
أبو خِراش الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏120‏.

حَامِز[اسْمُ فَاعِل]

ن22هـ=643م

الحَامِزُ مِنَ الهَمِّ: المُقْلِقُ المُوجِعُ.

فَلَمَّا شَرَاهَا فَاضَتِ العَيْنُ عَبْرَةً وَفِي الصَّدْرِ حُزَّازٌ مِنَ الوَجْدِ حَامِزُ
الشَّمّاخ بن ضرار الذُّبْيانيّ
ديوان الشَّمّاخ بن ضرار الذبياني: حققه وشرحه: صلاح الدين الهادي، دار المعارف، القاهرة، (1968م)، ‏190‏.

حَامِز[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

الحَامِزُ مِنْ كُلِّ شَيْءٍ: الشَّدِيدُ.

" الحَامِزُ : الشَّدِيدُ مِنْ كُلِّ شَيْءٍ. وَرُجُلٌ حَامِزُ الفُؤَادِ: شَدِيدُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏168‏.
ن215هـ=830م

الحَامِزُ مِنَ الأَطْعِمَةِ: الحَامِضُ اللَّاذِعُ.

"وَمِنَ اللَّبَنِ الرَّثِيئَةُ وَهُوَ أَنْ يُحْلَبَ عَلَى الحَامِضِ فَيَخْثُرَ... وَمِنْهُ المُثْمِرُ وَالمَغِيرُ الشَّدِيدُ الحُمُوضَةِ إِلَى المَرَارَةِ... ثُمَّ الحَامِضُ هُوَ الحَامِزُ "
أبو زيد الأنصاريّ
البُلْغة في شذور اللّغة: نشرها: أوغست هفنر، لويس شيخو، المطبعة الكاثوليكية، بيروت، ط2، (1914م)، ‏144‏.

حَمَّاز[صِيغَةُ مُبَالَغَة]

ن22هـ=643م

الحَمَّازُ : الشَّدِيدُ.

وَفِي النَّفْسِ حَمَّازٌ مِنَ اللَّوْمِ حَامِزُ
الشَّمّاخ بن ضرار الذُّبْيانيّ
المُنتَخَب من غريب كلام العَرَب: كُراع النَّمْل الهُنائيّ الأزديّ (ت، 309هـ)، تحقيق: محمّد بن أحمد العمري، جامعة أم القرى، مكة المكرمة، ط1، (1409هـ/ 1989م)، ‏2/‏673‏.

أَحْمَز[صِفَةٌ مُشَبَّهَة]

ن53هـ=673م

الأَحْمَزُ مِنَ الرِّجَالِ: الشَّدِيدُ الصُّلْبُ.

"أَمَا وَاللهِ، لَئِنْ وَصَلَ إِلَيَّ لَيَجِدُنِي أَحْمَزَ ، ضَرَّابًا بِالسَّيْفِ"
زياد ابن أبيه
تاريخ اليعقوبيّ: اليعقوبيّ (ت، 292هـ)، تحقيق: عبد الأمير مهنّا، شركة الأعلميّ للمطبوعات، بيروت، ط1، (1431هـ/ 2010م)، ‏2/‏126‏.

أَحْمَز[اسْمُ تَفْضِيل]

ن68هـ=687م

أَحْمَزُ الأَشْيَاءِ: أَشَدُّهَا وَأَمْتَنُهَا.

"أَفْضَلُ الأَشْيَاءِ أَحْمَزُهَا ؛ أَيْ: أَشَدُّهَا وَأَمْتَنُهَا"
ابن عبّاس
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏168‏.

مَحْمَز[مَصْدَرٌ مِيمِيّ]

ن79هـ=698م

المَحْمَزُ : المَشَقَّةُ وَالصُّعُوبَةُ.

إِنَّ الفَرَزْدَقَ بَرَّزَتْ حَلَبَاتُهُ عَفْوًا، وَغُودِرَ فِي الغُبَارِ جَرِيرُ
مَا كُنْتَ أَوَّلَ مَحْمَزٍ عَثَرَتْ بِهِ آبَاؤُهُ، إِنَّ اللَّئِيمَ عَثُورُ
سُراقة بن مرداس البارقيّ
كتاب جُمَل من أنساب الأشراف: أحمد بن يحيى البَلاذُري (ت، 279هـ)، ج(2- 13)، حققه وقدم له: سهيل زكّار، رياض زركلي، إشراف: مكتب البحوث والدراسات، دار الفكر، بيروت، ط1، (1417هـ/ 1996م)، ‏12/‏217‏.

حَمَازَة[مَصْدَر]

ن154هـ=771م

حَمَازَةُ الشَّيْءِ: لَذْعُهُ وَحُمُوضَتُهُ.

"شَرِبَ عُمَرُ شَرَابًا فِيهِ حَمَازَةٌ "
أبو عمرو بن العَلاء
غريب الحديث: ابن قتيبة (ت، 276هـ)، تحقيق: عبد الله جبوري، إحياء التراث الإسلامي- مطبعة العاني، بغداد، ط1، (1397هـ/ 1977م)، ‏1/‏270‏.
ن270هـ=883م

الحَمَازَةُ : الشِّدَّةُ وَالصَّلَابَةُ.

"الَّذِي عَلَاهُمْ بِنُكَتِه، وَقَهَرَهُمْ بِأَدِلَّتِه، وَبَايَنَهُمْ بِشَهَامَتِه، وَظَهَرَ عَلَيْهِمْ بِحَمَازَتِه ، التَّقِيُّ فِي دِينِه، النَّقِيُّ فِي حَسَبِه"
داود بن عليّ الأصبهانيّ
مناقب الشّافعيّ: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: السيد أحمد صقر، مكتبة دار التراث، القاهرة، د.ت. ‏1/‏83‏.

حَمَزَ[مُتَعَدٍّ]

ن175هـ=791م

حَمَزَ اللَّوْمُ وَنَحْوُهُ القَلْبَ: أَوْجَعَهُ.

" حَمَزَ اللَّوْمُ فُؤَادَهُ وَقَلْبَهُ؛ أَيْ: أَوْجَعَهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏167‏.
ن206هـ=821م

حَمَزَ الجِلْدَ وَنَحْوَهُ: لَذَعَهُ.

"قَدْ حَمَزَ جِلْدِي هَذَا الَّذِي جَعَلْتُمْ عَلَى جُرْحِي"
أبو عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏152‏.
ن220هـ=835م

حَمَزَهُ الأَمْرُ: قَبَضَهُ وَغَمَّهُ.

"كَلَّمْتُ فُلَانًا بِكَلِمَةٍ حَمَزَتْ فُؤَادَهُ"
عليّ بن المبارك اللّحيانيّ الهُذَليّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏379‏.

حَمُوز[صِفَةٌ مُشَبَّهَة]

ن207هـ=822م

الحَمُوزُ مِنَ النَّبِيذِ وَنَحْوِهِ: القَوِيُّ فِي الهَضْمِ.

"اشْرَبْ مِنْ نَبِيذِكَ فَإِنَّهُ حَمُوزٌ لِمَا تَجِدُ"
أبو زكريّا الفَرّاء
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏380‏.

حَمْز[مَصْدَر]

216هـ=831م

حَمْزُ الشَّيْءِ: لَذْعُهُ وَحَرَارَتُهُ.

"يُعْجِبُنِي حَمْزُهُ وَحَرَاوَتُهُ"
رجل من الأعراب
كتاب الأضداد في كلام العرب: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، عني بتحقيقه: عزة حسن، دار طلاس، دمشق، ط2، (1996م)، ‏257‏.

حَمِيز[صِفَةٌ مُشَبَّهَة]

ن216هـ=831م

حَمِيزُ الفُؤَادِ وَنَحْوِهِ: شَدِيدُهُ وَمُنْقَبِضُهُ.

"وَيُقَالُ: إِنَّهُ لَأَصْمَعُ الفُؤَادِ؛ إِذَا كَانَ حَمِيزَ الفُؤَادِ مُنْقَبِضَهُ"
الأصمعيّ
كتاب خلق الإنسان: الأصمعيّ (ت، 216هـ)- ضمن: الكنز اللغوي في اللَّسَن العربي: سعى في نشره وتعليق حواشيه: أوغست هفنر، المطبعة الكاثوليكية، بيروت، (1903م)، ‏171‏.
ن309هـ=921م

الحَمِيزُ مِنَ الأَشْيَاءِ: اللَّاذِعُ المَذَاقِ.

"وَكَانَ يُكَنَّى أَبَا حَمْزَةَ، وَكَانَتِ البَقْلَةُ حَمِيزَةً حَادَّةَ المَذَاقِ"
كُراع النَّمْل الهُنائيّ الأزديّ
المُنتَخَب من غريب كلام العَرَب: كُراع النَّمْل الهُنائيّ الأزديّ (ت، 309هـ)، تحقيق: محمّد بن أحمد العمري، جامعة أم القرى، مكة المكرمة، ط1، (1409هـ/ 1989م)، ‏2/‏758‏.
ن456هـ=1064م

الحَمِيزُ مِنَ الأَدْوِيَةِ وَنَحْوِهَا: السَّرِيعُ الأَثَرِ.

"وَيُقَالُ لِلْعِلَاجِ السَّرِيعِ الأَثَرِ: عِلَاجٌ حَمِيزٌ "
عبد الله بن محمّد الأزديّ الصُّحاريّ
الماء: عبد الله بن محمّد الأزديّ الصُّحاريّ (ت، 456هـ)، حققه: هادي حسن حمّودي، وزارة التراث والثقافة، مسقط، ط2، (1436هـ/ 2015م)، ‏1/‏487‏.

حَمِيز[اسْم]

ن291هـ=904م

الحَمِيزُ : اللَّبَنُ الحَامِضُ.

وَجِئْنَا بِشِيزَى مِنْ حَمِيزٍ نَبِيلَةٍ تُدَاوِي دَخِيلَ الجُوعِ مِنْ كُلِّ سَاغِبِ
من إنشاد السِّدْريّ
مجالس ثعلب: أبو العبّاس ثعلب (ت، 291هـ)، شرح وتحقيق: عبد السلام هارون، دار المعارف، القاهرة، ط2، (1948م)، ‏70‏.

حَمْزَة[اسْم]

ن255هـ=869م

حَمْزَةُ الشَّيْءِ: لَذْعَتُهُ وَحَرَارَتُهُ.

رجل من الأعراب

الحَمْزَةُ : الأَسَدُ.

"وَالأَسَدُ يُقَالُ لَهُ: الضَّرْضَمُ... وَالضِّرْغَامُ... وَالهِزَبْرُ... وَحَمْزَةُ "

حَمْزِيَّة[اسْم]

320هـ=932م

الحَمْزِيَّةُ : فِرْقَةٌ مِنَ الخَوَارِجِ، تَنْتَسِبُ إِلَى حَمْزَةَ بْنِ أَكْرَكَ. [الأديان والمذاهب والفرق]

"وَالفِرْقَةُ الرَّابِعَةُ مِنْهُمُ: الحَمْزِيَّةُ أَصْحَابُ رَجُلٍ يُدْعَى حَمْزَةَ"
أبو الحسن الأشعريّ
مقالات الإسلاميّين واختلاف المصلّين: أبو الحسن الأشعريّ (ت، 324هـ)، قدم له وكتب حواشيه: نعيم زَرْزور، المكتبة العصرية، صيدا- بيروت، (1430هـ/ 2009م)، ‏1/‏177‏.

حُمُوزَة[صِفَة]

ن844هـ=1440م

الحُمُوزَةُ فِي الطَّعَامِ وَنَحْوِهِ: الحُمُوضَةُ واللَّذَاعَةُ وَالحَرَارَةُ.

"لُقِّبَ بِذَلِكَ لِبَقْلَةٍ فِيهَا ‌ حُمُوزَةٌ كَانَ يَأْكُلُهَا"
شهاب الدّين ابن رسلان
شرح سنن أبي داود: شهاب الدّين ابن رسلان (ت، 844ه)، تحقيق: خالد الرباط وآخرين، دار الفلاح، الفيوم، ط1، 1437ه/ 2016م، ‏13/‏288‏.
