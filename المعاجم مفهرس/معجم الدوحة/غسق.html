غَسَّاق[صِيغَةُ مُبَالَغَة]

قبل 50ق.هـ=573م

الغَسَّاقُ : الشَّدِيدُ الظُّلْمَةِ.

لَكِنَّمَا عِوَلِي، إِنْ كُنْتُ ذَا عِوَلٍ، عَلَى بَصِيرٍ بِكَسْبِ الحَمْدِ سَبَّاقِ
عَارِي الظَّنَابِيبِ، مُمْتَدٍّ نَوَاشِرُهُ مِدْلَاجِ، أَدْهَمَ، وَاهِي المَاءِ، غَسَّاقِ
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏136‏.

غَسَّاق[اسْم]

3ق.هـ=619م

الغَسَّاقُ : مَا يَسِيلُ مِنَ الجُرُوحِ مِنْ صَدِيدٍ وَنَحْوِهِ.

قرآن كريم
قرآن كريم، ‏[ صٓ : 57 ].

غَاسِق[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الغَاسِقُ : المُظْلِمُ.

"وَاللُّوحِ الخَافِق، وَاللَّيْلِ الغَاسِق ، وَالصَّبَاحِ الشَّارِق... إِنَّ شَجَرَ الوَادِي لَيَأْدُو خَتْلَا، وَيَحْرِقُ أَنْيَابًا عُصْلَا"
زبراء الكاهنة
أسجاع الكهَّان الجاهليِّين وأشعارهم جمع وتحقيق ودراسة: إعداد: ياسين عبد الله جمول، رسالة ماجستير، جامعة دمشق، (2011- 2012م)، ‏403‏.
17هـ=638م

الغَاسِقُ مِنَ الأَعْيُنِ: السَّائِلُ بِالدَّمْعِ وَنَحْوِهِ.

"وَإِنَّ إِخْوَانَنَا مِنْ أَهْلِ الكُوفَةِ نَزَلُوا فِي مِثْلِ حَدَقَةِ البَعِيرِ الغَاسِقَةِ ، مِنَ العُيُونِ العِذَاب، وَالجِنَانِ الخِصَاب"
الأحنف بن قيس التَّميميّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏4/‏75‏.
ن225هـ=840م

الغَاسِقُ : الَّذِي تَأَذَّى بِشَيْءٍ مِنْ شُرُورِ الغَسَقِ.

"إِنَّ هَذِهِ الأَنْجَاسَ يُنَفَّرُ عَنْهَا كُلُّ الخَوَافِي، لَيْسَ نَافِثَ الغَسَقِ، لِأَنَّ الغَاسِقَ لَا يَنْفَعُهُ ذَلِكَ"
زيد بن كَثْوة العَنْبَريّ
الدّرّة الفاخرة في الأمثال السّائرة: حمزة بن الحسن الأصفهانيّ (ت، 360هـ)، حققه وقدم له ووضع حواشيه وفهارسه: عبد المجيد قطامش، دار المعارف، القاهرة، (1972م)، ‏564‏.
440هـ=1048م

الغَاسِقُ : الَّذِي يَأْتِي فِي الغَسَقِ.

"وَكَيْفَ ظَنُّهُ بِالشَّوْقِ وَنَارِه، وَالقَلْبِ وَأُوَارِه، وَقَدْ قَرُبَ المَزَار، وَتَدَانَتِ الدِّيَار؟ وَبَيْنَنَا مَسِيرَةُ غَاسِقٍ لِعَاشِقٍ"
شرف السّادة أبو الحسن البَلْخيّ
دُمية القصر وعُصرة أهل العصر: عليّ بن الحسن الباخَرْزيّ (ت، 467هـ)، تحقيق ودراسة: محمّد ألتونجي، دار الجيل، بيروت، ط1، (1414هـ/ 1993م)، ‏2/‏723‏.

غَاسِق[اسْم]

11هـ=632م

الغَاسِقُ : اللَّيْلُ إِذَا دَخَلَ وَأَظْلَمَ.

قرآن كريم
قرآن كريم، ‏[ الفَلَق : 3 ].
ن50هـ=670م

غَاسِقُ اللَّيْلِ: ظَلَامُهُ.

أَقُولُ لِفَتْلَاءِ المَرَافِقِ سَمْحَةٍ وَلِلَّيْلِ كِسْرٌ يَضْبَعُ البِيدَ غَاسِقُهْ
تَضَمَّنْتِ هَمِّي فَاسْتَقِيمِي وَشَمِّرِي عَلَى لَاحِبٍ تُنْضِي المَطِيَّ أَسَالِقُهْ
الحارث بن جَحدر الحضرميّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏8/‏342‏-‏343‏.

غَسَق[اسْم]

قبل 50ق.هـ=573م

الغَسَقُ : أَوَّلُ الظَّلَامِ إِذَا دَخَلَ اللَّيْلُ.

"وَالشَّفَقِ وَالغَسَق ، وَالفَلَقِ إِذَا اتَّسَق، إِنَّ مَا أَنْبَأْتُكَ بِهِ لَحَقّ"
سَطيح الذِّئْبيّ
أسجاع الكهَّان الجاهليِّين وأشعارهم جمع وتحقيق ودراسة: إعداد: ياسين عبد الله جمول، رسالة ماجستير، جامعة دمشق، (2011- 2012م)، ‏279‏.
ن225هـ=840م

الغَسَقُ : اسْمٌ لِلشُّرُورِ الَّتِي تَأْتِي مَعَ دُخُولِ الظَّلَامِ.

"إِنَّ هَذِهِ الأَنْجَاسَ يُنَفَّرُ عَنْهَا كُلُّ الخَوَافِي، لَيْسَ نَافِثَ الغَسَقِ ، لِأَنَّ الغَاسِقَ لَا يَنْفَعُهُ ذَلِكَ"
زيد بن كَثْوة العَنْبَريّ
الدّرّة الفاخرة في الأمثال السّائرة: حمزة بن الحسن الأصفهانيّ (ت، 360هـ)، حققه وقدم له ووضع حواشيه وفهارسه: عبد المجيد قطامش، دار المعارف، القاهرة، (1972م)، ‏564‏.

غَسَق[مَصْدَر]

ن85هـ=704م

غَسَقُ الدَّمْعِ: سَيَلَانُهُ.

شَأَتْكَ عَيْنٌ دُمُوعُهَا غَسَقُ فِي إِثْرِ حَيٍّ سُلَّافُهُمْ فِرَقُ
عُبيد الله بن قيس الرُّقَيّات
ديوان عُبيد الله بن قيس الرُّقَيّات: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، د.ت. ‏71‏.

غَسَاق[صِفَةٌ مُشَبَّهَة]

3ق.هـ=619م

الغَسَاقُ : البَارِدُ الشَّدِيدُ البَرْدِ.

قرآن كريم
قرآن كريم، ‏555‏.

تَغْسَاق[مَصْدَر]

ن22هـ=643م

تَغْسَاقُ العَيْنِ: سَيَلَانُهَا بِالدَّمْعِ.

حَتَّى إِذَا طَالَ الوُقُوفُ بِدِمْنَةٍ خَرْسَاءَ حَلَّ بِهَا الرَّبِيعُ نِطَاقَا
عُجْتُ القَلُوصَ بِهَا أُسَائِلُ آيَهَا وَالعَيْنُ تُذْرِي عَبْرَةً تَغْسَاقَا
الشَّمّاخ بن ضرار الذُّبْيانيّ
ديوان الشَّمّاخ بن ضرار الصحابي الغطفاني رضي الله عنه: شرح: أحمد بن الأمين الشنقيطي، مطبعة السعادة، مصر، (1327هـ)، ‏73‏.

غَسَقَ[لَازِم]

ن23هـ=644م

غَسَقَ اللَّيْلُ: أَظْلَمَ.

"فَإِنْ غُمَّ عَلَيْكُمْ فَعُدُّوا شَعْبَانَ ثَلَاثِينَ، ثُمَّ لَا تُفْطِرُوا حَتَّى يَغْسِقَ اللَّيْلُ عَلَى الظِّرَابِ"
عمر بن الخطّاب
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏8/‏431،،‏(رقم الحديث: 8031).
ن175هـ=791م

غَسَقَتِ العَيْنُ: سَالَ دَمْعُهَا.

فَالعَيْنُ مَطْرُوفَةٌ لِبَيْنِهِمُ تَغْسِقُ مَا فِي دُمُوعِهَا سَرَعُ
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏353‏.

غَسِق[صِيغَةُ مُبَالَغَة]

ن50هـ=670م

الغَسِقُ مِنَ الدَّمْعِ وَنَحْوِهِ: الكَثِيرُ السَّيَلَانِ.

فَقَلْبُهُ فَرِقٌ، وَمَأْقُهُ شَرِقٌ وَدَمْعُهُ غَسِقٌ ، مِنْ شِدَّةِ الكَمَدِ
عمرو بن مسعود السُّلَميّ
مكارم الأخلاق: ابن أبي الدُّنيا (ت، 281هـ)، تحقيق وتعليق: مجدي السيد إبراهيم، مكتبة القرآن، القاهرة، (1990م)، ‏148‏.

مُغْسِق[اسْمُ فَاعِل]

ن58هـ=678م

المُغْسِقُ : الَّذِي يَأْتِي فِي الغَسَقِ.

"كَانَ لَهُ غَنَمٌ، فَأَمَرَ عَامِرَ بْنَ فُهَيْرَةَ أَنْ يُعَزِّبَ بِهَا، فَكَانَ يُرَوِّحُ عَلَيْهِمَا مُغْسِقًا وَهُمَا فِي الغَارِ"
عائشة بنت أبي بكر الصِّدّيق
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏2/‏19‏-‏20‏.

مُغْسِق[صِفَةٌ مُشَبَّهَة]

ن366هـ=976م

المُغْسِقُ مِنَ اللَّيَالِي: المُظْلِمُ.

يَخْتَالُ بَيْنَ مُخَنَّثٍ وَمُوَاجِرٍ مُسْتَرْزِقِ
فَكَأَنَّ مَنْ يَغْشَاهُمَا فِي جُنْحِ لَيْلٍ مُغْسِقِ
مِنْ ذَاكِرٍ أَضْيَافَ جَفْـ نَةَ فِي الزَّمَانِ الأَسْبَقِ
عليّ بن الحسن اللَّحّام الحَرّانيّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏4/‏119‏.

أَغْسَقَ[لَازِم]

ن63هـ=683م

أَغْسَقَ المُؤَذِّنُ وَنَحْوُهُ: أَخَّرَ الأَذَانَ إِلَى وَقْتِ دُخُولِ الظَّلَامِ.

" أَغْسِقْ أَغْسِقْ"
الرَّبيع بن خُثيم الثَّوريّ التَّميميّ
كتاب الصّلاة: ابن دُكين (ت، 219هـ)، تحقيق: صلاح بن عايض الشلاحي، مكتبة الغرباء الأثرية، المدينة المنورة، ط1، (1417هـ/ 1996م)، ‏212‏.
ن428هـ=1037م

أَغْسَقَ اللَّيْلُ: أَظْلَمَ.

أَلَيْلَ سَوَادِي، مَا أَرَى الصُّبْحَ سَرَّنِي فَمَنْ رَدَّ لِي ذَاكَ الظَّلَامَ فَأَغْسَقَا ؟
مهيار الدّيلميّ
ديوان مِهْيار الدَّيْلميّ: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1344هـ/ 1925م- 1350هـ/ 1931م)، ‏2/‏307‏.

غَسِيق[صِيغَةُ مُبَالَغَة]

ن80هـ=699م

الغَسِيقُ : الشَّدِيدُ الحُمْرَةِ.

هِجَانٌ، فَلَا فِي اللَّوْنِ شَامٌ يَشِينُهُ وَلَا مَهَقٌ يَغْشَى الغَسِيقَاتِ ، مُغْرَبُ
أبو صخر الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏937‏.

إِغْسَاق[مَصْدَر]

ن81هـ=700م

الإِغْسَاقُ : التَّأَخُّرُ إِلَى وَقْتِ الغَسَقِ.

"فَعَنْ مَنْ أَخَذَ ابْنُ الزُّبَيْرِ الإِغْسَاقَ ؟"
ابن الحَنَفيّة
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏7/‏105،،‏(رقم الحديث: 7932).
ن360هـ=971م

الإِغْسَاقُ : الإِظْلَامُ.

وَإِذَا مَا أَلَمَّ خَطْبٌ فَرَأْيِي مِنْهُ مِثْلُ الشِّهَابِ فِي الإِغْسَاقِ
كُشاجم
ديوان كشاجم محمود بن الحسين (ت، 360): دراسة وشرح وتحقيق: النبوي عبد الواحد شَعلان، مكتبة الخانجي- مطبعة المدني، القاهرة، ط1، (1417هـ/ 1997م)، ‏279‏.
