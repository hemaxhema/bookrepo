زَئِير[مَصْدَر]

قبل 150ق.هـ=476م

الزَّئِيرُ : مَا يُصْدِرُهُ الأَسَدُ مِنْ صَوْتٍ يُرَدِّدُهُ فِي صَدْرِهِ، وَيَمُدُّهُ تَغَضُّبًا.

"أَحْمَرُ ضِرْغَامَةٌ، شَدِيدُ الزَّئِير ، قَلِيلُ الهَدِير"
منسوب إلى هند بنت الخُسّ الإياديّة
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏6/‏52‏.

زَائِر[اسْمُ فَاعِل]

قبل 100ق.هـ=525م

الزَّائِرُ : المُرَدِّدُ صَوْتَ الزَّئِيرِ.

بِمَقَالَةٍ مِنْ حَازِمٍ ذِي مِرَّةٍ يَدَأُ العَدُوَّ زَئِيرُهُ لِلزَّائِرِ
ثعلبة بن صُعير المازنيّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏383‏.

زَأَرَ[لَازِم]

قبل 13ق.هـ=609م

زَأَرَ الأَسَدُ: رَدَّدَ صَوْتَهُ فِي صَدْرِهِ، وَمَدَّهُ تَغَضُّبًا.

ضِعَافُ الأُسْدِ أَكْثَرُهَا زَئِيرًا وَأَصْرَمُهَا اللَّوَاتِي لَا تَزِيرُ
مُعَوِّد الحكماء
أشعار العامريين الجاهليين: جمعها ووثقها وقدم لها: عبد الكريم إبراهيم يعقوب، دار الحوار، اللاذقية، ط1، (1982م)، ‏57‏.

زَأَرَ[مُتَعَدٍّ]

ن42هـ=662م

زَأَرَهُ : صَاحَ عَلَيْهِ كَزَئِيرِ الأَسَدِ.

"فَأَرَدْتُ الإِسْلَامَ، فَإِذَا قَوْمِي يَزْئِرُونِي زَئِيرًا شَدِيدًا"
عثمان بن طلحة بن أبي طلحة العَبْدريّ الحَجَبيّ
المنتظم في تاريخ الملوك والأمم: ابن الجوزيّ (ت، 597هـ)، دراسة وتحقيق: محمد عبد القادر عطا، مصطفى عبد القادر عطا، راجعه وصححه: نعيم زرزور، دار الكتب العلمية، بيروت، ط2، (1415ه/ 1995م)، ‏5/‏189‏.

زَأْر[مَصْدَر]

قبل 13ق.هـ=609م

الزَّأْرُ : مَا يُصْدِرُهُ الأَسَدُ مِنْ صَوْتٍ يُرَدِّدُهُ فِي صَدْرِهِ، وَيَمُدُّهُ تَغَضُّبًا.

أُنْبِئْتُ أَنَّ أَبَا قَابُوسَ أَوْعَدَنِي ولَا قَرَارَ عَلَى زَأْرٍ مِنَ الأَسَدِ
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏87‏.

زَار[اسْمُ جِنْس]

قبل 13ق.هـ=609م

الزَّارُ : الغَابَةُ يَعْتَادُهَا الأُسْدُ.

يَكُبُّ السَّفِينَ لِأَذْقَانِهِ وَيَصْرَعُ بِالعِبْرِ أَثْلًا وَزَارَا
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏1/‏198‏.

تَزَاءَرَ[لَازِم]

ن10هـ=631م

تَزَاءَرَتِ الأُسُودُ: تَصَايَحَتْ بِالزَّئِيرِ.

لَنَا زَارَةٌ، ضَبِسٌ نَابُهَا يَهُونُ عَلَى حَامِيَيْهَا الوَعِيدُ
بِهَا قُضُبٌ هِنْدُوَانِيَّةٌ وَغَابٌ تَزَاءَرُ فِيهِ الأُسُودُ
الأخرم السِّنْبِسيّ الطّائيّ
رسالة الصّاهل والشّاحج: أبو العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: عائشة عبد الرحمن بنت الشاطئ، دار المعارف، القاهرة، ط2، (1404هـ/ 1984م)، ‏529‏.

تَزَاؤُر[مَصْدَر]

ن105هـ=723م

التَّزَاؤُرُ بَيْنَ الأُسُودِ: زَئِيرُ بَعْضِهِمْ عَلَى بَعْضٍ.

تُقَلِّعُ عُمْرِيَّ العِضَاهِ كأَنَّهَا بِأَجْوَازِهِ أُسْدٌ لَهُنَّ تَزَاؤُرُ
كُثَيِّر عَزّة
ديوان كُثَيِّر عَزَّة: جمعه وشرحه: إحسان عبّاس، دار الثقافة، بيروت، (1391هـ/ 1971م)، ‏375‏.

زَأْرَة[اسْمُ مَرَّة]

ن110هـ=728م

الزَّأْرَةُ : المَرَّةُ مِنَ الزَّأْرِ.

إِذَا سَمِعَتْ مِنِّي حَوِيزَةُ زَأْرَةً تَحَوَّزَ دَاءٌ فِي حَوَايَاهُمُ الأُدْرِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏595‏.

زَأْرَة[اسْم]

178هـ=794م

الزَّأْرَةُ مِنَ النَّاسِ: الجَمَاعَةُ.

"وَيُقَالُ: مَعَهُ زَأْرَةٌ مِنَ النَّاسِ"
أبو زيد الأنصاريّ
كتاب النّوادر في اللّغة: أبو زيد الأنصاريّ (ت، 215هـ)، تحقيق ودراسة: محمّد عبد القادر أحمد عطا، دار الشروق، بيروت- القاهرة، ط1، (1401هـ/ 1981م)، ‏589‏.
216هـ=831م

الزَّأْرَةُ : البُسْتَانُ.

زَأْرَةُ جَبَّارٍ مِنَ النَّخْلِ بَسَقْ
من إنشاد الأصمعيّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: محمد باسل عيون السود، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏1/‏406‏.

زَءَّارَة[صِيغَةُ مُبَالَغَة]

127هـ=745م

الزَّءَّارَةُ : الكَثِيرُ الزَّئِيرِ.

بَلْ إِنْ رَآنِي هَادِرٌ لَمْ يَهْدِرِ أَوْ أَسَدٌ زَءَّارَةٌ لَمْ يَزْأَرِ يَحْذَرُ مِنْ نَطَّاحَةٍ مُكَسِّرِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏297‏.

زَأَّار[صِيغَةُ مُبَالَغَة]

132هـ=749م

الزَّأَّارُ : الكَثِيرُ الزَّئِيرِ.

يَا حَكَمُ الوَارِثُ مِنْ عَبْدِ المَلِكْ مِيرَاثَ أَحْسَابٍ وَجُودٍ مُنْسَفِكْ فِي القَدَمِ العَادِيِّ وَالعِزِّ العَرِكْ مِنْ كُلِّ زَأَّارٍ وَمَخَّاضٍ عَلِكْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏38‏.

زَئِر[صِيغَةُ مُبَالَغَة]

170هـ=786م

الزَّئِرُ : الكَثِيرُ الزَّئِيرِ.

مَا مُخْدِرٌ خَدِرٌ، مُسْتَأْسِدٌ أَسَدٌ ضُبَارِمٌ خَادِرٌ، ذُو صَوْلَةٍ زَئِرُ
بِبَالِغٍ عُشْرَ عُشْرٍ مِنْ شَجَاعَتِهِ إِذَا تَنَازَلَتِ الأَبْطَالُ، وَاشْتَجَرُوا
أبو الخطّاب البَهْدليّ
طبقات الشّعراء: ابن المعتزّ (ت، 296هـ)، تحقيق: عبد الستار أحمد فرّاج، دار المعارف، القاهرة، د.ت. ‏133‏.

مُزَئِّر[اسْمُ فَاعِل]

630هـ=1232م

المُزَئِّرُ : المُرَدِّدُ صَوْتَ الزَّئِيرِ.

يَا رَاكِعًا، لَبِسَ السَّوَادَ، وَسَاجِدًا يَتْلُو بَنِي العَبَّاسِ، وَهْوَ مُزَئِّرُ
مجد الدين الروذراوريّ
مطالع البدور في منازل السَّرور: علي بن عبد الله الغزولي (ت، 815هـ)، مطبعة إدارة الوطن، القاهرة، ط1، (1299ه/ 1882م)، ‏2/‏121‏.

زَؤُور[صِيغَةُ مُبَالَغَة]

ن665هـ=1266م

الزَّؤُورُ : الكَثِيرُ الزَّئِيرِ.

وَاحْذَرِي مِنَ العُتُوِّ وَالقَهْرِ وَالكِبْـ رِ انْتِهَارًا مِنَ الهِزَبْرِ الزَّؤُورِ
ابن عَلْوان
الفتوح (كتاب وديوان): ابن عَلْوان (ت، 665هـ)، تحقيق: عبد العزيز سُلطان طاهِر المنصوب، دار الفكر المعاصر، بيروت، ط2، (1416ه/ 1995م)، ‏206‏.

تَزَاَّرَ[لَازِم]

ن676هـ=1277م

تَزَأَّرَ الأَسَدُ: رَدَّدَ صَوْتَهُ فِي صَدْرِهِ، وَمَدَّهُ تَغَضُّبًا.

رَبْعٌ يُغِيثُ المُعْتَفِينَ رِيَاضُهُ وَحِمًى يُجِيرُ الخَائِفِينَ حِبَالُهُ
مِثْلُ العَرِينِ تَزَأَّرَتْ آسَادُهُ وَتَمَنَّعَتْ فِي غَابِهَا أَشْبَالُهُ
السّتاليّ الخروصيّ
ديوان السّتاليّ: السّتاليّ الخروصيّ (ت، 676هـ)، تحقيق: عزّ الدِّين التنوخيّ، وزارة التُّراث والثَّقافة، مسقط، ط2، (1426ه/ 2005م)، ‏365‏.

تَزْآر[مَصْدَر]

ن915هـ=1509م

التَّزْآرُ : الزَّئِيرُ.

وَلَوْ تَسْمَعُ الأُسْدُ تَزْآرَنَا غَدَاةَ الوَقِيعَةِ، لَمْ تَنْأَمِ
سليمان بن سليمان النّبهاني
ديوان النّبهاني: سليمان بن سليمان النّبهاني (ت، 915هـ)، تحقيق: عز الدّين التّنوخيّ، وزارة التّراث والثَّقَافَة، مسقط، ط2، (1426ه/ 2005م)، ‏242‏.

مَزْأَر[اسْمُ مَكَان]

1010هـ=1601م

المَزْأَرُ : مَكَانُ الزَّئِيرِ.

جَيْشٌ طَلَائِعُهُ الأَوَابِدُ، إِنْ تُصِخْ لِوَجِيبِهِ مِنْ قَيْدِ شَهْرٍ تَنْفُرِ
يَقْتَادُهُ المَلِكُ المُشِيحُ، كَأَنَّهُ بَيْنَ العَوَالِي ضَيْغَمٌ فِي مَزْأَرِ
نصير الدين الحسين بن إبراهيم الحسيني
سلافة العصر في محاسن أهل العصر: ابن مَعْصُوم المَدَنِيّ (ت، 1120هـ)، تحقيق: محمود خلف البادي، دار كنان، دمشق، ط1، (1430ه/ 2009م)، ‏1/‏155‏.

مُزْئِر[اسْمُ فَاعِل]

1073هـ=1662م

المُزْئِرُ : المُرَدِّدُ صَوْتَ الزَّئِيرِ.

وَالرَّايَةُ الحَمْرَاءُ يَخْفِقُ ظِلُّهَا يَهْفُو عَلَيْهَا كُلُّ لَيْثٍ مُزْئِرِ
ابن مَعْصُوم المَدَنِيّ
رحلة ابن معصوم المدني أو سلوة الغريب وأسوة الأريب: ابن مَعْصُوم المَدَنِيّ (ت، 1120هـ)، تحقيق: شاكر هادي شكر، الدّار العربية للموسوعات، الحازمية، بيروت، ط1، (1426ه/ 2006م)، ‏291‏.

زَأَّرَ[مُتَعَدٍّ]

ن1291هـ=1873م

زَأَّرَهُ : جَعَلَهُ يَزْأَرُ.

نَاهِيكَهُ، لَيْثُ حَرْبٍ لَا يُنَاضِلُهُ لَيْثٌ يُزَئِّرُهُ فَخْرًا تَرَجُّزُهُ
ابن رزيق حميد بن محمد بن رزيق النخلي
ديوان فرند الجمان (دراسة وتحقيق): ابن رزيق حميد بن محمد بن رزيق النخلي (ت، 1291هـ)، إعداد: ليلى بنت أنور اللواتي، رسالة ماجستير، جامعة السلطان قابوس، مسقط، (2009م)، ‏159‏.

مُتَزَائِر[اسْمُ فَاعِل]

1354هـ=1934م

المُتَزَائِرُ مِنَ الأَمْوَاجِ وَنَحْوِهَا: الَّذِي يَتَرَدَّدُ كَصَوْتِ الأَسَدِ فِي هَيَجَانِهِ وَغَضَبِهِ.

أَلَا مَا لِهَذَا البَحْرِ غَضْبَانَ، مِثْلَمَا تَنَفَّسُ فِيهِ الرِّيحُ عَنْ صَدْرِ ثَائِرِ؟
لَقَدْ غَمَرَ الأَكْوَاخَ فَوْقَ صُخُورِهِ وَلَجَّ بِهَا فِي مَوْجِهِ المُتَزَائِرِ
علي محمود طه
ديوان علي محمود طه: علي محمود طه، مؤسسة هنداوي للتعليم والثقافة، القاهرة، (2012م)، ‏53‏.

زَئِيرِيّ[اسْمٌ مَنْسُوب]

1392هـ=1971م

الزَّئِيرِيُّ : المَنْسُوبُ إِلَى الزَّئِيرِ.

"وَإِنَّ كُلَّ صَوْتٍ يَدُلُّ دَلَالَةً بِعَيْنِهَا، فَمَثَلًا (عُو) يَدُلُّ عَلَى الحَيَوَانَاتِ الزَّئِيرِيَّةِ ، وَ(وَا) يَدُلُّ عَلَى الصَّوْتِ المُضَاعَفِ وَالمُتَكَرِّرِ بِحَرَكَةِ الفَكَّينِ"
محمد يوسف نور الدين
التطور اللغوي ونشوء العربية: ‏محمد يوسف نور الدين،‏مجلة اللّسان العربيّ: الرباط، المغرب، مج 8، ج1، 1971م، ‏128‏.
