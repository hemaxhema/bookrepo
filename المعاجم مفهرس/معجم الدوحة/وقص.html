أُوَيْقِص[اسْمٌ مُصَغَّر]

قبل 150ق.هـ=476م

الأُوَيْقِصُ : تَصْغِيرُ الأَوْقَصِ، وَهُوَ القَصِيرُ العُنُقِ.

" الأُوَيْقِصُ ، القَصِيرُ العَضُدِ، العَظِيمُ الحَاوِيَةِ"
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏2/‏286‏.

أَوْقَص[صِفَةٌ مُشَبَّهَة]

قبل 150ق.هـ=476م

الأَوْقَصُ مِنَ الأَشْخَاصِ: القَصِيرُ العُنُقِ.

"أَبْغَضُ المُتَجَرِّفَةَ الشَّوْهَاء، المَنْفُوحَةَ الكَبْدَاء، العِنْفِصَ الوَقْصَاء ، الحَمِشَةَ الزَّلَّاء"
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏61‏.
ن272هـ=885م

الأَوْقَصُ مِنَ الأَشْيَاءِ: القَصِيرُ.

"كُلُّ قَلَمٍ تُطِيلُ جِلْفَتَهُ، فَإِنَّ الخَطَّ يَخْرُجُ بِهِ أَوْقَصَ "
سليمان بن وهب الحارثيّ
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏4/‏195‏.

أَوْقَص[اسْمُ تَفْضِيل]

ن385هـ=995م

أَوْقَصُ الطُّرُقِ وَنَحْوِهَا: أَقْصَرُهَا.

"خُذْ أَوْقَصَ الطَّرِيقَيْنِ"
من رواية الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏5/‏468‏.

وَقَّاص[صِيغَةُ مُبَالَغَة]

قبل 150ق.هـ=476م

الوَقَّاصُ : الشَّدِيدُ الوَقْصِ، وَهُوَ دَقُّ العُنُقِ وَكَسْرُهَا.

إِنِّي زَعِيمٌ لَكِ، أُمَّ عَمْرِو بِمَاجِدِ الجِدِّ، كَرِيمِ النَّجْرِ أَشْجَعَ مِنْ ذِي لِبَدٍ هِزَبْرِ وَقَّاصِ أَقْرَانٍ، شَدِيدِ الأَسْرِ
قائل في المنام
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏11/‏35 ‏-‏ 36‏.

وَقَّاص[اسْم]

ن123هـ=741م

الوَقَّاصُ : شَبَكَةٌ لِاصْطِيَادِ الطَّيْرِ.

"كُنْتُ أَصْطَادُ فِي حَرَمِ المَدِينَةِ بِالوَقَاقِيصِ ، وَهِيَ شِبَاكُ الطَّيْرِ، فَاصْطَدْت نُهَسًا، فَأَخَذَهُ زَيْدُ بْنُ ثَابِتٍ، وَصَكَّ فِي قَفَايَ، ثُمّ أَرْسَلَهُ"
شُرَحْبيل بن سعد
الروض الأنف في شرح السيرة النبوية لابن هشام: أبو القاسم السُّهَيْليّ (ت، 581هـ)، تحقيق وتعليق وشرح: عبد الرحمن الوكيل، مكتبة ابن تيمية، القاهرة، مكتبة العلم، جدة، دار النصر للطباعة، القاهرة، (1387ه/ 1967م- 1410ه/ 1990م)، ‏5/‏49‏.

تَوَقَّصَ[لَازِم]

قبل 50ق.هـ=573م

تَوَقَّصَ الشَّخْصُ فِي سَيْرِهِ: شَدَّدَ وَطْأَهُ عَلَى الأَرْضِ، كَأَنَّهُ يَكْسِرُ مَا تَحْتَهُ.

وَمَنْ رَامَ ظُلْمِي مِنْهُمُ فَكَأَنَّمَا تَوَقَّصَ حِينًا مِنْ شَوَاهِقِ صِنْدِدِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏55‏.
ن571هـ=1175م

تَوَقَّصَ الشَّخْصُ: انْدَقَّتْ عُنُقُهُ وَانْكَسَرَتْ.

"فَانْطَلَقَ بِهِمْ أَبُو الجُعَيْدِ عَلَى مَدَقَّةِ الطَّرِيقِ وَجِسْرِ اليَرْمُوكِ حَتَّى وَاقَعَ عَسْكَرَهُمْ، فَقَاتَلُوهُمْ مَلِيًّا. فَلَمَّا أَنْشَبَ القِتَالُ انْحَازَ بِهِمْ فِي ظُلْمَةِ اللَّيْلِ عَلَى الطَّرِيقِ الَّتِي أَقْبَلَ عَلَيْهَا وَالجِسْرِ، وَتَنَادَتِ الرُّومُ: إِنَّ العَرَبَ قَدِ انْهَزَمَتْ، فَخَرَجَتْ تُرَاكِضُ تَؤُمُّ النِّيرَانَ؛ فَتَوَقَّصَ مِنْهُمْ، فِي وَادِي اليَرْمُوكِ، أَكْثَرُ مِنْ ثَمَانِينَ أَلْفًا، لَا يَعْلَمُ الآخِرُ مَا لَقِيَ الأَوَّلُ"
ابن عساكر
تاريخ مدينة دمشق وذكر فضلها وتسمية من حلها من الأماثل أو اجتاز بنواحيها من وارديها وأهلها: ابن عساكر (ت، 571هـ)، دراسة وتحقيق: عمر بن غرامة العمروي، دار الفكر، بيروت، ط1، (1415ه/ 1995م)، ‏66/‏117 ‏-‏ 118‏.

تَوَقَّصَ[مُتَعَدٍّ بِالحَرْف]

ن74هـ=693م

تَوَقَّصَ بِهِ الفَرَسُ: وَثَبَ بِهِ فِي عَدْوِهِ، وَهُوَ يُقَارِبُ الخَطْوَ.

"ثُمَّ أُتِيَ بِفَرَسٍ مَعْرُورٍ، فَعَقَلَهُ رَجُلٌ، فَرَكِبَهُ، فَجَعَلَ يَتَوَقَّصُ بِهِ"
جابر بن سَمُرة السُّوائيّ العامريّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏34/‏424،،‏(رقم الحديث: 20834).

وَقِيص[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

الوَقِيصُ مِنَ الجِحَاشِ وَنَحْوِهَا: الَّذِي دُقَّ عُنُقُهُ وَكُسِرَ.

فَأَصْدَرَهَا تَعْلُو النِّجَادَ عَشِيَّةً أَقَبُّ كَمِقْلَاءِ الوَلِيدِ، خَمِيصُ
فَجَحْشٌ، عَلَى أَدْبَارِهِنَّ، مُخَلَّفٌ وَجَحْشٌ، لَدَى مَكْرُوهِهِنَّ، وَقِيصُ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس وملحقاته: شرح: أبي سعيد السُّكَّريّ (ت، 275هـ): دراسة وتحقيق: أنور عليان أبو سويلم، محمّد علي الشوابكة، مركز زايد للتراث والتاريخ، العين، ط1، (1421هـ/ 2000م)، ‏2/‏618‏.
ن406هـ=1015م

الوَقِيصُ بِالشَّيْءِ: المُثْقَلُ بِهِ.

خَفَّ عَنْ عَاتِقِي الرَّجَاءُ، وَكَمْ بَا تَ بِمَنِّ الرِّجَالِ، وَهْوَ وَقِيصُ
الشّريف الرّضيّ
ديوان الشريف الرضي: دار صادر- دار بيروت، بيروت، (1380هـ/ 1961م)، ‏1/‏573‏.

وَقَصَ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

وَقَصَتِ الدَّابَّةُ الذُّبَابَ: ضَرَبَتْهُ بِذَنَبِهَا أَوْ بِجَفْنِهَا، فَقَتَلَتْهُ.

وَمُغِيرَةٍ نَسْجَ الجَنُوبِ شَهِدْتُهَا تَمْضِي سَوَابِقُهَا عَلَى غُلَوَائِهَا
بِمُحَالَةٍ، تَقِصُ الذُّبَابَ بِطَرْفِهَا خُلِقَتْ مَعَاقِمُهَا عَلَى مُطَوَائِهَا
المُرَقِّش الأكبر البكريّ
المرَقِّش الأكبر أخباره وشعره: تحقيق: نوري حمّودي القيسيّ، مجلة العرب، الرياض، س4، ج6، 1389هـ، ج10، (1390هـ)، ‏2/‏882‏-‏ 883‏.
قبل 13ق.هـ=609م

وَقَصَهُ : دَقَّ عُنُقَهُ وَكَسَرَهَا.

مَتَى نَعْقِدْ قَرِينَتَنَا بِحَبْلٍ نَجُذِّ الحَبْلَ أَوْ نَقِصِ القَرِينَا
عمرو بن كلثوم التَّغْلِبيّ
ديوان عَمرو بن كلثوم: جمعه وحققه وشرحه: إميل بديع يعقوب، دار الكتاب العربي، بيروت، ط2، (1416هـ/ 1996م)، ‏81‏.

وَقَصَ[مُتَعَدٍّ بِالحَرْف]

ن93هـ=712م

وَقَصَتْ بِهِ الدَّابَّةُ وَنَحْوُهَا: أَلْقَتْ بِهِ؛ فَدَقَّتْ عُنُقَهُ، وَكَسَرَتْهَا.

"فَرَكِبَتِ البَحْرَ مَعَ بِنْتِ قَرَظَةَ، فَلَمَّا قَفَلَتْ رَكِبَتْ دَابَّتَهَا، فَوَقَصَتْ بِهَا، فَسَقَطَتْ عَنْهَا، فَمَاتَتْ"
أنس بن مالك الخزرجيّ الأنصاريّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏2/‏326‏.

قبل 13ق.هـ=609م

المَوْقُوصُ : الَّذِي دُقَّ عُنُقُهُ وَكُسِرَ.

إذَا مَا بَدَا، ظَلَّتْ لَهُ الأُسْدُ عُكَّفًا فَهُنَّ، حِذَارَ المَوْتِ مِنْهُ، رُبُوضُ
تُرَى بَيْنَ مَوْقُوصٍ تَغَطْمَطَ فِي الرَّدَى وَذِي رَغْبَةٍ يَرْجُو الحَيَاةَ، نَحِيضُ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏82‏.
ن328هـ=940م

المَوْقُوصُ مِنْ أَجْزَاءِ البَيْتِ الشِّعْرِيِّ: المَحْذُوفُ مِنْهُ ثَانِيهِ المُتَحَرِّكُ. [العَروض والقوافي]

وَإِنْ وَجَدْتَ الثَّانِيَ المَنْقُوصَا مُحَرَّكًا، سَمَّيْتَهُ المَوْقُوصَا
ابن عبد ربّه الأندلسيّ

ديوان الأخرس (السيد عبد الغفار بن عبد الواحد بن وهب الموصلي البغدادي البصري): عبد الغفار بن عبد الواحد بن وهب الموصلي (ت، 1291هـ)، حققه وعلق عليه: وليد الأعظمي، راجعه وضبطه وأضاف إلى حواشيه: عبد العزيز محمد جمعة، إبراهيم عبد الحميد الأسود، مؤسسة جائزة عبد العزيز بن سعود البابطين للإبداع الشعري، الكويت، ط2، (2008م)، ‏596‏.

مُتَوَاقِص[اسْمُ فَاعِل]

423هـ=1032م

المُتَوَاقِصُ : الَّذِي يَنْحَنِي وَيَتَقَاصَرُ عَلَى الشَّيْءِ؛ لِيُمْسِكَهُ بِعُنُقِهِ.

لَوْ كَانَ إِيقَاعُ الزَّمَانِ مُسَاعِدِي لَوَجَدْتَنِي فِي سُكْرِ عَيْشِي رَاقِصَا
الذَّنْبُ لِلْأَيَّامِ حِينَ تَرَكْنَنِي ظُلْمًا عَلَى جِيدِي لَهَا، مُتَوَاقِصَا
أبو القاسم الطبريّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏5/‏191‏.

مُوَقَّص[اسْمُ مَفْعُول]

ن428هـ=1037م

المُوَقَّصُ مِنَ الأَشْخَاصِ: الَّذِي دُقَّ عُنُقُهُ وَكُسِرَ.

مَتَى تَتَعَنَّسُ الدُّنْيَا عَجُوزًا مُوَقَّصَةً ، وَتَتْرُكُهُ صَبِيَّا؟
مهيار الدّيلميّ
ديوان مِهْيار الدَّيْلميّ: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1344هـ/ 1925م- 1350هـ/ 1931م)، ‏4/‏198‏.

مُوقَص[صِفَةٌ مُشَبَّهَة]

ن567هـ=1171م

المُوقَصُ مِنَ الأَشْيَاءِ: القَصِيرُ.

يُنَافِرُ إِيقَاعُهُ صَوْتَهُ فَهَذَا يَزِيدُ، وَذَا يَنْقُصُ
وَيَتْبَعُهُ زَامِرٌ مِثْلُهُ بَلِيغٌ، لَهُ نَفَسٌ مُوقَصُ
ابن قَلاقس
الزّهر الباسم والعرف النّاسم في مديح الأجل أبي القاسم: ابن قَلاقس (ت، 567هـ)، تحقيق: عبد العزيز بن ناصر المانع، مطابع جامعة الملك سعود، الرياض، ط1، (1404ه/ 1984م)، ‏19‏.

مُتَوَقِّص[اسْمُ فَاعِل]

ن711هـ=1311م

المُتَوَقِّصُ فِي المَشْيِ: المُشَدِّدُ وَطْأَهُ عَلَى الأَرْضِ، كَأَنَّهُ يَكْسِرُ مَا تَحْتَهُ.

"وَهُوَ مَشْيُ الرَّجُلِ مُتَوَقِّصًا فِي جَانِبَيْهِ، وَمَشْيُ الفَرَسِ النَّشِيطِ"
ابن منظور
لسان العرب: ابن منظور (ت، 711هـ)، دار صادر، بيروت، ط1، (د.ت)، ‏5/‏429‏.

تَوْقِيص[مَصْدَر]

761هـ=1359م

تَوْقِيصُ الأَعْنَاقِ وَنَحْوِهَا: دَقُّهَا وَتَكْسِيرُهَا.

مِنْ عَدْلِهِ يَضَعُ الأَشْيَاءَ مَوْضِعَهَا لِلصَّدْرِ طَعْنٌ، وَلِلْأَعْنَاقِ تَوْقِيصُ
ابن جابر الأندلسيّ
ديوان المقصد الصّالِح في مدح الملك الصّالِح: ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: أحمد فوزي الهيب، دار سعد الدِّين، دمشق، ط1، (1429ه/ 2008م)، ‏108‏.

تَوَاقُص[مَصْدَر]

842هـ=1438م

تَوَاقُصُ الشَّخْصِ: انْحِنَاؤُهُ وَتَقَاصُرُهُ عَلَى الشَّيْءِ؛ لِيُمْسِكَهُ بِعُنُقِهِ، حَتَّى لَا يَسْقُطَ.

"إِذَا كَانَ ضَيِّقًا، فَإِنَّهُ يُجْزِئُهُ أَنْ يَتَّزِرَ بِهِ؛ لِأَنَّ القَصْدَ الأَصْلِيَّ سَتْرُ العَوْرَةِ، وَهُوَ يَحْصُلُ بِالِائْتِزَارِ، وَلَا يَحْتَاجُ إِلَى التَّوَاقُصِ المُغَايِرِ لِلِاعْتِدَالِ المَأْمُورِ بِهِ"
ابن حجر العسقلاني
فتح الباري بشرح صحيح البخاري: ابن حجر العسقلاني (ت، 852هـ)، تحقيق: عبد العزيز بن عبد الله بن باز، محمد فؤاد عبد الباقي، محبّ الدّين الخطيب، مكتبة المعرفة، بيروت، (د.ت)، ‏1/‏472‏.
