رَشَاش[اسْمُ جِنْس]

قبل 150ق.هـ=476م

الرَّشَاشُ : مَا يَنْضَحُ مِنَ الدَّمِ وَنَحْوِهِ وَيَتَنَاثَرُ.

وَيَوْمَ جُرَادٍ لَمْ نَدَعْ لِرَبِيعَةٍ وَإِخْوَتِهَا أَنْفًا بِهِ غَيْرَ أَجْدَعَا
بِضَرْبٍ تَظَلُّ الطَّيْرُ تَقْفُو رَشَاشَهُ عَلَى الصَّخْرِ، حَتَّى تَنْثَنِي مِنْهُ ظُلَّعَا
منسوب إلى عمارة بن عبيد الهَمْدانيّ
شعر هَمْدان وأخبارها في الجاهلية والإسلام: جمع وتحقيق ودراسة: حسن عيسى أبو ياسين، دار العلوم، الرِّياض، ط1، (1403هـ/ 1983م)، ‏271‏.

رَشَاش[اسْم]

8هـ=629م

الرَّشَاشُ : البُكَاءُ.

تَبِيتُ عُيُونُ النَّائِحَاتِ عَلَيْهِمُ تَجُودُ بِأَسْبَابِ الرَّشَاشِ وَبِالوَبْلِ
عليّ بن أبي طالب
ديوان أمير المؤمنين الإمام علي بن أبي طالب رضي الله عنه وكرم الله وجهه: جمع وترتيب: عبد العزيز الكرم، ط1، (1409هـ/ 1988م)، ‏153‏.

مُرِشّ[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

المُرِشُّ مِنَ الطَّعْنِ:الوَاسِعُ النَّاشِرُ الدَّمَ.

وَطَعْنَةِ خَلْسٍ قَدْ طَعَنْتَ مُرِشَّةٍ لَهَا نَفَذٌ تَضِلُّ فِيهِ المَسَابِرُ
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏80‏.
ن125هـ=743م

المُرِشُّ : المُمْطِرُ مَطَرًا خَفِيفًا.

سَقَى مَاءُ النَّدَى مِنْهَا رِيَاضًا وَسَارِحَةً مَعَ اليَوْمِ المُرِشِّ
النّابغة الشَّيْبانيّ
ديوان نابغة بني شَيْبان: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1351هـ/ 1932م)، ‏20‏.
169هـ=785م

المُرِشُّ مِنَ الأَشْيَاءِ: الَّذِي يَقْطُرُ مَاؤُهُ وَنَحْوُهُ.

وَلِلدَّمْعِ مِنْ عَيْنَيْ شَرِيجَا صَبَابَةٍ مُرِشُّ الرَّجَا، وَالجَائِلُ المُتَرَقْرِقُ
ابن المولى
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏3/‏201‏.

أَرَشَّ[لَازِم]

قبل 50ق.هـ=573م

أَرَشَّ السَّحَابُ وَنَحْوُهُ: نَضَحَ مِمَّا فِيهِ مِنْ مَاءٍ وَنَحْوِهِ.

وَلَا رِيِحُ فَغْوٍ أَوْ خُزَامَى وَحَنْوَةٍ أَرَشَّتْ عَلَيْهَا سَارِيَاتُ السَّحَائِبِ
بِأَطْيَبَ مِنْ فِيهَا إِذَا مَا تَقَلَّبَتْ مَعَ اللَّيْلِ وَسْنَى جَانِبًا بَعْدَ جَانِبِ
الحارث بن صُريم الوادعيّ الهَمْدانيّ
شعر هَمْدان وأخبارها في الجاهلية والإسلام: جمع وتحقيق ودراسة: حسن عيسى أبو ياسين، دار العلوم، الرِّياض، ط1، (1403هـ/ 1983م)، ‏248‏.

أَرَشَّ[مُتَعَدٍّ]

110هـ=728م

أَرَشَّهُ : جَعَلَهُ يَنْثُرُ مَا فِيهِ مِنْ مَاءٍ وَنَحْوِهِ.

وَعَيْنٌ أَرَشَّتْهَا، بِأَكْنَافِ مُشْرِفٍ مِنَ الزُّرْقِ فِي سَفْكٍ، دِيَارُ الحَبَائِبِ
ذو الرُّمَّة
ديوان ذي الرُّمّة غَيلان بن عقبة العَدَويّ: شرح: أبي نصر أحمد بن حاتم الباهلي صاحب الأصمعي، رواية: أبي العبّاس ثعلب، حققه وقدم له وعلق عليه: عبد القدوس أبو صالح، مؤسسة الرسالة، بيروت، ط3، (1414هـ/ 1993م)، ‏190‏.

أَرَشَّ[مُتَعَدٍّ بِالحَرْف]

ن206هـ=821م

أَرَشَّتِ النَّاقَةُ فِي الزِّمَامِ: ذَهَبَتْ.

"تَقُولُ: أَرَشَّتِ النَّاقَةُ فِي الزِّمَامِ"
من رواية أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏8‏.

رَشّ[مَصْدَر]

قبل 50ق.هـ=573م

رَشُّ الدَّمْعِ وَنَحْوِهِ: نَضْحُهُ وَتَنَاثُرُهُ.

أَمِنْ ذِكْرِ نَبْهَانِيَّةٍ حَلَّ أَهْلُهَا بِجِزْعِ المَلَا، عَيْنَاكَ تَبْتَدِرَانِ؟
فَدَمْعُهُمَا سَكْبٌ، وَسَحٌّ، وَدِيمَةٌ، وَرَشٌّ، وَتَوْكَافٌ، وَتَنْهَمِلَانِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏88‏.

رَشّ[اسْم]

ن50هـ=670م

الرَّشُّ : المَطَرُ الخَفِيفُ.

فَنَزَلْنَا بِمَلِيعٍ مُقْفِرٍ مَسَّهُ طَلُّ مِنَ الدَّجْنِ، وَرَشّْ
النّابغة الجَعْديّ
شعر النابغة الجعدي: تحقيق: عبد العزيز رباح، منشورات المكتب الإسلامي، دمشق، ط1، (1384هـ/ 1964م)، ‏244‏.
1380هـ=1960م

رَشُّ البُنْدُقِيَّةِ: رَصَاصُهَا المُسْتَدِيرُ.

"ثُقُوبُ الرَّشِّ: مَرَضٌ فِطْرِيُّ الحَجْمِ، تَتَجَمَّعُ أَفْرَادُهُ فِي أَعْدَادٍ كَبِيرَةٍ يُصِيبُ الأَوْرَاقَ، فَيُحْدِثُ بِهَا ثُقُوبًا فِي حَجْمِ رَشِّ البُنْدُقِيَّةِ"
دائرة المعارف الزّراعيّة العربيّة
دائرة المعارف الزراعية العربية: المجلة الزراعية، د.م، (1960م)، ‏164‏.

أَجَشُّ هَزِيمٌ جَوْشَنِيٌّ رَشِيشُهُ مَرِيشُ كَمِيشُ الرَّشِّ رِيٌّ يُرَيِّفُ

1374هـ=1954م

الرَّشِيشُ : البُنْدُقِيَّةُ الصَّغِيرَةُ.

"وَلَعَلَّ أَبَا سَلِيمٍ ظَنَّ حَسُّونَ مِنَ الهَاغَانَا، فَقَدْ أَمْسَكَ بِيَدِي بَيْنَمَا كُنْتُ أَحْمِلُ عَلَى كَتِفِي صُنْدُوقًا مِنْ ذَخِيرَةِ الرَّشِيشَاتِ ، وَأَشَارَ بِحَذَرٍ إِلَى وَقْفَةِ هَذَا الصَّبِيِّ الَّتِي تَدْعُو إِلَى الِاشْتِبَاهِ"
عبد السّلام العجيليّ

رَشِيش[صِفَةٌ مُشَبَّهَة]

ن231هـ=845م

الرَّشِيشُ مِنَ الدَّمِ وَنَحْوِهِ: الوَاسِعُ المُتَنَاثُرُ.

بِضَرْبٍ يُخَفِّتُ فَوَّارُهُ وَطَعْنٍ تَرَى الدَّمَ مِنْهُ رَشِيشَا
إِذَا قَتَلُوا مِنْكُمُ فَارِسًا ضَمِنَّا لَهُ خَلْفَهُ أَنْ يَعِيشَا
من إنشاد أبي عبد الله ابن الأعرابيّ اللُّغويّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏11/‏286‏.

إِرْشَاش[مَصْدَر]

قبل 50ق.هـ=573م

إِرْشَاشُ الفَرَسِ: تَعْرِيقُهُ بِالرَّكْضِ لِيَضْمُرَ.

طَوَاهُ القَنِيصُ وَتَعْدَاؤُهُ وَإِرْشَاشُ عِطْفَيْهِ حَتَّى شَسَبْ

إِرْشَاشُ الدَّمْعِ وَنَحْوِهِ: نَضْحُهُ وَسَيَلَانُهُ.

أَمِنْ دِمْنَةٍ، مِنْ آلِ لَيْلَى غَشِيتَهَا عَلَى تِمِّ حَوْلٍ، مَاءُ عَيْنَيْكَ سَافِحُ؟!
كَإِرْشَاشِ غَرْبٍ بَيْنَ قَرْنَيْ مَحَالَةٍ مُقَحِّمُهُ دَامِي السَّلَائِقِ، نَاضِحُ
رقيع الوالبيّ

"هَذِهِ البُزُوُر سِنْجَابِيَّةٌ مُسْتَدِيرَةٌ فِي حَجْمِ رُشِّ الرَّصَاصَ، وَفِيهَا خُطُوطٌ صَغِيرَةٌ مُنْتَهِيَةٌ بِانْتِفَاخٍ صَغِيرٍ، وَرَائِحَتُهَا كَرَائِحَةِ البَقِّ"
أحمد حسن الرشيدي
كتاب عمدة المحتاج في علمي الأدوية والعلاج (المادة الطبية): أحمد حسن الرشيدي (ت، 1282هـ)، دار الطباعة الخديوية، بولاق، القاهرة، (1283ه)، ‏2/‏638‏.

رَشَّاشِيَّة[اسْم]

1409هـ=1988م

الرَّشَّاشِيَّةُ : فُطْرٌ يَنْمُو بِشَكْلٍ زُغْبِيٍّ قُطْنِيٍّ، كَثِيرُ الِانْتِشَارِ فِي الطَّبِيعَةِ، مُتْلِفٌ لِلْفَوَاكِهِ وَالخُضَرِ خَاصَّةً.

"يُمْكِنُ لِلْأَجْوَافِ المُزْمِنَةِ فِي الرِّئَةِ أَنْ تَنْخَمِجَ بِفُطْرٍ، مِثْلِ الرَّشَّاشِيَّةِ ، الَّذِي يَكُونُ ـ عِنْدَئِذٍ ـ كُتْلَةً مُسْتَدِيرَةً دَاخِلَ الكَهْفِ"
منظمة الصحة العالمية
النظام الشعاعي الأساسي لمنظمة الصحة العالمية (دليل الممارس العام لقراءة الصور الشعاعية): منظمة الصحة العالمية ، منظمة الصحة العالمية، المكتب الإقليمي لشرق البحر المتوسط، الإسكندرية، (1988م)، ‏72‏.
