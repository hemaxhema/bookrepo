أَفْقَصَ[مُتَعَدٍّ]

35هـ=655م

أَفْقَصَ الشَّيْءَ: كَسَرَهُ وَفَلَقَهُ.

"جَاءَ رَجُلٌ مِنْ خَلْفِ عُثْمَانَ بِسَعْفَةٍ، فَضَرَبَ بِهَا جَبْهَتَهُ، فَرَأَيْتُ الدَّمَ يَسِيلُ وَهُوَ يَمْسَحُهُ وَيَقُولُ: اللَّهُمَّ، لَا يَطْلُبْ بِدَمِي غَيْرُكَ! وَجَاءَ آخَرُ، فَضَرَبَهُ بِالسَّيْفِ عَلَى صَدْرِهِ، فَأَفْقَصَهُ ، وَتَعَاوَرُوهُ بِأَسْيَافِهِمْ "
رَيْطة مولاة أسامة بن زيد
سمط النجوم العوالي في أنباء الأوائل والتّوالي: عبد الملك بن حسين العصامي (ت، 1111هـ)، تحقيق وتعليق: عادل أحمد عبد الموجود، علي محمَّد معوض، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏2/‏529‏.

فَقُّوص[اسْمُ جِنْس]

ن175هـ=791م

الفَقُّوصُ : البِطِّيخُ قَبْلَ النُّضْجِ.

" الفَقُّوصُ : البِطِّيخُ -بِلُغَةِ مِصْرَ- الَّذِي لَمْ يَنْضَجْ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏67‏.
ن313هـ=925م

الفَقُّوصُ : القِثَّاءُ الصَّغِيرُ الَّذِي لَا يَكْبُرُ وَلَا يَعْدُو أَطْوَلُهُ الفِتْرَ.

"أَمَّا الفَقُّوصُ فَرَدِيءٌ عَسِرُ الهَضْمِ، وَلَاسِيَّمَا مَا صَلُبَ وَكَبُرَ مِنْهُ، وَأَمَّا الصَّغِيرُ الرَّطْبُ مِنْهُ فَدُونَ ذَلِكَ"
أبو بكر الرّازيّ
كتاب منافع الأغذية ودفع مضارّها: أبو بكر الرّازيّ (ت، 313هـ)، المطبعة الخيريّة، مصر، ط1، (1305هـ)، ‏46‏.

تَفَقَّصَ[لَازِم]

ن255هـ=869م

تَفَقَّصَ البَيْضُ: تَكَسَّرَ قِشْرُهُ عَمَّا فِيهِ.

"إِذَا كَانَ ذَلِكَ، سَدَّتْ عَلَيْهِنَّ بَابَ الجُحْرِ، ثُمَّ تَدَعُهُنَّ أَرْبَعِينَ يَوْمًا؛ فَيَتَفَقَّصُ البَيْضُ، وَيَظْهَرُ مَا فِيهِ، فَتَحْفِرُ عَنْهُنَّ عِنْدَ ذَلِكَ"
من رواية الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏6/‏117‏.

فَقَصَ[مُتَعَدٍّ]

ن313هـ=925م

فَقَصَ البَيْضَةَ: كَسَرَ قِشْرَهَا عَمَّا فِيهَا.

" يُفْقَصُ البَيْضُ، وَيُضْرَبُ ضَرْبًا جَيِّدًا، وَيُصَبُّ فِيهِ مِثْلُ رُبُعِهِ مِنْ مَاءِ البَصَلِ المَدْقُوقِ المَعْصُورِ، وَيُجْعَلُ رِعَادًا، وَيُتَحَسَّى"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏3/‏392‏.

فَقَصَ[لَازِم]

ن369هـ=979م

فَقَصَتِ البَيْضَةُ: انْكَسَرَ قِشْرُهَا عَمَّا فِيهَا.

"تَبْدَأُ النِّسَاءُ بِتَحْضِينِ بَيْضِ دُودِ الحَرِيرِ حَتَّى تَفْقِصَ وَتَنْصَرِفَ الغَرَانِقُ إِلَى الجَزَائِرِ"
عَريب بن سعد القرطبيّ
كتاب الأنواء: عَريب بن سعد القرطبيّ (ت، 369هـ)- ضمن: رسالتان في الأنواء: عريب بن سعد القرطبي، ابن البنا المراكشي: تحقيق: إبراهيم القادري بوتشيش، سعيد بنحمادة، جامعة مولاي إسماعيل، مكناس، (2015م)، ‏76‏.

انْفَقَصَ[مُتَعَدٍّ بِالحَرْف]

ن313هـ=925م

انْفَقَصَ الفَرْخُ عَنِ البَيْضَةِ: انْكَسَرَ عَنْهُ قِشْرُهَا.

"آخِرُ مِثْقَالٍ مِنْ قُشُورِ البَيْضِ الَّذِي انْفَقَصَ عَنْهَا الفَرُّوجُ بِمَاءِ العَسَلِ"
أبو بكر الرّازيّ
مقالة في الحصى في الكلى والمثانة: أبو بكر الرّازيّ (ت، 313هـ)، تحقيق: ب. دو كونينغ، مطبعة بريل، ليدن، (1896م)، ‏208‏-‏210‏.

تَفَقُّص[مَصْدَر]

ن313هـ=925م

تَفَقُّصُ الفَرْخِ: تَكَسُّرُ قِشْرِ البَيْضَةِ عَنْهُ.

"مِنْ كُلِّ وَاحِدٍ مِثْقَالُ قُشُورِ بَيْضِ الفَرَارِيجِ سَاعَةَ تَفَقُّصِ الفَرْخِ"
أبو بكر الرّازيّ
مقالة في الحصى في الكلى والمثانة: أبو بكر الرّازيّ (ت، 313هـ)، تحقيق: ب. دو كونينغ، مطبعة بريل، ليدن، (1896م)، ‏200‏.

فَقْص[مَصْدَر]

418هـ=1027م

فَقْصُ البَيْضِ: إِخْرَاجُ مَا فِيهِ.

"الذَّكَرُ مِنْ زَوْجَيِ الحَمَامِ وَمَا أَشْبَهَهَا، يَحْضُنُ شَطْرًا مِنَ النَّهَارِ، وَالأُنْثَى تَحْضُنُ بَاقِيَ النَّهَارِ وَاللَّيْلَ أَجْمَعَ، وَتَتَوَلَّى كَسْرَ البَيْضِ وَفَقْصَهُ بَعْدَ عِشْرِينَ"
ابن سينا
الشّفاء الطّبيعيّات: ابن سينا (ت، 428هـ)، ج(1- 3)، تحقيق: سعيد زايد وآخرين، تصدير ومراجعة: إبراهيم مدكور، منشورات مكتبة آية الله العظمى المرعشي النجفي، قم، دار الكتاب العربي- الهيئة المصرية العامة للتأليف والنشر، القاهرة، ط2، (1406هـ)، ‏3/‏81 ‏-‏ 82‏.
ن1005هـ=1596م

فَقْصُ العَدُوِّ وَنَحْوِهِ: كَسْرُهُ وَقَهْرُهُ.

وَضَلُّوا عَلَى عِلْمٍ، فَزَادَ عِنَادُهُمْ لَهُ حَسَدًا، فَاسْتَوْجَبُوا الجَزْرَ وَالفَقْصَا
عبد الرّحمن بن أحمد الحميدي
ديوان الدّر المنظم في مدح النّبي الأعظم: عبد الرّحمن بن أحمد الحميدي (ت، 1005هـ)، المطبعة العامرة الشّرفية، مصر، (1322ه)، ‏82‏.

فَقِيص[صِفَةٌ مُشَبَّهَة]

418هـ=1027م

الفَقِيصُ مِنَ البَيْضِ: المُنْكَسِرُ قِشْرُهُ عَمَّا فِيهِ.

"البَيْضُ مَكَانُهُ الرَّحِمُ، لَا المَعِدَةُ، وَإِلَّا لَفَعَلَتْ فِيهِ المَعِدَةُ فِعْلَهَا. وَالأَنْكَلِيسُ فَإِنَّمَا يُوجَدُ البَيْضُ فِي مَعِدَتِهِ فَقَطْ فَقِيصًا "
ابن سينا
الشّفاء الطّبيعيّات: ابن سينا (ت، 428هـ)، ج(1- 3)، تحقيق: سعيد زايد وآخرين، تصدير ومراجعة: إبراهيم مدكور، منشورات مكتبة آية الله العظمى المرعشي النجفي، قم، دار الكتاب العربي- الهيئة المصرية العامة للتأليف والنشر، القاهرة، ط2، (1406هـ)، ‏3/‏65‏.

مَفْقُوص[اسْمُ مَفْعُول]

ن450هـ=1058م

المَفْقُوصُ مِنَ البَيْضِ: المَكْسُورُ قِشْرُهُ عَمَّا فِيهِ.

وَالبَيْضُ مَفْقُوصٌ بِهَا يُنَجَّمُ كَأَنَّهُ لَمَّا عَلَاهَا أَنْجُمُ
الشّريف العَقيليّ
كتاب الوافي بالوفيات: صلاح الدّين الصّفديّ (ت، 764هـ)، تحقيق واعتناء: أحمد الأرناؤوط، تركي مصطفى، دار إحياء التراث العربي، بيروت، ط1، (1420ه/ 2000م)، ‏21/‏33‏.

فَاقِص[اسْمُ فَاعِل]

ن538هـ=1143م

فَاقِصُ البَيْضِ وَنَحْوِهِ: كَاسِرُ قِشْرِهِ عَمَّا فِيهِ.

وَبَاضَتْ مِرَارًا فِتْنَةٌ جَاهِلِيَّةٌ فَكَانَ لَهَا فِي آلِ مَرْوَانَ فَاقِصُ
جار الله الزمخشريّ
ديوان جار الله الزّمخشريّ: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: فاطمة يُوسف الخيميّ، دار صادر، بيروت، ط1، (1429ه/ 2008م)، ‏326‏.

انْفِقَاص[مَصْدَر]

635هـ=1237م

انْفِقَاصُ البَيْضِ: انْكِسَارُ قِشْرِهِ عَمَّا فِيهِ.

"رَمَادُ قِشْرِ بَيْضِ الدَّجَاجِ بَعْدَ انْفِقَاصِهِ عَنِ الفَرْخِ، شَدِيدُ القُوَّةِ جِدًّا فِي إِخْرَاجِ الحَصَاةِ مِنَ المَثَانَةِ شُرْبًا"
ابن البَيْطار
المغني في الأدوية المفردة النافعة من الأمراض: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، مراجعة: حسيني الجندي، جامعة مصر للعلوم والتكنولوجيا، القاهرة، ط1، (1441ه/ 2019م)، ‏2/‏591‏.
1428هـ=2006م

الِانْفِقَاصُ : البُزُوغُ وَالظُّهُورُ.

"تَتَعَرَّضُ لِلْـ( م. خ. م ) بِفِعْلِ تَغَيُّرِ تَرْكِيزِ هُرْمُونَيْنِ اثْنَيْنِ؛ حَيْثُ يَنْخَفِضُ إِنْتَاجُ هُرْمُونِ الإِكْدِيزُونِ (Ecdyosone)، وَيَرْتَفِعُ إِفْرَازُ هُرْمُونِ الِانْفِقَاصِ (Hormone d ' éclosion)"
مجلة عالم الفكر
-: ‏مجلة عالم الفكر،‏مجلة عالم الفكر: الكويت، مج35، ع2، (2006م)، ‏400‏.

فُقَّيْص[اسْم]

ن646هـ=1248م

الفُقَّيْصُ : حَيَوَانٌ صَغِيرٌ، يَتَوَلَّدُ عَلَى الشَّوْكِ وَأَشْجَارِ البَلُّوطِ، وَيَنْتِجُ عَنْهُ صِبْغٌ أَحْمَرُ.

"الَّذِينَ يَتَّجِرُونَ بِهِ يَكْسِرُونَهُ، وَتَخْتَلِطُ مَائِيَّتُهُ وَدَمُهُ بِأَجْزَائِهِ، وَالَّذِي يَبْقَى صَحِيحًا يَخْرُجُ فِي شَهْرِ العُنْصُرَةِ حَيَوَانًا أَحْمَرَ كَأَنَّهُ الصِّئْبَانُ... وَكُلَّمَا قَدُمَ كَانَ أَجْوَدَ لِلصِّبْغِ، وَقَدْ يَتَوَلَّدُ عَلَى شَجَرِ البَلُّوطِ، وَيَجْمَعُهُ الرِّجَالُ وَالنِّسَاءُ، وَيُسَمُّونَهُ فُقَّيْصًا "
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏3/‏438 ‏-‏ 439‏.

تَفْقِيص[مَصْدَر]

786هـ=1384م

تَفْقِيصُ الأَصَابِعِ: ضَغْطُ مَفَاصِلِهَا حَتَّى تُصْدِرَ صَوْتَ فَرْقَعَةٍ.

"سَمِعَ ابْنُ القَاسِمِ كَرَاهَةَ تَفْقِيصِ الأَصَابِعِ بِهِ وَبِغَيْرِهِ، وَخَصَّهُ ابْنُ القَاسِمِ بِهِ"
أبو عبد الله محمد بن محمد ابن عرفة
المختصر الفقهي: أبو عبد الله محمد بن محمد ابن عرفة (ت، 803هـ)، صححه ونقحه وعلق هوامشه: حافظ عبد الرحمن محمد خير، مسجد ومركز الفاروق عمر بن الخطاب، دبي، ط1، (1435ه/ 2014م)، ‏1/‏357‏.
1324هـ=1905م

تَفْقِيصُ البَيْضِ: تَكْسِيرُ قِشْرِهِ عَمَّا فِيهِ.

"وَضَعَ البَيْضَ فِي الخَلَايَا، فَكَانَ يَتَفَقَّصُ بَعْدَ 19 يَوْمًا بِالتَّمَامِ، فَعَرَفَ بِذَلِكَ أَنَّ كُوَّارَاتِ النَّحْلِ تَقُومُ مَقَامَ الدَّجَاجِ الحَوَاضِنِ فِي تَفْقِيصِ البَيْضِ"
مجلّة المشرق
النحل الحواضن: ‏المشرق،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين، بيروت، س9، ع14، (24/ 05/ 1324هـ)، ‏675‏.
1366هـ=1946م

تَفْقِيصُ البُذُورِ: تَشَقُّقُ أَغْلِفَتِهَا عَنْ أَجِنَّةِ النَّبَاتِ.

"إِنَّ مِبْذَرَنَا يَفْحَصُ إِفْرَادِيًّا، وَالبَذْرَةُ المَذِرَةُ غَيْرُ الصَّالِحَةِ لِلتَّفْقِيصِ ، تُنْفَى خَارِجًا"
مارون عبود
على المِحَكِّ، نظرات وآراء في الشعر والشعراء: مارون عبود، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2014م)، ‏146‏.

تَفْقِيص[اسْم]

1427هـ=2005م

التَّفْقِيصُ : لُعْبَةٌ يَلْعَبُهَا الفِتْيَانُ بِالبَيْضِ، يُحَاوِلُ فِيهَا كُلٌّ مِنْهُمْ أَنْ يَكْسِرَ البَيْضَةَ الَّتِي بِيَدِ صَاحِبِهِ.

"فِي صَبَاحِ الفِصْحِ أَيْ ذِكْرَى قِيَامَةِ يَسُوعَ مِنَ المَوْتِ، يَذْهَبُ النَّاسُ إِلَى الكَنِيسَةِ، ثُمَّ يَزُورُونَ بَعْضُهُمُ البَعْضَ، حَيْثُ يَتَنَافَسُونَ فِي تَفْقِيصِ البَيْضِ"
مارون الحايك
عين علق بين القرن السابع والقرن الحادي والعشرين: مارون الحايك، ميا برس، بيروت، (2005م)، ‏467‏.

مُفَاقَصَة[مَصْدَر]

1357هـ=1937م

المُفَاقَصَةُ : المُكَاسَرَةُ وَالمُغَالَبَةُ.

"كَسَّرُوا السِّجْنَ، وَوَثَبُوا عَلَى عَامِلِ الإِمَامِ عَلِيِّ بْنِ سُلَيْمَانَ، مُفَاقَصَةً ، فَانْفَلَتَ مِنْ بَيْنِهِمْ فِي سِتَّةِ فَوَارِسَ مِنْ بَنِي عَمِّهِ، وَمِنَ الطَّبَرِيِّينَ"
أحمد بن أحمد بن محمّد المطاع
تاريخ اليمن الإسلامي: أحمد بن أحمد بن محمّد المطاع (ت، 1366هـ)، تحقيق: عبد الله محمد الحبشي، منشورات المدينة- دار التنوير، ط1، (1407ه/ 1986م)، ‏99‏.

مِفْقَص[اسْمُ آلَة]

1383هـ=1963م

المِفْقَصُ : حَاضِنَةٌ آلِيَّةٌ، تُوَفِّرُ الشُّرُوطَ المُنَاسِبَةَ لِاكْتِمَالِ نُمُوِّ الأَجِنَّةِ دَاخِلَ البَيْضِ، قَبْلَ أَنْ تَخْرُجَ مِنْهُ.

"نَذْكُرُ اَنَّ جَمَاعَةَ النَّحْلِ جِسْمٌ مُوَحَّدُ الحَرَارَةِ ثَابِتُهَا، بِمَعْنَى أَنَّ دَرَجَةَ الحَرَارَةِ فِي دَاخِلِ القَفِيرِ كُلِّهِ -بِمَا فِي ذَلِكَ مِفْقَصُ البَيْضِ وَخَلَايَا الدِّيدَانِ- هِيَ مُسْتَقِرَّةٌ عِنْدَ 33- 34 دَرَجَةً، لًا تَرْتَفِعُ وَلَا تَنْخَفِضُ، مَهْمَا كَانَتْ أَحْوَالُ الطَّقْسِ فِي الخَارِجِ"
مجلة العلوم
-: ‏مجلة العلوم،‏مجلة العلوم: دار العلم للملايين، بيروت، مج 8، 1963م، ‏74‏.

فَقَّاص[اسْم]

1427هـ=2005م

الفَقَّاصُ : حَلْوَى جَافَّةٌ، تُتَّخَذُ مِنَ البَيْضِ وَالسُّكَّرِ وَالدَّقِيقِ وَالخَمِيرَةِ، وَقَدْ يُضَافُ إِلَى مُكَوِّنَاتِهَا الزَّبِيبُ وَالفَوَاكِهُ المُجَفَّفَةُ وَنَحْوُهُمَا، تُؤْكَلُ فِي الأَعْرَاسِ وَالمُنَاسَبَاتِ الدِّينِيَّةِ فِي المَغْرِبِ الأَقْصَى.

"صَارَ الطِّفْلُ يَتَجَرَّعُ المَاءَ، وَيَلُوكُ قِطَعًا يَابِسَةً مِنَ الفَقَّاصِ "
محمّد حمداني
في الواحة، صور من أعماق الذاكرة: محمد حمداني، المؤلف، د.م، (2005م)، ‏31‏.

مُفَقَّص[اسْمُ مَفْعُول]

1430هـ=2008م

المُفَقَّصُ مِنَ البَيْضِ: المُكَسَّرُ قِشْرُهُ عَمَّا فِيهِ.

"يَسْتَعْمِلُ مُؤَلِّفُو الطَّبِيخِ فِي وَصْفِهِمْ لِتَزْيِينِ الطَّعَامِ بِالبَيْضِ المُفَقَّصِ أَوِ المَشْطُورِ شَطْرَيْنِ، تَرْکِیبَ عُيُونِ البَيْضِ"
سهام الدّبّابي الميساوي
الطعام والشراب في التراث العربي: سهام الدّبّابي الميساوي، منشورات كلية الآداب والفنون والإنسانيات، جامعة منوبة، تونس، (2008م)، ‏467‏.
