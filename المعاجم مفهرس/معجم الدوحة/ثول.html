ثَوْل[اسْمُ جَمْع]

قبل 50ق.هـ=573م

الثَّوْلُ : جَمَاعَةَ النَّحْلِ.

"وَابْنَاهُ وَابْنَ اللَّيْل... أَجَزْتَ بِاللَّيْل، تَضْرِبُ بِالذَّيْل، بِرَجْلٍ كَالثَّوْل "
أُميمة القَيْنيّة
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏21/‏124‏.

ثَوْل[اسْم]

ن110هـ=728م

الثَّوْلُ : الفَحْلُ مِنَ الضِّبَاعِ.

قَتَلْتُمْ بِهِ ثَوْلَ الضِّبَاعِ، فَغَادَرَتْ مَنَاصِلُكُم مِنْهُ خَصِيلًا مُوَضَّعَا
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏82‏.

انْثِيَال[مَصْدَر]

38هـ=658م

الِانْثِيَالُ عَلَى الشَّخْصِ: الإِقْبَالُ عَلَيْهِ، وَالِاحْتِشَادُ.

"فَمَا رَاعَنِي إِلَّا انْثِيَالُ النَّاسِ عَلَى أَبِي بَكْرٍ، وَإِجْفَالُهُمْ إِلَيْهِ لِيُبَايِعُوهُ"
عليّ بن أبي طالب
شرح نهج البلاغة: ابن أبي الحَديد (ت، 656هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية، القاهرة، منشورات مكتبة آية الله العظمى المرعشي النجفي- مؤسسة إسماعيليان، قم، ط2، (1383هـ/ 1963م- 1378هـ/ 1967م)، ‏6/‏95‏.
ن90هـ=709م

انْثِيَالُ الشَّيْءِ: انْهِيَالُهُ وَتَتَابُعُهُ.

"لَقَدْ قُلْتُ أُرْجُوزَتِي... وَأَنَّا بِالرَّمْلِ، فِي لَيْلَةٍ وَاحِدَةٍ، فَانْثَالَتْ عَلَيَّ قَوَافِيهَا انْثِيَالًا"
العجّاج
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏1/‏209‏.

انْثَالَ[مُتَعَدٍّ بِالحَرْف]

ن40هـ=660م

انْثَالَ النَّاسُ عَلَيْهِ: اجْتَمَعُوا عَلَيْهِ وَاحْتَشَدُوا.

"فَمَا رَاعَنِي إِلَّا وَالنَّاسُ كَعُرْفِ الضَّبُعِ إِلَيَّ، يَنْثَالُونَ عَلَيَّ مِنْ كُلِّ جَانِبٍ"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏1/‏36‏.
ن90هـ=709م

انْثَالَ عَلَيْهِ الشَّيْءُ: انْهَالَ عَلَيْهِ وَتَتَابَعَ.

"لَقَدْ قُلْتُ أُرْجُوزَتِي... وَأَنَّا بِالرَّمْلِ، فِي لَيْلَةٍ وَاحِدَةٍ، فَانْثَالَتْ عَلَيَّ قَوَافِيهَا انْثِيَالًا"
العجّاج
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏1/‏209‏.

انْثَالَ[لَازِم]

ن75هـ=694م

انْثَالَ التُّرَابُ وَنَحْوُهُ: انْصَبَّ وَانْهَالَ هَشًّا غَيْرَ مُتَمَاسِكٍ.

أَحَمِّ المَدَامِعِ يَبْنِي الكِنَا سَ فِي دَمِثِ التُّرْبِ يَنْثَالُ هَالِ
أُمَيّة بن أبي عائذ الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏176‏.

أَثْوَل[صِفَةٌ مُشَبَّهَة]

ن110هـ=728م

الأَثْوَلُ مِنَ الأَشْخَاصِ: المَجْنُونُ الأَحْمَقُ.

إِنَّ الحَيَاةَ إِلَى الرِّجَالِ بِغِيضَةٌ بَعْدَ الَّذِي فَعَلَ اللَّئِيمُ الأَثْوَلُ
الفرزدق
شرح نقائض جرير والفرزدق: رواية: أبي عبد الله اليزيدي عن أبي سعيد السكري عن ابن حبيب عن أبي عبيدة، تحقيق وتقديم: محمّد حُوّر، وليد محمود خالص، منشورات المجمع الثقافي، أبو ظبي، ط2، (1998م)، ‏2/‏384‏.
ن206هـ=821م

الأَثْوَلُ مِنَ الأَشْخَاصِ: البَطِيءُ النُّصْرَةِ وَالخَيْرِ.

بَرَّضْتَ لِي شَيْئًا وَلَمْ تُشَلْشِلِ بِمِثْلِ بَوْلِ الدَّفَئِيِّ الأَثْوَلِ
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏104‏.
ن240هـ=854م

الأَثْوَلُ مِنَ الغَنَمِ وَنَحْوِهَا: الَّذِي يُصِيبُهُ دَاءٌ يُشْبِهُ الجُنُونَ فَلَا يَتْبَعُ الغَنَمَ، وَيَسْتَديرُ فِي مَرْتَعِهِ.

تَلْقَى الأَمَانَ عَلَى حِيَاضِ مُحَمَّدٍ ثَوْلَاءُ مُخْرِفَةٌ وَذِئْبٌ أَطْلَسُ
مُخَلَّد بن بَكّار الموصليّ
الحماسة المغربيّة (مختصر كتاب صفْوة الأدب ونخبة ديوان العرب): الجُرَاويّ (ت، 609هـ)، حققه: محمَّد رضوان الدّاية، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط1، (1411ه/ 1991م)، ‏1/‏388‏.

ثَوِلَ[لَازِم]

ن175هـ=791م

ثَوِلَتِ الشَّاةُ وَنَحْوُهَا: أَصَابَهَا الثَّوَلُ.

"الثَّوَلُ: شِبْهُ جُنُونٍ فِي الشَّاءِ، يُقَالُ: شَاةٌ ثَوْلَاءُ، وَقَدْ ثَوِلَتْ تَثْوَلُ ثَوَلًا، وَالذَّكَرُ: أَثْوَلُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏239‏.

ثَوَل[مَصْدَر]

ن175هـ=791م

الثَّوَلُ : جُنُونٌ يُصِيبُ الشَّاةَ، فَلَا تَتْبَعُ الغَنَمَ، وَتَسْتَديرُ فِي مَرْتَعِهَا.

" وَالثَّوَلُ : شِبْهُ جُنُونٍ فِي الشَّاءِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏239‏.

ثَوَّالَة[اسْمُ جَمْع]

202هـ=817م

الثَّوَّالَةُ : الجَمَاعَةُ مِنَ الطَّيْرِ وَنَحْوِهِ.

"مَرَّتْ بِنَا زُمَّةٌ مِنْ طَيْرٍ، وَثَوَّالَةٌ "
قُطْرُب
كتاب الفَرْق في اللُّغة: قُطْرُب (ت، 206هـ)، حققه: خليل إبراهيم العطية، راجعه: رمضان عبد التوّاب، مكتبة الثقافة الدينية، القاهرة، (1987م)، ‏155‏.

مُنْثَال[اسْمُ فَاعِل]

278هـ=891م

المُنْثَالُ : المُنْهَالُ المُتَتَابِعُ.

وَإِمَّا عَرَاهُمْ مَادِحُوهُمْ تَحَاشَدُوا لِتَصْدِيقِهِمْ، فَالقَوْلُ لِلْفِعْلِ مُنْثَالُ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏5/‏1998‏.
352هـ=963م

المُنْثَالُ : المُنْصَبُّ المُنْهَالُ.

إِذَا عَدَّ مِنْ آلِ المُهَلَّبِ أُسْرَةً مَعَاقِلُهَا أَسْيَافُهَا وَدُرُوعُهَا
رَأَيْتَ العُلا مُنْثَالَةً مِنْ شِعَابِهَا عَلَيْهِ، وَمَجْمُوعًا إِلَيْهِ جَمِيعُهَا
السَّرِيّ الرَّفّاء الكنديّ
ديوان السَّرِيّ الرَّفّاء: تقديم وشرح: كرم البستاني، مراجعة: ناهد جعفر، دار صادر، بيروت، ط1، (1996م)، ‏281‏.
797هـ=1394م

المُنْثَالُ : المُحْتَشِدُ المُجْتَمِعُ مِنْ كُلِّ جِهَةٍ.

"وَأَصْبَحَ النَّاسُ مِنَ الغَدِ مِنَ النَّوَاحِي القَرِيبَةِ مُنْثَالِينَ لِلْبَيْعَةِ"
ابن خلدون
كتاب العبر وديوان المبتدأ والخبر (في أيام العرب والعجم والبربر ومن عاصرهم من ذوي السلطان الأكبر): ابن خلدون (ت، 808هـ)، تحقيق: إبراهيم شبوح وآخرين، دار القيروان، تونس، الدار العربية للكتاب، تونس- طرابلس، ط1، (2006- 2015م)، ‏5/‏277‏.

تَثَوُّل[مَصْدَر]

ن282هـ=895م

التَّثَوُّلُ : الِاجْتِمَاعُ وَالِالْتِفَافُ.

"نَحْسِبُ أَنَّهَا سُمِّيَتْ: ثَوْلًا لِتَثَوُّلِهَا ؛ وَهُوَ اجْتِمَاعُهَا وَالْتِفَافُهَا"
أبو حنيفة الدّينوريّ
كتاب العسل والنّحل والنباتات الذي تجرس منه: أبو حنيفة الدّينوريّ (ت، 282هـ)، حققه وقدم له: محمّد جبار المعيبد، مجلة المورد، بغداد، مج3، ع1، (1394هـ/ 1974م)، ‏126‏.

اثْوَلَّ[لَازِم]

ن283هـ=896م

اثْوَلَّ الشَّخْصُ: أَصَابَهُ الجُنُونُ.

وَفَارِسٍ أَجْبَنَ مِنْ صِفْرِدِ يَحْوَلُّ، أَوْ يَثْوَلُّ مِنْ صَفْرَهْ
لَوْ صَاحَ فِي اللَّيْلِ بِهِ صَائِحٌ لَكَانَتِ الأَرْضُ لَهُ طَفْرَهْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏978‏.
