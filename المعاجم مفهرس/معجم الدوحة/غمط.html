أَغْمَطَ[لَازِم]

قبل 13ق.هـ=609م

أَغْمَطَتِ السَّمَاءُ: دَامَ مَطَرُهَا زَمَنًا لَا يَنْقَطِعُ.

" أَغْمَطَتِ السَّمَاءُ فِي أَرْضِنَا ثَلَاثًا رَهْوًا؛ فَثَرَّتْ"
رجل من الأعراب
كتاب وصف المطر والسّحاب وما نَعَتَتْه العرب الرُّوّادُ من البقاع: ابن دُريد (ت، 321هـ)، تحقيق: عز الدين التنوخي، مجلة المجمع العلمي العربي، دمشق، مج38، (1382هـ/ 1963م)، ‏443‏.

أَغْمَطَ[مُتَعَدٍّ بِالحَرْف]

ن224هـ=839م

أَغْمَطَتِ الحُمَّى وَنَحْوُهَا عَلَيْهِ: لَزِمَتْهُ، وَدَامَتْ زَمَنًا لَا تَنْقَطِعُ.

"أَغْمَطَتْ عَلَيْهِ الحُمَّى"
أبو عُبيد الهَرَويّ
كتاب غريب الحديث: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: عبد السلام هارون وآخرين، الهيئة العامة لشؤون المطابع الأميرية، القاهرة، (1404هـ/ 1984م- 1419هـ/ 1999م)، ‏3/‏160‏.

غَمَطَ[مُتَعَدٍّ]

ن11هـ=632م

غَمَطَ الشَّخْصَ وَنَحْوَهُ: احْتَقَرَهُ، وَاسْتَصْغَرَهُ.

"وَلَكِنَّ البَغْيَ مَنْ بَطِرَ،  قَالَ: أَوْ قَالَ: سَفِهَ الحَقَّ، وَغَمَطَ النَّاسَ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏6/‏155،،‏(رقم الحديث: 3644).
ن158هـ=775م

غَمَطَ النِّعْمَةَ وَنَحْوَهَا: جَحَدَهَا، وَكَفَرَ بِهَا.

"أَفْضَى أَمْرُهُمْ إِلَى أَحْدَاثٍ مُتْرَفِينَ مِنْ أَبْنَائِهِمْ، فَغَمِطُوا النِّعْمَةَ، وَلَمْ يَشْكُرُوا العَافِيَةَ"
أبو جعفر المنصور
جمهرة خطب العرب في عصور العربية الزاهرة: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1352هـ/ 1933م- 1352هـ/ 1934م)، ‏3/‏33‏.
ن207هـ=822م

غَمَطَ الشَّيْءَ: غَطَّاهُ بِالتُّرَابِ وَنَحْوِهِ.

"وَدُفِنَ بِالكُوفَةِ لَيْلًا، وَغُمِطَ قَبْرُهُ"
الواقديّ
تاريخ مدينة دمشق وذكر فضلها وتسمية من حلها من الأماثل أو اجتاز بنواحيها من وارديها وأهلها: ابن عساكر (ت، 571هـ)، دراسة وتحقيق: عمر بن غرامة العمروي، دار الفكر، بيروت، ط1، (1415ه/ 1995م)، ‏42/‏13‏.

غَمَطَ[مَصْدَر]

ن780هـ=1378م

غَمَطُ الحَقِّ وَنَحْوِهِ: غَمْطُهُ.

وَأَلْزَمَ القَمْرَ المُنْشَقَّ طَاعَتَهُ فَانْشَقَّ رُغْمًا لِبَاغِ حَقَّهُ غَمَطَا
ابن جابر الأندلسيّ
ديوان المديح النبوي (ديوان نفائس المنح وعرائس المدح): ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: محمَّد طيب خطّاب، مكتبة الآداب، القاهرة، ط1، (1426ه/ 2005م)، ‏358‏.

غَمْط[مَصْدَر]

ن11هـ=632م

غَمْطُ الشَّخْصِ وَنَحْوِهِ: احْتِقَارُهُ، وَاسْتِصْغَارُهُ.

"إِنَّ اللهَ جَمِيلٌ يُحِبُّ الجَمَالَ. الكِبْرُ بَطَرُ الحَقِّ، وَغَمْطُ النَّاسِ"
حديث نبويّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏1/‏93،،‏(رقم الحديث: 147).
ن175هـ=791م

غَمْطُ النِّعْمَةِ وَنَحْوِهَا: جَحْدُهَا، وَالكُفْرُ بِهَا.

"يُقَالُ: لَا يُبْطِرَنَّ جَهْلُ فُلَانٍ حِلْمَكَ، أَيْ: لَا يُدْهِشُكَ. وَفِي مَعْنًى: كَالأَشَرِ وَغَمْطِ النِّعْمَةِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏422‏.
ن175هـ=791م

غَمْطُ المَاءِ وَنَحْوِهِ: جَرْعُهُ جَرْعًا مُتَتَابِعًا.

غَمْطُ غَمَالِيطَ غَمَلَّطَاتِ
من شواهد اللَّيْث بن المظفر الكناني
التكملة والذيل والصلة لكتاب تاج اللغة وصحاح العربية: الصّاغانيّ (ت، 650هـ)، حققه: إبراهيم إسماعيل الأبياري وآخرون، راجعه: محمد خلف الله أحمد وآخرون، مطبعة دار الكتب، القاهرة، (1390ه/ 1970م- 1400ه/ 1979م)، ‏4/‏158‏.
ن593هـ=1197م

غَمْطُ الشَّخْصِ وَنَحْوِهِ: تَغْطِيَتُهُ بِالتُّرَابِ حَتَّى المَوْتِ.

فَأَعْجِبْ بِهَا مِنْ آيَةٍ لِمُفَكِّرٍ يُقَصِّرُ عَنْ إِدْرَاكِهَا كُلُّ مَا احْطَطَّا
وَتَفْجِيرُهَا مِنْ صَخْرَةٍ عَشْرِ أَعْيُنٍ وَثِنْتَيْنِ، تَسْقِي كُلُّ وَاحِدَةٍ سِبْطَا
وَتَفْلِيقُهَا رَهْوًا مِنَ البَحْرِ طَرِيقًا، فَمِنْ نَاجٍ، وَمِنْ هَالِكٍ غَمْطَا
ابن أرفع رأس
شذور الذهب ديوان ابن أرفع رأس الجياني الأندلسي شاعر الحكماء وحكيم الشعراء: ابن أرفع رأس (ت، 593هـ)، حققه وقدمه وعلق عليه: الهواري غزالي، دار الكتب العلمية، بيروت، ط1، (1439ه/ 2018م)، ‏189‏-‏ 190‏.
1011هـ=1602م

غَمْطُ الشَّيْءِ: إِخْفَاؤُهُ، وَمُوَارَاتُهُ عَنِ الأَنْظَارِ.

وَسَالَتْ، بِسَلْسَالِ اللُّجَيْنِ، حِيَاضُهَا بِحَارًا، غَدَا عَرْضُ البَسِيطِ لَهَا شَطَّا
تَطَلَّعُ مِنْهَا، وَسْطَ وُسْطَاهُ، دُمْيَةٌ هِيَ الشَّمْسُ، لَا تَخْشَى كُسُوفًا، وَلَا غَمْطَا
أبو فارس عبد العزيز الفشتالي
روضة الآس العاطرة الأنفاس في ذكر من لقيته من أعلام الحضرتين مراكش وفاس: أحمد بن محمد المقريّ التلسمانيّ (ت، 1041هـ)، تحقيق: عبد الوهاب منصور، المطبعة الملكية، الرّباط، ط2، (1403ه/ 1983م)، ‏138‏.

غَامِط[اسْمُ فَاعِل]

ن142هـ=759م

الغَامِطُ الحَقَّ وَنَحْوَهُ: الجَاحِدُهُ، الكَافِرُ بِهِ.

"وَلَيْسَ بِنَاقِصِهِ فِي رَأْيِهِ، وَلَا غَامِطِهِ مِنْ حَقِّهِ أَلَّا يَكُونَ هُوَ اسْتَحْدَثَ ذَلِكَ، وَسَبَقَ إِلَيْهِ"
ابن المُقفَّع
الأدب الصغير: ابن المُقفَّع (ت، 142هـ)، تحقيق: أحمد زكي باشا، جمعية العروة الوثقى، الإسكندرية، ط1، (1329هـ/ 1911م)، ‏8‏.
ن167هـ=783م

الغَامِطُ الأَمْرَ: المُحْتَقِرُهُ، المُسْتَصْغِرُهُ.

وَغَامِطَةٍ لِفَقْدِكَ فِي التَّدَانِي تُسَائِلُ: كَيْفَ أَنْتَ عَلَى البِعَادِ؟
بشّار بن برد
ديوان بشار بن برد: جمع وتحقيق وشرح: محمّد الطاهر ابن عاشور، وزارة الثقافة، الجزائر، (2007م)، ‏3/‏23‏.

غُمُوط[مَصْدَر]

ن160هـ=777م

غُمُوطُ النِّعْمَةِ وَنَحْوِهَا: جَحْدُهَا، وَالكُفْرُ بِهَا.

"ثُمَّ رَأَيْنَا مَا عَزَمَ اللهُ بِهِ عَلَيْهِ مِنْ تَرْكِ الصَّوَائِفِ، مُرَاقِبًا لِلَّذِي كَانَ مِنْ غُمُوطِ أَهْلِ الشَّامِ لِمَا كَانُوا فِيهِ مِنَ النِّعْمَةِ"
يحيى بن زياد بن عُبيد الله الحارثيّ
جمهرة رسائل العرب في عصور العربية الزاهرة العصر الجاهلي عصر صدر الإسلام: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1356هـ/ 1937م- 1357هـ/ 1939م)، ‏3/‏213‏.

مُغْمِط[صِفَةٌ مُشَبَّهَة]

ن176هـ=792م

المُغْمِطُ مِنَ الأَشْيَاءِ: الدَّائِمُ زَمَنًا لَايَنْقَطِعُ.

ثَبْتٌ، إِذَا كَانَ الخَطِيبُ، كَأَنَّهُ شَاكٍ، يَخَافُ بُكُورَ وِرْدٍ مُغْمِطِ
ابن هَرْمَة
العباب الزاخر واللباب الفاخر: الصّاغانيّ (ت، 650هـ)، تحقيق: فير محمد حسن، محمد حسن آل ياسين، راجعته وأشرفت على طبعه: لجنة المجمع، مطبوعات المجمع العلمي العراقي- دار الشؤون الثقافية العامة- دار الرشيد، بغداد، ط1، (1398ه/ 1978م- 1402ه/ 1981م)، ‏3/‏144‏.

اغْتَمَطَ[لَازِم]

ن206هـ=821م

اغْتَمَطَ الشَّيْءُ: اخْتَفَى، وَتَوَارَى عَنِ الأَنْظَارِ.

"خَرَجَتْ شَاتُنَا، فَاغْتُمِطَتْ ، فَمَا رَأَيْنَا لَهَا أَثَرًا"
من رواية أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏1‏.

مَغْمُوط[اسْمُ مَفْعُول]

ن334هـ=945م

المَغْمُوطُ مِنَ النِّعْمَةِ وَنَحْوِهَا: المَجْحُودَةُ، المَكْفُورُ بِهَا.

نَعِمْنَا، فِي دِمَشْقٍ، نِعْـ مَةً لَيْسَتْ بِمَغْمُوطَهْ
الصَّنَوبريّ
ديوان الصّنوبريّ أحمد محمّد بن الحسن الضّبيّ: تحقيق: إحسان عبّاس، دار صادر، بيروت، ط2، (1998م)، ‏250‏.
895هـ=1489م

المَغْمُوطُ : المُغَطَّى، المُخْفَى.

وَكُلُّ مَنْ يَمْرَضُ فِيهَا، فَمَا لَهُ دَوَاءٌ غَيْرُ المَشَارِيطِ
قَدْ جَعَلَ الأَسَى دُرُورًا لَهُ كَنَائِمٍ فِي المَهْدِ مَغْمُوطِ
من رواية جلال الدّين السَّيوطي
كوكب الرّوضة في تاريخ النّيل وجزيرة الرّوضة: جلال الدّين السَّيوطي (ت، 911هـ)، تحقيق: محمَّد الشّشتاوي، دار الآفاق العربية، القاهرة، ط1، (1422ه/ 2002م)، ‏228‏.

غَامَطَ[مُتَعَدٍّ]

352هـ=963م

غَامَطَ الشَّخْصُ المَاءَ وَنَحْوَهُ: جَرَعَهُ جَرْعًا مُتَتَابِعًا.

"وَالغَمْجُ: جَرْعُ المَاءِ، وَهُوَ يُغَامِطُ المَاءَ، وَيُغَامِجُهُ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏65‏.

غَمُوط[صِيغَةُ مُبَالَغَة]

ن428هـ=1037م

الغَمُوطُ مِنَ الأَشْخَاصِ: الشَّدِيدُ الجَحْدِ لِلنِّعْمَةِ، وَالكُفْرِ بِهَا.

فَإِنِ اعْتَرَتْكُمْ هَفْوَةٌ، أَوْ صَدَّكُمْ غَضَبُ المُنِيلِ عَلَى الغَمُوطِ الكَافِرِ
وَرَأَيْتُمُ نَعْمَاءَكُمْ وَصَنِيعَكُمْ لَا فِي المُقِرِّ لَكُمْ، وَلَا فِي الشَّاكِرِ
فَلَكُمْ غَدًا أَيَّامُ وَصْلٍ، طُولُهَا مُوفٍ عَلَى اليَوْمِ القَصِيرِ الهَاجِرِ
مهيار الدّيلميّ
ديوان مِهْيار الدَّيْلميّ: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1344هـ/ 1925م- 1350هـ/ 1931م)، ‏2/‏74‏.

غِمَاط[مَصْدَر]

ن436هـ=1044م

غِمَاطُ النِّعْمَةِ وَنَحْوِهَا: جَحْدُهَا، وَالكُفْرُ بِهَا.

وَكَمْ نِعَمٍ لِجَدِّهِمُ عَلَيْكُمْ لَقِينَ بِكُمْ جُحُودًا، أَوْ غِمَاطَا
الشّريف المرتضى
ديوان الشريف المرتضى: شرح: محمّد ألتونجي، دار الجيل، بيروت، ط1، (1417هـ/ 1997م)، ‏2/‏217‏.

مُتَغَمِّط[صِفَةٌ مُشَبَّهَة]

503هـ=1109م

المُتَغَمِّطُ مِنَ المَاءِ وَنَحْوِهِ: الكَثِيرُ.

وَفِي كُلِّ فَضْلٍ بَحْرُهُ مُتَغَمِّطٌ تَحَارُ بِحَارُ العِلْمِ فِي لُجِّ بَحْرِهِ
فَلَمْ يَلْحَقِ العَوَّامُ نَازِحَ شَطِّهِ وَلَمْ يُدْرِكِ الغوَّاصُ غَامِضَ قَعْرِهِ
جار الله الزمخشريّ
ديوان جار الله الزّمخشريّ: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: فاطمة يُوسف الخيميّ، دار صادر، بيروت، ط1، (1429ه/ 2008م)، ‏277‏.

تَغَمُّط[مَصْدَر]

720هـ=1320م

تَغَمُّطُ الشَّيْءِ: مُلَازَمَتُهُ، وَالمُدَاوَمَةُ عَلَيْهِ بِلَا انْقِطَاعٍ.

أبو حيّان الأندلسيّ
البحر المحيط في تفسير القرآن العظيم: أبو حيّان الأندلسيّ (ت، 745هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز هجر، القاهرة، ط1، (1436ه/ 2015م)، ‏12/‏200‏.

تَغَمْطَمَ[لَازِم]

ن873هـ=1468م

تَغَمْطَمَ المَوْجُ وَنَحْوُهُ: كَثُرَ.

"يَجْرِي بِالزِّيَادَةِ وَالنُّقْصَانِ كَمَجَارِي الشَّمْسِ وَالقَمَرِ، لَهُ أَوَانٌ تَظْهَرُ إِلَيْهِ عُيُونُ الأَرْضِ وَيَنَابِيعُهَا مُسَخَّرَةً لَهُ... حَتَّى إِذَا أَظْلَمَ عَجَاجُه، وَتَغَمْطَمَتْ أَمْوَاجُه... نَكَصَ عَلَى عَقِبِه"
غرس الدين خليل بن شاهين الظاهري
كتاب زبدة كشف الممالك: غرس الدين خليل بن شاهين الظاهري (ت، 873هـ)، قد اعتنى بتصحيحه: بول راويس، المطبعة الجمهورية، باريس، (1894م)، ‏38‏.

مِغْمَاط[صِيغَةُ مُبَالَغَة]

ن1150هـ=1737م

المِغْمَاطُ مِنَ الأَشْخَاصِ: الغَمُوطُ.

بَيْضَاءُ تَخْتَالُ مَعْ وَصَائِفِهَا فِي السُّنْدُسِ الخُضْرِ وَالسِّقِلَّاطِ
لَيْسَتْ بِحَمَّالَةِ النَّمِيمِ، وَلَا بهَّاتَةٍ لِلْإِحْسَانِ مِغْمَاطِ
الحبسي
ديوان الحبسي: الحبسي (ت، 1150هـ)، تحقيق وتعليق : عبد العليم عيسى، وزارة التراث القومي والثقافة، مسقط، ط2، (1412ه/ 1992م)، ‏328‏-‏ 329‏.

تَغَمَّطَ[مُتَعَدٍّ بِالحَرْف]

1318هـ=1900م

تَغَمَّطَ عَلَيْهِ التُّرَابُ وَنَحْوُهُ: غَطَّاهُ حَتَّى قَتَلَهُ.

"ثُمَّ عِنْدَمَا كَوَّمُوا الحَطَبَ، وَكَادُوا يُضْرِمُونَ النَّارَ فِيهَا، إِذْ تَزَلْزَلَتِ الأَرْضُ؛ فَانْشَقَّتْ، وَتَغَمَّطَتْ عَلَى جُثَثِ القِدِّيسِينَ، فَسَتَرَتْهَا؛ لِئَلَّا يُحْرِقَهَا الأَثَمَةُ"
أدي شير الكلداني
كتاب سيرة أشهر شهداء المشرق القديسين: أدي شير الكلداني (ت، 1333هـ)، معهد التراث الكوردي، الموصل، ط1، (1900- 1906م)، ‏1/‏283‏.

إِغْمَاط[مَصْدَر]

1404هـ=1983م

إِغْمَاطُ الحَقِّ وَنَحْوِهِ: جَحْدُهُ، وَإِنْكَارُهُ.

"إِنَّ مَوْقِفَ أُورُبَّا مِنَ العَرَبِ، مُنْذُ نُزُولِ الوَحْيِ المُحَمَّدِيِّ، مَوْقِفٌ عِدَائِيٌّ بَعِيدٌ البُعْدَ كُلَّهُ عَنِ الإِنْصَافِ وَالعَدَالَةِ وَالتَّارِيخِ، وَقْتَذَاكَ كَانَ يُمْلَى، وَيُصْنَعُ... إِنَّ مِثْلَ هَذَا الوَضْعِ كَانَ مَفْهُومًا فِي عَصْرٍ كَانَ فِيهِ الشُّعُورُ السَّائِدُ هُوَ إِغْمَاطُ حَقِّ كُلِّ فَرْدٍ يُخَالِفُ الأُورُبِّيِّينَ عَقَائِدِيًّا"
فؤاد حسنين علي
طه حسين في ميزان العلماء والأدباء: محمود مهدي الإستانبولي (ت، 1420هـ)، المكتب الإسلامي، بيروت، ط1، (1403/ 1983م)، ‏539‏.
