مُنِيف[اسْمُ فَاعِل]

قبل 100ق.هـ=525م

المُنِيفُ : المُرْتَفِعُ، العَظِيمُ.

"إِنَّ أَبِي كَرِيمُ النِّزَال، مُنِيفُ المَقَال، كَثِيرُ النَّوَال، قَلِيلُ السُّؤَال، كَرِيمُ الفَعَال"
إحدى صواحب العَجْفاء بنت علقمة السَّعْديّة
مجمع الأمثال: أبو الفضل الميدانيّ (ت، 518هـ)، حققه وفصله وضبط غرائبه وعلق حواشيه: محمّد محيي الدين عبد الحميد، مطبعة السنة المحمدية، القاهرة، (1374هـ/ 1955م)، ‏2/‏135‏.
ن126هـ=744م

المُنِيفُ عَلَى الشَّيْءِ: الزَّائِدُ عَلَيْهِ.

إِذَا هُوَ أَمْسَى فِي عُبَابَيْ أَشِرَّةٍ مُنِيفًا عَلَى العَبْرَيْنِ بِالمَاءِ أَكْبَدَا
الكميت بن زيد الأسديّ
شعر الكُمَيْت بن زيد الأسدي: جمع وتقديم: داود سلوم، جامعة بغداد- مكتبة الأندلس، بغداد، مطبعة النعمان، جنيف، (1969م)، ‏1/‏164‏.

مُنِيف[صِفَةٌ مُشَبَّهَة]

ن360هـ=971م

المُنِيفُ عَلَى الشَّيْءِ: المُشْرِفُ عَلَيْهِ مِنْ مَكَانٍ عَالٍ.

"وَكُنَّا نَخْرُجُ لِلصَّيْدِ بِمِصْرَ فِي مَوْضِعٍ يُعْرَفُ بِدِيرِ القُصَيْرِ، مُنِيفٍ عَلَى ذِرْوَةِ الجَبَلِ المُقَطَّمِ"
كُشاجم
المصايد والمطارد: كُشاجم (ت، 360هـ)، حقّقه وعلّق عليه: محمّد أسعد طلس، دار اليقظة- مطبعة دار المعرفة، بغداد، د.ت. ‏6‏.

نِيَاف[صِفَة]

قبل 50ق.هـ=573م

النِّيَافُ مِنَ الأَشْيَاءِ: الطَّوِيلُ فِي ارْتِفَاعٍ.

نِيَافُ القُرْطِ غَرَّاءُ الثَّنَايَا وُرَيْدَاءُ الشَّبَابِ وَنِعْمَ خِيمُ
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏307‏.

أَنَافَ[لَازِم]

قبل 50ق.هـ=573م

أَنَافَ الشَّيْءُ: ارْتَفَعَ وَأَشْرَفَ.

إِذَا زُجِرَتْ أَلْفَيْتَهَا مُشْمَعِلَّةً تُنِيفُ بِعِذْقٍ مِنْ غِرَاسِ ابْنِ مُعْنِقِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏169‏.

أَنَافَ[مُتَعَدٍّ]

4ق.هـ=618م

أَنَافَ الشَّيْءَ: رَفَعَهُ.

يُنْبِي القُتُودَ بِمِثْلِ البُرْجِ مُتَّصِلًا مُؤَيَّدًا قَدْ أَنَافُوا فَوْقَهُ بَابَا
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏2/‏242‏.

أَنَافَ[مُتَعَدٍّ بِالحَرْف]

ن53هـ=673م

أَنَافَ عَلَى العَدَدِ وَنَحْوِهِ: زَادَ عَلَيْهِ.

"وَلَقَدْ أَفْنَى هَذَانِ اليَوْمَانِ ـ يَوْمُ الجَمَلِ وَصِفِّينَ ـ مَا يُنِيفُ عَلَى مِئَةِ أَلْفٍ، كُلُّهُمْ يَزْعُمُ أَنَّهُ طَالِبُ حَقٍّ"
زياد ابن أبيه
شرح نهج البلاغة: ابن أبي الحَديد (ت، 656هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية، القاهرة، منشورات مكتبة آية الله العظمى المرعشي النجفي- مؤسسة إسماعيليان، قم، ط2، (1383هـ/ 1963م- 1378هـ/ 1967م)، ‏16/‏185‏.

نَيَّفَ[مُتَعَدٍّ بِالحَرْف]

1ق.هـ=621م

نَيَّفَ عَلَى العَدَدِ: زَادَ عَلَيْهِ.

"إِنِّي قَدْ نَيَّفْتُ عَلَى التِّسْعِينَ، وَإِنَّمَا قَلْبِي بَضْعَةٌ مِنْ جِسْمِي، وَقَدْ نَحِلَ كَمَا نَحِلَ جِسْمِي"
أكثم بن صَيْفيّ الأُسَيِّديّ التّميميّ
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏5/‏224‏.

نَوْف[اسْم]

8هـ=629م

النَّوْفُ : بَظْرُ المَرْأَةِ.

لَعَمْرُكَ مَا أَدْرِي وَإِنِّي لَسَائِلُ مَهَانَةُ ذَاتِ النَّوْفِ أَلْأَمُ أَمْ سَعْدُ؟
حسّان بن ثابت الخزرجيّ الأنصاريّ
كتاب جُمَل من أنساب الأشراف: أحمد بن يحيى البَلاذُري (ت، 279هـ)، ج(2- 13)، حققه وقدم له: سهيل زكّار، رياض زركلي، إشراف: مكتب البحوث والدراسات، دار الفكر، بيروت، ط1، (1417هـ/ 1996م)، ‏11/‏20‏.
ن150هـ=767م

النَّوْفُ : سَنَامُ النَّاقَةِ وَنَحْوِهَا.

وَالرَّحْلُ فَوْقَ ذَاتِ نَوْفٍ جَامِسُ
زياد المِلقَطيّ
أراجيز المقلّين: تحقيق: محمّد يحيى زين الدّين، مجلّة مجمع اللُّغة العربيّة، دمشق، مج57، ج(1، 2)، 1982م، مج57، ج(1، 2)، 1982م، ج3، ج4، مج59، ج2، 1984م، ج4، مج68، ج2، 1993م، مج70، ج2، ‏272‏.

نَيِّف[اسْم]

13هـ=634م

النَّيِّفُ : مِقْدَارٌ مِنَ الأَعْدَادِ بَيْنَ الوَاحِدِ وَالثَّلَاثَةِ.

وَمَا كَانَ فِي صِهْرِ اليَمَامِيِّ رَغْبَةٌ وَلَوْ لَمْ يُصَبْ إِلَّا مِنَ النَّاسِ وَاحِدُ
فَكَيْفَ بِأَلْفٍ قَدْ أُصِيبُوا وَنَيِّفٍ عَلَى المَائِتَيْنِ اليَوْمَ أَوْ زَادَ زَائِدُ
حسّان بن ثابت الخزرجيّ الأنصاريّ
كتاب الرّدّة مع نبذة من فتوح العراق وذكر المُثنّى بن حارثة "الشَّيْبانيّ": الواقديّ (ت، 207هـ)، رواية: ابن أعثم الكوفيّ (ت، 314هـ)، تحقيق: يحيى الجبوري، دار الغرب الإسلاميّ، بيروت، ط1، (1410هـ/ 1990م)، ‏145‏.

مُنَيِّف[اسْمُ فَاعِل]

ن50هـ=670م

المُنَيِّفُ مِنَ الأَقْدَاحِ: المُمْتَلِئُ.

تَرَى تَحْتَهَا عُسَّ النُّضَارِ مُنَيِّفًا سَمَا فَوْقَهُ مِنْ بَارِدِ الغُزْرِ طَامِحُ
جُبيهاء الأشجعيّ
المفضليات: المفضَّل الضَّبّيّ (ت، 178هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السّلام هارون، دار المعارف، القاهرة، ط6، د.ت. ‏168‏.

نَافَ[مُتَعَدٍّ بِالحَرْف]

ن93هـ=712م

نَافَ عَلَى الأَمْرِ: عَلَا عَلَيْهِ وَزَادَ.

كَيْفَ اصْطِبَارِي عَنْ فَتَاةٍ طَفْلَةٍ بَيْضَاءَ فِي لَوْنٍ لَهَا ذِي زِبْرِجِ؟
نَافَتْ عَلَى العِذْقِ الرَّطِيبِ بِرِيقِهَا وَعَلَى الهِلَالِ المُسْتَبِينِ الأَبْلَجِ
عمر بن أبي ربيعة
شرح ديوان عمر بن أبي ربيعة المخزومي: تحقيق: محمّد محيي الدين عبد الحميد، المكتبة التجارية الكبرى- مطبعة السعادة، القاهرة، ط1، (1371هـ/ 1952م)، ‏480‏.

إِنَافَة[مَصْدَر]

ن103هـ=721م

الإِنَافَةُ : الرِّفْعَةُ وَالشَّرَفُ.

"بَغْدَادُ مَدِينَةُ السَّلَام، وَقُبَّةُ الإِسْلَام، مَعْدِنُ الخِلَافَة، وَمَعْقِلُ الإِنَافَة ، جَعَلَهَا اللهُ لِخَلِيفَتِهِ مَثْوَى، وَلِشِيعَتِهِ مَهْوَى"
يحيى بن وثّاب الأسديّ
البصائر والذّخائر: أبو حيّان التَّوحيديّ (ت، 414هـ)، تحقيق: وداد القاضي، دار صادر، بيروت، ط1، (1408هـ/ 1988م)، ‏9/‏10‏.
ن440هـ=1048م

الإِنَافَةُ عَلَى الأَمْرِ: الزِّيَادَةُ عَلَيْهِ.

الِإنَافَةُ عَلَى الثَّمَانِينَ أَفْسَدَتْ مِنَ المُتَخَيِّلَةِ قُوَّتَيْهَا العَمَلِيَّتَيْنِ"
أبو الرّيحان البيرونيّ
كتاب الصّيدنة في الطّبّ: أبو الرّيحان البيرونيّ (ت، 440هـ)، تصحيح ومقدمة وتحشية: عباس زرياب، مركز نشر دانشكاهي، طهران، ط1، (1370هـ)، ‏17‏.

مُتَنَايِف[صِفَةٌ مُشَبَّهَة]

ن135هـ=752م

المُتَنَايِفُ : المُرْتَفِعُ المُشْرِفُ عَلَى غَيْرِهِ.

"فَقَدِمْتُ عَلَيْهِ، وَقَدْ خَرَجَ مُبْتَدِيًا بِقَرَابَتِهِ وَأَهْلِهِ وَحَشَمِهِ، وَغَاشِيَتِهِ مِنْ جُلَسَائِهِ، فَنَزَلَ فِي أَرْضٍ قَاعٍ صَحْصَح، مُتَنَايِفٍ أَفْيَح"
خالد بن صفوان المِنْقَريّ التَّميميّ
المؤتلِف والمختلِف: أبو الحسن الدّارقطنيّ (ت، 385هـ)، دراسة وتحقيق: موفق بن عبد الله بن عبد القادر، دار الغرب الإسلاميّ، بيروت، ط1، (1406هـ/ 1986م)، ‏2/‏842‏.

نَيْف[صِفَةٌ مُشَبَّهَة]

161هـ=778م

النَّيْفُ مِنَ الأُمُورِ: الزَّائِدُ.

أَنْتَ ابْنُ أَكَّارٍ نَهِيجُ أَكَّارْ مُضْطَهَدُ الوَالِدِ نَيْفُ المِشْوَارْ
بشّار بن برد
ديوان بشار بن برد: قرأه وقدم له: إحسان عبّاس، دار صادر، بيروت، ط1، (2000م)، ‏348‏.

نَيَّاف[صِفَة]

ن175هـ=791م

النَّيَّافُ مِنَ الإِبِلِ وَنَحْوِهَا: الَّذِي ارْتَفَعَ فِي سَيْرِهِ.

يَتْبَعْنَ نَيَّافَ الضُّحَى عُزَاهِلَا
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏376‏.

تَنْيِيف[مَصْدَر]

ن198هـ=813م

التَّنْيِيفُ : التَّعْلِيَةُ.

لَهُ حِرَابٌ فَوْقَ قُفَّازِهِ يَجْمَعْنَ تَنْيِيفًا وَتَسْنِينَا
أبو نُواسٍ
كتاب التّشبيهات: أبو اسحاق ابن أبي عون (ت، 322هـ)، عني بتصحيحه: محمّد عبد المعيد خان، مطبعة جامعة كمبردج، لندن، (1369هـ/ 1950م)، ‏47‏.
ن672هـ=1273م

التَّنْيِيفُ : الزِّيَادَةُ.

"يُعْطَفُ العِشْرُونَ وَأَخَوَاتُهُ عَلَى النَّيِّفِ وَهُوَ إِنْ قَصَدَ التَّعْيِينَ وَاحِدٌ أَوْ أَحَدٌ ... وَإِنْ لَمْ يَقْصِدِ التَّعْيِينَ فِيهِمَا فَبِضْعَةٌ وَبِضْعٌ، وَيُسْتَعْمَلَانِ -أَيْضًا- دُونَ تَنْيِيفٍ "
ابن مالك
شرح التسهيل تسهيل الفوائد وتكميل المقاصد: ابن مالك (ت، 672هـ)، تحقيق: محمد عبد القادر عطا، طارق فتحي السيد، دار الكتب العلمية، بيروت، ط1، (1422ه/ 2001م)، ‏2/‏314‏.

مُنَوَّف[صِفَةٌ مُشَبَّهَة]

ن206هـ=821م

المُنَوَّفُ مِنَ الإِبِلِ: العَظِيمُ المُرْتَفِعُ.

وَقَدْ أُعَنِّي الأَرْحَبِيَّ المُشْنِفَا ذَا الغَيَفَانِ السَّلِسَ المُنَوَّفَا
دُكين الطّائيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏10‏.

نِيَافَة[اسْم]

228هـ=843م

النِّيَافَةُ : الشَّرَفُ وَالرِّفْعَةُ.

فَاتْرُكْ، جُعِلْتُ فِدَاكَ، إِكْرَامِي بِمَا أَخْشَى بِهِ عِنْدَ الوَرَى اسْتِغْبَائِي
فَالعَيْنُ تَصْغُرُ إِنْ تُقَدِّمْهَا عَلَى أَوْلَادِ حَرْبِ السَّادَةِ الكُبَرَاءِ
حَلُّوا مِنَ العِزِّ المَنِيعِ نِيَافَةً يَحْمُونَ غَيْرَهُمُ ذُرَى العَلْيَاءِ
عمرو بن عثمان القينيّ
أدب الكُتّاب: أبو بكر الصّوليّ (ت، 335هـ)، نسخه وعني بتصحيحه وتعليق حواشيه: محمّد بهجة الأثري، نظر فيه: السيد محمود شكري الآلوسي، المكتبة العربية، بغداد، المطبعة السلفية، القاهرة، (1341هـ)، ‏159‏.

نِيَافِيّ[صِفَة]

352هـ=963م

النِّيَافِيُّ مِنَ الإِبِلِ وَنَحْوِهَا: الطَّوِيلُ المُشْرِفُ.

لَمَّا رَأَيْتُ الهَمَّ قَدْ أَجْفَانِي قَرَّبْتُ لِلرَّحْلِ وَلِلظِّعَانِ كُلَّ نِيَافِيِّ القَرَا ضُوبَانِ
من شواهد الأزهريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏12/‏83‏.

نَائِف[صِفَةٌ مُشَبَّهَة]

ن906هـ=1500م

النَّايِفُ : العَالِي المُرْتَفِعُ.

وَهْوَ لَهُ بَحْرِيُّهُ جَزَائِرُ مِمَّا يَلِي الفَالَ لَهَا أَمَايِرُ أَيْضًا، وَأَفَّشَاتُ مَعْ شُعُوبِ وَكَوْنُهَا عَنْهُ إِلَى الجَنُوبِ اثْنَيْ عَشَرْ زَامًا أَيَا مُعَلِّمَا جُزْرٌ كِبَارٌ نَايِفَاتٌ لِلسَّمَا
شهاب الدّين أحمد بن ماجد النجديّ
حاوية الاختصار في أصول علم البحار: شهاب الدّين أحمد بن ماجد النجديّ (ت، 906هـ)، ضمن: أحمد بن ماجد (منظر الملاحة الفلكية في المحيط الهندي وبحاره الشاطئية في القرن التاسع الهجري/ الخامس عشر الميلادي)، تحقيق: إبراهيم خوري، ج2، مركز الدراسات والوثائق في الديوان الأميري، رأس الخيمة، (د.ت)، (9- 88)، ‏2/‏41‏.
1330هـ=1911م

النَّائِفُ مِنَ الأَشْيَاءِ: الجُزْءُ القَلِيلُ مِنْهُ.

وَهَكَذَا السَّارِقُ لِلثِّمَارِ إِنْ كَانَ فِي النَّخْلِ أَوِ الأَشْجَارِ
وَإِنْ تُوَارِي ذَلِكَ الحُصُونُ فَالقَطْعُ فِيهِ لَازِمًا يَكُونُ
وَمَنْ تَعَاطَى نَائِفًا مِنَ الثَّمَرْ لَا قَطْعَ فِيهِ إِنْ يَكُنْ خَلْفَ الجُدُرْ
نور الدين أبو محمد عبد الله بن حميد السالمي
جوهر النظام في علمي الأديان والأحكام: نور الدين أبو محمد عبد الله بن حميد السالمي (ت، 1331هـ)، دار الكتب القطرية، قطر، ط12، (1413ه/ 1993م)، ‏3/‏206‏.
