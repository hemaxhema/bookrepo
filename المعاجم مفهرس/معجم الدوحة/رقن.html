أَرْقَن[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الأَرْقَنُ : المُتَلَطِّخُ بِالدَّسَمِ وَنَحْوِهِ.

يَرُوحُ غِلْمَانُنَا دُسْمًا مَشَافِرُهُمْ رُقْنًا بِأَيْدِيهِمُ الأَحْرَادُ وَالسَّدَفُ
الأفوه الأوديّ
شُعراء مَذْحِج أخبارهم وأشعارهم في الجاهلية: صنعة: مُقبل التَّامّ عامر الأحمدي، مَجمَع العربيَّة السَّعيدة، صنعاء، ط2، (1435هـ/ 2014م)، ‏398‏.

إِرْقَان[اسْم]

قبل 13ق.هـ=609م

الإِرْقَانُ : نَبَاتٌ مِنْ نَحْوِ الحِنَّاءِ وَالزَّعْفَرَانِ.

فَيَتْرُكُ القِرْنَ مُصْفَرًّا أَنَامِلُهُ كَأَنَّ فِي رَيْطَتَيْهِ نَضْخَ إِرْقَانِ
أبو المُثَلِّم الهُذَليّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏23/‏8‏.

مُرْتَقِن[صِفَة]

قبل 13ق.هـ=609م

المُرْتَقِنُ بِالدَّمِ وَنَحْوِهِ (المُرْتَقِنُّ: ضَرُورَةٌ): المُتَلَطِّخُ بِهِ.

دَعَا لِحِبَائِهِ عَمْرُو بْنُ هِنْدٍ لِأَضْرِبَ رَأْسَ حَارِثَةَ بْنِ عَمْرِو
فَقُلْتُ لَهُ: عَلَيْكَ بِمُرْتَقِنٍّ وَلُوغٍ فِي دِمَاءِ سَرَاةِ بَكْرِ
الكيِّس النَّمَريّ
كتاب الأنوار ومحاسن الأشعار: الشِّمْشاطيّ (ت، 377هـ)، تحقيق: السيد محمّد يوسف، راجعه وزاده في حواشيه: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، (1397هـ/ 1977م)، ‏1/‏226‏-‏ 227‏.

رَاقِن[اسْمُ فَاعِل]

ن10هـ=631م

الرَّاقِنُ : الحَسَنُ اللَّوْنِ، المُخْتَضِبُ بِالزَّعْفَرَانِ وَنَحْوِهِ.

فَأَبَارَ جَمْعَهُمُ السُّيُوفُ، وَأَبْرَزُوا عَنْ كُلِّ رَاقِنَةٍ تُجَرُّ، وَتُسْلَبُ
ساعدة بن جُؤَيّة الهُذَليّ
شعر ساعدة بن جُؤيّة الهُذليّ دراسة وتحقيق: إعداد: ميساء قتلان، رسالة ماجستير، كلية الآداب، جامعة دمشق، (1424هـ/ 2003م)، ‏171‏.
1378هـ=1958م

الرَّاقِنُ مِنَ الآلَاتِ وَنَحْوِهَا: الَّذِي تُحَوَّلُ بِهِ النُّصُوصُ المَكْتُوبَةُ بِاليَدِ وَالمَنْطُوقَةُ إِلَى مَطْبُوعَةٍ.

"يُشْتَرَطُ فِي التَّآلِيفِ المُقَدَّمَةِ أَلَّا تَكُونَ قُدِّمَتْ لِأَيِّ مُبَارَاةٍ سَابِقَةٍ، وَأَنْ تَكُونَ مَكْتُوبَةً عَلَى الآلَةِ الرَّاقِنَةِ ، فِي عَشْرِ نُسَخٍ"
مجلة دعوة الحقّ
جائزة قدماء الصادقية ((الأدبية)): ‏ـ،‏مجلة دعوة الحق: الرباط، س2، ع1، (1378هـ/1958م)، ‏69‏.
ن1392هـ=1971م

الرَّاقِنُ : الكَاتِبُ عَلَى آلَةِ الكِتَابَةِ وَنَحْوِهَا.

"إِنَّ الَّذِي يَكْتُبُ بِالحُرُوفِ اللَّاتِنِيَّةِ يُنْجِزُ 17 صَفْحَةً مِنْ حَجْمِ مَجَلَّةِ الفِكْرِ، بَيْنَمَا الرَّاقِنُ بِالعَرَبِيَّةِ لَا يُنْجِزُ إِلَّا 7 صَفَحَاتٍ فِي نَفْسِ المُدَّةِ"
البشير بن سلامة
اللغة العربية ومشاكل الكتابة: البشير بن سلامة، الدار التونسيه للنشر، نونس 1971م، ‏79‏.

مُرَقِّن[اسْمُ فَاعِل]

126هـ=744م

المُرَقِّنُ مِنَ الكُتَّابِ: المُزَيِّنُ لِكِتَابَتِهِ.

دَارٌ كَرَقْمِ الكَاتِبِ المُرَقِّنِ بَيْنَ نَقَى المُلْقَى وَبَيْنَ الأَجْؤُنِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏81‏.

رَقِين[صِفَةٌ مُشَبَّهَة]

ن130هـ=748م

الرَّقِينُ : المَصْبُوغُ بِالزَّعْفَرَانِ وَنَحْوِهِ.

كَأَنَّ بِخَدِّهِ وَالجِيدِ مِنْهُ مِنَ الجِرْيَالِ مَحْلُوبًا رَقِينَا
ابن الدُّمَينة
ديوان ابن الدُّمَينة: صنعة: أبي العبّاس ثعلب، ابن حَبيب البغداديّ، تحقيق: أحمد راتب النفاخ، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1959م)، ‏152‏.

رَقِين[اسْم]

352هـ=963م

الرَّقِينُ : الدِّرْهَمُ.

أَلَا رُبَّ مُلْتَاثٍ يَجُرُّ كِسَاءَهُ نَفَى عَنْهْ وُجْدَانَ الرَّقِينِ القَرَائِمَا
ثُمامة بن المخبر/المُحَيِّر السَّدوسيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏15/‏129‏.
ن1351هـ=1932م

الرَّقِينُ : اللَّوْحُ المَنْقُوشُ بِالكِتَابَةِ وَنَحْوِهَا.

وَبِكُلِّ رُكْنٍ صُورَةٌ وَبِكُلِّ زَاوِيَةٍ رَقِينْ
أحمد شوقي
الشوقيات: أحمد شوقي، مؤسسة هنداوي، القاهرة، جمهورية مصر ، (2012م)، ‏2/‏471‏.

تَرْقِين[مَصْدَر]

136هـ=753م

تَرْقِينُ الكِتَابَةِ وَنَحْوِهَا: تَزْيِينُهَا.

إِذَا تَهَجَّى قَارِئٌ يُهَيْنِمُهْ أَخْرَجَ أَسْمَاءَ البَيَانِ مُعْجَمُهْ وَحَلَقُ التَّرْقِينِ أَوْ مُوَشَّمُهْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏219‏-‏ 220‏.
ن175هـ=791م

تَرْقِينُ الثَّوْبِ: صَبْغُهُ بِالزَّعْفَرَانِ وَالوَرْسِ.

" وَتَرْقِينُ الثَّوْبِ بِالزَّعْفَرَانِ وَالوَرْسِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏143‏.
ن387هـ=997م

التَّرْقِينُ: تَسْوِيدُ مَوْضِعٍ فِي جَدْوَلٍ أَوْ فِي وَثِيقَةٍ، أَوْ رَسْمُ خَطٍّ فِيهِ لِيُعْلَمَ أَنَّهُ خَالٍ مِنَ العَدَدِ. [الرّياضيّات]

" التّرْقِينُ خَطٌّ يُخَطُّ فِي التَّأْرِيجِ أَوِ العَرِيضَةِ، إِذَا خَلَا بَابٌ مِنَ السَّطْرِ؛ لِكَيْ يَكُونَ التَّرْتِيبُ مَحْفُوظًا بِهِ، وَهُوَ بَمَنْزِلَةِ الصِّفْرِ فِي حِسَابِ الهِنْدِ وَحِسَابِ الجُمَّلِ، وَاشْتِقَاقُهُ مِنْ: رِقَان، وَهِيَ بِالنَّبَطِيَّةِ: الفَارِغُ"
محمد بن أحمد البَلْخيّ الخوارزميّ
مفاتيح العلوم: محمد بن أحمد البَلْخيّ الخوارزميّ (ت، 387هـ)، حققه: إبراهيم الأبياريّ، دار الكتاب العربي، بيروت، ط2، (1409هـ/ 1989م)، ‏83‏.
1301هـ=1883م

تَرْقِينُ القَيْدِ وَنَحْوِهِ: إِبْطَالُهُ وَمَحْوُهُ.

"تُرْسَلُ بَرَاءَاتُهُمْ إِلَى خَزِينَةِ المَالِيَّةِ الجَلِيلَةِ؛ لِأَجْلِ تَرْقِينِ قُيُودِهَا فِي الخَزِينَةِ الجَلِيلَةِ"
نوفل بن نعمة الله الطرابلسي
الدستور: نوفل بن نعمة الله الطرابلسي (ت، 1304هـ)، ترجمة: نوفل بن نعمة الله الطرابلسي، مراجعة وتدقيق: خليل أفندي الخوري، المطبعة الأدبية، بيروت، (1301ه/ 1883م)، ‏2/‏705‏.

رَقُون[اسْمُ مَفْعُول]

167هـ=783م

الرَّقُونُ مِنَ الأَيْدِي وَنَحْوِهَا: المخْضُوبُ بِالحِنَّاءِ.

تَذَكَّرْتَ تُكْنَى وَإِهْلَاسَهَا فَلَمْ تَنْسَ، وَالشَّوْقُ ذُو مَطْرُؤَهْ
سَلَامًا بِرَخْصٍ لَهُ بَهْجَةٌ وَكَفٍّ رَقُونٍ ، لَهَا مَحْنُؤَهْ
أبو حزام العُكْليّ
شرح سِقط الزّند: ابن السيّد البَطَلْيَوسيّ (ت، 521هـ)، ضمن: شُروح سقط الزّند، تحقيق: مُصطفى السَّقا وآخرين، إشراف: طه حسين، الهيئة المصريَّة العامَّة للكتاب، القاهرة، ط3، (1406ه/ 1986م)، ‏4/‏126‏.

رَقْن[مَصْدَر]

1408هـ=1987م

رَقْنُ النُّصُوصِ وَنَحْوِهَا: طِبَاعَتُهَا.

"وَقَدْ كُنْتُ قُمْتُ أَوَّلًا بِنَسْخِ قِسْمٍ هَامٍّ مِنَ النُّسْخَةِ المَدَنِيَّةِ، وُقُدِّمَ لِلرَّقْنِ "
محمد الشّاذليّ النّيفر
المعْلم بفوائد مسلم: المازَرِي (ت، 536هـ)، تقديم وتحقيق: محمد الشاذلي النيفر، الدار التونسية- بيت الحكمة، تونس، المؤسسة الوطنية للكتاب، الجزائر، ط2، (1987م)، ‏2/‏258‏.

مُرْقَن[اسْمُ مَفْعُول]

ن206هـ=821م

المُرْقَنُ مِنَ الثِّيَابِ وَنَحْوِهَا: المَصْبُوغُ المُشْبَعُ بِالرِّقَانِ.

كَأَنَّ كِلْتَيْهِمَا فِي مِمْطَرٍ خَلَقٍ وَجَيْبُهُ مُرْقَنٌ فِي صِبْغِ شَوْرَانِ
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏143‏.

ارْتَقَنَ[لَازِم]

ن231هـ=845م

ارْتَقَنَ بِالزَّعْفَرَانِ وَنَحْوِهِ: اخْتَضَبَ، وَتَضَمَّخَ.

غِيَاثُ، إِنْ مُتُّ، وَعِشْتَ بَعْدِي وَأَشْرَفَتْ أُمُّكَ لِلتَّصَدِّي وَارْتَقَنَتْ بِالزَّعْفَرَانِ الوَرْدِ فَاضْرِبْ، فِدَاكَ وَالِدِي وَجَدِّي ضَرْبَةَ لَا وَانٍ وَلَا ابْنِ عَبْدِ
من إنشاد أبي عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏9/‏96‏.

مُتَرَقِّن[اسْمُ فَاعِل]

ن275هـ=888م

المُتَرَقِّنُ بِالزَّعْفَرَانِ وَنَحْوِهِ: المُتَضَمِّخُ بِهِ.

"ثَلَاثَةٌ لَا تَقْرَبُهُمُ المَلَائِكَةُ بِخَيْرٍ: جَنَازَةُ الكَافِرِ، وَالمُتَرَقِّنُ بِالزَّعْفَرَانِ، وَالجُنُبُ حَتَّى يَغْتَسِلَ"
أبو عوانة
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏3/‏1121‏.

"كَأَنَّمَا تَرَقَّنُوا فِي الحَرْبِ بِالدَّمِ الحَالِكِ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏14/‏148‏.
1301هـ=1883م

تَرَقَّنَ القَيْدُ وَنَحْوُهُ: أُبْطِلَ، وَمُحِيَ.

"كَمَا أَنَّه يَنْبَغِي أَنْ يَتَرَقَّنَ قَيْدُهَا الكَائِنُ فِي دَفَاتِرِ الأَوْقَافِ"
نوفل بن نعمة الله الطرابلسي
الدستور: نوفل بن نعمة الله الطرابلسي (ت، 1304هـ)، ترجمة: نوفل بن نعمة الله الطرابلسي، مراجعة وتدقيق: خليل أفندي الخوري، المطبعة الأدبية، بيروت، (1301ه/ 1883م)، ‏1/‏95‏.

رِقَان[اسْم]

ن366هـ=976م

الرِّقَانُ : الزَّعْفَرَانُ.

وَمُسْمِعَةٍ إِذَا مَا شِئْتُ غَنَّتْ مُضَمَّخَةِ التَّرَائِبِ بِالرِّقَانِ
من شواهد السَّرِيّ الرَّفّاء الكنديّ
المحبّ والمحبوب والمشموم والمشروب: السَّرِيّ الرَّفّاء الكنديّ (ت، 366هـ)، تحقيق: مصباح غلاونجي، مطبوعات مجمع اللغة العربية، دمشق، (1407هـ/ 1986م)، ‏3/‏187‏.

مَرْقُون[اسْمُ مَفْعُول]

1391هـ=1970م

المَرْقُونُ مِنَ النُّصُوصِ: المَكْتُوبُ بِوَاسِطَةِ الآلَةِ الكَاتِبَةِ وَنَحْوِهَا.

"وَتَجْدُرُ الإِشَارَةُ إِلَى أَنَّ صَحَائِفَ عَدِيدَةً، مِنْ هَذَا الأَصْلِ، مَرْقُونَةٌ بِحِبْرٍ بَاهِتٍ"
أحمد المعداوي المجاطي
ظاهرة الشعر الحديث: أحمد المعداوي المجاطي (ت، 1415هـ)، مراجعة وتقديم: نجيب العوفي، شركة النشر والتوزيع المدارس، الدار البيضاء، ط1، (1423ه/ 2002م)، ‏8‏.

رَقَنَ[لَازِم]

1391هـ=1970م

رَقَنَ الشَّخْصُ: اخْتَضَبَ بِالحِنَّاءِ أَوِ الزَّعْفَرَانِ.

"وَقَدْ يَرْقُنُ الرَّجُلُ كَمَا تَرْقُنُ المَرْأَةُ بِالحِنَّاءِ وَالزَّعْفَرَانِ"
جواد علي
المفصل فى تاريخ العرب قبل الإسلام: جواد علي (ت، 1407هـ)، جامعة بغداد، بغداد، ط2، (1413ه/ 1993م)، ‏4/‏615‏.

رَقَّنَ[مُتَعَدٍّ]

1408هـ=1987م

رَقَّنَ القَيْدَ وَنَحْوَهُ: أَبْطَلَهُ، وَمَحَاهُ.

"وَلَا يَتَحَلَّلُ المُؤَمِّنُ لَهُ فِي كُلِّ مَا سَبَقَ مِنْ هَذَا الِالْتِزَامِ بِدَفْعِ القِسْطِ السَّنَوِيِّ إِلَّا إِذَا تَلِفَتْ مَرْكَبَتُهُ تَلَفًا كُلِّيًا... وَرُقِّنَ قَيْدُهَا فِي دَوَائِرِ المُرُورِ أَوْ غَيْرِهَا"
شُكْر محمّد أحمد
شرح في قانون التأمين الإلزامي من حوادث السير في العراق: ‏شكر محمد أحمد،‏تنمية الرافدين: تنمية الرافدين، جامعة الموصل- كلية الإدارة والاقتصاد، الموصل، مج9، ع21، (سبتمبر- 1987م)، ‏86‏.
