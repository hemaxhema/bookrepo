تَكَاشَّ[لَازِم]

قبل 150ق.هـ=476م

تَكَاشَّتِ الأَفْعَى: صَوَّتَ جِلْدُهَا، إِذَا حَكَّتْ بَعْضَهَا بِبَعْضٍ.

"نَعَمْ بِرُحْبِ ذِرَاع، وَهُوَ أَبُو الرِّبَاع، تَكَاشُّ مِنْ حِسِّهِ الأَفَاع"
ابنة الخُسّ الإياديّة
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏6/‏397‏.
ن175هـ=791م

تَكَاشَّتِ الأَفَاعِي فِي البَلَدِ: أَصَابَهُ الجَدْبُ.

"وَبَلَدٌ تَكَاشُّ أَفَاعِيهِ : يُوصَفُ بِالمَحْلِ وَالجَدْبِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏269‏.

كَشَّة[مَصْدَر]

قبل 50ق.هـ=573م

كَشَّةُ الأَفْعَى: تَصْوِيتُ جِلْدِهَا، إِذَا احْتَكَّ بَعْضُهَا بِبَعْضٍ.

وَهَلْ عَلِمْتِ، يَا قُفَيِّ التَّتْفُلَهْ وَكَشَّةَ الأَفْعَى، وَنَفْخَ الأَصَلَهْ أَنَّي أُفِيتُ المِئَةَ المُؤَبَّلَهْ؟
صُحير بن عُمير التَّميميّ
الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏236‏.

كَشَّة[اسْم]

ن304هـ=916م

الكَشَّةُ : الصَّوْتُ.

"لَا تُسْمَعُ لِضَرِيبَتِهِ كَشَّةٌ "
أبو محمّد الأنباريّ
ديوان المفضّليّات: المفضَّل الضَّبّيّ (ت، 178هـ)، شرح: أبي محمّد القاسم بن محمّد بن بشّار الأنباريّ، تحقيق: كارلوس يعقوب لايل، مطبعة الآباء اليسوعيّين، بيروت، كلارندون، أكسفورد، (1920م)، ‏176‏.

كَشِيش[مَصْدَر]

قبل 13ق.هـ=609م

كَشِيشُ الأَفْعَى: كَشَّتُهَا.

وَقَدْ كَانَتْ يَكُونُ لَهَا كَشِيشٌ وَأَحْيَانًا يَكُونُ لَهَا وِثَابُ
الزُّبير بن عبد المُطَّلِب القُرَشيّ
ديوان الزبير بن عبد المُطَّلب: أبو الطاهر القرشي : تح: الحسين الهاشمي، العربيَّة الثَّقافيَّة، (1431هـ)، ‏9‏.
ن40هـ=660م

كَشِيشُ الضَّبِّ وَنَحْوِهِ: تَصْوِيتُهُ.

"وَكَأَنِّي أَنْظُرُ إِلَيْكُمْ تَكِشُّونَ كَشِيشَ الضِّبَابِ، لَا تَأْخُذُونَ حَقًّا، وَلَا تَمْنَعُونَ ضَيْمًا"
عليّ بن أبي طالب
شرح نهج البلاغة: ابن أبي الحَديد (ت، 656هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية، القاهرة، منشورات مكتبة آية الله العظمى المرعشي النجفي- مؤسسة إسماعيليان، قم، ط2، (1383هـ/ 1963م- 1378هـ/ 1967م)، ‏7/‏304‏.
ن110هـ=728م

كَشِيشُ البَعِيرِ: صَوْتُهُ دُونَ الهَدِيرِ.

أَمْسَتْ طُهَيَّةُ كَالبِكَارِ أَفَزَّهَا، بَعْدَ الكَشِيشِ ، هَدِيرُ قَرْمٍ بَازِلِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏580‏.

كَشِيش[صِفَةٌ مُشَبَّهَة]

105هـ=723م

الكَشِيشُ مِنَ الشَّرَابِ وَنَحْوِهِ: الِّذِي يُصْدِرُ صَوْتًا عِنْدَ أَوَّلِ الغَلَيَانِ.

لَا تَحْسَبَنَّ مِرَاسَ الحَرْبِ إِذْ لَقِحَتْ شُرْبَ الكَشِيشِ وَأَكْلَ الخُبْزِ بِالصِّيرِ
جرير
الموشَّح مآخذ العلماء على الشعراء في عدة أنواع من صناعة الشعر: أبو عُبيد الله المَرْزُبانيّ (ت، 384هـ)، تحقيق: علي محمّد البجاوي، نهضة مصر، القاهرة، د.ت. ‏158‏.

كَشِيش[اسْم]

1264هـ=1847م

الكَشِيشُ : الكُشَّةُ.

"أَيُّهَا الإِخْوَان، إِلَامَ وَحَتَّى مَتَى الشَّنَآن؟ تَسْعَوْنَ إِلَى انْحِصَاصِ رِيشي، وَلَا أَنْسَابُ إِلَيْكُمْ بِكَشِيشي "
ابن رزيق حميد بن محمد بن رزيق النخلي
الفتح المبين في سيرة السادة البُوسَعيديين: ابن رزيق حميد بن محمد بن رزيق النخلي (ت، 1291هـ)، تحقيق: محمد حبيب صالح، محمود بن مبارك السليمي، وزارة التراث القومي والثقافة، مسقط، ط6، (1437ه/ 2016م)، ‏1/‏438‏.

كَشَّ[لَازِم]

ن10هـ=631م

كَشَّ الشَّخْصُ: ضَحِكَ.

وَلَوْ أَنَّ شَيْخًا ذَا بَنِينَ، كَأَنَّمَا عَلَى رَأْسِهِ، مِنْ شَامِلِ الشَّيْبِ، قَوْنَسُ
وَقَدْ فَنِيَتْ أَضْرَاسُهُ، غَيْرَ وَاحِدٍ رَمِيمٍ، إِذَا مَا مُسَّ يُدْمَى، وَيَضْرِسُ
لَظَلَّ النَّهَارَ رَانِيًا، وَكَأَنَّهُ، إِذَا كَشَّ ، ثَوْرٌ مِنْ كَرِيصٍ مُنَمِّسُ
المُزَرِّد بن ضرار الغَطَفانيّ
ديوان المُزرِّد بن ضِرار الغطفاني: رواية: ابن السكيت وغيره، شرح: ثعلب، عني بتحقيقه: خليل إبراهيم العطية، قدم له: محمّد رضا الشبيبي، مطبعة أسعد، بغداد، ط1، (1382هـ/ 1962م)، ‏62‏.
ن40هـ=660م

كَشَّ الضَّبُّ وَنَحْوُهُ: صَوَّتَ.

"وَكَأَنِّي أَنْظُرُ إِلَيْكُمْ تَكِشُّونَ كَشِيشَ الضِّبَابِ، لَا تَأْخُذُونَ حَقًّا، وَلَا تَمْنَعُونَ ضَيْمًا"
عليّ بن أبي طالب
شرح نهج البلاغة: ابن أبي الحَديد (ت، 656هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية، القاهرة، منشورات مكتبة آية الله العظمى المرعشي النجفي- مؤسسة إسماعيليان، قم، ط2، (1383هـ/ 1963م- 1378هـ/ 1967م)، ‏7/‏304‏.
ن54هـ=674م

كَشَّ البَعِيرُ: صَاتَ دُونَ الهَدِيرِ.

إِذَا سَمِعُوا الشَّرَّ آدَوْا لَهُ بِكَارٌ تَكِشُّ ، وَلَا تَضْرِبُ
حسّان بن ثابت الخزرجيّ الأنصاريّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏276‏.
ن153هـ=770م

كَشَّتِ القِدْرُ وَنَحْوُهَا: صَوَّتَتْ عِنْدَ أَوَّلِ الغَلَيَانِ.

يَا حَشَرَاتِ القَاعِ مِنْ جُلَاجِلِ قَدْ نَشَّ مَا كَشَّ مِنَ المَرَاجِلِ
من شواهد نهْشل بن زيْد
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏6/‏397‏.
ن175هـ=791م

كَشَّتِ الأَفْعَى: تَكَاشَّتْ.

" وَكَشَّتِ الأَفْعَى تَكِشُّ كَشِيشًا، إِذَا احْتَكَّتْ سَمِعْتَ لِجِلْدِهَا مِثْلَ جَرْشِ الرَّحَى"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏269‏.

كَشَّ[مُتَعَدٍّ]

1223هـ=1807م

كَشَّهُ : طَرَدَهُ، وَأَبْعَدَهُ.

وَلَكَمْ قَدْ كَشَّ خَطْبًا دَهَمَا بِذَكَا فَرْزٍ، وَقَلْبٍ حَمِسِ!
نقولا بن يوسف الترك الإسطمبولي
ديوان المعلم نقولا الترك: نقولا بن يوسف الترك الإسطمبولي (ت، 1244هـ)، ضبط نصوصه ووضع مقدمته وفهارسه: فؤاد أفرام البستاني، وزارة التربية الوطنية والفنون الجميلة، بيروت، (1949م)، ‏394‏.
1441هـ=2019م

كَشَّ الحَمَامَ: طَيَّرَهُ فِي الجَوِّ فِي مَجْمُوعَةٍ كَبِيرَةٍ؛ لِجَلْبِ حَمَامٍ آخَرَ مَعَهَا.

"كَانَ جُنُودُ دَارِ الضُّبَّاطِ يَكِشُّونَ الحَمَامَ، وَيَنْثُرُونَ لَهُ قُشُورَ الخُبْزِ"
تحسين رزاق عزيز
صبية من متروبول، لليودميلا بيتروشيفسكايا: تحسين رزاق عزيز ، ترجمة: تحسين رزاق عزيز، دار المدى، بغداد- بيروت- دمشق، ط1، (2019م)، ‏72‏.

تَكْشَاش[مَصْدَر]

136هـ=753م

تَكْشَاشُ البَعِيرِ: صَوْتُهُ دُونَ الهَدِيرِ.

أَعْلُو وَعِرْضِي لَيْسَ بِالمُمَشَّغِ بِالهَدْرِ تَكْشَاشَ البِكَارِ المُرَّغِ لِلَّهِ دَرُّ المُحْدَثِينَ النُّبَّغِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏55‏.

فَيَا لَيْتَ شِعْرِي، هَلْ رَأَتْ أُمُّ جَحْدَرٍ أَكُشَّكَ ، أَوْ ذَاقَتْ مَغَابِنَكَ القُشْرَا؟

كَشُّ البَعِيرِ: صَوْتُهُ دُونَ الهَدِيرِ.

" وَالكَشُّ لِأَفْتَاءِ الإِبِلِ"

كَشُّ عَصِيرِ العِنَبِ وَنَحْوِهِ: تَصْوِيتُهُ عِنْدَ أَوَّلِ الغَلَيَانِ.

"فَإِنْ لَمْ يُطْبَخِ العَصِيرُ، وَتُرِكَ لِيَسْتَحْكِمَ، فَأَوَّلُ غَلَيَانِهِ النَّشُّ وَالكَشُّ "

1348هـ=1929م

الكَشُّ : الطَّرْدُ، وَالإِبْعَادُ.

"لَمْ يَبْقَ فِي عَكَّا الآنَ إِلَّا ثَلَاثَةُ حَوَانِيتَ يَهُودِيَّةٍ، وَلَا عَمَلَ لِأَصْحَابِهَا إِلَّا كَشَّ الذُّبَابِ، وَالنَّوْمَ"
جريدة اليرموك
لجنة إعانة المنكوبين العرب: ‏رئيس التّحرير،‏جريدة اليرموك: اليرموك، حيفا، س6، ع353، (08/ 10/ 1929م)، ‏3‏.

كَشَّاش[صِيغَةُ مُبَالَغَة]

ن283هـ=896م

الكَشَّاشُ : الشَّدِيدُ التَّصْوِيتِ.

لَهُ سَحِيفٌ لَدَى مَزَاحِفِهِ يُجِيبُ، مِنْهُ، كَشِيشَ كَشَّاشِ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏1257‏.
1398هـ=1977م

الكَشَّاشُ الحَمَامَ: الكَثِيرُ التَّطْيِيرِ لَهُ فِي الجَوِّ فِي مَجْمُوعَةٍ كَبِيرَةٍ؛ لِجَلْبِ حَمَامٍ آخَرَ مَعَهُ.

"حَلَّتْ مَحَلَّهَا رُوحٌ أُخْرَى كَثِيرَةُ التَّلَفُّتِ، مَشْغُولَةٌ بِأَلْفِ شَاغِلٍ، دَائِبَةُ الحَرَكَةِ، عَيْنُهَا دَائِمًا إِلَى الفَضَاءِ، مِثْلُ كَشَّاشِي الحَمَامِ"
حسيب كيالي
الأعمال الروائية والقصصية (الجزء الثاني): حسيب كيالي، منشورات الهيئة العامة السورية للكتاب، دمشق، (2012م)، ‏398‏.

كُشّ[اسْم]

ن291هـ=904م

الكُشُّ : مَا يُلْقَحُ بِهِ النَّخْلُ.

"فَإِنَّ كُشَّ النَّخْلِ الَّذِي يُسَمَّى الفَحْلَ، إِذَا جُعِلَ فِي حَمْلِ النَّخْلِ، بَلَغَ ذَلِكَ الحَمْلُ بِسُرْعَةٍ"
ابن وحشيّة النَّبَطيّ

مِكْشَاش[صِيغَةُ مُبَالَغَة]

ن393هـ=1003م

المِكْشَاشُ مِنَ الإِبِلِ: الكَثِيرُ التَّكْشَاشِ.

فِي العَنْبَرِيِّينَ ذَوِي الأَرْيَاشِ يَهْدِرُ هَدْرًا، لَيْسَ بِالمِكْشَاشِ
العَنْبريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏3/‏1018‏.

كِشْ[اسْمُ صَوْت]

ن421هـ=1030م

كِشْ : لَفْظٌ يُقَالُ لِطَرْدِ الطَّائِرِ.

"إِذَا مَشَتْ فَلَا تَقُلْ لَهَا: هِش، وَإِذَا طَارَتْ فَلَا تَقُلْ لَهَا: كِش "
مثل
رسالة الأمثال البغداديّة التي تجري بين العامة: عليّ بن الفضل المؤيّديّ الطّالقانيّ (ت، 421هـ)، تحقيق وشرح: عبد الرحمن التكريتي، دار الشؤون الثقافية العامة، بغداد، ط1، (1990م)، ‏28‏.

مُكِشّ[اسْمُ فَاعِل]

ن1291هـ=1873م

المُكِشُّ مِنَ الحَيَّاتِ وَنَحْوِهَا: المُصْدِرُ صَوْتًا.

وَتَوَقَّاهُ، فَهْوَ صِلٌّ مُكِشٌّ آكِلٌ كُلَّ أَرْقَمٍ لَدَّاغِ
ابن رزيق حميد بن محمد بن رزيق النخلي
ديوان فرند الجمان (دراسة وتحقيق): ابن رزيق حميد بن محمد بن رزيق النخلي (ت، 1291هـ)، إعداد: ليلى بنت أنور اللواتي، رسالة ماجستير، جامعة السلطان قابوس، مسقط، (2009م)، ‏199‏.

مِكَشَّة[اسْمُ آلَة]

1411هـ=1990م

المِكَشَّةُ : آلَةُ الطَّرْدِ وَالإِبْعَادِ.

"لَمْ يَكُنِ اسْتِثْنَاءً فِي هَذَا البَلَاطِ أَنْ أُشَاهِدَ أَنَا بِالذَّاتِ المَلِكَ (فَيْصَل) يُنَظِّفَ مِنْضَدَتَهُ بِنَفْسِهِ... مُسْتَخْدِمًا مِكَشَّةً مِنَ الرِّيشِ"
محمد مهدي الجواهري
مذكراتي: محمد مهدي الجواهري (ت، 1418هـ)، دار المجتبى، دمشق، ط1، (1426ه/ 2005م)، ‏1/‏198‏.
