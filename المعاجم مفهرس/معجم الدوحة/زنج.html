زَنْج[اسْمُ جِنْس]

ن11هـ=632م

الزَّنْجِيُّ : الوَاحِدُ مِنَ الزَّنْجِ، وَهُمْ جِيلٌ مِنَ السُّودَانِ.

"ثُمَّ عَادَ إِلَى الثَّدْيِ يَرْضَعُ، ثُمَّ مَرُّوا بِجِيفَةِ حَبَشِيَّةٍ، أَوْ زَنْجِيَّةٍ ، تُجَرُّ فَقَالَتْ: أُعِيذُ ابْنِي بِاللهِ أَنْ يَمُوتَ مِيتَةَ هَذِهِ الحَبَشِيَّةِ، أَوِ الزَّنْجِيَّةِ، فَتَرَكَ الثَّدْيَ، وَقَالَ: اللَّهُمَّ أَمِتْنِي مِيتَةَ هَذِهِ الحَبَشِيَّةِ، أَوِ الزَّنْجِيَّةِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏15/‏68،،‏(رقم الحديث: 9135).

زَنْج[اسْم]

ن782هـ=1380م

الزَّنْجُ مِنَ الدُّفِّ: إِحْدَى الصَّفَائِحِ المَعْدَنِيَّةِ الَّتِي تُرَكَّبُ عَلَيْهِ، وَتُحْدِثُ الإِيقَاعَ المُوسِيقِيَّ المَطْلُوبَ؛ بِتَحْرِيكِ الأَيْدِي لَهَا.

"وَأَمَّا سَمَاعُ الطَّارِ بِتِلْكَ الزُّنُوجِ المَعْرُوفَةِ، فَفِيهِ اخْتِلَافٌ بِالإِبَاحَةِ، وَالكَرَاهِيَةِ، وَالمَنْعِ"
أبو سعيد بن لُب
المعيار المعرب والجامع المغرب عن فتاوي أهل إفريقية والأندلس والمغرب: أبو العباس أحمد بن يحيى الونشريسي (ت، 914هـ)، خرجه: جماعة من الفقهاء: إشراف: محمد حجي، وزارة الأوقاف والشؤون الإسلامية، الرباط، دار الغرب الإسلامي، بيروت، (1401ه/ 1981م)، ‏11/‏73‏.

زَنْج[مَصْدَر]

1204هـ=1789م

الزَّنْجُ : حَرَكَةُ الطَّارِ؛ لاسْتِخْرَاجِ صَوْتِ صُنُوجِهِ. [الغناء والموسيقا]

"يَجِبُ أَنْ يَكُونَ مُطَابِقًا لِلْغِنَاءِ، مُقَابِلًا لَهُ: دَفًّا بِدَفٍّ، وَزَنْجًا بِزَنْجٍ، وَنَدْفًا بِنَدْفٍ"
الحائك
التراث العربي المغربي في الموسيقى: إدريس بن جلّون التويمي، دراسة وتنسيق وتصحيح: كناش الحايك، مطبعة الرّايس، الدار البيضاء، (1401هـ/1981م)، ‏40‏.

زَنَجَ[لَازِم]

53هـ=673م

زَنَجَ الشَّخْصُ: أَقْبَلَ مُنْدَفِعًا.

"إِنِّي لَمَعَ نَفَرٍ مِنَ الأَنْصَارِ، وَالنَّاسُ فِي أَمْرٍ عَظِيمٍ، إِذْ هَوَّمْتُ تَهْوِيمَةً، فَزَنَجَ شَيْءٌ؛ أَقْبَلَ، طَوِيلَ العُنُقِ، أَهْدَبَ أَهْدَلَ"
عبد الرّحمن بن السّائب الأنصاريّ
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏3/‏65‏.

زَنَجَ[مُتَعَدٍّ]

175هـ=791م

زَنَجَ الشَّخْصُ: أَقْدَمَ دُونَ تَرَدُّدٍ.

"اشْدُدْ قَضْمَ المَحَاجِمِ، وَازْنِجْ ، وَلَا تَرْبُجْ، اجْعَلْ طَعْنَكَ وَخْزا، وَمَصَّكَ حَفْزا"
أبو علقمة النَّحْويّ النُّمَيريّ
تاريخ مدينة دمشق وذكر فضلها وتسمية من حلها من الأماثل أو اجتاز بنواحيها من وارديها وأهلها: ابن عساكر (ت، 571هـ)، دراسة وتحقيق: عمر بن غرامة العمروي، دار الفكر، بيروت، ط1، (1415ه/ 1995م)، ‏67/‏90‏.

زِنْجِيَّة[اسْم]

ن68هـ=687م

الزِّنْجِيَّةُ : لُغَةُ الزِّنْجِ.

"الحَصَبُ بِلِسَانِ الزِّنْجِيَّةِ : هُوَ الحَطَبُ"
ابن عبّاس
تأويلات أهل السّنّة تفسير الماتُريديّ: أبو منصور الماتُريديّ (ت، 333هـ)، تحقيق: مجدي باسلّوم، دار الكتب العلمية، بيروت، ط1، (1426هـ/ 2005م)، ‏7/‏377‏.

زِنْجِيَّة[مَصْدَرٌ صِنَاعِيّ]

1407هـ=1986م

الزِّنْجِيَّةُ : الِانْتِمَاءُ إِلَى الزِّنْجِ.

"وَكَانَتْ هَذِهِ الجُمْهُورِيَّةُ تُغْرِي- بِدِينِهَا الإِسْلَامِيِّ، وَبِقُرْبِهَا، وَزِنْجِيَّتِهَا - العَبِيدَ النَّاقِمِينَ بِأَنْ يَأَبَقُوا إِلَيْهَا"
شاكر مصطفى
الأدب في البرازيل: شاكر مصطفى (ت، 1418هـ)، عالم المعرفة، المجلس الوطني للثقافة والفنون والآداب، الكويت، (1986م)، ‏27‏.

زَنِجَ[لَازِم]

ن130هـ=748م

زَنِجَ الشَّيْءُ: ضَمُرَ وَنَحُفَ.

وَمَخْصِرًا، كَالسَّابِرِيِّ المُدْرَجِ وَكَفَلًا، يَرْتَجُّ، فِي تَبَجْبُجِ رَيَّانَ، لَمْ يَزْنَجْ ، وَلَمْ يُزَنَّجِ
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏117‏.

زَنَّجَ[مُتَعَدٍّ]

ن130هـ=748م

زَنَّجَ الشَّخَّصَ: ضَمَّرَهُ حَتَّى ضَاقَ بَطْنُهُ.

وَمَخْصِرًا، كَالسَّابِرِيِّ المُدْرَجِ وَكَفَلًا، يَرْتَجُّ فِي تَبَجْبُجِ رَيَّانَ، لَمْ يَزْنَجْ، وَلَمْ يُزَنَّجِ
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏117‏.
ن283هـ=896م

زَنَّجَهُ : نَسَبَهُ لِلزَّنْجِ.

يُزَنِّجُ مِنْهَا النَّاسِبُونَ وَشِيظَةً وَجُمْهُورُهَا، فِي النَّاسِبِينَ، مُرَوَّمُ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏5/‏2094‏.

مُزَنَّج[اسْمُ مَفْعُول]

ن173هـ=789م

المُزَنَّجُ : المَخْلُوقُ عَلَى هَيْئَةِ الزَّنْجِ.

"كَانَ السَّيِّدُ جَارِي، وَكَانَ أَدْلَمَ، وَكَانَ يُنَادِمُ فِتْيَانًا مِنْ فِتْيَانِ الحَيِّ، فِيهِمْ فَتًى مِثْلُهُ أَدْلَمُ، غَلِيظُ الأَنْفِ وَالشَّفَتَيْنِ، مُزَنَّجُ الخِلْقَةِ"
شيبان بن محمد الحَرَّانيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏7/‏204‏.
231هـ=845م

المُزَنَّجُ مِنَ الأَشْخَاصِ: المُسْتَضْعَفُ حَزْمُهُ وَعَزْمُهُ.

وَاللهِ، [لَلنَّوْمُ]، وَبِيضٌ دُمَّجُ أَهْوَنُ مِنْ لَيْلِ قِلَاصٍ تَمْعَجُ مَخَارِمَ اللَّيْلِ، لَهُنَّ بَهْرَجُ حَتَّى يَنَامَ الوَرَعُ المُزَنَّجُ
من إنشاد أبي عبد الله ابن الأعرابيّ اللُّغويّ
كتاب الأزمنة والأمكنة: أبو عليّ المرزوقيّ (ت، 421هـ)، ضبطه وخرج آياته: خليل المنصور، دار الكتب العلمية، بيروت، ط1، (1417هـ/ 1996م)، ‏253‏.
ن811هـ=1408م

المُزَنَّجُ مِنَ الدُّفُوفِ: المُرَكَّبُ عَلَيْهِ صَفَائِحُ نُحَاسِيَّةٌ، وَنَحْوُهَا، تُحْدِثُ الإِيقَاعَ المُوسِيقِيَّ المَطْلُوبَ؛ بِتَحْرِيكِ الأَيْدِي لَهَا.

"وَالطَّارُ المُزَنَّجُ ، وَالضَّرْبُ بِالأَكُفِّ، لَا يُقَالَ فِي ذَلِكَ: إِنَّهُ حَرَامٌ بِمُجَرَّدِهِ، إِلَّا أَنْ يَقْتَرِنَ بِهِ مُحَرَّمٌ؛ فَيَحْرُمَ بِسَبَبِ مَا صَحِبَهُ"
أبو عبد الله الحفّار
المعيار المعرب والجامع المغرب عن فتاوي أهل إفريقية والأندلس والمغرب: أبو العباس أحمد بن يحيى الونشريسي (ت، 914هـ)، خرجه: جماعة من الفقهاء: إشراف: محمد حجي، وزارة الأوقاف والشؤون الإسلامية، الرباط، دار الغرب الإسلامي، بيروت، (1401ه/ 1981م)، ‏11/‏108‏.

زَنْجَاوِيّ[صِفَة]

ن313هـ=925م

الزَّنْجَاوِيُّ مِنَ الأَلْوَانِ: الأَسْوَدُ المُشْبِهُ سَوَادَ الزَّنْجِ.

"البَوْلُ الأَسْوَدُ، وَالزَّنْجَاوِيُّ ، بِعَقِبِ النَّفْثِ الشَّدِيدِ، يَدُلُّ عَلَى التَّشَنُّجِ أَنَّهُ سَيَحْدُثُ؛ لِأَنَّهُ يَدُلُّ عَلَى اليُبْسِ، وَفَنَاءِ الرُّطُوبَةِ مِنَ الجِسْمِ، وَاسْتِحْوَاذِ الحَرَارَةِ عَلَيْهِ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏5/‏220‏.

زَنْجَانِيَّة[مَصْدَرٌ صِنَاعِيّ]

1387هـ=1966م

الزَّنْجَانِيَّةُ : اكْتِسَابُ الهُوِيَّةِ الزَّنْجِيَّةِ.

"وَهَذِهِ السُّلَالَةُ هِيَ المَعْبَرُ الَّذِي تَنْتَقِلُ إِلَيْهِ السُّلَالَاتُ إِلَى الزَّنْجَانِيَّةِ مِنَ القَوْقَازِيَّةِ، فَهَؤُلَاءِ الحَامِيُّونَ قَوْقَازِيُّونَ فِي كَثِيرٍ مِنْ مَظَاهِرِهِمْ"
محمد رياض
أفريقيا: دراسة لمقوِّمات القارَّة: محمد رياض ، كوثر عبد الرسول (ت، 1423هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2015م)، ‏191‏.

زَنْجَانِيّ[صِفَة]

1387هـ=1966م

الزَّنْجَانِيُّ : المُكْتَسِبُ لِهُوِيَّةِ الزَّنْجِ.

"وَتَسْكُنُ المَجْمُوعَتَانِ الزَّنْجِيَّةُ وَالزَّنْجَانِيَّةُ السُّودَانَ الجَنُوبِيَّ"
محمد رياض
أفريقيا: دراسة لمقوِّمات القارَّة: محمد رياض ، كوثر عبد الرسول (ت، 1423هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2015م)، ‏343‏.

زُنُوجَة[مَصْدَر]

1401هـ=1980م

الزُّنُوجَةُ : الكَيْنُونَةُ الزِّنْجِيَّةُ المُمَيَّزَةُ، عِرْقِيًّا وَثَقَافِيًّا.

"وَثَمَّةَ قَضِيَّتَانِ أَسَاسِيَّتَانِ تَغْلِبَانِ عَلَى الإِيدُيُولُوجِيَّاتِ السَّائِدَةِ فِي غَرْبِ أَفْرِيقْيَا: الزُّنُوجَةُ ، أَوِ الشَّخْصِيَّةُ الإِفْرِيقِيَّةُ، وَالِاشْتِرَاكِيَّةُ الإِفْرِيقِيَّةُ"
شوقي جلال
إفريقيا في عصر التحول الاجتماعي، لب. س. لويد: شوقي جلال ، ترجمة: شوقي جلال، عالم المعرفة، المجلس الوطني للثقافة والفنون والآداب، الكويت، (1980م)، ‏265‏.

مُتَزَنِّج[اسْمُ فَاعِل]

1420هـ=1998م

المُتَزَنِّجُ : المُكْتَسِبُ هُوِيَّةَ الزُّنُوجِ.

"كَانَتْ هُنَاكَ نَحْوَ 250 قَرْيَةً مُخَطَّطَةً وَمَبْنِيَّةً جَيِّدًا، قُطَّانُهَا مِنَ العَنَاصِرِ المُتَزَنِّجَةِ "
مجموعة من المؤلفين
موجز دائرة المعارف الإسلامية: مجموعة من المؤلفين ، مركز الشارقة للإبداع الفكري، ط1، (1418ه/ 1998م)، ‏31/‏9773‏.

تَزَنُّج[مَصْدَر]

1439هـ=2017م

التَّزَنُّجُ : الِاتِّصَافُ بِالزُّنُوجَةِ.

"وَمِنْ أَجْلِ العَمَلِيَّاتِ الأَخِيرَةِ سَكَّ مُسَاعِدُ بُومَان- فِي مُتْحَفِ (فِييِنَّا فَالْتَر هِيرْشَبْرَغ)- مُصْطَلَحَ التَّزَنُّجِ ؛ لِيُوصَفَ الفَسَادُ العُنْصُرِيُّ المَزْعُومُ مِنَ الأَسْفَلِ"
أبوبكر أحمد باقادر، إيمان الوكيلي
الأنثروبولوجيا حقل علمي واحد وأربع مدارس، لفريدريك بارث وآخرين: أبوبكر أحمد باقادر، إيمان الوكيلي ، ترجمة: أبوبكر أحمد باقادر، إيمان الوكيلي، مراجعة: ساري حنفي، المركز العربي للأبحاث ودراسة السياسات، بيروت، ط1، (2017م)، ‏194‏.
