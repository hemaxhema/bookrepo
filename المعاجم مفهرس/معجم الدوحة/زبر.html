زُبْرَة[اسْم]

قبل 150ق.هـ=476م

الزُّبْرَةُ : القِطْعَةُ الضَّخْمَةُ مِنَ الحَدِيدِ وَنَحْوِهِ.

فَهَدَّمَ مِنْ بُرُوجِ الحَضْرِ صَخْرًا كَأَنَّ ثِقَالَهُ زُبَرُ الحَدِيدِ
منسوب إلى جُدَيّ بن الدِّلْهاث القُضاعيّ
معجم البلدان: ياقوت الحَمَوي (ت، 626هـ)، دار صادر، بيروت، (1397هـ/ 1977م)، ‏2/‏269‏.
قبل 50ق.هـ=573م

الزُّبْرَةُ : مَوْضِعُ الكَاهِلِ المُرْتَفِعُ بَيْنَ الكَتِفَيْنِ.

فَقُلْتُ: مَلِّ، فَاجْثَأَلَّ وَجَثَمْ عَنْ زُبْرَةٍ مِنْهُ، وَعَنْ رَأْسٍ مُتَمْ
شقصة الفَزاريّ
شعر قبيلة ذُبيان في الجاهلية: جمع وتحقيق ودراسة: سلامة عبد الله السويدي، مطبوعات جامعة قطر، الدوحة، (1408هـ/ 1987م)، ‏472‏.
35هـ=655م

زُبْرَةُ الأَسَدِ: الشَّعَرُ المُجْتَمِعُ بَيْنَ كَتِفَيْهِ.

"فَهَجْهَجْنَا بِهِ، فَكَرَّ مُقْشَعِرَّا بِزُبْرَتِهِ "
أبو زُبيد الطّائيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏12/‏88‏.
64هـ=684م

الزُّبْرَةُ : مَا بَيْنَ الوَرِكَيْنِ مِنَ الحِمَارِ وَالشَّاةِ.

ضَخْمُ المِلَاطَينِ مَوَّارُ الضُّحَى هَزِجٌ، كَأَنَّ زُبْرَتَهُ فِي الآلِ عُنْقُودُ
الأخطل
شعر الأخطل أبي مالك غياث بن غوث التغلبي: صنعة: السُّكّري، رواية: أبي جعفر محمّد بن حبيب، تحقيق: فخر الدين قباوة، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط4، (1416هـ/ 1996م)، ‏81‏.
ن175هـ=791م

الزُّبْرَةُ : سَنْدَانُ الحَدَّادِ.

"المَطِيلَةُ: اسْمُ الحَدِيدَةِ الَّتِي تُمْطَلُ مِنَ البَيْضَةِ، وَمِنَ الزُّبْرَةِ، وَالمَطَّالُ: الحَدَّادُ، وَالزُّبْرَةُ : العَلَاةُ الَّتِي يُضْرَبُ عَلَيْهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏434‏.
ن650هـ=1252م

الزُّبْرَةُ : مَنْزِلٌ مِنَ المَنَازِلِ، يَنْزِلُهُ القَمَرُ فِي اللَّيْلَةِ الثَّانِيَةِ عَشْرَةَ.

"وَهُوَ نَجْمٌ وَاحِدٌ ‌نَيِّرٌ ‌يَتْلُو ‌الزُّبْرَةَ "
الصّاغانيّ
العباب الزاخر واللباب الفاخر: الصّاغانيّ (ت، 650هـ)، تحقيق: فير محمد حسن، محمد حسن آل ياسين، راجعته وأشرفت على طبعه: لجنة المجمع، مطبوعات المجمع العلمي العراقي- دار الشؤون الثقافية العامة- دار الرشيد، بغداد، ط1، (1398ه/ 1978م- 1402ه/ 1981م)، ‏344‏.

زَبُور[اسْم]

قبل 50ق.هـ=573م

الزَّبُورُ : الكِتَابُ المُنْزَلُ عَلَى النَّبِيِّ دَاوُدَ # .

"أَجِدُ لِبَارٍّ مَبْرُور، أُيِّدَ بِالقَهُور، وَوُصِفَ فِي الزَّبُور "
شافع بن كُليب الصّدفيّ
أسجاع الكهَّان الجاهليِّين وأشعارهم جمع وتحقيق ودراسة: إعداد: ياسين عبد الله جمول، رسالة ماجستير، جامعة دمشق، (2011- 2012م)، ‏348‏.
قبل 50ق.هـ=573م

الزَّبُورُ : الكِتَابُ.

لِمَنْ طَلَلٌ أَبْصَرْتُهُ فَشَجَانِي كَخَطِّ زَبُورٍ فِي عَسِيبٍ يَمَانِ؟
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏85‏.

زَبُور[صِفَةٌ مُشَبَّهَة]

ن65هـ=685م

الزَّبُورُ مِنَ الكَتَائِبِ وَنَحْوِهَا: الصُّلْبَةُ المُتَمَاسِكَةُ.

صَبَحْنَاهُمْ غَدَاةَ بَنَاتِ قَيْنٍ مُلَمْلَمَةً مَنَاكِبُهَا، زَبُورَا
أرطاة ابن سُهَيَّة
شعر أرطأة بن سُهَيّة المُرّي: جمعه وحققه وشكله وشرحه وقدم له ووضع فهارسه: شريف علاونة، جامعة البتراء- دار المناهج، عمان، ط1، (1427هـ/ 2006م)، ‏66‏.

زَبِير[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الزَّبِيرُ : المُتَمَاسِكُ، المُحْكَمُ.

يَا مَعْشَرَ الأَبْنَاءِ، إِنْ فُزْتُمْ بِهَا، فَوْزَ الزَّبِيرَةِ ، جَمْعُنَا لَمْ يُثْأَرِ
مارية بنت الدّيّان الحارثيّة
شُعراء مَذْحِج أخبارهم وأشعارهم في الجاهلية: صنعة: مُقبل التَّامّ عامر الأحمدي، مَجمَع العربيَّة السَّعيدة، صنعاء، ط2، (1435هـ/ 2014م)، ‏81‏.
ن175هـ=791م

الزَّبِيرُ : الضَّخْمُ المُكْتَنِزُ.

"وَكَبْشٌ زَبِيرٌ ، أَيْ: ضَخْمٌ مُكْتَنِزٌ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏363‏.

زَبِير[مَصْدَر]

ن100هـ=718م

الزَّبِيرُ : الكَدَرُ.

وَقَدْ جَرَّبَ النَّاسُ آلَ الزُّبَيْـ ـرِ، فَلَاقَوْا مِنَ الِ الزُّبَيْرِ الزَّبِيرَا
عبد الله بن هَمّام السَّلوليّ
شعر عبد الله بن همّام السَّلُولي: جمع وتحقيق ودراسة: وليد محمّد السَّراقبي، مركز جمعة الماجد للثقافة والتراث، دبي، ط1، (1417هـ/ 1996م)، ‏70‏.

لَهَا ثُنَنٌ كَخَوَافِي العُقَا بِ سُودٌ، يَفِئْنَ إِذَا تَزْبَئِرّْ

11هـ=632م

ازْبَأَرَّ الكَلْبُ: تَهَيَّأَ لِلقِتَالِ.

لَحَا اللهُ جَرْماً كُلَّمَا ذَرَّ شَارِقٌ، وُجُوهَ كِلَابٍ، هَارَشَتْ ‌فَازْبَأَرَّتِ
عمرو بن مَعْدي كَرِب الزُّبَيديّ
شعر عَمرو بن مَعْدي كَرب الزُّبَيْدي: جمعه ونسّقه: مطاع الطرابيشي، مطبوعات مجمع اللغة العربية، دمشق، ط2، (1405هـ/ 1985م)، ‏72‏.
73هـ=692م

ازْبَأَرَّ الشَّعَرُ وَنَحْوُهُ: نَبَتَ.

حُوصَ العُيُونِ، مُجْهِضَاتٍ مَا اسْتَطَرْ مِنْهُنَّ إِتْمَامٌ شَكِيرًا فَاشْتَكَرْ بِحَاجِبٍ، وَلَا قَفًا، وَلَا ازْبَأَرْ مِنْهُنَ سِيسَاءٌ وَلَا اسْتَغْشَى الوَبَرْ
العجّاج

ن90هـ=709م

ن337هـ=948م

ن364هـ=974م

الزَّابِرُ الكَلَامَ وَنَحْوَهُ: الَّذِي يَكْتُبُهُ وَيُنَمِّقُهُ.

"وَكَذَلِكَ فِعْلُ الزَّابِرِينَ لِلْكَلَامِ، المُمْتَحِنِينَ لَهُ"
يحيى بن عَديّ
منطق أرسطو: حققه وقدم له: عبد الرحمن بدويّ، وكالة المطبوعات، الكويت، دار القلم، بيروت، ط1، (1980م)، ‏3/‏1047‏.

مُزْبَئِرّ[اسْمُ فَاعِل]

ن175هـ=791م

المُزْبَئِرُّ : المُقْشَعِرُّ المُنْتَفِشُ الشَّعَرِ.

" وَالمُزْبَئِرُّ : المُقْشَعِرُّ مِنَ النَّاسِ وَالدَّوَابِّ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏401‏.
ن183هـ=799م

المُزْبَئِرُّ مِنَ النَّبْتِ وَنَحْوِهِ: الطَّالِعُ النَّابِتُ.

أَذَلِكَ، أَمْ ذَبُّ الرِّيَادِ خَلَا لَهُ لِوًى، وَكَثِيبٌ مُزْبَئِرٌّ خَمَائِلُهْ؟
أبو حيّة النُّميريّ
شعر أبي حيَّة النُّميري: جمعه وحققه: يحيى الجبوري، منشورات وزارة الثقافة، دمشق، (1975م)، ‏72‏.

زُبَيْرِيَّة[اسْم]

ن182هـ=798م

الزُّبَيْرِيَّةُ : فِرْقَةٌ مِنَ المُسْلِمِينَ مَالُوا إِلَى عَبْدِ اللهِ بْنِ الزُّبَيْرِ وَبَايَعُوهُ.

"كَانَ عَبْدُ المَلِكِ بْنُ مَرْوَانَ لَا يَسْمَعُ لِشُعَرَاءِ مُضَرَ، وَلَا يَأْذَنُ لَهُمْ؛ لِأَنَّهُمْ كَانُوا زُبَيْرِيَّةً "
يونس بن حَبيب النَّحْويّ
طبقات فحول الشّعراء: محمّد بن سلّام الجُمَحيّ (ت، 231هـ)، قرأه وشرحه: محمود محمّد شاكر، دار المدني، جدة، (1974م)، ‏2/‏418‏.

زَبَّرَ[مُتَعَدٍّ]

ن209هـ=824م

زَبَّرَهُ بِالشَّيْءِ: طَوَّقَهُ، وقَوَّى جَوَانِبَهُ بِهِ.

"فَجَعَلُوا قَبْرَهُ هُنَاكَ، وَزَبَّرُوهُ بِذَلِكَ الشَّجَرِ وَالمَدَرِ"
المُنتجع بن نَبهان
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏18/‏34‏.
ن456هـ=1064م

زَبَّرَ الشَّجَرَ: قَطَعَ أَغْصَانَهُ، وَقَلَّمَهَا.

"وَيُؤَبِّرُ النَّخْلَ، وَيُزَبِّرُ الدَّوَالِيَ، وَيَحْرُثُ مَا احْتَاجَ إِلَى حَرْثِهِ"
ابن حزم الظّاهريّ الأندلسيّ
المحلَّى: ابن حزم الظّاهريّ الأندلسيّ (ت، 456هـ)، تحقيق: محمّد منير الدمشقي، أحمد محمد شاكر، إدارة الطباعة المنيرية- مطبعة النهضة، القاهرة، (1347- 1352هـ)، ‏8/‏229‏.

زَبْرَة[اسْمُ مَرَّة]

ن212هـ=827م

الزَّبْرَةُ : الزَّجْرَةُ.

مَنْ مُبْلِغٌ يَحْيَى وَدُونَ لِقَائِهِ زَبَرَاتُ كُلِّ خُنَابِسٍ هَمْهَامِ
يَا رَاعِيَ السُّلْطَانِ غَيْرَ مُفَرِّطٍ فِي لِينِ مُخْتَبَطٍ وَطِيبِ شِمَامِ
إسحاق بن حسّان الخُريميّ
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏3/‏352 ‏-‏ 353‏.

تَزْبِرَة[مَصْدَر]

ن216هـ=831م

تَزْبِرَةُ الشَّخْصِ: كِتَابَتُهُ وَخَطُّهُ.

"أَنَا أَعْرِفُ تَزْبِرَتِي "
رجل من الأعراب
كتاب المأثور من اللّغة (ما اتفق لفظه واختلف معناه): أبو العَمَيْثَل الأعرابيّ (ت، 240هـ)، تحقيق: محمّد عبد القادر أحمد عطا، مكتبة النهضة المصرية، القاهرة، ط1، (1408هـ/ 1988م)، ‏87‏.

مِزْبَر[اسْمُ آلَة]

ن262هـ=876م

المِزْبَرُ : آلَةُ الزَّبْرِ وَالكِتَابَةِ، وَهُوَ القَلَمُ.

"دَعَا فِي مَرَضِهِ ‌بِدَوَاةٍ ‌وَمِزْبَرٍ فَكَتَبَ اسْمَ الخَلِيفَةِ بَعْدَهُ"
ابن شَبَّة
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏2/‏29‏.
ن456هـ=1064م

المِزْبَرَةُ : آلَةٌ تُقْطَعُ بِهَا أَغْصَانُ الشَّجَرِ وَنَحْوِهِ.

"وَقَدْ عَلِمَ كُلُّ ذِي حَظٍّ مِنَ التَّمْيِيزِ أَنَّ مُرُورَهُمْ وَمَسَاحِيَهُمْ لِحَفِيرِ الأَرْضِ...وَسِكَكَهُمْ لِلْحَرْثِ، وَمَزَابِرَهُمْ لِلزَّرْجُونِ"
ابن حزم الظّاهريّ الأندلسيّ
المحلَّى: ابن حزم الظّاهريّ الأندلسيّ (ت، 456هـ)، تحقيق: محمّد منير الدمشقي، أحمد محمد شاكر، إدارة الطباعة المنيرية- مطبعة النهضة، القاهرة، (1347- 1352هـ)، ‏9/‏498‏.

تَزْبِير[مَصْدَر]

ن291هـ=904م

تَزْبِيرُ الأَغْصَانِ: قَطْعُ أَطْرَافِهَا وَتَقْلِيمُهَا.

"وَيَصْبِرُونَ عَلَى التَّزْبِيرِ ، وَتَقْوِيمِ أَغْصَانِ الشَّجَرِ، وَتَسْوِيَةِ عُمُدِ الكُرُومِ، وَالشَّجَرِ، وَسَائِرِ النَّبَاتِ"
ابن وحشيّة النَّبَطيّ
الفلاحة النَّبَطيّة: ابن وحشيّة النَّبَطيّ (ت، 291هـ)، تحقيق: توفيق فهد، المعهد العلميّ الفرنسيّ للدراسات العربيّة، دمشق، الجفان والجابي، قبرص، ط1، (1993م)، ‏1/‏196‏.
ن573هـ=1177م

التَّزْبِيرُ : الكِتَابَةُ.

"وَأَمَرَ بِتَزْبِيرِ أَيَّامِهِمْ فِي حِجَارَةِ القُصُورِ"
نَشْوان الحِمْيريّ
ملوك حمير وأقيال اليمن (قصيدة نشوان الحميريّ وشرحها المسمّى خلاصة السِّيرة الجامعة لعجائب أخبار ملوك التّبابعة): نَشْوان الحِمْيريّ (ت، 573هـ)، تحقيق: علي بن إِسماعيل المؤيد، إسماعيل بن أَحمد الجرافيّ، دار العودة، بيروت، ط2، (1978م)، ‏58‏.

زَبُورِيّ[اسْمٌ مَنْسُوب]

ن658هـ=1260م

الزَّبُورِيُّ مِنَ الأَقْوَالِ: المَنْسُوبُ إِلَى كِتَابِ الزَّبُورِ.

"وَإِنْ كَانَ الهَاجِعُ كَاهِنًا فَلْيُجْعَلْ أَمَامَ المَذْبَحِ الإِلَهِيِّ، وَإِنْ كَانَ مِنَ الشَّعْبِ فَيُجْعَلُ أَسْفَلَ مِنَ المَذْبَحِ، ثُمَّ يَبْتَدِئُ رَئِيسُ الكَهَنَةِ بِالصَّلَاةِ عَلَيْهِ، وَالشُّكْرِ لِأَجْلِهِ، وَبَعْدَ ذَلِكَ يَقْرَأُ الخُدَّامُ الأَقْوَالَ الزَّبُورِيَّةَ "
الصفيّ أبو الفضائل بن العسّال
المجموع الصفوي: الصفيّ أبو الفضائل بن العسّال (ت، 658هـ)، اعتنى به وشرح مواده: جرجس فيلوثاوس عوض، مطبعة التوفيق، القاهرة، ط1، (د.ت)، ‏206‏.
