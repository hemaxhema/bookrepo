ثُؤْلُول[اسْم]

ن11هـ=632م

الثُّؤْلُولُ : بَثْرَةٌ صَغِيرَةٌ مُسْتَدِيرَةٌ كَالخُرَّاجِ اليَابِسِ تَظْهَرُ عَلَى الجِلْدِ.

"لَا يَمُوتُ هَذَا حَتَّى يَذْهَبَ الثُّؤْلُولُ مِنْ وَجْهِهِ"
حديث نبويّ
المستدرك على الصّحيحين: أبو عبد الله الحاكم النّيسابوريّ، دراسة وتحقيق: مصطفى عبد القادر عطا، دار الكتب العلميّة، بيروت، لبنان، ط2، 1422هـ-2002م، ‏4/‏545،،‏(رقم الحديث: 8525).
ن209هـ=824م

ثُؤْلُولُ الدَّابَّةِ: هَنَةٌ لَحْمِيَّةٌ كَالبَظْرِ فِي طَرَفِ حَيَائِهَا.

"وَفِي مَوْضِعِ عِجَانِ الذَّكَرِ ظَبْيَةُ الأُنْثَى، وَفِي الظَّبْيَةِ: المُلْتَقَى، وَالحِتَارُ، وَالمَهْبِلُ، وَالخَاتَمُ، وَالمَقَرَّةُ، وَالشَّرِيجَةُ، وَالثُّؤْلُولُ "
أبو عُبيدة
كتاب الخيل: أبو عبيدة (ت، 209هـ)، رواية: أبي حاتم سهل بن محمّد السجستاني عنه رواية أبي يوسف الأصبهاني عنه، مطبعة دائرة المعارف العثمانية، حيدر آباد الدكن، ط1، (1358هـ)، ‏36‏.

ثُؤْلِلَ[مُلَازِمٌ لِلْبِنَاءِ لِلْمَجْهُول]

ن175هـ=791م

ثُؤْلِلَ الشَّخْصُ: انْتَشَرَتِ الثَّآلِيلُ فِي جَسَدِهِ.

"وَيُقَالُ مِنَ الثُّؤْلُولِ: ثُؤْلِلَ الرَّجُلُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏241‏.

تَثَأْلَلَ[لَازِم]

ن175هـ=791م

تَثَأْلَلَ الجَسَدُ: ظَهَرَتْ عَلَيْهِ الثَّآلِيلُ.

"وَيُقَالُ مِنَ الثُّؤْلُولِ: ثُؤْلِلَ الرَّجُلُ، وَقَدْ تَثَأْلَلَ جَسَدُهُ بِالثَّآلِيلِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏241‏.
557هـ=1162م

تَثَأْلَلَ اللَّحْمُ: تَصَلَّبَ وَصَارَ ثُؤْلُولًا.

"وَإِنْ كَانَ اللَّحْمُ الَّذِي فِي القُرْحَةِ قَدْ صَلُبَ حَتَّى تَثَأْلَلَ ، فَيَنْبَغِي حِينَئِذٍ أَنْ تَقْطَعَهُ بِالحَدِيدِ، ثُمَّ تَضَعَ عَلَيْهِ الدَّوَاءَ المُنْبِتَ"
ابن رشد الحفيد
الكليات في الطب مع معجم بالمصطلحات الطبية العربية: ابن رشد الحفيد (ت، 595هـ)، تحقيق: محمد عابد الجابري، مركز دراسات الوحدة العربية، بيروت، ط1، (1999م)، ‏573‏.

ثُؤَيْلِيل[اسْمٌ مُصَغَّر]

ن309هـ=921م

الثُّؤَيْلِيلُ : تَصْغِيرُ الثُّؤْلُولِ.

وَأَنْكَرْتَ مِنْ زَبَّانَ خُضْرَةَ لَوْنِهِ وَأَنْفًا لَهُ مِثْلَ الثُّؤَيْلِيلِ قَاعِيَا
من شواهد كُراع النَّمْل الهُنائيّ الأزديّ
المُنتَخَب من غريب كلام العَرَب: كُراع النَّمْل الهُنائيّ الأزديّ (ت، 309هـ)، تحقيق: محمّد بن أحمد العمري، جامعة أم القرى، مكة المكرمة، ط1، (1409هـ/ 1989م)، ‏2/‏639‏.

ثُؤْلُولِيّ[اسْمٌ مَنْسُوب]

ن313هـ=925م

الثُّؤْلُولِيُّ مِنَ اللَّحْمِ وَنَحْوِهِ: المَنْسُوبُ إِلَى الثُّؤْلُولِ.

"عِلَاجُ البَوَاسِيرِ المَنْكُوسَةِ: تُلَفُّ خِرْقَةٌ خَشِنَةٌ عَلَى مِيلٍ، وَتُدْخَلُ فِيهِ، وَيُحَكُّ حَتَّى يَدْمَى نِعِمَّا، ثُمَّ يُتْرَكُ يَوْمَهُ، ثُمَّ يُعَادُ عَلَيْهِ الإِدْمَاءُ وَالحَكُّ ثَلَاثَ مَرَّاتٍ، وَكُلَّمَا كَانَ أَعْتَقَ احْتِيجَ أَنْ يُسْتَقْصَى ذَلِكَ عَلَيْهِ أَكْثَرَ؛ لِأَنَّ اللَّحْمَ الثُّؤْلُولِيَّ فِي جَوْفِهِ أَكْثَرُ وَأَصْلَبُ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏3/‏439‏-‏ 440‏.

ثَالُولِيّ[صِفَة]

ن313هـ=925م

الثَّالُولِيُّ مِنَ البُثُورِ وَنَحْوِهَا: المُتَيَبِّسُ المُتَّصِفُ بِصِفَاتِ الثُّؤْلُولِ.

"وَأَمَّا مَا لَحِقَ الجُدَرِيَّ وَالحَصْبَةَ، فَهَذِهِ البِيضُ مِنَ الجُدَرِيِّ، الكِبَارُ المُتَفَرِّقَةُ القَلِيلَةُ العِدَّةِ، وَالسَّهْلَةُ الخُرُوجِ... وَأَمَّا البِيضُ الصِّغَارُ جِدًّا، المُتَقَارِبَةُ الصُّلْبَةُ الثَّالُولِيَّةُ الَّتِي لَا تَحْمِلُ مَاءً، فَرَدِيئَةٌ، وَرَدَاءَتُهَا بِمِقْدَارِ عُسْرِ نُضْجِهَا"
أبو بكر الرّازيّ
كتاب في الجدري والحصبة: أبو بكر الرّازيّ (ت، 313هـ)، المدرسة الكلية السورية الإنجيلية، بيروت، (1872م)، ‏70‏-‏ 71‏.
ن1081هـ=1670م

الثَّالُولِيُّ مِنَ الأَدْوِيَةِ: المُخَصَّصُ لِعِلَاجِ الثَّآلِيلِ.

"وَالثَّآلِيلُ: بُثُورٌ صِغَارٌ مُسْتَدِيرَةٌ صُلْبَةٌ. العِلَاجُ: يَجِبُ تَنْقِيَةُ البَدَنِ بِالفَصْدِ وَالإِسْهَالِ إِنْ كَثُرَتْ فِي البَدَنِ، وَكَانَ مُمْتَلِئًا، ثُمَّ يُوضَعُ عَلَيْهَا الأَدْوِيَةُ الثَّالُولِيَّةُ "
ابن سلوم
الطب المجرّب المسمى غاية الإتقان في تدبير بدن الإنسان: ابن سلوم (ت، 1081هـ)، تحقيق وشرح: محسن عقيل، دار المحجة البيضاء- دار الرسول الأكرم، بيروت، ط1، (1425ه/ 2004م)، ‏716‏-‏ 717‏.

أَثْلُول[اسْم]

ن456هـ=1064م

الأَثْلُولُ : الثُّؤْلُولُ.

"إِنَّ نَفْسَ كُلِّ أَحَدٍ لَا تَطِيبُ عَلَى مَنْ بِهَا فِي خَافِي جَسَدِهَا لَمْعَةٌ مِنْ بَرَصٍ... وَعَلَى مَنْ بِهَا أَكَلَةٌ فِي وَجْهِهَا، أَوْ أَثْلُولٌ ضَخْمٌ"
ابن حزم الظّاهريّ الأندلسيّ
المحلَّى: ابن حزم الظّاهريّ الأندلسيّ (ت، 456هـ)، تحقيق: محمّد منير الدمشقي، أحمد محمد شاكر، إدارة الطباعة المنيرية- مطبعة النهضة، القاهرة، (1347- 1352هـ)، ‏10/‏115‏.

مُثَأْلَل[اسْمُ مَفْعُول]

ن684هـ=1285م

المُثَأْلَلُ : الَّذِي انْتَشَرَتِ الثَّآلِيلُ فِي جَسَدِهِ.

بِمُرْسِيَةٍ قَاضٍ قَطِيمٌ مُثَأْلَلٌ ثَآلِيلُهُ مِنْ أَنْ يُنَالَ مَوَانِعُ
محفوظ بن مرعي الشّريف
الوافي في نظم القوافي: أبو الطيّب الرُّنديّ (ت، 684هـ)، دراسة وتحقيق: هدى شوكت بهنام، زينة عبد الجبار محمَّد، دار غيداء، عمّان، ط1، (1440ه/ 2019م)، ‏434‏.

تَثَأْلُل[مَصْدَر]

1415هـ=1994م

التَّثَأْلُلُ : انْتِشَارُ الثَّآلِيلِ عَلَى الجِلْدِ وَنَحْوِهِ.

"الفِطْرِيَّاتُ وَالأَمْرَاضُ الَّتِي تُسَبِّبُهَا هِيَ: Olipidium، يُسَبِّبُ أَمْرَاضَ الجُذُورِ لِكَثِيرٍ مِن النَّبَاتَاتِ؛ Synchytriun، يُسَبِّبُ التَّثَأْلُلَ الأَسْوَدَ فِي البَطَاطِسِ"
محمود موسى أبو عرقوب
أمراض النّبات: جورج أجريوس، ترجمة: محمود موسى أبو عرقوب، المكتبة الأكاديميّة، ط1، 1994م، ‏549‏.
