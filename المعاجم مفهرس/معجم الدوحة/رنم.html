تَرَنُّم[مَصْدَر]

قبل 13ق.هـ=609م

تَّرَنُّمُ الشَّيْءِ: إِصْدَارُهُ صَوْتًا حَسَنًا.

إِذَا انْقَلَبَتْ فَوْقَ الحَشِيَّةِ مَرَّةً تَرَنَّمَ وَسْوَاسُ الحُلِيِّ تَرَنُّمَا
حاتم بن عبد الله الطّائيّ
ديوان شعر حاتم بن عبد الله الطائي وأخباره: صنعة: يحيى بن مدرك الطائي، رواية: هشام بن محمّد الكلبي، دراسة وتحقيق: عادل سليمان جمال، مطبعة المدني، القاهرة، د.ت. ‏234‏.
ن11هـ=632م

التَّرَنُّمُ بِالكَلَامِ: تَحْسِينُ الصَّوْتِ، وَتَرْجِيعُهُ أَثْنَاءَ إِلْقَائِهِ.

"مَا أَذِنَ اللهُ لِشَيْءٍ أَذَنَهُ لِنَبِيٍّ حَسَنِ التَّرَنُّمِ بِالقُرْآنِ"
حديث نبويّ
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏21/‏159‏.

مُتَرَنِّم[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

المُتَرَنِّمُ : المُرَجِّعُ صَوْتَهُ فِي تَطْرِيبٍ.

فَتَرَى الذُّبَابَ بِهَا يُغَنِّي وَحْدَهُ هَزِجًا، كَفِعْلِ الشَّارِبِ المُتَرَنِّمِ
عنترة
ديوان عنترة: تحقيق ودراسة: محمّد سعيد مولوي، المكتب الإسلامي، بيروت، (1390هـ/ 1970م)، ‏197‏.

تَرَنَّمَ[لَازِم]

قبل 13ق.هـ=609م

تَرَنَّمَ الشَّخْصُ، وَنَحْوُهُ: حَسَّنَ صَوْتَهُ، وَرَجَّعَهُ عِنْدَ التَّغَنِّي.

أَحِنُّ إِلَى تِلْكَ المَنَازِلِ كُلَّمَا غَدَا طَائِرٌ فِي أَيْكَةٍ يَتَرَنَّمُ
عنترة
شرح ديوان عنترة: الخطيب التِّبْريزي (ت، 502هـ)، قدم له ووضع هوامشه وفهارسه: مجيد طراد، دار الكتاب العربي، بيروت، ط1، (1412هـ/ 1992م)، ‏141‏.

تَرْنِيم[مَصْدَر]

قبل 13ق.هـ=609م

التَّرْنِيمُ : تَرْجِيعُ الصَّوْتِ فِي تَطْرِيبٍ.

تَحُفُّهُ هِقْلَةٌ، سَطْعَاءُ، خَاضِعَةٌ تُجِيبُهُ بِزِمَارٍ، فِيهِ تَرْنِيمُ
علقمة الفَحْل
ديوان علقمة الفحل: شرح: الأعلم الشَّنْتَمَري (ت، 476هـ)، حقّقه: لطفي الصقال، درية الخطيب، راجعه: فخر الدين قباوة، دار الكتاب العربي- مطبعة الأصيل، حلب، ط1، (1389هـ/ 1969م)، ‏63‏.

رَنِم[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الرَّنِمُ : الحَسَنُ الصَّوْتِ وَالإِيقَاعِ.

قَدْ أَشْهَدُ الشَّرْبَ، فِيهِمْ مِزْهَرٌ رَنِمٌ وَالقَوْمُ تَصْرَعُهُمْ صَهْبَاءُ خُرْطُومُ
علقمة الفَحْل
ديوان علقمة الفحل: شرح: الأعلم الشَّنْتَمَري (ت، 476هـ)، حقّقه: لطفي الصقال، درية الخطيب، راجعه: فخر الدين قباوة، دار الكتاب العربي- مطبعة الأصيل، حلب، ط1، (1389هـ/ 1969م)، ‏68‏.

رَنَّمَ[مُتَعَدٍّ]

ن93هـ=712م

رَنَّمَ الشَّيْءَ: أَحْدَثَ مِنْهُ صَوْتًا حَسَنًا.

"دَخَلْتُ عَلَى البَرَاءِ بْنِ مَالِكٍ، وَهُوَ يَتَغَنَّى، وَيُرَنِّمُ قَوْسَهُ"
أنس بن مالك الخزرجيّ الأنصاريّ
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏9/‏17‏.
ن175هـ=791م

رَنَّمَ صَوْتَهُ: رَجَّعَهُ فِي تَطْرِيبٍ.

"التَّرْنِيمُ: مَا اسْتَلْذَذْتَ مِنْ صَوْتِ الطَّرَبِ، وَتَطْرِيبِ الصَّوْتِ... وَهُوَ يُرَنِّمُ الصَّوْتَ، وَيَتَرَنَّمُ فِي صَوْتِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏270‏.

رَنَّمَ[لَازِم]

324هـ=936م

رَنَّمَ : تَرَنَّمَ.

عَرَفَ الحَمَامُ لَهَا الغِنَاءَ، فَرَنَّمَتْ فَاعْتَادَنِي نُكْسٌ مِنَ الأَحْزَانِ
سعد بن الجرَّاح الحارثيّ
الموازنة بين شعر أبي تمّام والبُحْتُريّ: أبو القاسم الآمديّ (ت، 370هـ)، ج(1- 2)، تحقيق: السيد أحمد صقر، دار المعارف، القاهرة، ط4، (1992م)، ‏2/‏152‏.

مُرْتَنِم[اسْمُ فَاعِل]

ن198هـ=813م

المُرْتَنِمُ : المُتَرَنِّمُ.

فِي كُلِّ يَوْمٍ يَظَلُّ قَيِّمُهَا مُكَبَّلًا كَالأَسِيرِ، فِي صَفَدِ
مُزَمْزِمًا حَوْلَهَا، وَمُرْتَنِمًا يَرْجُو، بِصَوْنٍ لَهَا، غِنَى الأَبَدِ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏173‏.

رَنِيم[مَصْدَر]

ن210هـ=825م

الرَّنِيمُ : تَرْجِيعُ الصَّوْتِ، وَتَطْرِيبُهُ.

أَشَاقَكَ بَرْقٌ؟ أَمْ شَجَتْكَ حَمَامَةٌ لَهَا فَوْقَ أَطْرَافِ الأَرَاكِ رَنِيمُ
محمّد بن يزيد المَسْلَميّ الحِصْنيّ الأُمويّ
سرور النفس بمدارك الحواس الخمس: أبو العبّاس التِّيفَاشيّ (ت، 651هـ)، حققه: إحسان عباس، المؤسسة العربية للدراسات والنشر، بيروت، ط1، (1400ه/ 1980م)، ‏96‏.

رَنِيم[صِفَةٌ مُشَبَّهَة]

1287هـ=1869م

الرَّنِيمُ مِنَ الأَصْوَاتِ: المُتَرَنِّمُ.

فَأَرِينِي الوَجْهَ، مَا أَلْطَفَهُ! وَاسْمَعِي الصَّوْتَ الرَّنِيمِ، الأُغْنِيَهْ
رزق الله حسون
أشعر الشعر: رزق الله حسون (ت، 1297هـ)، المطبعة الأميركانية، ط1، (1870م)، ‏104‏.

رَنَمَة[اسْم]

ن216هـ=831م

الرَّنَمَةُ : نَوْعٌ مِنْ صِغَارِ الشَّجَرِ.

"مِنْ نَبَاتِ السَّهْلِ: الحُرْبُثُ، وَالرَّنَمَةُ ، وَالتَّرِبَةُ"
الأصمعيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏15/‏215‏.

تَرْنَمُوت[مَصْدَر]

255هـ=869م

التَّرْنَمُوتُ : التَّرَنُّمُ.

تُجَاوِبُ الصَّوْتَ بِتَرْنَمُوتِهَا تَسْتَخْرِجُ الحَبَّةَ مِنْ تَابُوتِهَا
من إنشاد أبي حاتم السِّجِسْتانيّ
أبو تُراب اللّغويّ وكتابه الاعتقاب: أبو تراب اللُّغويّ (ت، 275هـ)، إعداد: عبد الرزاق بن فرّاج الصاعدي، مجلة الجامعة الإسلامية للعلوم الشرعية، المدينة المنورة، مج34، ع114، 2002م، ع115، ‏2/‏453‏.

تَرْنَمُوت[صِفَة]

ن449هـ=1057م

التَّرْنَمُوتُ مِنَ الأَقْوَاسِ: الَّتِي لَهَا حَنِينٌ وَتَصْوِيتٌ عِنْدَ الرَّمْيِ بِهَا.

"فَوَرَدْنَ، وَقَدْ طَلَعَ ذَنَبُ السِّرْحَان، وَكَلَأَهَا بِالقَدَرِ حَان، فِي يَدِهِ صَفْرَاءُ تَرْنَمُوت ، كَأَنَّهَا تَقُولُ لِلرَّمِيِّ: مُتْ، وَيْبَكَ؛ فَيَمُوت"
أبو العلاء المَعَرّيّ
رسائل أبي العلاء المعريّ: أبو العلاء المَعَرّيّ (ت، 449هـ)، شرح: شاهين أفندي عطية اللبناني، وقف على طبعها: أحمد أفندي عباس الأزهري، المطبعة الأدبيّة، بيروت، (1894م)، ‏192‏.

280هـ=893م

الرَّنْمَةُ : المَرَّةُ مِنَ التَّرَنُّمِ.

"وَسَمِعْتُ رَنْمَةً حَسَنَةً"
ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏2/‏802‏.

تَرَنَّمُوت[صِفَة]

ن321هـ=933م

التَّرَنَّمُوتُ مِنَ الأَقْوَاسِ: الَّتِي لَهَا رَنِينٌ بَعْدَ الرَّمْيِ.

"قَوْسٌ تَرَنَّمُوتٌ ، بِتَشْدِيدِ النُّونِ؛ إِذَا كَانَ لَهَا حَنِينٌ بَعْدَ الرَّمْيِ"

مُرَنَّم[اسْمُ مَفْعُول]

390هـ=1000م

المُرَنَّمُ : الَّذِي اعْتَرَاهُ طَرَبٌ.

تُصْحِلُ صَوْتَ الجُنْدَبِ المُرَنَّمِ
من شواهد ابن الحدّاد المَعافريّ السَّرَقُسْطيّ
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏3/‏423‏.
ن750هـ=1349م

المُرَنَّمُ مِنَ الشِّعْرِ: مَا اشْتَرَكَ فِيهِ الإِعْرَابُ وَاللَّحْنُ، خِلَافًا لِلْمُوشَّحِ وَالزَّجَلِ.

"وَبَاقِي البِلَادِ لَا يَعْتَبِرُونَ هَذَا الفَرْقَ، بَلْ يُسَمُّونَ كُلَّ مَا أُعْرِبَ: مُوَشَّحًا، وُكُلَّ مَا خَلَا مِنَ الإِعْرَابِ: زَجَلًا، وَمَا اشْتَرَكَ فِيهِ الإِعْرَابُ وَاللَّحْنُ: مُرَنَّمًا "
صَفيّ الدِين الحِليّ
العاطِل الحالي والمرخص الغالي: صَفيّ الدِين الحِليّ (ت، 750هـ)، تحقيق: حسين نصّار، مطبعة دار الكتب والوثائق القوميَّة، القاهرة، ط2، (1424ه/ 2003م)، ‏8‏.
ن1348هـ=1929م

المُرَنَّمُ مِنَ الأَصْوَاتِ: المُحَسَّنُ التَّنْغِيمِ.

"وَهَذِهِ الفِيَلَةُ، وَهِيَ مِنْ أَكْبَرِ الحَيَوَانِ أَجْسَامًا، وَأَشَدِّهَا بَطْشًا، إِذَا سَمِعَتْ صَوْتًا مُرَنَّمًا ، أَوْ كَلَامًا مُنَغَّمًا، لَمْ يَلْبَثْ هَذَا الجِسُمُ العَظِيمُ أَنْ يَتَمَايَلَ تَرَنُّحًا، وَيَهْتَزَّ طَرَبًا"
محمد المويلحي
حديث عيسى بن هشام: محمد المويلحي (ت، 1348هـ)، دار المعارف، القاهرة، ط7، (1947م)، ‏146‏.

تَرْنِيمَة[اسْم]

ن724هـ=1324م

التَّرْنِيمَةُ : أُنْشُودَةٌ دِينِيَّةٌ تُنْشَدُ فِي الكَنَائِسِ أَثْنَاءَ القُدَّاسِ، وَمَوْضُوعُهَا الِابْتِهَالُ، وَالشُّكْرُ، وَقَدْ تَكُونُ مَصْحُوبَةً بِعَزْفٍ مُوسِيقِيٍّ خَاصٍّ. [الديانة المسيحية]

"لِأَنَّ مَعَانِيَ الكُتُبِ الإِلَهِيَّةِ، وَالمَزَامِيرِ الدَّاودِيَّةِ، تُؤَثِّرُ فِي العُقُولِ وَالنُّفُوسِ، وَتَقَاسِيمَ التَّرَانِيمِ البِيعِيِّةِ تُؤَثِّرُ فِي قُوَى المَحْسُوسِ"
أبو البركات ابن كَبَر
مصباح الظلمة في إيضاح الخدمة: أبو البركات ابن كَبَر (ت، 724هـ)، مكتبة الكاروز، القاهرة، (1971م)، ‏2/‏293‏.

مِرْنَام[صِيغَةُ مُبَالَغَة]

ن1081هـ=1670م

المِرْنَامُ : الكَثِيرُ التَّرَنُّمِ.

تَنَفَّسَتْ نَسَمَاتُ السِّحْرِ فِي السَّحَرِ وَهَيْنَمَتْ فِي مَيَادِينٍ مِنَ الزَّهَرِ
فنَبَّهَتْ كُلَّ مِرْنَامِ الضُّحَى غَرِدٍ حَالِي الطُّلَا، مُخْضَبِ الكَفَّيْنِ، مُسْتَحِرِ
عبد الرّحمن بن محمَّد الحسيني
ديوان ابن النّقيب: عبد الرّحمن بن محمَّد الحسيني (ت، 1081هـ)، حققه: عبد الله الجبوري، راجعه وأشرف على طبعه: أحمد الجندي، مطبوعات المجمع العلمي العربي، دمشق، (1383ه/ 1963م)، ‏132‏-‏133‏.

رَنُوم[صِفَةٌ مُشَبَّهَة]

1270هـ=1853م

الرَّنُومُ مِنَ النِّسَاءِ: المُغَنِّيَةُ المُجِيدَةُ.

"وَأَشْجَى مِنَ الرُّنُمِ ، وَمِنْ صَوْتِ كُلِّ دُعْبُبٍ غَرِيضٍ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏460‏.

مُرَنِّم[اسْمُ فَاعِل]

1282هـ=1865م

المُرَنِّمُ مِنَ الأَشْخَاصِ وَنَحْوِهِمْ: المُؤَدِّي لِلتَّرَانِيمِ المُنَغَّمَةِ.

"مَسِيحِ إِلهِ يَعْقُوبَ، وَمُرَنِّمِ إِسْرَائِيلَ الحُلْوِ"
كرنيليوس فان دايك وآخرون
الكتاب المقدس (أي كتب العهد القديم والعهد الجديد): كرنيليوس فان دايك وآخرون (ت، 1313هـ)، ترجمة: الجمعية الأميركانية، المطبعة الأميركانية، بيروت، (1910م)، ‏1/‏475‏.

رَنَّام[صِيغَةُ مُبَالَغَة]

1291هـ=1873م

الرَّنَّامُ : الكَثِيرُ التَّرَنُّمِ.

هَذَا صَدَى لُبْنَانَ أَصْبَحَ هَاتِفًا بِالحَمْدِ نَحْوَكَ صَوْتُهُ الرَّنَّامُ
يوسف بن فارس بن يوسف الخوري الشَّلْفُون
أنيس الجليس: يوسف بن فارس بن يوسف الخوري الشَّلْفُون (ت، 1314هـ)، المطبعة الكلية، بيروت، ط1، (1291ه/ 1874م)، ‏32‏.
