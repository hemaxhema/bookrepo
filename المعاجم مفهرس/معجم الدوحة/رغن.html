مُرْغِن[اسْمُ فَاعِل]

ن125هـ=743م

المُرْغِنُ : المُصْغِي رَاضِيًا بِمَا يُصْغِي إِلَيْهِ، قَابِلًا لَهُ.

مُرْغِنَاتٌ لِأَخْلَجِ الشِّدْقِ، سِلْعَا مٍ، مُمَرٍّ، مَفْتُولَةٍ عَضُدُهْ
الطِّرمّاح
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏3/‏341‏.
126هـ=744م

المُرْغِنُ : الطَّامِعُ فِي الشَّيْءِ.

أَلْيَسَ، مَلْوِيِّ المَلَاوِي، مِثْفَنِ، يَشْتَقُّ، أَوْ يَدْنُو دُنُوَّ المُرْغِنِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏123‏.

إِرْغَان[مَصْدَر]

ن175هـ=791م

الإِرْغَانُ : الإِصْغَاءُ عَنْ رِضًى وَقَبُولٍ.

وَأُخْرَى تُصَفِّقُهَا كُلُّ رِيحٍ سَرِيعٍ لَدَى الحَوْرِ إِرْغَانُهَا
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏405‏.

رَغَنَ[مُتَعَدٍّ بِالحَرْف]

ن175هـ=791م

رَغَنَ لَهُ: أَصْغَى إِلَيْهِ قَابِلًا رَاضِيًا.

"أَرْغَنَ فُلَانٌ لِفُلَانٍ، أَيْ: أَصْغَى قَابِلًا رَاضِيًا، وَفِي لُغَةٍ: رَغَنَ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏404‏.

أَرْغَنَ[مُتَعَدٍّ]

ن216هـ=831م

أَرْغَنَهُ بِالشَّيْءِ: أَطْمَعَهُ فِيهِ.

أَرْغَنَتْهُمْ ضَرْعَهَا الدُّنْـ ـيَا، وَقَامَتْ تَتَفَكَّكْ
من رواية الأصمعيّ
رد العامي إلى الفصيح: أحمد رضا العاملي (ت، 1371هـ)، دار العرفان، صيدا، (1371ه/ 1952م)، ‏305‏.
ن418هـ=1027م

أَرْغَنَ الشَّخْصَ: أَمَالَهُ وَسَكَّنَهُ مِنَ السُّكْرِ وَنَحْوِهِ.

"مُصَدِّعٌ يَضُرُّ الرَّأْسَ، وَيُشْرَبُ بِالمِيبَخْتَجِ لِلخُمَارِ، وَهُوَ مُنَوِّمٌ مُظْلِمٌ لِلَحَوَاسِّ، إِذَا سُقِيَ فِي الشَّرَابِ أَسْكَرَ حَتَّى يُرْغِنَ "
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏1/‏464‏.

رَغْن[مَصْدَر]

ن231هـ=845م

الرَّغْنُ : الأَكْلُ وَالشُّرْبُ فِي نِعْمَةٍ.

"يُقَالُ: هَذَا يَوْمُ رَغْنٍ ؛ إِذَا كَانَ ذَا أَكْلٍ، وَشُرْبٍ، وَنَعِيمٍ"
أبو عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏101‏.

رَاغِن[اسْمُ فَاعِل]

579هـ=1183م

الرَّاغِنُ : الطَّامِعُ فِي الأَمْرِ.

"ثُمَّ عُدْنَا فِيهِ رَاغِبِين، وَلِلْفَتْكِ بِهَا رَاغِنِين "
عماد الدين الأصفهانيّ
خريدة القصر وجريدة العصر: عماد الدين الأصفهانيّ (ت، 597هـ)، تحقيق: مبروك المناعي وآخرين، دار مداد يونيفارسيتي برس، قسنطينة، ط3، (2018م)، ‏13/‏394‏.
