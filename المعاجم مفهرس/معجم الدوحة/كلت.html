كِلِّيت[اسْم]

ن20هـ=641م

الكِلِّيتُ : حَجَرٌ مُسْتَطِيلٌ كَالبِرْطِيلِ، يُسْتَرُ بِهِ وِجَارُ الضَّبُعِ.

وَصَاحِبٍ صَاحَبْتُهُ زِمِّيتِ مُنْقَذِفٍ بِالقَوْمِ كَالكِلِّيتِ يُرَاقِبُ النَّجْمَ رِقَابَ الحُوتِ
أبو محمّد الحَذْلَميّ الفَقْعَسيّ الأسديّ
ما تبقّى من أراجيز أبي محمّد عبد الله بن ربعي بن خالد الحَذْلَميّ الفَقْعَسيّ الأسديّ: جمعها وحققها وشرحها: محمّد جبار المعيبد، وزارة الثقافة والإعلام، بغداد، ط1، (2001م)، ‏22‏.

اكْتَلَتَ[مُتَعَدٍّ]

ن207هـ=822م

اكْتَلَتَ النَّبِيذَ وَنَحْوَهُ: شَرِبَهُ.

"خُذْ هَذَا الإِنَاءَ، فَاقْمَعْهُ فِي فَمِهِ، ثُمَّ اكْلِتْهُ فِي فِيهِ، فَإِنَّهُ يَكْتَلِتُهُ "
من رواية أبي زكريّا الفَرّاء
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏10/‏138‏.
ن348هـ=959م

اكْتَلَتَ الشَّيْءَ: اسْتَوْعَبَهُ.

" اكْتَلَتُّ مَا فِي يَدِهِ"
أحمد بن محمّد الخارْزَنْجيّ
نصوص من كتاب تكملة العين للخارْزَنْجيّ جمع وتوثيق ودراسة: أحمد بن محمّد الخارْزَنْجيّ (ت، 348هـ)، تحقيق: عامر باهر أسمير الحيالي، مجلة أبحاث كليّة التّربية الأساسيّة، جامعة الموصل، مج4، ع4، (2007م)، ‏182‏.

كَلَتَ[مُتَعَدٍّ]

ن207هـ=822م

كَلَتَ الإِنَاءَ: صَبَّ مَا فِيهِ مِنْ شَرَابٍ وَنَحْوِهِ.

"خُذْ هَذَا الإِنَاءَ، فَاقْمَعْهُ فِي فَمِهِ، ثُمَّ اكْلِتْهُ فِي فِيهِ، فَإِنَّهُ يَكْتَلِتُهُ"
من رواية أبي زكريّا الفَرّاء
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏10/‏138‏.
398هـ=1007م

كَلَتَ الشَّيْءَ: جَمَعَهُ.

"وَالدُّنْيَا أُمُّ مُقْلِت، تَعُوقُ الإِنْسَانَ وَتَأْلِت... وَتَأْخُذُ مِنْهُ مَا يَكْلِت "
أبو العلاء المَعَرّيّ
الفصول والغايات في تمجيد الله والمواعظ: أبو العلاء المَعَرّيّ (ت، 449هـ)، صنعه وفسر غريبه: محمود حسن زناتي، مراجعة: لجنة إحياء التراث العربي، دار الآفاق الجديدة، بيروت، د.ت. ‏285‏.
