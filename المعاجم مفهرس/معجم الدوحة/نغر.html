نَغَرَ[لَازِم]

ن5هـ=626م

نَغَرَتِ الخَمْرُ وَنَحْوُهَا: غَلَتْ، وَفَارَتْ.

تُصْفَقُ الرَّاحُ وَالرَّحِيقُ عَلَيْهِمْ فِي دِنَانٍ مَصْفُوفَةٍ وَقِلَالِ
وَأَبَارِيقَ تَنْغِرُ الخَمْرُ فِيهَا وَرَحِيقٍ مِنَ الفُرَاتِ الزُّلَالِ
أُمَيّة بن أبي الصَّلْت الثَّقَفيّ
شرح القصائد السّبع الطّوال الجاهليّات: أبو بكر الأنباريّ (ت، 328هـ)، تحقيق وتعليق: عبد السلام هارون، دار المعارف، القاهرة، ط5، (1993م)، ‏110‏.
175هـ=791م

نَغَرَتِ النَّاقَةُ: ضَمَّتْ مُؤَخَّرَهَا، وَمَضَتْ.

وَعُجُزٌ تَنْغَرُ لِلتَّنْغِيرِ
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏405‏.
ن320هـ=932م

نَغَرَ الشَّخْصُ: اغْتَاظَ بِشِدَّةٍ.

"أَمَا إِنَّ أَهْلَكَ قَدْ أَسْلَمُوا، وَتَرَكُوكَ وَمَا أَنْتَ عَلَيْهِ، فَلَمَّا سَمِعَ ذَلِكَ نَغَرَ ، وَقَالَ: أَيُّهُمْ؟ قَالَ: خَتْنُكَ، وَابْنُ عَمِّكَ، وَأُخْتُكَ"
إبراهيم بن محمّد البيهقيّ
كتاب المحاسن والمساوي: إبراهيم بن محمّد البيهقيّ (ت، 320هـ)، وقف على طبعه: فريدريك شوالي، ي. ريكر، مدينة ‏ليبسيغ، (1320هـ/ 1902م)، ‏71‏.
330هـ=941م

نَغَرَ الجُرْحُ وَنَحْوُهُ: سَالَ مِنْهُ الدَّمُ.

وَإِنَّ الجُرْحَ يَنْغَرُ بَعْدَ حِينٍ إِذَا كَانَ البِنَاءُ عَلَى فَسَادِ
أبو الطَّيّب المُتَنَبّي
الأمثال السّائرة من شعر المتنبّيّ والرّوزنامجة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، مكتبة النهضة- مكتبة المعارف، بغداد، ط1، (1385هـ/ 1965م)، ‏25‏.

نَغَرَ[مُتَعَدٍّ بِالحَرْف]

ن175هـ=791م

نَغَرَ بِهِ: صَاحَ بِهِ.

" وَنَغَرْتُ بِهَا: صِحْتُ بِهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏405‏.

نُغَيْر[اسْمٌ مُصَغَّر]

ن11هـ=632م

النُّغَيْرُ : تَصْغِيرُ النُّغَرِ، وَهُوَ فَرْخُ العُصْفُورِ.

"يَا أَبَا عُمَيْرٍ، مَا فَعَلَ النُّغَيْرُ ؟"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏21/‏456،،‏(رقم الحديث: 14071).

نُغَر[اسْمُ جِنْس]

11هـ=632م

النُّغَرُ : نَوْعٌ مِنَ العَصَافِيرِ، حُمْرُ المَنَاقِيرِ.

"مَا لِي أَرَى أَبَا عُمَيْرٍ حَزِينًا؟ فَقَالُوا: مَاتَ نُغَرُهُ الَّذِي كَانَ يَلْعَبُ بِهِ"
بعض الصّحابة
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏20/‏282،،‏(رقم الحديث: 12957).

نُغَر[اسْمُ جَمْع]

ن175هـ=791م

النُّغَرُ : أَوْلَادُ الحَوَامِلِ إِذَا صَوَّتَتْ، وَبَدَا حَجْمُهَا فِي البَطْنِ كَالوَزَغِ.

" وَالنُّغَرُ : أَوْلَادُ الحَوَامِلِ إِذَا صَوَّتَتْ وَوَزَغَتْ؛ أَيْ: يَتَبَيَّنُ فِي بَطْنِهَا كَالوَزَغِ فِي خِلْقَتِهِ فِي الصِّغَرِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏405‏.

نَغْرَة[اسْمُ مَرَّة]

11هـ=632م

النَّغْرَةُ : المَرَّةُ مِنْ شِدَّةُ الغَيْظِ.

"ثُمَّ لَمْ تَلْبَثُوا إِلَّا رَيْثَ أَنْ تُسَكِّنَ نَفْسٌ نَغْرَتَهَا "
فاطمة بنت محمّد رسول الله
التّذكرة الحمدونيّة: ابن حمدون الكاتب (ت، 562هـ)، تحقيق: إحسان عبّاس، بكر عبّاس، دار صادر، بيروت، ط1، (1996م)، ‏6/‏257‏.
280هـ=893م

نَغْرَةُ الرِّيحِ وَنَحْوِهَا: المَرَّةُ مِنْ هُبُوبِهَا.

"وَيُقَالُ لِلرِّيحِ إِذْ هَبَّتْ، ثُمَّ سَكَنَتْ: هَذِهِ نَغْرَةُ نَجْمِ كَذَا وَكَذَا"
ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏3/‏1299‏.
393هـ=1003م

النَّغْرَةُ مِنَ الجُرْحِ وَنَحْوِهِ: المَرَّةُ مِنْ سَيَلَانِهِ.

وَكَأَنِّي، خِلَلَ الغَيْبِ، أَرَى نَغْرَةً مِنْ جُرْحِهَا، بَعْدَ انْدِمَالِ
الشّريف الرّضيّ
ديوان الشريف الرضي: دار صادر- دار بيروت، بيروت، (1380هـ/ 1961م)، ‏2/‏198‏.

نَغِر[صِفَةٌ مُشَبَّهَة]

40هـ=660م

النَّغِرُ مِنَ الأَشْخَاصِ: الَّذِي يَغْلِي جَوْفُهُ مِنَ الغَيْظِ وَالغَيْرَةِ.

"رُدُّونِي إِلَى أَهْلِي غَيْرَى نَغِرَةٌ "
امرأة من المسلمين
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏17/‏245،،‏(رقم الحديث: 17161).

نَغَّار[صِيغَةُ مُبَالَغَة]

ن126هـ=744م

النَّغَّارُ مِنَ العُرُوقِ: السَّيَّالُ بِالدَّمِ.

وَعَاثَ فِيهِنَّ مِنْ ذِي لَيَّةٍ نَتَقَتْ أَوْ نَازِفٌ مِنْ عُرُوقِ الجَوْفِ نَغَّارُ
الكميت بن زيد الأسديّ
شعر الكُمَيْت بن زيد الأسدي: جمع وتقديم: داود سلوم، جامعة بغداد- مكتبة الأندلس، بغداد، مطبعة النعمان، جنيف، (1969م)، ‏1/‏174‏.
1071هـ=1660م

النَّغَارُ : الشَّدِيدُ غَلَيَانِ الصَّدْرِ بِالغَيْظِ وَنَحْوِهِ.

فَمَا مُطَوَّقَةٌ غِيلَتْ بِوَاحِدِهَا أَشْجَى، وَقَدْ نُغِرَتْ مِنِّي بِنَغَّارِ
شهاب الدين أحمد بن محمد الآنسي
لذة الوسن: الكوكباني (ت، 1071هـ)، تحقيق ودراسة: نهى عبد الرّازق الحفناوي، دار الكتب العلمية، بيروت، ط1، (1441ه/ 2020م)، ‏163‏.

يُوَقِّرُ النَّفْسَ عَلَى تَوْقِيرِهَا يَعْلَمُ أَلَّا شَيْءَ فِي تَنْغِيرِهَا
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏214‏.

وَعُجُزٌ تَنْغَرُ لِلتَّنْغِيرِ

ن167هـ=783م

النَّغِيرُ مِنَ الأَشْخَاصِ: المُمْتَلِئُ غَيْظًا.

وَإِذْ سُلَيْمَى سَجِنٌ مَحْصُورُ لَيْسَ عَلَيْنَا فِي الهَوَى نَغِيرُ

نَغَر[اسْم]

178هـ=794م

النَّغَرُ فِي الشَّاءِ وَنَحْوِهَا: دَاءٌ يُخَالِطُ اللَّبَنَ فِيهِ دَمٌ.

"أَنَّ الشَّاةَ وَالنَّاقَةَ تَبْرُكُ عَلَى نَدًى، فَيَخْرُجُ اللَّبَنُ كَقِطَعِ الأَوْتَارِ أَحْمَرَ، فَيُقَالُ لِذَلِكَ الدَّاءِ: النَّغَرُ ، وَالمَغَرُ"
الأصمعيّ
كتاب النّوادر في اللّغة: أبو زيد الأنصاريّ (ت، 215هـ)، تحقيق ودراسة: محمّد عبد القادر أحمد عطا، دار الشروق، بيروت- القاهرة، ط1، (1401هـ/ 1981م)، ‏291‏.
178هـ=794م

النَّغَرُ : اللَّبَنُ المُخْتَلِطُ فِي الضَّرْعِ بِالدَّمِ.

"فَإِذَا حُلِبَتِ الشَّاةُ كَانَ أَوَّلُ شَخْبَةٍ تَشْخَبُهَا فِي الأَرْضِ؛ مَخَافَةَ أَنْ تَكُونَ نَغَرًا فَاسِدًا"
أعرابيٌّ
كتاب النّوادر في اللّغة: أبو زيد الأنصاريّ (ت، 215هـ)، تحقيق ودراسة: محمّد عبد القادر أحمد عطا، دار الشروق، بيروت- القاهرة، ط1، (1401هـ/ 1981م)، ‏290‏.

نَغَر[مَصْدَر]

198هـ=813م

النَّغَرُ : غَلَيَانُ الصَّدْرِ بِالغَيْظِ وَنَحْوِهِ.

"قَدْ أَفْرَخَ اللهُ عَنْهُمْ نَغَرَ التَّجَارِبِ وَالتَّجَاذُبِ، وَجَعَلَ مَا كَانَ يَسْعَى بِهِ بَعْضُهُمْ مِنَ الإِعْدَادِ لِبَعْضٍ زِيَادَةً فِي رِيحِهِم، وَحَدًّا فِي شَوْكَتِهِمْ"
أحمد بن يوسف الكاتب
جمهرة رسائل العرب في عصور العربية الزاهرة العصر الجاهلي عصر صدر الإسلام: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1356هـ/ 1937م- 1357هـ/ 1939م)، ‏3/‏326‏.
ن891هـ=1485م

النَّغَرُ : النُّكْرَانُ وَالجُحُودُ.

"وَنَعُوذُ بِاللهِ مِنْ شَرِّ مَنْ رَضِيَ لِنَفْسِهِ بِالنَّغَرِ وَالخِلَاف، وَكَانَتْ قُصَارَى هِمَّتِهِ عَدَمَ الإِنْصَاف"
أبو الحسن علي بن محمد بن علي القرشي البسطي الشهير بالقلصادي
رحلة القلصادي: أبو الحسن علي بن محمد بن علي القرشي البسطي الشهير بالقلصادي (ت، 891هـ)، دراسة وتحقيق: محمد أبو الأجفان، الشركة التونسية، تونس، (1978م)، ‏83‏.

تَنَغَّرَ[مُتَعَدٍّ بِالحَرْف]

ن224هـ=839م

تَنَغَّرَ عَلَى الشَّخَصِ: غَلَى جَوْفُهُ عَلَيْهِ غَيْظًا.

"وَيُقَالُ مِنْهُ: رَأَيْتُ فُلَانًا يَتَنغَّرُ عَلَى فُلَانٍ"
من رواية أبي عُبيد الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏100‏.
375هـ=985م

تَنَغَّرَ عَلَى الشَّخْصِ: تَنَكَّرَ لَهُ، وَتَوَعَّدَهُ.

"حَتَّى إِذَا اسْتَقَى مَا عِنْدَ ذَلِكَ الإِنْسَانِ بِهَذِهِ الزَّخَارِفِ وَالحِيَلِ، وَسَالَ الرَّجُلُ مَعَهُ فِي حُدُورِهِ عَلَى مَذْهَبِ الثِّقَةِ، وَرَكِبَ فِي مُنَاظَرَتِهِ... وَشاكَعَهُ، وَوَضَعَ يَدَهُ عَلَى النُّكْتَةِ الفَاصِلَةِ، وَالأَمْرِ القَاطِعِ، تَنَمَّرَ لَهُ، وَتَنَغَّرَ عَلَيْهِ"
أبو حيّان التَّوحيديّ
أخلاق الوزيرين "مثالب الوزيرين الصاحب ابن عباد وابن العميد": أبو حيّان التَّوحيديّ (ت، 414هـ)، حققه وعلق حواشيه: محمّد بن تاويت الطنجي، دار صادر، بيروت، (1412هـ/ 1992م)، ‏111‏-‏ 112‏.

تَنَاغُر[مَصْدَر]

ن228هـ=843م

التَّنَاغُرُ : التَّغَايُظُ وَالتَّخَاصُمُ.

"وَكَانَ بَيْنَهُمَا تَنَاغُرٌ "
ابن عائشة التَّيْميّ القُرَشيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏4/‏72‏.

نَاغِر[اسْمُ فَاعِل]

388هـ=998م

النَّاغِرُ مِنَ العُرُوقِ وَنَحْوِهَا: الَّذِي يَسِيلُ مِنْهُ الدَّمُ.

أَسْعَى عَلَى أَثَرِ النَّوَائِبِ مُنْصِفًا مِنْهَا، وَآسِي كُلَّ عِرْقٍ نَاغِرِ
الشّريف الرّضيّ
ديوان الشريف الرضي: دار صادر- دار بيروت، بيروت، (1380هـ/ 1961م)، ‏1/‏481‏.
ن406هـ=1015م

النَّاغِرُ مِنَ القُدُورِ وَنَحْوِهَا: الَّذِي يَغْلِي بِمَا فِيهِ.

وَرَاحَتْ طِرَابًا، لَمْ تُشَمِّسْ رِحَالَهَا وَلَا نَغَرَتْ مِنْهَا القُدُورُ النَّوَاغِرُ
الشّريف الرّضيّ
ديوان الشريف الرضي: دار صادر- دار بيروت، بيروت، (1380هـ/ 1961م)، ‏1/‏533‏.
412هـ=1021م

النَّاغِرُ مِنَ الأَشْيَاءِ: الحَارُّ السَّاخِنُ.

"وَيُزِيلُونَ الغَيْمَ الغَالِبَ عَلَى الأَفْئِدَةِ بِالقَارِسِ فِي حَرِّ الظَّهِيرَةِ، وَيُذْهِبُونَ الدَّرَنَ بِالنَّاغِرِ مِنْ ذَاتِ الزُّبَرِ وَالصَّرَفَانِ"
أبو العلاء المَعَرّيّ
رسالة الصّاهل والشّاحج: أبو العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: عائشة عبد الرحمن بنت الشاطئ، دار المعارف، القاهرة، ط2، (1404هـ/ 1984م)، ‏526‏.

مُنْغِر[صِفَةٌ مُشَبَّهَة]

398هـ=1007م

المُنْغِرُ مِنَ النُّوقِ وَنَحْوِهَا: الَّتِي خَالَطَ لَبَنَهَا دَمٌ.

"المُلْكُ لَكَ، غَالِبَ الغَالِبِينَ، لَوْ شِئْتَ لَجَعَلْتَنِي رَاعِيَ فِرْقٍ، أَرْقُبُ ثَرَّتَهُ وَالعَزُوزَ... أَوْ صَاحِبَ هَجْمَةٍ أَتَلَكَّدُ بِهَا أُنُوفَ الكَلَأِ، هِمَّتِي فِي المُنْغِرَةِ وَالمِخْزَابِ"
أبو العلاء المَعَرّيّ
الفصول والغايات في تمجيد الله والمواعظ: أبو العلاء المَعَرّيّ (ت، 449هـ)، صنعه وفسر غريبه: محمود حسن زناتي، مراجعة: لجنة إحياء التراث العربي، دار الآفاق الجديدة، بيروت، د.ت. ‏67‏.

مُنْغِر[اسْمُ فَاعِل]

1373هـ=1953م

المُنْغِرُ مِنَ الأُمُورِ: المُقْلِقُ المُزْعِجُ.

"كَمَا بِسَبَبِ الفِكْرَةِ المُنْغِرَةِ الَّتِي طَرَأَتْ عَلَى رَأْسِهِ مِنْ أَنَّ كُتْلَةَ الأَشْخَاصِ ذَوِي النُّفُوذِ فِي الأَرْكَانِ سَتُصْبِحُ فِي غَلَيَانٍ لَدَى اطِّلَاعِهَا عَلَى الأَنْبَاءِ"
مجموعة من المترجمين
الحرب والسلم، لليو تولستوي: مجموعة مؤلّفين ، ترجمة: مجموعة مترجمين، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2020م)، ‏4/‏167‏.

نَغَّرَ[مُتَعَدٍّ]

400هـ=1009م

نَغَّرَ النَّاقَةَ وَنَحْوَهَا: صَاحَ بِهَا.

نَغَّرْتُهَا ، وَالبَدْرُ مُطَّلِعٌ حَتَّى اسْتَجَابَ لِقَائِدِ الأُفْلِ
الشّريف الرّضيّ
ديوان الشريف الرضي: دار صادر- دار بيروت، بيروت، (1380هـ/ 1961م)، ‏2/‏164‏.
ن1270هـ=1853م

نَغَّرَ الصَّبِيَّ وَنَحْوَهُ: دَغْدَغَهُ.

"وَمَا الفَائِدَةُ مِنْ وُقُوفِ الخُوَيْدِمِ بَيْنَ يَدَيْكَ وَفِي حِزَامِهِ المِلْعَقَةُ أَوْ عَلَى رَأْسِهِ الخِوَانُ... وَأَنْتَ لَا تَأْكُلُ مَعَ السَّيِّدَةِ وَأَوْلَادِهَا، وَلَا تَأْخُذُ وَلَدَكَ وَتَضَعَهُ عَلَى رُكْبَتِكَ... وَلَا تُنَغِّرُهُ ، وَلَا تُرَخِّمُهُ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏459‏-‏ 460‏.

نَغِرَ[مُتَعَدٍّ بِالحَرْف]

ن538هـ=1143م

نَغِرَ الجُرْحُ عَلَيْهِ: انْتَكَأَ عَلَيْهِ، وَسَالَ دَمُهُ.

"خَرَجَتْ بِهِ بَثْرَةٌ، فَعَصَرَهَا، فَنَغِرَتْ عَلَيْهِ"
جار الله الزمخشريّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: محمد باسل عيون السود، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏1/‏44‏.

نَغَّارَة[اسْم]

1422هـ=2000م

النَّغَّارَةُ : إِحْدَى الآلَاتِ المُوسِيقِيَّةِ.

"وَيَزْدَهِرُ عِنْدَهُمُ الذِّكْرُ، وَيَصِيرُ فَنًّا قَائِمَ الذَّاتِ، يَعْتَمِدُ إِيقَاعَاتٍ مَخْصُوصَةً، وَتُسْتَعْمَلُ فِيهِ آلَاتُ الوَزْنِ؛ كَالرَّقِ وَالنَّغَّارَاتِ "
محمّد الكحلاويّ
السّماع والإنشاد بين الشّعر والموسيقى: ‏محمد الكحلاوي،‏مجلة الحياة الثقافية: الحياة الثقافية، تونس، ع113، (01/ 04/ 2000م)، ‏146‏.
