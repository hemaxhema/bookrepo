قَمْقَام[صِفَة]

قبل 50ق.هـ=573م

القَمْقَامُ : العَظِيمُ الشَّأْنِ.

فَاسْأَلْ لِكِنْدَةَ حِينَ أَقْبَلَ جَمْعُهَا حَوْلَ ابْنِ كَبْشَةَ وَابْنِ أُمِّ قَطَامِ
مَلِكَانِ قَدْ قَادَا الجُيُوشَ، وَأَثْخَنَا بِالقَتْلِ كُلَّ مُتَوَّجٍ قَمْقَامِ
الحارث بن عُباد البَكْريّ
ديوان الحارث بن عباد: جمعه وحققه: أنس عبد الهادي أبو هلال، هيئة أبو ظبي للثقافة والتراث، أبو ظبي، ط1، (1429هـ/ 2008م)، ‏227‏.
قبل 50ق.هـ=573م

القَمْقَامُ : الشُّجَاعُ الهُمَامُ.

مُتَبَارِيَاتٍ فِي الأَعِنَّةِ قُطَّبًا يَحْمِلْنَ كُلَّ مُنَازِلٍ قَمْقَامِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏123‏.
قبل 13ق.هـ=609م

القَمْقَامُ : المُتَضَامُّ المُجْتَمِعُ.

وَاللَّهِ، لَا زِلْتُ أَبْكِيهِ، وَأَنْدُبُهُ حَتَّى تَزُورَكَ أَخْوَالِي وَأَعْمَامِي
بِكُلِّ أَسْمَرَ لَدْنِ الكَعْبِ مُعْتَدِلٍ وَكُلِّ أَبْيَضَ صَافِي الحَدِّ قَمْقَامِ
الهيفاء بنت صُبيح القُضاعيّة
شاعرات العرب في الجاهلية والإسلام: جمعه ورتبه ووقف على طبعه: بشير يموت، المكتبة الأهلية- المطبعة الوطنية، بيروت، ط1، (1353هـ/ 1934م)، ‏41‏.
436هـ=1044م

القَمْقَامُ مِنَ الأُمُورِ: العَظِيمُ مِنْهُ.

"وَقَعَ فِي قَمْقَامٍ مِنَ الأَمْرِ"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏6/‏92‏.

قَمْقَام[اسْم]

قبل 50ق.هـ=573م

القَمْقَامُ : العَدَدُ الكَثِيرُ.

قَدْ قَرَّتِ العَيْنُ مِنْ عِمْرَانَ إِذْ قُتِلَتْ وَمِنْ عَدِيٍّ مَعَ القَمْقَامِ إِذْ جُهِدُوا
الحارث بن عُباد البَكْريّ
ديوان الحارث بن عباد: جمعه وحققه: أنس عبد الهادي أبو هلال، هيئة أبو ظبي للثقافة والتراث، أبو ظبي، ط1، (1429هـ/ 2008م)، ‏156‏.
قبل 50ق.هـ=573م

القَمْقَامُ : السَّيِّدُ الكَثِيرُ الخَيْرِ، الوَاسِعُ الفَضْلِ.

فَيُعْطِي الَّتِي فِيهَا الغِنَى، وَكَأَنَّهُ لِتَصْغِيرِهِ تِلْكَ العَطِيَّةَ جَارِمُ
بِذَلِكَ أَوْصَاهُ عَدِيٌّ وَحَشْرَجٌ وَسَعْدٌ وَعَبْدُ اللهِ تِلْكَ القَمَاقِمُ
عبد قيس بن خُفاف البُرْجُميّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏8/‏175‏.
ن20هـ=641م

القَمْقَامُ : الجَمَاعَةُ.

وَجَعَلَتْ تَأْوِي إِلَى قَمْقَامِهَا وَانْصَرَفَتْ، وَالشَّمْسُ مِنْ أَمَامِهَا
أبو محمّد الحَذْلَميّ الفَقْعَسيّ الأسديّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏88‏.
ن40هـ=660م

القَمْقَامُ : مُقَدِّمَةُ الجَيْشِ.

وَفَالِقِ الهَامَاتِ وَالْمنَاكِبِ أَحْمِي بِهِ قَمَاقِمَ الكَتَائِبِ
عليّ بن أبي طالب
ديوان الإمام علي بن أبي طالب رضي الله عنه: اعتنى به: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط3، (1426هـ/ 2005م)، ‏35‏.
ن110هـ=728م

القَمْقَامُ : البَحْرُ، يُغْمَرُ فِي لُجَجِهِ مَنْ يَدْخُلُهُ.

وَحَسِبْتَ بَحْرَ بَنِي كُلَيْبٍ مُصْدِرًا فغَرِقْتَ حِينَ وَقَعْتَ فِي القَمْقَامِ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏555‏.
ن110هـ=728م

القَمْقَامُ : الجَيْشُ يُجْمَعُ مِنْ هَاهُنَا وَهَاهُنَا.

يَحُوطُكَ مَنْ يَحُوطُ ذِمَارَ قَيْسٍ وَمَنْ وَسَطَ القَمَاقِمَ مِنْ تَمِيمِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏588‏.

قَمْقَام[اسْمُ جِنْس]

ن175هـ=791م

القَمْقَامُ : صِغَارُ القِرْدَانِ.

"الحَمْنَانُ، الوَاحِدَةُ حَمْنَانَةٌ: صِغَارُ القِرْدَانِ... وَتَكُونُ حَمْنَانًا ثُمَّ قَمْقَامًا ثُمَّ قِرْدَانًا ثُمَّ حَلَمًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏253‏.

قُمَاقِم[صِفَة]

قبل 13ق.هـ=609م

القُمَاقِمُ مِنْ كُلِّ شَيْءٍ: العَظَيمُ الوَاسِعُ.

وَجَدْتُ أَبَاكَ الحُرَّ بَحْرًا بِنَجْوَةٍ بَنَاهَا لَهُ مَجْدٌ أشَمُّ قُمَاقِمُ
الرَّقّاص الكلبيّ
ديوان شُعراء بني كلب: صنعة: محمّد شفيق البيطار، دار صادر، بيروت، ط1، (2002م)، ‏1/‏99‏.

قُمَاقِم[اسْم]

ن37هـ=657م

القُمَاقِمُ : السَّيِّدُ الكَثِيرُ الخَيْرِ، الوَاسِعُ الفَضْلِ.

إِذَا الرِّفَاقُ أَنَاخُوا فِي مَبَاءَتِهِ حَلُّوا بِذِي فُجَرَاتٍ، زَنْدُهُ وَارِي
قُمَاقِمٍ ، بَارِعٍ، خَضَّامَةٍ، أُنُفٍ جَمِّ المَوَاهِبِ، بَدْءٍ، غَيْرِ عُوَّارِ
ابن مقبل العَجْلانيّ التَّميميّ
ديوان ابن مقبل: عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏97‏.

قُمْقُمَان[اسْم]

64هـ=684م

القُمْقُمَانُ : العَدَدُ الكَثِيرُ.

وَالرَّأْسُ مِنْ خُزَيْمَةَ الأَشَمُّ وَقَيْسُ عَيْلَانَ أَخٌ وَعَمُّ فَاجْتَمَعَ الخِضَمُّ وَالخِضَمُّ وَقُمْقُمَانٌ عَدَدٌ قُمْقُمُّ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏2/‏129‏.
ن110هـ=728م

قُمْقُمَانُ البَحْرِ: مُعْظَمُهُ وَلُجَّتُهُ.

وَلَوْ شِئْتُ غَمَّ التَّيْمَ عَمْروٌ وَمَالِكٌ وَطَمَّ عَلَيْهِمْ قُمْقُمَانٌ مِنَ البَحْرِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏596‏.

قُمْقُمّ[صِفَة]

64هـ=684م

القُمْقُمُّ مِنَ العَدَدِ: الكَثِيرُ.

فَاجْتَمَعَ الخِضَمُّ وَالخِضَمُّ وَقُمْقُمَانٌ عَدَدٌ قُمْقُمُّ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏2/‏129‏.

تَقَمْقُم[مَصْدَر]

ن90هـ=709م

تَقَمْقُمُ الشَّخْصِ لِقِرْنِهِ: اعْتِلَاؤُهُ لَهُ لِيَضْرِبَهُ.

يَقْتَسِرُ الأَقْرَانَ بِالتَّقَمْقُمِ
العجّاج
التكملة والذيل والصّلة لكتاب تاج اللّغة وصحاح العربيّة، الحسن بن محمد بن الحسن الصّغاني، عبد العليم الطّحاوي وآخرون، مطبعة دار الكتب، القاهرة، 1970- 1979/، ‏6/‏128‏.

تَقَمْقَمَ[لَازِم]

122هـ=740م

تَقَمْقَمَ الشَّخْصُ: ذَهَبَ فِي المَاءِ الكَثِيرِ، وَغُمِرَ حَتَّى غَرِقَ.

مَنْ خَرَّ فِي قَمْقَامِنَا تَقَمْقَمَا كَمَا هَوَى فِرْعَوْنُ إِذْ تَغَمْغَمَا
رؤبة بن العجّاج
شرح ديوان رؤبة بن العجّاج المنسوب لأبي سعيد الضَّرير: تقديم وتحقيق ودراسة: خالد حجوط، أطروحة دكتوراه، جامعة سيدي محمّد بن عبد الله، فاس، (1426هـ/ 2006م- 1407هـ/ 2007م)، ‏2/‏518‏.
ن385هـ=995م

تَقَمْقَمَ الغُرَابُ وَنَحْوُهُ: جَمَعَ طَعَامَهُ مِمَّا حَوْلَهُ.

"وَغُرَابُ البَيْنِ سُمِّيَ بِذَلِكَ لِأَنَّهُ إِذَا قَصَدَ أَهْلُ الدَّارِ لِلنُّجْعَةِ وَقَعَ فِي بُيُوتِهِمْ يَتَقَمْقَمُ "
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏10/‏407‏.

قُمْقُم[اسْم]

136هـ=753م

القُمْقُمُ : رَأْسُ الحُلْقُومِ النَّاتِئُ فِي الحَلْقِ.

إِذَا ثَنَى فَرْغَ اللَّهَاةِ قُمْقُمُهْ وَرَدَّها عُثْنُونُهُ وَغَلْصَمُهْ مُجَّ عَلَى هَامَاتِهِنَّ بَلْغَمُهْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏246‏-‏ 247‏.

" قَمْقَمَ اللهُ عَصَبَهُ"
ابن قتيبة
أدب الكاتب: ابن قتيبة (ت، 276هـ)، حققه وعلق حواشيه ووضع فهارسه: محمّد أحمد الدالي، مؤسسة الرسالة، بيروت، ط1، (1402هـ/ 1981م)، ‏48‏.
