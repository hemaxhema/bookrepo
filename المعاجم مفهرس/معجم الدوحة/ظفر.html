ظَفِرَ[لَازِم]

قبل 150ق.هـ=476م

ظَفِرَ الشَّخْصُ: فَازَ بِمَا أَرَادَ.

أُعَلِّمُهُ الرِّمَــايَةَ كُلَّ يَـــوْمٍ فَلَمَّا اسْتَدَّ سَاعِــدُهُ رَمَـانِي
فَلَا ظَفِرَتْ يَدَاهُ حِينَ يَرْمِي وَشَلَّتْ مِنْهُ حَامِلَـةُ البَنَـانِ
منسوب إلى مالك بن فَهْم الأزديّ
شُعراء عُمان في الجاهلية وصدر الإسلام: جمع وتحقيق: أحمد محمّد عبيد، المجمع الثقافي، أبو ظبي، (1420هـ/ 2000م)، ‏86‏-‏ 87‏.
ن175هـ=791م

ظَفِرَتِ العَيْنُ: غَشِيَ بَيَاضَهَاجُلَيْدَةٌ مِمَّا يَلِي الأَنْفَ إِلَى سَوَادِهَا.

"وَيُقَالُ: ظُفِرَ فُلَانٌ، فَهُوَ مَظْفُورٌ، وَعَيْنٌ ظَفِرَةٌ، وَقَدْ ظَفِرَتْ عَيْنُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏158‏.

ظَفِرَ[مُتَعَدٍّ بِالحَرْف]

قبل 50ق.هـ=573م

ظَفِرَ بِالشَّيْءِ: فَازَ بِهِ وَنَالَهُ.

لَنَا الغُرَفُ العُلْيَا مِنَ المَجْدِ وَالعُلَا ظَفِرْنَا بِهَا وَالنَّاسُ بَعْدُ تَوَابِعُ
عبيد بن عبد العُزَّى السَّلاميّ الأزديّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏8/‏279‏.
ن11هـ=632م

ظَفِرَ بِالخَصْمِ: تَمَكَّنَ مِنْهُ وَغَلَبَهُ.

"إِنِّي قَدْ كُنْتُ أَمَرْتُكُمْ بِتَحْرِيقِ هَذَيْنِ الرَّجُلَيْنِ، ثُمَّ رَأَيْتُ أَنَّهُ لَا يَنْبَغِي لأَحَدٍ أَنْ يُعَذِّبَ بِالنَّارِ إِلَّا اللهُ، فَإِنْ ظَفِرْتُمْ بِهِمَا فَاقْتُلُوهُمَا"
حديث نبويّ
مُسند الدّارمي المعروف بـ(سنن الدارمي): الدّارمي (ت، 255هـ)، تحقيق: حسين سليم أسد الداراني، دار المغني، الرياض، ط1، (1421هـ/ 2000م)، ‏3/‏1599،،‏(رقم الحديث: 2504).

ظَفَر[مَصْدَر]

قبل 150ق.هـ=476م

الظَّفَرُ : الفَوْزُ وَالغَلَبَةُ.

شُدُّوا وَثُورُوا جَمِيعًا فِي وُجُوهِهِمُ فَإِنَّ ذَلِكَ مِنْكُمْ، فَاعْلَمُوا، ظَفَرُ
منسوب إلى زرقاء اليمامة
أسجاع الكهَّان الجاهليِّين وأشعارهم جمع وتحقيق ودراسة: إعداد: ياسين عبد الله جمول، رسالة ماجستير، جامعة دمشق، (2011- 2012م)، ‏410‏.
ن189هـ=805م

ظَفَرُ العَيْنِ: نُتُوءُ جُلَيْدَةٍ تُغَشِّي بَيَاضَهَا مِمَّا يَلِي الأَنْفَ إِلَى سَوَادِهَا.

"وَالحَوَلُ عَيْبٌ، وَالحَوَصُ عَيْبٌ، وَالظَّفَرُ عَيْبٌ، وَالشَّعْرُ يَكُونُ فِي جَوْفِ العَيْنِ عَيْبٌ"
محمّد بن الحسن الشَّيبانيّ
الأصل: محمّد بن الحسن الشَّيبانيّ (ت، 189هـ)، تحقيق ودراسة: محمّد بوينوكالن، وزارة الأوقاف والشؤون الإسلامية، الدوحة، دار ابن حزم، بيروت، ط1، (1433هـ/ 2012م)، ‏2/‏499‏.

ظَافِر[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الظَّافِرُ بِالشَّيْءِ: الفَائِزُ بِهِ، الحَاصِلُ عَلَيْهِ.

طَرَحْتُ لَهُ نَعْلًا مِنَ السِّبْتِ طَلَّةً خِلَافَ نَدًى مِنْ آخِرِ اللَّيْلِ مُخْضِلِ
فَوَلَّى بِهَا جَذْلَانَ يَنْفُضُ رَأْسَهُ كَصَاحِبِ غُنْمٍ ظَافِرٍ بِالتَّمَوُّلِ
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏185‏.
36هـ=656م

الظَّافِرُ : المُنْتَصِرُ الغَالِبُ.

"أَلَا تَرْبَعُ أَيُّهَا الإِنْسَانُ عَلَى ظَلْعِك، وَتَعْرِفُ قُصُورَ ذَرْعِك، وَتَتَأَخَّرُ حَيْثُ أَخَّرَكَ القَدَرُ؟ فَمَا عَلَيْكَ غَلَبَةُ المَغْلُوبِ، وَلَا لَكَ ظَفَرُ الظَّافِرِ "
عليّ بن أبي طالب
جمهرة رسائل العرب في عصور العربية الزاهرة العصر الجاهلي عصر صدر الإسلام: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1356هـ/ 1937م- 1357هـ/ 1939م)، ‏1/‏394‏.

ظُفْر[اسْم]

قبل 50ق.هـ=573م

الظُّفْرُ : مَا يُغَطِّي أَعْلَى طَرَفِ الإِصْبَعِ مِنَ اليَدِ وَالرِّجْلِ.

صَعْبُ البَدِيهَةِ مَشْبُوبٌ أَظَافِرُهُ مُوَاثِبٌ أَهْرَتُ الشِّدْقَيْنِ نِبْرَاسُ
مالك بن خالد الخُناعيّ الهُذَليّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏9/‏267‏.
132هـ=749م

ظُفْرُ القَوْسِ: الطَّرَفُ الَّذِي فَوْقَ الوَتَرِ.

فَمَا أَدْرَكَتْنَا يَا ابْنَ مَرْوَانَ دُونَكُمْ صَلَاةٌ لِأُولَى فِي مُنَاخٍ وَلَا فَجْرِ
وَلَا هِيَ إِلَّا وَقْعَةٌ كُلَّمَا الْتَظَى أُوَارُ الحَصَى فِي كُلِّ هَاجِرَةٍ وَغْرِ
وَتََحْلِيلِ شُعْثٍ غَوَّرُوا رَفَّعُوا لَهُمْ بِنَاءً بَنَوْهُ فَوْقَ ظُفْرٍ عَلَى ظُفْرِ
أبو حيّة النُّميريّ
شعر أبي حيَّة النُّميري: جمعه وحققه: يحيى الجبوري، منشورات وزارة الثقافة، دمشق، (1975م)، ‏56‏.
1406هـ=1985م

الظُّفْرَانِ : المُزْدَوِجَتَانِ، وَهُمَا عَلَامَتَا تَرْقِيمٍ تُسْتَعْمَلَانِ لِلِاقْتِبَاسِ أَوْ تَمْيِيزِ عِبَارَةٍ أَوْ قَوْلٍ أَوْ نَحْوِ ذَلِكَ.

"وَهَذَا مَا حَدَا بِنَا إِلَى أَنْ نَسْتَعْمِلَ عِبَارَةَ: "تَجَاوُرِ الحُقُولِ الدَّلَالِيَّةِ"، بَيْنَ ظُفْرَيْنِ ، لِأَنَّ هَذَا الِاسْتِعْمَالَ يَفْتَرِضُ...، أَنَّ الحُقُولَ الدَّلَالِيَّةِ أَوْ بَعْضَهَا، عَلَى الأَقَلِّ، تَخْضَعُ لِنَوْعٍ مِنَ التَّنْظِيمِ وَالهَيْكَلَةِ"
أحمد محمد أمبيريك
صورة بخيل الجاحظ الفنية من خلال خصائص الأسلوب في كتاب البخلاء: أحمد محمد أمبيريك، الدار التونسية للنّشر، تونس، (1985م)، ‏38‏.

اظِّفَار[مَصْدَر]

قبل 50ق.هـ=573م

الِاظِّفَارُ : الفَوْزُ وَالغَلَبَةُ.

نَحْنُ أَصْحَابُ شَبًا يَوْمَ شَبَا بِصِفَاحِ البِيضِ فِيهِنَّ اظِّفَارُ
الأفوه الأوديّ
ديوان الأفوه الأودي: شرح وتحقيق: محمّد ألتونجي، دار صادر، بيروت، ط1، (1998م)، ‏78‏.

تَظَافَرَ[لَازِم]

قبل 13ق.هـ=609م

تَظَافَرَ القَوْمُ: تَنَاصَرُوا وَتَعَاوَنُوا.

"فَتَفَاقَمَ الأَمْرُ بَيْنَ الحَيَّيْنِ،... فَتَظَافَرُوا عَلَيْنَا حَسَدًا"
الحارث بن ذُبيان
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏1/‏102‏.

ظَفِر[صِيغَةُ مُبَالَغَة]

10ق.هـ=612م

الظَّفِرُ : الكَثِيرُ الظَّفَرِ.

ظَفِرٌ بِالأُمُورِ، جَلْدٌ نَجِيبٌ وَإِذَا مَا سَمَا لِحَرْبٍ أَبَاحَا
الخنساء
ديوان الخنساء: شرحه: أبو العبّاس ثعلب (ت، 291هـ)، حققه: أنور أبو سُوَيْلم، دار عمار، عمان، ط1، (1409هـ/ 1988م)، ‏242‏.

النّابغة الجَعْديّ
شعر النابغة الجعدي: تحقيق: عبد العزيز رباح، منشورات المكتب الإسلامي، دمشق، ط1، (1384هـ/ 1964م)، ‏50‏-‏ 51‏.

ن11هـ=632م

الظَّفَرَةُ : جُلَيْدَةٌ تَغْشَى بَيَاضَ العَيْنِ مِمَّا يَلِي الأَنْفَ إِلَى سَوَادِهَا.

"إِنَّ الدَّجَّالَ خَارِجٌ، وَهُوَ أَعْوَرُ بِعَيْنِ الشِّمَالِ، عَلَيْهَا ظَفَرَةٌ غَلِيظَةٌ"

تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏18/‏519‏.

مُسْتَظْفِر[اسْمُ فَاعِل]

ن270هـ=883م

المُسْتَظْفِرُ : النَّاصِرُ المُعِينُ.

"أَسْبَغَ اللهُ عَلَيْكَ نِعَمَهُ الرَّاهِنَةَ بِنِعْمَةِ المُسْتَظْفِرِ ، وَصَانَهَا لَدَيْكَ بِإِيزَاعِ الشُّكْرِ عَلَيْهَا"
ابن عبد كان
كتاب صبح الأعشى: أبو العباس أحمد بن علي القلقشنديّ (ت، 821هـ)، مطبعة دار الكتب المصرية- المطبعة الأميرية، القاهرة، (1331ه/ 1913م- 1340ه/ 1992م)، ‏8/‏161‏.

مُظَافَرَة[مَصْدَر]

320هـ=932م

المُظَافَرَةُ : المُنَاصَرَةَ وَالمُعَاوَنَةَ.

"ثُمَّ نَزَلَ عَلَيْهِ بَنُو شِهَابٍ دِينًا، وَبَنُو بَنِيهِ عَائِذِينَ مِنْ ضَرِيَّةَ إِلَى اليَمَنِ لِيَسْكُنُوا مَعَ آلِ عَبْدِ اللهِ ذِي الأَفْرَاسِ بْنِ سَكَكٍ، فَسَأَلُوهُ الحِلْفَ وَالمُظَافَرَةَ لَمَّا أَعْجَبَهُمْ حَقْلُ صَعْدَةَ"
الحسن بن أحمد الهَمْدانيّ
كتاب الإكليل من أخبار اليمن وأنساب حِمْير: الحسن بن أحمد الهَمْدانيّ (ت، 334هـ)،ج1، حقّقه وعلّق عليه: محمّد بن علي الأكوع، إصدارات وزارة الثّقافة والسّياحة، صنعاء، (1425هـ/ 2004م)، ‏1/‏309‏.

مُتَظَافِر[اسْمُ فَاعِل]

334هـ=945م

المُتَظَافِرُ مَعَ غَيْرِهِ: المُتَنَاصِرُ المُتَعَاوِنُ مَعَهُ.

"وَأَنَّ المُلْكَ صَارَ بَعْدَ مِنُوشَهْرَ إِلَى أَخَوَيْنِ، وَقِيلَ: بَلْ كَانَا شَرِيكَيْنِ فِي المُلْكِ مُتَظَافِرَيْنِ مُتَعَاوِنَيْنِ عَلَى عِمَارَةِ الأَرْضِ"
عليّ بن الحسين المسعوديّ
مروج الذّهب ومعادن الجوهر: عليّ بن الحسين المسعوديّ (ت، 346هـ)، ج(1- 2)، تحقيق: محمّد محيي الدين عبد الحميد، دار الفكر، بيروت، ط5، (1393هـ/ 1973م)، ‏1/‏232‏.

مُظَفِّر[اسْمُ فَاعِل]

345هـ=956م

"وَاللهُ فِي ذَلِكَ كُلِّهِ نَاصِرُنَا وَخَاذِلُه، وَمُظَفِّرُنَا وَقَاتِلُه"
أبو إسحاق الصّابي

ظُفَيْر[اسْمٌ مُصَغَّر]

ن374هـ=984م

الظُّفَيْرُ : تَصْغِيرُ الظُّفْرِ.

وَغُيُومُ السَّحَابِ تَنْشْرُ سَكْبًا ثُمَّ تَنْجَابُ تَارَةً عَنْ قُمَيْرِ
أَنْحْفَتْهُ أَوَاخِرُ الشَّهْرِ حَتَّى عَادَ فِي أُفْقِهِ كَقَصِّ ظُفَيْرِ
تميم بن المعزّ لدين الله الفاطميّ
ديوان تميم بن المعزّ لدين الله الفاطميّ: تحقيق: محمّد حسن الأعظمي وآخرين، مطبعة دار الكتب المصرية، القاهرة، (1377هـ/ 1957م)، ‏196‏.

مَظْفُور[اسْمُ مَفْعُول]

375هـ=985م

المَظْفُورُ بِهِ: المُدْرَكُ، المُتَمَكَّنُ مِنْهُ.

"وَالخَوْضُ فِي هَذَا الطَّرَفِ قَدِيمٌ، وَفَضْلُهُ فِي الحَقِّ شَاقٌّ، وَالتَّنَازُعُ فِيهِ قَائِمٌ، وَالظَّنُّ يَعْمَلُ عَمَلُهُ، وَاليَقِينُ غَيْرُ مَظْفُورٍ بِهِ، وَلَا مَوْصُولٍ إِلَيْهِ"
أبو حيّان التَّوحيديّ
