أَيْقَنَ[مُتَعَدٍّ]

قبل 150ق.هـ=476م

أَيْقَنَ الأَمْرَ: عَلِمَهُ حَقِيقَةً لَا شَكَّ فِيهَا.

لَقَدْ أَيْقَنَتْ نَفْسُ الفَتَى، غَيْرَ بَاطِلٍ، وَإِنْ عَاشَ حِينًا، أَنَّهُ سَوْفَ يَهْلِكُ
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏63‏.

مُسْتَيْقِن[اسْمُ فَاعِل]

قبل 100ق.هـ=525م

المُسْتَيْقِنُ بِالأَمْرِ: العَالِمُ بِهِ حَقِيقَةً لَا شَكَّ فِيهَا.

فَخَرَجْنَ، حِينَ ثَوَى كُلَيْبٌ، حُسَّرًا مُسْتَيْقِنَاتٍ ، بَعْدَهُ، بِهَوَانِ
مُهَلْهِل بن ربيعة التَّغْلِبيّ
ديوان المُهَلْهِل: شرح وتحقيق: أنطوان محسن القوَّال، دار الجيل، بيروت، ط1، (1415هـ/ 1995م)، ‏86‏.

اسْتَيْقَنَ[مُتَعَدٍّ]

قبل 100ق.هـ=525م

اسْتَيْقَنَ الأَمْرَ: أَيْقَنَهُ.

فَلَمَّا اسْتَيْقَنُوا بِالصَّبْرِ، مِنَّا تُذُكِّرَتِ العَشَائِرُ، وَالحَزِيقُ
المفضّل بن معشر النُّكْريّ
شعراء عبد القيس وشعرهم في العصر الجاهلي: جمع وتحقيق ودراسة: عبد الحميد المعيني، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري، الكويت، (2002م)، ‏328‏.

يَقِين[اسْم]

قبل 50ق.هـ=573م

اليَقِينُ : مَا يَسْتَقِرُّ فِي النَّفْسِ حَقِيقَةً، لَا شَكَّ فِيهِ.

هَذَا، عَلَى إِثْرِ الَّذِي هُوَ قَبْلَهُ وَهِيَ المَنَايَا، وَالسَّبِيلُ المَهْيَعُ!
هَذَا اليَقِينُ ، فَكَيْفَ أَنْسَى فَقْدَهُ إِنْ رَابَ دَهْرٌ، أَوْ نَبَا بِيَ مَضْجَعُ؟
سعدى بنت الشَّمَرْدَل الجُهَينيّة
الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏104‏.
12ق.هـ=610م

اليَقِينُ : المَوْتُ.

قرآن كريم
قرآن كريم، ‏[ المُدَّثِّر : 47 ].
ن11هـ=632م

اليَقِينُ فِي الصَّلَاةِ وَنَحْوِهَا: مَا تَحَقَّقَهُ الشَّخْصُ مِنْ الأَقَلِّ فِي أَعْمَالِهَا، عِنْدَ التَّرَدُّدِ بَيْنَ الزِّيَادَةِ وَالنُّقْصَانِ.

"إِذَا شَكَّ أَحَدُكُمْ فِى صَلاتِهِ فَلْيُلْقِ الشَّكَّ، وَلْيَبْنِ عَلَى اليَقِينِ ، وَلْيُصَلِّ سَجْدَتَيْنِ، فَإِنْ كَانَتْ خَمْسًا شَفَعَ بِهِمَا، وَإِنْ كَانَ صَلَّى أَرْبَعًا كَانَتَا تَرْغِيمًا لِلشَّيْطَانِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏18/‏346،،‏(رقم الحديث: 11830).

يَقِين[مَصْدَر]

ن200هـ=815م

اليَقِينُ فِي العِلْمِ: القَطْعُ فِي حَقَائِقِهِ لِبِنَائِهَا عَلَى أَدِلَّةٍ بُرْهَانِيَّةٍ مُقَدِّمَاتُهَا كُلِّيَّةٌ ضَرُورِيَّةٌ ذَاتِيَّةٌ. [الفلسفة]

"حَتَّى إِنَّ قَوْمًا قَدْ ظَنَّوا أَنَّهُ يُمْكِنُ أَنْ يَكُونَ فِي هَذَا البَابِ عِلْمٌ بُرْهَانِيٌّ يَقِينٌ ، وَذَلِكَ إِذَا لَمْ يُوجَدْ فِي كُلِّ مَا يَسْبِقُهُ أَمْرٌ وَاحِدٌ مُخَالِفٌ لِمَا يَشْهَدُ بِأَمْرٍ مَا مِنَ الأُمُورِ"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏286‏.

يَقَّنَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

يَقَّنَهُ الأَمْرَ: جَعَلَهُ يَتَيَقَّنُهُ مِنْ غَيْرِ شَكٍّ.

"فَكَانَ ذَلِكَ مِمَّا يَقَّنَ عِنْدِي الحَمْلَ"
آمنة بنت وهب الزُّهْريّة القُرَشيّة
المنتظم في تاريخ الملوك والأمم: ابن الجوزيّ (ت، 597هـ)، دراسة وتحقيق: محمد عبد القادر عطا، مصطفى عبد القادر عطا، راجعه وصححه: نعيم زرزور، دار الكتب العلمية، بيروت، ط2، (1415ه/ 1995م)، ‏2/‏242‏.

يَقَن[مَصْدَر]

قبل 13ق.هـ=609م

اليَقَنُ بِالأَمْرِ: العِلْمُ بِهِ حَقِيقَةً لَا شَكَّ فِيهَا.

وَمَا بِالَّذِي أَبْصَرَتْهُ العُيُو نُ، مِنْ قَطْعِ يَأْسٍ، وَلَا مِنْ يَقَنْ
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏1/‏142‏.

يَقَن[صِفَةٌ مُشَبَّهَة]

ن215هـ=830م

اليَقَنُ مِنَ النَّاسِ: الَّذِي لَا يَسْمَعُ بِشَيْءٍ إِلَّا أَيْقَنَ بِهِ.

"رَجُلٌ أَذَنٌ يَقَنٌ "
أبو زيد الأنصاريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏9/‏325‏.
ن728هـ=1327م

اليَقَنُ مِنَ المَاءِ: السَّاكِنُ المُسْتَقِرُّ.

"وَاليَقِينُ يَتَضَمَّنُ مَعْنَى الطُّمَأْنِينَةِ وَالسُّكُونِ، وَمِنْهُ: مَاءٌ يَقَنٌ "
ابن تيميّة
مجموع الفتاوى: ابن تيميّة (ت، 728هـ)، جمع وترتيب: عبد الرحمن بن قاسم، محمد بن عبد الرحمن بن قاسم، مجمع الملك فهد لطباعة المصحف الشريف، المدينة المنورة، ط1، (1425ه/ 2004م)، ‏5/‏570‏-‏ 571‏.

مُوقِن[اسْمُ فَاعِل]

3ق.هـ=619م

المُوقِنُ بِالأَمْرِ: العَالِمُ بِهِ حَقِيقَةً لَا شَكَّ فِيهَا.

قرآن كريم، ‏[ الأَنعَام : 75 ].
ن200هـ=815م

المُوقِنُ عِنْدَ الشِّيعَةِ: البَالِغُ رُتْبَةَ اليَقِينِ بَيْنَ الأَشْخَاصِ التَّابِعِينَ لِمَقَامَيِ النُّبُوءَةِ وَالإِمَامَةِ.

"النَّبِيُّ، الإِمَامُ، الحِجَابُ، البَسِيطُ... المُوقِنُ ، الكَمِينُ، الصِّرَاطُ"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏440‏.

تَيَقَّنَ[مُتَعَدٍّ]

ن3ق.هـ=619م

تَيَقَّنَ الأَمْرَ: أَيْقَنَهُ حَقِقَةً.

فَجَاءُوا، وَقَدْ هَمُّوا بِقَتْلِ مُحَمَّدٍ؛ فَرَدَّهُمُ عَنْهُ، بِحُسْنِ خِصَامِ
بِتَأْوِيلِهِ التَّوْرَاةَ، حَتَّى تَيَقَّنُوا وَقَالَ لَهُمْ: رُمْتُمْ أَشَدَّ مَرَامِ
أبو طالب
ديوان أبي طالب بن عبد المطلب: صنعة: أبي هَفّان المِهْزَميّ (ت، 257هـ)، عليّ بن حمزة البصريّ (ت، 375هـ): تحقيق: محمّد حسن آل ياسين، دار ومكتبة الهلال، بيروت، ط1، (1421هـ/ 2000م)، ‏133‏.

تَيَقُّن[مَصْدَر]

8هـ=629م

تَيَقُّنُ الأَمْرِ: العِلْمُ بِهِ حَقِيقَةً لَا شَكَّ فِيهَا.

وَظَنُّهُمُ بِي أَنَّنِي، لِعَشِيرَتِي عَلَى أَيِّ حَالٍ، كَانَ، حَامٍ، وَذَائِدُ
فَإِنْ لَمْ أُحَقِّقْ ظَنَّهُمْ، بِتَيَقُّنٍ فَلَا سَقَتِ الأَوْصَالَ، مِنِّي، الرَّوَاعِدُ
حسّان بن ثابت الخزرجيّ الأنصاريّ
ديوان حسان بن ثابت: حقّقه وعلّق عليه: وليد عرفات، دار صادر، بيروت، (2006م)، ‏1/‏49‏.
ن430هـ=1038م

التَّيَقُّنُ : إِدْرَاكُ المُبْصَرَاتِ القَرِيبَةِ الوَاضِحَةِ إِدْرَاكًا حِسِّيًّا غَيْرَ حَدْسِيٍّ.

"وَقَدْ تَقَدَّمَ أَنَّ مَقَادِيرَ أَبْعَادِ المُبْصَرَاتِ مِنْهَا مَا يُدْرَكُ بِالتَّيَقُّنِ ، وَمِنْهَا مَا يُدْرَكُ بِالحَدْسِ"
ابن الهيثم
كتاب المناظر للحسن بن الهيثم المقالات 1- 2- 3 في الإبصار على الاستقامة: ابن الهيثم (ت، 430هـ)، حققها وراجعها: عبد الحميد صبرة، المجلس الوطني للثقافة والفنون والآداب، الكويت، (1404هـ/ 1983م)، ‏280‏.

اسْتِيقَان[مَصْدَر]

ن93هـ=712م

لَا تُجْمِعِي صَرْمِي، وَهَجْرِيَ، بَاطِلًا وَتَفَهَّمِي، وَاسْتَيْقِنِي اسْتِيقَانَا

شرح ديوان عمر بن أبي ربيعة المخزومي: تحقيق: محمّد محيي الدين عبد الحميد، المكتبة التجارية الكبرى- مطبعة السعادة، القاهرة، ط1، (1371هـ/ 1952م)، ‏261‏.

يَقِن[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

"وَإِذَا ظَهَرَ دَلِيلٌ يُؤَيِّدُ الِافْتِرَاضَ (ق)، وَيُفِيدُ الِافْتِرَاضَ (ب)؛ عِنْدَئِذٍ قَدْ يَحُلُّ (ق) مَحَلَّ (ب). إِنَّ تَعْدِيلَ الِاعْتِقَادِ الِاسْتِيقَانِيِّ سَبَبُهُ الدَّلِيلُ الجَدِيدُ الَّذِي أَثَارَ بَدَاءَةَ الشُّكُوكِ حَوْلَ حَقِيقَةِ الِافْتِرَاضِ (ب)، ثُمَّ أَقْنَعَنَا بِإِحْلَالِ (ق) مَحَلَّ (ب)"
فولكر نينهاوس
آراء جديدة في علم الاقتصاد المعياري: النهوج الغربية والمنظور الإسلامي: ‏فولكر نينهاوس،‏مجلة المسلم المعاصر: https://almuslimalmuaser.org/، ‏، نُشر في: 02-08-1993 م. شوهد في: 17-07-2025 م.

مُتَيَاقِن[اسْمُ فَاعِل]

1422هـ=2000م

المُتَيَاقِنُ مِنَ الأُمُورِ: المُتَّجِهُ إِلَى أَنْ يَكُونَ يَقِينًا مُطَابِقًا لِلْوَاقِعِ.

"يَتَعَذَّرُ جِدًّا -فِي رَأْيِنَا عَلَى الأَقَلِّ- تَنَاوُلُ نَصٍّ طَوِيلٍ ... بِمَنْهَجٍ جَانِحٍ إِلَى السِّيمْيَائِيَّةِ؛ لِمَا يَتَطَلَّبُ تَتَبُّعُ كُلِّ مَنْطُوقَاتِهِ مِنْ تَحْلِيلٍ فَرْدَانِيٍّ، وَمُزْدَوَجٍ، وَمُرَكَّبٍ، وَجَمْعَانِيٍّ، وَنَحْوِيٍّ، وَدِلَالِيٍّ، مُورْفُولُوجِيٍّ، مِنْ تَحْلِيلٍ مُتَشَاكِلٍ وَغَيْرِ مُتَشَاكِلٍ، وَمُتَيَاقِنٍ وَمُتَحَايِلٍ"
حبيب مونسي
القراءة والحداثة، مقاربة الكائن والممكن في القراءة العربية: حبيب مونسي، اتحاد الكتّاب العرب، دمشق، (2000م)، ‏247‏.

مُتَيَاقَن[اسْمُ مَفْعُول]

1428هـ=2006م

المُتَيَاقَنُ مِنَ الأُمُورِ: المَنْحُوُّ بِهِ إِلَى أَنْ يَكُونَ يَقِينًا مُطَابِقًا لِلْوَاقِعِ.

"هُنَاكَ لُغَتَانِ بَيْنَهُمَا اخْتِلَافٌ وَاضِحٌ: لُغَةُ الحُلْمِ، وَلُغَةُ التَّفْسِيرِ، وَلَكِنَّ اللُّغَتَيْنِ مَعًا تَنْدَرِجَانِ فِي إِطَارٍ يَشْمَلُهُمَا، هُوَ التَّخْيِيلُ. أ- بِالنِّسْبَةِ لِلُغَةِ الحُلْمِ؛ جُمَلٌ وَاصِفَةٌ، تَوْلِيدِيَّةٌ، مُتَدَرِّجَةُ البِنَاء، تَبْنِي العَالَمَ فِي تَهَيُّبٍ، تُحَاكِي أُسْلُوبِيَّةَ الحُلْمِ القَائِمَةَ عَلَى التَّوَهُّمِ المُتَيَاقَنِ ، أَوِ اليَقِينِ المُتَوَاهَمِ"
ندوة كلية الآداب-عين الشق
ندوة تكريم أحمد بوزفور: مجموعة من الباحثين، جامعة الحسن الثاني، الدار البيضاء، (2006م)، ‏207‏.

مَيْقُونِيَّة[مَصْدَرٌ صِنَاعِيّ]

1430هـ=2008م

المَيْقُونِيَّةُ : اليَقِينِيَّةُ.

"أَمَّا المِيزَةُ الأُخْرَى الَّتِي يَتَّسِمُ بِهَا هَذَا المَذْهَبُ -الخَلَاصُ الزَّائِفُ- فَهِيَ المَيْقُونِيَّةُ المُطْلَقَةُ، فِيمَا يَخُصُّ هُوِيَّةَ المَسِيحِ المُخَلِّصِ"
داوود عبد الله
قانون العودة الإسرائيلي وتأثيره على الصراع في فلسطين: مجموعة مؤلّفين، تحرير: داوود عبد الله، مركز باحث للدراسات، بيروت، (2008م)، ‏24‏.

تَيَاقُنِيَّة[مَصْدَرٌ صِنَاعِيّ]

1432هـ=2010م

التَّيَاقُنِيَّةُ : اتِّجَاهُ العِلْمِ وَنَحْوِهِ إِلَى أَنْ يَكُونَ يَقِينًا مُطَابِقًا لِلْوَاقِعِ.

"تَعْتَبِرُ التَّيَاقُنِيَّةُ أَنَّ العِلْمَ يَتَيَاقَنُ، أَيْ: يَتَّجِهُ إِلَى أَنْ يَغْدُوَ يَقِينًا، وَمُطَابِقًا لِلْوَاقِعِ"
حسن عجمي
السوبر جدلية، التحاقلية، التياقنية: حسن عجمي، شمس للنشر والتوزيع، القاهرة، ط1، (2010م)، ‏55‏.

تَيَاقُن[مَصْدَر]

1432هـ=2010م

التَّيَاقُنُ : اتِّجَاهُ العِلْمِ وَنَحْوِهِ إِلَى أَنْ يَكُونَ يَقِينًا مُطَابِقًا لِلْوَاقِعِ.

"بِمَا أَنَّ مَعْرِفَتَنَا عَمَلِيَّةُ تَيَاقُنٍ ، بَدَلًا مِنْ يَقِينٍ، إِذَنْ؛ مَعْرِفَتُنَا مُعَرَّضَةٌ لِلتَّكْذِيبِ وَالتَّصْحِيحِ بِشَكْلٍ دَائِمٍ"
حسن عجمي
السوبر جدلية، التحاقلية، التياقنية: حسن عجمي، شمس للنشر والتوزيع، القاهرة، ط1، (2010م)، ‏57‏.

تَيَاقَنَ[لَازِم]

1432هـ=2010م

تَيَاقَنَ العِلْمُ وَنَحْوُهُ: اتَّجَهَ إِلَى أَنْ يَكُونَ يَقِينًا مُطَابِقًا لِلْوَاقِعِ.

"بِالنِّسْبَةِ إِلَى التَّيَاقُنِيَّةِ؛ الأَشْيَاءُ وَالمَعَارِفُ تَتَيَاقَنُ ، أَيِ المَعْرِفَةُ فِي صَيْرُورَةٍ أَنْ تَغْدُوَ مُطَابِقَةً لِلْوَاقِعِ"
حسن عجمي
السوبر جدلية، التحاقلية، التياقنية: حسن عجمي، شمس للنشر والتوزيع، القاهرة، ط1، (2010م)، ‏59‏.
