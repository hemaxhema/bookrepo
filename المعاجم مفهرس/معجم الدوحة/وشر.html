وُشُر[مَصْدَر]

قبل 150ق.هـ=476م

وُشُرُ أَسْنَانِ الشَّخْصِ: تَحْدِيدُهَا، وَتَرْقِيقُ أَطْرَافِهَا.

وَوَاضِحٍ أَشْنَبِ الأَنْيَابِ ذِي أُشُرٍ كَالأُقْحُوَانِ، إِذَا مَا نَوْرُهُ لَمَعَا
لَقيط بن يَعْمَر الإياديّ
ديوان لقيط بن يَعمر: حققه وقدّم له: عبد المعيد خان، دار الأمانة- مؤسّسة الرسالة، بيروت، (1391هـ/ 1971م)، ‏52‏.

وَاشِر[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

الوَاشِرُ مِنَ الأَيْدِي وَنَحْوِهَا: المَقْطُوعُ.

لَقَدْ عَيَّلَ الأَيْتَامَ طَعْنَةُ نَاشِرَهْ أَنَاشِرَ، لَا زَالَتْ يَمِينُكَ آشِرَهْ
أمّ ناشرة التَّغْلِبيّة
كتاب الأضداد: قُطْرُب (ت، 206هـ)، عني بتحقيقه والتقديم له: حنّا جميل الحداد، دار العلوم، الرياض، ط1، (1405هـ/ 1984م)، ‏87‏.
315هـ=927م

الوَاشِرُ مِنَ الأَسْنَانِ: المُحَدَّدَةُ، المُرَقَّقَةُ أَطْرَافُهَا.

فِي فَرْعِ سَامِقَةٍ عَلَى خَلْقَاءَ بَاسِقَةِ العَرَاعِرْ
رَوْقَاءَ لَمْ تَكْسَسْ، وَكَمْ شَرِيَتْ نَوَاجِذُهَا الأَوَاشِرْ
زيد بن أحمد البردعيّ
كتاب الأشباه والنّظائر من أشعار المتقدّمين والجاهليّة والمخضرمين: أبو بكر الخالديّ (ت، 380هـ)، أبو عثمان الخالديّ (371هـ)، حقّقه وعلّق عليه: السيد محمّد يوسف، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، (1965م)، ‏2/‏180‏.

وَاشِر[اسْمُ فَاعِل]

ن11هـ=632م

الوَاشِرُ مِنَ الأَشْخَاصِ: المُحَدِّدُ أَسْنَانَهُ، وَالمُرَقِّقُ أَطْرَافَهَا.

"لَعَنَ النَّامِصَةَ وَالمُتَنَمِّصَة، وَالوَاشِرَةَ وَالمُسْتَوْشِرَة"
حديث نبويّ
كتاب غريب الحديث: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: عبد السلام هارون وآخرين، الهيئة العامة لشؤون المطابع الأميرية، القاهرة، (1404هـ/ 1984م- 1419هـ/ 1999م)، ‏3/‏214‏.

مُوَشَّر[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المُوَشَّرُ مِنَ أَسْنَانِ الأَشْخَاصِ: المُحَدَّدَةُ، المُرَقَّقَةُ أَطْرَافُهَا.

وَمُؤَشَّرٍ ، حُمْشِ اللِّثَاتِ، كَأَنَّمَا شَرِكَتْ مَنَابِتُهُ رَضِيضَ الإِثْمِدِ
زهير بن أبي سُلْمى المُزَنيّ
شعر زُهير بن أبي سُلمى: صنعة: الأعلم الشَّنْتَمَريّ (ت، 476هـ)، تحقيق: فخر الدين قباوة، منشورات الآفاق الجديدة، بيروت، ط3، (1400هـ/ 1980م)، ‏230‏.

مُوَشَّر[اسْم]

قبل 13ق.هـ=609م

المُوَشَّرُ العَضُدَيْنِ : الجُعْلُ.

كَأَنَّ مُؤَشَّرَ العَضُدَيْنِ حَجْلًا هَدُوجًا بَيْنَ أَقْلِبَةٍ مِلَاحِ
تَضَمَّنَ نِعْمَتِي، فَعَدَا عَلَيْهَا بُكُورًا، أَوْ تَعَجَّلَ فِي الرَّوَاحِ
عنترة
ديوان عنترة: تحقيق ودراسة: محمّد سعيد مولوي، المكتب الإسلامي، بيروت، (1390هـ/ 1970م)، ‏290 ‏-‏291‏.

مِيشَار[اسْمُ آلَة]

قبل 13ق.هـ=609م

المِيشَارُ : آلَةُ شَّقِّ الخَشَبِ وَقَطْعِهِ.

أَنْحَوْا عَلَى حُرِّ الوُجُو هِ بِحَدِّ مِئْشَارٍ ضَرُوسَا
ذو الإصبع العَدْوانيّ
ديوان ذي الإصبع العَدْواني حُرثان بن مُحَرِّث: جمعه وحققه: عبد الوهاب محمّد علي العَدْواني، محمّد نائف الدُّليمي، خطّ أشعاره: يوسف ذنون، مطبعة الجمهور، الموصل، (1393هـ/ 1973م)، ‏44‏.

وَشَرَ[مُتَعَدٍّ]

ن11هـ=632م

وَشَرَ الشَّيْءَ: شَقَّهُ وَقَطَعَهُ بِالمِيشَارِ.

"يَقُولُ: أَوَمَا تُؤْمِنُ بِي؟ قَالَ: فَيَقُولُ: أَنْتَ المَسِيحُ الكَذَّابُ، قَالَ: فَيُؤْمَرُ بِه، فَيُؤْشَرُ بِالمِئْشَارِ مِنْ مَفْرِقِهِ، حَتَّى يُفَرَّقَ بَيْنَ رِجْلَيْهِ"
حديث نبويّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏2257،،‏(رقم الحديث: 2938).
ن175هـ=791م

وَشَرَ الشَّخْصُ أَسْنَانَهُ: حَدَّدَهَا، وَرَقَّقَ أَطْرَافَهَا.

"الوَاشِرَةُ، وَهِيَ الآشِرَةُ، تَأْشِرُ أَسْنَانَهَا، أَيْ: تُحَزِّزُهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏282‏.
ن200هـ=815م

وَشَرَ الخَشَبَ وَنَحْوَهُ: شَقَّهُ وَقَطَعَهُ بِالمِيشَارِ.

"وَيُقَالُ: وَشَرْتُ الخَشَبَةَ"
من رواية أبي مِسْحَل الأعرابيّ
كتاب النّوادر: أبو مِسْحَل الأعرابيّ (ت، 230هـ)، عني بتحقيقه: عزة حسن، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏429‏.

مُسْتَوْشِر[اسْمُ فَاعِل]

ن11هـ=632م

المُسْتَوْشِرُ مِنَ الأَشْخَاصِ: الَّذِي يَطْلُبُ تَحْدِيدَ أَسْنَانِهِ، وَتَرْقِيقَ أَطْرَافِهَا.

"لَعَنَ اللَّهُ الوَاشِمَةَ وَالمُسْتَوْشِمَةَ، وَالمُتَنَمِّصَةَ وَالنَّامِصَةَ، وَالوَاشِرَةَ وَالمُسْتَوْشِرَةَ "
حديث نبويّ
مسند أمير المؤمنين عمر بن عبد العزيز رضي الله عنه: أبو بكر محمد بن محمد بن سليمان الباغندي (ت، 312هـ)، خرج أحاديثه وعلق عليه: محمد عوامة، دار ابن كثير، دمشق-بيروت، ط3، (1407هـ/1987م)، ‏1/‏81،،‏(رقم الحديث: 29).

وَشْر[مَصْدَر]

ن30هـ=651م

الوَشْرُ : تَحْدِيدُ الأَسْنَانِ وَتَرْقِيقُ أَطْرَافِهَا.

"بَلَغَنَا أَنَّ رَسُولَ اللهِ  @  نَهَى عَنِ الوَشْرِ ، وَالوَشْمِ، وَالنَّتْفِ، وَالمُشَاغَرَةِ، وَالمُكَامَعَةِ، وَالوِصَالِ، وَالمُلَامَسَةِ"
أبو ريحانة الأزديّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏28/‏440،،‏(رقم الحديث: 17208).
ن414هـ=1023م

وَشْرُ الشَّيْءِ: شَقُّهُ، وَقَطْعُهُ بِالمِيشَارِ.

"فَإِنَّهُ مَنْ شَهِدَ ذَلِكَ مِنْكَ سَامَكَ خَسْفا، وَنَسَفَكَ نَسْفا، وَقَشَرَكَ قَشْرا، وَوَشَرَكَ وَشْرا "
أبو حيّان التَّوحيديّ
الإشارات الإلهيّة: أبوحيّان التّوحيديّ (ت، 414هـ)، حققه وقدم له: عبد الرحمن بدويّ، مطبعة جامعة فؤاد الأول، القاهرة، (1950م)، ‏126‏.

مَوْشُور[اسْمُ مَفْعُول]

ن37هـ=657م

المَوْشُورُ مِنَ الأَشْخَاصِ: المُحَدَّدَةُ أَسْنَانُهُ، المُرَقَّقَةُ أَطْرَافُهَا.

سَبَتْكَ بِمَأْشُورِ الثَّنَايَا، كَأَنَّهُ أَقَاحِي غَدَاةٍ بَاتَ بِالدَّجْنِ يُنْضَحُ
ابن مقبل العَجْلانيّ التَّميميّ
ديوان ابن مقبل: عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏53‏.
412هـ=1021م

المَوْشُورُ مِنَ الخَشَبِ وَنَحْوِهِ: المَشْقُوقُ، المَقْطُوعُ بِالمِيشَارِ.

"وَلَوْ أَنَّهُمَا مَأْشُورَانِ مِنْ خَشَبِ القُطْرِ لَمَا زَادَ الأَمْرُ عَلَى مَا ادَّعَيْتَ"
أبو العلاء المَعَرّيّ
رسالة الصّاهل والشّاحج: أبو العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: عائشة عبد الرحمن بنت الشاطئ، دار المعارف، القاهرة، ط2، (1404هـ/ 1984م)، ‏331‏.

تَوْشِير[مَصْدَر]

ن130هـ=748م

التَّوْشِيرُ : صَوْتُ المِيشَارِ يَشُقُّ الخَشَبَ وَيَقْطَعُهُ.

وَبَاتَتِ الأَفْعَى عَلَى مَحْفُورِهَا بِالَّلجْفِ تَسْتَحْيِيهِ مِنْ تَصْغِيرِهَا تَأْشِيرُهَا يَحْتَكُّ فِي تَأْشِيرِهَا
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏213‏-‏214‏.
ن449هـ=1057م

تَوْشِيرُ أَسْنَانِ الشَّخْصِ: تَحْدِيدُهَا، وَتَرْقِيقُ أَطْرَافِهَا.

يَا سِيدُ، هَلْ لَكَ فِي ظَبْيٍ تُغَازِلُهُ تُلْقِي نُيُوبُكَ فِي تَأْشِيرِهِ قُبَلَكْ؟
أبو العلاء المَعَرّيّ
اللّزوميّات: أبو العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: أمين عبد العزيز الخانجي، مكتبة الهلال، بيروت، مكتبة الخانجي، القاهرة، (1342هـ)، ‏2/‏167‏.

تَوْشِير[اسْم]

ن153هـ=770م

تَوْشِيرُ الجَرَادَةِ وَنَحْوِهَا: مَا تَعَضُّ بِهِ.

"لِلْجَرَادَةِ قَرْنَانِ، وَهُمَا مِثْلُ الشَّعْرَتَيْنِ، وَلَهَا تَآشِيرُ ، وَهِيَ الَّتِي تَعَضُّ بِهَا"
نهْشل بن زيْد
كتاب النّبات: أبو حنيفة الدّينوريّ (ت، 282هـ)، حققه وشرحه وقدم له: برنهارد لفين، فرانز شتاينر بقيسبادن، (1394هـ/ 1974م)، ‏59‏.

وَشِر[صِفَةٌ مُشَبَّهَة]

ن167هـ=783م

الوَشِرُ مِنَ الأَسْنَانِ: المُحَدَّدَةُ، المُرَقَّقةُ أَطْرَافُهَا.

مِنْ لُؤْلُؤٍ أَشِرِ الأَطْرَافِ مَنْبِتُهُ فِي طَيِّبِ الطَّعْمِ عَذْبٍ، بَارِدٍ، خَصِرِ
بشّار بن برد
ديوان بشار بن برد: قرأه وقدم له: إحسان عبّاس، دار صادر، بيروت، ط1، (2000م)، ‏349‏.

وُشْرَة[مَصْدَر]

ن175هـ=791م

الوُشْرَةُ : حِدَّةُ الأَسْنَانِ، وَرِقَّةُ أَطْرَافِهَا.

"يُقَالُ: إِنَّ ثَغْرَهَا لَذُو أُشْرَةٍ وَعِتْرَةٍ"
من رواية الخليل بن أحمد الفراهيدي
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏2/‏66‏.

وَشَّرَ[مُتَعَدٍّ]

ن250هـ=864م

وَشَّرَ الشَّخْصُ أَسْنَانَهُ: بَالَغَ فِي تَحْدِيدِهَا، وَتَرْقِيقِ أَطْرَافِهَا.

"وَقَدْ تُؤَشِّرُ المَرْأَةُ الكَبِيرَةُ أَسْنَانَهَا تَتَشَبَّهُ بِالأَحْدَاثِ"
ثابت بن أبي ثابت اللُّغويّ
كتاب خلق الإنسان: ثابت بن أبي ثابت اللُّغويّ (ت، 250هـ)، تحقيق: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، ط2، د.ت. ‏168‏.

وُشَارَة[اسْم]

ن298هـ=910م

وُشَارَةُ الخَشَبِ وَنَحْوِهَا: مَا سَقَطَ مِنْهَا عِنْدَ شَقِّهَا وَقَطْعِهَا بِالمِيشَارِ.

"وَأُحِبُّ أَنْ تَجْعَلَ بَدَلَ الأُشَارَةِ التُّرَابَ، فَإِنَّ النَّبِيَّ  @  قَالَ: (أَتْرِبُوا كُتُبَكُمْ؛ فَإِنَّهُ أَنْجَحُ لِلْحَاجَةِ)"
إبراهيم بن محمّد الشَّيْبانيّ
الرّسالة العذراء في موازين البلاغة وأدوات الكتابة: إبراهيم بن محمّد الشّيبانيّ (ت، 298هـ)، تحقيق ودراسة: يوسف محمّد فتحي عبد الوهاب، دار الطلائع، القاهرة، (2005م)، ‏53‏.

وُشَارَة[مَصْدَر]

ن638هـ=1240م

وُشَارَةُ الخَشَبِ وَنَحْوِهِ: شَقُّهُ، وَقَطْعُهُ بِالمِيشَارِ.

"إِعْطَاءُ الوَلَدِ لِلْمُعَلِّمِ، وَاتِّخَاذُ الشَّرَائِفِ، وَعَمَلُ الخَوَاتِمِ مِنَ الكُتُبِ... وَوُشَارَةُ المَرْكَبِ وَالخَشَبِ، وَالسَّفَرُ، وَتَكْوِيرُ المَرَاكِبِ وَغَيْرِهَا... وَكُلُّ مَا يَعْمَلُهُ فِي هَذِهِ السَّاعَةِ مِمَّا ذُكِرَ جَيِّدَةٌ لَهُ"
محيي الدين ابن عربيّ
مجموعة رسائل ابن عربي: محيي الدين ابن عربيّ (ت، 638هـ)، دار المحجة البيضاء، دار الرسول الأكرم، بيروت، ط1، (1421ه/ 2000م)، ‏3/‏388‏.

مُوتَشِر[اسْمُ فَاعِل]

352هـ=963م

المُوتَشِرُ مِنَ الأَشْخَاصِ: المُسْتَوْشِرُ.

"رُوِيَ عَنِ النَّبِي @  أَنَّهُ لَعَنَ الوَاشِرَةَ وَالمُؤْتَشِرَةَ "
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏11/‏409‏.

اسْتِيشَار[مَصْدَر]

ن751هـ=1350م

اسْتِيشَارُ الشَّخْصِ: طَلَبُهُ تَحْدِيدَ أَسْنَانِهِ، وَتَرْقِيقَ أَطْرَافِهَا.

"وَمِنَ الكَبَائِرِ: تَرْكُ الصَّلَاةِ، وَمَنْعُ الزَّكَاةِ... وَتَخَنُّثُ الرَّجُلِ، وَتَرَجُّلُ المَرْأَةِ... وَالوَشْمُ وَالِاسْتِيشَامُ، وَالوَشْرُ وَالِاسْتِيشَارُ ، وَالنَّمْصُ وَالتَّنْمِيصُ"
ابن قيّم الجوزيّة
إعلام الموقعين عن رب العالمين: ابن قيّم الجوزيّة (ت، 751هـ)، قرأه وقدم له وعلق عليه وخرج أحاديثه وآثاره: أبو عبيدة مشهور بن حسن آل سلمان، شارك في التخريج: أبو عمر أحمد عبد الله أحمد، دار ابن الجوزي، الدمام، ط1، (1423 ه)، ‏6/‏569‏-‏ 570‏.
