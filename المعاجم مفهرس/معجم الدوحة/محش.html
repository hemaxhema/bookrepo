مِحَاش[اسْمُ جَمْع]

قبل 13ق.هـ=609م

المِحَاشُ : القَوْمُ المُجْتَمِعُونَ مِنْ قَبَائِلَ شَتَّى، فَيَتَحَالَفُونَ عِنْدَ النَّارِ.

جَمِّعْ مِحَاشَكَ ، يَا يَزِيدُ، فَإِنَّنِي أَعْدَدْتُ يَرْبُوعًا لَكُمْ وَتَمِيمَا
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏224‏.

امْتَحَشَ[مُتَعَدٍّ]

ن11هـ=632م

امْتَحَشَتْهُ النَّارُ: أَحْرَقَتْهُ.

"فَيُخْرَجُونَ مِنَ النَّارِ وَقَدِ امْتُحِشُوا "
حديث نبويّ
مُسند الدّارمي المعروف بـ(سنن الدارمي): الدّارمي (ت، 255هـ)، تحقيق: حسين سليم أسد الداراني، دار المغني، الرياض، ط1، (1421هـ/ 2000م)، ‏1/‏199،،‏(رقم الحديث: 53).
ن125هـ=743م

امْتَحَشَ الخَمْرَةَ وَنَحْوَهَا: طَبَخَهَا عَلَى النَّارِ.

خُرُسُ الأَلْسُنِ مِمَّا صَابَهُمْ بَيْنَ مَصْدُوعٍ وَصَاحٍ مُنْتَعِشْ
مِنْ حُمَّيَّا قَرْقَفٍ خُصِّيَّةٍ قَهْوَةٍ حَوْلِيَّةٍ لَمْ تُمْتَحَشْ
النّابغة الشَّيْبانيّ
ديوان نابغة بني شَيْبان: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1351هـ/ 1932م)، ‏85‏.

امْتَحَشَ[لَازِم]

ن11هـ=632م

امْتَحَشَ الشَّيْءُ: احْتَرَقَ.

"فَيُخْرِجُونَهُمْ قَدِ امْتَحَشُوا "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏13/‏145،،‏(رقم الحديث: 7717).

مَحَشَ[مُتَعَدٍّ]

ن11هـ=632م

مَحَشَتْهُ النَّارُ: أَحْرَقَتْهُ.

"إِنَّ أَقْوَامًا يَخْرُجُونَ مِنَ النَّارِ بَعْدَمَا مُحِشُوا فِيهَا"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏23/‏290،،‏(رقم الحديث: 15048).
ن198هـ=813م

مَحَشَ الجِلْدَ: قَشَرَهُ وَخَدَشَهُ.

مَحَلٌّ لَيْسَ يُعْدِمُنِي بِهِ ذُو غُمَّةٍ جَحِدُ
مِنَ الأَعْرَابِ قَدْ مَحَشَتْ ضَوَاحِي جِلْدِهِ النُّجُدُ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏359‏.
ن216هـ=831م

مَحَشَ اللَّحْمَ عَلَى النَّارِ: اشْتَوَاهُ.

"وَخَالَفَ الأَصْمَعِيُّ النَّاسَ فِي هَذَا، وَقَالَ: إِنَّمَا سُمُّوا مِحَاشًا؛ لِأَنَّهُمْ مَحَشُوا بَعِيرًا عَلَى النَّارِ"
الأصمعيّ
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏1/‏539‏.
1317هـ=1899م

مَحَشَ الكِتَابَةَ عَنِ الرَّقِّ: كَشَطَهَا وَأَزَالَهَا.

"فَإِنَّ الكَرْدِينَالَ مَايَ اكْتَشَفَ كِتَابًا خَطِّيًّا مُسَطَّرًا عَلَى رَقِّ غَزَالٍ، كَانَ رُقِمَ عَلَيْهِ كِتَابَةٌ سَابِقَةٌ مَحَشَهَا النُّسَّاخُ لِخَطِّ تَأْلِيفٍ جَدِيدٍ"
لويس بولوموا
الطَّاعُونُ الدُّمَّلِيُّ: ‏لويس بولوموا،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين، بيروت، س2، ع12، (06/ 02/ 1317هـ)، ‏533‏.

مُحَاش[اسْم]

ن11هـ=632م

مُحَاشُ الشَّيْءِ: مَا احْتَرَقَ مِنْهُ.

"فَيُخْرِجُونَهُمْ قَدِ امْتَحَشُوا، فَيُلْقُونَهُمْ فِي نَهْرٍ... فَتَسْقُطُ مُحَاشُهُمْ عَلَى حَافَّةِ النَّهْرِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏22/‏374،،‏(رقم الحديث: 14491).

مُحَاش[صِفَة]

ن637هـ=1239م

المُحَاشُ : المُحَرَّقُ.

"فَلَمَّا أَضَاءَ النَّهَارُ رَكِبَتْ خَيْلُ الكُفْرِ وَخَيْلُ الإِيمَان، وَتَقَابَلَ حِزْبُ اللهِ وَحِزْبُ الشَّيْطَان، وَلَمْ تَزَلِ الحَرْبُ تُسَنُّ وَالكَرُّ نَاش، وَنَارُهَا تُشَبُّ وَالمَوْتُ غَاش، إِلَى أَنْ تَصَافَحُوا بِالصِّفَاح، بَعْدَ أَنْ تَحَيَّوْا بِالرِّمَاح، فَلَمْ نَرَ إِلَّا لُبًّا مُطَاشا، وَرَأْسًا مُحَاشا "
ابن الأثير الكاتب
رسائل ابن الأثير: ابن الأثير الكاتب (ت، 637هـ)، تحقيق: نوري حمودي القيسيّ، هلال ناجي، منشورات جامعة الموصل، الموصل، (1402ه/ 1982م)، ‏68‏.

مَحْش[مَصْدَر]

ن145هـ=762م

المَحْشُ : الإِحْرَاقُ.

كَمْ سَاقَ مِنْ دَارِ امْرِىءٍ جَحِيشِ إِلَيْكَ نَأْشُ القَدَرِ النَؤُوشِ وَطُولُ مَحْشِ السَّنَةِ المَحُوشِ!
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏200 ‏-‏ 201‏.
352هـ=963م

مَحْشُ الجِلْدِ: سَحْجُهُ وَخَدْشُهُ مِنْ غَيْرِ أَنْ يُسْلَخَ.

"قَالَ بَعْضُهُمْ: مَرَّ بِي حِمْلٌ، فَمَحَشَنِي مَحْشًا "
من رواية الأزهريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏197‏.

مَحُوش[صِفَةٌ مُشَبَّهَة]

ن145هـ=762م

المَحُوشُ مِنَ السِّنِينَ: المُحْرِقُ بِالجَدْبِ.

كَمْ سَاقَ مِنْ دَارِ امْرِىءٍ جَحِيشِ إِلَيْكَ نَأْشُ القَدَرِ النَّؤُوشِ وَطُولُ مَحْشِ السَّنَةِ المَحُوشِ !
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏200 ‏-‏ 201‏.

مَاحِش[اسْمُ فَاعِل]

ن220هـ=835م

المَاحِشُ مِنَ الحَرِّ وَنَحْوِهِ: المُحْرِقُ.

وَغَبْرَاءَ لَا تَجْرِي بِهَا الرِّيحُ عَاقِرٍ حَمَى ظِلَّهَا وَقْعٌ مِنَ الصَّيْفِ مَاحِشُ
بَرَقْتُ بِأَيْدِي العِيسِ مِنْهَا كَمَا مَضَى مِنَ الغَرَضِ الأَقْصَى المُرَامِي المُرَايِشُ
ناهض بن ثومة العامريّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏889‏.

مَاحِش[صِفَةٌ مُشَبَّهَة]

ن1188هـ=1773م

المَاحِشُ : الكَثِيرُ الأَكْلِ، حَتَّى يَعْظُمَ بَطْنُهُ.

مَنْ يُكْثِرِ الشُّرْبَ، وَيَأْكُلْ مَاحِشَا يَذْهَبْ بِهِ البَطْنُ ذَهَابًا فَاحِشَا
من شواهد المرتضى الزّبيديّ
تاج العروس من جواهر القاموس: المرتضى الزبيدي (ت، 1205هـ)، تحقيق: عبد الستار أحمد فراج وآخرين، المجلس الوطني للثقافة والفنون والآداب، الكويت، ط1، (1385ه/ 1965م- 1422ه/ 2001م)، ‏17/‏376‏.

مَحَّاش[صِيغَةُ مُبَالَغَة]

ن334هـ=945م

المَحَّاشُ مِنَ الحَرِّ وَنَحْوِهِ: الشَّدِيدُ الإِحْرَاقِ.

لَا يُبَالِي وَحْشَ الهَجِيرِ، إِذَا وَا جَهَ وَجْهَ الهَوَاجِرِ المَحَّاشَا
الصَّنَوبريّ
ديوان الصّنوبريّ أحمد محمّد بن الحسن الضّبيّ: تحقيق: إحسان عبّاس، دار صادر، بيروت، ط2، (1998م)، ‏194‏.

امْتِحَاش[مَصْدَر]

ن398هـ=1007م

الِامْتِحَاشُ : الِاحْتِرَاقُ.

"مَا رَأْيُكُمْ- يَا فِتْيَانُ- فِي عَنَاقٍ نَجْدِيَّة، عُلْوِيَّةٍ بَرِّيَّة... تُشْحَطُ مُعْتَبَطَةً، ثُمَّ تُنْكَسُ فِي وَطِيسٍ حَتَّى تَنْضَجَ مِنْ غَيْرِ امْتِحَاشٍ أَوْ إِنْهَاءٍ؟"
بديع الزّمان الهَمَذانيّ
مقامات بديع الزّمان الهَمَذانيّ: بديع الزّمان الهَمَذانيّ (ت، 398هـ)، قدم لها وشرح غوامضها: الشّيخ محمّد عبده، دار الكتب العلميّة، بيروت، ط3، (1426هـ/ 2005م)، ‏205‏-‏ 206‏.

إِمْحَاش[مَصْدَر]

543هـ=1148م

الإِمْحَاشُ : الإِحْرَاقُ.

"يَمُدُّونَ أَيْدِيَهُمْ إِلَى ضَرْبِ النَّاسِ بِالسِّيَاطِ إِبْلَاغًا فِي الِانْتِهَاءِ بِكَثْرَتِهَا وَإِمْحَاشا ، وَيَتَسَبَّبُونَ بِذَلِكَ إِلَى أَخْذِ أَمْوَالِ النَّاسِ إِيغَارًا لِلصُّدُورِ وَإِيحَاشا"
عبد المؤمن بن علي الكومي
نُظم الجمان لترتيب ما سلف من أخبار الزمان: ابن القطّان المراكُشيّ (ت، 650هـ)، درسه وقدم له وحققه: محمود علي مكي، دار الغرب الإسلامي، بيروت، ط2، (1410ه/ 1990م)، ‏193‏.

أَمْحَشَ[مُتَعَدٍّ]

ن633هـ=1235م

أَمْحَشَهُ الحَرُّ وَنَحْوُهُ: أَحْرَقَهُ.

"هَذَا الغِطْرِيسُ فِي اليَمِّ أَخَذَ رَجُلًا، لَا يَمْلِكُ حَذْرَفُوتًا، ... وَلَهُ قُرْحَةٌ أُمْحِشَتْ مِنَ الحَرِّ"
عمر بن الحسن بن علي بن دحية الكلبي
عنوان الدراية فيمن عرف من العلماء في المائة السابعة ببجاية: الغِبْريني (ت، 714هـ)، حققه وعلق عليه: عادل نويهض، دار الآفاق الجديدة، بيروت، ط2، (1979م)، ‏271‏.

مَحَّشَ[مُتَعَدٍّ]

1270هـ=1853م

مَحَّشَ المَرْأَةَ: نَكَحَهَا بِشِدَّةٍ.

"فَأَوَّلُ مَا تُحَرِّكُ شَفَتَيْهَا، تَسْكُنُ القُلُوبُ إِلَيْهَا، وَعِنْدَ مُغَازَلَةِ عَيْنَيْهَا، تَنْهَالُ المَسَرَّاتُ عَلَى مَنْ هُوَ بَيْنَ يَدَيْهَا، فَيُحَنْبِشُ وَيُمَحِّش ، وَيُحَفِّشُ وَيَنْتَعِش"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏606‏.
