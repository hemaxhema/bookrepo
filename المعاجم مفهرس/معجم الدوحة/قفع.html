قَفْعَاء[اسْم]

قبل 13ق.هـ=609م

القَفْعَاءُ : نَبْتَةٌ مِنْ نَبَاتِ الرَّبِيعِ فِيهَا رَخَاوَةٌ، لَهَا نَوْرٌ أَحْمَرُ، وَرَقُهَا صَغِيرٌ، خَشِنٌ مُسْتَعْلٍ، وَثَمَرَتُهَا مِنْ تَحْتُ مُتَقَبِّضَةٌ صُلْبَةٌ.

كَأَنَّهَا مِنْ قَطَا الأَجْبَابِ، حَلَّأَهَا وِرْدٌ، وَأَفْرَدَ، عَنْهَا أُخْتَهَا، الشَّرَكُ
جُونِيَّةٌ، كَحَصَاةِ القَسْمِ، مَرْتَعُهَا بِالسِّيِّ مَا تُنْبِتُ القَفْعَاءُ ، وَالحَسَكُ
زهير بن أبي سُلْمى المُزَنيّ
شعر زُهير بن أبي سُلمى: صنعة: الأعلم الشَّنْتَمَريّ (ت، 476هـ)، تحقيق: فخر الدين قباوة، منشورات الآفاق الجديدة، بيروت، ط3، (1400هـ/ 1980م)، ‏82‏.
ن887هـ=1482م

القَفْعَاءُ : تُرْسٌ يُتَّخَذُ مِنَ الخَشَبِ وَنَحْوِهِ، يَمْشِي تَحْتَهُ المُقَاتِلُونَ إِلَى الحُصُونِ فِي الحَرْبِ.

وَاللُّطْفُ يُجْزِئُ عَنْ قَفْعَاءَ كَانِفَةٍ مِنَ الحَدِيدِ، وَعَنْ عَالٍ مِنَ الأُطُمِ
الشهاب المنصوري
ديوان الشّهاب المنصوري (تحقيق ودراسة): أحمد بن محمَّد بن علي (ت، 887هـ)، تحقيق: جيهان أحمد إبراهيم السَّجيني، رسالة ماجستير، جامعة طنطا، طنطا، (1424ه/ 2003م)، ‏536‏.

تَقَفَّعَ[لَازِم]

12هـ=633م

تَقَفَّعَ الشَّيْءُ: تَيَبَّسَ وَتَقَبَّضَ؛ مِنْ بَرْدٍ وَنَحْوِهِ.

فَعَيْنَيَّ، هَلَّا تَبْكِيَانِ لِمَالِكٍ إِذَا أَذْرَتِ الرِّيحُ الكَنِيفَ المُرَفَّعَا
وَهَبَّتْ شَمَالًا مِنْ تُجَاهِ أُظَايِفٍ إِذَا صَادَفَتْ كَفَّ المُفِيضِ، تَقَفَّعَا
مُتَمِّم بن نُويرة اليَرْبوعيّ التَّميميّ
مالك ومتمّم ابنا نُويرة اليربوعي: تحقيق: ابتسام مرهون الصفار، مطبعة الإرشاد، بغداد، (1968م)، ‏109‏.

مُنْقَفِع[صِفَةٌ مُشَبَّهَة]

ن20هـ=641م

المُنْقَفِعُ مِنَ الأَشْيَاءِ: اليَابِسُ المُتَقَبِّضُ؛ مِنْ بَرْدٍ وَنَحْوِهِ.

فَخَيَّمَتْ فِي ذَنَبَانٍ مُنْقَفِعْ وَفِي رُفُوضِ كَلَإٍ غَيْرِ قَشِعْ
أبو محمّد الحَذْلَميّ الفَقْعَسيّ الأسديّ
ما تبقّى من أراجيز أبي محمّد عبد الله بن ربعي بن خالد الحَذْلَميّ الفَقْعَسيّ الأسديّ: جمعها وحققها وشرحها: محمّد جبار المعيبد، وزارة الثقافة والإعلام، بغداد، ط1، (2001م)، ‏66‏.

قَفْعَة[اسْم]

ن23هـ=644م

القَفْعَةُ : الوِعَاءُ الصَّغِيرُ المُسْتَدِيرُ بِلَا عُرْوَةٍ، الوَاسِعُ الأَعْلَى، الضَيِّقُ الأَسْفَلِ، يُتَّخَذُ مِنْ سَعَفِ النَّخْلِ، وَيُجْتَنَى فِيهِ الرُّطَبُ وَنَحْوُهُ.

"وَدِدْتُ أَنَّ عِنْدَنَا قَفْعَةً نَأْكُلُ مِنْهَا"
عمر بن الخطّاب
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏19/‏243،،‏(رقم الحديث: 19030).
ن175هـ=791م

القَفْعَةُ : الزَّائِدَةُ المُسْتَدِيرَةُ الَّتِي تَتَذَبْذَبُ.

"وَشَهِدَ عِنْدَ بَعْضِ القُضَاةِ قَوْمٌ عَلَيْهِمْ خِفَافٌ لَهَا قُفَعٌ ، أَيْ: هَنَاتٌ مُسَتَدِيرَةٌ تَتَذَبْذَبُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏176‏.
ن175هـ=791م

القَفْعَةُ : قِطْعَةٌ دَائِرِيَّةٌ تُتَّخَذُ مِنْ لِيفِ النَّخْلِ، يَجْعَلُ فِيهَا الدَّهَّانُونَ السِّمْسِمَ المَطْحُونَ وَنَحْوَهُ، ثُمَّ يُوضَعُ بَعْضُهَا عَلَى بَعْضٍ؛ حَتَّى يَسِيلَ مِنْهَا الدُّهْنُ بَعْدَ ضَغْطِهَا.

"وَتُسَمَّى هَذِهِ الدُّوَارَاتُ الَّتِي يَجْعَلُ فِيهَا الدَّهَّانُونَ السِّمْسِمَ المَطْحُونَ قَفَعَاتٍ . وَهِيَ هَنَاتٌ يُوضَعُ بَعْضُهَا عَلَى بَعْضٍ؛ حَتَّى يَسِيلَ مِنْهَا الدُّهْنُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏176‏.

قَفْعَة[اسْمُ مَرَّة]

ن127هـ=745م

القَفْعَةُ : المَرَّةُ مِنَ القَفْعِ، وَهُوَ الضَّرْبُ بِالمِقْفَعَةِ وَنَحْوِهَا.

"مَرَّ غُلَامٌ بِالقَاسِمِ بْنِ مُخَيْمِرَةَ، فَعَبِثَ بِهِ الغُلَامُ،  فَتَنَاوَلَهُ القَاسِمُ، وَقَفَعَهُ قَفْعَةً شَدِيدَةً"
أبو إسحاق السَّبيعيّ الهَمْدانيّ
غريب الحديث: إبراهيم بن إسحاق الحربيّ (ت، 285هـ)، تحقيق ودراسة: سليمان بن إبراهيم بن محمّد العايد، جامعة أم القرى، مكة المكرمة، دار المدني، جدة، ط1، (1405هـ/ 1985م)، ‏3/‏747‏.

أَقْفَع[صِفَةٌ مُشَبَّهَة]

ن90هـ=709م

الأَقْفَعُ مِنَ الدَّوَابِّ: القَصِيرُ الذَّنَبِ.

إِنَّا وَجَدْنَا العِيسَ خَيْرًا بَقِيَّةً مِنَ القُفْعِ أَذْنَابًا، إِذَا مَا اقْشَعَرَّتِ
الرّاعي النُّمَيريّ
ديوان الراعي النُّمَيْري: جمعه وحققه: راينهرت فايبرت، المعهد الألماني للأبحاث الشرقيّة، بيروت، (1401هـ/ 1980م)، ‏21‏.
ن110هـ=728م

الأَقْفَعُ مِنَ الأَصَابِعِ وَنَحْوِهَا: اليَابِسُ المُتَقَبِّضُ؛ مِنْ بَرْدٍ وَنَحْوِهِ.

يُثَمِّرُ أَوْلَادَ المَخَاضِ ابْنُ دَيْسَقٍ وَيَقْرِي الضِّبَابَ الضَّيْفَ قُفْعًا رَوَاجِبُهْ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏1/‏151‏.

ن175هـ=791م

القَفْعُ : تُرْسٌ تُتَّخَذُ مِنَ الخَشَبِ وَنَحْوِهِ، يَمْشِي تَحْتَهَا المُقَاتِلُونَ بِسِلاَحِهِمْ إِلَى الحُصُونِ فِي الحَرْبِ.

" القَفْعُ : ضَرْبٌ مِنَ الخَشَبِ، يَمْشِي الرِّجَالُ تَحْتَهُ إِلَى الحُصُونِ فِي الحَرْبِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏175‏.

قَفْع[مَصْدَر]

ن458هـ=1066م

القَفْعُ : الضَّرْبُ بِالمِقْفَعَةِ.

"فَأَمَّا القَفْعُ ، فَلَا يَكُونُ إِلَّا عَلَى شَيْءٍ أَجْوَفَ"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
كتاب المخصَّص: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، المطبعة الكبرى الأميرية، بولاق، ط1، (1316- 1321هـ)، ‏6/‏103‏.

قُفَاعِيّ[صِفَة]

ن175هـ=791م

القُفَاعِيُّ مِنَ الأَشْخَاصِ: الأَحْمَرُ الَّذِي يَتَقَشَّرُ أَنْفُهُ مِنْ شِدَّةِ حُمْرَتِهِ.

" القُفَاعِيُّ : الرَّجُلُ الأَحْمَرُ الَّذِي يَتَقَشَّرُ أَنْفُهُ مِنْ شِدَّةِ حُمْرَتِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏176‏.

قُفَّاعَة[اسْم]

ن198هـ=813م

القُفَّاعَةُ : المِصْيَدَةُ تُصْنَعُ مِنْ جَرِيدِ النَّخْلِ تُصَادُ بِهَا الطُّيُورُ وَنَحْوُهَا.

رَأَيْتُ المَسْجِدَ الجَامِـ ـعَ قُفَّاعَةَ إِبْلِيسِ
بَنَاهُ اللهُ، وَالطَّالِـ ـعُ بُرْجٌ غَيْرُ مَنْحُوسِ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏723‏.
ن255هـ=869م

القُفَّاعَةُ : قَفَصٌ مِنْ جَرِيدٍ يُكْفَأُ عَلَى الحَمَامِ وَنَحْوِهِ.

"وَإِذَا سَقَطَ أُخِذَ بِالبَايَكِيرِ وَبِالقُفَّاعَةِ "
الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏3/‏218‏-‏219‏.

قَفِع[صِفَةٌ مُشَبَّهَة]

215هـ=830م

القَفِعُ مِنَ الأَطْرَافِ: اليَابِسُ المُتَقَبِّضُ؛ مِنْ بَرْدٍ وَنَحْوِهِ.

"فَخَرَجْتُ لَا أَتَلَفَّعُ بِوَصِيدَةٍ... وَالرُّكَبَاتُ زَلِعَة، وَالأَطْرَافُ قَفِعَة ، وَالجِسْمُ مُسْلَهِمّ، وَالنَّظَرُ مُدْرَهِمّ"
رجل من الأعراب
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏1/‏147‏.

مُقَفِّعِيّ[صِفَة]

246هـ=860م

المُقَفَّعِيُّ مِنَ النَّثْرِ وَنَحْوِهِ: البَلِيغُ المُحْكَمُ.

مُقَفَّعِيِّ النَّثْرِ عَتَّابِيِّهْ لَا مُجْتَوَى القُرْبِ وَلَا مَابِيِّهْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏6/‏2624‏.

ن282هـ=895م

"وَإِذَا قَبَضَهُ اليُبْسُ، انْقَفَعَ انْقِفَاعًا "

تَقْفِيع[مَصْدَر]

ن283هـ=896م

التَّقْفِيعُ : البُخْلُ.

فَاتْرُكِ التَّقْفِيعَ لِلْأَغْـ مَارِ، وَالْحَقْ بِالكَرَارِزْ
لَا تُصَادِفْ لَيِّنَ الصُّو فِ، فَأَنْتَ اليَوْمَ مَاعِزْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏1157‏.
ن296هـ=909م

التَّقْفِيعُ : التَّيَبُّسُ والتَّقَبُّضُ فِي الأَعْضَاءِ.

كَأَنَّمَا لُؤْلُؤَةٌ رَطْبَةٌ قَدْ أُفْرِغَتْ فِي ذَهَبٍ رَطْبِ
كَأَنَّمَا التَّقْفِيعُ فِي جِسْمِهَا أَنَامِلٌ تَحْسُبُ عَنْ قَلْبِ
ابن المعتزّ
الكشف والتّنبيه على الوصف والتّشبيه: صلاح الدّين الصّفديّ (ت، 764هـ)، حققه وعلق عليه: هلال ناجي، وليد بن أَحمد الحُسين، إِصدارات مجلة الحكمة، لندن، ط1، (1420ه/ 1999م)، ‏350‏.

مُقَفَّع[صِفَةٌ مُشَبَّهَة]

ن291هـ=904م

المُقَفَّعُ مِنَ الأَعْضَاءِ: اليَابِسُ المُتَقَبِّضُ؛ مِنْ بَرْدٍ وَنَحْوِهِ.

"فَتَرْتَعِشُ أَعْضَاؤُهُ كُلُّهَا، ثُمَّ يَتَشَنَّجُ عَقِبُهُ تَشَنُّجًا لَا بُرْءَ مِنْهُ، فَيَمُوتُ مُتَشَنِّجَ الأَطْرَافِ، مُسْوَدَّ الوَجْهِ، مُقَفَّعَ الأَصَابِعِ، مُتَطَحِّنَ الأَسْنَانِ"
ابن وحشيّة النَّبَطيّ
الفلاحة النَّبَطيّة: ابن وحشيّة النَّبَطيّ (ت، 291هـ)، تحقيق: توفيق فهد، المعهد العلميّ الفرنسيّ للدراسات العربيّة، دمشق، الجفان والجابي، قبرص، ط1، (1993م)، ‏2/‏1129‏.

"رَأَيْتُ التَّقَفُّعَ إِنَّمَا يَحْدُثُ بِأَصْحَابِ الأَمْزَاجِ الشَّدِيدَةِ الحَرَارَةِ"

قَافِع[اسْمُ فَاعِل]

ن1311هـ=1893م

القَافِعُ : الضَّارِبُ بِالمِقْفَعَةِ وَنَحْوِهَا.

"وَاعْلَمْ أَنَّ الإِنْكَارَ عَلَى السَّادَاتِ الصُّوفِيَّة، وَالطَّرِيقَةِ العَلِيَّة، المُتَّبِعِينَ لِلسُّنَّةِ السَّنِيَّة، وَالقَافِعِينِ لِلْبِدْعِيَّةِ الرَّدِيَّة، خُصُوصًا أَهْلُ العِلْمِ... سُمٌّ قَاتِلٌ"
ضياء الدين أحمد النقشبندي الخالدي
جامع الأصول في الأولياء: ضياء الدين أحمد النقشبندي الخالدي (ت، 1311هـ)، تحقيق: أديب نصر الدين، مؤسسة الانتشار العربي، بيروت، ط1، (1997م)، ‏2/‏281‏-‏282‏.
