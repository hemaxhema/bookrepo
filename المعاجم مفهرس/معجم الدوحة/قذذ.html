قُذَّة[اسْم]

قبل 50ق.هـ=573م

القُذَّةُ : رِيشَةُ الطَّائِرِ، تُبْرَى وَتُسَوَّى لِتُرَكَّبَ فِي السَّهْمِ.

فَقَلَّدْتُ سَوَّارَ بْنَ عَمْرِو بْنِ مَالِكٍ بِأَسْمَرَ حَشْرِ القُذَّتَيْنِ طَمِيلِ
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏188‏.
ن145هـ=762م

القُذَّةُ : البُرْغُوثُ.

يَا أَبَتَا، أَرَّقَنِي القِذَّانْ فَالنَّوْمُ لَا تَطْعَمُهُ العَيْنَانْ مِنْ وَخْزِ بُرْغُوثٍ لَهُ أَسْنَانْ
رؤبة بن العجّاج
مجموع أشعار العرب وهو مشتمل على ديوان رؤبة بن العجاج وعلى أبيات مفردات منسوبة إليه: اعتنى بتصحيحه وترتيبه: وليم بن الورد البروسيّ، دار ابن قتيبة، الكويت، د.ت. ‏186‏.
ن175هـ=791م

القُذَّةُ : كَلِمَةٌ يَسْتَعْمِلُهَا الصِّبْيَانُ فِي ضَرْبٍ مِنَ اللَّعِبِ.

" القُذَّةُ : كَلِمَةٌ يَقُولُهَا صِبْيَانُ العَرَبِ؛ يَقُولُونَ: لَعِبْنَا شَعَارِيرَ قُذَّةَ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏20‏.

مَقَذّ[اسْم]

قبل 13ق.هـ=609م

المَقَذُّ : مُنْتَهَى مَنْبِتِ الشَّعْرِ مِنْ مُؤَخَّرِ الرَّأْسِ.

حَلَبَ الهَجِيرُ بِلِيتِهَا وَمَقَذِّهَا حَتَّى كَأَنَّ بِهَا عَنِيَّةَ قَارِ
بَيْهس بن عبد الحارث الغَطَفانيّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏9/‏60‏.

أَقَذّ[صِفَةٌ مُشَبَّهَة]

ن27هـ=648م

الأَقَذُّ مِنَ السِّهَامِ: مَا أُلْزِقَ بِهِ الرِّيشُ بَعْدَ قَشْرِهِ وَبَرْيِهِ.

فَجَاءَ بِهَا بَعْدَ الكَلَالِ كَأَنَّهُ مِنَ الأَيْنِ مِحْرَاسٌ أَقَذُّ سَحِيجُ
أبو ذؤيب الهُذَليّ
ديوان أبي ذؤيب الهُذلي: تحقيق وإخراج: أحمد خليل الشال، مركز الدراسات والبحوث الإسلامية، بورسعيد، ط1، (1435هـ/ 2014م)، ‏84‏.

قُذَاذَة[اسْم]

ن100هـ=718م

القُذَاذَةُ مِنَ الشَّيْءِ: مَا قُطِعَ مِنْهُ وَسَقَطَ.

سَجَنْتَ لِسَانِي يَا ابْنَ حَيَّانَ بَعْدَمَا تَوَلَّى شَبَابِي، إِنَّ عَقْدَكَ مُحْكَمُ
وَعِيدُكَ أَبْقَى مِنْ لِسَانِي قُذَاذَةً هَيُوبًا، وَصَمْتًا بَعْدُ لَا يَتَكَلَّمُ
شَبيب بن البرصاء المُرّيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏12/‏200‏.

مُقَذَّذ[اسْمُ مَفْعُول]

126هـ=744م

المُقَذَّذُ مِنَ الأَشْيَاءِ: مَا زُيِّنَ وَسُوِّيَ وَأُلْطِفَ.

لَمْ تَرَ العَيْنُ مِثْلَهُ فِي مَنَامٍ، وَلَا كَذَا
تَامِكًا كَالسَّنَامِ إِذْ بُذَّ عَنْهُ مُقَذَّذَا
ذو كِبار الهَمْدانيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏24/‏122‏.
ن175هـ=791م

المُقَذَّذُ مِنَ الرِّجَالِ: المُقَصَّصُ شَعْرُهُ حَوَالَيْ قُصَاصِهِ كُلِّهِ.

"وَرَجُلٌ مُقَذَّذٌ ، أَيْ مُقَصَّصٌ شَعْرُهُ حَوَالَيْ قُصَاصِهِ كُلِّهِ
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏20‏.
ن231هـ=845م

المُقَذَّذُ مِنَ السِّهَامِ: الَّذِي رُكِّبَتْ عَلَيْهِ القُذَّةُ، فَلَا يَطِيشُ.

فَأَعِزَّ ذِلَّةَ رُجْلَتِي بِمُهَذَّبٍ حُلْوِ المَخِيلِ مُقَذَّذٍ مَقْدُودِ
أبو تمّام
شرح ديوان أبي تمّام: الخطيب التِّبْريزي (ت، 502هـ)، قدم له ووضع هوامشه وفهارسه: راجي الأسمر، دار الكتاب العربي، بيروت، ط2، (1414هـ/ 1994م)، ‏1/‏306‏.

قَذَّ[مُتَعَدٍّ]

ن130هـ=748م

قَذَّ الشَّخْصَ: ضَرَبَهُ عَلَى قَفَاهُ.

قَامَ إِلَيْهَا رَجُلٌ، فِيهِ عُنُفْ لَهُ ذِرَاعٌ ذَاتُ نِيرَيْنِ وَكَفْ فَقَذَّهَا بَيْنَ قَفَاهَا وَالكَتِفْ
أبو وَجْزة السُّلَميّ السَّعْديّ
شعر أبي وَجْزة السَّعْديّ (ت، 130هـ): تحقيق: وليد السراقبي، مجلة معهد المخطوطات العربیة، القاهرة، مج34، ج(1، 2)، (1410هـ)، ‏60‏.
ن206هـ=821م

قَذَّ الشَّيْءَ: قَطَعَهُ مِنْ أَطْرَافِهِ وَسَوَّاهُ.

"الرِّيشَةُ يُقَذُّ مِنْهَا جَنْبَاهَا، فَوَاحِدٌ يُسَمَّى: الظُّهْرَانَ، وَالآخَرُ يُسَمَّى: البُطْنَانَ"
أبو عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏221‏.
352هـ=963م

قَذَّهُ : ضَرَبَهُ بِأَطْرَافِ الأَصَابِعِ.

"يُقَالُ: قَدْ قَذَذْتُ قَفَا الغُلَامِ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏2/‏188‏.

مَقْذُوذ[اسْمُ مَفْعُول]

132هـ=749م

المَقْذُوذُ مِنَ الأَشْيَاءِ: مَا بُرِيَ وَحُدِّدَ.

مَقْذُوذَةُ الآذَانِ صَدْقَاتُ الحَدَقْ قَدْ أَحْصَنَتْ مِثْلَ دَعَامِيصِ الرَّنَقْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏15‏.
ن300هـ=912م

المَقْذُوذُ مِنَ المَلَابِسِ وَنَحْوِهَا: المُسَوَّى، المُزَيَّنُ.

يَا لَيْتَنِي كُنْتُ لِعَمْرٍو عُوذَهْ أَوْ حُلَّةً يَلْبَسُهَا مَقْذُوذَهْ
مُدرِك بن عليّ الشَّيْبانيّ
نِشوار المحاضرة وأخبار المذاكرة: أبو عليّ التَّنوخيّ (ت، 384هـ)، تحقيق: عبّود الشالجي، دار صادر، بيروت، ط2، (1995م)، ‏4/‏268‏.
765هـ=1363م

المَقْذُوذُ مِنَ الأَشْيَاءِ: المَقْطُوعُ المَقْصُوصُ.

إنَّ الخَطُوبَ إِذَا عَشَوْتَ لِنَارِهِ تُمْسِي الخُطُوبُ وَعُودُهَا مَقْذُوذُ
ابن جابر الأندلسيّ
ديوان المقصد الصّالِح في مدح الملك الصّالِح: ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: أحمد فوزي الهيب، دار سعد الدِّين، دمشق، ط1، (1429ه/ 2008م)، ‏164‏.

قَذّ[مَصْدَر]

ن175هـ=791م

قَذُّ الشَّيْءِ: قَطْعُ أَطْرَافِهِ وَتَسْوِيَتُهُ.

" القَذُّ : قَطْعُ أَطْرَافِ الرِّيشِ عَلَى مِثَالِ الحَذْفِ، وَالتَّحْذِيفِ، وَكَذَلِكَ كُلُّ قَطْعٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏20‏.
ن296هـ=909م

قَذُّ السَّهْمِ: إِلْصَاقُ القُذَّةِ بِهِ.

أَنْعَتُ أَمْثَالًا قُذِذْنَ قَذَّا يَشْخَذُهَا الشَّوْطُ البَطِينُ شَخْذَا
ابن المعتزّ
ديوان أشعار الأمير أبي العبّاس عبد الله بن محمّد المعتزّ بالله الخليفة العباسي: دراسة وتحقيق: محمّد بديع شريف، دار المعارف، القاهرة، (1977م)، ‏124‏.

ن250هـ=864م

"يُقَالُ لِلسِّكِّينِ وَمَا قُذَّ بِهِ الرِّيشُ: مِقَذٌّ "

تَقْذِيذ[مَصْدَر]

ن282هـ=895م

تَقْذِيذُ السِّهَامِ وَنَحْوِهَا: قَطْعُهَا مِنْ أَطْرَافِهَا وَتَسْوِيَتُهَا.

"اللَّغْبُ أَنْ يَكُونَ رِيشَتَانِ مِنْ ظُهُورِ الرِّيشِ، وَالثَّالِثَةُ مِنَ البَطْنِ، فَلَا يَزَالُ السَّهْمُ مُضْطَرِبًا. قَالَ: وَلَا يَفْعَلُ هَذَا مَنْ لَهُ عِلْمٌ بِتَقْذِيذِ السِّهَامِ"
أبو حنيفة الدّينوريّ
كتاب النّبات: أبو حنيفة الدّينوريّ (ت، 282هـ)، حققه وشرحه وقدم له: برنهارد لفين، فرانز شتاينر بقيسبادن، (1394هـ/ 1974م)، ‏354‏.
543هـ=1148م

تَقْذِيذُ العَيْنِ: إِخْرَاجُ القَذَى مِنْهَا.

"فَلَعَنَ اللهُ أَبَا الحُسَيْن، إِنْ عَادَ إِلَى لُبْسِ خُفَّيْ حُنَيْن، بَيْدَ أَنْ يَجْرِيَ القَدَرُ بِإِذْهَابِ الجَفَا، وَتَقْذِيذٍ فِي العَيْنِ مِنْ قَذَى"
ابن منير الطرابلسيّ
خريدة القصر وجريدة العصر: عماد الدين الأصفهانيّ (ت، 597هـ)، تحقيق: مبروك المناعي وآخرين، دار مداد يونيفارسيتي برس، قسنطينة، ط3، (2018م)، ‏13/‏129‏.
