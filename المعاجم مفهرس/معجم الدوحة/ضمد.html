ضَمَد[اسْم]

قبل 13ق.هـ=609م

الضَّمَدُ : الغَيْظُ الكَامِنُ فِي الصَّدْرِ.

فَمَنْ أَطَاعَكَ فَانْفَعْهُ بِطَاعَتِهِ كَمَا أَطَاعَكَ، وَادْلُلْهُ عَلَى الرَّشَدِ
وَمَنْ عَصَاكَ فَعَاقِبْهُ مُعَاقَبَةً تَنْهَى الظَّلُومَ، وَلَا تَقْعُدْ عَلَى ضَمَدِ
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏82‏.
قبل 13ق.هـ=609م

الضَّمَدُ : الدَّمُ الجَافُّ اليَابِسُ.

وَمَا هُرِيقَ عَلَى غَرِيِّكَ الضَّمَدُ
النّابغة الذُّبْيانيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏12/‏5‏.

ضَمَدَ[مُتَعَدٍّ]

ن11هـ=632م

ضَمَدَ العُضْوَ: شَدَّهُ بِالضِّمَادِ وَعَصَبَهُ.

" اضْمِدْهُمَا بِالصَّبِرِ"
حديث نبويّ
الجامع الصحيح وهو سُنن الترمذي: التّرمذيّ (ت، 279هـ)، تحقيق وشرح: أحمد محمد شاكر، وآخرين، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1382هـ/ 1962م- 1398هـ/ 1978م)، ‏3/‏278،،‏(رقم الحديث: 952).
ن27هـ=648م

ضَمَدَتِ المَرْأَةُ زَوْجَهَا وَغَيْرَهُ: اتَّخَذَتْ مَعَهُ خَلِيلًا أَوْ أَكْثَرَ.

تُرِيدِينَ كَيْمَا تَضْمِدِينِي وَخَالِدًا وَهَلْ يُجْمَعُ السَّيْفَانِ، وَيْحَكِ، فِي غِمْدِ؟
أبو ذؤيب الهُذَليّ
إصلاح المنطق: ابن السّكّيت (ت، 244هـ)، شرح وتحقيق: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط4، (1368هـ/ 1949م)، ‏50‏.
ن206هـ=821م

ضَمَدَ الثَّوْرَيْنِ: قَرَنَهُمَا بِالضِّمَادِ.

" ضَمَدْتُ الثَّوْرَيْنِ"
الحارثي
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏198‏.

اضْطَمَدَ[لَازِم]

ن58هـ=678م

اضْطَمَدَ الشَّخْصُ: الْتَفَّ بِالضِّمَادِ.

"كُنَّ يَخْرُجْنَ مَعَ رَسُولِ اللهِ  @  عَلَيْهِنَّ الضِّمَادُ، قَدِ اضْطَمَدْنَ قَبْلَ أَنْ يُحْرِمْنَ"
عائشة بنت أبي بكر الصِّدّيق
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏41/‏50،،‏(رقم الحديث: 24502).

ضِمَاد[اسْم]

ن58هـ=678م

الضِّمَادُ : مَا يُلَفُّ عَلَى العُضْوِ المَمْسُوحِ أَوِ المَجْرُوحِ مِنْ خِرْقَةٍ وَنَحْوِهَا.

"كُنَّ يَخْرُجْنَ مَعَ رَسُولِ اللهِ  @  عَلَيْهِنَّ الضِّمَادُ ، قَدِ اضْطَمَدْنَ قَبْلَ أَنْ يُحْرِمْنَ"
عائشة بنت أبي بكر الصِّدّيق
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏41/‏50،،‏(رقم الحديث: 24502).
235هـ=849م

الضِّمَادُ : مَا يُلَفُّ عَلَى العُضْوِ مِنْ خِرْقَةٍ وَنَحْوِهَا مَعَ الدَّوَاءِ.

"وَتَكُونُ فَاكِهَتُهُ الرُّمَّانَ وَالسَّفَرْجَلَ، وَشَرَابُهُ المَيْبَةُ، أَوْ طِلَاءً طَيِّبَ الرِّيحِ، وُتُضَمَّدُ بِضِمَادَاتٍ طَيِّبَةِ الرِّيحِ مُقَوِّيَةٍ لِلْكَبِدِ"
ابن سهل الطَّبريّ
فردوس الحكمة في الطّب: ابن سهل الطَّبريّ (ت، 247هـ)، اعتنى بنسخه وتصحيحه: محمّد زبير الصديقي، مطبع آفتاب، برلين، (1928م)، ‏221‏.

ضِمَاد[مَصْدَر]

ن100هـ=718م

الضِّمَادُ : اتِّخَاذُ المَرْأَةِ خَلِيلَيْنِ أَوْ أَكْثَرَ.

لَا يُخْلِصُ، الدَّهْرَ ، خَلِيلٌ عَشْرَا ذَاقَ الضِّمَادَ ، أَوْ يَزُورَ القَبْرَا إِنِّي رَأَيْتُ الضَّمْدَ شَيْئًا نُكْرَا
مدرك بن حصن الفَقْعَسيّ الأسديّ
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏242‏.

ضَمَّدَ[مُتَعَدٍّ]

ن58هـ=678م

ضَمَّدَ وَجْهَهُ بِالطِّيبِ: ضَمَّخَهُ وَلَطَّخَهُ.

"كُنَّا نَخْرُجُ مَعَ النَّبِىِّ @  صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ إِلَى مَكَّةَ، فَنُضَمِّدُ جِبَاهَنَا بِالسُّكِّ المُطَيَّبِ عِنْدَ الإِحْرَامِ"
عائشة بنت أبي بكر الصِّدّيق
سنن أبي داود: أبو داود السِّجِسْتانيّ (ت، 275هـ)، حكم على أحاديثه وآثاره وعلّق عليه: محمّد ناصر الدين الألبانيّ، اعتنى به: مشهور حسن سلمان، مكتبة المعارف، الرياض، ط2، (1424هـ)، ‏319،،‏(رقم الحديث: 1830).

ضَمْد[اسْم]

ن83هـ=702م

الضَّمْدُ : نَبَاتُ الأَرْضِ رَطْبًا وَيَابِسًا.

إِذَا مَا هَبَطْنَ بِنَا سَبْسَبًا وَجَاوَزْنَ بَعْدَ جِبَالٍ جِبَالَا
وَمَارَتْ قَلَائِدُ أَعْنَاقِهَا وَغادَرْنَ فِي كُلِّ ضَمْدٍ نِعَالَا
أعشى هَمْدان
ديوان أعشى هَمدان وأخباره: تحقيق: حسن عيسى أبو ياسين، دار العلوم، الرياض، (1403هـ/ 1983م)، ‏153‏.
ن651هـ=1253م

الضَّمْدُ مِنَ الغَنَمِ: الصَّغِيرُ وَالكَبِيرُ.

أَمَّا الحِرَاثَةُ سَرَّحُوا أَضْمَادَهُمْ مَا أَنْ بَقِي أَحَدٌ يُرَكِّبُ مِضْمَدَا
الوصَابيّ الهَمْدَانيّ
ديوان أبي عبد الله جمال الدِّين محمّد بن حمير الوصَابيّ الهَمْدَانيّ: الوصَابيّ الهَمْدَانيّ (ت، 651هـ)، حققه وعلق عليه: محمَّد بن علي بن الحُسين الأَكوع الحوالي، دار العودة، بيروت، ط1، (1405ه/ 1985م)، ‏98‏.

مِضْمَدَة[اسْم]

ن255هـ=869م

المِضْمَدَةُ : خَشَبَةٌ تُجْعَلُ عَلَى أَعْنَاقِ الثَّوْرَيْنِ، فِي طَرَفَيْهَا ثَقْبَانِ، فِي كُلِّ وَاحِدَةٍ مِنْهَا ثُقْبَةٌ بَيْنَهُمَا فَرْضٌ فِي ظَهْرِهَا، ثُمَّ يُجْعَلُ في الثَّقْبَيْنِ خَيْطٌ يُخْرَجُ طَرَفَاهُ مِنْ بَاطِنِ المِضْمَدَةِ، وَيُوثَقُ فِي طَرَفِ كُلِّ خَيْطٍ عُودٌ، يُجْعَلُ عُنُقُ الثَّوْرِ بَيْنَ العُودَيْنِ".

"الجَرُّ: الحَبْلُ الَّذِي فِي طَرَفِ اللُّؤْمَةِ إِلَى وَسَطِ المِضْمَدَةِ "
أبو حاتم السِّجِسْتانيّ
البارع في اللُّغة: أبو عليّ القالي (ت، 356هـ)، تحقيق: هاشم الطعّان، مكتبة النهضة، بغداد، دار الحضارة العربية، بيروت، ط1، (1975م)، ‏568‏.

أَضْمَدَ[مُتَعَدٍّ]

ن288هـ=901م

أَضْمَدَ الجُرْحَ: ضَمَدَهُ بِالضِّمَادِ.

"فَإِنْ كَانَ وَقْتُ وَرَقِ البَنْجِ فَاتُّخِذَتْ مِنْهُ وَمِنْ لُبَابِ الخُبْزِ الحُوَّارِيِّ ضِمَادٌ أُضْمِدَ بِهِ نَفَعَ"
ثابت بن قُرّة الحَرّانيّ
كتاب الذّخيرة في علم الطّبّ: ثابت بن قُرّة الحَرّانيّ (ت، 288هـ)، تحقيق: جورج صبحي، المطبعة الأميريّة، القاهرة، (1928م)، ‏133‏.
ن1008هـ=1599م

أَضْمَدَ الطِّحَالَ وَنَحْوَهُ: عَالَجَهُ.

"الطَّرْفَا: إِذَا شُرِبَ مِنْ عَصِيرِ وَرَقِهَا، أَوْ طُبِخَ مِنْ زَهْرِهَا أَضْمَدَ الطِّحَالَ"
داود بن عمر الأنطاكي
مجمع المنافع البدنية: داود بن عمر الأنطاكي (ت، 1008هـ)، تحقيق: أحمد فريد المزيدي، دار الكتب العلمية، بيروت، ط1، (1424ه/ 2002م)، ‏68‏.

مُضَمِّد[اسْمُ فَاعِل]

ن313هـ=925م

المُضَمِّدُ : الَّذِي يُعَالِجُ الجُرُوحَ بِوَضْعِ الدُّهْنِ وَنَحْوِهِ عَلَيْهَا، ثُمَّ يَشُدُّ عَلَيْهِ بِالضَّمَادِ.

"وَالصَّيَادِلَةُ العَطَّارُونَ، وَالطَّبَّاخُونَ، وَالمُضَمِّدُونَ ... وَإِنْ كَانَ الطَّبِيبُ قَدْ يَعْمَلُ هَذِهِ الأَعْمَالَ فِي بَعْضِ الأَحَايِينِ، فَإِنَّمَا يَعْمَلُ عَمَلَ مَا يَعْمَلُ مِنْ ذَلِكَ بِمَنْزِلَةِ مَا يَمُدُّ يَدَهُ النَّجَّارُ، فِي بَعْضِ الأَحَايِينِ، إِلَى المِجْذَافِ أَوِ الدَّقَلِ، فَيَعْمَلُ بِهِ شَيْئًا مِنْ أَعْمَالِ المَلَّاحِينَ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏7/‏8‏.

ضِمَادَة[مَصْدَر]

ن393هـ=1003م

الضِّمَادَةُ مِنَ الأَمْرِ: الإِشْرَافُ عَلَيْهِ، وَالدُّنُوُّ مِنْهُ.

"أَنَا عَلَى ضِمَادَةٍ مِنَ الأَمْرِ"
أبو نصر الجوهريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏2/‏501‏.

تَضَمُّد[مَصْدَر]

418هـ=1027م

التَّضَمُّدُ بِالشَّيْءِ: الِادِّهَانُ بِهِ، وَشَدُّهُ بِالضِّمَادِ.

"أَعْضَاءُ الغِذَاءِ: بَعَرُ المَاعِزِ، خُصُوصًا الجَبَلِيُّ لِلْيَرَقَانِ، يُشْرَبُ بِبَعْضِ الأَفَاوِيهِ، مُجَرَّبٌ، وَيَنْفَعُ فِي الِاسْتِسْقَاءِ ضِمَادًا وَشُرْبًا، وَلْيَكُنِ التَّضَمُّدُ وَالتَّطَلِّي بِهِ فِي الشَّمْسِ"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏1/‏468‏.

ضِمَادِيّ[اسْمٌ مَنْسُوب]

418هـ=1027م

الضِّمَادِيُّ مِنَ الأَدْوِيَةِ: المَنْسُوبُ إِلَى الضِّمَادِ.

"وَأَمَّا الأَدْوِيَةُ الضِّمَادِيَّةُ المُفْرَدَةُ، وَالضِّمَادَاتُ المُرَكَّبَةُ النَّافِعَةُ، فِي هَذِهِ العِلَّةِ، فَقَدْ ذَكَرْنَا كَثِيرًا مِنْهَا فِي الأَقْرَابَاذِينِ"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏2/‏546‏.

مَضْمُود[اسْمُ مَفْعُول]

418هـ=1027م

المَضْمُودُ بِهِ مِنَ الأَدْوِيَةِ: الَّذِي يُدَّهَنُ بِهِ، وَيُشَدُّ عَلَيْهِ بِالضِّمَادِ.

"وَإِنْ كَانَ مِنْ بَرْدٍ عُولِجَ بِالمُسَخِّنَاتِ المَشْرُوبَةِ وَالمَضْمُودِ بِهَا، مِمَّا قَدْ شُرِحَ فِي مَوْضِعِهِ، وَجُعِلَ غِذَاؤُهُ مِنَ القَنَابِرِ وَالعَصَافِيرِ المَشْوِيَّةِ وَالفِرَاخِ أَيْضًا"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏2/‏471‏.

مُضَمَّد[اسْمُ مَفْعُول]

418هـ=1027م

المُضَمَّدُ بِهِ مِنَ الأَدْوِيَةِ: الَّذِي يُدَّهَنُ بِهِ، وَيُشَدُّ عَلَيْهِ بِالضِّمَادِ.

"إِنْ كَانَ السَّبَبُ وَرَمًا، عُولِجَ الوَرَمُ، وَأُرْخِيَ، وَلُيِّنَ، وَاسْتُعْمِلَ التَّعْرِيقُ فِي حَمَّامٍ مَائِيٍّ، وَالمُلَيِّنَاتُ المُضَمَّدُ بِهَا"

انْضِمَاد[مَصْدَر]

ن656هـ=1258م

انْضِمَادُ مَوْضِعِ المَرَضِ: انْكِمَاشُهُ وَتَجَمُّعُهُ.

"وَعَلَامَتُهُ: جَفَافُ المُلْتَحِمِ وَانْضِمَادُهُ ، وَفِي العَيْنِ غُؤُورٌ وَيُبْسٌ، وَرُبَّمَا احْمَرَّتْ قَلِيلًا مَعَ حُمْرَةِ الأَجْفَانِ، وَصُدَاعٍ مُبَرِّحٍ"
الحلبيّ
الكافي في الكحل: الحلبيّ (ت، 656هـ)، تحقيق: محمد ظافر وفائي، محمد رواس قلعه جي، المنظمة الإسلامية للتربية والعلوم والثقافة (إيسيسكو)، الرباط، ط1، (1410ه/ 1990م)، ‏183‏.

ضَمِيدَة[اسْم]

ن911هـ=1505م

الضَّمِيدَةُ : الضِّمَادُ.

"تًأْخُذُ الفَيْجَلَ، وَتَهْرِسُهُ وَتُغَرْبِلُهُ وَتَخْلِطُهُ بِحَلِيبِ النِّسَاءِ، وَيُحْلَقُ رَأْسُهُ وَيُجْعَلُ عَلَيْهِ ضَمِيدَةٌ ، يَبْرَأُ بِإِذْنِ اللهِ تَعَالَى"
جلال الدّين السَّيوطي
الرحمة في الطب والحكمة: جلال الدّين السَّيوطي (ت، 911هـ)، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، (د.ت)، ‏32‏.

ضَامِد[اسْمُ فَاعِل]

ن1111هـ=1699م

ضَامِدُ وَجْهِهِ بِالطِّيبِ وَنَحْوِهِ: مُضَمِّخُهُ وَمُلَطِّخُهُ.

وَكَبَّرَتْ كُلُّ عَجُوزٍ غَوْرَمِ ضَامِدَةً جَبْهَتَهَا بِالكُرْكُمِ
من شواهد محمد الأمين بن فضل الله المحبي
قصد السبيل فيما في اللغة العربية من الدخيل: محمد الأمين بن فضل الله المحبي (ت، 1111هـ)، تحقيق وشرح: عثمان محمود الصيني، مكتبة التوبة، الرياض، ط1، (1415ه/ 1994م)، ‏2/‏393‏.
1270هـ=1853م

الضَّامِدُ مِنَ النِّسَاءِ: الَّتِي تَتَّخِذُ مَعَ زَوْجِهَا خَلِيلًا، أَوْ تَتَّخِذُ خَلِيلَيْنِ.

"وَمِنْهُنَّ الشَّفِيرَةُ، وَهْيَ القَانِعَةُ مِنَ البِعَالِ بِأَيْسَرِهِ، وَمِنْهُنَّ الضَّامِدُ ، وَهْيَ الَّتِي تَتَّخِذُ خَلِيلَيْنِ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏530‏.
