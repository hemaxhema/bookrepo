احْتَنَطَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

احْتَنَطَ المَيِّتَ: طَيَّبَهُ بِالحَنُوطِ.

وَكُلُّ مُجْتَمِعٍ لَا بُدَّ مُفْتَرِقٌ وَكُلُّ ذِي عُمُرٍ يَوْمًا سَيُحْتَنَطُ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: شرح: أشرف أحمد عدرة، دار الكتاب العربي، بيروت، ط1، (1414هـ/ 1994م)، ‏81‏.

حِنْطِيّ[صِفَة]

ن7ق.هـ=615م

الحِنْطِيُّ : الَّذِي يَأْكُلُ الحِنْطَةَ، وَيَسْمَنُ عَلَيْهَا.

وَالحِنْطِئُ الحِنْطِيُّ يُمْـ ـثَجُ بِالعَظِيمَةِ وَالرَّغَائِبْ
الأعلم الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏1/‏316‏.
ن320هـ=932م

الحِنْطِيُّ مِنَ الدَّقِيقَ: الخَالِصُ المَحْضُ، لَمْ يُنْزَعْ مِنْهُ شَيْءٌ.

"وَمِنْهُ مَا لَا يُنْزَعُ مِنْهُ شَيْءٌ أَصْلًا، وَهُوَ المَعْرُوفُ بِخُبْزِ الخُشْكَارِ، وَالأَوَائِلُ تُسَمِّيهِ الدَّقِيقَ الحِنْطِيَّ ؛ أَيْ: حِنْطَةٌ خَاصَّةٌ مَحْضَةٌ لَمْ يُمَيَّزْ مِنْ أَجْزَائِهَا شَيْءٌ"
إسحاق بن سليمان الإسرائيليّ
كتاب الأغذية والأدوية: إسحاق بن سليمان الإسرائيليّ (ت، 320هـ)، تحقيق: محمّد الصبّاح، مؤسسة عز الدين، بيروت، ط1، (1412هـ/ 1992م)، ‏191‏.
418هـ=1027م

الحِنْطِيُّ مِنَ الأَغْذِيَةِ وَنَحْوِهَا: المُتَّخَذُ مِنَ الحِنْطَةِ.

"إِنْ لَمْ يَكُنْ بُدٌّ مِنْ عِلَاجِهِ فَلْيَكُنِ الغَرَضُ تَسْكِينَ الوَجَعِ، وَأَنْ يُنَقَّى البَدَنُ وَنَاحِيَةُ الرَّأْسِ مِنَ الخَلْطِ العَكِرِ، وَيَغْتَذِيَ بِالأَغْذِيَةِ الجَيِّدَةِ الكَيْمُوسِ الحِنْطِيَّةِ الَّتِي لَا تَسْخِينَ فِيهَا"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏2/‏181‏.
ن456هـ=1064م

الحِنْطِيُّ اللَّوْنِ: الدَّاكِنُ، بَيْنَ السُّمْرَةِ وَالحُمْرَةِ.

"وَيَنْفَعُ مِنْهَا التَّدْهِينُ دَائِمًا بِدُهْنِ الوَرْدِ وَالشُّحُومِ، وَقَدْ تُقْطَعُ أَوْ تُقْلَعُ بِالدَّوَاءِ الحَادِّ، وَخَاصَّةً مَا يُعْرَفُ مِنْهَا بِالثَّآلِيلِ العَدَسِيَّةِ وَالحِنْطِيَّةِ "
عبد الله بن محمّد الأزديّ الصُّحاريّ
الماء: عبد الله بن محمّد الأزديّ الصُّحاريّ (ت، 456هـ)، حققه: هادي حسن حمّودي، وزارة التراث والثقافة، مسقط، ط2، (1436هـ/ 2015م)، ‏1/‏323‏.

حِنْطَة[اسْم]

6هـ=627م

الحِنْطَةُ : حَبُّ القَمْحِ المَعْرُوفُ بِالبُرِّ.

"وَاللهِ لَا يَأْتِيكُمْ مِنَ اليَمَامَةِ حَبَّةُ حِنْطَةٍ حَتَّى يَأْذَنَ فِيهَا رَسُولُ اللهِ @ "
ثُمامة بن أُثال الحَنَفيّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏3/‏1387،،‏(رقم الحديث: 1764).
ن313هـ=925م

الحِنْطَةُ السَّوْدَاءُ : نَبْتَةٌ آسِيَوِيَّةٌ سَنَوِيَّةٌ، ذَاتُ ثِمَارٍ صَغِيرَةٍ، مُثَلَّثِيَّةُ الشَّكْلِ، شَبِيهَةٌ بِالبُذُورِ.

" حِنْطَةٌ سَوْدَاءُ : قَالَ جَالِينُوس: إِنَّهُ غِذَاءٌ كَالحِنْطَةِ وَلَهُ تَقْوِيَةٌ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏6/‏100‏.

حَنَّطَ[مُتَعَدٍّ]

10هـ=631م

حَنَّطَ المَيِّتَ: طَيَّبَهُ بِالحَنُوطِ.

"اغْسِلُوهُ بِمَاءٍ وَسِدْرٍ، وَكَفِّنُوهُ فِي ثَوْبَيْنِ، وَلَا تُحَنِّطُوهُ ، وَلَا تُخَمِّرُوا رَأْسَهُ؛ فَإِنَّهُ يُبْعَثُ يَوْمَ القِيَامَةِ مُلَبِّيًا"
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏1/‏391،،‏(رقم الحديث: 1265).
1371هـ=1951م

حَنَّطَ الجُثَّةَ: صَانَهَا مِنَ التَّفَسُّخِ وَالتَّعَفُّنِ بِطُرُقٍ وَمَوَادَّ تَحْفَظُهَا مِنَ البِلَى.

"نَعْلَمُ أَنَّ العُلَمَاءَ الَّذِينَ رَافَقُوا الحَمْلَةَ كَانُوا يَدْعُونَ أَحْيَانًا بَعْضَ المِصْرِيِّينَ ـ وَلَا سِيَّمَا المُوَظَّفِينَ مِنْهُمْ ـ إِلَى زِيَارَةِ مَقَرِّ أَعْمَالِهِمْ، وَكَانُوا يُطْلِعُونَهُمْ خِلَالَ هَذِهِ الزِّيَارَاتِ عَلَى الآلَاتِ الَّتِي أَتَوْا بِهَا وَالصُّوَرِ الَّتِي رَسَمُوهَا، وَالحَيَوَانَاتِ الَّتِي حَنَّطُوهَا "
ساطع الحصري
آراء وأحاديث في التاريخ والاجتماع: ساطع الحصري (ت، 1388هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2022م)، ‏45‏.
1407هـ=1986م

حَنَّطَ الأَمْرَ: جَمَّدَهُ وَحَافَظَ عَلَيْهِ دُونَ تَغْيِيرٍ.

"أَصْبَحَ مِنَ الضَّرُورِي أَنْ تُرَاجَعَ مَبَادِئُ حُنِّطَتْ فِي قَوَالِبَ أَمْثِلَةٍ، فَبَدَا فِي مَرْحَلَةٍ أُولَى أَنَّ الضَّرُورَةَ أَوْجَبَتِ التَّخَلِّيَ عَنْ أَهَمِّ تِلْكَ المَبَادِئِ، أَيْ مَبْدَأِ السَّبَبِيَّةِ"
علي بلحاج
التفكير الجديد في الفيزياء الحديثة، لأرتور مارش: علي بلحاج ، ترجمة: علي بلحاج، المؤسسة الوطنية بيت الحكمة، قرطاج، (1986م)، ‏144‏.

حَنُوط[اسْم]

ن11هـ=632م

الحَنُوطُ : طِيبٌ يُخْلَطُ لِلْمَيِّتِ خَاصَّةً؛ كَيْلَا يُنْتِنَ.

"إِنَّ المُؤْمِنَ إِذَا كَانَ فِي إِقْبَالٍ مِنَ الآخِرَةِ، وَانْقِطَاعٍ مِنَ الدُّنْيَا، تَنَزَّلَتْ إِلَيْهِ المَلَائِكَةُ كَأَنَّ عَلَى وُجُوهِهِمُ الشَّمْسَ، مَعَ كُلِّ وَاحِدٍ مِنْهُمْ كَفَنٌ وَحَنُوطٌ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏30/‏576،،‏(رقم الحديث: 18614).

حِنَاط[اسْم]

ن73هـ=692م

الحِنَاطُ : الحَنُوطُ.

"أَجْمِرُوا ثِيَابِي إِذَا مِتُّ، ثُمَّ حَنِّطُونِي، وَلَا تَذُرُّوا عَلَى كَفَنِي حِنَاطًا ، وَلَا تَتْبَعُونِي بِنَارٍ"
أسماء بنت أبي بكر الصِّدّيق
الموطأ: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ): صححه ورقمه وخرج أحاديثه وعلق عليه: مُحمَّد فؤاد عبد الباقي، دار إحياء التُّراث العربيّ، بيروت، (1406هـ/ 1985م)، ‏1/‏226،،‏(رقم الحديث: 12).

تَحَنُّط[مَصْدَر]

ن82هـ=701م

التَّحَنُّطُ : اسْتِعْمَالِ الحَنُوطِ.

"وَفِيهِ جَوَازُ اسْتِهْلَاكِ النَّفْسِ فِي الجِهَادِ، وَتَرْكُ الأَخْذِ بِالرُّخْصَةِ، وَالتَّهْيِئَةُ لِلْمَوْتِ بِالتَّحَنُّطِ وَالتَّكْفِينِ"
المُهَلَّب بن أبي صُفْرة الأزديّ
فتح الباري بشرح صحيح البخاري: ابن حجر العسقلاني (ت، 852هـ)، تحقيق: عبد العزيز بن عبد الله بن باز، محمد فؤاد عبد الباقي، محبّ الدّين الخطيب، مكتبة المعرفة، بيروت، (د.ت)، ‏6/‏52‏.
ن1432هـ=2010م

تَحَنُّطُ الفِكْرِ وَنَحْوِهِ: تَجَمُّدُهُ، وَعَدَمُ تَطَوُّرِهِ.

"نَعَمْ، نَحْنُ المُسْلِمِينَ نُعَانِي مِنَ التَّحَنُّطِ الفِكْرِيِّ وَالحَضَارِيِّ! هَذِهِ حَقِيقَةٌ وَاقِعَةٌ... بَعْدَ انْتِهَاءِ العَصْرِ الذَّهَبِيِّ، وَمَوْتِ الفَلْسَفَةِ"
محمد أركون
التشكيل الإنساني للإسلام: محمد أركون، ترجمة: هاشم صالح، دار الساقي، بيروت، 2022م، ‏135‏.

تَحَنَّطَ[لَازِم]

ن93هـ=712م

تَحَنَّطَ الشَّخْصُ: تَطَيَّبَ بِالحَنُوطِ؛ اسْتِعْدَادًا لِلْمَوْتِ.

"مَرَرْتُ يَوْمَ اليَمَامَةِ بِثَابِتِ بْنِ قَيْسِ بْنِ شَمَّاسٍ وَهُوَ يَتَحَنَّطُ "
أنس بن مالك الخزرجيّ الأنصاريّ
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏18/‏151،،‏(رقم الحديث: 17977).

تَحَنَّطَ[مُتَعَدٍّ بِالحَرْف]

1352هـ=1933م

تَحَنَّطَ فِي الشَّيْءِ: تَجَمَّدَ فَيِهِ، لَا يَخْرُجُ عَنْهُ.

"وَبَعْدَ إِخْفَاقِ الأَمِيرِ فِي مَعَارِكِهِ ضِدَّ المُسْتَعْمِرِينَ، وَخُرُوجِهِ مِنَ الوَطَنِ، عَادَ الفِكْرُ الجَزَائِرِيُّ إِلَى جُمُودِهِ وَتَحَجَّرَ، وَتَحَنَّطَتِ الأَخْلَاقِيَّةُ الِإسْلَامِيَّةُ وَالجِهَادُ فِي صُورَةِ زَوَايَا وَطُرُقٍ وَأَصْنَامٍ"
عبد الحميد بن باديس
آثار ابن باديس: عبد الحميد بن باديس (ت، 1358هـ)، إعداد وتصنيف: عمار الطالبي، الشركة الجزائرية، الجزائر، ط3، (1417ه/ 1997م)، ‏17‏.

حَنَّاط[صِيغَةُ مُبَالَغَة]

ن105هـ=723م

الحَنَّاطُ : مَنْ حِرْفَتُهُ بَيْعُ الحِنْطَةِ.

‌إِنَّا ‌وَجَدْنَا ‌عُرُسَ الحَنَّاطِ مَذْمُومَةً لَئِيمَةَ الحِوَاطِ
دُكين بن رَجاء الفُقيميّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏2/‏277‏.

مُحْنِط[صِفَةٌ مُشَبَّهَة]

ن125هـ=743م

المُحْنِطُ مِنَ الشَّجَرِ: الَّذِي نَضِجَ ثَمَرُهُ وَأَمْكَنَ اجْتِنَاؤُهُ.

"كَمَا أَنَّ كَهَنَةَ آمُونَ كَانُوا يُقِيمُونَ فِي السُّودَانِ، بِالإِضَافَةِ إِلَى أَعْدَادٍ كَبِيرَةٍ مِنَ المُهَاجِرِينَ المِصْرِيِّينَ؛ مِنْ حِرَفِيِّينَ وَرَسَّامِينَ وَخَطَّاطِينَ وَصُنَّاعَ وَكَهَنَةٍ وَمُحَنِّطِينَ "
شوقي عبد الحكيم
مدخل لدراسة الفولكلور والأساطير العربية: شوقي عبد الحكيم (ت، 1425هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2017م)، ‏124‏.
1408هـ=1987م

المُحَنِّطُ لِلْأَفْكَارِ وَنَحْوِهَا: المُجَمِّدُ لَهَا.

"وَلَعَلَّهُ مِنَ الأَجْدَى إِدْخَالُهَا تَحْتَ مَفَاهِيمِ: التَّقْلِيدِ وَالتَّحْدِيثِ، أَوِ الغَرْبِ وَالِإسْلَامِ، أَوِ التَّفَاعُلِ الثَّقَافِيِّ الَّذِي يَتَجَاوَزُ القِسْمَةَ الطَّائِفِيَّةَ وَيَتَعَدَّاهَا لِيُدْخِلَ المَوْضُوعَ فِي مَنْهَجٍ تَارِيخِيٍّ - اجْتِمَاعِيٍّ، لَا فِي مَنْهَجٍ أَنْثْرُوبُّولُوجِيٍّ مُؤَبِّدٍ لِلثَّوَابِتِ، وَمُحَنِّطٍ لَهَا فِي حَقْلٍ ضَيِّقٍ كَالحَقْلِ اللُّبْنَانِي"
وجيه كوثراني
الإشكال الثقافي الإيديولوجي في لبنان بين حاجات السلام ودعوات التغيير: ‏وجيه كوثراني،‏المستقبل العربي: المستقبل العربي، مركز دراسات الوحدة العربية، بيروت، مج10، ع108، (فبراير- 1988م)، ‏39‏.

حَنْطَة[اسْم]

1394هـ=1973م

الحَنْطَةُ : تَجَمُّعٌ لِأَصْحَابِ الحِرْفَةِ الوَاحِدَةِ دَاخِلَ سُوقٍ أَوْ خَانٍ، وَنَحْوِ ذَلِكَ.

"وَكَانَ أَعْضَاءُ المِهْنَةِ يَنْتَسِبُونَ فِي الغَالِبِ لِطَرِيقَةٍ مِنَ الطُّرُقِ الَّتِي تَسُدُّ حَاجَتَهُمْ لِلْاجْتِمَاعَاتِ الدِّينِيَّةِ فِي شَكْلِهَا الخَاصّ ... وَلَكِنَّ هَذِهِ الحَنْطَاتِ لَمْ تَكُنْ تَتَعَدَّى دَوَائِرَ الفَلَّاحِينَ وَالصُّنَّاعِ "
علال الفاسي
النقد الذاتي: علال الفاسي (ت، 1394هـ)، المطبعة العالمية، القاهرة، ط1، (د.ت)، ‏405‏.

حِنْطَاوِيّ[صِفَة]

1435هـ=2013م

الحِنْطَاوِيُّ مِنَ الأَلْوَانِ: الدَّاكِنُ بَيْنَ السُّمْرَةِ وَالحُمْرَةِ.

"عِمْرَانُ مُوسَى، أَحَدُ المُعَمِّرِينَ الَّذِينَ عَاصَرُوا الزَّهَاوِي وَجَالَسُوهُ فِي هَذِهِ المَقْهَى، فَقَالَ وَهُوَ يَصِفُهُ: إِنَّهُ كَانَ يُشَاهِدُ رَجُلًا حِنْطَاوِيًّا نَحِيفًا، مُتْعَبَ العَيْنَيْنِ"
شفيق مهدي
مقاهي بغداد: مهدي شفيق، دار الشؤون الثقافية العامة، بغداد، 2013م، ‏203‏.
