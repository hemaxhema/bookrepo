ازْدَقَمَ[مُتَعَدٍّ]

1ق.هـ=621م

ازْدَقَمَ الشَّيْءَ: ابْتَلَعَهُ.

"هَاتِي، يَا جَارِيَةُ، تَمْرًا وَزُبْدًا نَزْدَقِمُهُ "
أبو جهل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏94‏.

تَزَقَّمَ[مُتَعَدٍّ]

1ق.هـ=621م

تَزَقَّمَ الشَّيْءَ: بَلَعَهُ فِي جُهْدٍ وَتَكَلُّفٍ.

"يُخَوِّفُنَا مُحَمَّدٌ بِشَجَرَةِ الزَّقُّومِ؟ هَاتُوا تَمْرًا وَزُبْدًا فَتَزَقَّمُوا "
أبو جهل
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏5/‏477،،‏(رقم الحديث: 3546).
280هـ=893م

تَزَقَّمَ اللَّبَنَ: أَفْرَطَ فِي شُرْبِهِ.

"بَاتَ يَتَزَقَّمُ اللَّبَنَ"
ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏2/‏823‏.

زَقُّوم[اسْمُ جِنْس]

1ق.هـ=621م

الزَّقُّومُ : شَجَرَةٌ فِي جَهَنَّمَ.

قرآن كريم
قرآن كريم، ‏[ الصَّافَّات : 62 ].
ن11هـ=632م

الزَّقُّومُ: ثَمَرُ شَجَرٍ كَرِيهُ الرَّائِحَةِ، مُرُّ الطَّعْمِ.

"وَلَوْ أَنَّ قَطْرَةً مِنَ الزَّقُّومِ قُطِرَتْ، لَأَمَرَّتْ عَلَى أَهْلِ الأَرْضِ عَيْشَهُمْ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏4/‏467‏.

زَقُّوم[اسْم]

1ق.هـ=621م

الزَّقُّومُ : طَعَامٌ مِنْ تَمْرٍ وَزُبْدٍ.

"إِنَّمَا الزَّقُّومُ التَّمْرُ وَالزُّبْدُ نَتَزَقَّمُهُ"
أبو جهل
تفسير الإمام مجاهد بن جبر: مجاهد بن جبر المكّيّ (ت، 104هـ)، تحقيق: محمّد عبد السلام أبو النيل، دار الفكر الإسلامي الحديثة، القاهرة، ط1، (1410هـ/ 1989م)، ‏568‏.

زَقَّمَ[مُتَعَدٍّ]

1ق.هـ=621م

زَقَّمَهُ الشَّيْءَ: بَلَّعَهُ إِيَّاهُ.

"أَيُخَوِّفُنِي ابْنُ أَبِي كَبْشَةَ بِشَجَرَةِ الزَّقُّومِ؟ ثُمَّ دَعَا بِتَمْرٍ وَزُبْدٍ، فَجَعَلَ يَقُولُ: زَقِّمْنِي "
أبو جهل
تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏14/‏648‏.

تَزَقُّم[مَصْدَر]

1ق.هـ=621م

تَزَقُّمُ الشَّيْءِ: بَلْعُهُ فِي جُهْدٍ وَتَكَلُّفٍ.

"وَاللهِ لَئِنِ اسْتَمْكَنَّا مِنْهَا، لَنَتَزَقَّمَنَّهَا تَزَقُّمًا "
أبو جهل
السّيرة النّبويّة: ابن هشام المَعافريّ الحِمْيريّ (ت، 213هـ)، حققها وضبطها وشرحها ووضع فهارسها: مصطفى السّقا وآخرون، شركة مكتبة ومطبعة البابي الحلبي وأولاده، القاهرة، ط2، (1375هـ/ 1955م)، ‏1/‏362‏.

زَقْم[مَصْدَر]

ن175هـ=791م

الزَّقْمُ : أَكْلُ الزَّقُّومِ.

" الزَّقْمُ : أَكْلُ الزَّقُّومِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏94‏.
ن244هـ=858م

زَقْمُ اللُّقْمَةِ وَنَحْوِهَا: بَلْعُهَا بِشِدَّةٍ.

"إِنَّهُ لَيَزْقُمُ اللُّقَمَ زَقْمًا جَيِّدًا"
من رواية ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏482‏.
ن321هـ=933م

زَقْمُ اللَّبَنِ: شُرْبُهُ بِإِفْرَاطٍ.

"بَاتَ يَتَزَقَّمُ اللَّبَنَ: يَشْرَبُهُ وَيُفْرِطُ فِيهِ، وَهُوَ الزَّقْمُ "
ابن دُريد
كتاب المخصَّص: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، المطبعة الكبرى الأميرية، بولاق، ط1، (1316- 1321هـ)، ‏11/‏97‏.

زَقَمَ[مُتَعَدٍّ]

ن244هـ=858م

زَقَمَ الشَّيْءَ: بَلَعَهُ.

"وَيُقَالُ: إِنَّهُ لَيَزْقُمُ اللَّقْمَ زَقْمًا جَيِّدًا"
ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏482‏.

أَزْقَمَ[مُتَعَدٍّ]

423هـ=1032م

أَزْقَمَهُ الشَّيْءَ: أَبْلَعَهُ إِيَّاهُ.

"وَلَا يُزْقِمُ أُسْتَاذَهُ مِنْ طَرِيِّهِ لُقَمًا"
أبو الرّيحان البيرونيّ
كتاب البيروني في تحقيق ما للهند من مقولة مقبولة في العقل أو مرذولة: أبو الرّيحان البيرونيّ (ت، 440هـ)، مطبعة مجلس دائرة المعارف العثمانية، حيدر آباد الدكن، (1377هـ/ 1958م)، ‏150‏.

زَقَام[اسْم]

ن1231هـ=1815م

الزَّقَامُ : الطَّعَامُ الَّذِي يَقْتُلُ.

وَاسْتَعْذَبَتْ بِطِبَاعِهَا مِنْ فِعْلِهَا المُرْدِي، زَقَامَهْ
هاشم بن حردان الكعبي
ديوان الكعبي (قسم المراثي الحسينية): هاشم بن حردان الكعبي (ت، 1231هـ)، قدم له وعلق عليه وصححه: محمد حسن آل الطالقاني، انتشارات الشريف الرضي- شريعت، ط1، (1420ه/ 1999م)، ‏147‏.
