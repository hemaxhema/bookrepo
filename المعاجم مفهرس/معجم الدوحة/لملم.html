مُلَمْلَم[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُلَمْلَمُ مِنَ الأَشْيَاءِ: المُدَوَّرُ المُجْمُوعُ بَعْضُهُ إِلَى بَعْضٍ.

وَإِنْ أَدْبَرَتْ قُلْتَ: أُثْفِيَّةٌ مُلَمْلَمَةٌ ، لَيْسَ فِيهَا أُثُرْ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏166‏.
قبل 13ق.هـ=609م

المُلَمْلَمُ مِنَ الكَتَائِبِ وَنَحْوِهَا: المُجْتَمِعُ المُتَضَامُّ.

كَتِيبَتُهُ مُلَمْلَمَةٌ رَدَاحٌ إِذَا يَرْمُونَهَا تُفْنِي النِّبَالَا
عمرو بن كلثوم التَّغْلِبيّ
ديوان عَمرو بن كلثوم: جمعه وحققه وشرحه: إميل بديع يعقوب، دار الكتاب العربي، بيروت، ط2، (1416هـ/ 1996م)، ‏51‏.
ن90هـ=709م

المُلَمْلَمُ مِنَ الشَّعَرِ: المَدْهُونُ.

وَمَا التَّصَابِي لِلعُيُونِ الحُلَّمِ بَعْدَ ابْيِضَاضِ الشَّعَرِ المُلَمْلَمِ إِلَّا تَضَالِيلُ الفُؤَادِ الأَيْهَمِ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏1/‏446‏.

مُلَمْلَم[صِفَة]

ن82هـ=701م

المُلَمْلَمُ مِنَ النُّوقِ وَنَحْوِهَا: الغَلِيظُ المُعْتَدِلُ الخَلْقِ.

"فَأَتَاهُ رَجُلٌ بِنَاقَةٍ عَظِيمَةٍ مُلَمْلَمَةٍ ، فَأَبَى أَنْ يَأْخُذَهَا"
سُويد بن غَفَلَة الجُعْفيّ
سنن الحافظ أبي عبد الله محمّد بن يَزيد القَزويني ابن ماجه: ابن ماجَهْ (ت، 273هـ)، حقق نصوصه ورقم كتبه وأبوابه وأحاديثه وعلق عليه: محمّد فؤاد عبد الباقي،دار إحياء الكتب العربية، د.ت. ‏1/‏576‏.

تَلَمْلَمَ[لَازِم]

قبل 50ق.هـ=573م

تَلَمْلَمَ الشَّيْءُ: اجْتَمَعَ بَعْضُهُ إِلَى بَعْضٍ.

وَإِذَا دَعَوْا بِأَبِي رَبِيعَةَ، أَقْبَلُوا بِكَتَائِبٍ، دُونَ النِّسَاءِ، تَلَمْلَمُ
حَمَصيصة الشَّيْبانيّ
كتاب الأنوار ومحاسن الأشعار: الشِّمْشاطيّ (ت، 377هـ)، تحقيق: السيد محمّد يوسف، راجعه وزاده في حواشيه: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، (1397هـ/ 1977م)، ‏1/‏102‏.
ن530هـ=1135م

تَلَمْلَمَتْ شَفَتَاهُ: تَحَرَّكَتَا بِالذِّكْرِ وَنَحْوِهِ.

"لَقَدْ كَانَ لَهُ فِي نَفْسِهِ شُغْلٌ، وَكَأَنَّ شَفَتَيْهِ لَا تَزَالَانِ تَتَلَمْلَمَانِ بِذِكْرِ اللهِ، وَمَا يَتَعَلَّقُ بِطَاعَتِهِ"
محمّد بن منصور
من كتاب أخبار الزيدية [المُطَرِّفِيَّة]: اللَّحْجيّ (ت، 530هـ)، اعتنى به وضبطه: مقبل التام عامر الأحمدي، الجيل الجديد ناشرون، صنعاء، (2020م)، ‏166‏.
ن854هـ=1450م

تَلَمْلَمَتِ الأَمُورُ: صَلَحَتْ وَاسْتَقَامَتْ.

"وَلَمَّا وَصَلَ بِير مُحَمَّد إِلَى قَنْدَهَار، وَاسْتَقَرَّتْ بِهِ الدَّار، وَتَلَمْلَمَتْ أُمُورُه، وَحَامَتْ حَوْلَ قُصُورِهِ صُقُورُه، وَدَارَتْ مِن سَيَّارَاتِ عَسْكَرِهِ بِدُورِهِ بُدُورُه، تَسَعَّرَتْ سَمُومُهُ وَحَرُورُه، وَتَطَايَرَ شَرَارُهُ وَشُرُورُه"
ابن عرب شاه الدمشقي
عجائب المقدور في نوائب تيمور: ابن عرب شاه الدمشقي (ت، 854هـ)، تحقيق: أحمد فايز الحمصي، مؤسسة الرسالة، بيروت، ط1، (1407ه/ 1986م)، ‏430‏.
1392هـ=1971م

تَلَمْلَمَ الجُرْحُ وَنَحْوُهُ: الْتَأَمَ وَتَدَاوَى.

"أَتَكَمَّشُ بِالفَرَحِ الآتِي.. وَأُضِيءْ يَتَلَمْلَمُ جُرْحِي بِالفُقَراءِ.. وَيَأْمُرُنِي.. فَأُضِيءْ ‏وَأُحَارِبُ، حَتَّى تُصْبِحَ يَا وَطَنِي.. وَطَنِي‏"
أحمد دحبور
مصيدتان: ‏أحمد دحبور،‏مجلة الموقف الأدبي: دمشق، مج1، ع7-12، (1971م)، ‏64‏.

لَمْلَمَ[مُتَعَدٍّ]

ن73هـ=692م

لَمْلَمَ الشَّيْءَ: جَمَعَهُ.

"فَأَقْبَلَتْ، وَهِيَ تُلَمْلِمُ الفِهْرَ فِي يَدِهَا"
أسماء بنت أبي بكر الصِّدّيق
أخبار مكّة وما جاء فيها من الآثار: محمّد بن عبد الله الأزرقيّ (ت، 250هـ)، دراسة وتحقيق: عبد الملك بن عبد الله بن دهيش، مكتبة الأسدي، ط1، (1424هـ/ 2003م)، ‏436‏.
423هـ=1032م

لَمْلَمَ الشَّيْءَ: جَعَلَهُ مُدَوَّرًا.

"وَذَكَرَ بَرَاهْمِهِرُ فِي صَنْعَتِهِ: بَعْدَ اخْتِيَارِ الحَجَرِ لَهُ سَلِيمًا مِنَ المَعَايِبِ، أَنْ يُؤْخَذَ الطُّولُ الَّذِي يُرَادُ أَنْ يُعْمَلَ لَهُ، وَيُقْسَمَ أَثْلَاثًا... وَيُدَوَّرُ الثُّلُثُ الأَعْلَى، وَيُلَمْلَمُ رَأْسُهُ، حَتَّى يَصِيرَ شَبِيهًا بِالكَمَرَةِ"
أبو الرّيحان البيرونيّ
كتاب البيروني في تحقيق ما للهند من مقولة مقبولة في العقل أو مرذولة: أبو الرّيحان البيرونيّ (ت، 440هـ)، مطبعة مجلس دائرة المعارف العثمانية، حيدر آباد الدكن، (1377هـ/ 1958م)، ‏430‏.
1374هـ=1954م

لَمْلَمَ الجِرَاحَ وَنَحْوَهَا: دَاوَاهَا.

تُلَمْلِمُ ، يَا أُخْتُ، الدُّرُوبُ جِرَاحَنَا فَفِي كُلِّ شِبْرٍ لَاجِئٌ وَمُصَابُ
سليمان العيسى
أعاصير في السلاسل، ديوان شعر: سليمان العيسى، دار العلم للملايين، بيروت، (1963م)، ‏170‏.

لَمْلَمَ[مُتَعَدٍّ بِالحَرْف]

ن145هـ=762م

لَمْلَمَ بِالقَوْمِ: اجْتَمَعَ بِهِمْ.

وَإِنْ تُلَمْلِمْ خِنْدِفِي بِالأَنْضَادْ وَقَيْسُنَا تَرْحَمْ بِعِزٍّ مَيَّادْ تَزِلُّ عَنْهُ نَاطِحَاتُ الأَضْدَادْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏62‏.

لَمْلَم[صِفَة]

ن75هـ=694م

اللَّمْلَمُ : المُجْتَمِعُ بَعْضُهُ إِلَى بَعْضٍ.

وَلَقَدْ يَحِلُّ بِهَا، وَيَسْكُنُهَا حَيٌّ حِلَالٌ، لَمْلَمٌ ، عَكْرُ
عمرو بن أحمر الباهليّ
غريب الحديث: إبراهيم بن إسحاق الحربيّ (ت، 285هـ)، تحقيق ودراسة: سليمان بن إبراهيم بن محمّد العايد، جامعة أم القرى، مكة المكرمة، دار المدني، جدة، ط1، (1405هـ/ 1985م)، ‏1/‏327‏.

لَمْلَمَة[مَصْدَر]

ن175هـ=791م

اللَّمْلَمَةُ : إِدَارَةُ الحَجَرِ وَنَحْوِهِ.

" وَاللَّمْلَمَةُ : إِدَارَةُ الحَجَرِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏323‏.
ن385هـ=995م

لَمْلَمَةُ الشَّيْءِ: جَمْعُهُ.

"سَمْلَكْتُ اللُّقْمَةَ، وَهُوَ أَنْ تُطَوِّلَهَا فِي لَمْلَمَةٍ وَتَدْوِيرٍ"
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏6/‏366‏.
ن560هـ=1165م

اللَّمْلَمَةُ : الِاسْتِدَارَةُ.

"خُشْكَنَانَجٌ كَأَسَاوِرِ الكَوَاعِب، كَسَّرَهَا فَضْلُ التَّجَاذُبِ وَالتَّدَاعُب، أَوْ كَقُرُونِ الظِّبَاءِ قَدًّا وَالْتِفَافَا، وَلَمْلَمَةً وَاسْتِحْصَافَا"
المظفّر بن أردشير
التّذكرة الحمدونيّة: ابن حمدون الكاتب (ت، 562هـ)، تحقيق: إحسان عبّاس، بكر عبّاس، دار صادر، بيروت، ط1، (1996م)، ‏9/‏119‏.
1395هـ=1974م

لَمْلَمَةُ الجِرَاحِ وَنَحْوِهَا: مُدَاوَاتُهَا.

"يَبْحَثُ عَنْ هُوُيَّةٍ فِكْرِيَّةٍ وَيُفَتِّشُ، غَيْرَ أَنَّهُ -عَلَى مَا يَبْدُو- لَمْ يَجِدَّ بَعْدَ الفِكْرِ الَّذِي يَتَقَمَّصُهُ وَيُصَفِّحُهُ ضِدَّ القَدِيمِ، بِالدَّرَجَةِ الَّتِي تَكْفِي، لِلَمْلَمَةِ جِرَاحِ رَفِيقَتِهِ وَحَبِيبَتِهِ تَمِيمَةَ"
توفيق يوسف عوَّاد
مقابلة أدبية مع توفيق يوسف عواد: ‏عفيف فرّاج،‏مجلة الآداب: بيروت، س22، ع6، (01/ 06/ 1974م)، ‏22‏.

لَمْلَمِيّ[صِفَة]

246هـ=860م

اللَّمْلَمِيُّ : المُجْتَمِعُ المُتَضَامُّ.

سَيْفِيِّ وَقْعِ القِدْحِ، نَشَّابِيِّهْ كِلَاهُمَا أَذْعَنَ مِنْ سَبِيِّهْ بِلَمْلَمِيِّ الحُكْمِ، كَبْكَبِيِّهْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏6/‏2623‏.

مُلَمْلِم[اسْمُ فَاعِل]

255هـ=869م

المُلَمْلِمُ مِنَ الجُيُوشِ وَنَحْوِهَا: المُجْتَمِعُ المُتَرَابِطُ.

وَاغْتَرَّ أَهْلُ البَذِّ فِي شُرَفَاتِهِمْ حَتَّى أَصَابَهُمُ بِسَيْفِ الهَيْثَمِ
فِي وَقْعَةٍ، وَلِيَتْ غَنِيٌّ حَدَّهَا بِأَجَشَّ مِنْ زَجَلِ الحَديدِ مُلَمْلِمِ
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏4/‏2082‏.
ن1349هـ=1930م

المُلَمْلِمُ الأَشْيَاءَ: الجَامِعُهَا.

يُغْوِيكَ فِيهِ البَحْرُ، وَهْوَ مُلَمْلِمٌ أَمْوَاجَهُ، حَتَّى تَخَافَ زِحَامَهَا
فوزي المعلوف
ديوان فوزي المعلوف: فوزي المعلوف (ت، 1349هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2013م)، ‏36‏.
1405هـ=1984م

المُلَمْلِمُ الجِرَاحَ وَنَحْوَهَا: المُدَاوِيهَا.

وَيَؤُوبُ عَنْ أَرْضِ الفِدَاءِ مُلَمْلِمًا تِلْكَ الجِرَاحَ، وَهَارِبًا مَرْهُوبَا
محمّد فريد عبد الله
أرض البطولة: محمد فريد عبد الله، دار الريحاني، بيروت، (1984م)، ‏180‏.

تَلَمْلُم[مَصْدَر]

"وَتُشَبَّهُ المَرْأَةُ بِالبَيْضِ؛ لِتَلَمْلُمِهَا ، وَزَوَالِ الحُجُومِ عَنْهَا"

مُتَلَمْلِم[اسْمُ فَاعِل]

ن684هـ=1285م

المُتَلَمْلِمُ مِنَ الأَشْيَاءِ: المُجْتَمِعُ بَعْضُهُ إِلَى بَعْضٍ.

هَلْ شِمْتَ بَرْقًا بِالشَّآمِ القَارِبِ مُتَلَمْلِمًا مِثْلَ اخْتِلَاجِ الحَاجِبِ؟
