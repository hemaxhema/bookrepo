شَاذّ[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الشَّاذُّ مِنَ الجَرَادِ وَنَحْوِهِ: المُتَفَرِّقُ المُتَنَاثِرُ.

يُطَايِرْنَ الأَكُفَّ عَنِ التَّرَاقِي كَشُذَّانِ الجَرَادِ لَدَى المَطِيرِ
حراب بن الورد الحربيّ النِّهْميّ الهَمْدانيّ
شعر هَمْدان وأخبارها في الجاهلية والإسلام: جمع وتحقيق ودراسة: حسن عيسى أبو ياسين، دار العلوم، الرِّياض، ط1، (1403هـ/ 1983م)، ‏250‏.
قبل 13ق.هـ=609م

الشَّاذُّ : المُنْفَرِدُ عَنِ الجَمَاعَةِ، الخَارِجُ عَنْهَا.

"وَإِنِّي لَأَكْرَهُ أَنْ أُجَبِّهَ وُفُودِي، أَوْ أُحْنِقَ صُدُورَهُمْ. وَالَّذِي أُحِبُّ هُوَ إِصْلَاحُ مَدَارِكُمْ، وَتَأَلُّفُ شَوَاذِّكُمْ "
ترجمان كسرى
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏2/‏19‏.
ن154هـ=771م

الشَّاذُّ مِنَ الكَلِمِ: الَّذِي خَالَفَ نَظَائِرَهُ فِي بَابِهِ، وَخَرَجَ عَنْ أَصْلِهِ. [النّحو والصّرف]

"هُوَ مَفْعُولٌ مِنْ (أَبْرَزْتُ) شَاذٌّ "
أبو عمرو بن العَلاء
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏3/‏194‏.
ن160هـ=777م

الشَّاذُّ مِنَ الحَدِيثِ: الَّذِي يَرْوِيهِ الثِّقَاتُ، فَيَنْفَرِدُ عَنْهُمْ وَاحِدٌ فَيُخَالِفُهُمْ. [علوم الحديث]

"لَا يَجِيئُكَ الحَدِيثُ الشَّاذُّ إِلَّا مِنَ الرَّجُلِ الشَّاذِّ"
شعبة بن الحجّاج
الكفاية في معرفة أصول علم الرّواية: الخطيب البغداديّ (ت، 463هـ)، تحقيق وتعليق: أبي إسحاق إبراهيم بن مصطفى آل بحبح الدمياطيّ، دار الهدى، الدقهلية، ط1، (1423هـ/ 2003م)، ‏1/‏420‏.
ن318هـ=930م

الشَّاذُّ مِنَ الأَقْوَالِ: المُخَالِفُ لِمَا قَالَ بِهِ سَائِرُ العُلَمَاءِ. [أصول الفقه]

"وَلَا نَعْلَمُ أَحَدًا يُخَالِفُ ذَلِكَ إلَّا مَكْحُولًا؛ فَإِنَّهُ قَالَ: إِذَا كَانَتِ المَأْمُومَةُ عَمْدًا فَفِيهَا ثُلُثَا الدِّيَةِ، وَإِذَا كَانَتْ خَطَأً فَفِيهَا ثُلُثُ الدِّيَةِ... وَهَذَا قَوْلٌ شَاذٌّ ، وَبِالقَوْلِ الأَوَّلِ أَقُولُ".
ابن المنذر النَّيْسابوريّ
الإشراف على مذاهب العلماء: ابن المنذر النَّيْسابوريّ (ت، 318هـ)، تحقيق: أبي حمّاد صغير أحمد الأنصاريّ، مكتبة مكة الثقافية، رأس الخيمة، ط1، (1425هـ/ 2004م- 1426هـ/ 2005م)، ‏7/‏406‏.

شَاذّ[صِفَةٌ مُشَبَّهَة]

204هـ=819م

الشَّاذُّ مِنَ القِرَاءاتِ: الَّذِي رُوِيَ عَلَى خِلَافِ قِرَاءَةِ الحُجَّةِ مِنَ القُرَّاءِ، أَوْ خَالَفَ رَسْمَ المُصْحَفِ الإِمَامِ. [علوم القرآن والقراءات]

"وَيُقْرَأُ (سُرِّقَ) وَلَا أَشْتَهِيهَا؛ لِأَنَّهَا شَاذَّةٌ "
أبو زكريّا الفَرّاء
معاني القرآن: أبو زكريّا الفَرّاء (ت، 207هـ)، تحقيق ومراجعة: أحمد يوسف النجاتي وآخرين، الدار المصرية، القاهرة، (1955م)، ‏2/‏53‏.

أَشَذَّ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

أَشَذَّ الشَّيْءَ: فَرَّقَهُ.

كَأَنَّ صَلِيلَ المَرْوِ، حِينَ تُشِذُّهُ ، صَلِيلُ زُيُوفٍ، يُنْتَقَدْنَ بِعَبْقَرَا
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏96‏.
263هـ=877م

أَشَذَّ اللَّفْظَ: عَدَّهُ مُخَالِفًا لِنَظَائِرِهِ فِي بَابِهِ، وَخَارِجًا عَنْ أَصْلِهِ.

"وَفِي (قُرَيْشٍ): قُرَشِيٌّ، وَفِي (هُذَيْلٍ): هُذَلِيٌّ، وَفِي (سُلَيْمٍ): سُلَمِيٌّ، هَذَا هُو القِيَاسُ، إِلَّا مَا أَشَذُّوا "
ابن قتيبة
أدب الكاتب: ابن قتيبة (ت، 276هـ)، حققه وعلق حواشيه ووضع فهارسه: محمّد أحمد الدالي، مؤسسة الرسالة، بيروت، ط1، (1402هـ/ 1981م)، ‏280‏.

شَذَّان[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

شَذَّانُ الحَصَا وَنَحْوِهِ: المُتَفَرِّقُ المُتَنَاثِرُ.

تُطَايِرُ شَذَّانَ الحَصَا، بِمَنَاسِمٍ صِلَابِ العُجَى، مَلْثُومُهَا غَيْرُ أَمْعَرَا
امرؤ القيس بن حُجْر الكنديّ
كتاب المعاني الكبير في أبيات المعاني: ابن قتيبة (ت، 276هـ)، صححه: سالم الكرنكوى، دار النّهضة الحديثة، بيروت، د.ت. ‏1/‏165‏.

شَذَّ[لَازِم]

ن11هـ=632م

شَذَّ الشَّخْصُ وَنَحْوُهُ: انْفَرَدَ عَنِ الجَمَاعَةِ، وَخَرَجَ عَنْهَا.

"إِنَّ اللهَ لَا يَجْمَعُ أُمَّتِي...عَلَى ضَلَالَةٍ، وَيَدُ اللهِ مَعَ الجَمَاعَةِ، وَمَنْ شَذَّ شَذَّ إِلَى النَّارِ"
حديث نبويّ
الجامع الصحيح وهو سُنن الترمذي: التّرمذيّ (ت، 279هـ)، تحقيق وشرح: أحمد محمد شاكر، وآخرين، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1382هـ/ 1962م- 1398هـ/ 1978م)، ‏4/‏466،،‏(رقم الحديث: 2167).
ن180هـ=796م

شَذَّ اللَّفْظُ: خَالَفَ نَظَائِرَهُ فِي بَابِهِ، وَخَرَجَ عَنْ أَصْلِهِ. [النّحو والصّرف]

"وَتَكُونُ النُّونُ فِي نَفْسِ الحَرْفِ بِمَنْزِلَةِ نُونِ (مِنْ) وَ (عَنْ)؛ فَقَدْ يَشُذُّ الشَّيْءُ مِنْ كَلَامِهِمْ عَنْ نَظَائِرِهِ، وَيَسْتَخِفُّونَ الشَّيْءَ فِي مَوْضِعٍ، وَلَا يَسْتَخِفُّونَهُ فِي غَيْرِهِ"
سيبويه
كتاب سيبويه: سيبويه (ت، 180هـ)، تحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط3، (1408هـ/ 1988م- 1416هـ/ 1996م)، ‏1/‏210‏.

شُذُوذ[مَصْدَر]

16هـ=637م

شُذُوذُ الشَّخْصِ وَنَحْوِهِ: انْفِرَادُهُ عَنِ الجَمَاعَةِ، وَخُرُوجُهُ عَنْهَا.

"فَمَنْ أَرَادَ بُحْبُوحَةَ الجَنَّةِ فَلْيَلْزَمِ الجَمَاعَةَ، وَلَا يُبَالِي اللهُ شُذُوذَ مَنْ شَذَّ"
عمر بن الخطّاب
جمهرة خطب العرب في عصور العربية الزاهرة: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1352هـ/ 1933م- 1352هـ/ 1934م)، ‏1/‏255‏.
ن180هـ=796م

شُذُوذُ اللَّفْطِ: مُخَالَفَتُهُ نَظَائِرَهُ فِي بَابِهِ وَخَرُوجُهُ عَنْ أَصْلِهِ.

"إِنَّ هَذِهِ اللَّامَ دَخَلَتْ عَلَى جِهَةِ الشُّذُوذِ "
سيبويه
الأصول في النّحو: ابن السَّرّاج النَّحْويّ البغداديّ (ت، 316هـ)، تحقيق: عبد الحسين الفتلي، مؤسسة الرسالة، بيروت، ط3، (1417هـ/ 1996م)، ‏1/‏242‏.
204هـ=819م

شُذُوذُ القِرَاءَةِ: رِوَايَتُهَا عَلَى خِلَافِ قِرَاءَةِ الحُجَّةِ مِنَ القُرَّاءِ، أَوْ مُخَالَفَتُهَا رَسْمَ المُصْحَفِ الإِمَامِ.

أبو زكريّا الفَرّاء
معاني القرآن: أبو زكريّا الفَرّاء (ت، 207هـ)، تحقيق ومراجعة: أحمد يوسف النجاتي وآخرين، الدار المصرية، القاهرة، (1955م)، ‏1/‏416‏.
270هـ=883م

الشُّذُوذُ فِي الفِقْهِ: مُخَالَفَةُ المُجْمَعِ عَلَيْهِ، بِلَا حُجَّةٍ مُعْتَبَرَةٍ. [أصول الفقه]

"وَأَمَّا مَنْ قَالَ: عَنَى بِذَلِكَ مَا ذَبَحَهُ المُسْلِمُ فَنَسِيَ ذِكْرَ اسْمِ اللهِ، فَقَوْلٌ بَعِيدٌ مِنَ الصَّوَابِ؛ لِشُذُوذِهِ وَخُرُوجِهِ عَمَّا عَلَيْهِ الحُجَّةُ مُجْمِعَةٌ مِنْ تَحْلِيلِهِ"
الطّبريّ
تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏9/‏529‏.
ن320هـ=932م

شُذُوذُ الثِّيرَانِ وَنَحْوِهَا: تَفَرُّقُهَا.

لَمَّا تَوَلَّى الأَمْرَ شَذَّ عِصَابَةٌ عَنْهُ شُذُوذَ نَوَافِرِ الثِّيرَانِ
أحمد بن عَلَويّة الكاتب الأصفهانيّ
شعر أحمد بن عَلَوِيّة الكاتب: جمع وتحقيق ودراسة: عبد المجيد الإسداوي، مجلة الخزانة، كربلاء، س3، ع3، (1439هـ/ 2018م)، ‏80‏.
ن436هـ=1044م

شُذُوذُ الحَدِيثِ: تَفَرُّدُ الثِّقَةِ بِرِوَايَةٍ فِيهِ مُخَالِفَةٍ لِمَا يَرْوِيهِ نُظَرَاؤُهُ. [علوم الحديث]

"وَمِنَ البَدِيعِ أَنْ يَقُولَ فِي مِثْلِ مَا رُوِيَ مِنْ قَوْلِهِ صَلَّى اللهُ عَلَيْهِ وَآلِهِ: (عَلِيٌّ وَلِيُّ كُلِّ مُؤْمِنٍ بَعْدِي)... إِنَّهُ لَا يُعْرَفُ وَيَرْمِيهِ بِالشُّذُوذِ "
الشّريف المرتضى
الشّافي في الإمامة: الشّريف المرتضى (ت، 436هـ)، حقّقه وعلّق عليه: عبد الزهراء الحسيني الخطيب، راجعه: السيد فاضل الميلاني، مؤسسة الصادق، طهران، ط2، (1407هـ/ 1986م)، ‏3-4/‏111‏.
1365هـ=1945م

الشُّذُوذُ العَقْلِيُّ : نَقْصٌ فِي بَعْضِ المَرَاكِزِ العَصَبِيَّةِ؛ يُؤَدِّي إِلَى اضْطِرَابٍ فِي النُّمُوِّ العَقْلِيِّ والخُلُقِيِّ، وَيَجْعَلُ الشَّخْصَ الَّذِي يُصَابُ بِهِ لَا يَسْتَطيعُ أَنْ يَتَكَيَّفَ مَعَ الوَسَطِ الَّذِي يَعِيشُ فِيهِ. [علم النّفس]

"أَثْبَتَتِ الإِحْصَائِيَّاتُ أَنَّ نِسْبَةَ الشُّذُوذِ العَقْلِيِّ وَالخُلُقِيِّ تَرْتَفِعُ ارْتِفَاعًا كَبِيرًا فِي المَرَاكِزِ الصِّنَاعِيَّةِ؛ حَيْثُ يَكْثُرُ الفَقْرُ، وَتَقِلُّ العِنَايَةُ بِرَفَاهِيَّةِ العُمَّالِ مِنَ النَّاحِيَةِ المَادِيَّةِ وَالعَقْلِيَّةِ."
السيِّد محمد بدوي
الشذوذ العقلي عند الأطفال في السن المدرسية.: ‏السيِّد محمد بدوي،‏مجلة علم النفس: القاهرة، مجلد 2 عدد 1، (1365ه/ 1946م)، ‏121‏.
1417هـ=1996م

الشُّذُوذُ الكُرُومُوزُومِيُّ : اضْطِرَابٌ يَتَمَيَّزُ بِتَغْيِيرٍ عَدَدِيٍّ فِي الكُرُومُوزُومَاتِ الفَرْدِيَّةِ، أَوِ المُتَعَدِّدَةِ. [الأحياء]

"أُثْبِتَ أَنَّ مُتَلَازِمَةَ دَاوِنْ تَنْشَأُ عَنْ شُذُوذٍ كُرُومُوزُومِيٍّ "
أحمد مستجير
الشفرة الوراثية للإنسان، القضايا العلمية والاجتماعية لمشروع الجينوم البشري: دانييل كيفلس، ليروي هود، ترجمة: أحمد مستجير، المجلس الوطني للثقافة والفنون والآداب، الكويت، (1418هـ/1997م)، ‏26‏.

شُذُوذ[اسْم]

1387هـ=1966م

الشُّذُوذُ الْجِنْسِيُّ : اللِّوَاطُ أَوْ السِّحَاقُ أَوْ نَحْوِهُمَا. [القانون]

"لَقَدْ قَالَ قَوْمٌ مِنْ عُلَمَاءِ البَيُولُوجْيَا: إِنَّ بَعْضَ الأَفْرَادِ يُصَابُونَ بِالشُّذُوذِ الجِنْسِيِّ ، بِحَسَبِ تَكْوِينِهِمْ، فَيَدْعُوهُمْ ذَلِكَ إِلَى الإِفْرَاطِ فِي هَذَا البَابِ"

تَشْذِيذ[مَصْدَر]

436هـ=1044م

تَشْذِيذُ اللَّفْظِ: الحُكْمُ بِمُخَالَفَتِهِ نَظَائِرَهُ فِي بَابِهِ وَبِخُرُوجِهِ عَنْ أَصْلِهِ.

"وَمَا ذَهَبُوا إِلَيْهِ مِنْ تَشْذِيذِ (دِمْتَ تَدُومُ) أَخَفُّ مِمَّا ذَهَبُوا إِلَيْهِ مِنْ تَسَوُّغِ (دُمْتَ تَدَامُ)؛ إِذِ الأُولَى ذَاتُ نَظَائِرَ"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏10/‏143‏.

"لَوْ كَانَ هَذَا الحُكْمُ المَشْذُوذُ عَنْهُ أَصْلًا لِلشَّاذِّ لَمَا شَذَّ عَنْهُ مَا شَذَّ"

شَذَّذَ[مُتَعَدٍّ]

ن793هـ=1390م

شَّذَّذَهُ : عَدَّهُ شَاذًّا، خَارِجَ القَاعِدَةِ.

وَجَزْءَهُ، وَالشَّطْرَ فِيهِ شَذِّذِ "إِنَّ الغَزَالَ" مَعَهُ "إِنَّ الَّذِي"
أَبو سعيد شعبان بن محمد القرشيّ الآثاريّ
الوجه الجميل في علم الخليل: أَبو سعيد شعبان بن محمد القرشيّ الآثاريّ (ت، 828هـ)، حققها على ثلاثة أصول مخطوطة: هلال ناجي، عالم الكتب، بيروت، ط1، (1418ه/ 1998م)، ‏104‏.
