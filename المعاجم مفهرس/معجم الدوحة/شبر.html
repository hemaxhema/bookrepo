شِبْر[اسْم]

قبل 13ق.هـ=609م

الشِّبْرُ : اسْمُ وَحْدَةٍ لِقِيَاسِ الأَبْعَادِ، تَزِيدُ عَلَى الكَفِّ، وَهُوَ المَسَافَةُ بَيْنَ طَرَفَيِ الإِبْهَامِ وَالخِنْصَرِ.

وَإِنِّي، إِذَا مَا المَوْتُ لَمْ يَكُ دُونَهُ قِدَى الشِّبْرِ ، أَحْمِي الأَنْفَ أَنْ أَتَأَخَّرَا
حاتم بن عبد الله الطّائيّ
ديوان شعر حاتم بن عبد الله الطائي وأخباره: صنعة: يحيى بن مدرك الطائي، رواية: هشام بن محمّد الكلبي، دراسة وتحقيق: عادل سليمان جمال، مطبعة المدني، القاهرة، د.ت. ‏269‏.

شَبْر[اسْم]

قبل 13ق.هـ=609م

الشَّبْرُ : العَطَاءُ.

وَلَمْ نَقْرِهِمْ شَيْئًا، وَلَكِنَّ قَصْدَهُمْ صَبُوحٌ لَدَيْنَا، مَطْلَعَ الشَّمْسِ، حَازِرُ
صَبَحْنَاهُمُ، عِنْدَ الشُّرُوقِ، كَتَائِبًا كَأَرْكَانِ سَلْمَى شَبْرُهَا مُتَوَاتِرُ
مُعقِّر بن أوس البارقيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏11/‏111‏.
ن50هـ=670م

الشَّبْرُ : القَدُّ وَالقَامَةُ.

فَإِنْ جَلَسَتْ وَسْطَ النِّسَاءِ شَهَرْنَهَا، وَإِنْ هِيَ قَامَتْ فَهْيَ كَامِلَةُ الشَّبْرِ
الرّحّال بن مَجدوح النُّمَيريّ
ديوان جران العَوْد النميري وضمنه شعر عُروة الرّحّال: تحقيق: كارين صادر، دار صادر، بيروت، ط1، (1999م)، ‏46‏.
ن80هـ=699م

الشَّبْرُ : الطُّولُ.

يَدِي، يَا أَمِيرَ المُؤْمِنِينَ، أُعِيذُهَا بِحِقْوَيْكَ أَنْ تُلْقَى بِمُلْقًى يُهِينُهَا
فَقَدْ كَانَتِ الحَسْنَاءَ لَوْ تَمَّ شَبْرُهَا ، وَلَا تَعْدِمُ الحَسْنَاءُ عَابًا يَشِينُهَا
طَهْمان بن عمرو الكِلابيّ
ديوان طهمان بن عمرو الكلابي: شرح: أبي سعيد السكري، تحقيق: محمّد جبار المعيبد، مطبعة الإرشاد، بغداد، (1968م)، ‏40‏.
ن175هـ=791م

شَبْرُ المَرْأَةِ: حَقُّهَا فِي النِّكَاحِ.

"وَأَعْطَاهَا شَبْرَهَا ، أَيْ: حَقَّهَا فِي النِّكَاحِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏258‏.

شَبْر[مَصْدَر]

ن129هـ=747م

الشَّبْرُ : الجِمَاعُ وَالضِّرابُ.

"أَأَنْ سَأَلَتْكَ ثَمَنَ شَكْرِهَا وَشَبْرِكَ ، أَنْشَأْتَ تَطُلُّهَا وَتَضْهَلُهَا"
يحيى بن يَعْمَر العَدْوانيّ
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏1/‏378‏.

شَبَر[اسْم]

قبل 13ق.هـ=609م

الشَّبَرُ : القُرْبَانُ مِنْ هَدِيَّةٍ وَنَحْوِهَا.

إِذْ أَتَانِي خَبَرٌ مِنْ مُنْعِمٍ لَمْ أَخُنْهُ وَالَّذِي أَعْطَى الشَّبَرْ
عَديّ بن زيد العِباديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏259‏.

أَشْبَرَ[مُتَعَدٍّ]

ن2ق.هـ=620م

أَشْبَرَهُ الشَّيْءَ: أَعْطَاهُ إِيَّاهُ.

وَبَيضَاءَ زَغْفٍ نَثْلَةٍ سُلَمِيَّةٍ لَهَا رَفْرَفٌ فَوْقَ الأَنَامِلِ مُرْسَلُ
وَأَشْبَرَنِيهِ الهَالِكِيُّ كَأَنَّهُ غَدِيرٌ جَرَتْ فِي مَتْنِهِ الرِّيحُ سَلْسَلُ
أوس بن حَجَر المازنيّ التَّميميّ
ديوان أوس بن حجر: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، ط3، (1399هـ/ 1979م)، ‏96‏.

شَابِر[اسْم]

ن18هـ=639م

الشَّابِرُ : الإِصْبَعُ، يُقَاسُ بِهَا الشِّبْرُ.

غَلَبْنَا عَلَى مَاءِ الفُرَاتِ وَأَرْضِهِ عَشِيَّةَ جُزْنَا بِالسُّيُوفِ الأَكَابِرَا
فَدَرَّتْ عَلَيْنَا جِزْيَةُ القَوْمِ بَعْدَما ضَرَبْنَاهُمُ ضَرْبًا يَعُطُّ الشَّوَابِرَا
أبو مُفَزِّر الأسود بن قُطْبَة الأُسَيِّديّ التَّميميّ
شعراء إسلاميون: تحقيق: نوري حمّودي القيسيّ، عالم الكتب- مكتبة النهضة العربية، بيروت، ط2، (1405هـ/ 1984م)، ‏121‏.

شَابِر[اسْمُ فَاعِل]

53هـ=673م

الشَّابِرُ : الَّذِي يَقِيسُ المَسَافَةَ بِالشِّبْرِ.

فَمَا لَكُمُ فِي أَرْضِ نَجْدٍ وَغَوْرِهَا، إِذَا اقْتَسَمُوا بِالحَقِّ، شِبْرٌ لِشَابِرِ
فَقُمْ صَاغِرًا، يَا شَيْخَ جَرْمٍ، فَإِنَّمَا يُقَالُ لِشَيْخِ الصِّدْقِ: قُمْ غَيْرَ صَاغِرِ
زياد الأعجم
شعر زياد الأعجم: جمع وتحقيق ودراسة: يوسف حسين بكار، دار المسيرة، بيروت، ط1، (1403هـ/ 1983م)، ‏73‏.
125هـ=743م

الشَّابِرُ مِنَ الأَشْبَارِ: الفَائِقُ غَيْرَهُ فِي الطُّولِ.

إِذَا اجْتَلَتْ أَيَّامُهُ المَفَاخِرَا مَدَّ ابْنُ عَبْدِ اللهِ شِبْرًا شَابِرَا إِلَى عُلَا الأَعْلَى، وَضَوْءًا زَاهِرَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏185‏.

شَبَّرَ[مُتَعَدٍّ]

ن61هـ=681م

شَبَّرَ الشَّيْءَ: شَبَرَهُ.

"عَنْ أُمِّ الحَسَنِ البَصْرِيِّ أَنَّ أُمَّ سَلَمَةَ حَدَّثَتْهُمْ أَنَّ النَّبِيَّ @ شَبَّرَ لِفَاطِمَةَ شِبْرًا مِنْ نِطَاقِهَا"
أمّ سَلَمَة
الجامع الصحيح وهو سُنن الترمذي: التّرمذيّ (ت، 279هـ)، تحقيق وشرح: أحمد محمد شاكر، وآخرين، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1382هـ/ 1962م- 1398هـ/ 1978م)، ‏4/‏224،،‏(رقم الحديث: 1732).
ن216هـ=831م

شَبَّرَهُ : عَظَّمَهُ.

" شُبِّرَ فَتَشَبَّرَ "
مثل
غريب الحديث: ابن قتيبة (ت، 276هـ)، تحقيق: عبد الله جبوري، إحياء التراث الإسلامي- مطبعة العاني، بغداد، ط1، (1397هـ/ 1977م)، ‏2/‏582‏.

شَبَرَ[مُتَعَدٍّ]

ن61هـ=681م

شَبَرَ الشَّيْءَ: قَاسَ مِقْدَارَهُ بِالشِّبْرِ.

"عَنْ أُمِّ الحَسَنِ أَنَّ أُمَّ سَلَمَةَ حَدَّثَتْهُمْ أَنَّ رَسُولَ اللهِ @ شَبَرَ لِفَاطِمَةَ شِبْرًا مِنْ نِطَاقِهَا"
أمّ سَلَمَة
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏44/‏178،،‏(رقم الحديث: 26554).
67هـ=686م

شَبَرَهُ الشَّيْءَ: مَنَحَهُ لَهُ وَأَعْطَاهُ إِيَّاهُ.

فَيَا وَاهِبَ الوُصَفَاءِ الصِّبَا حِ إِنْ شُبِرَتْ تَمَّ إِشْبَارُهَا
أعشى هَمْدان
ديوان أعشى هَمدان وأخباره: تحقيق: حسن عيسى أبو ياسين، دار العلوم، الرياض، (1403هـ/ 1983م)، ‏123‏.
ن175هـ=791م

شَبَرَ الشَّخْصَ: قَايَسَهُ سَعَةَ شِبْرِهِ فَغَلَبَهُ.

"يُقَالُ: هَذَا أَشْبَرُ مِنْ هَذَا، أَيْ: أَوْسَعُ مِنْهُ شِبْرًا، وَأَنَا أَشْبُرُهُ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏258‏.

مَشْبَر[اسْم]

65هـ=685م

المَشْبَرُ : القِطْعَةُ وَالحُزَّةُ.

يَا طَلْحَ، يَا لَيْتَكَ عَنَّا تُخْبَرُ! إِنَّا أَتَانَا جُرَذٌ مُوَزَّرُ شِبْرٌ مِنَ المَشَابِرِ حِينَ يُشْبَرُ أَنْكَرَهُ سَرِيرُنَا وَالمِنْبَرُ
أبو حُزابة التَّميميّ
الإشراف في مَنازل الأشراف: ابن أبي الدُّنيا (ت، 281هـ)، قدم له وحقّقه وعلّق عليه: نجم عبد الرحمن خلف، مكتبة الرشد، الرياض، ط1، (1411هـ/ 1990م)، ‏1/‏245‏.
280هـ=893م

المَشْبَرَةُ : النَّهْرُ الصَّغِيرُ المُنْخَفِضُ بَيْنَ نَهْرَيْنِ يَأْتِيهِ المَاءُ مِنْهُمَا مَعًا.

وَبَاتَ مَحَلُّهُمْ أَضْوَاجَ طِبْنٍ لِمَشْبَرَةٍ لِعَانَتِهِ تَهَارِي
من إنشاد ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏3/‏1305‏.

مَشْبَر[مَصْدَرٌ مِيمِيّ]

127هـ=745م

المَشْبَرُ : قِيَاسُ الطُّولِ بِالشِّبْرِ.

قَامَرْتَ عَنْ مَجْدِ امْرِئٍ لَمْ يُقْهَرِ يَزْدَادُ فِي البَاعِ، وَعِنْدَ المَشْبَرِ طُولًا، إِذَا قَصَّرَ بَاعُ الأَقْصَرِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏309‏.

مِشْبَر[اسْمُ آلَة]

65هـ=685م

المِشْبَرُ : مِقْيَاسٌ لِلطُّولِ مُقَدَّرًا بِالشِّبْرِ.

قَطَعْتُ لَهَا مِنْ جَانِبِ الحَوْضِ مَشْرَبًا صَغِيرًا كَقِيدِ الشِّبْرِ، أَوْ هُوَ أَصْغَرُ
إِذَا شَرَعَتْ فِيهِ، فَلَيْسَ لِمُلْتَقَى مَشَافِرِهَا مِنْهُ قِدَى الشِّبْرِ مِشْبَرُ
عمر بن أبي ربيعة
أمالي المرزوقيّ: أبو عليّ المرزوقيّ (ت، 421هـ)، تحقيق: يحيى وهيب الجبوري، دار الغرب الإسلاميّ، بيروت، ط1، (1995م)، ‏355‏.

مِشْبَار[اسْمُ آلَة]

ن70هـ=689م

المِشْبَارُ : مِقْيَاسٌ لِلطُّولِ مُقَدَّرًا بِالشِّبْرِ.

يَا قَبَّحَ اللهُ صِبْيَانًا تَجِيءُ بِهِمْ أُمُّ الهُنَيْبِرِ مِنْ زَنْدٍ لَهَا وَارِي
مِنْ كُلِّ أَعْلَمَ مُنْشَقٍّ مَشَافِرُهُ، وَمُؤْدَنٍ مَا وَفَى شِبْرًا بِمِشْبَارِ

الرّحلة العياشية: أبو سالم العياشي (ت، 1090هـ)، حققها وقدم لها: سعيد الفاضلي، سليمان القرشي، دار السُّويدي، أبو ظبي، ط1، (1426ه/ 2006م)، ‏2/‏385‏.

أُشْبُور[اسْمُ جِنْس]

ن463هـ=1070م

الأُشْبُورُ : سَمَكٌ يُنْتِجُ بَيْضًا صَغِيرَ الحَجْمِ، دَقِيقَ الحَبِّ، يُسْتَخْدَمُ فِي الطِّيبِ، وَتَتَمَيَّزُ ذُكُورُهُ بِتَتَبُّعِ الإِنَاثِ وَالْتِقَامِ مَا تَبِيضُ.

"وَالبَيْضُ المُقَدَّمُ فِي الطِّيبِ ثَلَاثَةُ أَجْنَاسٍ: بَيْضُ الأُشْبُورِ ، وَبَيْضُ الدَّجَاجِ، وَبَيْضُ الجَرَادِ"
ابن عبد البَرّ القرطبيّ
بهجة المَجالس وأنس المُجالِس وشحذ الذاهن والهاجس: ابن عبد البَرّ القرطبيّ (ت، 463هـ)، تحقيق: محمّد مرسي الخولي، دار الكتب العلمية، بيروت، ط2، (1981م)، ‏2/‏82‏.

شُبَيْر[اسْمٌ مُصَغَّر]

ن596هـ=1199م

الشُّبَيْرُ : تَصْغِيرُ الشِّبْرِ.

"وَلَيْتَ هَذِهِ المَحَاسِنَ وَلِيَتِ الأَسْمَاع، وَأَلْقَتِ القِنَاع، وَفِي العُمْرِ مُسْتَمْتَع، وَفِي قَوْسِ الشَّبِيبَةِ مَنْزَع، وَلَكِنْ ضَاقَ فِتْرٌ عَنْ شُبَير ، وَجَاءَ فَضْلُهَا الأَوَّلُ فِي الزَّمَنِ الأَخِير"
القاضي الفاضِل
خزانة الأدب وغاية الأرب: ابن حجة الحمويّ (ت، 837هـ)، دراسة وتحقيق: كوكب دياب، دار صادر، بيروت، ط2، (1425ه/ 2005م)، ‏4/‏368‏.

شَبِير[اسْم]

667هـ=1268م

الشَّبِيرُ : المِهْمَازُ.

"وَذَلِكَ خَمْسَةٌ وَعِشْرُونَ أَلْفَ دِينَارٍ مِنَ الذَّهَبِ، وَسَبْعُونَ فَرَسًا، مِنْهَا خَمْسَةٌ وَعِشْرُنَ مُجَهَّزَةٌ بِفَاخِرِ الجِهَازَاتِ، وَسَبْعُونَ سَيْفًا مُحَلَّاةٌ، وَعِشْرُونَ مِنَ الأَشَابِرِ المُذَهَّبَةِ"
ابن عِذَاري المراكشيّ
البيان المغرب في اختصار أخبار ملوك الأندلس والمغرب: ابن عِذَاري المراكشيّ (ت، 695هـ)، حققه وضبط نصه وعلق عليه: بشار عواد معروف، محمود بشار عواد، دار الغرب الإسلامي، تونس، ط1، (1434ه/ 2013م)، ‏3/‏21‏.

شَبَار[اسْم]

ن1098هـ=1686م

الشَّبَارُ : نَوْعٌ مِنَ السَّمَكِ.

"وَنَوْعٌ فِي السَّمَكِ يُقَالُ لَهُ: شَبَارٌ ، لَهُ لَذَّةٌ فِي الطَّعْمِ وَالمَأْكَلِ"
الشّربيني
هز القحوف في شرح قصيدة أبي شادوف: الشّربيني (ت، 1098هـ)، اعتنى به: عبد العزيز جمال الدّين، الهيئة العامة لقصور الثَّقَافَة، القاهرة، ط1، (1440ه/ 2019م)، ‏367‏.

مُتَشَبِّر[اسْمُ فَاعِل]

ن1239هـ=1823م

المُتَشَبِّرُ عَلَى الشَّيْءِ: المُتَمَسِّكُ بِهِ وَالمُتَشَبِّثُ.

"فَقُلْتُ لَهُ: وَأَنْتُمَا مُتَشَبِّرَانِ عَلَيْهَا، إِنْ كُنْتُمَا كَذَلِكَ، نَعْمَلُ لَكُمَا هَكَذَا، ثُمَّ قَبَضْتُ يَدَهُ وَنَفَضْتُهَا، وَإِذَا بِالرَّجُلِ قَدْ جَاءَنِي غَدًا، وَلَمَّا رَآنِي رَدَّ يَدَهُ وَرَاءَهُ، وَحَنَى عَلَى رِجْلِي وَقَبَّلَهَا، ثُمَّ قَالَ لِي: اللهُ يُرَبِّحُكَ بِابْنَتِي؛ أَعْطِهَا لِمَنْ شِئْتَ"
محمد العربي الدرقاوي
مجموعة رسائل مولاي العربي الدرقاوي الحسني: محمد العربي الدرقاوي (ت، 1239هـ)، تحقيق: بسام محمد بارود، قدم له: أحمد بن الخياط الزكاري، المجمع الثقافي، أبو ظبي، ط1، (1420ه/ 1999م)، ‏328‏.

شَبُّورَة[اسْم]

"أَبْخِرَةٌ خَفِيفَةٌ مُنْتَشِرَةٌ، وَشَبُّورَةٌ خَفِيفَةٌ صَبَاحًا"

تَشَبُّر[مَصْدَر]

ن1308هـ=1890م

تَشَبُّرُ الشَّخْصِ: تَعَظُّمُهُ وَاعْتِدَادُهُ بِنَفْسِهِ.

شُبِّرَ، فَاغْتَدَى أَخَا تَشَبُّرِ ذَاكَ الشَّقِيُّ، ابْنُ الشَّقِيِّ المُجْتَرِي

تَشْبِير[مَصْدَر]

1435هـ=2013م

التَّشْبِيرُ : القِيَاسُ بِالشِّبْرِ.

"وَلِذَلِكَ الدَّيْرِ قِصَّةٌ مُلْغَزَةٌ أَعْيَا النَّاسَ تَفْسِيرُهَا، ذَلِكَ أَنَّ المُحَاوَلَاتِ المُتَكَرِّرَةَ لِحِسَابِ مِسَاحَتِهِ، أَوْ تَشْبِيرِهِ ، لَمْ تُفْلِحْ، فَكَانَتْ نَتِيجَةُ الحِسَابِ تَخْتَلِفُ فِي كُلِّ مَرَّةٍ"
عقيل سعيد محفوض
الأكراد واللغة والسياسة: دراسة في البنى اللغوية وسياسات الهوية: عقيل سعيد محفوض ، المركز العربي للأبحاث ودراسة السياسات، الدوحة، ط1، (2013م)، ‏11‏.
