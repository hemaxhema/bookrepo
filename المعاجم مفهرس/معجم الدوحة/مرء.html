مَرْء[اسْم]

قبل 150ق.هـ=476م

المَرْءُ : الإِنْسَانُ رَجُلًا وَامْرَأَةً.

إِذَا الجَوْزَاءُ أَرْدَفَتِ الثُّرَيَّا ظَنَنْتُ بِآلِ فَاطِمَةَ الظُّنُونَا
ظَنَنْتُ بِهَا، وَظَنُّ المَرْءِ حُوبٌ وَإِنْ أَوْفَى، وَإِنْ سَكَنَ الحَجُونَا
منسوب إلى حَزيمة بن نَهْد القُضاعيّ
الشُّعراء الجاهليون الأوائل: تحقيق: عادل الفريجات، دار المشرق، بيروت، ط2، (2008م)، ‏128‏-‏129‏.
قبل 100ق.هـ=525م

المَرْءُ : الرَّجُلُ.

مَضَوْا قَصْدَ السَّبِيلِ، وَخَلَّفُونِي إِلَى خَلَفٍ مِنَ الأَبْرَامِ خَلْفِ
سُدًى، لَا يَكْتَفُونَ، وَلَا أَرَاهُمْ يُطِيعُونَ امْرَأً ، إِنْ كَانَ يَكْفِي
أُحَيحة بن الجُلاح الأوسيّ
الشُّعراء الجاهليون الأوائل: تحقيق: عادل الفريجات، دار المشرق، بيروت، ط2، (2008م)، ‏444‏.

مَرْأَة[اسْم]

قبل 100ق.هـ=525م

المَرْأَةُ : أُنْثَى الإِنْسَانِ البَالِغَةُ.

"وَلَوْ أَنَّ امْرَأَةً اسْتَغْنَتْ عَنِ الزَّوْجِ؛ لِغِنَى أَبَوَيْهَا، وَشِدَّةِ حَاجَتِهِمَا إِلَيْهَا، كُنْتِ أَغْنَى النَّاسِ عَنْهُ"
أُمامة بنت الحارث الشَّيْبانيّة
مجمع الأمثال: أبو الفضل الميدانيّ (ت، 518هـ)، حققه وفصله وضبط غرائبه وعلق حواشيه: محمّد محيي الدين عبد الحميد، مطبعة السنة المحمدية، القاهرة، (1374هـ/ 1955م)، ‏2/‏263‏.

مَرِيء[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

المَرِيءُ مِنَ الأَشْيَاءِ: السَّائِغُ.

أَغَرَّكُمُ أَنِّي بِأَحْسَنِ شِيمَةٍ رَفِيقٌ، وَأَنِّي بِالفَوَاحِشِ أَخْرَقُ
وَأَنَّكَ قَدْ فَاحَشْتَنِي فَغَلَبْتَنِي؟ هَنِيئًا مَرِيئًا ، أَنْتَ بِالفُحْشِ أَحْذَقُ
لَقيط بن زُرارة بن عُدَس الدّارميّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏320‏.
ن175هـ=791م

المَرِيءُ مِنَ النَّاسِ: الظَّاهِرُ المُرُوءَةِ.

"وَهُوَ مَرِيءٌ ، بَيِّنُ المُرُوءَةِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏299‏.

مَرِيء[اسْم]

ن26هـ=647م

المَرِيءُ : مَجْرَى الطَّعَامِ وَالشَّرَابِ فِي رَأْسِ المَعِدَةِ.

تَرَى المَرِيءَ كَنَصْلِ السَّيْفِ إِذْ ضَمِنَتْ أَوِ النَّضِيَّ الفَضَا، بَطَّنْتَهُ العُنُقَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏236‏.

مُرُوءَة[مَصْدَر]

قبل 13ق.هـ=609م

المُرُوءَةُ : صِفَةٌ نَفْسِيَّةٌ، تَحْمِلُ صَاحِبَهَا عَلَى الأَخْذِ بِمَحَاسِنِ الأَخْلَاقِ، وَاجْتِنَابِ مَسَاوِئِهَا.

"فَمَا أَفْضَلُ المُرُوءَةِ ؟"
قُسّ بن ساعدة الإياديّ
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏2/‏43‏.

أَمْرَأ[اسْمُ تَفْضِيل]

ن11هـ=632م

الأَمْرَأُ مِنَ الطَّعَامِ وَنَحْوِهِ: الأَكْثَرُ مَرَاءَةً.

"كَانَ النَّبِىُّ  @  يَتَنَفَّسُ فِي الإِنَاءِ ثَلَاثًا، وَيَقُولُ: هَذَا أَهْنَأُ، وَأَمْرَأُ ، وَأَبْرَأُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏19/‏224،،‏(رقم الحديث: 12186).

مُرَيْئَة[اسْمٌ مُصَغَّر]

ن11هـ=632م

المُرَيْئَةُ : تَصْغِيرُ المَرْأَةِ.

"إِنِّي لَأَكْرَهُ أَنْ أَرَى الرَّجُلَ ثَائِرًا فَرِيصُ رَقَبَتِهِ، قَائِمًا عَلَى مُرَيْئَتِهِ يَضْرِبُهَا"
حديث نبويّ

اسْتَمْرَأَ[مُتَعَدٍّ]

ن37هـ=657م

اسْتَمْرَأَ الشَّيْءَ: اسْتَطَابَهُ، وَهَنِئَ بِهِ.

"وَلَكِنَّ القَوْمَ ذَاقُوا الدُّنْيَا، فَاسْتَحَبُّوهَا وَاسْتَمْرَؤُوهَا "
عمّار بن ياسر العَنْسيّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏5/‏39‏.
ن70هـ=689م

سَائِلْ بِنَا حَيَّ عِلْبَاءٍ، فَقَدْ شَرِبُوا مِنَّا بِكَأْسٍ، فَلَمْ يَسْتَمْرِئُوا الشُّرُبَا

الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏56‏.

اسْتَمْرَأَ[مُتَعَدٍّ بِالحَرْف]

ن749هـ=1348م

اسْتَمْرَأَ مِنْهُ: اسْتَفَادَ مِنْهُ المُرُوءَةَ.

"وَهُوَ مِنْ أَوَّلِ شُيُوخِي، الَّذِينَ عَلَيْهِمْ قَرَأْتُ، وَمِنْهُمُ اسْتَمْرَأْتُ ، قَرَأْتُ عَلَيْهِ النَّحْوَ"

مُسْتَمْرَأ[اسْمُ مَفْعُول]

ن313هـ=925م

المُسْتَمْرَأُ مِنَ الطَّعَامِ وَنَحْوِهِ: المُسْتَطَابُ المُسْتَسَاغُ.

"وَالجَوْزُ الَّذِي لَمْ يَسْتَحْكِمْ بَعْدُ، وَلَمْ يَجِفَّ مِثْلُ جَمِيعِ غَيْرِهِ مِنَ الثِّمَارِ، مَمْلُوءٌ رُطُوبَةً غَيْرَ مُسْتَمْرَأَةٍ "
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏6/‏93‏.

تَمَرُّؤ[مَصْدَر]

ن313هـ=925م

التَّمَرُّؤُ : وَجَعُ المَرِيءِ.

أَقْوَى الأَدْوِيَةِ -فِي مَا جَرَّبْنَاهُ- لِلتَّمَرُّؤِ ؛ مِنْ أَجْلِ البَوْرَقِ وَالسَّذَابِ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏3/‏111‏.

مَرِيئِيّ[اسْمٌ مَنْسُوب]

ن369هـ=979م

المَرِيئِيُّ : المَنْسُوبُ إِلَى المَرِيءِ.

"صِفَةُ شَرَابٍ أَلَّفْتُهُ، نَافِعٍ مِنْ ضَعْفِ المَعِدَةِ وَفَسَادِهَا، وَالقَيْءِ المُتَوَلِّدُ عَنِ الصَّفْرَاءِ، وَالإِسْهَالِ المَرِيئِيِّ ، وَهُوَ عَجِيبٌ"
ابن الجزّار القيروانيّ
كتاب في المَعِدة وأمراضها ومداواتها: ابن الجزّار القيروانيّ (ت، 369هـ)، تحقيق: سلمان قطاية، دار الرشيد، بغداد، (1980م)، ‏196‏.

مَرَأَ[مُتَعَدٍّ]

440هـ=1048م

مَرَأَهُ الطَّعَامُ: سَاغَ لَهُ، وَخَفَّ عَلَيْهِ.

"هَنَأَنِي الطَّعَامُ، وَمَرَأَنِي "
أبو العلاء المَعَرّيّ
اللّامع العزيزيّ شرح ديوان المتنبّيّ: أبو العلاء المَعَرّيّ (ت، 449هـ)، حققه: محمّد سعيد المولوي، مركز الملك فيصل للبحوث والدراسات الإسلامية، الرياض، ط1، (1429هـ/ 2008م)، ‏1/‏523‏.

مَرَأَ[لَازِم]

ن650هـ=1252م

مَرَأَ الشَّخْصُ: طَعِمَ.

"مَالَكَ لَا تَمْرَأُ ؟"
الصّاغانيّ
العباب الزاخر واللباب الفاخر: الصّاغانيّ (ت، 650هـ)، تحقيق: فير محمد حسن، محمد حسن آل ياسين، راجعته وأشرفت على طبعه: لجنة المجمع، مطبوعات المجمع العلمي العراقي- دار الشؤون الثقافية العامة- دار الرشيد، بغداد، ط1، (1398ه/ 1978م- 1402ه/ 1981م)، ‏1/‏112‏.

إِمْرَاء[مَصْدَر]

ن487هـ=1094م

إِمْرَاءُ الكَلَإِ وَنَحْوِهِ: اسْتِطَابَتُهُ.

"فَكَانَتْ بَنُو عَامِرٍ يَتَصَيَّفُونَ الطَّائِفَ؛ لِطِيبِهَا وَثِمَارِهَا، وَيَتَشَتَّوْنَ بِلَادَهُمْ مِنْ أَرْضِ نَجْدٍ؛ لِسَعَتِهَا، وَكَثْرَةِ مَرَاعِيهَا، وَإِمْرَاءِ كَلَئِهَا، وَيَخْتَارُونَهَا عَلى الطَّائِفِ"
أبو عُبيد البَكْريّ الأندلسيّ
معجم ما اسْتَعْجَم من أسماء البلاد والمواضع: أبو عُبيد البَكْريّ الأندلسيّ (ت، 487هـ)، حققه وضبطه: مصطفى السقا، عالم الكتب، بيروت، (1364هـ/ 1945م)، ‏1/‏77‏.

تَمْرِئَة[مَصْدَر]

ن619هـ=1222م

التَّمْرِئَةُ : الِاسْتِرَاحَةُ فِي أَثْنَاءِ الرَّعْيِ.

"وَذَلِكَ أَنَّ الإِبِلَ تَرِدُ المَاءَ فَتَشْرَبُ مِنْهُ، ثُمَّ تَخْرُجُ تَرْعَى سَاعَةً وَتَسْتَرِيحُ، وَتُسَمَّى تِلْكَ الِاسْتِرَاحَةُ فِي الرَّعِيِّ التَّمْرِئَةَ ، ثُمّ تَرِدُ مَرَّةً أُخْرَى فَتَشْرَبُ المَاءَ"
الشّريشيّ
شرح مقامات الحريريّ: الشّريشيّ (ت، 619هـ)، تحقيق: محمَّد أَبو الفضل إِبراهيم، المكتبة العصريَّة- شركة أبناء شريف الأنصاري- الدار النموذجية- المطبعة العصرية، بيروت- صيدا، (1413ه/ 1992م)، ‏1/‏159‏.
ن1294هـ=1876م

تَمْرِئَةُ الأَرْضِ: تَجْفِيفُهَا وَتَصْفِيَةُ مَا فِيهَا مِنَ المِيَاهِ الرَّاكِدَةِ.

"وَلْنَشْرَعْ فِي ذِكْرِ الطُّرُقِ النَّافِعَةِ؛ لِتَتْمِيمِ هَذِهِ الشُّرُوطِ المُخَصِّبَةِ، مُبْتَدِئِينَ بِتَجْفِيفِ مَنَاِقعِ المِيَاهِ وَتَمْرِئَةِ الأَرَاضِي، أَيْ تَصْفِيَةِ مَا فِيهَا مِنَ المِيَاهِ"
أحمد ندا بك
حسن الصناعة في علم الزراعة: أحمد ندا بك (ت، 1294هـ)، دار الطباعة العامرة، مصر، (1291ه)، ‏1/‏84‏.

مُمْرِئ[اسْمُ فَاعِل]

ن730هـ=1329م

المُمْرِئُ لِلطَّعَامِ وَنَحْوِهِ: الجَاعِلُهُ لَذِيذًا مُسْتَطَابًا.

"فَإِنَّ عُمَرَ <  إِنَّمَا اسْتَشَارَهُمْ فِي المُشْتَدِّ دُونَ الحُلْوِ، وَهُوَ مَا يَكُونُ مُمْرِئًا لِطَعَامٍ، مُقَوِّيًا عَلَى الطَّاعَةِ فِي لَيَالِيِ الصِّيَامِ"
عبد العزيز البخاريّ
كشف الأسرار عن أصول فخر الإسلام البزدوي: عبد العزيز البخاريّ (ت، 730هـ)، مطبعة الشركة الصحفية العثمانية، إسطنبول، (1308ه)، ‏3/‏361‏.

مَرِئَ[لَازِم]

1282هـ=1865م

مَرِئَ الهَوَاءُ وَنَحْوُهُ: رَقَّ وَطَابَ.

شَجْرَاءُ تَسْلُكُهَا السَّمُومُ، فَتَغْتَدِي رَهْوًا، وَيَسْكُنُهَا الهَجِيرُ، فَيَمْرَأُ
محمود سامي البارودي
ديوان محمود سامي البارودي: محمود سامي البارودي (ت، 1322هـ)، مؤسسة هنداوي، وند سور، المملكة المتحدة، (1435ه/ 2014م)، ‏39‏.

اسْتِمْرَائِيّ[صِفَة]

ن1437هـ=2015م

الِاسْتِمْرَائِيُّ مِنَ الأَشْخَاصِ: الَّذِي يَسْتَطِيبُ الأُمُورَ وَيَسْتَسِيغُهَا.

"بِالنِّسْبَةِ لِأُولَئِكَ الَّذِينَ لَا هَمَّ لَهُمْ غَيْرَ مَا يَسْتَعْذِبُونَهُ وَيَتَلَذَّذُونَ بِهِ، كَوْنَهُمْ يَمْتَلِكُونَ شَخْصِيَّةً اسْتِمْرَائِيَّةً ، لَا تَتَجَاوَبُ إِلَّا مَعَ الأَحْدَاثِ المُصْطَنَعَةِ فِي ثَوْبِهَا البَرَّاقِ"
بشرى عبد المجيد تاكفراست
المرأة في الخطاب الأدبي الإعلامي والثقافي، وقائع المؤتمر الدولي الأدبي الثالث : عبد الرحيم مراشدة، هيثم أحمد العزام، دار الكتاب الثقافي، عمان، (2015م)، ‏726‏.
