جُبْن[مَصْدَر]

قبل 150ق.هـ=476م

الجُبْنُ : تَهَيُّبُ الإِقْدَامِ عَلَى الأُمُورِ.

وَذُو الجُبْنِ مِمَّا يُسْعِرُ الحَرْبَ نَفْخُهُ يُهَيِّجُ مِنْهَا نَارَهَا، ثُمَّ يَخْنُسُ
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏63‏.

جُبْن[اسْمُ جِنْس]

ن23هـ=644م

الجُبْنُ : اللَّبَنُ يُخَثَّرُ عَلَى نَحْوٍ مَخْصُوصٍ، حَتَّى يَغْلُظَ وَيَجْمُدَ.

"بَلَغَنِي أَنَّكُمْ فِي أَرْضٍ تَأْكُلُونَ طَعَامًا يُقَالُ لَهُ: الجُبْنُ ، فَانْظُرُوا مَا حَلَالُهُ مِنْ حَرَامِهِ؟"
عمر بن الخطّاب
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏19/‏442‏.

جَبَان[صِفَةٌ مُشَبَّهَة]

قبل 100ق.هـ=525م

الجَبَانُ : الَّذِي يَهَابُ الإِقْدَامَ عَلَى الأُمُورِ.

إِذْ لَا يُقَارِعُ أَطْرَافَ الظُّبَاتِ إِذَا اسْـ ـتَوْقَدْنَ إِلَّا كُمَاةٌ غَيْرُ أَجْبَانِ
أبو قِلابة الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏712‏.

جَبِين[اسْم]

قبل 100ق.هـ=525م

الجَبِينُ : مَا فَوْقَ الصُّدْغِ عَنْ يَمِينِ الجَبْهَةِ أَوْ شِمَالِهَا.

وَلَرُبَّ وَاضِحَةِ الجَبِينِ غَرِيرَةٍ مِثْلِ المَهَاةِ، تَرُوقُ عَيْنَ النَّاظِرِ
قَدْ بِتُّ أُلْعِبُهَا، وَأَقْصُرُ هَمَّهَا حَتَّى بَدَا وَضَحُ الصَّبَاحِ الجَاشِرِ
ثعلبة بن صُعير المازنيّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏383‏.
قبل 13ق.هـ=609م

الجَبِينُ : الجَبْهَةُ.

جَزَى اللهُ الأَغَرَّ جَزَاءَ صِدْقٍ إِذَا مَا أُوقِدَتْ نَارُ الحُروبِ
يَقِينِي بِالجَبِينِ وَمَنْكِبَيْهِ وَأَنْصُرُهُ بِمُطَّرِدِ الكُعُوبِ
عنترة
ديوان عنترة: تحقيق ودراسة: محمّد سعيد مولوي، المكتب الإسلامي، بيروت، (1390هـ/ 1970م)، ‏320‏.

جَبُنَ[لَازِم]

قبل 50ق.هـ=573م

جَبُنَ : تَهَيَّبَ الإِقْدَامَ عَلَى الأُمُورِ.

وَمَا جَبُنَتْ خَيْلِي، وَلَكِنْ تَذَكَّرَتْ مَرَابِطَهَا مِنْ بَرْبَعِيصَ وَمَيْسَرَا
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏97‏.

أَجْبَن[اسْمُ تَفْضِيل]

2هـ=623م

الأَجْبَنُ : الأَشَدُّ جُبْنًا.

"يَا قَوْمِ، اعْصِبُوهَا اليَوْمَ بِرَأْسِي، وَقُولُوا: جَبُنَ عُتْبَةُ بْنُ رَبِيعَةَ؛ وَقَدْ عَلِمْتُمْ أَنِّي لَسْتُ بِأَجْبَنِكُمْ "
عتبة بن ربيعة بن الأُمويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏2/‏260‏.

أَجْبَن[صِفَة]

220هـ=835م

مَا أَجْبَنَهُ : تَعَجُّبٌ مِنْ شِدَّةِ جُبْنِهِ.

"أَخْزَاكَ اللهُ مَا أَجْبَنَكَ !"
الهيثم الغنويّّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏9/‏15‏.

جَبَّنَ[مُتَعَدٍّ]

ن11هـ=632م

جَبَّنَ الشَّخْصَ: حَمَلَهُ عَلَى الجُبْنِ.

"وَاللهِ إِنَّكُمْ لَتُجَبِّنُونَ ، وَتُبَخِّلُونَ، وَإِنَّكُمْ لَمِنْ رَيْحَانِ اللهِ   "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏45/‏293،،‏(رقم الحديث: 27314).
ن85هـ=704م

جَبَّنَ اللَّبَنَ: خَثَّرَهُ وَجَعَلَهُ غَلِيظًا كَالثَّرِيدِ.

"فَأُتِي بِهِ، فَفَتَّ فَتًّا دَقِيقًا، ثُمَّ صَبَّ عَلَيْهِ اللَّبَنَ، ثُمَّ جَبَّنَهُ بِيَدِهِ، حَتَّى جَعَلَهُ كَالثَّرِيدِ"
واثلة بن الأسقع اللَّيْثيّ
دلائل النُّبوّة: أبو نُعيم الأصبهانيّ (ت، 430هـ)، حققه: محمّد رواس قلعه جي، عبد البر عباس، دار النفائس، بيروت، ط2، (1406هـ/ 1986م)، ‏421 ‏-‏ 422‏.
102هـ=720م

جَبَّنَ الشَّخْصَ: اتَّهَمَهُ بِالجُبْنِ.

أَرِقْتُ، وَلَمْ تَأْرَقْ مَعِي أَمُّ خَالِدٍ وَقَدْ أَرِقَتْ عَيْنَايَ حَوْلًا مُجَرَّمَا
عَلَى مَلِكٍ، يَا صَاحِ، بِالعَقْرِ جُبِّنَتْ كَتَائِبُهُ، وَاسْتَوْرَدَ المَوْتَ مُعْلَمَا
ثابت قُطْنة العَتَكيّ الأزديّ
شعر ثابت قُطْنة العَتَكيّ: جمع وتحقيق: ماجد أحمد السامرائي، وزارة الثقافة والإعلام، بغداد، (1388هـ/ 1968م)، ‏57‏.
ن142هـ=759م

جَبَّنَ الشُّجَاعَ وَنَحْوَهُ: صَيَّرَهُ جَبَانًا.

"وَالقَدَرُ هُوَ الَّذِي يَسْلُبُ الأَسَدَ قُوَّتَهُ وَشِدَّتَهُ... وَهُوَ الَّذِي يَجْعَلُ العَاجِزَ حَازِمًا، وَيُثَبِّطُ الشَّهْمَ، وَيُوسِعُ عَلَى المُقْتِرِ، وَيُشَجِّعُ الجَبَانَ، وَيُجَبِّنُ الشُّجَاعَ"
ابن المُقفَّع
كتاب كليلة ودمنة: بيدبا، ترجمة: ابن المُقفَّع (ت، 142هـ)، المطبعة الأميرية، القاهرة، (1937م)، ‏131‏.

ن11هـ=632م

المَجْبَنَةُ : المَدْعَاةُ إِلَى كَثْرَةِ الجُبْنِ.

"إِنَّ الوَلَدَ مَبْخَلَةٌ مَجْبَنَةٌ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏29/‏104،،‏(رقم الحديث: 17562).

مَجْبَنَة[اسْمُ مَكَان]

ن897هـ=1491م

المَجْبَنَةُ : المَقْبَرَةُ.

"مَنْ أَذِنَ لَهَا فِي المَجْبَنَةِ حَتَّى يُدْفَنَ فِيهَا هَؤُلَاءِ؟"
محمد بن يوسف العبدوسي الغرناطي
المعيار المعرب والجامع المغرب عن فتاوي أهل إفريقية والأندلس والمغرب: أبو العباس أحمد بن يحيى الونشريسي (ت، 914هـ)، خرجه: جماعة من الفقهاء: إشراف: محمد حجي، وزارة الأوقاف والشؤون الإسلامية، الرباط، دار الغرب الإسلامي، بيروت، (1401ه/ 1981م)، ‏7/‏35‏.
ن1444هـ=2022م

المَجْبَنَةُ : مَصْنَعُ الجُبْنِ.

"كَانَ رُعَاتُهَا يَعْمَلُونَ، يَصْنَعُونَ الجُبْنَ فِي مَجْبَنَةٍ بِدَائِيَّةٍ"
سارة ياقوت
خضرة: رحلات في وقت الربيع، لتيم دي: سارة ياقوت ، مراجعة: هبة عبد المولى أحمد، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2022م)، ‏207‏.

أَجْبَنَ[مُتَعَدٍّ]

عمرو بن مَعْدي كَرِب الزُّبَيديّ
إصلاح المنطق: ابن السّكّيت (ت، 244هـ)، شرح وتحقيق: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط4، (1368هـ/ 1949م)، ‏250‏.

كتاب الدّلائل في غريب الحديث: قاسم بن ثابت السَّرَقُسْطيّ (ت، 302هـ)، تحقيق: محمّد بن عبد الله القناص، مكتبة العبيكان، الرياض، ط1، (1422هـ/ 2001م)، ‏2/‏693 ‏-‏ 694‏.

اسْتَجْبَنَ[مُتَعَدٍّ]

ن73هـ=692م

اسْتَجْبَنَهُ : اتَّهَمَهُ بِالجُبْنِ.

"خَرَجَتْ سَرِيَّةٌ وَأَنَا فِيهِمْ، فَفَرُّوا، فَلَمَّا رَجَعُوا إِلَى المَدِينَةِ اسْتُجْبِنُوا مِنَ النَّاسِ"
ابن عمر
دَرج الدّرر في تَفِسيِر الآيِ والسّوَر: عبد القاهر الجرجاني، تح: وليد أحمد بن صالح الحسين، بريطانيا- مانشيستر-ط1، 1429هـ= 2008م، ‏834‏-‏ 835‏.

جَبَّان[اسْم]

ن82هـ=701م

الجَبَّانُ : المَكَانُ المُسْتَوِي الوَاسِعُ مِنَ الأَرْضِ.

"أَخَذَ بِيَدِي عَلِيُّ بْنُ أَبِي طَالِبٍ عَلَيْهِ السَّلام فَأَخْرَجَنِي إِلَى الجَبَّانِ ، فَلَمَّا أَصْحَرَ جَلَسَ ثُمَّ تَنَفَّسَ، ثُمَّ قَالَ: يَا كُمَيْلُ بْنَ زِيَادٍ، القُلُوبُ أَوْعِيَةٌ فَخَيْرُهَا أَوْعَاهَا"
كُميل بن زياد النَّخَعيّ
الجليس الصّالح الكافي والأنيس الناصح الشافي: أبو الفرج ابن طَرار الجَريريّ النَّهْروانيّ (ت، 390هـ)، دراسة وتحقيق: إحسان عبّاس، محمّد مرسي الخولي، عالم الكتب، بيروت، ط1، (1413هـ/ 1993م)، ‏4/‏135‏-‏ 136‏.
ن85هـ=704م

الجَبَّانُ : المَقْبَرَةُ.

"صَلَّى عَلَى الحَارِثِ بْنِ قَيْسٍ الجُعْفِيِّ، بَعْدَ مَا صُلِّيَ عَلَيْهِ، أَدْرَكَهُمْ بِالجَبَّانِ "
خَيْثَمة بن عبد الرّحمن الجُعْفيّ
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏7/‏410‏.

جَبَّان[صِيغَةُ مُبَالَغَة]

843هـ=1439م

الجَبَّانُ : بَائِعُ الجُبْنِ وَصَانِعُهُ.

"وَبِهِ عِدَّةٌ كَثِيرَةٌ مِنَ الزَّيَاتِينَ، وَكَثِيرٌ مِنَ الجَبَّانِينَ ، وَالخَبَّازِينَ، وَاللَّبَانِينَ، وَالطَّبَّاخِينَ وَالشَّوَّائِينَ"

مُجَبَّن[اسْمُ مَفْعُول]

ن85هـ=704م

المُجَبَّنُ مِنَ الأَشْخَاصِ: المُصَابُ بِالجُبْنِ.

تُحِبُّهُمْ عُوَّذُ النِّسَاءِ إِذَا مَا احْمَرَّ تَحْتَ القَوَانِسِ الحَدَقُ
وَأَنْكَرَ الكَلْبُ أَهْلَهُ، وَأَتَى الشْـ ـشَرُّ، وَخَافَ المُجَبَّنُ الفَرِقُ
عُبيد الله بن قيس الرُّقَيّات
ديوان عُبيد الله بن قيس الرُّقَيّات: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، د.ت. ‏74‏.
ن313هـ=925م

المُجَبَّنُ مِنَ اللَّبَنِ: المُصَيَّرُ جُبْنًا.

" وَالمُجَبَّنُ بِالقُرْطُمِ قَوِيٌّ فِي إِسْهَالِهِ، وَمَتَى طُبِخَ بَعْدَ أَخْذِهِ، وَجُعِلَ فِيهِ المِلْحُ أَسْهَلَ بِقُوَّةٍ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏6/‏367‏.
ن405هـ=1014م

المُجَبَّنُ مِنَ الأَشْخَاصِ: المُتَّهَمُ بِالجُبْنِ.

وَجَبُنْتُ غيْرَ مُجَبَّنٍ فِي مَاقِطٍ عَنْ ذِكْرِ حَالِي لِلْوَزِيرِ الخِضْرِمِ

ن200هـ=815م

تَجْبِينُ اللَّبْنِ: تَصْيِيرُهُ جُبْنًا.

"وَلِبَعْضِ الحَيَوَانِ لَبَنٌ كَثِيرٌ، أَكْثَرُ مِمَّا يَحْتَاجُ لِرَضَاعِ الوَلَدِ، وَهُوَ مُوَافِقٌ لِتَهْيِئَةِ الجُبْنِ مِنْهُ... وَمِنْ لَبَنِ البَقَرِ يُهَيَّأُ أَكْثَرَ مِمَّا يُهَيَّأُ مِنْ لَبَنِ المِعْزَى، قَدْرَ مَرَّةٍ وَنِصْفٍ. وَأَمَّا سَائِرُ الأَلْبَانِ فَلَيْسَتْ بِمُوَافِقَةٍ لِشَيْءٍ مِنَ التَّجْبِينِ ، وَلَا سِيَّمَا لَبَنُ الحَيَوانِ الَّذِي لَهُ أَثْدَاءٌ كَثِيرَةٌ"

مَا زِلْتَ تَكْلَؤُهُمْ بِعَيْنِ نَصِيحَةٍ وَتَبِيتُ لِلْفِكْرِ الطَّوِيلِ مُثَافِنَا
مُتَحَرِّزًا، حَتَّى لَخَالَكَ خَائِلٌ رَجُلًا شَدِيدَ الجُبْنِ، أَوْ مُتَجَابِنَا
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏6/‏2596 ‏-‏ 2597‏.

جُبْنِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن288هـ=901م

جُبْنِيَّةُ اللَّبَنِ: صَيْرُورَتُهُ كَالجُبْنِ فِي التَّجَمُّدِ.

"إِنَّ اللَّبَنَ، وَإِنْ كَانَ بَسِيطًا عِنْدَ الحِسِّ؛ فَإِنَّهُ مُرَكَّبٌ مِنْ جَوَاهِرَ مُخْتَلِفَةِ الطَّبَايِعِ، مُتَضَادَّةِ الأَفْعَالِ، مِثْلِ شَيْءٍ يُطْلِقُ البَطْنَ، وَيُلَطِّفُ الأَخْلَاطَ الغَلِيظَةَ، وَيُنَقِّي مَجَارِيَ الكَبِدِ فِي مَائِيَّتِهِ، وَشَيْءٍ يَعْقِلُ البَطْنَ، وَيُغَلِّظُ، وَيُحْدِثُ سَدَدًا فِي مَجَارِي الكَبِدِ، وَالحَصَاةِ فِي الكُلَى وَالمَثَانَةِ، وَذَلِكَ جُبْنِيَّتُهُ ، وَشَيْءٍ يُلَيِّنُ الأَعْضَاءَ، وَيُرْخِيهَا، وَذَلِكَ زُبْدِيَّتُهُ"
ثابت بن قُرّة الحَرّانيّ
كتاب الذّخيرة في علم الطّبّ: ثابت بن قُرّة الحَرّانيّ (ت، 288هـ)، تحقيق: جورج صبحي، المطبعة الأميريّة، القاهرة، (1928م)، ‏172‏.

جَبَانَة[مَصْدَر]

324هـ=936م

"وَإِنْ كَانَ المَرِّيخُ رَدِيءَ الحَالِ فِي التَّغْرِيبِ وَالرُّجُوعِ وَالسُّقُوطِ، وَهُوَ المُنْفَرِدُ بِالدَّلَالَةِ فِي ابْتِدَاءِ تَدْبِيرِهِ عَلَى بَعْضِ المُدُنِ؛ فَإِنَّهُ يَظْهَرُ فِي أَهْلِهَا الدِّيَانَاتُ وَالسُّنَنُ وَالشَّرَائِعُ الرَّدِيئَةِ المُخْتَلِفَةِ، وَقِلَّةُ الوَزَعِ وَكَثْرَةُ الجَبَانَةِ وَاللُّصُوصِيَّةِ"
موسى بن الحسن النّوبَخْتيّ
كتاب الأزمنة والدّهور: موسى بن الحسن النّوبَخْتيّ (ت، 330هـ)، تحقيق: آنا لابارتا، جامعة فلانسيا، فلانسيا، (2005م)، ‏68‏.

"كَأَنَّهُم إِذَا نَاقَشُوا ابْتِدَاءً رُمِقُوا بِعَيْنِ الِاسْتِجْبَانِ ، وَظُنَّ بِهِمْ أَنَّ الِاخْتِنَاقَ يُلْجِئُهُمْ إِلَى المُعَاسَرَةِ"

مُجَبَّنَة[اسْم]

ن638هـ=1240م

المُجَبَّنَةُ : نَوْعٌ مِنَ الأَطْعِمَةِ، تُشْبِهُ الإِسْفِنْجَ، تُؤْكَلُ سَاخِنَةً فِي الصَّبَاحِ، شَاعَتْ فِي المَغْرِبِ وَالأَنْدَلُسِ.

"وَيُصْنَعُ مِنْ بَاقِي العَجِينَ أَمْثَالُ مُجَبَّنَاتٍ صِغَارٍ مُجَوَّفَةٍ بِأَغْطِيَةٍ، وَتُقْلَى فِي زَيْتٍ عَذْبٍ بَعْضَ قَلْيٍ هِيَ وَأَغْطِيَتُهَا"

جَبِينِيّ[اسْمٌ مَنْسُوب]

ن1192هـ=1777م

"وَأَمَّا الشُّعْبَةُ الأُخْرَى فَغَلِيظَةٌ مُسْتَدِيرَةُ الرَّأْسِ، مُرَكَّبَةٌ فِي نُقْرَةٍ تَحْتَ الزَّائِدَةِ الشَّبِيهَةِ بِحَلْمَتَيِ الثَّدْيِ فِي العَظْمِ الجَبِينِيِّ ، وَبِهَذَا المِفْصَلِ تَكُونُ حَرَكَةُ اللَّحْيِ الأَسْفَلِ"
أحمد بن عبد المنعم بن يوسف الدمنهوري
القول الصريح في علم التشريح: أحمد بن عبد المنعم بن يوسف الدمنهوري (ت، 1192هـ)، تحقيق ودراسة: وليد الإمام مبارك، إشراف ومراجعة: أحمد فؤاد باشا، دار الكتب والوثائق القومية، القاهرة، (1439ه/ 2018م)، ‏119‏.
