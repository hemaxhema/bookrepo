أَشْقَح[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الأَشْقَحُ : الضَّارِبُ إِلَى الحُمْرَةِ.

وَأَشْقَحَ مِنْ رَاحِ العِرَاقِ مُمَلَّأٍ مَخِيطٍ عَلَيْهِ الخَيْشُ، جَلْدٍ مَرَائِرُهْ
سَبَقْتُ بِهِ طَلْقًا يَرَاحُ إِلَى النَّدَى إِذَا مَا انْتَشَى لَمْ تَحْتَضِرْهُ مَفَاقِرُهْ
الزُّبير بن عبد المُطَّلِب القُرَشيّ
كتاب حذف من نسب قريش: مؤرِّج السَّدوسيّ (ت، 195هـ)، نشره: صلاح الدين المنجد، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1960هـ)، ‏21‏-‏ 22‏.

أَشْقَح[صِفَة]

ن351هـ=962م

مَا أَشْقَحَهُ : تَعَجُّبٌ مِنْ شِدَّةِ شَقَاحَتِهِ.

"وَيَقُولُونَ: قُبْحًا لَهُ وَشُقْحًا... وَمَا أَقْبَحَهُ وَأَشْقَحَهُ ! "
أبو الطّيّب اللُّغويّ الحلبيّ
كتاب الإتباع: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، حققه وشرحه وقدم له: عز الدين التنوخي، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏56‏.

أَشْقَح[اسْمُ تَفْضِيل]

ن385هـ=995م

الأَشْقَحُ : الأَشَدُّ بالظُّلْمِ وَنَحْوِهِ.

"بَلْ أَنْتَ أَشْقَحُ وَأَلْقَحُ بِالظُّلْمِ"
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏2/‏333‏.

مَشْقُوح[اسْمُ مَفْعُول]

4هـ=625م

المَشْقُوحُ : المُبْعَدُ.

"أَيْنَ هَذِهِ المَشْقُوحَةُ المَقْبُوحَةُ الَّتِي قَدْ آذَيْتِ بِهَا رَسُولَ اللهِ #  ؟"
عمّار بن ياسر العَنْسيّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏44/‏151،،‏(رقم الحديث: 26529).
ن37هـ=657م

المَشْقُوحُ : المَكْسُورُ.

"أَأَنْتَ تَسُبُّ حَبِيبَةَ رَسُولِ اللهِ @ ؟ اقْعُدْ مَنْبُوحًا مَقْبُوحًا مَشْقُوحًا "
عمّار بن ياسر العَنْسيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏22‏.

شَقْحِيّ[صِفَة]

ن43هـ=663م

الشَّقْحِيُّ مِنَ الثِّيَابِ وَنَحْوِهَا: الأَحْمَرُ.

"أُتِيَ بِحُيَيِّ بْنِ أَخْطَبَ مَجْمُوعَةً يَدَاهُ إلَى عُنُقِهِ، عَلَيْهِ حُلَّةٌ شَقْحِيَّةٌ قَدْ لَبِسَهَا لِلْقَتْلِ"
محمّد بن مسلمة الحارثيّ الأنصاريّ
كتاب المغازي: الواقديّ (ت، 207هـ)، تحقيق: مارسدن جونس، عالم الكتب، القاهرة، ط3، (1404هـ/ 1984م)، ‏2/‏513‏.

شَقَّحَ[لَازِم]

ن75هـ=694م

شَقَّحَ الأَرَاكُ وَنَحْوُهُ: تَلَوَّنَ ثَمَرُهُ بِالحُمْرَةِ، أَوِ الصُّفْرَةِ.

كِنَانِيَّةٌ، أَوْتَادُ أَطْنَابِ بَيْتِهَا أَرَاكٌ، إِذَا صَافَتْ بِهِ المَرْدُ شَقَّحَا
عمرو بن أحمر الباهليّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏10/‏50‏.

أَشْقَحَ[لَازِم]

ن78هـ=697م

( أَشْقَهَ : لُغة في: أَشْقَحَ).

"نَهَى عَنِ المُحَاقَلَةِ، وَالمُزَابَنَةِ، وَالمُخَابَرَةِ، وَأَنْ تُشْتَرَى النَّخْلُ حَتَّى تُشْقِهَ "
جابر بن عبد الله الخزرجيّ الأنصاريّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏3/‏1175،،‏(رقم الحديث: 1536).
ن78هـ=697م

أَشْقَحَ البُسْرُ وَنَحْوُهُ: تَغَيَّرَ لَوْنُهُ إِلَى الحُمْرَةِ، أَوِ الصُّفْرَةِ.

"نَهَى رَسُولُ اللهِ @  عَنْ بَيْعِ الثَّمَرَةِ حَتَّى تُشْقِحَ "
جابر بن عبد الله الخزرجيّ الأنصاريّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏23/‏165،،‏(رقم الحديث: 14884).
ن105هـ=723م

أَشْقِحْ بِهِ : تَعَجُّبٌ مِنْ شِدَّةِ شَقَاحَتِهِ.

أَقْبِحْ بِهِ مِنْ وَلَدٍ، وَأَشْقِحِ ! مِثْلِ جُرَيِّ الكَلْبِ لَمْ يُفَقِّحِ
الأحوص الأوسيّ الأنصاريّ
شعر الأحوص الأنصاريّ: جمعه وحققه: عادل سليمان جمال، قدم له: شوقي ضيف، مكتبة الخانجي- مطبعة المدني، القاهرة، ط2، (1411هـ/ 1990م)، ‏271‏.

شُقْحَة[اسْم]

ن78هـ=697م

الشُّقْحَةُ : البُسْرَةُ الَّتِي تَغَيَّرَ لَوْنُهَا إِلَى الحُمْرَةِ، أَوِ الصُّفْرَةِ.

"ثُمَّ عَمَدْتُ إِلَى جُمَّاعٍ مِنَ التَّمْرِ مِثْلِ: نُخْبَةٍ، وَقَرْنٍ، وَشُقْحَةٍ ، وَغَيْرِهَا مِنَ الأَنْوَاعِ، وَهُوَ أَقَلُّ التَّمْرِ، فَجَعَلْتُهُ حَبْلًا وَاحِدًا"
جابر بن عبد الله الخزرجيّ الأنصاريّ
كتاب المغازي: الواقديّ (ت، 207هـ)، تحقيق: مارسدن جونس، عالم الكتب، القاهرة، ط3، (1404هـ/ 1984م)، ‏1/‏401‏.

مُشْقِح[اسْمُ فَاعِل]

ن120هـ=738م

المُشْقِحُ مِنَ النَّخْلِ وَنَحْوِهِ: المُتَغَيِّرُ لَوْنُ ثَمَرِهِ إِلَى الحُمْرَةِ، أَوِ الصُّفْرَةِ.

إِذَا ذَكَرَ النَّخْلَ أَرْبَابُهَا وَقَالُوا: مُبَكِّرُهَا المُبْلِحُ
تَعَجَّلَ عَنْ جَرْيَةِ المَاذِيَانِ فَنَوَّرَ، أَوْ بَعْضُهُ المُشْقِحُ
عبد الله بن عمر العَرْجيّ
ديوان العَرجي: رواية: أبي الفتح ابن جنّي (ت، 392هـ)، شرحه وحققه: خضر الطائي، رشيد العبيدي، الشركة الإسلامية، بغداد، ط1، (1375هـ/ 1956م)، ‏114‏.

"وَالسَّدَى مِنَ البَلَحِ يُقَالُ لَهُ: الرِّمَخُ... وَهُوَ أَخْضَرُ بَعْدُ. فَإِذَا اخْضَرَّ وَتَلَوَّنَ قَلِيلًا قِيلَ: قَدْ تَشَقَّحَ "

كتاب النّخْلة: أبو حاتم السِّجِسْتانيّ (ت، 255هـ)، تحقيق: حاتم الضامن، دار البشائر الإسلامية، بيروت، ط1، (1422هـ/ 2002م)، ‏71‏.

شُقَّاح[اسْمُ جِنْس]

ن275هـ=888م

الشُّقَّاحُ : نَبَاتٌ جَبَلِيٌّ لَاذِعُ الطَّعْمِ، يُشْبِهُ الكَبَرَ، تَنْبُتُ فِي أُصُولِهِ الكَمْأَةُ.

"شَمَنْصِيرٌ: وَهُوَ جَبَلٌ مُلَمْلَمٌ... وَيُقَالُ: إِنَّ أَكْثَرَ نَبَاتِهِ النَّبْعُ وَالشَّوْحَطُ، وَالمِيَاهُ حَوَالَيْهِ يَنَابِيعُ عَلَيْهَا النَّخِيلُ وَالحَمَاطُ، وَفِي كُلِّ جِبَالِ تِهَامَةَ الشُّقَّاحُ "
عَرّام بن الأصبغ السُّلَميّ
كتاب أسماء جبال تهامة وسكّانها وما فيها من القرى وما ينبت عليها من الأشجار وما فيها من المياه: عَرّام بن الأصبغ السُّلَميّ (ت، 275هـ)، تحقيق: عبد السلام هارون، عني بنشره: يوسف زينل، محمّد نصيف، مطبعة أمين عبد الرحمن، القاهرة، ط1، (1373هـ)، ‏26‏.
