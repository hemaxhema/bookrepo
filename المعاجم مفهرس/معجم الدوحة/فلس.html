مُفْلِس[اسْمُ فَاعِل]

قبل 150ق.هـ=476م

المُفْلِسُ : الفَاقِدُ مَا كَانَ يَمْلِكُ.

وَلَيْسَ الفَتَى، عِنْدِي، بِشَيْءٍ أَعُدُّهُ إِذَا كَانَ ذَا مَالٍ، مِنَ العَقْلِ مُفْلِسُ
منسوب إلى هند بنت الخُسّ الإياديّة
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏63‏.
قبل 100ق.هـ=525م

المُفْلِسُ مِنَ الأُمُورِ: الخَالِي مِنَ النَّفْعِ وَالفَائِدَةِ.

يَا حَبَّ، مَا حُبُّ القَتُولِ؟ وَحُبُّهَا فَلَسٌ، فَلاَ يُنْصِبْكَ حُبٌّ مُفْلِسٌ
أبو قِلابة الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏715‏.
ن11هـ=632م

المُفْلِسُ : الَّذِي اجْتَمَعَتْ عَلَيْهِ دُيُونٌ لَا يَفِي بِهَا مَالُهُ. [الفقه]

"إِذَا أَفْلَسَ الرَّجُلُ، فَوَجَدَ غَرِيمُهُ مَتَاعَهُ عِنْدَ المُفْلِسِ بِعَيْنِهِ، فَهُوَ أَحَقُّ بِهِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏14/‏544‏.

فَلَس[صِفَةٌ مُشَبَّهَة]

قبل 100ق.هـ=525م

الفَلَسُ مِنَ الأُمُورِ: العَدِيمُ الفَائِدَةِ.

يَا حِبُّ، مَا حُبُّ القَتُولِ؟ وَحُبُّهَا فَلَسٌ ، فَلَا يُنْصِبْكَ حُبٌّ مُفْلِسُ
أبو قِلابة الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏3/‏32‏.

فَلَس[مَصْدَر]

ن179هـ=795م

الفَلَسُ : الخَسَارَةُ وَذَهَابُ المَالِ.

"وَإِنْ تَكَارَى إِبِلًا فَحَمَلَ عَلَيْهَا مَتَاعًا، أَوْ دَفَعَ إِلَى صَابِغٍ مَتَاعًا يَصْبِغُهُ، أَوْ يَخِيطُهُ، أَوْ يَغْسِلُهُ، كَانَ المُكْرِي أَوِ الصَّبَّاغُ أَوْلَى بِمَا فِي أَيْدِيهِمْ، فِي الفَلَسِ وَالمَوْتِ، مِنَ الغُرَمَاءِ"
مالك بن أنس الأصبحيّ الحِمْيريّ
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏4/‏559‏.

فِلْس[اسْم]

قبل 13ق.هـ=609م

الفِلْسُ : صَنَمٌ كَانَ يُعْبَدُ فِي الجَاهِلِيَّةِ بِأَرْضِ طَيِّئٍ.

فَأُقْسِمُ بِالعَتَائِرِ، حَيْثُ فِلْسٌ وَمَنْ نَسَكَ الأُقَيْصِرَ مِ العِبَادِ
لَقَدْ حُزْتَ الكَهَانَةَ عَنْ سَطِيحٍ وَشِقٍّ، وَالمُرَفَّلِ، مِنْ إِيَادِ
عارق الطّائيّ
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏2/‏322‏.

أَفْلَسَ[لَازِم]

ن11هـ=632م

أَفْلَسَ الشَّخْصُ: ذَهَبَ مَالُهُ.

"مَنْ وَجَدَ عَيْنَ مَالِهِ عِنْدَ رَجُلٍ قَدْ أَفْلَسَ فَهُوَ أَحَقُّ بِهِ مِمَّنْ سِوَاهُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏12/‏21،،‏(رقم الحديث: 7124).

أَفْلَسَ[مُتَعَدٍّ]

ن179هـ=795م

أَفْلَسَهُ الغُرَمَاءُ: طَلَبُوا مِنَ القَاضِي الحُكْمَ بِإِفْلَاسِهِ.

"لَوْ أَنَّ رَجُلًا أَفْلَسَهُ رَجُلَانِ، لِكُلِّ وَاحِدٍ مِنْهُمَا مِئَةُ دِرْهَمٍ، وَلِرَجُلٍ غَائِبٍ عَلَيْهِ مِئَةُ دِرْهَمٍ أَيْضًا، وَلَمْ يُعْلَمْ بِالغَائِبِ، فَفَلَّسُوا هَذَا الغَرِيمَ، فَلَمْ يَجِدُوا لَهُ إِلَّا مِئَةَ دِرْهَمٍ، فَقُسِّمَتِ المِئَةُ بَيْنَ هَذَيْنِ الرَّجُلَيْنِ... ثُمَّ قَدِمَ الغَائِبُ، وَأَثْبَتَ دَيْنَهُ، فَإِنَّهُ يَصِيرُ لَهُ فِي المُحَاصَّةِ مِنَ المِئَةِ ثَلَاثَةٌ وَثَلَاثُونَ وَثُلُثٌ"
مالك بن أنس الأصبحيّ الحِمْيريّ
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏5/‏229‏-‏ 230‏.
ن180هـ=796م

أَفْلَسَهُ : أَذْهَبَ مَالَهُ، وَصَيَّرَهُ إِلَى حَالِ الإِفْلَاسِ.

الجُودُ أَفْلَسَهُمْ ، وَأَذْهَبَ مَالَهُمْ فَاليَوْمَ، إِنْ رَامُوا السَّمَاحَةَ، يَبْخَلُوا
أبو الشَّمَقْمَق البَصْريّ
ديوان أبي الشمقمق: جمعه وحققه وشرحه: واضح محمّد الصمد، دار الكتب العلمية، بيروت، ط1، (1415هـ/ 1995م)، ‏82‏.
389هـ=999م

أَفْلَسَهُ المَالَ وَنَحْوَهُ: جَعَلَهُ يَخْسَرُهُ.

ثُمَّ بَرَّزْتَ تُفْلِسُ السُّبَقَ الجَرْ يَةَ، لَا تَابِعًا، وَلَا مَتْبُوعَا
ابن نُباتة السَّعْديّ
ديوان ابن نُباتة السَّعْديّ أبي نصر عبد العزيز بن عمر بن نُباتة السَّعْديّ (ت، 405هـ): دراسة وتحقيق: عبد الأمير مهدي الطائي، دار الحرية، بغداد، (1397هـ/ 1977م)، ‏2/‏326‏.

إِفْلَاس[مَصْدَر]

ن11هـ=632م

الإِفْلَاسُ : ذَهَابُ المَالِ.

"مَنِ احْتَكَرَ عَلَى المُسْلِمِينَ طَعَامَهُمْ ضَرَبَهُ اللهُ بِالإِفْلَاسِ أَوْ بِجُذَامٍ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏1/‏284،،‏(رقم الحديث: 135).
ن179هـ=795م

الإِفْلَاسُ : أَنْ يَكُونَ الدَّيْنُ الَّذِي عَلَى الشَّخْصِ أَكْثَرَ مِنْ مَالِهِ. [الفقه]

"وَلَوْ دَايَنَهُ قَوْمٌ آخَرُونَ بَعْدَ إفْلَاسِهِ لَمْ يَكُنْ لِلْغَائِبِ فِي مَالِهِ قَلِيلٌ وَلَا كَثِيرٌ، وَإِنَّمَا يَتْبَعُ الأَوَّلِينَ الَّذِينَ فَلَّسُوهُ، وَقَسَّمَ لَهُمْ مَالَهُ، وَكَذَلِكَ السَّارِقُ"
مالك بن أنس الأصبحيّ الحِمْيريّ
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏6/‏290‏.

إِفْلَاس[اسْم]

ن1293هـ=1875م

الإِفْلَاسُ : مَا يُتَّخَذُ مِنْ إِجْرَاءَاتٍ تَضْمَنُ حُقُوقَ الدَّائِنِينَ، وَتَعْمَلُ عَلَى إِعَادَةِ دُيُونِهِمْ بِشَكْلٍ عَادِلٍ، دُونَ أَنْ تَمِيلَ كَفَّةُ الِميزَانِ لِطَرَفٍ عَلَى حِسَابِ طَرَفٍ آخَرَ. [القانون]

"إِنَّ كَيْفِيَّةَ الإِفْلَاسِ تُعْلَنُ، وَتُجْرَى مُؤَقَّتًا، حَسَبَ مَنْطُوقِ الحُكْمِ وَالإِعْلَامِ الَّذِي يُعْطَى؛ إِمَّا بِنَاءً عَلَى إِنْهَاءِ المُفْلِسِ، أَوْ عَلَى اسْتِدْعَاءِ وَاحِدٍ أَوْ أَكْثَرَ مِنْ أَصْحَابِ الدُّيُونِ"
نقولا النقاش
قانون التجارة: نقولا النقاش (ت، 1312هـ)، ترجمة: نقولا النقاش، المطبعة العمومية، بيروت، (1297ه/ 1880م)، ‏1/‏29‏.

أَفْلَس[اسْمُ تَفْضِيل]

ن13هـ=634م

الأَفْلَسُ : الأَكْثَرُ إِفْلَاسًا.

فَقَوْمٌ، إِلَى نَارِ الجَحِيمِ، مَصِيرُهُمْ بِإِفْلَاسِهِمْ، وَالعَابِدُ الصَّخْرَ أَفْلَسُ
أبو بكرٍ الصِّدّيق
ديوان أبي بكر الصديق رضي الله عنه: حققه وشرحه: محمّد شفيق البيطار، شراع، دمشق، ط1، (1993م)، ‏91‏.

تَفْلِيس[مَصْدَر]

125هـ=743م

التَّفْلِيسُ : ذَهَابُ المَالِ بِسَبَبٍ مَا.

أَمَرْتَ نَفْسًا، تَكْرُمُ النُّفُوسَا لَيْسَتْ لِخَبٍّ، يَرْهَبُ التَّفْلِيسَا وَلَا لِنِكْسٍ، يَعْمُرُ التَّنْكِيسَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏236‏-‏237‏.
ن175هـ=791م

تَفْلِيسُ المَدِينِ: حُكْمُ القَاضِي بِإِفْلَاسِهِ، بِحَيْثُ يُمْنَعُ مِنَ التَّصَرُّفِ فِي مَالِهِ. [الفقه]

"وَقَدْ فَلَّسَهُ الحَاكِمُ تَفْلِيسًا "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏260‏.
1301هـ=1883م

التَّفْلِيسُ : إِجْرَاءٌ قَانُونِيٌّ هَدَفُهُ تَسْوِيَةُ الدُّيُونِ بِشَكْلٍ مُنَظَّمٍ حِمَايَةً لِلدَّائِنِينَ. [القانون]

"إِذَا عُقِدَتْ، وَتَمَّتْ بَعْدَ اليَوْمِ الَّذِي يَكُونُ خُصِّصَ مِنْ طَرَفِ مَحْكَمَةِ التِّجَارَةِ؛ لِيَكُونَ بِدَايَةَ التَّفْلِيسِ ، أَوْ قَبْلَ ذَلِكَ اليَوْمِ بِعَشْرَةِ أَيَّامٍ؛ تَكُونُ سَاقِطَةً وَغَيْرَ مُعْتَبَرَةٍ لَدَى المَاسَّةِ، وَتُحْسَبُ كَأَنَّهَا لَمْ تَكُنْ"
نوفل بن نعمة الله الطرابلسي
الدستور: نوفل بن نعمة الله الطرابلسي (ت، 1304هـ)، ترجمة: نوفل بن نعمة الله الطرابلسي، مراجعة وتدقيق: خليل أفندي الخوري، المطبعة الأدبية، بيروت، (1301ه/ 1883م)، ‏1/‏215‏.
