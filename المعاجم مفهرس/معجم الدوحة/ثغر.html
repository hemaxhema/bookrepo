ثَغْر[اسْم]

قبل 150ق.هـ=476م

الثَّغْرُ : النَّاحِيَةُ المَخُوفَةُ مِنْ أَطْرَافِ البِلَادِ.

وَقَدْ أَظَلَّكُمُ مِنْ شَطْرِ ثَغْرِكُمُ هَوْلٌ لَهُ ظُلَمٌ تَغْشَاكُمُ قِطَعَا
لَقيط بن يَعْمَر الإياديّ
ديوان لقيط بن يَعمر: حققه وقدّم له: عبد المعيد خان، دار الأمانة- مؤسّسة الرسالة، بيروت، (1391هـ/ 1971م)، ‏43‏.
قبل 50ق.هـ=573م

الثَّغْرُ : الفَمُ، وَمَا تَقَدَّمَ مِنَ الأَسْنَانِ.

لَهَا كَفَلٌ كَالدِّعْصِ لَبَّدَهُ النَّدَى وَثَغْرٌ نَقِيٌّ كَالأَقَاحِي المُنَوِّرِ
أمّ النُّحيف، وهو سعد بن قُرْط أحد بني جَذيمة
شعراء عبد القيس وشعرهم في العصر الجاهلي: جمع وتحقيق ودراسة: عبد الحميد المعيني، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري، الكويت، (2002م)، ‏430‏.
قبل 13ق.هـ=609م

الثَّغْرُ : المَنْفَذُ وَالفُرْجَةُ فِي الشَّيْءِ.

فَشَكَّ بِالرَّوْقِ مِنْهُ صَدْرَ أَوَّلِهَا شَكَّ المُشَاعِبِ أَعْشَارًا بِأَعْشَارِ
ثُمَّ انْثَنَى، بَعْدُ، لِلثَّانِي فَأَقْصَدَهُ بِذَاتِ ثَغْرٍ بَعِيدِ القَعْرِ نَعَّارِ
النّابغة الذُّبْيانيّ
ديوان النابغة الذبياني: جمعه وشرحه وكمّله وعلّق عليه: محمّد الطاهر ابن عاشور، الشركة التونسية، تونس، الشركة الوطنية، الجزائر، (1976م)، ‏152‏-‏ 153‏.

ثَغْر[صِفَةٌ مُشَبَّهَة]

ن5هـ=626م

الثَّغْرُ مِنَ النَّاسِ وَنَحْوِهِمْ: المُتَفَرِّقُ الضَّائِعُ.

سِنْفَةٌ أُرْسِلَتْ تُخَبِّرُ عَنْهُمْ أَهْلَ قُرْحٍ بِهَا قَدَ امْسَوْا ثُغُورَا
أُمَيّة بن أبي الصَّلْت الثَّقَفيّ
ديوان أمية بن أبي الصلت: صنعة: عبد الحفيظ السطلي، المطبعة التعاونية، دمشق، (1974م)، ‏407‏.

ثَغْر[اسْمُ جِنْس]

86هـ=705م

الثَّغْرُ : نَبَاتٌ بَرِّيٌّ مِنَ العُشْبِ الحَارِّ، زَهْرُهُ أَبْيَضُ، تَأْكُلُهُ الإِبِلُ.

إِذَا ذَكَرَتْهَا النَّفْسُ ظَلَّتْ كَأَنَّمَا عَلَيْهَا مِنَ الوَرْدِ التِّهَامِيِّ أَفْكَلُ
وَفَاضَتْ دُمُوعُ العَيْنِ حَتَّى كَأَنَّمَا بِوَادِي القُرَى مِنْ يَابِسِ الثَّغْرِ تُكْحَلُ
كُثَيِّر عَزّة
ديوان كُثَيِّر عَزَّة: جمعه وشرحه: إحسان عبّاس، دار الثقافة، بيروت، (1391هـ/ 1971م)، ‏254‏.

ثُغْرَة[اسْم]

قبل 50ق.هـ=573م

الثُّغْرَةُ : مَوْضِعُ المَخَافَةِ مِنْ أَطْرَافِ البِلَادِ.

وَيْلُمِّهِ رَجُلًا تَأْبَى بِهِ غَبَنًا إِذَا تَجَرَّدَ لَا خَالٌ وَلَا بَخَلُ
السَّالِكُ الثُّغْرَةَ اليَقْظَانَ كَالِئُهَا مَشْيَ الهَلُوكِ عَلَيْهَا الخَيْعَلُ الفُضُلُ
المُتَنَخِّل الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏34‏.
قبل 13ق.هـ=609م

ثُغْرَةُ النَّحْرِ: النُّقْرَةُ بَيْنَ التَّرْقُوَتَيْنِ بِأَسْفَلِ العُنُقِ.

يَدْعُونَ عَنْتَرَ وَالرِّمَاحُ كَأَنَّهَا أَشْطَانُ بِئْرٍ فِي لَبَانِ الأَدْهَمِ
مَا زِلْتُ أَرْمِيهِمْ بِثُغْرَةِ نَحْرِهِ وَلَبَانِهِ حَتَّى تَسَرْبَلَ بِالدَّمِ
عنترة
ديوان عنترة: تحقيق ودراسة: محمّد سعيد مولوي، المكتب الإسلامي، بيروت، (1390هـ/ 1970م)، ‏216‏-‏ 217‏.
8هـ=629م

ثُغْرَةُ الشَّيْءِ: طَرِيقُهُ المَسْلُوكَةُ.

عَوْدُ الرِّيَاسَةِ شَامِخٌ عِرْنِينُهُ مُتَطَلِّعٌ ثُغَرَ المَكَارِمِ خِضْرِمُ
العبّاس بن مرداس السُّلَميّ
ديوان العباس بن مرداس السَّلمي: جمعه وحققه: يحيى الجبوري، مؤسسة الرسالة، بيروت، ط1، (1412هـ/ 1991م)، ‏139‏.
ن175هـ=791م

الثُّغْرَةُ : النَّاحِيَةُ مِنَ الأَرْضِ.

" وَالثُّغْرَةُ: النَّاحِيَةُ مِنَ الأَرْضِ، يُقَالُ: مَا فِي تِلْكَ الثُّغْرَةِ مِثْلُ فُلَانٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏400‏.
ن231هـ=845م

ثُغْرَةُ الشَّيْءِ: الثُّلْمَةُ والفُرْجَةُ فِيهِ.

شَابَ رَأْسِي، وَمَا رَأَيْتُ مَشِيبَ الرْ رَأْسِ إِلَّا مِنْ فَضْلِ شَيْبِ الفُؤَادِ
نَالَ رَأْسِي مِنْ ثُغْرَةِ الهَمِّ مَا لَمْ يَسْتَنِلْهُ مِنْ ثُغْرَةِ المِيلَادِ
أبو تمّام
شرح ديوان أبي تمّام: الخطيب التِّبْريزي (ت، 502هـ)، قدم له ووضع هوامشه وفهارسه: راجي الأسمر، دار الكتاب العربي، بيروت، ط2، (1414هـ/ 1994م)، ‏1/‏191‏.

مُثْغِر[صِفَة]

قبل 50ق.هـ=573م

المُثْغِرُ مِنَ الطَّعْنَاتِ وَنَحْوِهَا: الفَاغِرُةُ النَّافِذَةُ.

فَهَتَكْتُ بِالرُّمْحِ الطَّوِيلِ إِهَابَهُ فَهَوَى صَرِيعًا لِلْيَدَيْنِ وَلِلْفَمِ
وَمَنَحْتُ آخَرَ بَعْدَهُ جَيَّاشَةً نَجْلَاءَ مُثْغِرَةً ، كَشِدْقِ الأَضْجَمِ
ربيعة بن مُكَدَّم الفِراسيّ الكِنانيّ
نشوة الطّرب في تاريخ جاهليّة العرب: ابن سعيد المغربيّ (ت، 685هـ)، تحقيق: نصرت عبد الرَّحمن، مكتبة الأَقصى، عمّان، ط1، (1982م)، ‏377‏-‏ 378‏.

مُثْغِر[صِفَةٌ مُشَبَّهَة]

460هـ=1068م

المُثْغِرُ : الَّذِي نَبَتَتْ أَسْنَانُهُ بَعْدَ سُقُوطِهَا.

"إِنْ ثَبَتَ أَنَّهُ بَاعَهَا وَهِيَ غَيْرُ مُثْغِرَةٍ ، وَأَبَتِ الأُمُّ مِنَ الخُرُوجِ فِي طَلَبِهَا؛ فَعَلَى البَائِعِ الخُرُوجُ فِيهَا وَاسْتِرْجَاعُهَا إِلَى مَوْضِعِ الأُمِّ"
عيسى بن سهل الأسديّ الجيّانيّ
ديوان الأحكام الكبرى أو الإعلام بنوازل الأحكام وقِطر من سِير الحكّام: عيسى بن سهل الأسديّ الجيّانيّ (ت، 486هـ)، تحقيق: يحيى مراد، دار الحديث، القاهرة، (1428هـ/ 2007م)، ‏319‏.

مُثْغِر[اسْمُ فَاعِل]

ن861هـ=1456م

المُثْغِرُ مِنَ الأَسْنَانِ: النَّابِتُ.

"وَكَانَتْ لَهُ عَجَائِبُ... وَمِنْهَا أَنَّهُ وُلِدَ فَبَقِيَتْ أَسْنَانُهُ مُثْغِرَةٌ ، فَمَاتَ وَلَمْ يُثْغَرْ"
يحيى بن أبي بكر العامريّ الحَرَضيّ
غربال الزمان في وفيات الأعيان: يحيى بن أبي بكر العامريّ الحَرَضيّ (ت، 893هـ)، صححه وعلق عليه: محمد ناجي زعبي العمر، إشراف: عبد الرحمن بن يحيى الإرياني، مطبعة زيد بن ثابت، دمشق، (1405ه/ 1985م)، ‏170‏.

ثَاغِر[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

الثَّاغِرُ مِنَ الأَكْبَادِ وَنَحْوِهَا: المَكْلُومُ المَثْلُومُ.

قَوْمٌ يَهُودٌ، قَدْ رَأَوْا مَا قَدْ رَأَوْا ظِلَّ الغَمَامَةِ ثَاغِرِي الأَكْبَادِ
أبو طالب
ديوان أبي طالب عم النبي صلى الله عليه وسلم: جمعه وشرحه: محمّد ألتونجي، دار الكتاب العربي، بيروت، ط1، (1414هـ/ 1994م)، ‏39‏.
125هـ=743م

الثَّاغِرُ : الكَاسِرُ الثَّغْرَ.

يَلْقَوْنَ تَعْوِيرًا وَصَكًّا بَادِرَا مِنَّا إِذَا الشَّرُّ اكْتَسَى الأَنَامِرَا وَالجُرْدَ يَعْلُكْنَ الشَّكِيمَ الثَّاغِرَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏172‏.

ثَاغِر[صِفَةٌ مُشَبَّهَة]

ن126هـ=744م

الثَّاغِرُ : ذُو الأَسْنَانِ التَّامَّةِ الخَلْقِ.

وَإِذَا ضَحِكْنَ عَنِ العِذَا بِ لَنَا المُسَفَّاتِ الثَّوَاغِرْ
كَانَ التَّهَلُّلُ بِالتَّبَسْـ ـسُمِ لَا القَهَاقِهُ بِالقَرَاقِرْ
الكميت بن زيد الأسديّ
ديوان الكُمَيْت بن زيد الأسديّ: جمع وشرح وتحقيق: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (2000م)، ‏136‏.

ثُغْرُور[اسْم]

ن5هـ=626م

الثُّغْرُورُ : الفَمُ وَمَا تَقَدَّمَ مِنَ الأَسْنَانِ.

وَأَبْدَتِ الثُّغْرُورَا
أُمَيّة بن أبي الصَّلْت الثَّقَفيّ
الشّعر والشّعراء: ابن قتيبة (ت، 276هـ)، تحقيق وشرح: أحمد محمد شاكر، دار المعارف، القاهرة، (1982م)، ‏461‏.

ثُغْر[اسْم]

ن7هـ=628م

الثُّغْرُ مِنَ النَّحْرِ: النُّقْرَةُ الَّتِي بَيْنَ التَّرْقُوَتَيْنِ أَسْفَلَ الرَّقَبَةِ.

فَلَسْتُ أُسْلِمُ أَوْسًا أَوْ أَمُوتَ إِذًا حَتَّى أُرَدَّ، وَثُغْرُ النَّحْرِ مَبْلُولُ
المُثَلَّم بن حُذافة العَدَويّ القُرَشيّ
كتاب نسب قريش: المصعب بن عبد الله الزُّبيريّ (ت، 236هـ)، عني بنشره لأول مرة وتصحيحه والتعليق عليه: إ. ليفي. بروفنسال، دار المعارف، القاهرة، ط3، (1982م)، ‏374‏.

مَثْغَر[اسْمُ مَكَان]

30هـ=651م

المَثْغَرُ : مَكَانُ نَبَاتِ الأَسْنَانِ.

خُبَعْثِنَةٌ فِي سَاعِدَيْهِ تَزَايُلٌ تَقُولُ: وَعَى مِنْ بَعْدِمَا قَدْ تَكَسَّرَا
شِبَالًا وَأَشْبَاهَ الزِّجَاجِ مَغَاوِلًا مَطَلْنَ وَلَمْ يَلْقَيْنَ فِي الرَّأْسِ مَثْغَرَا
أبو زُبيد الطّائيّ
شعر أبي زُبيد الطائي: جمعه وحققه: نوري حمّودي القيسيّ، المجمع العلمي العراقي- مطبعة المعارف، بغداد، (1967م)، ‏74‏.
ن80هـ=699م

المَثْغَرُ : مُقَدَّمُ الأَسْنَانِ.

شُنْبٌ مَثَاغِرُهَا ، يَرْضَى مُعَاشِرُهَا لَذٌّ مَبَاشِرُهَا، تَشْفِي مِنَ السَّقَمِ
أبو صخر الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏969‏.

ثَغَرَ[مُتَعَدٍّ]

ن37هـ=657م

ثَغْرِيّ[صِفَة]

272هـ=885م

الثَّغْرِيُّ : المُرَابِطُ فِي الثُّغُورِ لِدَفْعِ الأَعْدَاءِ.

"ثَمَّ نُتْبِعُ ذَلِكَ بِوَصْفِ أَحَدِ أَيَّامِ الغَزَوَاتِ؛ لِيَكُونَ عِلْمُ ذَلِكَ مُحَصَّلًا مَحْفُوظًا، فَنَقُولُ: إِنَّ أَجْهَدَهَا مِمَّا يَعْرِفُهُ أَهْلُ الخِبْرَةِ مِنَ الثَّغْرِيِّينَ ؛ أَنْ تَقَعَ الغُزَاةُ الَّتِى تُسَمَّى الرَّبِيعِيَّةَ لِعَشَرِةِ أَيَّامٍ تَخْلُو مِنْ أَيَّارَ بَعْدَ أَنْ يَكُونَ النَّاسُ قَدْ أَرْبَعُوا دَوَابَّهُمْ"
ابن خُرْداذْبُه
المسالك والممالك: ابن خُرْداذْبُه (ت، 280هـ)، تحقيق: ميخيل يوهنا دى خويه، مطبعة بريل، ليدن، (1889م)، ‏259‏.

أَثْغَر[صِفَةٌ مُشَبَّهَة]

216هـ=831م

الأَثْغَرُ : الَّذِي ذَهَبَ مُقَدَّمُ أَسْنَانِهِ.

وَرَدَ الحَبَائِلَ وَهْيَ صُورٌ نَحْوَهُ طَمَعًا لِتَعْلُقَهُ وَإِنْ لَمْ تَنْشُبِ
صَرَخَتْ بِهِ نَفْسُ النَّجِيِّ مَخَافَةً إِنَّ النَّجَاحَ لِأَثْغَرٍ مُتَشَعِّبِ
شوسة الفَقْعَسيّ
الممتع في صنعة الشّعر: عبد الكريم بن إبراهيم النَّهْشليّ القيروانيّ (ت، 405هـ)، تحقيق: محمّد زغلول سلام، منشأة المعارف، الإسكندرية، د.ت. ‏353‏.

مُثَّغِر[صِفَةٌ مُشَبَّهَة]

ن255هـ=869م

المُثَّغِرُ : الَّذِي نَبَتَتْ أَسْنَانُهُ.

" الِاثِّغَارُ يَكُونُ فِي النَّبَاتِ وَالسُّقُوطِ؛ وَمِنَ النَّبَاتِ، حَدِيثُ الضَّحَّاكِ: أَنَّهُ وُلِدَ وَهُوَ مُثَّغِرٌ "
شَمِر بن حَمْدَوَيْه الهَرَويّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏88‏.

مُثَّغِر[اسْمُ فَاعِل]

ن428هـ=1037م

المُثَّغِرُ : الَّذِي سَقَطَتْ أَسْنَانُهُ.

وَ"جَعْفَرُ"، مَا "جَعْفَرُ" المَكْرُمَا تِ، لَمْ يَسْلُ عَنْهَا، وَلَمْ يَنْزِعِ
وَكَمْ جَذَعٍ مِنْكَ أَقْرَحْتَهُ وَمُثَّغِرٍ بَعْدُ لَمْ يُجْذَعِ!
مهيار الدّيلميّ
ديوان مِهْيار الدَّيْلميّ: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1344هـ/ 1925م- 1350هـ/ 1931م)، ‏2/‏248‏.

ثَغْرَاوَا[اسْمُ جِنْس]

ن291هـ=904م

الثَّغْرَاوَا : شَجَرٌ كَبِيرٌ ذُو شَوْكٍ ضَعِيفٍ، يَنْبُتُ فِي البَرِّ.

"وَمِنَ الشَّجَرِ الكِبَارِ المُتَشَوِّكَةِ النَّابِتَةِ فِي البَرِّ: الثَّغْرَاوَا ، الوَاحِدَةُ مِنْهَا ثَغْرَاوَةٌ، شَوْكُهَا أَضْعَفُ مِنْ شَوْكِ شَجَرَةِ إِبْرَاهِيمَ"
ابن وحشيّة النَّبَطيّ
الفلاحة النَّبَطيّة: ابن وحشيّة النَّبَطيّ (ت، 291هـ)، تحقيق: توفيق فهد، المعهد العلميّ الفرنسيّ للدراسات العربيّة، دمشق، الجفان والجابي، قبرص، ط1، (1993م)، ‏1142‏.

مُثَاغَرَة[مَصْدَر]

745هـ=1344م

المُثَاغَرَةُ : المُرَابَطَةِ عَلَى ثُغُورِ البِلَادِ.

"مِنْ عَبْدِ اللهِ، أَمِيرِ المُسْلِمِينَ، المُجَاهِدِ فِي سَبِيلِ اللهِ رَبِّ العَالَمِينَ... مُؤْثِرِ المُرَابَطَةَ وَالمُثَاغَرَةَ ، مُؤَازِرِ حِزْبِ الإِسْلَامِ حَقَّ المُؤَازَرَةِ"
المنصور المريني
نفح الطيب من غصن الأندلس الرطيب: أحمد بن محمد المقريّ التلسمانيّ (ت، 1041هـ)، حققه: إحسان عباس، دار صادر، بيروت، (1388ه/ 1968م)، ‏4/‏386‏.

ثَغْرَة[اسْم]

1403هـ=1982م

الثَّغْرَةُ فِي الشَّيْءِ: مَوْضِعُ الضَّعْفِ وَالخَلَلِ فِيهِ.

"وَرُغْمَ أَنَّ هَذَا القَانُونَ كَانَ بِهِ بَعْضُ الثَّغَرَاتِ ... فَإِنَّنَا نَجِدُ أَنَّ القَانُونَ المَذْكُورَ فَتَحَ البَابَ فِعْلًا لِمُسَاهَمَةِ رُؤُوسِ الأَمْوَالِ المِصْرِيَّةِ"
نبيل عبد الحميد سيد أحمد
الفكر الاستراتيجي العربي: الفكر الاستراتيجي العربي، بيروت، ع( 6- 7)، (مايو- 1983م)، ‏353‏.
