خُذْرُوف[اسْم]

قبل 50ق.هـ=573م

الخُذْرُوفُ : لُعْبَةٌ مِنْ عُوَيْدٍ أَوْ قَصَبَةٍ مَشْقُوقَةٍ، تُثْقَبُ فِي وَسَطِهَا، وَيُجْعَلُ فِيهَا خَيْطٌ، ثُمَّ يُجْذَبُ الخَيْطُ، فَتَدُورُ مُحْدِثَةً دَوِيًّا.

دَرِيرٍ كَخُذْرُوفِ الوَلِيدِ أَمَرَّهُ تَقَلُّبُ كَفَّيْهِ بِخَيْطٍ مُوَصَّلِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏21‏.
8هـ=629م

الخُذْرُوفُ : البَرْقُ اللَّامِعُ فِي المُنْقَطِعِ مِنَ السَّحَابِ.

صَبَرْنَا مَعَ الضَّحَّاكِ، لَا يَسْتَفِزُّنَا قِرَاعُ الأَعَادِي مِنْهُمُ وَالوَقَائِعُ
أَمَامَ رَسُولِ اللهِ يَخْفِقُ فَوْقَنَا لِوَاءٌ كَخُذْرُوفِ السَّحَابَةِ لَامِعُ
العبّاس بن مرداس السُّلَميّ
ديوان العباس بن مرداس السَّلمي: جمعه وحققه: يحيى الجبوري، مؤسسة الرسالة، بيروت، ط1، (1412هـ/ 1991م)، ‏108‏.
ن37هـ=657م

الخُذْرُوفُ مِنَ الشَّيْءِ: القِطْعَةُ مِنْهُ.

وَحَيُّ نُمَيْرٍ، إِنْ دَعَوْتُ، أَجَابَنِي كِرَامٌ، إِذَا شُلَّ السَّعَامُ المُصَبَّحُ
لِأَسْيَافِهِمْ فِي كُلِّ يَوْمِ كَرِيهَةٍ خَذَارِيفُ هَامٍ، أَوْ مَعَاصِمُ سُنَّحُ
ابن مقبل العَجْلانيّ التَّميميّ
ديوان ابن مقبل: عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏56‏.
ن60هـ=680م

الخُذْرُوفُ : لُعْبَةٌ مُسْتَدِيرَةٌ كَقِطْعَةِ السُّكَّرِ، تُعْجَنُ مِنْ طِينٍ.

بِمَأثُورَةٍ مِنْ عِنْدِ دَاوُودَ يُخْتَلَى بِهَا الهَامُ وَالأَيْدِي، حَدِيثٍ صِقَالُهَا
تُغَشَّى بِهِنَّ الهَامُ حَتَّى كَأَنَّهَا خَذَارِيفُ ، أَوْ بَيْضٌ يُجَرُّ قِلَالُهَا
أُنيف بن زَبّان/حكيم الطّائيّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏7/‏77‏.
ن203هـ=818م

الخُذْرُوفُ : الخَشَبَةُ العَرِيضَةُ كَاللَّوْحِ.

"الدَّوَامِغُ عَلَى حَاقِّ رُؤُوسِ الأَحْنَاءِ مِنْ فَوْقِهَا، وَاحِدَتُهَا دَامِغَةٌ، وَرُبَّمَا كَانَتْ مِنْ خَشَبٍ وَتُؤْسَرُ بِالقِدِّ أَسْرًا شَدِيدًا، وَهِيَ الخَذَارِيفُ وَاحِدُهَا خُذْرُوفٌ"
النَّضْر بن شُميل المازنيّ التّميميّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏80‏.

خُذْرُوف[صِفَة]

قبل 13ق.هـ=609م

الخُذْرُوفُ مِنْ عَدْوِ الخَيْلِ وَنَحْوِهَا: السَّرِيعُ الَّذِي يُبْذَلُ فِيهِ أَقْصَى الوُسْعِ.

وَفِينَا رِبَاطُ الخَيْلِ، كُلُّ مُطَهَّمٍ رَجِيلٍ كَسِرْحَانِ الغَضَا المُتَأَوِّبِ
يُذِيقُ الَّذِي يَعْلُو عَلَى ظَهْرِ مَتْنِهِ ظِلَالَ خَذَارِيفٍ مِنَ الشَّدِّ مُلْهِبِ
الطُّفَيل بن عوف الغَنَويّ
ديوان طُفيل الغَنَوي: شرح: الأصمعي، تحقيق: حسان فلاح أوغلي، دار صادر، بيروت، ط1، (1997م)، ‏27 ‏-‏ 28‏.
ن110هـ=728م

الخُذْرُوفُ مِنَ السُّيُوفِ وَنَحْوِهَا: المُحَدَّدُ السَّرِيعُ القَطْعِ.

أَلَمْ يَأْتِ بِالشَّأْمِ الخَلِيفَةَ أَنَّنَا ضَرَبْنَا لَهُ مَنْ كَانَ عَنْهُ يُخَالِفُ
صَنَادِيدَ أَهْدَيْنَا إِلَيْهِ رُؤُوسَهُمْ وَقَدْ بَاشَرَتْ مِنْهَا السُّيُوفُ الخَذَارِفُ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏110‏.
330هـ=941م

الخُذْرُوفُ مِنَ النَّخْلِ وَنَحْوِهِ: المَقْطُوعُ المَشْقُوقُ مِنْهُ.

تَخَالُ سَوْسَنَهَا فِي كُلِّ شَارِقَةٍ عَلَى المَيَادِينِ أَذْنَابَ الطَّوَاوِيسِ
خِلَالَهُ الآسُ وَالخِيرِيُّ عَنْ كَثَبٍ وَالنَّخْلُ مِنْ بَيْنِ خُذْرُوفٍ وَمَغْرُوسِ
من رواية محمد بن سهل بن المَرْزُبان البغداديّ
كتاب المنتهى في الكمال: محمد بن سهل بن المَرْزُبان البغداديّ (ت، 330هـ)، رواية: أبي محمّد عبيد الله بن محمّد بن جميل الكرخيّ، تحقيق ودراسة: سالم مرعي الهَدْروسيّ، معهد الدّراسات الشّرقيّة، جامعة يوستس ليبغ، ألمانيا، (1987م)، ‏190‏.

خَذْرَفَ[لَازِم]

قبل 13ق.هـ=609م

خَذْرَفَتِ النَّاقَةُ وَنَحْوُهَا: أَسْرَعَتْ وَرَمَتْ بِقَوَائِمِهَا فِي سَيْرِهَا مِنْ سُرْعَتِهَا.

أَنْضَيْتُهَا بَعْدَ المَرَاحِ إِلَى امْرِئٍ جَلْدِ القُوَى فِي كُلِّ سَاعَةِ مَحْبِسِ
إِلَى ابْنِ هِنْدٍ خَذْرَفَتْ أَخْفَافُهَا تَهْوِي لِمُعْتَمَدٍ، بَعِيدِ المَحْدِسِ
أبو اللَّحّام التَّغْلِبيّ
شعراء تغلب في الجاهلية أخبارهم وأشعارهم: صنعة: علي أبو زيد، المجلس الوطني للثقافة والفنون والآداب، الكويت، ط1، (1421هـ/ 2000م)، ‏2/‏265‏.

خَذْرَفَ[مُتَعَدٍّ]

280هـ=893م

خَذْرَفَ الشَّخْصَ بِالسَّيْفِ: قَطَعَ أَطْرَافَهُ بِهِ.

"يُقَالُ: خَذْرَفَهُ بِالسَّيْفِ"
من رواية ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏2/‏1144‏.
436هـ=1044م

خَذْرَفَ الرَّحَى: وَضَعَ عُودًا فِي خَرْقِ الحَجَرِ الأَعْلَى مِنْهَا.

"قَدْ خَذْرَفَ الرَّحَى"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏5/‏208‏.

خَذْرَفَ[مُتَعَدٍّ بِالحَرْف]

ن351هـ=962م

خَذْرَفَ فِي مِشْيَتِهِ: اهْتَزَّ وَتَبَخْتَرَ.

"مَرَّ يُخَذْرِفُ فِي مِشْيَتِهِ خَذْرَقَةً"
من رواية أبي الطّيّب اللُّغويّ الحلبيّ
كتاب الإبدال: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، حققه وشرحه ونشر حواشيه الأصلية وأكمل نواقصه: عز الدين التنوخي، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏2/‏19‏.

خَذْرَفَة[مَصْدَر]

4ق.هـ=618م

خَذْرَفَةُ الدَّابَةِ: إِسْرَاعُهَا وَرَمْيُهَا بِقَوَائِمِهَا تَوْسِيعًا لِخَطْوِهَا.

فَانْصَاعَ لَا يَأْتَلِي شَدًّا بِخَذْرَفَةٍ تَرَى لَهُ مِنْ يَقِينِ الخَوْفِ إِهْذَابَا
الأعشى الكبير
كتاب الصبح المنير في شعر أبي بصير ميمون بن قيس بن جندل الأعشى والأعْشَيْن الآخَرَيْن: مطبعة آدلف هلزهوسن، دار ابن قتيبة، الكويت، ط2، (1993م)، ‏229‏.
ن351هـ=962م

الخَذْرَفَةُ فِي المَشْيِ: الِاهْتِزَازُ والتَّبَخْتُرُ فِيهِ.

"وَيُقَالُ: مَرَّ يُخَذْرِفُ فِي مِشْيَتِهِ خَذْرَقَةً "
أبو الطّيّب اللُّغويّ الحلبيّ
كتاب الإبدال: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، حققه وشرحه ونشر حواشيه الأصلية وأكمل نواقصه: عز الدين التنوخي، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏2/‏19‏.

مُخَذْرَف[اسْمُ مَفْعُول]

ن68هـ=687م

المُخَذْرَفُ مِنَ الأَظْلَافِ وَنَحْوِهَا: المُحَدَّدُ الأَطْرَافِ.

تُذْرِي الخُزَامَى بِأَظْلَافٍ مُخَذْرَفَةٍ وَوَقْعُهُنَّ إِذَا وَقَعْنَ تَحْلِيلُ
جِران العَود النُّميريّ
ديوان جران العَوْد النميري وضمنه شعر عُروة الرّحّال: تحقيق: كارين صادر، دار صادر، بيروت، ط1، (1999م)، ‏95‏.

خِذْرَاف[اسْمُ جِنْس]

86هـ=705م

الخِذْرَافُ : نَبْتٌ حَمْضِيٌّ، لَهُ وُرَيْقَةٌ صَغِيرَةٌ، يَرْتَفِعُ قَدْرَ الذِّرَاعِ، يَجِفُّ صَيْفًا.

إِذَا حَمَلَتْ مَاءَ الصَّرَائِمِ قَلَّصَتْ رَوَايَا لِأَطْفَالٍ بِمَعْمِيَّةٍ زُغْبِ
تَوَائِمَ أَشْبَاهٍ بِأَرْضٍ مَرِيضَةٍ يَلُذْنَ بِخِذْرَافِ المِتَانِ وَبِالعِرْبِ
الأخطل
شعر الأخطل أبي مالك غياث بن غوث التغلبي: صنعة: السُّكّري، رواية: أبي جعفر محمّد بن حبيب، تحقيق: فخر الدين قباوة، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط4، (1416هـ/ 1996م)، ‏39‏.

خِذْرَاف[مَصْدَر]

ن351هـ=962م

الخِذْرَافُ : الِاهْتِزَازُ وَالتَّبَخْتُرُ فِي المَشْيِ.

"وَيُقَالُ: مَرَّ يُخَذْرِفُ فِي مِشْيَتِهِ خَذْرَقَةً وَخِذْرَافًا "
من رواية أبي الطّيّب اللُّغويّ الحلبيّ
كتاب الإبدال: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، حققه وشرحه ونشر حواشيه الأصلية وأكمل نواقصه: عز الدين التنوخي، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏2/‏19‏.

خُذْرُوفَة[اسْم]

ن398هـ=1007م

الخُذْرُوفَةُ : اللَّبِنَةُ مِنْ طِينٍ وَنَحْوِهِ كَالأُثْفِيَّةِ الَّتِي يُوضَعُ عَلَيْهَا قِدْرُ الطَّبْخِ.

"يَا نَكْهَةَ الصُّقُورِ، يَا وَتِدَ الدُّورِ، يَا خُذْرُوفَةَ القُدُورِ"
بديع الزّمان الهَمَذانيّ
مقامات بديع الزّمان الهَمَذانيّ: بديع الزّمان الهَمَذانيّ (ت، 398هـ)، قدم لها وشرح غوامضها: الشّيخ محمّد عبده، دار الكتب العلميّة، بيروت، ط3، (1426هـ/ 2005م)، ‏367‏.
ن398هـ=1007م

الخُذْرُوفَةُ : الخُذْرُوفُ، وَهُوَ لُعْبَةٌ لِلصِّبْيَانِ.

أَنَا جَوَّالَةُ البِلَا دِ، وَجَوَّابَةُ الأُفْقِ
أَنَا خُذْرُوفَةُ الزَّمَا نِ وَعَمَّارَةُ الطُّرْقِ
بديع الزّمان الهَمَذانيّ
زهر الآداب وثمر الألباب: أبو إسحاق الحُصْريّ القيروانيّ (ت، 413هـ)، عارضه بمخطوطات القاهرة وحققه وضبطه وشرحه ووضع فهارسه: علي محمّد البجاوي، دار إحياء الكتب العربية، القاهرة، ط1، (1372هـ/ 1953م)، ‏2/‏892‏.
