رُفْغ[اسْم]

قبل 50ق.هـ=573م

الرُّفْغُ : مَا يُحِيطُ بِالفَرْجِ مِنْ أُصُولِ أَعَالِي الفَخِذَيْنِ مِنْ بَاطِنٍ.

تَحْمِلُ رُفْغًا وَاسِعَ الفُضُولْ مِثْلَ إِهَابِ المَيْحَةِ المَبْخُولْ يَبِيتُ فِيهِ الذِّئْبُ، أَوْ يَقِيلْ
طارقة مولاة امرئ القيس بن زيد الأوسيّ
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏115‏.
قبل 50ق.هـ=573م

الرُّفْغُ : الإِبِطُ.

مُرَسَّعَةٌ وَسْطَ أَرْفَاغِهِ بِهِ عَسَمٌ، يَبْتَغِي أَرْنَبَا
امرؤ القيس بن حُجْر الكنديّ
الغريبين في القرآن والحديث: أبو عُبيد الهَرَويّ الفاشانيّ (ت، 401هـ)، تحقيق ودراسة: أحمد فريد المزيدي، قدم له وراجعه: فتحي حجازي، مكتبة نزار مصطفى الباز، مكّة المكرّمة- الرّياض، ط1، (1419هـ/ 1999م)، ‏3/‏740‏.
ن11هـ=632م

الرُّفْغُ : الوَسَخُ الَّذِي بَيْنَ الأَنْمُلَةِ وَالظُّفْرِ.

"وَكَيْفَ لَا أُوهِمُ، وَرُفْغُ أَحَدِكُمْ بَيْنَ ظُفْرِهِ وَأَنْمُلَتِهِ؟!"
حديث نبويّ
كتاب غريب الحديث: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: عبد السلام هارون وآخرين، الهيئة العامة لشؤون المطابع الأميرية، القاهرة، (1404هـ/ 1984م- 1419هـ/ 1999م)، ‏1/‏262‏.
ن11هـ=632م

الرُّفْغُ : الفَرْجُ.

"إِذَا الْتَقَى الرُّفْغَانِ ، فَقَدْ وَجَبَ الغُسْلُ"
حديث نبويّ
غريب الحديث: أبو عُبيد القاسم بن سلّام الهرويّ (ت، 224هـ)، مراقبة: محمد عبد المعيد، مطبعة دائرة المعارف العثمانية، حيدر آباد الدكن، ط1، 1384هـ/1964م، ‏1/‏263‏.

رَافِغ[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الرَّافِغُ مِنَ العَيْشِ: الوَاسِعُ الرَّاغِدُ.

صُعُدٌ، نُحَرِّزُ أَهْلَنَا بِفُرُوعِهِ، فِيهِ لَنَا حِرْزٌ، وَعَيْشٌ رَافِغُ
أبو سُلْمى المُزَنيّ
شرح شعر زهير بن أبي سلمى: أبو العبّاس ثعلب (ت، 291هـ)، تحقيق: فخر الدين قباوة، مكتبة هارون الرشيد، دمشق، ط3، (1428هـ/ 2008م)، ‏272‏.
398هـ=1007م

الرَّافِغُ مِنَ النَّاسِ: الَّذِي يَعِيشُ فِي سَعَةٍ وَرَغَدٍ.

"فَسُبْحَانَ اللهِ مَعَ المُسَبِّحِينَ، مَا وَصَلَ الشَّادِنُ إِلَى البَرِير، إِلَّا بَعْدَ ضَرِير، وَاللهُ يَسَّرَ المَعِيشَةَ لِأَهْلِ الخِصْبِ الرَّافِغِينَ "
أبو العلاء المَعَرّيّ
الفصول والغايات في تمجيد الله والمواعظ: أبو العلاء المَعَرّيّ (ت، 449هـ)، صنعه وفسر غريبه: محمود حسن زناتي، مراجعة: لجنة إحياء التراث العربي، دار الآفاق الجديدة، بيروت، د.ت. ‏239‏.
ن780هـ=1378م

الرَّافِغُ العَيْشَ: الجَاعِلُهُ وَاسِعًا رَاغِدًا.

وَالوَرَى قِسْمَانِ: ذُو حَسَدٍ فَهْوَ يَوْمَ الحَرْبِ دَامِغُهُ
وَفَتًى ضَاقَ الزَّمَانُ بِهِ فَهْوَ بِالإِحْسَانِ رَافِغُهُ
ابن جابر الأندلسيّ
ديوان المقصد الصّالِح في مدح الملك الصّالِح: ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: أحمد فوزي الهيب، دار سعد الدِّين، دمشق، ط1، (1429ه/ 2008م)، ‏74‏.

رَفْغ[اسْم]

10هـ=631م

الرَّفْغُ : الأَرْضُ الكَثِيرَةُ التُّرَابِ.

مَا حُمِّلَ البُخْتِيُّ عَامَ غِيَارِهِ عَلَيْهِ الوُسُوقُ، بُرُّهَا وَشَعِيرُهَا
أَتَى قَرْيَةً كَانَتْ كَثِيرًا طَعَامُهَا كَرَفْغِ التُّرَابِ، كُلُّ شَيْءٍ يَمِيرُهَا
أبو ذؤيب الهُذَليّ
ديوان أبي ذؤيب الهُذلي: تحقيق وإخراج: أحمد خليل الشال، مركز الدراسات والبحوث الإسلامية، بورسعيد، ط1، (1435هـ/ 2014م)، ‏114‏.
136هـ=753م

الرَّفْغُ : النَّاحِيَةُ.

إِنْ لَمْ يَعُقْنِي عَائِقُ التَّسَغْسُغِ فِي الأَرْضِ، فَارْقُبْنِي وَعَجْمَ المُضَّغِ لَاجْتَبْتُ مَسْحُولًا جَدِيبَ الأَرْفُغِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏50‏.
ن175هـ=791م

رَفْغ[صِفَةٌ مُشَبَّهَة]

369هـ=979م

الرَّفْغُ مِنَ الأَشْخَاصِ: السَّافِلُ الرَّذِيلُ.

"فَأَجْمَعَتِ العَامَّةُ بِبَغْدَادَ بِالصُّرَاخِ وَالنِّدَاءِ بِالنَّفِيرِ، وَانْضَمَّتْ إِلَيْهَا الأَبْنَاءُ وَالشَّاكِرِيَّةُ، تُظْهِرُ أَنَّهَا تَطْلُبُ الأَرْزَاقَ، فَفَتَحُوا السُّجُونَ، وَأَخْرَجُوا رُفُوغَ خَرَاسَانَ وَالصَّعَالِيكَ مِنْ أَهْلِ الجِبَالِ"
ابن مسكويه
تجارِب الأُمَم وتعاقب الهِمَم: ابن مسكويه (ت، 421هـ)، حققه وقدم له: أبو القاسم إمامي، دار سروش، طهران، ط2، (1422هـ/ 2001م)، ‏4/‏324‏.

رَفِيغ[صِفَةٌ مُشَبَّهَة]

13هـ=634م

الرَّفِيغُ مِنَ العَيْشِ: الوَاسِعُ الخَصِيب.

"وَأَمَّا مَا ذَكَرْتُمْ مِنْ بِلَادِنَا، وَزَعَمْتُمْ أَنَّهَا بِلَادُ البُؤْسِ وَالشَّقَاءِ، فَقَدْ صَدَقْتُمْ، وَقَدْ أَبْدَلَنَا اللهُ بِهَا بِلَادَكُمْ، بِلَادَ العَيْشِ الرَّفِيغِ "
أبو عُبيدة ابن الجَرّاح الفِهْريّ القُرَشيّ
جمهرة رسائل العرب في عصور العربية الزاهرة العصر الجاهلي عصر صدر الإسلام: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1356هـ/ 1937م- 1357هـ/ 1939م)، ‏1/‏154‏.
101هـ=719م

الرَّفِيغُ مِنَ السِّعْرِ: الرَّخِيصُ.

"سَرِّحْ مَعَ أَهْلِ بَيْتِكَ خَيْلًا مِنْ خَيْلِكَ عَظِيمَةً، فَتَأْتِي الجَزِيرَةَ، وَتُبَادِرُ إِلَيْهَا، حَتَّى يَنْزِلُوا حِصْنًا مِنْ حُصُونِهَا... فَيَأْتِيكَ مَنْ بِالمَوْصِلِ مِنْ قَوْمِكَ، وَيَنْفَضُّ إِلَيْكَ أَهْلُ العِرَاقِ وَأَهْلُ الثُّغُورِ، وَتُقَاتِلُهُمْ فِي أَرْضٍ رَفِيغَةِ السِّعْرِ، وَقَدْ جَعَلْتَ العِرَاقَ كُلَّهُ وَرَاءِ ظَهْرِكَ"
حبيب بن المهلَّب بن أبي صُفْرة
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏6/‏588‏.

"فَلَمْ تُصْبِحْ أُمَّةٌ مُخَالِفَةٌ لِدِينِكُمْ إِلَّا أُمَّتَانِ: أُمَّةٌ مُسْتَعْبَدَةٌ لِلْإِسْلَامِ وَأَهْلِهِ... وَأُمَّةٌ تَنْتَظِرُ وَقَائِعَ اللهِ وَسَطَوَاتِهِ فِي كُلِّ يَوْمٍ وَلَيْلَةٍ، قَدْ مَلَأَ اللهُ قُلُوبَهُمْ رُعْبًا... قَدْ دَهَمَتْهُمْ جُنُودُ اللهِ  ، وَنَزَلَتْ بِسَاحَتِهِمْ، مَعَ رَفَاغَةِ العَيْشِ، وَاسْتِفَاضَةِ المَالِ"

أَرْفَغَ[لَازِم]

375هـ=985م

أَرْفَغَ الشَّخْصُ: رَغِدَ عَيْشُهُ وَاتَّسَعَ.

"وَقَامَتْ لَهُمُ الأَسْوَاقُ، وَطَابَتِ المَرَابِعُ، وَفَشَا الخِصْبُ، وَتَوَالَى النِّتَاجُ... وَأَرْفَغَ المُنْتَجِعُ"
أبو حيّان التَّوحيديّ
كتاب الإمتاع والمؤانسة: أبو حيّان التَّوحيديّ (ت، 414هـ)، صححه وضبطه وشرح غريبه: أحمد أمين، أحمد الزين، المكتبة العصرية، بيروت- صيدا، (1373هـ/ 1953م)، ‏1/‏80‏-‏ 81‏.

مَرَافِغ[اسْمُ جَمْع]

ن58هـ=678م

المَرَافِغُ : أُصُولُ اليَدَيْنِ وَالفَخِذَيْنِ مِنْ بَاطِنٍ، الَّتِي تَتَجَمَّعُ فِيهَا الأَوْسَاخُ.

"كَانَ رَسُولُ اللهِ  @  إِذَا أَرَادَ أَنْ يَغْتَسِلَ مِنَ الجَنَابَةِ، بَدَأَ بِكَفَّيْهِ فَغَسَلَهُمَا، ثُمَّ غَسَلَ مَرَافِغَهُ ، وَأَفَاضَ عَلَيْهِ المَاءَ"
عائشة بنت أبي بكر الصِّدّيق
سنن أبي داود: أبو داود السِّجِسْتانيّ (ت، 275هـ)، حكم على أحاديثه وآثاره وعلّق عليه: محمّد ناصر الدين الألبانيّ، اعتنى به: مشهور حسن سلمان، مكتبة المعارف، الرياض، ط2، (1424هـ)، ‏1/‏46،،‏(رقم الحديث: 243).

رَفَاغ[مَصْدَر]

86هـ=705م

رَفَاغُ العَيْشِ: سَعَتُهُ وَرَغَدُهُ.

"وَكَانَتْ أَرْضُ الحَبَشَةِ مَتْجَرًا لِقُرَيْشٍ يَتَّجِرُونَ فِيهَا، يَجِدُونَ فِيهَا رَفَاغًا مِنَ الرِّزْقِ، وَأَمْنًا، وَمَتْجَرًا حَسَنًا"
عروة بن الزُّبير بن العَوّام
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏2/‏328‏.

رَفَّغَ[مُتَعَدٍّ]

96هـ=715م

رَفَّغَ المَرْأَةَ: دَلَكَ رُفْغَهَا قَبْلَ الجِمَاعِ.

"تَزَوَّجَ الوَلِيدُ بْنُ عَبْدِ المَلِكِ بْنِ مَرْوَانَ امْرَأَةً مِنَ العَرَبِ، فَلَمَّا صَارَ إِلَيْهَا، قَالَتْ: رَفِّغْ، رَفِّغْ، قَبَّحَ اللهُ أُمًّا عَوَّدَتْكَ مَا أَرَى"
امرأة من العرب
البصائر والذّخائر: أبو حيّان التَّوحيديّ (ت، 414هـ)، تحقيق: وداد القاضي، دار صادر، بيروت، ط1، (1408هـ/ 1988م)، ‏5/‏172‏.

رَفَاغِيَة[مَصْدَر]

ن160هـ=777م

رَفَاغِيَةُ العَيْشِ:سَعَتُهُ وَرَغَدُهُ.

"وَخُيِّرَ أَمِيرُ المُؤْمِنِينَ مِنْ مَنْزِلِهِ الَّذِي هُوَ بِهِ مَنْزِلًا جَمَعَ مِنْ بَسْطَةٍ فِي المَوْضِعِ، وَرَفَاغِيَةٍ فِي المَعَاشِ، أَنَّهُ حَامِلٌ لِلْجُنُودِ، جَامِعٌ لِلْمَرَافِقِ"
يحيى بن زياد بن عُبيد الله الحارثيّ
جمهرة رسائل العرب في عصور العربية الزاهرة العصر الجاهلي عصر صدر الإسلام: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1356هـ/ 1937م- 1357هـ/ 1939م)، ‏3/‏213‏.

رَفْغَاء[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

الرَّفْغَاءُ مِنَ النِّسَاءِ: الدَّقِيقَةُ الفَخِذَيْنِ، الضَّيِّقَةُ الرُّفْغَيْنِ، الصَّغِيرَةُ المَتَاعِ.

"وَامْرَأَةٌ رَبْلَاءُ رَفْغَاءُ ، أَيْ: ضَيِّقَةُ الأَرْفَاغِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏265‏.
ن175هـ=791م

الرَّفْغَاءُ مِنَ النُّوقِ: الوَاسِعَةُ أُصُولِ الفَخِذَيْنِ مِنَ البَاطِنِ.

"وَنَاقَةٌ رَفْغَاءُ : وَاسِعَةُ الرُّفْغِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏407‏.

مُرَفَّغ[اسْمُ مَفْعُول]

ن384هـ=994م

المُرَفَّغُ مِنَ العَيْشِ: المُوَسَّعُ المُرَغَّدُ.

بَقِيتَ، وَلَا تَعْدَمْ بَقَاءً مُرَفَّهًا وَعِشْتَ، وَلَا تَعْدَمْ مَعَاشًا مُرَفَّغَا
أبو إسحاق الصّابي
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏1/‏314‏.

رُفَيْغَة[اسْمٌ مُصَغَّر]

ن483هـ=1090م

الرُّفَيْغَةُ : تَصْغِيرُ الرَّفْغٍ، وَهُوَ اللَّيِّنُ مِنَ الطِّينِ وَنَحْوِهِ.

فَهِيلَ بِهِ كَثِيبُ الكُفْرِ هَيْلًا وَكُلُّ رُفَيْغَةٍ مِنْهُ رُكَامُ
ابن وهبون المرسيّ الأندلسيّ
الذخيرة في محاسن أهل الجزيرة: ابن بسّام الشّنْتَرِيني (ت، 542هـ)، تحقيق: إحسان عبّاس، دار الثَّقافة، بيروت، ط1، (1417ه/ 1997م)، ‏2/‏246‏.

رَفَغَ[مُتَعَدٍّ]

ن488هـ=1095م

رَفَغَ عَيْشَهُ: وَسَّعَهُ وَرَغَّدَهُ.

عَبْدَ الغَنِيِّ اقْتَرِبْ، فَلَا وَأَبِي مَا رُفِّهِ العَيْشُ لِي، وَلَا رُفِغَا
أبو الحسن الحُصْريّ الفِهْريّ القيروانيّ
أبو الحسن الحصري القيرواني: تحقيق: محمّد المرزوقيّ، الجيلانيّ بن الحاج يحيى، مكتبة المنار، تونس، (1963م)، ‏403‏.

تَرَفَّغَ[لَازِم]

1039هـ=1629م

تَرَفَّغَ العَيْشُ وَنَحْوُهُ: تَوَسَّعَ.

غَمْرُ الرِّدَا، بَحْرُ النَّدَى يَتَرَفَّغُ غَمَرَ البِلَادَ، بِذِكْرِهِ يُسْتَفْرَغُ
محمد بن عفيف الدين الإيجي الحسني
نفح الطيب من غصن الأندلس الرطيب: أحمد بن محمد المقريّ التلسمانيّ (ت، 1041هـ)، حققه: إحسان عباس، دار صادر، بيروت، (1388ه/ 1968م)، ‏7/‏478‏.
