شَنِجَ[لَازِم]

قبل 50ق.هـ=573م

شَنِجَ الشَّيْءُ: تَقَبَّضَ.

لَمَّا رَأَيْتَ الخَيْلَ قَدْ طَافَتْ بِهِ شَنِجَتْ شِمَالُكَ فِي عِنَانِ الأَشْقَرِ
مارية بنت الدّيّان الحارثيّة
شُعراء مَذْحِج أخبارهم وأشعارهم في الجاهلية: صنعة: مُقبل التَّامّ عامر الأحمدي، مَجمَع العربيَّة السَّعيدة، صنعاء، ط2، (1435هـ/ 2014م)، ‏588‏.

شَنِج[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الشَّنِجُ مِنَ العُرُوقِ وَنَحْوِهَا: المُنْقَبِضُ.

لَهُ سَاقَا ظَلِيمٍ خَا ضِبٍ فُوجِئَ بِالرُّعْبِ
وَقُصْرَى شَنِجِ الأَنْسَا ءِ، نَبَّاحٍ مِنَ الشُّعْبِ
عقبة بن سابق الهِزّانيّ العَنَزيّ
الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏41‏.
ن20هـ=641م

الشَّنِجُ مِنَ النَّاسِ: البَخِيلُ.

بِكُلِّ فَتًى يَعْصَى بِكُلِّ مُهَنَّدٍ نَدٍ، غَيْرِ مِبْطَانِ العَشِيَّاتِ، عَثْجَلِ
كَعِجْلِ الهِجَانِ الأُدْمِ، لَيْسَ بِرُمَّحٍ وَلَا شَنِجٍ ، كَزِّ الأَنَامِلِ، زُمَّلِ
عمرو بن شَأْس الأسديّ
شعر عَمرو بن شأس الأسدي: تحقيق: يحيى الجبوري، دار القلم، الكويت، ط2، (1403هـ/ 1983م)، ‏48‏.

أَشْنَجَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

أَشْنَجَ الشَّيْءَ: قَبَّضَهُ.

فَمَنْ لِلْخَيْلِ بَعْدَ أَبِي سِرَاجٍ إِذَا مَا أَشْنَجَ الصِّرُّ الكَلِيبَا؟
عُبادة/عِياذ بن مُحَبَّر السَّعْديّ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏2/‏77‏.

شَنَج[مَصْدَر]

ن8هـ=629م

شَنَجُ الشَّيْءِ: شِدَّةُ تَقَبُّضِهِ.

وَأَنْتُمُ مَعْشَرٌ فِي عِرْقِكُمْ شَنَجٌ بُزْخُ الظُّهُورِ، وَفِي الأَسْتَاهِ تَأْخِيرُ
دريد بن الصِّمَّة الجُشَميّ
ديوان دريد بن الصِّمَّة الجُشَمي: جمع وتحقيق وشرح: محمّد خير البقاعي، قدم له: شاكر الفحّام، دار قتيبة، دمشق، (1401هـ/ 1981م)، ‏74‏.

شَنَج[اسْم]

ن175هـ=791م

الشَّنَجُ : الجَمَلُ.

"وَهُذَيْلٌ تَقُولُ: عَنَجٌ عَلَى شَنَجٍ ؛ أَيْ: رَجُلٌ عَلَى جَمَلٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏232‏.

أَشْنَج[صِفَةٌ مُشَبَّهَة]

ن50هـ=670م

الأَشْنَجُ مِنَ العُرُوقِ وَنَحْوِهَا: المُتَقَبِّضُ.

وَسَطْعَاءَ لَمْ تُجْنِنْ حُوَارًا، وَلَمْ تُرَعْ لِهَدْرٍ، وَلَمْ تَغْمُمْ يَدَيْهَا الزَّوَامِلُ
زَلُوجٍ بِشَنْجَاءِ النَّسَا مُسْتَقِلَّةٍ بِرَجْعِ السُّلَامَى، لَمْ تَخُنْهَا الحَوَامِلُ
مُلَيح بن الحَكَم الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏3/‏1059‏.

تَشَنَّجَ[لَازِم]

ن58هـ=678م

تَشَنَّجَ الشَّيْءُ: تَقَبَّضَ.

"إِذَا حَشْرَجَ الصَّدْرُ، وَطَمَحَ البَصَرُ، وَاقْشَعَرَّ الجِلْدُ، وَتَشَنَّجَتِ الأَصَابِعُ، فَعِنْدَ ذَلِكَ مَنْ أَحَبَّ لِقَاءَ اللهِ، أَحَبَّ اللهُ لِقَاءَهُ، وَمَنْ أَبْغَضَ لِقَاءَ اللهِ، أَبْغَضَ اللهُ لِقَاءَهُ"
عائشة بنت أبي بكر الصِّدّيق
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏14/‏230،،‏(رقم الحديث: 8556).

مُشَنَّج[اسْمُ مَفْعُول]

ن68هـ=687م

المُشَنَّجُ مِنَ الجِلْدِ وَنَحْوِهِ: المُقَبَّضُ.

فَمَا إِنْ بَرِحْنَا السِّجْنَ حَتَّى بَدَا لَنَا جَبِينٌ كَقَرْنِ الشَّمْسِ غَيْرُ مُشَنَّجِ
عُبيد الله بن الحُرّ الجُعْفيّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏3/‏306‏.

مُشَنَّج[صِفَةٌ مُشَبَّهَة]

ن121هـ=739م

المُشَنَّجُ مِنَ السَّرَاوِيلِ: الوَاسِعُ الَّذِي يَسْقُطُ عَلَى الخَلْفِ حَتَّى يُغَطِّيَ نِصْفَ القَدَمِ.

"امْنَعِ النَّاسَ مِنَ السَّرَاوِيلِ المُشَنَّجَةِ "
مَسْلَمة بن عبد الملك بن مروان
المجموع المغيث في غريبي القرآن والحديث: الحافظ أبو موسى محمد بن أبي بكر بن أبي عيسى المدينى الأصفهاني، (ت،581ه)، تحقيق: عبد الكريم العزباوي، جامعة أم القرى، مكة المكرمة، ط1، (1406هـ/ 1986م)، ‏2/‏223‏.

انْشِنَاج[مَصْدَر]

ن110هـ=728م

انْشِنَاجُ الشَّيْءِ: انْقِبَاضُهُ.

حَوْضُ الحِمَارِ أَبُو الفَرَزْدَقِ، فَاعْلَمُوا عَقَدَ الأَخَادِعِ، وَانْشِنَاجَ المِرْفَقِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏936‏.

شَنَّجَ[مُتَعَدٍّ]

ن110هـ=728م

شَنَّجَ الشَّيْءَ: جَعَلَهَ مُتَقَلِّصًا مُنْقَبِضًا.

وَقَدْ قُوِّسَتْ أُمُّ البَعِيثِ، وَأُكْرِهَتْ عَلَى الزِّفْرِ، حَتَّى شَنَّجَتْهَا الأَخَادِعُ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏924‏.

مُنْشَنِج[اسْمُ فَاعِل]

132هـ=749م

مُنْشَنِجُ اليَدِ: مُنْقَبِضُهَا.

إِنْ أُمْسِ مُنْشَنِجَ اليَدَيْنِ عَنِ النَّدَى وَعَنِ العَدُوِّ مُحَبَّسَ الشَّيْطَانِ
فَلَقَدْ أَرُوحُ عَلَى اللِّئَامِ مُسَلَّطًا ثَلِجَ المَقِيلِ، مُنَعَّمَ النَّدْمَانِ
بشّار بن برد
أمالي المرتضى غُرر الفوائد ودُرر القلائد: الشّريف المرتضى (ت، 436هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، ط1، (1373هـ/ 1954م)، ‏1/‏510‏.

تَشَنُّج[مَصْدَر]

ن175هـ=791م

التَّشَنُّجُ : التَّقَبُّضُ الَّذِي يُصِيبُ الجِلْدَ أَوِ العَصَبَ أَوِ العَضَلَاتِ، فَيَمْنَعُهَا مِنَ الِانْبِسَاطِ.

"عَجُوزٌ هَمَّرِشٌ: جَحْمَرِشٌ فِي اضْطِرَابِ خَلْقِهَا، وَتَشَنُّجِ جِلْدِهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏119‏.
ن313هـ=925م

التَّشَنُّجُ الِامْتِلَائِيُّ : دَاءٌ يَحْدُثُ مِنْ أَخْلَاطٍ لَزِجَةٍ بَلْغَمِيَّةٍ تَرْسَخُ فِي الأَعْضَاءِ العَصَبِيَّةِ. [الطّبّ والصّيدلة]

"الشَّرَابُ المُرُّ إِذَا سُقِيَ صِرْفًا عَلَى قَلِيلٍ مِنَ الغِذَاءِ؛ أَعَانَ عَلَى التَّشَنُّجِ الِامْتِلَائِيِّ ، وَالكَثِيرُ المِزَاجِ رَدِيءٌ لِذَلِكَ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏1/‏107‏.
1410هـ=1989م

التَّشَنُّجُ فِي الأَمْرِ: التَّشَدُّدُ وَالتَّوَتُّرُ.

"بَدَأَ المَوْقِفُ الغَرْبِيُّ مِنْ رَسُولِ الإِسْلَامِ  @  يَتَشَكَّلُ فِي إِطَارٍ دِينِيٍّ صِرْفٍ، مُتْرَعٍ بِالتَّعَصُّبِ وَالتَّشَنُّجِ وَالِانْفِعَالِ، مَلِيءٍ بِالحِقْدِ وَالغَضَبِ وَالكَرَاهِيَةِ، تُحِيطُهُ جَهَالَةٌ عَمْيَاءُ، مُتَعَمَّدَةٌ حِينًا، وَغَيْرُ مُتَعَمَّدَةٍ أَحْيَانًا"
عماد الدين خليل
المستشرقون والسيرة النبوية: عماد الدين خليل ، دار ابن كثير، دمشق، بيروت، ط1، (1426ه/ 2005م)، ‏19‏.
1429هـ=2007م

التَّشَنُّجُ السِّيَاسِيُّ : هُوَ حَالَةُ الِاضْطِرَابِ وَالِاهْتِيَاجِ فِي الأَوْسَاطِ السِّيَاسِيَّةِ. [السّياسة]

"مَصَادِرُ الحَرِيرِيِّ وَعُونٍ اعْتَبَرَتْ أَنَّ مُجَرَّدَ اللِّقَاءِ بَيْنَ الِاثْنَيْنِ هُوَ إِنْجَازٌ؛ لِأَنَّهُ يَحُدُّ مِنَ التَّشَنُّجِ السِّيَاسِيِّ الرَّاهِنِ، وَيَجْعَلُ المَنَاخَ أَكْثَرَ قَابِلِيَّةً لِاسْتِيعَابِ التَّطَوُّرَاتِ"
نبيه البرجي

تَشْنِيج[مَصْدَر]

ن193هـ=809م

التَّشْنِيجُ : التَّقَبُّضُ.

"هَذَا جَشِيشُ الأَرُزِّ وَالحِنْطَةِ، وَمَاءُ نُخَالَةِ السَّمِيدِ، وَهُوَ نَافِعٌ لِلْأَطْرَافِ المُعْوَجَّةِ، وَتَشْنِيجِ الأَعْصَابِ، وَيُصَفِّي البَشَرَةَ، وَيَذْهَبُ بِالكَلَفِ، وَيُسَمِّنُ البَدَنَ، وَيَجْلُو الأَوْسَاخَ"
هارون الرّشيد
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏8/‏351‏.
1444هـ=2022م

تَشْنِيجُ الأَمْرِ: جَعْلُهُ مُضْطَرِبًا مُتَهَيِّجًا.

"هَكَذَا دَخَلَتِ الأَحْزَابُ، بِسَبَبِ القُوَى التَّقَدُّمِيَّةِ، فِي مُنَافَسَةٍ، لَا عَلَى البَرَامِجِ وَالمَشَارِيعِ، وَلَكِنْ عَلَى الهُوِيَّاتِ وَالشَّرْعِيَّاتِ... ذلِكَ أَنَّ تَشْنِيجَ المُنَافَسَةِ الدَّاخِلِيَّةِ فِي الأَحْزَابِ قَدْ أَضْعَفَ اللُّحْمَةَ بَيْنَهَا"
أيمن البوغانمي
الشعب يريد حين تأكل الديمقراطية نفسها: أيمن البوغانمي ، المركز العربي للأبحاث ودراسة السياسات، بيروت، ط1، (2022م)، ‏69‏.

"المَوْقِفَانِ: عِرْقَانِ مُكْتَنِفَا القُحْقُحِ، إِذَا انْشَنَجَا لَمْ يَقُمِ الإِنْسَانُ، وَإِذَا قُطِعَا مَاتَ"

مُتَشَنِّج[اسْمُ فَاعِل]

235هـ=849م

المُتَشَنِّجُ مِنَ الأَعْضَاءِ: المُتَقَبِضُّ.

"وَالآذَانُ البَارِدَةُ المُتَشَنِّجَة ، وَشُحُومُ الآذَانِ المُتَقَلِّبَة، وَجُلُودُ الجِبَاهِ المُتَمَرِّدَة، وَالأَلْوَانُ المُخْضَرَّةُ أَوِ المُسْوَدَّة؛ كُلُّهَا رِدْيَةٌ تَدُلُّ عَلَى المَوْتِ"
ابن سهل الطَّبريّ
فردوس الحكمة في الطّب: ابن سهل الطَّبريّ (ت، 247هـ)، اعتنى بنسخه وتصحيحه: محمّد زبير الصديقي، مطبع آفتاب، برلين، (1928م)، ‏313‏.

مُتَشَنِّج[صِفَةٌ مُشَبَّهَة]

ن313هـ=925م

المُتَشَنِّجُ مِنَ الأَشْخَاصِ: المُصَابُ بِدَاءِ الشُّنَاجِ أَوْ أَحَدِ صُوَرِ التَّشَنُّجِ.

"وَقَدْ يَبْرَأُ المُتَشَنِّجُ وَالمَفْلُوجُ بِالخَوْضِ فِي العُيُونِ الحَامِيَةِ بُرْءًا سَرِيعًا يُعْجَبُ مِنْهُ"

1397هـ=1976م

المُتَشَنِّجُ مِنَ الأُمُورِ: المُتَشَدِّدُ المُتَوَتِّرُ.

علي مزاحم عباس

مُشْنِج[اسْمُ فَاعِل]

352هـ=963م

مُشْنِجُ الأَنَامِلِ وَنَحْوِهَا: المُنْقَبِضُ مِنْهَا.

قَامَ إِلَيْهَا مُشْنِجُ الأَنَامِلِ أَغْثَى، خَبِيثُ الرِّيحِ بِالأَصَائِلِ
من شواهد الأزهريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏10/‏541‏.

شَنَاج[مَصْدَر]

ن391هـ=1001م

شَنَاجُ الجِلْدِ وَنَحْوِهِ: تَقَبُّضُهُ.

يُعَرِّقُ خَدِّي جَفَافُ الهُزَالِ وَحَافَ الشَّنَاجُ عَلَى وَجْنَتِي
وَقَوَّسَنِي الهَمُّ حَتَّى انْطَوَيْتُ فَصِرْتُ كَأَنِّي أَبُو جَدَّتِي
ابن الحجّاج النِّيلي البغداديّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏3/‏64‏.

شُنَاج[مَصْدَر]

574هـ=1178م

الشُّنَاجُ مِنَ الأَدْوَاءِ: التَّقَبُّضُ الَّذِي يُصِيبُ العَضَلَاتِ، فَلَا يُسْتَطَاعُ التَّحَكُّمِ بِهَا، وَتَتَيَبَّسُ. [الطّبّ والصّيدلة]

"وَفِيهِ بَدَأَ الطَّاعُونُ بِبَغْدَادَ وَنَوَاحِيهَا... ثُمَّ عَرَضَ لَهُمْ شُنَاجٌ وَبِرْسَامٌ وَصُدَاعٌ، وَكَانَ الأَطِبَّاءُ يَصِفُونَ مَعَ هَذِهِ الأَمْرَاضِ أَكْلَ اللَّحْمِ لِحِفْظِ القُوَّةِ، فَإِنَّهُمْ مَا كَانَتْ تَزِيدُهُمُ الحِمْيَةُ إِلَّا قُوَّةَ مَرَضٍ"
ابن الجوزيّ
المنتظم في تاريخ الملوك والأمم: ابن الجوزيّ (ت، 597هـ)، دراسة وتحقيق: محمد عبد القادر عطا، مصطفى عبد القادر عطا، راجعه وصححه: نعيم زرزور، دار الكتب العلمية، بيروت، ط2، (1415ه/ 1995م)، ‏16/‏240‏.

مُشَنِّج[صِفَةٌ مُشَبَّهَة]

1321هـ=1902م

المُشَنِّجُ مِنَ الأَسْمَاكِ: الَّذِي تَصْدُرُ عَنْهُ طَاقَةٌ كَهْرَبَائِيَّةٌ إِذَا ذُعِرَ، أَوْ مُسَّ. [الأحياء]
