رَاشَى[مُتَعَدٍّ]

قبل 50ق.هـ=573م

رَاشَاهُ : دَارَاهُ وَصَانَعَهُ.

نَزِيفٌ إِذَا قَامَتْ لِوَجْهٍ تَمَايَلَتْ تُرَاشِي الفُؤَادَ الرَّخْصَ أَلَّا تَخَتَّرَا
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏61‏.
11هـ=632م

رَاشَى الشَّخْصَ: أَعْطَاهُ رِشْوَةً.

" أَتُرَاشُونِي عَنْ دِينِي؟"
امرأة عجوز من بني عديّ
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏3/‏167‏.
ن283هـ=896م

رَاشَاهُ الشَّيْءَ: أَعْطَاهُ إِيَّاهُ، وَأَعَانَهُ بِهِ.

مَا أَكْرَمَ البَيْهَقِيَّ مِنْ رَجُلٍ! كَمْ مِنْ نَدِيمٍ لَهُ وَمِنْ غَاشِ
كَسَّبْتُهُ صُحْبَةَ المُلُوكِ بِشَتْـ ـمِيهِ، فَرَاشُوهُ خَيْرَ أَرْيَاشِ!
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏1254‏.

رِشَاء[اسْم]

قبل 50ق.هـ=573م

الرِّشَاءُ : الحَبْلُ.

وَأَعْدَدْتُ لِلْحَرْبِ وَثَّابَةً جَوَادَ المَحَثَّةِ وَالمَرْوَدِ
وَمُطَّرِدًا كَرِشَاءِ الجَرُو رِ، مِنْ خُلُبِ النَّخْلَةِ الأَجْرَدِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏88‏.
ن175هـ=791م

الرِّشَاءُ : مَا يَمْتَدُّ مِنْ أَغْصَانِ شَجَرِ الحَنْظَلِ وَالبِطِّيخِ وَنَحْوِهِمَا، كَأَنَّهُ سَيْرٌ أَوْ حَبْلٌ.

" وَأَرْشِيَةُ شَجَرِ الحَنْظَلِ وَالبِطِّيخِ، وَمَا يُشْبِهُهُ: سُيُورُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏281‏.
ن387هـ=997م

الرِّشَاءُ : أَحَدُ مَنَازِلِ القَمَرِ، وَيُسَمَّى بَطْنَ الحُوتِ.

"مَنَازِلُ القَمَرِ فِي ضِمْنِ هَذِهِ الصُّورَةِ، وَهِيَ ثَمَانِيَةٌ وَعِشْرُونَ مَنْزِلًا: أَوَّلُهَا: الشَّرَطَانِ...، ثُمَّ البُطَيْنُ، ثُمَّ الثُّرَيَّا، ثُمَّ الدَّبَرَانُ...، ثُمَّ الرِّشَاءُ ، وَيُقَالُ لَهُ أَيْضًا: بَطْنُ الحُوتِ"
محمد بن أحمد البَلْخيّ الخوارزميّ
مفاتيح العلوم: محمد بن أحمد البَلْخيّ الخوارزميّ (ت، 387هـ)، حققه: إبراهيم الأبياريّ، دار الكتاب العربي، بيروت، ط2، (1409هـ/ 1989م)، ‏238 ‏-‏ 239‏.

أَرْشَى[مُتَعَدٍّ]

قبل 13ق.هـ=609م

أَرْشَى السِّلَاحَ إِلَى القَوْمِ: صَوَّبَهُ نَحْوَهُمْ، وَأَشْرَعَهُ فِيهِمْ.

وَأَمْنَعُ مَنْ أَرْشَى إِلَيْهِمْ سِلَاحَهُ وَأَرْفَعُ يَوْمَ الضَّرْبِ بِالسَّيْفِ مِعْصَمِي
مرداس بن أبي عامر السُّلَميّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏31‏.
ن179هـ=795م

أَرْشَى الشَّخْصَ: أَعْطَاهُ رِشْوَةً.

"وَإِنْ كَانَ الَّذِي أَقَرَّ بِقَتْلِهِ مِنَ الأَبَاعِدِ، مِمَّنْ لَا يُتَّهَمُ فِيهِ؛ رَأَيْتُ أَنْ يُقْبَلَ قَوْلُهُ، إِذَا كَانَ ثِقَةً مَأْمُونًا، وَلَمْ يُخَفْ أَنْ يَكُونَ أُرْشِيَ عَلَى ذَلِكَ؛ لِيُحَابِيَ بِهِ أَحَدًا"
مالك بن أنس الأصبحيّ الحِمْيريّ
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏6/‏406‏.
ن187هـ=803م

أَرْشَى الشَّخْصَ: كَافَأَهُ.

" أَرْشِهِ عَلَيْهَا"
الفُضيل بن عِياض اليَرْبوعيّ التَّميميّ
التّبصرة: عليّ بن محمّد اللَّخْميّ (ت، 478هـ)، دراسة وتحقيق: أحمد عبد الكريم نجيب، وزارة الأوقاف والشؤون الإسلامية، الدّوحة، ط1، د.ت. ‏1/‏391‏.

أَرْشَى[مُتَعَدٍّ بِالحَرْف]

ن125هـ=743م

أَرْشَى لَهُ: دَنَا مِنْهُ، وَمَدَّ حَبْلَ الوَصْلِ إِلَيْهِ.

فَأَضْحَتْ دَارُهَا مِنْهَا قِفَارًا قَطُوعُ الوُدِّ، لَا تُرْشِي لِمُرْشِي
النّابغة الشَّيْبانيّ
ديوان نابغة بني شَيْبان: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1351هـ/ 1932م)، ‏20‏.
ن206هـ=821م

أَرْشَى المَطَرُ وَنَحْوُهُ بِالمَكَانِ: أَصَابَهُ وَنَزَلَ بِهِ.

أَرَأَيْتَ عَزَّةَ، أَمْ رَأَيْتَ غَمَامَةً غَرَّاءَ بَيْنَ أَكِلَّةٍ وَحِجَالِ
أَمْ رَوْضَةً رَجَبِيَّةً أَرْشَى بِهَا طَفَلٌ بِغِبِّ دُجُنَّةٍ وَطِلَالِ؟
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏37‏.

رَشَا[مُتَعَدٍّ]

قبل 13ق.هـ=609م

رَشَا الشَّخْصَ: كَافَأَهُ.

فَمَنْ يَهْدِي أَخًا لِذِنَابِ لَوٍّ؟ فأَرْشُوهُ ، فَإِنَّ اللهَ جَارُ
عَديّ بن زيد العِباديّ
ديوان عَديّ بن زيد العِباديّ: حققه وجمعه: محمّد جبار المعيبد، شركة دار الجمهورية، بغداد، (1385هـ/ 1965م)، ‏133‏.
7ق.هـ=615م

رَشَاهُ المَالَ وَنَحْوَهُ: أَعْطَاهُ إِيَّاهُ رِشْوَةً.

"أَرْجِعُوا إِلَى هَذَا هَدِيَّتَهُ... وَاللهِ لَوْ رَشَوْنِي فِي هَذَا دَبْرَ ذَهَبٍ... مَا قَبِلْتُهُ"
النّجاشي
دلائل النُّبُوَّة ومعرفة أحوال صاحب الشّريعة: أبو بكر البَيْهقيّ (ت، 458هـ)، وثق أصوله وخرج حديثه وعلق عليه: عبد المعطي قلعجي، دار الكتب العلمية، بيروت، دار الريان للتراث، القاهرة، ط1، (1408هـ/ 1988م)، ‏2/‏296‏.
ن760هـ=1358م

رَشَاهُ بِالشَّيْءِ: أَمَدَّهُ وَزَوَّدَهُ بِهِ.

"يَكُونُ مُرَادُهُ أَنَّ الأَغْصَانَ تَرْشُوهُ بِزَهْرِهَا لِيَغْذُوَهَا بِمَائِهِ"
الشّريف السَّبْتيّ
رفع الحجب المستورة عن محاسن المقصورة: الشّريف السَّبْتيّ (ت، 760هـ)، تحقيق وشرح: محمد الحجوي، وزارة الأوقاف والشؤون الإسلامية، الرباط، (1418ه/ 1997م)، ‏2/‏464‏.

رَشَا[لَازِم]

ن291هـ=904م

رَشَا الطَّيْرُ: مَدَّ رَأْسَهُ إِلَى أُمِّهِ لِتُطْعِمَهُ.

"الرُّشْوَةُ مَأْخُوذَةٌ مِنْ رَشَا الفَرْخُ إِذَا مَدَّ رَأْسَهُ إِلَى أُمِّهِ لِتَزُقَّهُ"
أبو العبّاس ثعلب
لسان العرب: ابن منظور (ت، 711هـ)، دار صادر، بيروت، ط1، (د.ت)، ‏14/‏322‏.

رَشَا[مُتَعَدٍّ بِالحَرْف]

ن750هـ=1349م

رَشَا لَهُ الأَمْرُ: عَرَضَ لَهُ.

شَمُولُ عُقَارٍ فِي أَكُفِّ أَهِلَّةٍ لَهَا لَهَبٌ، وَهْمُ الظَّلَامِ بِهَا يَرْشُو
صَفيّ الدِين الحِليّ
ديوان صفي الدِّين الحِليّ: صَفيّ الدِين الحِليّ (ت، 750هـ)، تحقيق: كرم البستاني، دار صادر، بيروت، (د.ت)، ‏729‏.

رِشْوَة[اسْم]

قبل 13ق.هـ=609م

الرِّشْوَةُ : المُكَافَأَةُ.

فَلَا ذَا غِنًى يُرْجِئْنَ عَنْ فَضْلِ مَالِهِ وَإِنْ قَالَ: أَخِّرْنِي، وَخُذْ رِشْوَةً أَبَى
حنظلة بن أبي عفر الطّائيّ
شعر طيِّئ وأخبارها في الجاهلية والإسلام: جمع وتحقيق ودراسة: وفاء فهمي السنديوني، دار العلوم، الرِّياض، ط1، (1403هـ/ 1983م)، ‏387‏.
ن11هـ=632م

الرِّشْوَةُ : مَا يُعْطَى لِلْحَاكِمِ وَنَحْوِهِ؛ لِإِبْطَالِ حَقٍّ، أَوْ إِحْقَاقِ بَاطِلٍ. [الفقه]

"وَمَا مِنْ قَوْمٍ يَظْهَرُ فِيهِمُ الرُّشَا إِلَّا أُخِذُوا بِالرُّعْبِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏29/‏356،،‏(رقم الحديث: 17822).

"ثُمَّ بَعَثَ المَأْمُونُ بِرَجُلٍ مِنَ العَجَمِ، يُقَالُ لَهُ: ابْنُ بِسْطَامٍ، فَوَلَّاهُ الشُّرَطَ، فَعَزَلَهُ كَيْدُرُ لِرَشْوَةٍ ارْتَشَاهَا "

مُرْتَشِي[اسْمُ فَاعِل]

ن11هـ=632م

المُرْتَشِي : الَّذِي يَأْخُذُ المَالَ وَنَحْوَهُ؛ لِيُبْطِلَ حَقًّا، أَوْ يُحِقَّ بَاطِلًا.

"لَعْنَةُ اللهِ عَلَى الرَّاشِي وَالمُرْتَشِي "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏11/‏565،،‏(رقم الحديث: 6984).
ن206هـ=821م

"وَكَانَ قَدْ أَطْبَقَ غَيْمٌ عَظِيمٌ ... لَكِنَّ الهَوَاءَ الَّذِي فَوْقَ الغَيْمِ كَانَ رَشْيًا "
فخر الدين الرازي
المباحث المشرقية في علم الإلهيات والطبيعيات: فخر الدين الرازي (ت، 606هـ)، انتشارات بيدار، ط1، (1370ه)، ‏2/‏183‏.

رَشْيَة[اسْم]

ن696هـ=1296م

الرَّشْيَةُ : وَرَمٌ فِي مُوقِ العَيْنِ يَكُونُ فِيهِ قَيْحٌ. [الطّبّ والصّيدلة]

"وَأَمَّا العِلَاجُ ... فَلَا بُدَّ مِنْ رَدِّ الرَّشْيَةِ إِلَى الأَنْفِ"
الكَحّال الحَمَويّ
نور العيون وجامع الفنون: الكَحّال الحَمَويّ (ت، 696هـ)، حققه وعلق عليه علميا: محمد ظافر الوفائي، راجعه وضبطه وزاد في تعليقاته: محمد رواس قلعجي، مركز الملك فيصل للدراسات والبحوث الإسلامية، الرياض، ط1، (1407ه/ 1987م)، ‏245‏.

رَاشِيَة[اسْم]

793هـ=1390م

الرَّاشِيَةُ : المِبْضَعُ يُشَقُّ بِهِ.

"الفَصْدُ: تَفَرُّقُ اتِّصَالٍ إِرَادِيٍّ وَاقِعٍ فِي العِرْقِ بِالرَّاشِيَةِ "
حاجي باشا خضر بن علي الآيديني
شفاء الأسقام ودواء الآلام: حاجي باشا خضر بن علي الآيديني (ت، 820هـ)، تحقيق: سيدة حامد عبد العال وأخريات، مراجعة: عامر النجار، تصدير: أحمد فؤاد باشا، دار الكتب والوثائق القومية، القاهرة، (2016- 2018م)، ‏1/‏361‏.

مُرْتَشَى[اسْمُ مَفْعُول]

1369هـ=1949م

المُرْتَشَى مِنَ الأَشْيَاءِ: المُعْطَى رِشْوَةً.

وَالمَعْبَدُ مَلْآنٌ بِالرَّوَائِحِ الزَّكِيَّةِ كَلَحْمِ الأَطْفَالِ وَمَلْآنٌ بِأَشْيَاءَ أُخْرَى مُرْتَشَاةٍ وَغَنِيَّةٍ وَمُنْتَصِرَةٍ
عبد الفتّاح الدّيدي
بودلير وفنّ الشّعر: ‏عبد الفتّاح الدّيدي،‏مجلة الرسالة: الرسالة، القاهرة، س17، ع860، (26/ 12/ 1949م)، ‏1761‏.

رَشْوِيَّة[مَصْدَرٌ صِنَاعِيّ]

1377هـ=1957م

الرَّشْوِيَّةُ : الاِرْتِشَاءُ.

"أَرْسَلَ المُسْتَعْمِرُونَ أَسَاطِيلَهُمْ ... بَحْثًا فِيمَا زَعَمُوا عَنِ النَّخَّاسِينَ وَالأَرِقَّاءِ، وَلَكِنَّهُمْ لَمْ يَفْعَلُوا ذَلِكَ إِلَّا بَعْدَ أَنْ أَصْبَحَ تَحْرِيرُ الرَّقِيقِ مَأْثُرَةً يَدَّعِيهَا المُدَّعِي، وَيَرْشُو بِهَا الضَّمَائِرَ، فَتُقْبَلُ الرَّشْوِيَّةُ خَالِصَةَ النِّيَّةِ"
عباس محمود العقاد
لا شيوعية ولا استعمار: عباس محمود العقاد (ت، 1384هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2014م)، ‏90‏.
