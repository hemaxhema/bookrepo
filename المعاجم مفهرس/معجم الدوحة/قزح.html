قُزْحَة[اسْم]

قبل 50ق.هـ=573م

القُزْحَةُ : الطَّرِيقَةُ المُلَوَّنَةُ الَّتِي تَبْدُو فِي قَوْسِ السَّحَابِ أَيَّامَ الرَّبِيعِ.

قُلْتُ لَمَّا نَصَلَا مِنْ قُنَّةٍ: كَذَبَ العَيْرُ وَإِنْ كَانَ بَرَحْ
وَتَرَى خَلْفَهُمَا إِذْ مَضَيَا مِنْ غُبَارٍ سَاطِعٍ قَوْسَ قُزَحْ
أبو دُواد الإياديّ
ديوان أبي دواد الإياديّ: جمعه وحققه: أنوار محمود الصالحي، أحمد هاشم السامرائي، دار العصماء، دمشق، ط1، (1431هـ/ 2010م)، ‏74‏.

قَزَحَ[مُتَعَدٍّ]

ن11هـ=632م

قَزَحَ الطَّعَامَ: أَضَافَ إِلَيْهِ أَبْزَارَ التَّوَابِلِ.

"إِنَّ مَطْعَمَ ابْنِ آدَمَ جُعِلَ مَثَلًا لِلدُّنْيَا، وَإِنْ قَزَحَهُ وَمَلَحَهُ، فَانْظُرُوا إِلَى مَا يَصِيرُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏35/‏161،،‏(رقم الحديث: 21239).

قَزَحَ[مُتَعَدٍّ بِالحَرْف]

ن255هـ=869م

قَزَحَ الكَلْبُ بَوْلَهُ: رَمَى بِهِ دَفْعَةً.

"فَالتَّيْسُ كَالكَلْبِ؛ لِأَنَّهُ يَقْزَحُ بِبَوْلِهِ"
الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏5/‏465‏.

قَزَحَ[لَازِم]

749هـ=1348م

قَزَحَ : الشَّيْءُ صَلُبَ وَاشْتَدَّ.

وَضَاقَتِ الأَنْفَاسُ مِنْ فَرْطِ مَا يَنْدِفُ مِنْ قُطْنٍ، وَظَهْرٍ قَزَحْ
ابن نُباتة المصريّ
كتاب الشتويات: ابن فضل الله العُمريّ (ت، 749هـ)، تحقيق وتقديم: حسن محمد عبد الهادي، عالم الكتب، القاهرة، ط1، (2017م)، ‏72‏.
1327هـ=1908م

قَزَحَ الشَّخْصُ: قَفَزَ.

"وَقَلَّدَ الرُّومَانُ مَدَارِسَ أَفْلَاطُونَ وَسُقْرَاطَ مِنْ أَهْلِ أَثِينَا، فَلَمْ يُفْلِحُوا فِي تَعَالِيمِهِمْ، ثُمَّ يَقْزِحُونَ خُطْوَةً، وَيَنْسَوْنَ عُلَمَاءَ الإِسْلَامِ الَّذِينَ هُمْ أَصْلُ نِعْمَتِهِمْ"
طنطاوي جوهري
نهضة الأمة وحياتها: طنطاوي جوهري (ت، 1358هـ)، تقديم: حازم زكريا محيي الدين، دار الكتاب المصري- القاهرة، دار الكتاب اللبناني- بيروت، ط2، (1433ه/ 2012م)، ‏212‏-‏ 213‏.

قَزَّاح[صِيغَةُ مُبَالَغَة]

ن21هـ=642م

القَزَّاحُ مِنَ الكِلَابِ: الشَّدِيدُ الدَّفْعِ بِبَوْلِهِ وَالرَّمْيِ بِهِ.

إِذَا تَخَازَرْتُ، وَمَا بِي مِنْ خَزَرْ وَجَدْتَنِي أَلْوَى، بَعِيدَ المُسْتَمَرْ أَبْذَى، إِذَا بُوذِيتُ، مِنْ كَلْبٍ ذَكَرْ أَسْوَدَ قَزَّاحٍ يُعَوِّي فِي السَّحَرْ
الأغلب بن عمرو العِجْليّ
الأغلب العجلي حياته وشعره: تحقيق: نوري حمّودي القيسيّ، مجلة المجمع العلمي العراقي، مج31، ج3، (1400هـ)، ‏131‏.
855هـ=1451م

القَزَّاحُ : مَنْ حِرْفَتُهُ بَيْعُ القَزْحِ.

"وَإِذَا وُضِعَ البَصَلُ بَيْنَ حَجَرَيِ الطَّاحُونِ مَنَعَهُ أَنْ يُدُورَ، وبِزْرُهُ يُسَمَّى: القِزْحَ بِالكَسْرِ، وَبَائِعُهُ القَزَّاح "
ابن داود الدمشقي
نزهة النفوس والأفكار في خواص الحيوان والنبات والأحجار: ابن داود الدمشقي (ت، 856هـ)، تحقيق: أشرف غنام، تقديم ومراجعة: عامر النجار، دار الكتب والوثائق القومية، القاهرة، (1438ه/ 2017م)، ‏301‏.

قَازِح[اسْمُ فَاعِل]

ن21هـ=642م

القَازِحُ : المُتَصَلِّبُ اليَابِسُ.

فَبِتُّ أَمْرِيهَا، وَأَدْنُو لِلثُّنَنْ بِقَازِحِ الجِلْدِ، مَتِينٍ كَالرَّسَنْ
الأغلب بن عمرو العِجْليّ
الأغلب العجلي حياته وشعره: تحقيق: نوري حمّودي القيسيّ، مجلة المجمع العلمي العراقي، مج31، ج3، (1400هـ)، ‏126‏.
ن250هـ=864م

القَازِحُ مِنْ ذُكُورِ الرِّجَالِ: الَّذِي اشْتَدَّ نَعْظُهُ.

وَقَدْ أَكُونُ لِلنِّسَاءِ صَالِحَا إِذَا تَشَكَّيْنَ عُرَامًا آزِحَا أَقْبَلْتُهُنَّ قَيْسَبَانًا قَازِحَا
من شواهد ثابت بن أبي ثابت اللُّغويّ
كتاب خلق الإنسان: ثابت بن أبي ثابت اللُّغويّ (ت، 250هـ)، تحقيق: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، ط2، د.ت. ‏288‏.

قَزَّحَ[مُتَعَدٍّ]

ن30هـ=651م

قَزَّحَ الطَّعَامَ: أَضَافَ إِلَيْهِ أَبْزَارَ التَّوَابِلِ.

"أَلَا إِنَّ مَطْعَمَ ابْنِ آدَمَ ضُرِبَ مَثَلًا لِلدُّنْيَا، أَلَا وَإِنْ قَزَّحَهُ ، وَمَلَّحَهُ"
أُبَيّ بن كعب الخزرجيّ الأنصاريّ
من كتاب الزّهد: أبو حاتم الحَنْظليّ الرّازيّ (ت، 277هـ)، دراسة وتحقيق وتعليق: عامر حسن صبري، دار البشائر الإسلامية، بيروت، ط1، (1422هـ/ 2001م)، ‏47‏.
ن538هـ=1143م

قَزَّحَ الشَّيْطَانُ فِي الأُذْنِ: زَيَّنَ وَسَاوِسَهُ فِيهَا.

وَمُصْبِحٍ، خَاطَ الكَرَى عَيْنَيْهِ وَقَزَّحَ الشَّيْطَانُ فِي أُذُنَيْهِ
جار الله الزمخشريّ
ديوان جار الله الزّمخشريّ: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: فاطمة يُوسف الخيميّ، دار صادر، بيروت، ط1، (1429ه/ 2008م)، ‏571‏.

قُزَح[اسْم]

ن40هـ=660م

قُزَحُ : الشَّيْطَانُ.

"لَا تَقُولُوا: قَوْسَ قُزَحَ، فَإِنَّ قُزَحَ الشَّيْطَانُ"
عليّ بن أبي طالب
الجامع في الحديث: عبد الله بن وهب بن مسلم القُرَشيّ (ت، 197هـ)، ضبط وتخريج وتحقيق: مصطفى حسن حسين محمّد أبو الخير، دار ابن الجوزي، الرياض، ط1، (1416هـ/ 1995م)، ‏1/‏105‏.

مُقَزِّح[اسْمُ فَاعِل]

ن68هـ=687م

المُقَزِّحُ مِنَ الأَشْجَارِ: الَّذِي تَشَعَّبَ رَأْسُهُ مِثْلَ بُرْثُنِ الكَلْبِ.

"نُهِيَ عَنِ الصَّلَاةِ خَلْفَ الشَّجَرَةِ المُقَزِّحَةِ "
ابن عبّاس
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏2/‏394‏.
ن530هـ=1135م

المُقَزِّحُ مِنَ الأَشْخَاصِ: المُشَعْوِذُ.

"هَذِهِ عِبْرَةٌ عَظِيمَةٌ مُشَاهَدَةٌ فِي الزَّمَانِ، وَالمَكَانِ، دَالَّةٌ عَلَى قُبْحِ التَّفْرِيطِ فِي النَّفْسِ، وَالمُخَاطَرَةِ بِهَا لِلنَّارِ، وَعَلَى قُبْحِ الِاتِّكَالِ عَلَى خُرَافَاتِ المُنَجِّمِينَ وَالكُهَّانِ، وَالمُقَزِّحِينَ لِلْجُهَّالِ بِالمُحَالِ"
اللَّحْجيّ
من كتاب أخبار الزيدية [المُطَرِّفِيَّة]: اللَّحْجيّ (ت، 530هـ)، اعتنى به وضبطه: مقبل التام عامر الأحمدي، الجيل الجديد ناشرون، صنعاء، (2020م)، ‏234‏.

مُقَزَّح[اسْمُ مَفْعُول]

ن68هـ=687م

المُقَزَّحُ مِنَ الشَّجَرِ: الَّذِي بَالَتِ الكِلَابُ وَنَحْوُهَا مِنَ السِّبَاعِ عَلَى أَصْلِهِ.

"كَرِهَ أَنْ يُصَلِّيَ الرَّجُلُ إِلَى الشَّجَرَةِ المُقَزَّحَةِ "
ابن عبّاس
الغريبين في القرآن والحديث: أبو عُبيد الهَرَويّ الفاشانيّ (ت، 401هـ)، تحقيق ودراسة: أحمد فريد المزيدي، قدم له وراجعه: فتحي حجازي، مكتبة نزار مصطفى الباز، مكّة المكرّمة- الرّياض، ط1، (1419هـ/ 1999م)، ‏5/‏1537‏.
ن175هـ=791م

المُقَزَّحُ مِنَ الطَّعَامِ: الَّذِي أُضِيفَ إِلَيْهِ أَبْزَارُ التَّوَابِلِ.

"القَزْحُ: أَبْزَارُ القِدْرِ، وَقِدْرٌ مُقَزَّحَةٌ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏38‏.
ن440هـ=1048م

المُقَزَّحُ مِنَ الأَلْوَانِ: المُشَبَّهُ بِأَلْوَانِ قَوْسِ قُزَحٍ.

"حَجَرُ الأَلْمَاسِ: وَهُوَ جَوْهَرٌ مُشِفٌّ فِي لَوْنِهِ كَالزُّجَاجِيَّةِ، وَمِنْهُ مَا يَضْرِبُ إِلَى الصُّفْرَةِ، وَيَنْبَعِثُ مِنْهُ فِي ضِيَاءِ الشَّمْسِ أَلْوَانٌ مُقَزَّحَةٌ "
أبو الرّيحان البيرونيّ
كتاب الصّيدنة في الطّبّ: أبو الرّيحان البيرونيّ (ت، 440هـ)، تصحيح ومقدمة وتحشية: عباس زرياب، مركز نشر دانشكاهي، طهران، ط1، (1370هـ)، ‏72‏.

مُقَزَّح[اسْم]

ن231هـ=845م

المُقَزَّحُ : شَجَرٌ عَلَى صُورَةِ التِّينِ، لَهُ غُصُونٌ قِصَارٌ فِي رُؤُوسِهَا؛ مِثْلُ بُرْثُنِ الكَلْبِ.

"وَمِنْ غَرِيبِ شَجَرِ البَرِّ: المُقَزَّحُ "
أبو عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏29‏.

قَازِحَة[اسْم]

ن130هـ=748م

القَازِحَةُ : نُفَّاخَةُ المَاءِ، تَنْتَفِخُ فَتَظْهَرُ ثُمَّ تَذْهَبُ.

فَلَمْ أَرَ قَوْمًا مِثْلَ قَوْمِيَ، إِذْ هُمُ بِأَوْطَانِهِمْ أَعْطَى وَأَغْلَى المَرَابِحِ
لَهُمْ حَاضِرٌ لَا يَجْهَلُونَ، وَصَارِخٌ كَسَيْلِ الغَوَادِي تَرْتَمِي بِالقَوَازِحِ
أبو وَجْزة السُّلَميّ السَّعْديّ
شعر أبي وَجْزة السَّعْديّ (ت، 130هـ): تحقيق: وليد السراقبي، مجلة معهد المخطوطات العربیة، القاهرة، مج34، ج(1، 2)، (1410هـ)، ‏47‏.

قَزْح[اسْم]

ن175هـ=791م

القَزْحُ : التَّابَلُ الَّذِي يُطْرَحُ فِي القِدْرِ مَعَ الطَّعَامِ.

" القَزْحُ : أَبْزَارُ القِدْرِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏38‏.

قَزْح[مَصْدَر]

ن1078هـ=1667م

القَزْحُ : القَفْزُ.

لَوْ رَأَوْهُ فِي الكَرَى لَانْتَبَهُوا وَلَهُمْ؛ مِنْ خَوْفِهِ بِالرُّعْبِ، قَزْحُ
محمد بن حجازي بن أحمد بن محمد الرقباوي الأنبابي
نفحة الرّيحانة ورشحة طلاء الحانة: محمد الأمين بن فضل الله المحبي (ت، 1111هـ)، تحقيق: عبد الفتاح محمَّد الحلو، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، ط1، (1387ه/ 1968م)، ‏602‏.

تَقْزِيح[مَصْدَر]

ن175هـ=791م

تَقْزِيحُ النَّبَاتِ: انْشِعَابُ أَغْصَانِهِ؛ مِثْلَ بُرْثُنِ الكَلْبِ.

" التَّقْزِيحُ فِي رَأْسِ شَجَرَةٍ أَوْ نَبْتٍ: إِذَا انْشَعَبَ شُعَبًا مِثْلَ بُرْثُنِ الكَلْبِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏38‏.
ن200هـ=815م

تَقْزِيحُ الطَّعَامِ: تَلْوِينُهُ بِالأَبْزَارِ.

"ضِفْنَا حَيًّا مِنَ العَرَبِ، فَقَدَّمُوا لَنَا غَوْطَةً دَكْنَاءَ مِنَ التَّقْزِيحِ ، وَهِيَ تَرْذُمُ"
أبو ثَرْوان العُكْليّ الوحشيّ البدويّ
المثلث: ابن السيّد البَطَلْيَوسيّ (ت، 521هـ)، تحقيق ودراسة: صلاح مهدي الفرطوسي، دار الرشيد للنشر، بغداد، (1401ه/ 1981م)، ‏2/‏11‏.
ن606هـ=1209م

التَّقْزِيحُ : التَّحْسِينُ، وَالتَّزْيِينُ.

"سُمِّيَ بِهِ؛ لِتَسْوِيلِهِ لِلنَّاسِ وَتَحْسِينِهِ إِلَيْهِمْ المَعَاصِيَ، مِنَ التَّقْزِيحِ "
ابن الأثير أبو السعادات
النهاية في غريب الحديث والأثر: ابن الأثير أبو السعادات (ت، 606هـ)، تحقيق: محمود محمد الطناحي، طاهر أحمد الزاوي، المكتبة الإسلامية، بيروت، ط1، (1383ه/ 1963م)، ‏4/‏57‏.

تَقَزَّحَ[مُتَعَدٍّ بِالحَرْف]

ن206هـ=821م

تَقَزَّحَ عَلَى السَّائِلِ: مَنَعَهُ، مُتَذَرِّعًا بِعُدْمِهِ.

"سَأَلْتُهُ فَتَقَزَّحَ عَلَيَّ"
أبو عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏81‏.

تَقَزَّحَ[لَازِم]

418هـ=1027م

تَقَزَّحَتْ هَالَةُ الشَّمْسِ وَنَحْوُهَا: تَلَوَّنَتْ طَرَائِقَ مِثْلَ قَوْسِ قُزَحَ.

"وَإِنَّمَا تَتَقَزَّحُ هَالَةُ الشَّمْسِ أَحْيَانًا، إِذَا كَثُفَ السَّحَابُ، وَأَظْلَمَ"
ابن سينا
الشّفاء الطّبيعيّات: ابن سينا (ت، 428هـ)، ج(1- 3)، تحقيق: سعيد زايد وآخرين، تصدير ومراجعة: إبراهيم مدكور، منشورات مكتبة آية الله العظمى المرعشي النجفي، قم، دار الكتاب العربي- الهيئة المصرية العامة للتأليف والنشر، القاهرة، ط2، (1406هـ)، ‏2- 2/‏49‏.

قَزَح[مَصْدَر]

ن206هـ=821م

القَزَحُ : صَلَابَةُ ذَكَرِ الرَّجُلِ وَشِدَّةُ إِنْعَاظِهِ.

ثُمَّ اعْتَلَاهَا قَزَحًا ، وَارْتَهَزَا وَسَاقَهَا ثَمَّ سِيَاقًا بَزْبَزَا
من إنشاد أبي عمرو الشّيبانيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏13/‏173‏.

مَقْزُوح[اسْمُ مَفْعُول]

330هـ=941م

المَقْزُوحُ مِنَ الطَّعَامِ: المَخْلُوطُ بِأَبْزَارِ التَّوَابِلِ.

"فَمَعْنَى قَوْلِهِمْ: مَلِيحٌ قَزِيحٌ: كَامِلُ الحُسْنِ؛ لِأَنَّ كَمَالَ طِيبِ القِدْرِ أَنْ تَكُونَ مَقْزُوحَةً مَمْلُوحَةً"
أبو عليّ القالي
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏2/‏235‏.

قَزِيح[صِفَةٌ مُشَبَّهَة]

ن351هـ=962م

القَزِيحُ مِنَ الطَّعَامِ: المُطَيَّبُ بِالأَبْزَارِ.

"وإِنَّهُ لَمَليحٌ قَزِيحٌ "
من شواهد أبي الطّيّب اللُّغويّ الحلبيّ
كتاب الإتباع: أبو الطّيّب اللُّغويّ الحلبيّ (ت، 351هـ)، حققه وشرحه وقدم له: عز الدين التنوخي، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏71‏.

قُزَاح[اسْم]

ن416هـ=1025م

القُزَاحُ : قَوْسُ قُزَحَ.

وَمْجَهَلٍ مُشْتَبِهٍ طُرْقُهُ كَأَنَّمَا هُنَّ خُطُوطُ قُزَاحْ
أبو الحسن التِّهاميّ
الذخيرة في محاسن أهل الجزيرة: ابن بسّام الشّنْتَرِيني (ت، 542هـ)، تحقيق: إحسان عبّاس، دار الثَّقافة، بيروت، ط1، (1417ه/ 1997م)، ‏4/‏538‏.

قُزَحِيّ[صِفَة]

418هـ=1027م

القُزَحِيُّ مِنَ الهَالَاتِ وَنَحْوِهَا: الشَّبِيهُ بِقَوْسِ قُزَحَ فِي طَرَائِقِهِ وَأَلْوَانِهِ.

"وَمَعَ ذَلِكَ فَقَدْ تَكُونُ حَوْلَ الشَّمْسِ هَالَةٌ، وَهُوَ الطُّفَاوَةُ، وَذَلِكَ فِي النُّدْرَةِ. وَالَّتِي تَكُونُ مِنَ الهَالَاتِ تَحْتَ الشَّمْسِ أَدَلُّ عَلَى المَطَرِ مِنَ الخَيَالَاتِ القُزَحِيَّةِ الَّتِي تَكُونُ قُبَالَتَهَا"
ابن سينا
الشّفاء الطّبيعيّات: ابن سينا (ت، 428هـ)، ج(1- 3)، تحقيق: سعيد زايد وآخرين، تصدير ومراجعة: إبراهيم مدكور، منشورات مكتبة آية الله العظمى المرعشي النجفي، قم، دار الكتاب العربي- الهيئة المصرية العامة للتأليف والنشر، القاهرة، ط2، (1406هـ)، ‏2- 2/‏49‏.

قُزَحِيّ[اسْمٌ مَنْسُوب]

1256هـ=1839م

القُزَحِيُّ مِنْ أَغْشِيَةِ العَيْنِ: المَنْسُوبُ إِلَى قُزَحِيَّتِهَا.

"وَمِنَ المَعْلُومِ أَنَّ فِي العَيْنِ أَغْشِيَةً، وَمَوَادَّ، فَالأَغْشِيَةُ خَمْسَةٌ، وَهِيَ: اللِّيفِيُّ اليَابِسُ، وَالقَرَنِيُّ الشَّفَّافُ، وَالعِنَبِيُّ، وَالقُزَحِيُّ ، وَالشَّبَكِيُّ"
يوسف فرعون
كتاب الفيسلوجيا: لافارج: يوسف فرعون (ت، 1265هـ)، تحقيق: مصطفى حسن كساب، ترجمة: يوسف فرعون، مطبعة بولاق، القاهرة، (1256ه)، ‏16‏.

تَقْزِيحَة[اسْم]

ن430هـ=1038م

التَّقْزِيحَةُ : الطَّرِيقَةُ المُلَوَّنَةُ مِنْ طَرَائِقِ قَوْسِ قُزَحَ.

"وَقَدِ اعْتَقَدَ قَوْمٌ أَنَّ اللَّوْنَ لَا حَقِيقَةَ لَهُ، وَأَنَّهُ شَيْءٌ يَعْرِضُ بَيْنَ البَصَرِ وَالضَّوْءِ كَمَا تَعْرِضُ التَّقَازِيحُ ، وَلَيْسَ اللَّوْنُ صُورَةً فِي الجِسْمِ المُتَلَوِّنِ"
ابن الهيثم
كتاب المناظر للحسن بن الهيثم المقالات 1- 2- 3 في الإبصار على الاستقامة: ابن الهيثم (ت، 430هـ)، حققها وراجعها: عبد الحميد صبرة، المجلس الوطني للثقافة والفنون والآداب، الكويت، (1404هـ/ 1983م)، ‏116‏.

قُزُوح[مَصْدَر]

436هـ=1044م

"وَقَدْ قِيلَ: إِنَّ اشْتِقَاقَ قُزُوحِ الكَلْبِ مِنْهُ، وَلَيْسَ بِقَوِيٍّ مِنَ الِاشْتِقَاقِ"

قُزَّاح[اسْمُ جِنْس]

ن646هـ=1248م

القُزَّاحُ : نَبَاتٌ يُشْبِهُ نِسْبِيًّا الرَازْيَانْجَ شَكْلًا وَطَعْمًا، إِلَّا أَنَّ أَغْصَانَهُ أَصْغَرُ، وَأَكْثَرُ تَشَعُّبًا وَتَدَاخُلًا، وَهِيَ مُزَوَّاةٌ عَلَى أَطْرَافِهَا زَهْرٌ أَصْفَرُ، وَكُلُّهُ عَطِرُ الرَّائِحَةِ طَيِّبُ الثَّمَرِ.

"وَيَسْتَعْمِلُهُ أَهْلُ الجِهَةِ فِي التَّوَابِلِ فِي مَاءِ الشَّرَابِ؛ لِطِيبِ رَائِحَتِهِ، وَأَهْلُ البَوَادِي بِالقَيْرَوَانِ، وَأَعْمَالِ المَهْدِيَّةِ، وَمَا هُنَالِكَ يُسَمُّونَهُ بِالقُزَّاحِ أَيْضًا"
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏3/‏452‏.

مُتَقَزِّح[اسْمُ فَاعِل]

ن719هـ=1319م

المُتَقَزِّحُ مِنَ الأَشْيَاءِ: الآخِذُ لَوْنَ قَوْسِ قُزَحَ.

"ثُمَّ تَتَشَكَّلُ حَوْلَ الدَّائِرَةِ مِنْ أَجْلِ انْعِطَافِ صُورَةِ النَّيِّرِ إِلَى البَصَرِ فِي طَبَقَتَيْنِ مِنَ الرَّذَاذِ ـ حَلْقَةٌ مُتَقَزِّحَةٌ عَلَى عَرْضِ الطَّبَقَةِ الأُولَى المَذْكُورِ بَيِّنَةَ التَّقَازِيحِ"
كمال الدين الفارسيّ
كتاب البصائر في علم المناظر: كمال الدين الفارسيّ (ت، 719هـ)، تحقيق: مصطفى موالدي، مؤسسة الكويت للتقدم العلمي، الكويت، (2009م)، ‏428‏-‏ 429‏.

قُزَحِيَّة[اسْم]

1248هـ=1832م

القُزَحِيَّةُ : غِشَاءُ العَيْنِ الَّذِي يُعْطِيهَا تَلَوُّنًا طَيْفِيًّا،وَيَحْمِيهَا مِنَ الضَّوْءِ، وَيُسَاعِدُهَا فِي عَمَلِيَّةِ الإِبْصَارِ.

"فَإِنَّ عِنْدِي أَدِلَّةً تَحْمِلُ عَلَى ظَنِّ أَنَّ القُزَحِيَّةَ تَتَوَلَّدُ أَوَّلًا بِهَيْئَةِ حَلْقَةٍ تَنْمُو بَعْدَ ذَلِكَ مِنْ مَرْكَزِهَا، حَتَّى أَنَّهَا- فِي الآخِرِ- لَا تَتْرُكُ إِلَّا الفُوَّهَةَ المُسَمَّاةَ بِالحَدَقَةِ"
أحمد حسن الرشيدي
كتاب الولادة: أحمد حسن الرشيدي (ت، 1282هـ)، دار الطباعة العامرة، بولاق، القاهرة، (1258ه)، ‏234‏.

تَقَزُّحِيّ[صِفَة]
