غَتْم[مَصْدَر]

قبل 13ق.هـ=609م

الغَتْمُ : شِدَّةُ الحَرِّ الَّذِي يَكَادُ يَأْخُذُ بِالنَّفَسِ.

حَرَّقَهَا حَمْضُ بِلَادٍ فِلِّ وَغَتْمُ نَجْمٍ غَيْرِ مُسْتَقِلِّ فَمَا تَكَادُ نِيبُهَا تُوَلِّي
مسعود بن قَيْد الفَزاريّ
شعر قبيلة ذُبيان في الجاهلية: جمع وتحقيق ودراسة: سلامة عبد الله السويدي، مطبوعات جامعة قطر، الدوحة، (1408هـ/ 1987م)، ‏479‏.
ن538هـ=1143م

الغَتْمُ : الِاتِّخَامُ: مِنْ كَثْرَةِ الأَكْلِ.

"وَيُقَالُ: لَا تُغْتِمِ الزِّيَارَةَ؛ فَتُمَلَّ: مِنِ أَغْتَمَ الرَّجُلُ إِذَا أَكْثَرَ مِنَ الأَكْلِ حَتَّى أَخَذَهُ الغَتْمُ مِنْ كَرْبِ الكَظَّةِ"
جار الله الزمخشريّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، مطبعة دار الكتب المصرية، القاهرة، (1341ه/ 1923م)، ‏2/‏156 ‏-‏ 157‏.

تَغَتَّمَ[لَازِم]

قبل 13ق.هـ=609م

تَغَتَّمَ الشَّخْصُ: عَجَزَ عَنِ الإِفْصَاحِ.

وَهُمْ خَشُّوا مَعَ الدَّيَّانِ حَتَّى تَغَتَّمَ كُلُّ عُضْرُوطٍ وَعَبْدِ
عمرو بن مَعْدي كَرِب الزُّبَيديّ
شعر عَمرو بن مَعْدي كَرب الزُّبَيْدي: جمعه ونسّقه: مطاع الطرابيشي، مطبوعات مجمع اللغة العربية، دمشق، ط2، (1405هـ/ 1985م)، ‏99‏.

أَغْتَم[صِفَةٌ مُشَبَّهَة]

ن50هـ=670م

الأَغْتَمُ مِنَ النَّاسِ: الَّذِي لَا يُفْصِحُ شَيْئًا؛ لِعُجْمَةٍ فِي مَنْطِقِهِ.

فَهَمَمْتُ، ثُمَّ ذَكَرْتُ لَيْلَ لِقَاحِنَا بِلِوَى عُنَيْزَةَ، أَوْ بِقُفِّ بَشَامِ
إِذْ هُنَّ عَنْ حَسَبِي مَذَاوِدُ كُلَّمَا نَزَلَ الظَّلَامُ بِعُصْبَةٍ أَغْتَامِ
جُبيهاء الأشجعيّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏18/‏70‏.
ن436هـ=1044م

الأَغْتَمُ مِنَ الطَّيْرِ: الَّذِي لَا يَكَادُ يُظْهِرُ صَوْتَهُ إِذَا صَوَّتَ.

"وَهُنَّ غُتْمٌ أَيْضًا؛ إِنَّمَا تُغَطْغِطُ إِحْدَاهُنَّ بِصَوْت فِي حَلْقِهَا، وَإِنَّمَا تُصَوِّت حِينَ تَطِيرُ، ثُمَّ تَقْطَعُ التَّصْوِيتَ"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
كتاب المخصَّص: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، المطبعة الكبرى الأميرية، بولاق، ط1، (1316- 1321هـ)، ‏8/‏157‏.

غُتَيْم[اسْم]

ن100هـ=718م

غُتَيْمُ : المَوْتُ.

وَكُنْتُ امْرَأً، مَنْ يَبْغِ حَرْبِي أَرِدْ بِهِ حِيَاضَ غُتَيْمٍ حَيْثُ يَلْقَى مَنُونَهَا
مدرك بن حصن الفَقْعَسيّ الأسديّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏21‏.

مَغْتُوم[اسْمُ مَفْعُول]

105هـ=723م

المَغْتُومُ : المَلْفُوحُ بِالحَرِّ وَهُوَ جَائِعٌ.

يَا تَيْمُ، إِنَّكَ عَبْدٌ مِنْ بَنِي كُسَعٍ مَا كُنْتَ أَوَّلَ عَبْدٍ، ضَلَّ، مَغْتُومِ
جرير
شرح ديوان جرير: تحقيق: محمّد إسماعيل عبد الله الصاوي، المكتبة التجارية الكبرى- مطبعة الصاوي، القاهرة، ط1، (1935م)، ‏489‏.

غُتْمِيّ[صِفَة]

ن175هـ=791م

الغُتْمِيُّ : الَّذِي لَا يُفْصِحُ شَيْئًا؛ لِعُجْمَةٍ فِي مَنْطِقِهِ.

"الغُتْمَةُ: عُجْمَةٌ فِي المَنْطقِ. وَرَجُلٌ أَغْتَمُ، وَغُتْمِيٌّ ، أَيْ: لَا يُفْصِحُ شَيْئًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏398‏.
ن231هـ=845م

الغُتْمِيُّ : الثَّقِيلُ الرُّوحِ.

"الغُتْمُ: قِطَعُ اللَّبَنِ الثِّخَانُ، وَمِنْهُ قِيلَ لِلثَّقِيلِ الرُّوحِ: غُتْمِيٌّ "
من رواية أبي عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏83‏.
ن571هـ=1175م

الغُتْمِيُّ : الجَاهِلُ.

"فَإِنْ قِيلَ: إِنَّ الجَمَّ الغَفِيرَ فِي سَائِرِ الأَزْمَانِ، وَأَكْثَرَ العَامَّةِ فِي جَمِيعِ البُلْدَانِ لَا يَقْتَدُونَ بِالأَشْعَرِيِّ... قِيلَ: لَا عِبْرَةَ بِكَثْرَةِ العَوَامِّ، وَلَا الْتِفَاتَ إِلَى الجُهَّالِ الأَغْتَامِ "
ابن عساكر
تبيين كذب المفتري فيما نسب إلى الإمام أبي الحسن الأشعري: ابن عساكر (ت، 571هـ)، تحقيق: حسام الدين القدسي، مطبعة التوفيق، دمشق، (1347ه/ 1949م)، ‏331‏.

غُتْمَة[اسْم]

ن175هـ=791م

الغُتْمَةُ : العُجْمَةُ فِي المَنْطِقِ.

" الغُتْمَةُ : عُجْمَةٌ فِي المَنْطِقِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏398‏.

غُتْمَة[مَصْدَر]

ن487هـ=1094م

الغُتْمَةُ : الجَهَالَةُ.

"وَيُرْوَى غَيْرَ أَغْتَمَ بِالتَّاءِ مُعْجَمَةً بِاثْنَتَيْنِ، مِنَ الغُتْمَةِ ، وَهِيَ الجَهَالَةُ"
أبو عُبيد البَكْريّ الأندلسيّ
الجزء الأول من سمط اللآلي ويحتوي على النصف الأول من اللآلي في شرح أمالي القالي: أبو عُبيد البَكْريّ الأندلسيّ (ت، 487هـ)، نسخه وصححه ونقحه وحقق ما فيه واستخرجه من بطون دواوين العلم: عبد العزيز الميمني، دار الكتب العلمية، بيروت، د.ت. ‏1/‏302‏.

أَغْتَمَ[مُتَعَدٍّ]

280هـ=893م

أَغْتَمَ الشِّعْرَ وَنَحْوَهُ: أَكْثَرَهُ، وَأَطَالَهُ.

"وَكَانَ العَجَّاجُ يُغْتِمُ الشِّعْرَ"
من رواية ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏3/‏1288‏.
538هـ=1143م

أَغْتَمَ الزِّيَارَةَ: أَكْثَرَ مِنْهَا حَتَّى سُئِمَ.

"وَيُقَالُ: لَا تُغْتِمِ الزِّيَارَةَ؛ فَتُمَلَّ"
من رواية الزَّمخشريّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، مطبعة دار الكتب المصرية، القاهرة، (1341ه/ 1923م)، ‏2/‏156‏.

اسْتِغْتَام[مَصْدَر]

282هـ=895م

الِاسْتِغْتَامُ : الِاسْتِكْثَارُ.

"يَا مَوْلَانَا، أَخْذِي لِهَذِهِ الثِّيَابِ يُصْبِحُ بِي اسْتِغْتَامًا ، وَالأَمِيرُ أَحْوَجُ مِنِّي إِلَيْهَا"
تاجر سنانير
كتاب المقفى الكبير: تقي الدين أحمد بن علي المقريزي (ت، 845هـ)، تحقيق: محمد اليعلاوي، دار الغرب الإسلامي، بيروت، ط1، (1411ه/ 1991م)، ‏3/‏830‏.

غَتَامَة[مَصْدَر]

ن320هـ=932م

الغَتَامَةُ : الغَبَاءُ، وَثِقَلُ الرُّوحِ.

"فَلَوْ وَقَفَ هَذَا المَدْعُوُّ عَلَى نُزْهَةٍ فِي هَذَا الطَّرِيقِ، فَقَالَ: أَبْنِي هَهُنَا قَصْرًا لِأَمِيرِ المُؤْمِنِينَ، وَأَتَقَرَّبُ بِهِ إِلَيْهِ، لَعُدَّ ذَلِكَ بَلَاهَةً وَغَتَامَةً ، وَانْحَطَّتْ مَنْزِلَتُهُ عِنْدَهُ"
الحكيم التِّرْمِذيّ
رسالة مكر النفس: الحكيم التِّرْمِذيّ (ت، 320هـ)، تحقيق: عبد الفتاح عبد الله بركة، مجلة معهد المخطوطات، القاهرة، مج20، ج2، (1394هـ/ 1974م)، ‏157‏.

مُغْتَتَم[اسْمُ فَاعِل]

ن327هـ=939م

المُغْتَتِمُ : مَنْ يُكْثِرُ الأَكْلَ حَتَّى يُتَّخَمَ.

صَدِيقُكَ غَيْرُ مُحْتَشِمِ وَأَنْتَ فَغَيْرُ مُغْتَتِمِ
عبد الرّحمن بن عيسى الهَمَذانيّ
المنتحَل: أبو منصور الثّعالبيّ (ت، 429هـ)، نظر فيه وصحح روايته وترجم شعراءه وشرح ألفاظه اللغوية: أحمد أبو علي، المطبعة التجارية، الإسكندرية، (1319هـ/ 1901م)، ‏42‏.

غُتْمِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن334هـ=945م

الغُتْمِيَّةُ : الجَهَالَةُ.

فَرْطُ غُتْمِيَّةٍ سَبَقْتُمْ بِهَا بَرْ كَشَ، أَوْ بَاكَ بَاكَ، أَوْ خُرْخَاشَا
الصَّنَوبريّ
ديوان الصّنوبريّ أحمد محمّد بن الحسن الضّبيّ: تحقيق: إحسان عبّاس، دار صادر، بيروت، ط2، (1998م)، ‏196‏.

غَتَمَة[مَصْدَر]

398هـ=1007م

الغَتَمَةُ : شِدَّةُ الحَرِّ، وَسُكُونُ الرِّيحِ.

"مَا تَقُولُ غَبْرَاءُ مُتَرَنِّمَة، هِيَ بِالتَّسْبِيحِ مُهَيْنِمَة، تَسْتَتِرُ فِي الأَوْقَاتِ الشَّبِمَة، وَتَبْرُزُ أَوَانَ الغَتَمَة ؟"
أبو العلاء المَعَرّيّ
الفصول والغايات في تمجيد الله والمواعظ: أبو العلاء المَعَرّيّ (ت، 449هـ)، صنعه وفسر غريبه: محمود حسن زناتي، مراجعة: لجنة إحياء التراث العربي، دار الآفاق الجديدة، بيروت، د.ت. ‏70‏.

تَغَاتَمَ[لَازِم]

ن482هـ=1089م

تَغَاتَمَ الشَّخْصُ: تَظَاهَرَ بِالعُجْمَةِ وَعَدَمِ القُدْرَةِ عَلَى الإِبَانَةِ.

"وَإِنْ كَانَ لِخَوْفٍ مِنْ ثَقِيل، وَحَذَرٍ مِنْ غَلُول، فَمَا كَانَ هُنَاكَ إِلَّا مَنْ يَفْرَقُ السَّوْرَة، وَيَسْتُرُ العَوْرَة... وَإِنْ تَفَادَمَ وَتَغَاتَمَ دَلَّكَ عَلَيْهِ شَرْحُ أُمُورِ قَدِيمَة، وَظُهُورُ أَنْبَاءٍ مَكْتُومَة"
المجيد العسقلانيّ
الذخيرة في محاسن أهل الجزيرة: ابن بسّام الشّنْتَرِيني (ت، 542هـ)، تحقيق: إحسان عبّاس، دار الثَّقافة، بيروت، ط1، (1417ه/ 1997م)، ‏4/‏634‏.
ن811هـ=1408م

تَغَاتَمَ الشَّخْصَانِ: كَلَّمَ كِلَاهُمَا صَاحِبَهُ بِمَا لَا يُفْهَمُ.

وَإِذَا تَغَاتَمَتِ الكُمَاةُ بِجَحْفَلٍ كَلَّمْتُهُمْ فِيهِ بِكُلِّ لِسَانِ
شمس الدين المزين
المنهل الصافي والمستوفي بعد الوافي: جمال الدين أبو المحاسن يوسف بن تغري بردي الأتابكي (ت، 874هـ)، حقّقه ووضع حواشيه: محمد محمد أمين، تقديم: سعيد عبد الفتاح عاشور، الهيئة المصرية العامة للكتاب، القاهرة، (1405ه/ 1984م- 1430ه/2009م)، ‏9/‏227‏.

غُتَمِيّ[صِفَة]

ن692هـ=1293م

الغُتَمِيُّ مِنَ الأَلْفَاظِ: الأَعْجَمِيُّ.

وَبِرُوحِي هَوَيْتُهُ أَعْجَمِيَّا لِيَ لَذَّتْ أَلْفَاظُهُ الغُتَمِيَّهْ
كَمْ حَلَا عُجْمَةً، فَقُلْتُ لِخِلِّي: خَلِّنِي وَالحَلَاوَةَ العَجَمِيَّهْ
محيي الدّين عبد الله بن عبد الظّاهر الجذاميّ السعديّ
تأهيل الغريب: ابن حجة الحمويّ (ت، 837هـ)، تحقيق: محمود حسن المِصْرِيّ، نادي المدينة المنورة الأدبي- دار النّوادر، دمشق، ط1، (1438ه/ 2017م)، ‏438‏.

المُتَغَاتِمُ : المُتَظَاهِرُ بَعَدَمِ الإِفْصَاحِ.

وَمُشَبِّبٍ أَبْدَى لَنَا***قَوْلًا بِنَغْمَتَهِ الشَّهِيَّهْ مُتَغَاتِمٍ ، فَكَأَنَّهُ***مُتَكَلِّمٌ بِالفَارِسِيَّهْ
إبراهيم المعمار
سفينة الملك ونفيسة الفلك: محمد بن إسماعيل بن عمر شهاب الدين المصري (ت، 1274هـ)، المطبعة الجامعة، القاهرة، (1309ه)، ‏472‏.

انْغِتَام[مَصْدَر]

791هـ=1389م

الِانْغِتَامُ : العُجْمَةُ.

"وَنَظَرْتُ إِلَى النَّارِ، وَقَدْ أَرَادَتْ سَبْيَ ذَلِكَ النَّادِي، فَشَنَّتْ عَلَيْهِ مِنْ فَوَارِسِ الغَارَةِ... . وَتَأَمَّلْتُ تِلْكَ الأَلْسُنَ الجَمْرِيَّةَ، وَقَدِ انْطَلَقَتْ فِي ثُغُورِ تِلْكَ الرُّبُوعِ، وَكَلَّمَتِ السُّكَّانَ، وَتَطَاوَلَتْ بِأَلْسُنِ الأَسِنَّةِ وَانْغِتَامِ الأَتْرَاكِ، فَانْدَهَشَ أَهْلُ دِمَشْقَ، وَقَدْ كُلِّمُوا بِكُلِّ لِسَانٍ"
ابن حجة الحمويّ
ياقوت الكلام في ما ناب الشّام: ابن حجة الحمويّ (ت، 837هـ)، تحقيق: أحمد طربين، مجلة المجمع العلمي العربي، دمشق، مج31، ج4، (1956م)، (611- 630)، ‏615‏.
