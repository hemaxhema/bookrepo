أَرْذَلَ[مُتَعَدٍّ]

قبل 100ق.هـ=525م

أَرْذَلَ لَهُ الشَّيْءَ: أَعْطَاهُ الرَّذْلَ مِنْهُ.

"إِنَّا لَمْ نَأْتِكَ لِتُرْذِلَ لَنَا بَنِيكَ، وَلَا لِتَسُومَنَا اللَّبَنَ"
بعض أشراف تغلب
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏5/‏28‏.
قبل 50ق.هـ=573م

أَرْذَلَ الشَّخْصَ: جَعَلَهُ رَذْلًا.

وَلِكلِّ سَاعٍ سُنَّةٌ مِمَّنْ مَضَى تَنْمِي بِهِ فِي سَعْيِهِ أَوْ تُرْذِلُ
الأفوه الأوديّ
الحماسة: البُحْتُريّ (ت، 284هـ)، تحقيق: محمّد إبراهيم حوّر، أحمد محمّد عبيد، المجمع الثقافي، أبو ظبي، ط1، (1428هـ/ 2007م)، ‏432‏.
ن110هـ=728م

أَرْذَلَ الشَّيْءَ: عَدَّهُ رَذْلًا.

"أَلَمْ تَرَوْا إِلَى ثِمَارِ الدُّنْيَا كَيْفَ تُرْذِلُونَ بَعْضَهُ؟ وَإِنَّ ذَلِكَ لَيْسَ فِيهِ رَذْلٌ"
الحسن البصري
تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏1/‏413‏.

أَرْذَلَ[لَازِم]

90هـ=709م

أَرْذَلَ الشَّخْصُ: أَخَذَ الرَّدِيءَ وَتَرَكَ الجَيِّدَ.

"أَعَلَيْنَا تُرْذِلُ ؟خُذِ الأَكْبَرَ"
عبد الله بن عبد الملك
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏15/‏220‏.

أَرْذَل[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الأَرْذَلُ مِنَ الأَشْخَاصِ: الدَّنِيءُ الخَسِيسُ.

بِأَيِّ مَشِيئَةٍ عَمْرُو بْنَ هِنْدٍ تَرَى أَنَّا نَكُونُ الأَرْذَلِينَا
عمرو بن كلثوم التَّغْلِبيّ
ديوان عَمرو بن كلثوم: جمعه وحققه وشرحه: إميل بديع يعقوب، دار الكتاب العربي، بيروت، ط2، (1416هـ/ 1996م)، ‏79‏.

أَرْذَل[اسْمُ تَفْضِيل]

4ق.هـ=618م

الأَرْذَلُ مِنَ الأُمُورِ: الأَسْوَأُ وَالأَرْدَأُ.

قرآن كريم
قرآن كريم، ‏[ النَّحْل : 70 ].
ن11هـ=632م

الأَرْذَلُ مِنَ الأَشْخَاصِ: الأَكْثَرُ رَذَالَةً.

"إِنَّ سُنَّتَنَا النِّكَاحُ. شِرَارُكُمْ عُزَّابُكُمْ، وَأَرَاذِلُ مَوْتَاكُمْ عُزَّابُكُمْ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏35/‏355،،‏(رقم الحديث: 21450).

رَذْل[صِفَةٌ مُشَبَّهَة]

2ق.هـ=620م

الرَّذْلُ مِنَ الأَشْخَاصِ: الدَّنِيءُ الخَسِيسُ.

قرآن كريم
قرآن كريم، ‏[ هُود : 27 ].
ن117هـ=735م

الرَّذْلُ مِنَ الثِّيَابِ وَنَحْوِهَا: الرَّثُّ المُتَّسِخُ.

وَبِالشَّمَائِلِ مِنْ جَلَّانَ مُقْتَنِصٌ رَذْلُ الثِّيَابِ، خَفِيُّ الشَّخْصِ مُنْزَرِبُ
ذو الرُّمَّة
ديوان ذي الرُّمّة غَيلان بن عقبة العَدَويّ: شرح: أبي نصر أحمد بن حاتم الباهلي صاحب الأصمعي، رواية: أبي العبّاس ثعلب، حققه وقدم له وعلق عليه: عبد القدوس أبو صالح، مؤسسة الرسالة، بيروت، ط3، (1414هـ/ 1993م)، ‏1/‏64‏.
ن325هـ=937م

الرَّذْلُ مِنَ الطَّعَامِ وَنَحْوِهِ: الرَّخِيصُ الحَقِيرُ.

"وَلَا يَنْتَقِلُونَ عَلَى شَرَابِهِمْ بِالأَشْيَاءِ الرَّذْلَةِ ، مِثْلِ البَاقِلَّى، وَالبَلُّوطِ، وَالبُسْرِ المَقْلُوِّ... وَمَا أَشْبَهَ ذَلِكَ مِنَ الأَنْقَالِ"
أبو الطّيّب الوَشّاء
الموشّى أو الظّرف والظّرفاء: أبو الطّيّب الوَشّاء (ت، 325هـ)، تحقيق: كمال مصطفى، مكتبة الخانجي- مطبعة الاعتماد، القاهرة، ط2، (1372هـ/ 1953م)، ‏171‏.

رُذَالَة[اسْمُ جَمْع]

1ق.هـ=621م

رُذَالَةُ النَّاسِ: خِسَاسُهُمْ وَسَفِلَتُهُمْ.

"إِنَّهُ لَمْ يُبْعَثْ نَبِيٌّ إِلَّا اتَّبَعَهُ رُذَالَةُ النَّاسِ وَمَسَاكِينُهُمْ"
رجل من العرب
تفسير القرآن العظيم : أبو الفداء إسماعيل بن كثير القرشي (ت، 774هـ)، تحقيق: سامي بن محمد السلامة، دار طيبة للنشر والتوزيع، الرياض، ط2، 1420هـ/1999م، ‏6/‏521‏.

رُذَالَة[اسْم]

ن100هـ=718م

رُذَالَةُ الشَّيْءِ: نُفَايَتُهُ وَحُثَالَتُهُ.

"فَنَهَى رَسُولُ اللهِ @  أَنْ تُؤْخَذَ فِي الصَّدَقَةِ الرُّذَالَةُ "
أبو أُمامة الأوسيّ الأنصاريّ
المجتبى من السنن الصغرى: أبو عبد الرحمن أحمد ب النسائي (ت: 303هـ)تح: عبد الفتاح أبو غدة، الناشر: مكتب المطبوعات الإسلامية - حلب، ط. 2، 1406هـ - 1986م، ‏5/‏43،،‏(رقم الحديث: 2492).

رَاذِل[صِفَةٌ مُشَبَّهَة]

ن11هـ=632م

الرَّاذِلُ مِنَ الأَشْخَاصِ: الدَّنِيءُ الخَسِيسُ.

"إِذَا كَانَتِ الفَاحِشَةُ فِي كِبَارِكُمْ، وَالمُلْكُ فِي صِغَارِكُمْ، وَالعِلْمُ فِي رُذَّالِكُمْ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏20/‏273،،‏(رقم الحديث: 12943).

رَاذِل[اسْمُ فَاعِل]

ن1272هـ=1855م

الرَّاذِلُ الشَّيْءَ: الَّذِي يَذُمُّهُ وَيَعِيبُهُ.

"يُمْكِنُنَا بِوَاسِطَةِ نِعْمَةِ اللهِ أَنْ نَفُوزَ بِالغَلَبَةِ ضِدَّ تَجَارِبِ أَعْدَاءِ خَلَاصِنَا كَافَّةً، رَاذِلِينَ تَنَعُّمَاتِ العَالَمِ"
مكسيموس مظلوم
كتاب الكنز الثمين في أخبار القديسين: مكسيموس مظلوم (ت، 1272هـ)، مطبعة المرسليين اليسوعيين، بيروت، (1863م)، ‏2/‏219‏.

رَذَلَ[مُتَعَدٍّ]

11هـ=632م

رَذَلَ الشَّخْصَ: أَبْعَدَهُ وَفَضَّلَ عَلَيْهِ غَيْرَهُ.

" رَذَلْتَنِي ، وَاللهِ، يَا رَسُولَ اللهِ"
عَقيل بن أبي طالب القُرَشيّ
كتاب الأمالي: اليزيديّ (ت، 310هـ)، مطبعة مجلس دائرة المعارف العثمانية، حيدر آباد الدكن، ط1، (1367هـ/ 1948م)، ‏92‏.
ن150هـ=767م

رَذَلَ الشَّيْءَ: أَهْمَلَهُ وَتَرَكَهُ.

"ثَمَرُ الدُّنْيَا مِنْهُ مَا يُرْذَلُ ، وَمِنْهُ نَقَاوَةٌ، وَثَمَرُ الجَنَّةِ نَقَاوَةٌ كُلُّهُ، يُشْبِهُ بَعْضُهُ بَعْضًا فِي الطِّيبِ، لَيْسَ مِنْهُ مَرْذُولٌ"
ابن جُريج
تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏1/‏413‏.
269هـ=882م

رَذَلَ الشَّيْءَ: احْتَقَرَهُ وَاسْتَصْغَرَهُ.

أَدِلَّةُ المَرْءِ أَيَّامٌ عُدِدْنَ لَهُ يُرِينَهُ القَصْدَ تَقْوِيمًا وَإِرْشَادَا
وَقَدْ يُطَالِبْنَ مَا قَدَّمْنَ مِنْ سَلَفٍ فِيهِ فَيَنْقُصْنَهُ الفَضْلَ الَّذِي ازْدَادَا
حَتَّى يَعُودَ الجَدِيدُ المُشْتَرَى خَلَقًا تَرْذُلُهُ العَيْنُ، وَالمُنْصَاتُ مُنْآدَا
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏1/‏608‏.

رُذَال[صِفَةٌ مُشَبَّهَة]

37هـ=657م

الرُّذَالُ مِنَ النَّاسِ: الدَّنِيءُ الخَسِيسُ.

"إِنَّ يَوْمَنَا وَيَوْمَهُمْ لَيَوْمٌ عَصِيبٌ، مَا يَصْبِرُ عَلَيْهِ إِلَّا كُلُّ مُشَيَّعِ القَلْبِ، صَادِقِ النِّيَّةِ، رَابِطِ الجَأْشِ، وَايْمُ اللهِ مَا أَظُنُّ ذَلِكَ اليَوْمَ يُبْقِي مِنْهُمْ وَلَا مِنَّا إِلَّا الرُّذَالَ "
زياد بن النَّضْر الحارثيّ
وقعة صفّين: نصر بن مُزاحم المِنْقَريّ (ت، 212هـ)، تحقيق وشرح: عبد السلام هارون، دار الجيل، بيروت، (1410هـ/ 1990م)، ‏111‏.
57هـ=677م

رُذَالُ الشَّيْءِ: رُذَالَتُهُ، وَرَدِيئُهُ.

"يَا أَبَا هُرَيْرَةَ، يَأْتِينَا مُصَدِّقُونَ يُصَدِّقُونَ أَمْوَالَنَا، أَفَنُغَيِّبُ عَنْهُمْ خِيَارَهَا، وَنُظْهِرُ لَهُمْ رُذَالَهَا ؟"
شيخ من بني فزارة
كتاب الأموال: ابن زَنْجَوَيْه (ت، 251هـ)، تحقيق: شاكر ذيب فيّاض، مركز الملك فيصل للبحوث والدراسات الإسلامية، الرياض، ط1، (1406هـ/ 1986م)، ‏3/‏893‏.

رَذِيل[صِفَةٌ مُشَبَّهَة]

65هـ=685م

الرَّذِيلُ : الدَّنِيءُ الرَّدِيءُ.

وَأَخَّاذُ أَقْصَى الحَقِّ، لَا مُتَهَضَّمٌ أَخُوهُ، وَلَا هَشُّ القَنَاةِ، رَذِيلُهَا
الأخطل
شعر الأخطل أبي مالك غياث بن غوث التغلبي: صنعة: السُّكّري، رواية: أبي جعفر محمّد بن حبيب، تحقيق: فخر الدين قباوة، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط4، (1416هـ/ 1996م)، ‏409‏.

ن110هـ=728م

رَذُلَ الشَّخْصُ: صَارَ رَذْلًا.

"قَدْ أُسْرِعَ بِخِيَارِكُمْ، وَأَنْتُمْ كُلَّ يَوْمٍ تَرْذُلُونَ "
الحسن البصريّ
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏3/‏132‏.

مَرْذُول[صِفَةٌ مُشَبَّهَة]

ن110هـ=728م

المَرْذُولُ مِنَ الأَشْيَاءِ: الخَسِيسُ الرَّدِيءُ.

 " يُشْبِهُ بَعْضُهُ بَعْضًا لَيْسَ فِيهِ مَرْذُولُ "
الحسن البصري
تفسير القرآن: عبد الرّزّاق بن هَمّام الصَّنْعانيّ (ت، 211هـ)، تحقيق: مصطفى مسلم محمد، مكتبة الرشد، ط1، (1410هـ/ 1989م)، ‏1/‏40‏.

مَرْذُول[اسْمُ مَفْعُول]

209هـ=824م

المَرْذُولُ مِنَ النَّاسِ: المُحْتَقَرُ المُهَانُ.

تِلْكَ دَعْوَى لَا نُنَاقِشُهَا وَأُبُوَّاتٌ مَرَاذِيلُ

الإِرْذَالُ بِالشَّخْصِ: الِاحْتِقَارُ وَالإِهَانَةُ لَهُ.

"وَكَانَ سُحْنُونٌ قَدْ تَلَقَّفَهُ عَنْ أَسَدِ بْنِ الفُرَاتِ، فَحَكَمَ بِهِ أَيَّامَ قَضَائِهِ، وَلَعَلَّهُ كَانَ يَفْعَلُ ذَلِكَ بِأَهْلِ الظُّلْمِ وَالجَبَرُوتِ، إِنْصَافًا مِنْهُمْ، وَإِرْذَالًا بِهِمْ، وَهُوَ الحَقُّ؛ فَإِنَّ الوَكَالَةَ مَعُونَةٌ، وَلَا تَكُونُ لِأَهْلِ البَاطِلِ"
أبو بكر ابن العربيّ
أحكام القرآن: أبو بكر ابن العربيّ (ت، 543هـ)، تحقيق: علي محمد البجاوي، مطبعة عيسى البابي الحلبي وشركاه، القاهرة، ط3، (1392ه/ 1972م)، ‏3/‏1231‏.

أَرْذَلِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن546هـ=1151م

الأَرْذَلِيَّةُ : الرَّذَالَةُ.

"ثُمَّ إِنِ اسْتَقَالَنِي وَاحِدٌ مِنْكُمَا أَقَلْتُه، وَمَنِ اسْتَغَاثَ بِي وَاسْتَجَارَ بِي أَجَرْتُهُ وَأَغَثْتُه، وَلْيَكُنْ بَعْدَ انْسِلَاخٍ مِنْ جِلْدَةِ المَعْصِيَة، وَانْخِلَاعٍ مِنْ كُسْوَةِ الأَدْنَاسِ مِنَ الأَرْذَلِيَّةِ ... فَإِنَّ الخَطِيئَةَ رِجْسٌ، وَالرِّجْسُ لَا يَدْخُلُ حَضْرَتِي"
ابن قَسِيّ
كتاب خلع النعلين واقتباس النور من موضع القدمين: ابن قَسِيّ (ت، 546هـ)، دراسة وتحقيق: محمد الأمراني، مطبعة IMBH، آسفي، ط1، (1418ه/ 1997م)، ‏271‏.

مُرْذَل[اسْمُ مَفْعُول]

1250هـ=1834م

المُرْذَلُ مِنَ الأُمُورِ: المَعِيبُ المَذْمُومُ.

وَإِنَّ المَنَايَا جَيْشُهَا كُلَّ سَاعَةٍ لِرُوحِ جَمِيعِ العَالَمِينَ سَلِيبُ
فَقَدْ فَجَعَتْنَا بِالإِمَامِ وَبِابْنِهِ عَلِيٍّ، فَفَاضَتْ بِالدُّمُوعِ ذَنُوبُ
وَإِنَّ عَلِيًّا فِي الفَضَائِلِ وَاحِدٌ وَعَنْ فِعْلِ كُلِّ المُرْذَلَاتِ نَكُوبُ
ابن عاكش الحسن بن أحمد بن عاكش الضمدي
حدائق الزهر في ذكر الأشياخ أعيان الدهر: ابن عاكش الحسن بن أحمد بن عاكش الضمدي (ت، 1290هـ)، حققه ودرسه وعلق عليه: إسماعيل بن محمد البشري، ط1، (1413ه/ 1992م)، ‏58‏.

مُتَرَذِّل[اسْمُ فَاعِل]

ن1391هـ=1970م

المُتَرَذِّلُ مِنَ الأَشْخَاصِ: المُنْحَطُّ، المُتَّصِفُ بِالرَّذَالَةِ.

"وَمَا شُعَرَاءُ تِلْكَ الأَيَّامِ، إِلَّا كَشُعَرَاءِ أَيَّامِ النَّابِغَةِ وَمَا بَعْدَهُ، فِيهِمُ الشَّاعِرُ المُتَرَفِّعُ، وَفِيهِمُ الشَّاعِرُ المُتَرَذِّلُ "
جواد علي
المفصل فى تاريخ العرب قبل الإسلام: جواد علي (ت، 1407هـ)، جامعة بغداد، بغداد، ط2، (1413ه/ 1993م)، ‏9/‏109‏.
