حُزَق[صِفَةٌ مُشَبَّهَة]

قبل 150ق.هـ=476م

الحُزَقُ : القَصِيرُ الغَلِيظُ، العَظِيمُ البَطْنِ.

زَلَّتْ مَحَافِيرُكِ يَا أُمَّ الحُبَقْ زَلَّتْ مَحَافِيرُكِ عَنْ جَهْمٍ حُزَقْ
منسوب إلى جمعة/خمعة بنت الخُسّ الإياديّة
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏193‏.

حَزِيق[اسْم]

قبل 100ق.هـ=525م

الحَزِيقُ : الجَمَاعَةُ مِنَ كُلِّ شَيْءِ.

فَلَمَّا اسْتَيْقَنُوا بِالصَّبْرِ مِنَّا تُذُكِّرَتِ العَشَائِرُ وَالحَزِيقُ
المفضّل بن معشر النُّكْريّ
شعراء عبد القيس وشعرهم في العصر الجاهلي: جمع وتحقيق ودراسة: عبد الحميد المعيني، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري، الكويت، (2002م)، ‏328‏.

حَزِيق[صِيغَةُ مُبَالَغَة]

ن206هـ=821م

الحَزِيقُ : الشَّدِيدُ.

هُمُ السَّالِبُونَ حَلْيَ النِّسَا ءِ المَانِعُوهُنَّ مَنْعًا حَزِيقَا
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏217‏.

حَزِيقَة[اسْم]

قبل 50ق.هـ=573م

الحَزِيقَةُ : الجَمَاعَةُ مِنْ كُلِّ شَيْءٍ.

كَأَنَّ مَجَرَّةَ النَّسْرَيْنِ نَهْجٌ لكُلِّ حَزيقَةٍ تُحْدَى وَعِيرِ
مُهَلْهِل بن ربيعة التَّغْلِبيّ
ديوان المُهَلْهِل: شرح وتحقيق: أنطوان محسن القوَّال، دار الجيل، بيروت، ط1، (1415هـ/ 1995م)، ‏36‏.

مُحَزَّق[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُحَزَّقُ : المَشْدُودُ وَثَاقُهُ بِإِحْكَامٍ.

كَأَنَّ التَّمِيمَ البِيضَ فِي كَوْرِ صَغْوِهَا إِذَا شِمْتَهَا، وَالشَّمْسُ بَادٍ شُرُوقُهَا
مُجَزَّعَةُ الأَحْقَابِ، بِالرِّيشِ رِكْزُهُا مُحَزَّقَةٌ أَوْسَاطُهَا وَحُلُوقُهَا
عامر بن سعد النَّمَريّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏9/‏90‏.
ن314هـ=926م

المُحَزَّقُ : المُحَاطُ المُسَوَّرُ.

"ثُمَّ تَقَدَّمَ رَجُلٌ مِنْ شُجْعَانِ الفُرْسِ... فِي قَرِيبٍ مِنْ أَلْفِ فَارِسٍ... عَلَى فِيلٍ مُزَيَّنٍ، وَعَلَى رَأْسِهِ تَاجٌ مِنَ الذَّهَبِ، مُرَصَّعٌ بِالجَوْهَرِ، وَفِي يَدِهِ طَبَرْزِينٌ مُحَزَّقٌ بِالذَّهَبِ"
ابن أعثم الكوفيّ
كتاب الفتوح: ابن أعثم الكوفيّ (ت، 314هـ)، تحقيق: علي شيري، دار الأضواء، بيروت، ط1، (1411هـ/ 1991م)، ‏2/‏304‏.
ن368هـ=978م

المُحَزَّقُ مِنَ الحُرُوفِ: المَضْغُوطُ المَحْبُوسُ.

"يَنْبَغِي إِذَا أَرَدْتَ امْتِحَانَ ذَلِكَ أَنْ تَبْتَدِئَ بِحَرْفٍ مِنَ الحُرُوفِ، وَتُثَنِّيَ بِأَحَدِ هَذِهِ الحُرُوفِ الخَمْسَةِ، فَتَقِفَ عَلَيْهِ، فَإِنَّكَ تَسْمَعُ صُوَيْتًا عِنْدَ الوَقْفِ عَلَيْهِ... وَذَلِكَ أَنَّ هَذِهِ الحُرُوفَ لَمَّا انْضَغَطَ مَوْضِعُهَا، وَلَمْ يَكُنْ لِلصَّوْتِ مَنْفَذٌ، صَارَ الوَقْفُ عَلَيْهِ، وَقَطْعُهُ، بِمَنْزِلَةِ قَطْعِ شَيْءٍ شَدِيدِ التَّحْزِيقِ... لِأَنَّ مَا كَانَ مُنْفَذًا لَمْ يَكُنْ لَهُ فِي التَّصْوِيتِ مِنَ الأَثَرِ مَا لِلْمُحَزَّقِ "
أبو سعيد السِّيرافيّ
شرح كتاب سيبويه: أبو سعيد السِّيرافيّ (ت، 368هـ)، تحقيق: أحمد حسن مهدلي، علي سيّد علي، دار الكتب العلميّة، بيروت، ط1، (1429هـ/ 2008م)، ‏5/‏46‏.

حُزُقَّة[صِفَة]

قبل 50ق.هـ=573م

الحُزُقَّةُ : المُتَقَارِبُ الخَطْوِ لِقِصَرٍ، أَوْ ضَعْفٍ.

وَأَعْجَبَنِي مَشْيُ الحُزُقَّةِ خَالِدٍ كَمَشْيِ أَتَانٍ حُلِّئَتْ بِالمَنَاهِلِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏140‏.
127هـ=745م

الحُزُقَّةُ : المُتَشَدِّدُ عَلَى مَا فِي يَدَيْهِ بُخْلًا.

فَمَا تِسْعُونَ تَحْفِزُهَا ثَلَاثٌ يَضُمُّ حِسَابَهَا رَجُلٌ شَدِيدُ
بِكَفِّ حُزُقَّةٍ جُمِعَتْ لِوَجْءٍ بِأَنْكَدَ مِنْ عَطَائِكَ يَا يَزِيدُ
يحيى بن نوفل الحِمْيريّ
يَحيى بن نَوفل أخباره وشعره: تحقيق: عبد المجيد الإسداوي، مجلة العرب، الرياض، س32، (ج1، 2)، (1417هـ/ 1996م)، ‏181‏.
ن398هـ=1007م

الحُزُقَّةُ : القَصِيرُ الغَلِيظُ، العَظِيمُ البَطْنِ.

"وَإِذَا هُنَاكَ قَوْمٌ مُجْتَمِعُونَ عَلى رَجُلٍ، يَسْتَمِعُونَ إِلَيْهِ... فَمَا زِلْتُ بِالنَّظَّارَةِ؛ أَزْحَمُ هَذَا، وَأَدْفَعُ ذَاكَ، حَتَّى وَصَلْتُ إِلَى الرَّجُلِ، وَسَرَّحْتُ الطَّرْفَ مِنْهُ إِلَى حُزُقَّةٍ كَالَقَرَنْبَى، أَعْمَى مَكْفُوف، فِي شَمْلَةِ صُوف، يَدُورُ كَالخُذْرُوف"
بديع الزّمان الهَمَذانيّ
مقامات بديع الزّمان الهَمَذانيّ: بديع الزّمان الهَمَذانيّ (ت، 398هـ)، قدم لها وشرح غوامضها: الشّيخ محمّد عبده، دار الكتب العلميّة، بيروت، ط3، (1426هـ/ 2005م)، ‏93‏.

حُزُقَّة[اسْم]

ن103هـ=721م

الحُزُقَّةُ : لُعْبَةٌ تَقُومُ عَلَى تَجَمُّعِ الأَفْرَادِ وَانْضِمَامِ بَعْضِهِمْ إِلَى بَعْضٍ.

"اجْتَمَعَ جَوَارٍ، فَأَرِنَّ، وَأَشِرْنَ، وَلَعِبْنَ الحُزُقَّةَ "
الشَّعْبيّ
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏3/‏118‏.
ن207هـ=822م

الحُزُقَّةُ : الضِّيقُ.

"يُقَالُ: إِنَّ فِي خُلُقِهِ لَحُزُقَّةً وَحُظُبَّةً"
أبو زكريّا الفَرّاء
المذكّر والمؤنّث: أبو بكر ابن الأنباريّ (ت، 328هـ)، تحقيق: محمّد عبد الخالق عضيمة، راجعه وصنع فهارسه: رمضان عبد التوّاب، وزارة الأوقاف، القاهرة، (1401هـ/ 1981م- 1419هـ/ 1999م)، ‏2/‏159‏.

حِزْقَة[اسْم]

قبل 50ق.هـ=573م

الحِزْقَةُ : الجَمَاعَةُ مِنْ كُلِّ شَيْءٍ.

زَجِلُ الأَصْوَاتِ، حَتَّى مَا بِهِ لَيْسَ شَتَّى حِزَقِ القَوْمِ، شِعَارُ
الأفوه الأوديّ
اللّامع العزيزيّ شرح ديوان المتنبّيّ: أبو العلاء المَعَرّيّ (ت، 449هـ)، حققه: محمّد سعيد المولوي، مركز الملك فيصل للبحوث والدراسات الإسلامية، الرياض، ط1، (1429هـ/ 2008م)، ‏459‏.
ن216هـ=831م

الحِزْقَةُ : القِطْعَةُ مِنْ كُلِّ شَيْءٍ.

فَمَا بِهَذَا تَقَومُ النَّادِبَاتُ، وَلَا تَبْكِي عَلَيْكَ، إِذَا مَا ضَمَّكَ الحِزَقُ
الأصمعيّ
الورقة: ابن الجَرّاح البغداديّ الأديب (ت، 296هـ)، تحقيق: عبد الوهاب عزام، عبد الستار أحمد فرّاج، دار المعارف، القاهرة، ط3، (1986م)، ‏32‏.

مَحْزُوق[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المَحْزُوقُ : المَضْغُوطُ المَعْصُورُ.

وَالطَّاعِنُ الطَّعْنَةَ النَّجْلَاءَ تَحْسَبُهَا شَنًّا هَزِيمًا، تَمُجُّ المَاءَ مَحْزُوقَا
الأسود بن يَعْفُر النَّهْشليّ
ديوان الأسود بن يَعفُر: صنعة: نوري حمّودي القيسيّ، وزارة الثقافة والإعلام، بغداد، (1390هـ/ 1970م)، ‏52‏.
ن283هـ=896م

المَحْزُوقُ : المَشْدُودُ المَرْبُوطُ.

لَمْ تَدَعْنِي حَبَائِلُ الشَّادِنِ الأَكْـ ـحَلِ حَتَّى نَشِبْتُ فِيهَا نُشُوقَا
أَحَلَالٌ أَنْ يَحْزِقَ الصَّيْدَ صَبْرًا مَنْ رَأَى فِي حِبَالِهِ مَحْزُوقَا ؟
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏4/‏1669‏.

"وَيَحْزُقُ أَحَدَ الأَوْتَارِ حَزْقًا لَطِيفَا مِقْدَارَ مَا يَسْمَعُ مِنْ نَقْرِ صَوْتٍ... ثُمَّ يُسَوَّى الوَتَرُ الثَّالِثُ تَسْوِيَةً حَازِقَةً ؛ حَتَّى يَحْصُلَ مِنْهَا نَغْمَةٌ هِي صَيْحَةُ النَّغْمَةِ الأُولَى"

حَازِق[اسْم]

حَزَق[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

الحَزَقُ : الشَّدِيدُ عَلَى مَا فِي يَدَيْهِ ضَنْكًا.

"وَالرَّجُلُ المُتَحَزِّقُ: المُتَشَدِّدُ عَلَى مَا فِي يَدَيْهِ ضَنْكًا، وَكَذَلِكَ الحُزُقَّةُ، وَالحُزُقُّ، وَيُقَالُ: الحَزَقُ أَيَضًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏38‏.
ن337هـ=948م

الحَزَقُ مِنَ الأَشْخَاصِ: السَّيِّئُ الخُلُقِ، الشَّدِيدُ الخُصُومَةِ.

"رَجُلٌ عَضٌّ شَرِسٌ، وَوَعْقَةٌ شَكِسٌ... وَحَزَقٌ عُوَقٌ، وَعَزَقٌ أَزِقٌ"
قُدامة بن جعفر البغداديّ
جواهر الألفاظ: أبو الفرج قُدامة بن جعفر البغداديّ (ت، 337هـ)، تحقيق: محمّد محيي الدين عبد الحميد، دار الكتب العلمية، بيروت، ط1، (1405هـ/ 1985م)، ‏295‏.

تَحْزِيق[مَصْدَر]

ن255هـ=869م

التَّحْزِيقِ : إِحْكَامُ الرَّبْطِ وَالشَّدِّ.

"وَمِنْهَا مَا يَكُونُ بِالشَّدِّ وَالعَصْبِ، وَشِدَّةِ التَّحْزِيقِ ، وَالعَقْدِ بِالخَيْطِ الشَّدِيدِ الوَتِيرِ الشَّدِيدِ الفَتْلِ"
الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏1/‏130‏.
ن368هـ=978م

التَّحْزِيقُ : الضَّغْطُ.

"يَنْبَغِي إِذَا أَرَدْتَ امْتِحَانَ ذَلِكَ أَنْ تَبْتَدِئَ بِحَرْفٍ مِنَ الحُرُوفِ، وَتُثَنِّيَ بِأَحَدِ هَذِهِ الحُرُوفِ الخَمْسَةِ، فَتَقِفَ عَلَيْهِ، فَإِنَّكَ تَسْمَعُ صُوَيْتًا عِنْدَ الوَقْفِ عَلَيْهِ، كَقَوْلِكَ: أَقْ، وَأَجْ، وَأَطْ، وَأَدْ، وَأَبْ... وَذَلِكَ أَنَّ هَذِهِ الحُرُوفَ لَمَّا انْضَغَطَ مَوْضِعُهَا، وَلَمْ يَكُنْ لِلصَّوْتِ مَنْفَذٌ، صَارَ الوَقْفُ عَلَيْهِ، وَقَطْعُهُ، بِمَنْزِلَةِ قَطْعِ شَيْءٍ شَدِيدِ التَّحْزِيقِ . وَالتَّحْزِيقُ الَّذِي يُوجِبُ التَّصْوِيتَ؛ لِأَنَّ مَا كَانَ مُنْفَذًا لَمْ يَكُنْ لَهُ فِي التَّصْوِيتِ مِنَ الأَثَرِ مَا لِلْمُحَزَّقِ"
أبو سعيد السِّيرافيّ
شرح كتاب سيبويه: أبو سعيد السِّيرافيّ (ت، 368هـ)، تحقيق: أحمد حسن مهدلي، علي سيّد علي، دار الكتب العلميّة، بيروت، ط1، (1429هـ/ 2008م)، ‏5/‏46‏.
1270هـ=1853م

تَحْزِيقُ الثَّوْبِ وَنَحْوِهِ: تَضْيِيقُهُ.

"فَلَوْ لَبِسَتْ إِحْدَاهُنَ، مَثَلًا، عَبْعَبَا، أَوْ حَزَّقَتْ ثَوْبَهَا لَعَبَّ النَّاسُ حُبَّ ذَلِكَ العَبْعَبَ، وَصَارَ التَّحْزِيقُ سُنَّةً فِيهِمْ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏634‏.

تَحَزَّقَ[مُتَعَدٍّ بِالحَرْف]

ن337هـ=948م

تَحَزَّقَ القَوْمُ عَلَيْهِ: اجْتَمَعُوا حَوْلَهُ وَاحْتَشَدُوا.

"وَيُقَالُ: صَارُوا إِلَيْهِ لِبَدًا... وَتَحَزَّقُوا عَلَيْهِ، وَتَحَزَّبُوا إِلَيْهِ"
قُدامة بن جعفر البغداديّ
جواهر الألفاظ: أبو الفرج قُدامة بن جعفر البغداديّ (ت، 337هـ)، تحقيق: محمّد محيي الدين عبد الحميد، دار الكتب العلمية، بيروت، ط1، (1405هـ/ 1985م)، ‏51‏.
ن845هـ=1441م

تَحَزَّقَ بِهِ: أَحَاطَ بِهِ.

"وَعَلِمَ بَكْجُورُ بِمَا دَبَّرَهُ الوَزِيرُ، فَأَخَذَ يُعَارِضُهُ فِي ضِيَاعِهِ، وَيُهِينُ عُمَّالَهُ، وَتَحَزَّقَ بِابْنِ أَبِي العُودِ الصَّغِيرِ"
تقي الدين أحمد بن علي المقريزي
اتِّعاظ الحنفا بأخبار الأئمة الفاطميين الخلفا: تقي الدين أحمد بن علي المقريزي (ت، 845هـ)، تحقيق: جمال الدين الشيال، محمد حلمي محمد أحمد، وزارة الأوقاف، القاهرة، ط2، (1416ه/ 1996م)، ‏1/‏259‏.

تَحَزُّق[مَصْدَر]

ن337هـ=948م

التَّحَزُّقُ : التَّقَبُّضُ وَالتَّضَيُّقُ.

"وَيُقَالُ: مَا أَشَدَّ تَعَسُّرَهُ... وَتَحَزُّقَهُ ، وَتَعَوُّقَهُ!"
قُدامة بن جعفر البغداديّ
جواهر الألفاظ: أبو الفرج قُدامة بن جعفر البغداديّ (ت، 337هـ)، تحقيق: محمّد محيي الدين عبد الحميد، دار الكتب العلمية، بيروت، ط1، (1405هـ/ 1985م)، ‏295‏.
ن751هـ=1350م

التَّحَزُّقُ : التَّغَضُّبُ.

"وَيُدْخِلُ عَلَيْهِمْ مِنَ الغَضَاضَةِ وَالعَارِ وَاحْتِمَالِ الضَّيْمِ وَالحَمِيَّةِ وَالتَّحَزُّقِ لِأَخْذِ الثَّأْرِ مَا لَا يَجْبُرُهُ المَالُ أَبَدًا"
ابن قيّم الجوزيّة
إعلام الموقعين عن رب العالمين: ابن قيّم الجوزيّة (ت، 751هـ)، قرأه وقدم له وعلق عليه وخرج أحاديثه وآثاره: أبو عبيدة مشهور بن حسن آل سلمان، شارك في التخريج: أبو عمر أحمد عبد الله أحمد، دار ابن الجوزي، الدمام، ط1، (1423 ه)، ‏3/‏353‏.

حَزُوق[اسْم]

ن395هـ=1005م

الحَزُوقُ : الفُسْتُقُ.

"وَالمَوْزُ عَرَبِيٌّ مَعْرُوفٌ. وَيُقَالُ لِلْفُسْتُقِ: الحَزُوقُ . وَالمِزْجُ: اللَّوْزُ المُرُّ"
أبو هلال العسكريّ
كتاب التّلخيص في معرفة أسماء الأشياء: أبو هلال العسكريّ (ت، 395هـ)، عني بتحقيقه: عزة حسن، دار طلاس، دمشق، ط2، (1996م)، ‏315‏.

انْحِزَاق[مَصْدَر]

418هـ=1027م

انْحِزَاقُ الشَّيْءِ: شِدَّتُهُ وَإِحْكَامُ رَبْطِهِ.

"وَقَدْ عَلِمْتَ أَنَّ الحِدَّةَ سَبَبُهَا القَرِيبُ تَلَزُّزٌ، وَقُوَّةٌ، وَمُلَامَسَةُ سَطْحٍ... وَأَنَّ الثِّقَلَ سَبَبُهُ أَضْدَادُ ذَلِكَ، وَأَنَّ أَسْبَابَ سَبِبِ الحِدَّةِ صَلَابَةُ المُقَاوِمِ المَقْرُوعِ، أَوْ مُلَامَسَتُهُ، أَوْ قِصَرُهُ، أَوِ انْحِزَاقُهُ ، أَوْ ضِيقُهُ"
ابن سينا
الشّفاء الرّياضيّات: ابن سينا (ت، 428هـ)، ج(1- 2)، تحقيق: عبد الحميد صبرة وآخرين، مراجعة وتصدير: إبراهيم بيومي مدكور، مكتبة آية الله المرعشي، قم، ط2، (1405هـ)، ‏مج1 ج2/‏10‏.

حِزَاق[اسْم]

ن694هـ=1294م

الحِزَاقُ : الرِّبَاط.

"ارْبَطْ حِمَارَكَ مَعْ حَمِيرِ الظُّلَمِ، إِنْ لَمْ يَتَعَلَّمِ النُّهَاق، يَتَعَلَّمِ الحِزَاق"
الزجاليّ القُرْطبيّ
أمثال العوام في الأندلس: الزجاليّ القُرْطبيّ (ت، 694هـ)، تحقيق وشرح ومقارنة: محمَّد بن شريفة، منشورات وزارة الدَّولة المكلفة للشُّؤون الثَّقافيَّة والتَّعليم الأَصلي، فاس، (1395ه/ 1975م)، ‏2/‏97‏.

أَحْزَقَ[مُتَعَدٍّ بِالحَرْف]

839هـ=1435م

أَحْزَقَ بِهِ: أَحَاطَ بِهِ.

"فَأَخَذَ الحَاجُّ عَليٌّ الجَابِرُ فِي الإِنْكَارِ عَلَى أَتْبَاعِ المَسْعُودِ، فَأَحْزَقَ بِهِ المَسْعُودُ، وَضَرَبَهَ وَنَتَفَ لِحْيَتَهُ"
تقي الدين أحمد بن علي المقريزي
درر العقود الفريدة في تراجم الأعيان المفيدة: تقي الدين أحمد بن علي المقريزي (ت، 845هـ)، حققه وعلق عليه: محمود الجليلي، دار الغرب الإسلامي، بيروت، ط1، (1423ه/ 2002م)، ‏3/‏309‏.

حَازُوقَة[اسْم]

1075هـ=1664م

الحَازُوقَةُ : الفُوَاقُ، وَهُوَ تَشَنُّجٌ فِي الحِجَابِ الحَاجِزِ، يَلِيهِ إِغْلَاقٌ مُزْعِجٌ سَرِيعٌ لِلْمِزْمَارِ، تَنْتُجُ عَنْهُ شَهَقَاتٌ مُتَكَرِّرَةٌ. [الطّبّ والصّيدلة]

"وَإِذَا خُلِطَ مَاؤُهُ بِالخَلِّ وَشُرِبَ، فَإِنَّهُ يَقْطَعُ بَصْقَ الدَّمِ، وَالحَازُوقَةَ ، وَيَمْنَعُ القَيْءَ"
ابن سلوم
غاية البيان في تدبير بدن الإنسان: ابن سلوم (ت، 1081هـ)، تحقيق ودراسة: محمد ياسر زكور، وزارة الثقافة، دمشق، ط1، (2013م)، ‏274‏.

حُزَاق[اسْم]

1413هـ=1992م

الحُزَاقُ : الحَازُوقَةُ.

"لَكِنَّ الأُمُورَ لَا يُمْكِنُهَا أَنْ تَنْقَلِبَ أَبَدًا، مَا دَامَ الهَرَاوِي قَدْ أَتَى عَلَى المِقْلَاةِ كُلِّهَا، وَانْقَلَبَ عَلَى بَطْنِهِ الآنَ، وَبَدَأَ يُصْدِرُ زَفِيرًا وَحُزَاقًا "
محمد زفزاف
الأعمال الروائية الكاملة، محمد زفزاف: محمد زفزاف (ت، 1422هـ)، المركز الثقافي العربي، الدار البيضاء، بيروت، ط1، (1438ه/ 2017م)، ‏669‏-‏ 700‏.
