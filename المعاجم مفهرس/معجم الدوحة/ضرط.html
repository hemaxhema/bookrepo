ضَرِط[مَصْدَر]

قبل 50ق.هـ=573م

ضَرِطُ المَرْأَةِ: وِلَادَتُهَا، وَمَا يَصْحَبُهَا مِنْ زَحِيرٍ وَنَحْوِهِ.

"أَعْرِفُ ضَرِطِي بِهِلَالٍ"
رُقيّة بنت جُشَم البَكْريّة
مجمع الأمثال: أبو الفضل الميدانيّ (ت، 518هـ)، حققه وفصله وضبط غرائبه وعلق حواشيه: محمّد محيي الدين عبد الحميد، مطبعة السنة المحمدية، القاهرة، (1374هـ/ 1955م)، ‏2/‏30‏.

ضَرِط[اسْم]

قبل 50ق.هـ=573م

الضَّرِطُ : الصَّوْتُ الخَارِجُ مِنَ الدُّبُرِ.

أَكْثَرَ مِنْهُ الأَكْلَ حَتَّى خَرِطَا فَأَكْثَرَ المَذْبُوبُ مِنْهُ الضَّرِطَا
نجاد الخَيْبريّ
لسان العرب: محمد بن مكرم أبو الفضل، جمال الدين ابن منظور الأنصاري، دار صادر - بيروت، ط 3، 1414 هـ ، ‏7/‏356‏.

ضَرِط[صِيغَةُ مُبَالَغَة]

ن175هـ=791م

الضَّرِطُ مِنَ النَّاسِ: الكَثِيرُ الضُّرَاطِ.

"وَرَجُلٌ ضَرِطٌ ، مِنَ الضُّرَاطِ، نَعْتٌ لَهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏22‏.

ضَرُوط[صِيغَةُ مُبَالَغَة]

قبل 13ق.هـ=609م

الضَّرُوطُ : الكَثِيرُ الضَّرْطِ.

كَأَنَّكَ عَيْرُ سَالِئَةٍ ضَرُوطٌ كَثِيرُ الجَهْلِ، شَتَّامُ الكِرَامِ
أوس بن غَلْفاء الهُجَيميّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏441‏.

ضَرَطَ[لَازِم]

قبل 13ق.هـ=609م

ضَرَطَ الشَّخْصُ وَغَيْرُهُ: أَخْرَجَ مِنْ دُبُرِهِ رِيحًا مَعَ صَوْتٍ.

"قَدْ يَضْرِطُ العَيْرُ، وَالمِكْوَاةُ فِي النَّارِ"
مسافر بن أبي عمرو بن أميّة القُرَشيّ
أمثال العرب: المفضَّل الضَّبّيّ (ت، 178هـ)، قدم له وعلق عليه: إحسان عبّاس، دار الرائد العربي، بيروت، ط2، (1403هـ/ 1983م)، ‏165‏.

ضَرْط[مَصْدَر]

قبل 13ق.هـ=609م

الضَّرْطُ : صَوْتُ الرِّيحِ عِنْدَ خُرُوجِهِ مِنَ الدُّبُرِ.

حَتَّى إِذَا قَمِلَتْ بُطُونُكُمُ وَرَأَيْتُمُ أَبْنَاءَكُمْ شَبُّوا
أَسْتَاهَ أَحْمِرَةٍ صَدَرْنَ مَعًا نَبَتَ الثَّغَامُ لَهُنَّ وَالعِرْبُ
يَمْلَأْنَ جَوْفَ مُتَالِعٍ ضَرْطًا فَضًّا يَرُدُّ فَضِيضَهُ الهَضْبُ
الأسود بن يَعْفُر النَّهْشليّ
ديوان الأسود بن يَعفُر: صنعة: نوري حمّودي القيسيّ، وزارة الثقافة والإعلام، بغداد، (1390هـ/ 1970م)، ‏19‏-‏20‏.

مِضْرَاط[صِيغَةُ مُبَالَغَة]

ن7هـ=628م

المِضْرَاطُ : الكَثِيرُ الضَّرْطِ.

بَنُو شُرَحْبِيلَ سِوًى بِسَاطْ وَعَمُّهُمْ ضُبَيْعَةُ المِضْرَاطْ صَمَحْمَحٌ مُجَرَّبٌ عَيَّاطْ
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏2/‏139‏.

أَضْرَطَ[مُتَعَدٍّ بِالحَرْف]

ن11هـ=632م

أَضْرَطَ بَيْنَ أَلْيَتَيْهِ: عَمِلَ فِيهِ صَوْتًا يُشْبِهُ الضُّرَاطَ.

"إِنَّ أَحَدَكُمْ إِذَا كَانَ فِي الصَّلَاةِ جَاءَ الشَّيْطَانُ، فَأَبَسَّ بِهِ كَمَا يُبِسُّ الرَّجُلُ بِدَابَّتِهِ، فَإِذَا سَكَنَ لَهُ أَضْرَطَ بَيْنَ أَلْيَتَيْهِ لِيَفْتِنَهُ عَنْ صَلاتِهِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏14/‏105،،‏(رقم الحديث: 8369).
ن135هـ=752م

أَضْرَطَ الشَّخْصُ بِغَيْرِهِ: أَخَرْجَ مِنْ بَيْنِ شَفَتَيْهِ صَوْتًا يُشْبِهُ الضُّرَاطَ عَلَى سَبِيلِ الِاسْتِخْفَافِ وَالإِنْكَارِ.

"جَاءَ رَجُلٌ إِلَى عَلِيٍّ }  فَسَأَلَهُ عَنِ الوُضُوءِ... فَأَضْرَطَ عَلِيٌّ  }  بِهِ، ثُمَّ دَعَا بِمَاءٍ فَبَدَأَ بِالشِّمَالِ قَبْلَ اليَمِينِ"
زياد بن أبي زياد/ميسرة
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏1/‏261،،‏(رقم الحديث: 409).

ضُرَاط[اسْم]

ن11هـ=632م

الضُّرَاطُ : صَوْتُ الرِّيحِ الخَارِجَةِ مِنَ الدُّبُرِ.

"إِذَا نُودِيَ بِالصَّلَاةِ أَدْبَرَ الشَّيْطَانُ وَلَهُ ضُرَاطٌ حَتَّى لَا يَسْمَعَ التَّأْذِينَ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏13/‏486،،‏(رقم الحديث: 8139).

ضَرَّطَ[مُتَعَدٍّ]

21هـ=642م

ضَرَّطَ غَيْرَهُ: جَعَلَهُ يَضْرِطُ.

"وَاللَّهِ، لَئِنْ جِئْتُكَ لَأُضَرِّطَنَّكَ "
الأشعث بن قيس الكنديّ
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏6/‏271‏.

ضَرْطَة[اسْمُ مَرَّة]

ن35هـ=655م

الضَّرْطَةُ : المَرَّةُ مِنَ الضَّرْطِ.

"ثُمَّ وَعَظَهُمْ فِي الضَّحِكِ مِنَ الضَّرْطَةِ "
عبد الله بن زَمْعَة القُرَشيّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏26/‏160،،‏(رقم الحديث: 16222).

مُضَارَطَة[مَصْدَر]

ن58هـ=678م

المُضَارَطَةُ : المُغَالَبَةُ فِي الضُّرَاطِ.

"هُوَ المُضَارَطَةُ "
عائشة بنت أبي بكر الصِّدّيق
غرائب التفسير وعجائب التأويل: محمود بن حمزة الكرماني، تحقيق: شمران سركال يونس العجلي، دار القبلة للثقافة الإسلامية، جدّة المملكة العربية السعودية، ومؤسسة علوم القرآن، بيروت، لبنان، (1403ه- 1983م)، ‏1/‏881‏.

مُضْرِط[اسْمُ فَاعِل]

ن80هـ=699م

المُضْرِطُ العَجِينَ وَنَحْوَهُ: المُخْرِجُ مِنْهُ صَوْتًا يُشْبِهُ الضُّرَاطَ.

فَلَا أَسَدًا أَسُبُّ، وَلَا تَمِيمًا وَكَيْفَ يَحِلُّ سَبُّ الأَكْرَمِينَا؟!
وَلَكِنَّ التَّقَارُضَ حَلَّ بَيْنِي وَبَيْنَكَ، يَا ابْنَ مُضْرِطَةِ العَجِينَا
الأُقيشر الأسديّ
ديوان الأُقَيْشِر الأسدي أخباره وشعره: صنعة: محمّد علي دقّة، دار صادر، بيروت، ط1، (1997م)، ‏123‏.

ضَارِط[اسْمُ فَاعِل]

86هـ=705م

الضَّارِطُ : المُخْرِجُ مِنْ دُبُرِهِ رِيحًا ذَاتَ صَوْتٍ.

وَيَضْرُطُ ضَارِطٌ مِنْ غَمْزِ قَوْسٍ فَيَحْبُوهُ الأَمِيرُ بِهَا بُدُورَا؟
الأعور الكلبيّ
ديوان شُعراء بني كلب: صنعة: محمّد شفيق البيطار، دار صادر، بيروت، ط1، (2002م)، ‏1/‏492‏.

ضَرِيط[اسْم]

ن100هـ=718م

الضَّرِيطُ : الضُّرَاطُ.

ثُمَّ أَنْحَى بِجَعْرِهِ حَاجِبَ الشَّمْـ ـسِ، فَأَلْقَى كَالمِعْلَفِ المَهْدُومِ
بِضَرِيطٍ تَرَى الخَنَازِيرَ مِنْهُ عَامِدَاتٍ لِتَلِّهِ المَرْكُومِ
الحكم بن عَبْدَل الأسديّ
شعر الحكم بن عبدل الأسدي: صنعة: محمّد نايف الدليمي، مجلة المورد، بغداد، مج5، ع4، (1397هـ/ 1976م)، ‏113‏.

أَضْرَط[اسْمُ تَفْضِيل]

ن167هـ=783م

الأَضْرَطُ : الأَكْثَرُ ضَرْطًا.

وَأَضْرَطَ مِنْ أُمِّ مُبْتَاعِهَا إِنِ اقْتَحَمَتْ بَكْرَةً حَرْمَلَا
بشّار بن برد
ديوان بشار بن برد: قرأه وقدم له: إحسان عبّاس، دار صادر، بيروت، ط1، (2000م)، ‏399‏.

ضُرَّيْط[مَصْدَر]

ن175هـ=791م

الضُّرَّيْطُ : إِحْدَاثُ صَوْتٍ بِالفَمِ يُشْبِهُ الضُّرَاطَ اسْتِخْفَافًا.

"الأَكْلُ سُرَّيْط، وَالقَضَاءُ ضُرَّيْط "
أبو الدُّقَيْش القَناني الغَنَويّ الأعرابيّ
كتاب الأمثال: مؤرِّج السَّدوسيّ (ت، 195هـ)، حققه وقدم له ووضع فهارسه: رمضان عبد التوّاب، دار النهضة العربية، بيروت، (1983م)، ‏80‏.

تَضَارَطَ[لَازِم]

ن207هـ=822م

تَضَارَطَ الشَّخْصَانِ: تَغَالَبَا، وَتَنَافَسَا فِي الضَّرْطِ.

" تَضَارَطَ أَعْرَابِيَّانِ عِنْدَ خَالِدِ بْنِ عَبْدِ اللهِ"
الهيثم بن عَديّ الطّائيّ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏4/‏412‏.

ضُرَّيْطَى[اسْم]

ن215هـ=830م

الضُّرَّيْطَى : الَّذِي إِذَا تَقَاضَاهُ صَاحِبُهُ أَضْرَطَ بِهِ.

"الأَكْلُ سُرَّيْطَى، وَالقَضَاءُ ضُرَّيْطَى "
مثل
المختار من كتاب الأمثال: أبو زيد الأنصاريّ (ت، 215هـ)، تحقيق: جليل العطية، مجلة المورد، بغداد، مج15، ع2، (1406هـ/ 1986م)، ‏82‏.

مَضْرَط[اسْم]

232هـ=846م

المَضْرَطُ : الِاسْتُ، مَوْضِعُ خُرُوجِ الرِّيحِ مِنَ الشَّرْجِ مَعَ صَوْتٍ.

لَرَهْزَةٌ مِنْ غَوِيٍّ فِي مَضَارِطِهَا أَشْهَى إِلَى قَلْبِهَا مِنْ أَلْفِ دِينَارِ
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏2/‏1107‏.

مُتَضَارِط[اسْمُ فَاعِل]

246هـ=860م

المُتَضَارِطُ : الَّذِي يَتَكَلَّفُ الضُّرَاطَ.

وَمَتَى رَأَوْا دِيكًا، وَلَوْ مِنْ فَرْسَخٍ أَبْصَرْتَهُمْ يَعْدُونَ عَدْوَ مُبَالِطِ
فَهُمُ لِغَوْغَاءِ القَبِيلَةِ لُعْبَةٌ فِي عَسْكَرٍ مُتَضَاحِكٍ مُتَضَارِطِ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏4/‏1447‏-‏1448‏.

ضَارَطَ[مُتَعَدٍّ]

ن260هـ=874م

ضَارَطَ الشَّخْصُ غَيْرَهُ: غَالَبَهُ فِي الشِّدَّةِ وَالضَّرْطِ.

مَنْ يُضَارِطْنِي يُضَارِطْ مُوسِرًا يُخْرِجُ الضَّرْطَةَ كَالرَّعْدِ القَصِفْ
سعيد بن حُميد الكاتب
شعراء عبّاسيّون: تحقيق: يونس أحمد السامرائيّ، عالم الكتب- مكتبة النهضة العربية، بيروت، ط2، (1411هـ/ 1990م)، ‏3/‏291‏.

تَضَارُط[مَصْدَر]

270هـ=883م

التَّضَارُطُ : المُشَارَكَةُ فِي فِعْلِ الضُّرَاطِ.

"وَكَانَتِ الخَبَائِثُ الَّتِي يَعْمَلُونَهَا: إِتْيَانَ الذُّكْرَانِ فِي أَدْبَارِهِمْ، وَخَذْفَهُمُ النَّاسَ، وَتَضَارُطَهُمْ فِي أَنْدِيَتِهِمْ"
الطّبريّ
تفسير الطّبريّ جامع البيان عن تأويل آي القرآن: الطّبريّ (ت، 310هـ)، تحقيق: عبد الله بن عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1422هـ/ 2001م)، ‏16/‏318‏.

ضَرَّاط[صِيغَةُ مُبَالَغَة]

ن283هـ=896م

الضَّرَّاطُ : الكَثِيرُ الضَّرْطِ.

إِذَا تَغَنَّتْ شُنْطُفٌ مَرَّةً فَاصْفَعْ، وَدَعْ عَنْكَ الأَبَاطِيلَا
ضَرَّابَةٌ بِالطَّبْلِ ضَرَّاطَةٌ تُجِيبُ بِالتَّطْبِيلِ تَطْبِيلَا
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏5/‏1983‏.

تَضْرِيط[مَصْدَر]

295هـ=908م

تَضْرِيطُ الشَّخْصِ: أَنْ يَجْعَلَهُ غَيْرُهُ يَضْرَطُ.

ثُمَّ وَلَّى مِنْ زَعْقَةٍ وَمُحَامُو هُ وَمَنْ خَلْفَهُمْ لَهُمْ تَضْرِيطُ
يحيى بن عليّ بن يحيى المُنجّم
الكامل في التاريخ: ابن الأثير الجَزريّ (ت، 630هـ)، حققه واعتنى به: عمر عبد السلام تدمري، دار الكتاب العربي، بيروت، (2012م)، ‏6/‏572‏.
1416هـ=1995م

التَّضْرِيطُ : الضُّرَاطُ.

"وَكَانَتْ زَوْجَةُ الخَلِيفَةِ- المَشْغُولِ عَنْهَا بِظِلِّه وِسْرَارِي ظِلِّه- تُنَفِّسُ عَنْ كُرْبَتِهَا بِالتَّسَوُّقِ، وَالمُوَاظَبَةِ عَلَى تَعَلُّمِ وَصَفَاتٍ مَطْبَخِيَّة، وَتَكْثِيفِ مُشَاهَدَةِ المُسَلْسَلَاتِ الرَّمَضَانِيَّة، وَالتَّضْرِيطِ ، وَمُدَاعَبَةِ خَادِمَاتِهَا وَوَصِيفَاتِهَا"
رامز عوض
إعادة نظر: رامز عوض، دار الحلم للنشر والتوزيع والترجمة، القاهرة، مصر، (1416ه- 1995م)، ‏82‏.

مُضَارِط[اسْمُ فَاعِل]

ن360هـ=971م

المُضَارِطُ غَيْرَهُ: المُغَالِبُ لَهُ فِي الضُّرَاطِ.

وَمَنْ يَلْقَ حُمْرَ الوَحْشِ يومًا مُضَارِطًا يَصِيرُ حِتارُ الدُّبْرِ مِنْهُ مُمَزَّقًا
مثَل مِن الشِّعْر
كتاب الأمثال الصّادرة عن بيوت الشِّعر: حمزة بن الحسن الأصفهانيّ (ت، 360هـ)، تحقيق: أحمد بن محمّد الضبيب، دار المدار الإسلامي، بيروت، ط1، (2009م)، ‏684‏.

مَضْرَطَة[مَصْدَرٌ مِيمِيّ]

ن383هـ=993م

المَضْرَطَةُ : المَدْعَاةُ إِلَى الضُّرَاطِ.

"لَيْسَ لَهُ ذَنْبٌ إِلَّا أَنَّ جَدَّهُ النَّبِيُّ، وَأَبُوهُ الوَصِيُّ... وَحُقُوقُهُ مَصْرُوفَةٌ إِلَى القَهْرَمَانَةِ وَالمَضْرَطَةِ "
أبو بكر الخوارزميّ
رسائل أبي بكر الخوارزميّ: أبو بكر الخوارزميّ (ت، 383هـ)، مطبعة الجوائب، قسطنطينية، ط1، (1297هـ)، ‏138‏.

مُضَرِّط[اسْمُ فَاعِل]

ن425هـ=1034م

المُضَرِّطُ : المُسَبِّبُ لِلضُرَاطِ.

"وَيُقَالُ لِنَبِيذِ التَّمْرِ: المُنْفِخَةُ المُضَرِّطَةُ "
الرّاغب الأصفهانيّ
مجْمع البلاغة: الرّاغب الأصفهانيّ (ت، 425هـ)، تحقيق: عُمر عبد الرَّحمن السّاريسيّ، مكتبة الأَقصى، عمّان، ط1، (1406ه/ 1986م)، ‏2/‏602‏.

ضَرَطَان[مَصْدَر]

436هـ=1044م

الضَّرَطَانُ : إِخْرَاجُ صَوْتٍ مِنَ الشَّفَتَيْنِ يُشْبِهُ الضُّرَاطَ عَلَى وَجْهِ الِاسْتِخْفَافِ وَالإِنْكَارِ.

"الأَكْلُ سَرَطَان، وَالقَضَاءُ ضَرَطَان "
من رواية ابن سِيدَهْ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏8/‏119‏.

اضْطَرَطَ[مُتَعَدٍّ بِالحَرْف]

ن594هـ=1197م

اضْطَرَطَ الشَّخْصُ بِغَيْرِهِ: أَخَرْجَ مِنْ بَيْنِ شَفَتَيْهِ صَوْتًا يُشْبِهُ الضُّرَاطَ عَلَى سَبِيلِ الِاسْتِخْفَافِ وَالإِنْكَارِ.

"إِذَا أَخَذَ المَالَ سَرَطَ، وَإِذَا طُولِبَ بِالأَدَاءِ اضْطَرْطَ بِصَاحِبِهِ"
الخُوَيِّي
فرائِد الخرائِد في الأمثال: الخُوَيِّي (ت، 549هـ)، تحقيق: عبد الرزّاق حسين، دار النَّفائس، عمّان، ط1، (1421ه/ 2000م)، ‏46‏.

ضِرَاطَة[مَصْدَر]

ن686هـ=1287م

الضِّرَاطَةُ : الضَّرْطُ.

"يَا خَصَافِ، وَيَا حَبَاقِ، كِلَاهُمَا بِمَعْنَى الضِّرَاطَةِ "
الرّضِيّ الأَسْتَرآباذي
شرح الرضي لكافية ابن الحاجب: الرّضِيّ الأَسْتَرآباذي (ت، 686هـ)، دراسة وتحقيق: حسن بن محمد بن إبراهيم الحفظي، يحيى بشير مصري، جامعة الإمام محمد بن سعود الإسلامية، القاهرة، ط1، (1414ه/ 1993م- 1417ه/ 1996م)، ‏2/‏329‏.

ضُرَاطِيّ[اسْمٌ مَنْسُوب]

690هـ=1291م

الضُّرَاطِيُّ : المَنْسُوبُ إِلَى الضُّرَاطِ.

"لُقِّبَ بِذَلِكَ لِأَنَّ شَخْصًا مَازَحَهُ، فَكَتَبَ إِلَيْهِ كِتَابًا، وَجَعْلَ عُنْوَانَهُ: المَجْلِسُ الشَّامِيُّ الفَاضِلِيُّ الضُّرَاطِيُّ "
شخص يكتب إلى شمس الدين الخابوري
نزهة الألباب في الألقاب: أبو الفضل أحمد بن علي بن محمد بن أحمد بن حجر العسقلاني، المحقق: عبد العزيز محمد بن صالح السديري، مكتبة الرشد، الرياض، المملكة العربية السعودية، ط1، (1409ه- 1989م)، ‏1/‏436‏.

ضَرْطِيّ[اسْمٌ مَنْسُوب]

ن1315هـ=1897م

الضَّرْطِيُّ : المَنْسُوبُ إِلَى الضَّرْطِ.

"وَذَاتَ يَوْمٍ دَخَلَ الأَمِيرُ عَلَى السُّجَنَاءِ يَتَفَقَّدُ أَحْوَالَهُمْ بِنَفْسِهِ...، وَمَرَّ عَلَى الهَجْلِيِّ، فَقَالَ لَهُ بِلَهْجَةِ المُسْتَهْزِئِ: يَا الهَجْلِيُّ، يَا الضَّرْطِيُّ "
محمد بن عبد الله الرشيد
فتافيت من المواقف والطرائف والتنكيت: عبدالرحمن بن زيد السويداء، دار السويداء للنشر والتوزيع، الرياض، المملكة العربية السعودية، (1414ه- 1993م)، ‏1/‏36‏.

إِضْرَاط[مَصْدَر]

1417هـ=1996م

الإِضْرَاطُ : أَنْ يَعْمَلَ بِفِيهِ شِبْهَ الضُّرَاطِ إِنْكَارًا وَاسْتِهْزَاءً.

"أَلَسْتَ بِقَادِرٍ عَلَى مُعَارَضَةِ إِضْرَاطِ صَاحِبِكَ بِإِضْرَاطٍ مِثْلِهِ؟"
الحسين بن علي بن عبد الله
قصص وأمثال من المغرب: الحسين بن علي بن عبد الله، نشر [s.n]،(1417ه- 1996م)، ‏1/‏138‏.

تَضْرِيطِيّ[اسْمٌ مَنْسُوب]

1437هـ=2015م

التَّضْرِيطِيُّ : المَنْسُوبُ إِلَى التَّضْرِيطِ.

"وَرَجَاءً: لَا أَحَدَ يَضْحَكُ؛ فَقَدْ وَجَدْتُهَا، أَثَابَهُ اللهُ عَلَى مَجْهُودِهِ التَّضْرِيطِيِّ وَجَامِعَتَهُ المُحْتَضِنَةَ لِبَحْثِهِ"
الشّبكة اللّيبراليّة الحُرّة
رسالة أوكتابة أو تعابير عن حكم "الضراط" في الاسلام وأنواعه: ‏معلق في الشبكة الليبرالية الحرة،‏الشبكة الليبرالية الحرة: https://libral.org/vb/archive/index.php، ‏، نُشر في: 23-02-2015 م. شوهد في: 06-09-2024 م.

تَضَارُطَة[اسْمُ مَرَّة]

1441هـ=2019م

التَّضَارُطَةُ : المَرَّةُ مِنَ التَّضَارُطِ.

"مَا جَنَيْنَاهُ فِي أَوَّلِ مَرْحَلَةٍ مِنَ التَّضَارُطَاتِ الكَاذِبَةَ- وَلْيَعْذُرْنِي الإِخْوَةُ عَلَى التَّعْبِيرِ- هُوَ صِفْرٌ"
