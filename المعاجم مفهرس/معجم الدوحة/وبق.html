أَوْبَقَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

أَوْبَقَ الشَّخْصَ: أَهْلَكَهُ.

وَهَزَزْتَ سَيْفَكَ؛ كَيْ تُحَارِبَنَا فَانْظُرْ بِسَيْفِكَ، مَنْ بِهِ تُرْدِي
وَأَرَدْتَ خُطَّةَ حَازِمٍ بَطَلٍ حَيْرَانَ، أَوْبَقَهُ الَّذِي يُسْدِي
يزيد بن الخَذّاق الشَّنّيّ العَبْديّ
شعراء عبد القيس وشعرهم في العصر الجاهلي: جمع وتحقيق ودراسة: عبد الحميد المعيني، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري، الكويت، (2002م)، ‏352‏.
5ق.هـ=617م

أَوْبَقَ السُّفُنَ وَنَحْوَهَا: حَبَسَهَا.

قرآن كريم
قرآن كريم، ‏[ الشُّورَىٰ : 34 ].
14هـ=635م

أَوْبَقَ الشَّيْءَ: أَفْسَدَهُ.

"فَإِنْ تَزْهَدُوا فِي الدُّنْيَا، وَتَرْغَبُوا فِي الآخِرَةِ، جَمَعَ اللهُ لَكُمُ الدُّنْيَا وَالآخِرَةِ... وَإِنْ تَفْشَلُوا، وَتَهِنُوا وَتَضْعُفُوا، تَذْهَبْ رِيحُكُمْ، وَتُوبِقُوا آخِرَتَكُمْ"
سعد بن أبي وَقّاص الزُّهْريّ القُرَشيّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏3/‏532‏.

مُوبِق[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

المُوبِقُ : المُهْلِكُ.

وَمَنْ يَلْتَمِسْ حُسْنَ الثَّنَاءِ بِمَالِهِ يَصُنْ عِرْضَهُ مِنْ كُلِّ شَنْعَاءَ مُوبِقِ
زهير بن أبي سُلْمى المُزَنيّ
شعر زُهير بن أبي سُلمى: صنعة: الأعلم الشَّنْتَمَريّ (ت، 476هـ)، تحقيق: فخر الدين قباوة، منشورات الآفاق الجديدة، بيروت، ط3، (1400هـ/ 1980م)، ‏262‏.
ن11هـ=632م

المُوبِقُ مِنَ الذُّنُوبِ: مَا يُعَدُّ مِنَ الكَبَائِرِ، وَيَكُونُ سَبَبًا لِلْهَلَاكِ وَالعُقُوبَةِ فِي الدُّنْيَا وَالآخِرَةِ.

"اجْتَنِبُوا السَّبْعَ المُوبِقَاتِ "
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏4/‏264،،‏(رقم الحديث: 6857).
342هـ=953م

المُوبِقُ لِلْأَمْرِ: المُفْسِدُهُ.

"لَيْسَ المَعْنَى فِيهِ أَنْ لَا يَقْصِدَ إِلَى مَعْرِفَتِهَا، وَلَكِنَّ المَعْنَى أَنْ لَا يَقْصِدَهُمْ بِالعُقُوبَةِ فِيهَا إِذَا كَانَتْ مِمَّا يَجُوزُ تَسْوِيغُهَا وَاحْتِمَالُهَا؛ وَذَلِكَ بِأَنْ لَا تَكُونَ مُوبِقًا لِلدِّينِ، وَلَا مُؤَثِّرًا فِي المَمْلَكَةِ"
محمّد بن يوسف العامريّ النَّيسابوريّ
السّعادة والإسعاد في السّيرة الإنسانيّة: محمّد بن يوسف العامريّ النّيسابوريّ (ت، 381هـ)، كتابة ومباشرة: مجتبى مينوي، انتشارات دانشكاه، طهران، (1957م)، ‏284‏.

مَوْبِق[اسْمُ مَكَان]

1ق.هـ=621م

المَوْبِقُ : مَكَانُ الهَلَاكِ.

قرآن كريم
قرآن كريم، ‏[ الكَهْف : 52 ].
ن209هـ=824م

المَوْبِقُ : المَوْعِدُ.

وَحَادَ شَرَوْرَى وَالسِّتَارَ، فَلَمْ يَدَعْ تِعَارًا لَهُ، وَالوَادِيَيْنِ بِمَوْبِقِ
من شواهد أبي عُبيدة مَعْمَر بن المُثَنَّى
مجاز القرآن: صنعة: أبي عبيدة (ت، 209هـ)، عارضه بأصوله وعلق عليه: محمّد فؤاد سزِكين، مكتبة الخانجي- دار غريب، القاهرة، (1988م)، ‏1/‏406‏.
507هـ=1113م

المَوْبِقُ : المَكَانُ الحَاجِزُ بَيْنَ شَيْئَيْنِ.

"لَوْلَا أَنَّ مَلَاذِي... أَخَذَ عَلَيَّ مَوْثِقا، وَقَدْ جَعَل البُعْدُ بَيْنِي وَبَيْنَهُ مَوْبِقا ، فِي إِعْلَامِي مَجْدَه، بِمَا تَجَرَّدَ مِنْ أَنْبَائِي بَعْدَه، لَتَثَاقَلْتُ هَيْبَة، وَتَوَارَيْتُ غَيْبَة"
ابن أبي الخِصال
رسائل ابن أبي الخصال: ابن أبي الخِصال (ت، 540هـ)، تحقيق: محمَّد رضوان الدّاية، دار الفكر، دمشق، ط1، (1408ه/ 1988م)، ‏459‏.

مَوْبِق[مَصْدَرٌ مِيمِيّ]

1065هـ=1654م

المَوْبِقُ : الهَلَاكُ.

"فَبُعْدًا لِلْمُغْتَرِّ بِهَذِهِ الدَّار، الَّتِي مَا بَرِحَتْ تَرْمِي أَبْنَاءَهَا بِالمَوَابِقِ الكِبَار"
يحيى بن الحسين بن القاسم بن محمد
غاية الأماني في أخبار القطر اليماني: يحيى بن الحسين بن القاسم بن محمد (ت، 1100هـ)، تحقيق وتقديم: سعيد عبد الفتاح عاشور، مراجعة: محمد مصطفى زيادة، دار الكاتب العربي، القاهرة، (1388ه/ 1968م)، ‏2/‏728‏.

مُوبَق[اسْمُ مَفْعُول]

ن11هـ=632م

المُوبَقُ : المُهْلَكُ.

"فَتَخْطَفُ النَّاسَ بِأَعْمَالِهِمْ؛ مِنْهُمُ المُوبَقُ بِعَمَلِهِ، وَمِنْهُمُ المُخَرْدَلُ"
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏4/‏204،،‏(رقم الحديث: 6573).
102هـ=720م

المُوبَقُ مِنَ الأَشْخَاصِ وَنَحْوِهِمْ: المَحْبُوسُ.

وَمَتَى يُؤَامِرْ نَفْسَهُ مُسْتَخْلِيًا فِي أَنْ تَجُودَ لَدَى السُّؤَالِ، تَقُولُ: جُدْ
أَوْ فِي الوُفُودِ عَلَى أَسِيرٍ مُوبَقٍ بَخِلَتْ أَقَارِبُهُ عَلَيْهِ، تَقُولُ: فِدْ
حمزة بن بيض الحَنَفيّ
معجم الأدباء إرشاد الأريب إلى معرفة الأديب: ياقوت الحَمَوي (ت، 626هـ)، تحقيق: إحسان عبّاس، دار الغرب الإسلاميّ، بيروت، ط1، (1993م)، ‏3/‏1216‏-‏ 1217‏.

إِيبَاق[مَصْدَر]

ن40هـ=660م

الإِيبَاقُ : الإِهْلَاكُ.

"الدُّنْيَا غُرُورٌ حَائِل، وَزُخْرُفٌ زَائِل... فَكَمْ مِنْ وَاثِقٍ بِهَا، راكِنٍ، قَدْ أَرْهَقَتْهُ بِإِيبَاقِها ، وَأَعْلَقَتْهُ بِأَرْبَاقِها"
عليّ بن أبي طالب
كتاب المجتنى: ابن دُريد (ت، 321هـ)، تحقيق: محمّد عبد المعيد خان، دائرة المعارف العثمانية، حيدر آباد، ط3، (1382هـ/ 1963م)، ‏20‏.

وَبِق[صِفَةٌ مُشَبَّهَة]

ن40هـ=660م

الوَبِقُ : الهَالِكُ.

"تَمِيدُ بِأَهْلِهَا مَيَدَانَ السَّفِينَةِ تَقْصِفُهَا العَوَاصِفُ فِي لُجَجِ البِحَارِ، فَمِنْهُمُ الغَرِقُ الوَبِقُ ، وَمِنْهُمُ النَّاجِي عَلَى بُطُونِ الأَمْوَاجِ"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏2/‏170‏-‏ 171‏.

وَبَقَ[لَازِم]

ن83هـ=702م

وَبَقَ الشَّخْصُ وَغَيْرُهُ: هَلَكَ.

أَسْتَغْفِرُ اللهَ أَعْمَالِي الَّتِي سَلَفَتْ مِنْ عَثْرَةٍ، إِنْ يُعَاقِبْنِي بِهَا أَبِقِ
أعشى هَمْدان
ديوان أعشى هَمدان وأخباره: تحقيق: حسن عيسى أبو ياسين، دار العلوم، الرياض، (1403هـ/ 1983م)، ‏147‏.

مَوْبُوق[اسْمُ مَفْعُول]

ن175هـ=791م

المَوْبُوقُ : المَحْبُوسُ.

"صَادَفْتُ -وَاللهِ- هُنَاكَ بَابًا مَغْلُوقا، وَعِلْمًا مَوْبُوقا "

أخبار القضاة: وَكيع الضَّبّيّ (ت، 306هـ)، مراجعة: سعيد محمّد اللحّام، عالم الكتب، بيروت، د.ت. ‏3/‏177‏.

اسْتَوْبَقَ[مُتَعَدٍّ]

272هـ=885م

اسْتَوْبَقَ الشَّيْءَ: حَبَسَهُ.

كَانَتْ بَشَاشَتُكَ الأُولَى الَّتِي ابْتَدَأَتْ بِالبِشْرِ، ثُمَّ اقْتَبَلْنَا بَعْدَهَا النِّعَمَا
كَالمُزْنَةِ اسْتَوْبَقَتْ أُولَى مَخِيلَتِهَا ثُمَّ اسْتَهَلَّتْ بِغُزْرٍ تَابَعَ الدِّيَمَا
البُحْتُريّ
أخبار البُحتريّ: أبو بكر الصّوليّ (ت، 335هـ)، حققها وعلق عليها: صالح الأشتر، مطبوعات المجمع العلمي العربي، دمشق، ط1، (1378هـ/ 1958م)، ‏61‏.
1250هـ=1834م

اسْتَوْبَقَ الشَّخْصَ: أَهْلَكَهُ.

"(اشْتَرُوا أَنْفُسَكُمْ) قَالَ ذَلِكَ  @  لَهُمْ؛ لِأَنَّهُمْ أَرْهَنُوا أَنْفُسَهُمْ، وَاسْتَوْبَقُوهَا بِالشِّرْكِ، إِنْ لَمْ يَسْتَنْقِذُوهَا بِكَلِمَةِ التَّوْحِيدِ مِنَ النَّار، وَغَضَبِ الجَبَّار"
عثمان بن عبد العزيز التميمي
فتح الحميد في شرح التوحيد: عثمان بن عبد العزيز بن منصور التميميّ (ت،1282هـ)، تحقيق: سعود بن عبد العزيز العريفي، حسين بن جليعب السعيدي، دار عالم الفوائد، مكة المكرمة، ط1، 1425هـ، ‏2/‏743‏.

أَوْبَق[اسْمُ تَفْضِيل]

ن328هـ=940م

الأَوْبَقُ : الأَكْثَرُ إِهْلَاكًا.

"أَلْهَمَكَ اللهُ -يَا أَخِي- مِنَ الأُمُورِ البَهِيَّةِ أَحْسَنَهَا وَأَوْفَقَها، وَصَرَفَ عَنْكَ مِنَ المُحْزِنَاتِ الرَّدِيئَةِ أَعْظَمَهَا وَأَوْبَقَها "
أفتيشيوس المكنّى بسعيد بن بطريق
كتاب التّاريخ المجموع على التّحقيق والتّصديق: أفتيشيوس المكنّى بسعيد بن بطريق (ت، 328هـ)، مطبعة الآباء اليسوعيّين، بيروت، (1905م)، ‏1/‏3‏.

مُوَابَقَة[مَصْدَر]

ن437هـ=1045م

المُوَابَقَةُ : الوُقُوعُ فِي المَهَالِكِ.

"وَيَأْتِي فِي هَذَا المَعْنَى بِمَا يَسْتَوْفِيه، وَيَكْفِي فِيه، مِمَّا يُرْهِبُ أَهْلَ الخِلَاف، وَيَقُودُهُمْ إِلَى المُوَابَقَةِ بِجَرَائِمِ الِاعْتِسَاف"

وَبَق[مَصْدَر]

ن1260هـ=1843م

الوَبَقُ : الهَلَاكُ.

فَمَنْ يُضَاهِيهِ فِي عِزٍّ، وَفِي شَرَفٍ فَهْوَ الجَهُولُ، الَّذِي يَنْحُو إِلَى الوَبَقِ
عبد اللطيف فتح الله
ديوان المفتي عبد اللطيف فتح الله: عبد اللطيف فتح الله (ت، 1260هـ)، حققه: زهير فتح الله، مراجعة: محمد الحجيري، المعهد الألماني للأبحاث الشرقية، بيروت، (1404ه/ 1984م)، ‏1/‏307‏.
