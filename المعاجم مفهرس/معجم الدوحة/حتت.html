حَتّ[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الحَتُّ مِنَ الخَيْلِ وَنَحْوِهَا: السَّرِيعُ العَرَقِ.

حَتُّ الجِرَاءِ، مُعَاوِدٌ سَبْقَ الحَلَائِبِ، وَالقَنِيصْ
أبو دُواد الإياديّ
ديوان أبي دواد الإياديّ: جمعه وحققه: أنوار محمود الصالحي، أحمد هاشم السامرائي، دار العصماء، دمشق، ط1، (1431هـ/ 2010م)، ‏122‏.
قبل 13ق.هـ=609م

الحَتُّ مِنَ الخَيْلِ وَنَحْوِهَا: السَّرِيعُ العَدْوِ.

مِنْ كُلِّ حَتٍّ ، إِذَا مَا ابْتَلَّ مُلْبَدُهُ ضَافِي السَّبِيبِ، أَسِيلِ الخَدِّ، يَعْبُوبِ
سَلامة بن جَنْدل التَّميميّ
ديوان سلامة بن جندل: صنعة: محمّد بن الحسن الأحول، تحقيق: فخر الدين قباوة، دار الكتب العلمية، بيروت، ط2، (1407هـ/ 1987م)، ‏96‏.
ن7ق.هـ=615م

الحَتُّ : المَتَسَاقِطُ مَا عَلَيْهِ مِنْ وَرَقٍ وَرِيشٍ وَنَحْوِهِمَا.

كَأَنَّ مُلَاءَتَيَّ عَلَى هِزَفٍّ يَعُنُّ، مَعَ العِشِيَّةِ، لِلرِّئَالِ
عَلى حَتِّ البُرَايَةِ زَمْخَرِيِّ السْ سَوَاعِدِ فِي شَرْيٍ طِوَالِ
الأعلم الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏83‏-‏ 84‏.
215هـ=830م

الحَتُّ مِنَ الأَشْخَاصِ: الوَاسِعُ الخَطْوِ.

لَسَوْفَ نَعْلُوكِ بِكُلِّ بَازِلِ حَتِّ الفُرُوجِ، لَيِّنِ المَفَاصِلِ
من شواهد الأخفش
صفة جزيرة العرب: الحسن بن أحمد الهَمْدانيّ (ت، 334هـ)، تحقيق: محمّد بن علي الأكوع، مكتبة الإرشاد، صنعاء، ط1، (1410هـ/ 1990م)، ‏382‏.
ن244هـ=858م

الحَتُّ مِنَ الأَشْخَاصِ: المُسْتَأْصَلُ مَا عِنْدَهُ مِنْ خَيْرٍ.

"تَرَكَهُ اللهُ حَتًّا فَتَّا، لَا يَمْلَأُ كَفَّا"
ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏425‏.
ن412هـ=1021م

الحَتُّ مِنَ التَّمْرِ: الَّذِي لَا يَلْتَصِقُ بَعْضُهُ بِبَعْضٍ.

"جَاءَنَا بِتَمرٍ حَتٍّ "
محمّد بن جعفر القزّاز القيروانيّ
كتاب العشرات في اللُّغة: محمّد بن جعفر القزّاز القيروانيّ (ت، 412هـ)، تحقيق وتعليق: يحيى عبد الرؤوف جبر، عمان، (1404هـ/ 1984م)، ‏60‏.

حَتّ[مَصْدَر]

قبل 13ق.هـ=609م

حَتُّ الشَّيْءِ: نَحْتُهُ وَتَقْشِيرُهُ.

وَفِي كُلِّ عَامٍ لَهُ غَزْوَةٌ تَحُتُّ الدَّوَابِرَ حَتَّ السَّفَنْ
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏1/‏141‏.
ن175هـ=791م

حَتُّ الشَّيْءِ: فَرْكُهُ وَإِزَالَتُهُ عَنِ الثَّوْبِ وَنَحْوِهِ.

" الحَتُّ : فَرْكُكَ شَيْئًا عَنْ ثَوْبٍ وَنَحْوِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏337‏.

حَتّ[اسْم]

ن385هـ=995م

الحَتُّ : الشَّيْءُ.

"وَمَا فِي يَدِي مِنْكَ حَتٌّ "
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏2/‏308‏.

حَتَّ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

حَتَّ الشَّيْءَ: نَحَّاهُ وَأَزَالَهُ.

يَنَامُ عِشَاءً، ثُمَّ يُصْبِحُ نَاعِسًا يَحُتُّ الحَصَى، عَنْ جَنْبِهِ المُتَعَفِّرِ
عروة بن الورد العَبْسيّ
شعر عُروة بن الورد العبسي: صنعة:ابن السِّكّيت (ت، 244هـ)، تحقيق: محمّد فؤاد نعناع، مكتبة دار العروبة، الكويت، مكتبة الخانجي، القاهرة، ط1، (1415هـ/ 1995م)، ‏46‏.
3هـ=624م

حَتَّ الشَّخْصَ: رَدَّهُ وَدَفَعَهُ.

" احْتُتْهُمْ ، يَا سَعْدُ، فِدَاكَ أَبِي، وَأُمِّي"
حديث نبويّ
النهاية في غريب الحديث والأثر: ابن الأثير، تح: طاهر أحمد الزاوي وآخرين، البابي الحلبي، القاهرة، 1963م، ‏1/‏337‏.

حُتَات[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الحُتَاتُ مِنَ الأَلْسِنَةِ: السَّلِيطُ اللَّاذِعُ.

أَدِبُّ عَلَى العَصَا، لَمْ يَبْقَ إِلَّا لِسَانٌ، صَارِمٌ، عَضْبٌ، حُتَاتُ
مرداس بن صُبيح الحَكَميّ المَذْحِجيّ
المعمرون والوصايا: أبو حاتم السجستاني، تح: محمد أمين الخانجي، السعادة، مصر، 1905م، ‏35‏.

ن175هـ=791م

حُتَاتُ الشَّيْءِ: مَا تَقَشَّرَ مِنْهُ وَتَنَاثَرَ.

" حُتَاتُ كُلِّ شَيْءٍ: مَا تَحَاتَّ مِنْهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏21‏.

حِتَّات[مَصْدَر]

ن10هـ=631م

الحِتَّاتُ : الإِفْنَاءُ والإِبَادَةُ.

وَبَعْدَ حِمْيَرَ، إِذْ شَالَتْ نَعَامَتُهُمْ حَتَّتَهُمْ رَيْبُ هَذَا الدَّهْرِ حِتَّاتَا
علقمة ذو جَدَن الحِمْيريّ
معجم البلدان: ياقوت الحَمَوي (ت، 626هـ)، دار صادر، بيروت، (1397هـ/ 1977م)، ‏1/‏535‏.

تَحَاتَّ[لَازِم]

ن11هـ=632م

تَحَاتَّ الشَّيْءُ: تَقَشَّرَ وَتَنَاثَرَ.

"مَثَلُ المُؤْمِنِ كَمَثَلِ شَجَرَةٍ خَضْرَاءَ، لَا يَسْقُطُ وَرَقُهَا، وَلَا يَتَحَاتُّ "
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏4/‏113،،‏(رقم الحديث: 6122).

حَاتَّ[مُتَعَدٍّ]

ن11هـ=632م

حَاتَّ الشَّيْءَ: أَسْقَطَهُ وَأَزَالَهُ.

"مَا مِنْ مُسْلِمٍ يُصِيبُهُ أَذًى إِلَّا حَاتَّ اللهُ عَنْهُ خَطَايَاهُ، كَمَا تَحَاتُّ وَرَقُ الشَّجَرِ"
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏4/‏24،،‏(رقم الحديث: 5647).

انْحَتَّ[لَازِم]

ن34هـ=654م

انْحَتَّ الشَّيْءُ: انْقَشَرَ وَتَسَاقَطَ.

"يَبْعَثُ اللهُ مِنْ بَقِيعِ الغَرْقَدِ سَبْعِينَ أَلْفًا، هُمْ خِيَارُ مَنْ يَنْحَتُّ عَنْ خَطْمِهِ المَدَرُ"
كعب الأحبار
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏3/‏6‏.
231هـ=845م

انْحَتَّ الأَمْرُ: زَالَ وَاخْتَفَى.

قالَ يَصِف ما فَعَلَهُ مَمْدُوحُهُ بِأَحَدِ الخُصُومِ: حَتَّى انْثَنَى العُسْرُ إِلَى يُسْرِهِ
وَانْحَتَّ عَنْ خَدَّيْهِ ذَاكَ العُبُوسْ
أبو تمّام
شرح ديوان أبي تمّام: الخطيب التِّبْريزي (ت، 502هـ)، قدم له ووضع هوامشه وفهارسه: راجي الأسمر، دار الكتاب العربي، بيروت، ط2، (1414هـ/ 1994م)، ‏1/‏378‏.

انْحِتَات[مَصْدَر]

ن40هـ=660م

انْحِتَاتُ الشَّيْءِ: تَقَشُّرُهُ وَتَسَاقُطُهُ.

"وَقَدْ يَتَحَسَّرُ مِنْ رِيشِهِ، وَ يَعْرَى مِنْ لِبَاسِهِ؛ فَيَسْقُطُ تَتْرَى، وَ يَنْبُتُ تِبَاعًا، فَيَنْحَتُّ مِنْ قَصَبِهِ انْحِتَاتَ أَوْرَاقِ الأَغْصَانِ"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏2/‏74‏-‏ 75‏.

"فَلْتَكُنِ الدُّنْيَا فِي عُيُونِكِمْ أَقَلَّ مِنْ حُتَاتَةِ القَرَظِ، وَقُرَاضَةِ الجَلَمِ"

حَيْتُوت[صِفَة]

ن60هـ=680م

الحَيْتُوتُ : الَّذِي يَحُتُّ كُلَّ شَيْءٍ.

يَا جُمْلُ، هَلْ أَنْتِ، قَبْلَ المَوْتِ، سَاقِيَتِي كَأْسَ الحَيَاةِ؟ نَعَمْ، يَا جُمْلُ، لَوْ شِيتِ
أَحْيَيْتِ نَفْسًا، كَمَا أَبْتَتِّهَا قَعَصًا بِمُرْهَفٍ مِنْ سِهَامِ المَوْتِ، حَيْتُوتِ
كعب بن مشهور المُخَبَّليّ الخَثْعَميّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏809‏.

تَحَاتّ[مَصْدَر]

ن110هـ=728م

تَحَاتُّ الوَرَقِ وَنَحْوِهِ: تَسَاقُطُهُ.

"ثُمَّ مَا أَعْلَمُ مِنْ بَلْدَةٍ عَلَى وَجْهِ الأَرْضِ أَنَّ أَحَدًا يَمْشِي، فَيَكُونُ فِي مَشْيِهِ ذَلِكَ تَكْفِيرُ الخَطَايَا وَتَحَاتُّ الذُّنُوبِ، كَمَا تَتَحَاتُّ وَرَقُ الشَّجَرِ اليَابِسِ، إِلَّا بِمَكَّةَ"
الحسن البصريّ
أخبار مكّة في قديم الدّهر وحديثه: محمّد بن إسحاق الفاكهيّ (ت، 272هـ)، دراسة وتحقيق: عبد الملك بن عبد الله بن دهيش، دار خضر، بيروت، ط2، (1414هـ/ 1994م)، ‏2/‏293‏.
1406هـ=1985م

التَّحَاتُّ: تَلَفٌ لِلْمَادَّةِ عَبْرَ حَكِّهَا بِالسَّوَائِلِ. [الميكانيكا]

"وَأَيْضًا فَإِنَّ المَوَانِئَ كَثِيرًا مَا تَكُونُ مَحْمِيَّةً بِأَحَدِ الرُّؤُوسِ البَحْرِيَّةِ، وَالرَّأْسُ عِبَارَةٌ عَنْ نُتُوءٍ مِنْ صَخْرٍ صَلْدٍ قَاوَمَ التَّحَاتَّ أَكْثَرَ مِمَّا حَوْلَهُ" الغَرْبِيِّ.
محمد التّازي
شمال إفريقيا في عالم البحث الأبيض المتوسط، اسطيفان كسيل، تعريب محمد التازي سعود، مجلة كلية الآداب والعلوم الانسانية: مجلة كلية الآداب والعلوم الانسانية، جامعة سيدي محمد بن عبد الله، فاس، عدد خاص، (1985م)، ‏15‏.

مُنْحَتّ[صِفَةٌ مُشَبَّهَة]

ن145هـ=762م

المُنْحَتُّ مِنَ الشَّعَرِ وَنَحْوِهِ: المُتَسَاقِطُ.

وَإِنْ حَدَا مِنْ قَلِقَاتِ الخُرْتِ خِمْسٌ، كَحَبْلِ الشَّعَرِ المُنْحَتِّ إِذَا بَنَاتُ الأَرْحَبِيِّ الأَفْتِ قَارَبْنَ أَقْصَى غَوْلِهِ بِالمَتِّ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏24‏.

مُحَتَّت[اسْمُ مَفْعُول]

ن167هـ=783م

المُحَتَّتُ : المُقَشَّرُ.

لِنِسَاءِ الزِّنْجِيِّ، فِيمَنْ يُصَلِّي، صَدَقَاتٌ، فَضَحْنَ بِنْتًا، وَأُخْتَا
وَعَلَى وَجْهِهِ المُحَتَّتِ سِيمَا خَبَّرَتْنَا عَنِ اسْتِهِ، خَبُثَ اسْتَا
بشّار بن برد
ديوان بشار بن برد: قرأه وقدم له: إحسان عبّاس، دار صادر، بيروت، ط1، (2000م)، ‏156‏.

مُنْحَاتّ[اسْمُ مَفْعُول]

ن231هـ=845م

المُنْحَاتُّ : المَحْتُوتُ.

"هُوَ شَجَرٌ يَشْتَهِيهِ الضَّبُّ، فَيَسْحَجُهَا بِذَنَبِهِ، حَتَّى تَحَاتَّ، فَيَأْكُلُ المُنْحَاتَّ "
أبو عبد الله ابن الأعرابيّ اللُّغويّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏2/‏297‏.

حَتَّة[اسْمُ مَرَّة]

ن255هـ=869م

الحَتَّةُ : المَرَّةُ مِنَ التَّقَشُّرِ.

فَإِذَا اسْتَأْنَيْتَهُ جُمُعَةً، ثُمَّ أَطْلَقْتَهُ، فَلَهُ نَفْضَةٌ، وَسَمْطَةٌ، وَحَتَّةٌ ، وَحِينَئِذٍ يُتَائِمُ البُسْرُ؛ يَخْرُجُ ثَلَاثٌ فِي قِمَعٍ"
أبو حاتم السِّجِسْتانيّ
كتاب النّخْلة: أبو حاتم السِّجِسْتانيّ (ت، 255هـ)، تحقيق: حاتم الضامن، دار البشائر الإسلامية، بيروت، ط1، (1422هـ/ 2002م)، ‏69‏.

مَحْتُوت[اسْمُ مَفْعُول]

ن283هـ=896م

المَحْتُوتُ مِنَ الوَرَقِ وَنَحْوِهِ: المَنْزُوعُ المُسَاقَطُ.

لَنْ يَضُرَّ الأُصُولَ، وَهْيَ رَوَاسٍ، وَرَقٌ، مِنْ فُرُوعِهَا، مَحْتُوتُ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏367‏.

مُتَحَاتّ[اسْمُ فَاعِل]

316هـ=928م

المُتَحَاتُّ : المُتَفَتِّتُ المُتَآكِلُ.

"ثُمَّ يُقَالُ: فِي الأَسْنَانِ إِنْ كَانَتْ فُلْجًا، قِيلَ: أَفْلَجُ، وَإِنْ كَانَتْ طِوَالًا جِدًّا، قِيلَ: أَشْغَى، وَإِنْ كَانَتْ صِغَارًا مُتَحَاتَّةً ، قِيلَ: أَكَسُّ"
قُدامة بن جعفر البغداديّ
الخَراج وصناعة الكتابة: قُدامة بن جعفر البغداديّ (ت، 337هـ)، شرح وتعليق: محمّد حسين الزبيدي، دار الرشيد، بغداد، (1981م)، ‏25‏.

تَحَتَّتَ[لَازِم]

ن366هـ=976م

تَحَتَّتَ الشَّيْءُ: تَقَشَّرَ وَتَسَاقَطَ.

وَأَحْضَرَنَا مَحْبُوسَةً، طُولَ لَيْلِهَا مُعَذَّبَةً، بِالنَّارِ، مُسْعَرَةً كَرْبَا
تحَتَّتَ مِنْ رَطْبِ الذُّؤَابَةِ لَحْمُهَا وَمِنْ يَابِسِ الحَبِّ الثَّقِيلِ، لَهَا الحُبَّا
السَّرِيّ الرَّفّاء الكنديّ
ديوان السَّرِيّ الرَّفّاء: تقديم وشرح: كرم البستاني، مراجعة: ناهد جعفر، دار صادر، بيروت، ط1، (1996م)، ‏54‏.

حَاتّ[اسْمُ فَاعِل]

ن414هـ=1023م

حَاتُّ الوَرَقِ: جَاعِلُهُ يَتَسَاقَطُ.

"وَصُنْ أَسْرَارَنَا مَعَكَ عَنْ أَعْدَائِنَا فِيكَ، الحَاتِّينَ لِوَرَقِنَا مِنْ أَجْلِكَ"
أبو حيّان التَّوحيديّ
الإشارات الإلهيّة: أبوحيّان التّوحيديّ (ت، 414هـ)، حققه وقدم له: عبد الرحمن بدويّ، مطبعة جامعة فؤاد الأول، القاهرة، (1950م)، ‏197‏.

1270هـ=1853م

تَحَتُّتُ الشَّيْءِ: تَآكُلُهُ وَتَسَاقُطُ بَعْضٍ مِنْهُ.

حِتَّة[اسْم]

ن1398هـ=1977م

الحِتَّةُ : المَوْضِعُ وَالمَحَلُّ.

"وَلَمَّا كُنَّا " الأَفَنْدِيَّةَ" الوَحِيدِينَ المَوْجُودِينَ فِي الحِتَّةِ فَقَدْ لَجَأَ إِلَيْنَا عَسَى أَنْ نَجْبُرَ بِخَاطِرِهِ، وَأَنْ نَنْفَعَهُ"

نفحة من الإيمان صور طبق الأصل: يوسف السباعي (ت، 1398هـ)، لجنة النشر للجامعيين، القاهرة، (1407ه/ 1987م)، ‏173‏.
ن1398هـ=1977م

الحِتَّةُ : القِطْعَةُ النَّوْعِيَّةُ مِنَ الشَّيْءِ.

"وَأَشَارَ عَلَيْهِ .. بِاسْتِعْمَالِ الدُّودِ، لِمَصِّ الدَّمِ الفَاسِدِ الَّذِي يُسَبِّبُ لَهُ هَذَا الصُّدَاعَ، وَأَنْبَأَهُ أَنَّ لَدَيْهِ حِتَّتَيْنِ هَدِيَّةً، يَفُوقَانِ الثَّعَابِينَ حَجْمًا وَقُوَّةً"
يوسف السباعي
يا أمة ضحكت: يوسف السباعي (ت، 1398هـ)، مكتبة مصر، القاهرة، (د.ت)، ‏172‏.

حَتِّيّ[اسْمٌ مَنْسُوب]

1402هـ=1981م

الحَتِّيُّ مِنَ الأَشْكَالِ: المَنْسُوبُ إِلَى حَتِّ الصَّخْرِ وَنَحْوِهِ.

"تَشْتَمِلُ الصُّخُورُ الرَّمْلِيَّةُ عَلَى صُوَرٍ كَثِيرَةٍ مِنَ الأَشْكَالِ الحَتِّيَّةِ "
عبد الله يوسف الغنيم
