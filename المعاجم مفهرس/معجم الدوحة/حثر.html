حَوْثَرَة[اسْم]

ن100هـ=718م

الحَوْثَرَةُ : رَأْسُ الذَّكَرِ.

حَوْثَرَةٌ، مِنْ أَعْظَمِ الحَوَاثِرِ نِيطَتْ بِحَقْوَيْ صَمَيَانٍ عَاهِرِ أَهْدَيْتُهَا إِلَى ابْنَةِ العُذَافِرِ
عمرة بنت الحُمارِس
مجمع الأمثال: أبو الفضل الميدانيّ (ت، 518هـ)، حققه وفصله وضبط غرائبه وعلق حواشيه: محمّد محيي الدين عبد الحميد، مطبعة السنة المحمدية، القاهرة، (1374هـ/ 1955م)، ‏2/‏348‏.

أَحْثَرَ[لَازِم]

206هـ=821م

أَحْثَرَ الشَّيْءُ: تَحَبَّبَ وَصَارَ حَبُّهُ كَالبَثَرَاتِ الصِّغَارِ.

"يُقَالُ: قَدْ نَفَضَتْ عُقَالَاهُ، ثُمَّ يُقَالُ: قَدْ أَحْثَرَ "
من رواية أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏182‏.

أَحْثَرَ[مُتَعَدٍّ]

ن320هـ=932م

أَحْثَرَ البَصَرَ: أَحْدَثَ فِيهِ خُشُونَةً.

"السِّمْسِمُ فِيهِ مِنَ الجَوْهَرِ اللَّزِجِ الدُّهْنِيِّ مَا لَيْسَ بِاليَسِيرِ ... وَلِذَلِكَ صَارَ مُضِرًّا ... وَيُحْدِثُ عَطَشًا، وَيُحْثِرُ البَصَرَ، وَيُوَلِّدُ غِذَاءً دَسِمًا"
إسحاق بن سليمان الإسرائيليّ
كتاب الأغذية والأدوية: إسحاق بن سليمان الإسرائيليّ (ت، 320هـ)، تحقيق: محمّد الصبّاح، مؤسسة عز الدين، بيروت، ط1، (1412هـ/ 1992م)، ‏249‏.

حَثِرَ[لَازِم]

206هـ=821م

حَثِرَتِ البِئْرُ: كَدِرَ مَاؤُهَا وَفَسَدَ.

"قَدْ حَثِرَتْ قَلِيبُكُمْ هَذَا وَرَدُؤَ مَاؤُهَا"
من رواية أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏145‏.
ن385هـ=995م

حَثِرَ اللَّبَنُ وَنَحْوُهُ: تَغَيَّرَ طَعْمُهُ وَكَدُرَ لَوْنُهُ.

"مَاءٌ عَاتِكٌ وَلَبَنٌ عَاتِكٌ: قَدُمَا حَتَّى أَجِنَ وَحَثِرَ "
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏1/‏214‏.
1378هـ=1958م

حَثِرَ الذَّوْقُ: ضَعُفَ وَاعْتَلَّ.

"إِنَّ الدِّينَ الإِسْلَامِيَّ يَنْفَرِدُ عَنْ سَائِرِ الأَدْيَانِ بِاعْتِمَادِ دَعْوَتِهِ عَلَى الأَدَبِ، وَقِيَامِ مُعْجِزَتِهِ عَلَى البَلَاغَةِ، فَإِذَا حَثِرَ ذَوْقُ العَرَبِيَّةِ فِي رِجَالِهِ ... تَقَطَّعَتِ الأَسْبَابُ بَيْنَهُمْ وَبَيْنَ مُحَمَّدٍ، فَضَلُّوا سَبِيلَه، وَجَهِلُوا عِلْمَه"
أحمد حسن الزيات
وحي الرسالة: أحمد حسن الزيات (ت، 1389هـ)، مطبعة الرسالة، القاهرة، مج1: ط7، (1381ه/ 1962م)؛ مج2: ط4، (1377ه/ 1958م)؛ مج3: ط5، (1384ه/ 1964م)؛ مج4: ط2، (1385ه/ 1966م)، ‏2/‏184‏.

حَثَر[اسْم]

ن206هـ=821م

حَثَرُ الآرَاكِ وَنَحْوِهِ: حَبُّهُ الأَخْضَرُ حِينَ يَخْرُجُ مِنْ بُرْعُمِهِ.

"أَوَّلُ مَا يُثْمِرُ الأَرَاكُ فَهُوَ الحَثَرُ "
أبو عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏8‏.

حَثَر[مَصْدَر]

216هـ=831م

حَثَرُ العَيْنِ: خُشُونَتُهَا مِنَ الرَّمَصِ.

"يُقَالُ: أَجِدُ فِي عَيْنِي حَثَرًا "
من رواية الأصمعيّ
كتاب خلق الإنسان: الأصمعيّ (ت، 216هـ)- ضمن: الكنز اللغوي في اللَّسَن العربي: سعى في نشره وتعليق حواشيه: أوغست هفنر، المطبعة الكاثوليكية، بيروت، (1903م)، ‏185‏.
1364هـ=1944م

الحَثَرُ فِي الذَّوْقِ وَنَحْوِهِ: ضَعْفُهُ وَاعْتِلَالُهُ.

"إِنَّمَا هِيَ نَزَوَاتٌ فِي بَعْضِ الرُّؤُوس، أَوْ نَزَعَاتٌ فِي بَعْضِ النُّفُوس، تَصْدُرُ عَنْ شُذُوذٍ فِي الفِكْرِ، أَوْ حَثَرٍ فِي الذَّوْقِ، أَوْ عَجْزٍ عَنِ الكَمَالِ"
أحمد حسن الزيات
مجلة الرسالة: الرسالة، القاهرة، ع566، (08/ 05/ 1944م)، ‏1‏.

حَثِر[صِفَةٌ مُشَبَّهَة]

ن231هـ=845م

الحَثِرُ مِنَ الأَشْيَاءِ: المُتَحَبِّبُ كَالبَثَرِ الصَّغِيرِ.

"الدَّوَاءُ إِذَا بُلَّ وَعُجِنَ، فَلَمْ يَجْتَمِعْ وَتَنَاثَرَ، فَهُوَ حَثِرٌ "
أبو عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏479‏.
393هـ=1003م

الحَثِرُ مِنَ الأَشْخَاصِ: الَّذِي بَثِرَ جِلْدُهُ، وَخَرَجَ لَهُ خَرَارِيجُ صِغَارٌ.

رَأَتْهُ شَيْخًا حَثِرَ المَلَامِجِ
من شواهد الجوهريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏1/‏339‏.

حَثُرَ[لَازِم]

235هـ=849م

حَثُرَ الدَّمُ وَنَحْوُهُ: ثَخُنَ وَغَلُظَ.

"فَإِنْ تَقَيَّحَ الوَرَمُ سَالَ مِنْهُ شَيْءٌ يُشْبِهُ القِيحَ وَالدَّمَ، وَخَرَجَ فِي المَشْيِ، ثُمَّ يَحْثُرُ قَلِيلًا قَلِيلًا"
ابن سهل الطَّبريّ
فردوس الحكمة في الطّب: ابن سهل الطَّبريّ (ت، 247هـ)، اعتنى بنسخه وتصحيحه: محمّد زبير الصديقي، مطبع آفتاب، برلين، (1928م)، ‏220‏.

حَثَرَة[اسْم]

393هـ=1003م

الحَثَرَةُ : حَبَّةُ العُنْقُودِ فِي أَوَّلِ خُرُوجِهَا.

"أَحْثَرَ النَّخْلُ، إِذَا تَشَقَّقَ طَلْعُهُ وَكَانَ حَبُّهُ كَالحَثَرَاتِ الصِّغَارِ قَبْلَ أَنْ يَصِيرَ خَصَلًا"
من رواية الجوهريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏2/‏623‏.

تَحْثِير[مَصْدَر]

1401هـ=1980م

التَّحْثِيرُ : تَكْبِيرُ الجُسَيْمَاتِ بِوَاسِطَةِ تِقْنِيَةِ التَّكَتُّلِ، وَهُوَ أَحَدُ أَهَمِّ عَمَلِيَّاتِ الوِحْدَةِ فِي إِنْتَاجِ أَشْكَالِ الجُرْعَاتِ الصَّيْدَلَانِيَّةِ، وَمُعْظَمُهَا أَقْرَاصٌ وَكَبْسُولَاتٌ. [الطّبّ والصّيدلة]

"يُضَافُ إِلَى المَسْحُوقِ بَعْدَ ذَلِكَ كَمِّيَّةٌ كَافِيَةٌ مِنْ سَائِلٍ مُنَاسِبٍ لِلْحُصُولِ عَلَى عَجِينَةٍ ذَاتِ قِوَامٍ قَابِلٍ لِلتَّحْثِيرِ "
بديع كعيد
كتاب الصّيدلانيّات، القسم النظريّ: بديع كعيد، المطبعة الجديدة، دمشق، ط1، (1981م)، ‏43‏.
1432هـ=2010م

تَحْثِيرُ السُّكَّرِ وَنَحْوِهِ: تَجَمُّعُ حُبَيْبَاتِهِ مَعَ بَعْضِهَا.

"يُوجَدُ السُّكَّرُوزُ بِأَسْمَاءَ تِجَارِيَّةٍ وَمَأْلُوفَةٍ كَثِيرَةٍ، قَدْ تَتَعَلَّقُ هَذِهِ الأَسْمَاءُ بِدَرَجَةِ نَقَاوَتِهِ ... وَبِمَدَى التَّحْثِيرِ أَوْ حَجْمِ البَلُّورَاتِ"
حسن كلاوي وآخرون
كيمياء الغذاء: حسن كلاوي وآخرون ، إشراف: زيد العساف، مراجعة: كرم عودة، تقديم: حسين عبد الرزاق الجزائري، المركز العربي للتعريب، دمشق، (2010م)، ‏1094‏.

حُثَيْرَات[اسْم]

1401هـ=1980م

الحُثَيْرَاتُ : مُحَضَّرَاتٌ صُلْبَةٌ بِشَكْلِ قِطَعٍ صَغِيرَةٍ مُخْتَلِفَةِ الحَجْمِ تَتَأَلَّفُ بِصُورَةٍ رَئِيسِيَّةٍ مِنَ السُّكَّرِ، وَتَحْتَوِي عَلَى عَجِينَةٍ ذَاتِ قِوَامٍ قَابِلٍ لِلتَّحْثِيرِ. [الطّبّ والصّيدلة]

"لِتَحْضِيرِ الحُثَيْرَاتِ بِصُورَةٍ عَامَّةٍ، يَجِبُ أَوَّلًا سَحْقُ المَوَادِّ الدَّاخِلَةِ فِي تَرْكِيبِهَا، ثُمَّ مَزْجُهَا جَيَّدًا حَتَّى الحُصُولِ عَلَى مَسْحُوقٍ مُتَجَانِسٍ"
بديع كعيد
كتاب الصّيدلانيّات، القسم النظريّ: بديع كعيد، المطبعة الجديدة، دمشق، ط1، (1981م)، ‏43‏.

تَحَثَّرَ[لَازِم]

1432هـ=2010م

تَحَثَّرَ العَسَلُ وَنَحْوُهُ: تَجَمَّدَ وَغَلُظَ قَوَامُهُ.

"وَعِنْدَمَا تُصْنَعُ هَذِهِ الأَنْوَاعُ مِنَ العَسَلِ، تَكُونُ سَوَائِلَ سَمِيكَةً وَشَفَّافَةً، تَتَحَبَّبُ/ تَتَحَثَّرُ تَدْرِيجِيًّا بِسَبَبِ نُمُوِّ بَلُّورَاتِ السُّكَّرِ"
حسن كلاوي وآخرون
كيمياء الغذاء: حسن كلاوي وآخرون ، إشراف: زيد العساف، مراجعة: كرم عودة، تقديم: حسين عبد الرزاق الجزائري، المركز العربي للتعريب، دمشق، (2010م)، ‏1108‏.
