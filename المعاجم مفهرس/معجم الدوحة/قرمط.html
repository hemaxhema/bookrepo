اقْرَمَّطَ[لَازِم]

ن9هـ=630م

اقْرَمَّطَ الجِلْدُ وَنَحْوُهُ: تَقَبَّضَ وَتَقَارَبَ بَعْضُهُ إِلَى بَعْضٍ.

تَكَسَّبْتُهَا فِي كُلِّ أَطْرَافِ شِدَّةٍ إِذَا اقْرَمَّطَتْ ، يَوْمًا، مِنَ الفَزَعِ، الخُصَى
زيد الخير
لسان العرب: ابن منظور (ت، 711هـ)، دار صادر، بيروت، ط1، (د.ت)، ‏7/‏377‏.

قَرْمَطَ[مُتَعَدٍّ بِالحَرْف]

ن40هـ=660م

قَرْمَطَ فِي كِتَابَتِهِ: قَارَبَ بَيْنَ حُرُوفِهَا.

"أَلِقْ دَوَاتَكَ، وَأَطِلْ شَبَاةَ قَلَمِكَ، وَفَرِّجْ بَيْنَ السُّطُورِ، وَقَرْمِطْ بَيْنَ الحُرُوفِ"
عليّ بن أبي طالب
كتاب الوزراء والكُتّاب: محمّد بن عبدوس الجَهْشَيَاري (ت، 331هـ)، تحقيق: مصطفى السقا وآخرين، مطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1357هـ/ 1938م)، ‏23‏.
ن43هـ=663م

قَرْمَطَ فِي مَشْيِهِ: قَارَبَ بَيْنَ خُطَاهُ.

"مَنْ يَعِشْ يَكْبَرْ، وَمَنْ يَكْبَرْ يُقَرْمِطْ "
عمرو بن العاص
تعليق من أمالي ابن دريد (ت، 321هـ): رواية: أبي مسلم محمّد بن أحمد بن علي الكتاب وآخرين، تحقيق: السيد مصطفى السنوسي، المجلس الوطني للثقافة والفنون والأدب، الكويت، ط1، (1404هـ/ 1984م)، ‏207‏.

قَرْمَطَ[مُتَعَدٍّ]

411هـ=1020م

قَرْمَطَ الشَّيْءَ: جَعَلَهُ يَتَقَبَّضُ.

"إِذَا دَخَلْتُمْ هَجَرَ، فَعَبِّسُوا وُجُوهَكُمْ، وَقَرْمِطُوا آنَافَكُمْ عَلَى أَهْلِهَا"
صرصر
السجل المعلق: ‏حمزة بن عليّ،‏مجلة المقتبس: المقتبس، سوريا،ع52، ج5، مج5، (يونيو- 1910م)، ‏305‏.
464هـ=1071م

قَرْمَطَ الخَطَّ: دَقَّقَهُ.

خَطَّ الجَمَالُ عَلَى لَأْلَاءِ عَارِضِهِ دَقِيقَ خَطٍّ بِنَقْطِ الخَالِ مَوْسُومِ
كَمَا يُقَرْمَطُ عُنْوَانٌ بِغَالِيَةٍ عَلَى كِتَابٍ بِطِينِ المِسْكِ مَخْتُومِ
أبو المعالي، محمّد بن عليّ العُقَيليّ السَّلّار
دُمية القصر وعُصرة أهل العصر: عليّ بن الحسن الباخَرْزيّ (ت، 467هـ)، تحقيق ودراسة: محمّد ألتونجي، دار الجيل، بيروت، ط1، (1414هـ/ 1993م)، ‏2/‏1155‏.
ن643هـ=1245م

قَرْمَطَ المَوَاعِيدَ: جَمَعَ بَعْضَهَا إِلَى بَعْضٍ، وَلَمْ يَفِ بِهَا.

"يُقَالُ لِمَنْ يُقَرْمِطُ المَوَاعِيدَ: عُرْقُوبٌ"
ابن يعيش
ما يعوَّل عليه في المضاف والمضاف إليه: محمد الأمين بن فضل الله المحبي (ت، 1111هـ)، تحقيق: محمد حسن عبد العزيز وآخرين، مراجعة: حسن الشافعي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1424ه/ 2003م- 1433ه/ 2012م)، ‏3/‏254‏.

قَرْمَطَة[مَصْدَر]

ن175هـ=791م

القَرْمَطَةُ فِي الخَطِّ: دِقَّةُ الكِتَابَةِ وَتَقَارُبُ الحُرُوفِ وَالسُّطُورِ.

" القَرْمَطَةُ : دِقَّةُ الكِتَابَةِ، وَتَدَانِي الحُرُوفِ وَالسُّطُورِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏258‏.
ن175هـ=791م

القَرْمَطَةُ فِي المَشْيِ: مُقَارَبَةُ الخَطْوِ.

"القَرْمَطَةُ: دِقَّةُ الكِتَابَةِ، وَتَدَانِي الحُرُوفِ وَالسُّطُورِ، وَالقَرْمَطَةُ فِي مَشْيِ القَطُوفِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏258 ‏-‏ 259‏.
ن255هـ=869م

القَرْمَطَةُ فِي المَوَاعِيدِ: جَمْعُ بَعْضِهَا إِلَى بَعْضٍ، دُونَ الوَفَاءِ بِهَا.

"يَعْنِي بِذَلِكَ القَرْمَطَةَ فِي المَوَاعِيدِ"
الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏2/‏174‏.
ن346هـ=957م

القَرْمَطَةُ : اتِّبَاعُ مَذْهَبِ القَرَامِطَةِ.

"وَشُهِدَ لَهُمْ بِالنَّزَاهَةِ مِنَ القَرْمَطَةِ ، فَخُلِّيَ عَنْهُمْ"
أبو إسحاق الإصطخريّ
المسالك والممالك: أبو إسحاق الإصطخريّ (ت، 346هـ)، تحقيق: محمّد جابر عبد العال الحسيني، مراجعة: محمّد شفيق غربال، قدم هذه الطبعة: عبد العال عبد المنعم الشامي، الهيئة العامة لقصور الثقافة، القاهرة، د.ت. ‏90‏.

قُرْمُوط[اسْم]

ن175هـ=791م

القُرْمُوطُ : ثَمَرُ شَجَرِ الغَضَا، أَحْمَرُ، يُشَبَّهُ بِالرُّمَّانِ.

وَيُنْشِزُ جَيْبَ الدِّرْعِ عَنْهَا، إِذَا مَشَتْ خَمِيلٌ كَقُرْمُوطِ الغَضَا الخَضِلِ النَّدِي
من شواهد الخليل
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏259‏.

قُرْمُوط[اسْمُ جِنْس]

773هـ=1371م

القُرْمُوطُ : سَمَكٌ يَعِيشُ فِي الأَنْهَارِ، أَسْمَرُ الجِلْدِ، أَحْمَرُ اللَّوْنِ، يُدْعَى (دُحْرُوجَةَ الجُعَلِ).

"فَإِنْ رَأَى عِنْدَهُ شَيْئًا مِمَّا يَأْنَسُ بِالإِنْسَانِ، أَوْ يُرَبَّى فِي البُيُوتِ، كَاللَّجَأَةِ، وَالقُرْمُوطِ ، وَمَا أَشْبَهَهُمَا، كَانَ دَلِيلًا عَلَى الإِحْسَانِ لِلْأَيْتَامِ وَالغُرَبَاءِ"
كمال الدين محمد بن موسى الدَّميري
حياة الحيوان الكبرى: كمال الدين محمد بن موسى الدَّميري (ت، 808هـ)، عني بتحقيقه: إبراهيم صالح، دار البشائر، دمشق، ط1، (1426ه/ 2005م)، ‏2/‏568‏.

مُقَرْمَط[اسْمُ مَفْعُول]

ن255هـ=869م

المُقَرْمَطُ مِنَ الأُمُورِ: المَوْصُولُ بَعْضُهُ بِبَعْضٍ.

"يَقُولُ: مَطْلًا مُقَرْمَطًا دَائِمًا"
الجاحظ
كتاب الحيوان: الجاحظ (ت، 255هـ)، تحقيق وشرح: عبد السلام هارون، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، القاهرة، ط2، (1385هـ/ 1965م- 1389هـ/ 1969م)، ‏1/‏317‏.
ن385هـ=995م

المُقَرْمَطُ مِنَ المَشْيِ: الَّذِي يُقَارَبُ فِيهِ الخُطَا.

"وَالوَكَّاتُ: الَّذِي يَمْشِي مَشْيًا مُقَرْمَطًا "
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏6/‏308‏.
387هـ=997م

المُقَرْمَطُ مِنَ الوُعُودِ: المُجَمَّعُ دُونَ الوَفَاءِ بِهِ.

"وَعْدُهُ مُقَرْمَط ، وَتَسْوِيفُهُ مُفَرَّط"
أبو منصور الثّعالبيّ
لُباب الآداب: أبو منصور الثّعالبيّ (ت، 429هـ)، حرره وحققه: أحمد حسن بسج، دار الكتب العلمية، بيروت، ط1، (1417هـ/ 1997م)، ‏73‏.
ن391هـ=1001م

المُقَرْمَطُ مِنَ الكِتَابَةِ: الَّذِي دُقِّقَ.

كَتَبَتْ رُقْعَةً إِلَيَّ وَقَدْ عَبَّـ ـتْ بِسَطْرٍ مُقَرْمَطٍ خَلْفَ سَطْرِ
ابن الحجّاج النِّيلي البغداديّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏3/‏100‏.

قَرْمَطِيّ[اسْمٌ مَنْسُوب]

ن292هـ=905م

القَرْمَطِيُّ : المَنْسُوبُ إِلَى القَرَامِطَةِ.

"قَدْ تَقَدَّمَتْ كُتُبِي إِلَى الوَزِيرِ- أَعَزَّهُ اللهُ- فِي خَبَرِ القَرْمَطِيِّ اللَّعِينِ وَأَشْيَاعِهِ، بِمَا أَرْجُو أَنْ يَكُونَ قَدْ وَصَلَ إِنْ شَاءَ اللهُ"
محمّد بن سليمان الكاتب
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏10/‏109‏.

مُقَرْمِط[اسْمُ فَاعِل]

ن293هـ=906م

المُقَرْمِطُ فِي الخَطِّ: الَّذِي يُدَقِّقُهُ.

"فَكُنْتُ آخُذُ الكَاغَذَ الفِرْعَوْنِيَّ، وَأَكْتُبُ مُقَرْمِطًا "
صالح جَزَرَة
تاريخ مدينة السلام وأخبار مُحدّثيها وذكر قُطّانها العلماء من غير أهلها ووارديها: الخطيب البغداديّ (ت، 463هـ)، حققه وضبط نصه وعلق عليه: بشار عواد معروف، دار الغرب الإسلاميّ، بيروت، ط1، (1422هـ/ 2001م)، ‏10/‏443‏.
ن728هـ=1327م

المُقَرْمِطُ : المُتَّبِعُ مَذْهَبَ القَرَامِطَةِ.

"فَإِذَا أَخَذَ الخَصْمُ يُنَازِعُ فِي دَلَالَةِ النَّصِّ، أَوِ العَقْلِ، جَعَلَهُ مُسَفْسِطًا أَوْ مُقَرْمِطًا "
ابن تيميّة
مجموع الفتاوى: ابن تيميّة (ت، 728هـ)، جمع وترتيب: عبد الرحمن بن قاسم، محمد بن عبد الرحمن بن قاسم، مجمع الملك فهد لطباعة المصحف الشريف، المدينة المنورة، ط1، (1425ه/ 2004م)، ‏3/‏301‏.

قَرَامِطَة[اسْم]

ن300هـ=912م

القَرَامِطَةُ : جَمَاعَةٌ مِنَ الشِّيعَةِ البَاطِنِيَّةِ، قِيلَ: إِنَّهَا مَنْسُوبَةٌ إِلَى رَجُلٍ مُلَقَّبٍ بِقَرْمَطٍ. [الأديان والمذاهب والفرق]

"وَكَانَ يَبْرِينُ لِبَنِي سَعْدٍ مِنْ تَمِيمٍ، فَغَلَبَتْهُمُ القَرَامِطَةُ عَلَيْهِ"
أبو عليّ الهَجَريّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏1639‏.

345هـ=956م

" وَتَقَرْمَطَ أَكْثَرُ مَنْ حَوْلَ دِمَشْقَ مِنَ الغُوطَةِ وَغَيْرِهَا، وَعَاضَدُوهُ"
عليّ بن الحسين المسعوديّ
التّنبيه والإشراف: عليّ بن الحسين المسعوديّ (ت، 346هـ)، عني بتصحيحه ومراجعته: عبد الله إسماعيل الصاوي، مطبعة النهضة، القاهرة، (1357هـ/ 1938م)، ‏322‏.

مُتَقَرْمِط[اسْمُ فَاعِل]

709هـ=1309م

المُتَقَرْمِطُ : المُنْتَمِي إِلَى مَذْهَبِ القَرَامِطَةِ.

"وَهَذَا يَرْفَعُ قَوْلَ بَعْضِ الجُهَّالِ المُتَقَرْمِطَةِ مِنْ هَؤُلَاءِ: إِنَّهُ لَنْ يَرَى رَبَّهُ حَتَّى يَمُوتَ، أَيْ: تَمُوتَ نَفْسُهُ وَهَوَاهُ"

تَقَرْمُط[مَصْدَر]

ن1050هـ=1640م

التَّقَرْمُطُ : التَّقَبُّضُ.

وَالَّذِي يَمْنَعُهَا عَنِ التَّقَرْمُطِ وَالِاسْتِعْلَاء، وَيَرْدَعُهَا عَنِ الغَلَبَةِ وَالِاسْتِيلَاء، هُوَ النُّورُ البَارِقُ القُدُسِيّ، وَالبُرْهَانُ النَّيِّرُ العَرْشِيّ"

القُرْمُوطِيُّ : المَنْسُوبُ إِلَى القُرْمُوطِ.

"فَفِيهِ الشَّكْلُ القُمْقُمِيُّ، وَالرُّمَّانِيُّ، وَالقُرْمُوطِيُّ ، وَالإِطَارِيُّ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏197‏.
