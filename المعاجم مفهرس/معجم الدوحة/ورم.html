مُوَرَّم[صِفَة]

قبل 50ق.هـ=573م

المُوَرَّمُ مِنَ الأَشْخَاصِ: الضَّخْمُ السَّمِينُ.

لَهُ شَرْبَتَانِ بِالنَّهَارِ، وَأَرْبَعٌ مِنَ اللَّيْلِ، حَتَّى آضَ سُخْدًا، مُوَرَّمَا
طَرَفة بن العبد البكريّ
ديوان طرفة بن العبد شرح الأعلم الشَّنْتَمَري وتليه طائفة من الشعر المنسوب إلى طرفة: تحقيق: درية الخطيب، لطفي الصقّال، إدارة الثقافة والفنون، البحرين، المؤسسة العربية، بيروت، ط2، (2000م)، ‏106‏.

مُوَرَّم[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المُوَرَّمُ : المُنَفَّخُ المُنَفَّشُ.

يَنَامُ الضُّحَى، حَتَّى إِذَا يَوْمُهُ اسْتَوَى تَنَبَّهَ مَثْلُوجَ الفُؤَادِ، مُوَرَّمَا
حاتم بن عبد الله الطّائيّ
ديوان شعر حاتم بن عبد الله الطائي وأخباره: صنعة: يحيى بن مدرك الطائي، رواية: هشام بن محمّد الكلبي، دراسة وتحقيق: عادل سليمان جمال، مطبعة المدني، القاهرة، د.ت. ‏240‏.

وَارِم[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

الوَارِمُ مِنَ الأَعْضَاءِ وَنَحْوِهَا: المُنْتَفِخُ مِنْ سِمَنٍ أَوْ غَيْرِهِ.

جَزِيزُ القَفَا، شَبْعَانُ، يَرْبِضُ حَجْرَةً حَدِيثُ الخِصَاءِ، وَارِمُ العَفْلِ، مُعْبَرُ
بشر بن أبي خازم الأسديّ
ديوان بشر بن أبي خازم الأسدي: عني بتحقيقه: عزة حسن، وزارة الثقافة والإرشاد القومي، دمشق، (1379هـ/ 1960م)، ‏88‏.
ن206هـ=821م

الوَارِمُ الأَنْفِ: المُغْتَاظُ، المُمْتَلِئُ غَضَبًا.

"جَاءَنَا نَافِخًا قِبِرَّاهُ، وَوَارِمًا خَوْرَمَتُهُ"
أبو عمرو الشّيبانيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏9/‏138‏.

وَارِم[صِفَةٌ مُشَبَّهَة]

37هـ=657م

الوَارِمُ مِنَ النَّبَاتِ: الطَّوِيلُ المُمْتَلِئُ رِيًّا.

فَتَمَطَّى زَمْخَرِيٌّ، وَارِمٌ مِنْ رَبِيعٍ، كُلَّمَا خَفَّ هَطَلْ
النّابغة الجَعْديّ
شعر النابغة الجعدي: تحقيق: عبد العزيز رباح، منشورات المكتب الإسلامي، دمشق، ط1، (1384هـ/ 1964م)، ‏95‏.

وَرِمَ[لَازِم]

13هـ=634م

وَرِمَ أَنْفُهُ : غَضِبَ، وَامْتَلَأَ غَيْظًا.

"إِنِّي وَلَّيْتُ أُمُورَكُمْ خَيْرَكُمْ فِي نَفْسِي؛ فَكُلُّكُمْ وَرِمَ أَنْفُهُ أَنْ يَكُونَ لَهُ الأَمْرُ مِنْ دُونِهِ"
أبو بكرٍ الصِّدّيق
جمهرة خطب العرب في عصور العربية الزاهرة: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1352هـ/ 1933م- 1352هـ/ 1934م)، ‏1/‏206‏.
ن32هـ=653م

وَرِمَ العُضْوُ وَنَحْوُهُ: انْتَفَخَ مِنْ مَرَضٍ أَوْ غَيْرِهِ.

"إِنَّ امْرَأَتِي وَرِمَ ثَدْيُهَا، فَمَصِصْتُهُ، فَدَخَلَ حَلْقِي شَيْءٌ سَبَقَنِي"
رجل من المسلمين
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏16/‏32،،‏(رقم الحديث: 15752).
ن629هـ=1231م

وَرِمَ أَنْفُهُ : شَمَخَ وَتَكَبَّرَ.

ثُمَّ انْتَحَيْنَا لِعَوْفٍ بَعْدَمَا وَرِمَتْ أُنُوفُهَا ، فَفَشَشْنَا ذَلِكَ الوَرَمَا
ابن المُقَرَّب العُيونيّ
شرح ديوان ابن المقرّب: ابن المُقَرَّب العُيونيّ (ت، 629هـ)، أعده وحققه وعلق عليه: عبد الخالق بن عبد الجليل الجنبي وآخرون، منشورات ملتقى الواحتين، الرويس، دار المحجة البيضاء، بيروت، ط2، (1433ه/ 2012م)، ‏4/‏2176‏.

أَوْرَم[اسْم]

ن20هـ=641م

الأَوْرَمُ : الجَمَاعَةُ الكَثِيرَةُ مِنَ الجُنْدِ وَنَحْوِهِمْ.

وَحَيٍّ حُلُولٍ أُولِي بَهْجَةٍ شَهِدْتُ، وَشِعْبُهُمُ مُفْرَمُ
بِأَلْبِ أَلُوبٍ، وَحَرَّابَةٍ لَدَى مَتْنِ وَازِعِهَا الأَوْرَمُ
البُريق الخُناعيّ الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏753‏.
ن209هـ=824م

الأَوْرَمُ : النَّاسُ.

"مَا أَدْرِي أَيُّ الطَّبْلِ هُوَ... وَأَيُّ الأَوْرَمِ ؟"
أبو عُبيدة
شرح نقائض جرير والفرزدق: رواية: أبي عبد الله اليزيدي عن أبي سعيد السكري عن ابن حبيب عن أبي عبيدة، تحقيق وتقديم: محمّد حُوّر، وليد محمود خالص، منشورات المجمع الثقافي، أبو ظبي، ط2، (1998م)، ‏1/‏304‏.

تَوَرَّمَ[لَازِم]

ن50هـ=670م

تَوَرَّمَ العُضْوُ وَنَحْوُهُ: انْتَفَخَ مِنْ مَرَضٍ أَوْ غَيْرِهِ.

"قَامَ رَسُولُ اللهِ،  @ ، حَتَّى تَوَرَّمَتْ قَدَمَاهُ"
المغيرة بن شعبة الثَّقَفيّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏30/‏138،،‏(رقم الحديث: 18198).
ن243هـ=857م

تَوَرَّمَ الشَّيْءُ: تَوَسَّعَ.

"إِنَّ الوَرَمَ الدَّمَوِيَّ حَارٌّ ثَقِيلٌ رَطْبٌ، كَثِيرُ الدَّمْعَةِ وَالرَّمَصِ وَالغَمَصِ، وَرُبَّمَا تَوَرَّمَ الإِكْلِيلُ، حَتَّى يُنْظَرَ إِلَى سَوَادِ العَيْنِ فِي وَسَطِهِ كَهَيْئَةِ فَصِّ الخَاتَمِ قَدْ تَوَرَّمَ البَيَاضُ، وَعَلَا عَلَيْهِ"
ابن ماسويه
معرفة مِحْنة الكَحّالين: ابن ماسويه (ت، 243هـ)، تحقيق ودراسة: نشأت الحمارنة، اكتمال رجب، منشورات مجمع اللغة العربيّة الأردنيّ، ط1، (1440هـ/ 2018م)، ‏294‏.
1441هـ=2019م

تَوَرَّمَ الشَّيْءُ: تَضَخَّمَ.

"لَا يُمْكِنُ اعْتِبَارُهَا آلِيَّةً مَشْرُوعَةً مِنْ آلِيَّاتِ التَّوْلِيدِ المُعْجَمِيِّ، لَكِنْ لَا أَحَدَ يُنْكِرُ أَنَّهَا، مِنَ النَّاحِيَةِ التَّارِيخِيَّةِ وَالعَمَلِيَّةِ، قَدْ قَامَتْ بِإِنْتَاجِ عَدَدٍ كَبِيرٍ مِنَ الأَلْفَاظِ الَّتِي تَوَرَّمَتْ بِهَا القَوَامِيسُ العَرَبِيَّةُ"
عبدالعلي الودغيري
القاموسية العربية الحديثة بين تنمية الفصحى وتحديث القاموس والتأريخ للمعجم: عبدالعلي الودغيري ، المركز العربي للأبحاث ودراسة السياسات، الدوحة، ط1، (2019م)، ‏180‏.
1442هـ=2020م

تَوَرَّمَ الرَّأْسُ وَنَحْوُهُ: ثَقُلَ.

"أَجْوَاءٌ مِنْ جَزَعٍ وَإِحْبَاطٍ كَانَتْ تُسَيْطِرُ عَلَيْنَا فِي تَرَقُّبِ المَجْهُولِ... هَوَاجِسُ كَانَتْ تُهَشِّمُ كُلَّ صُوَرِ العَالَمِ الخَارِجِيِّ الَّتِي بَدَأَتْ بِدَوْرِهَا تَتَلَاشَى وَتَغْرَقُ فِي يَمِّ النِّسْيَانِ، تَوَرَّمَ رَأْسِي حِينَهَا؛ لِفَرْطِ مَا كَانَ يَحْمِلُ مِنَ الوَجَعِ فِي مُلَاحَقَةِ القَادِمِ"
ناهض الهندي
من شقوق الظلام قصة سجين أكثر ما يؤرق فيها أنها لم تنته بعد: ناهض الهندي ، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2022م)، ‏80‏.
1443هـ=2021م

تَوَرَّمَتِ الأَشْيَاءُ: كَثُرَتْ.

"تَمَّ إِنْشَاءُ بِنًى شَبِيهَةٍ، مَجَالِسَ، وَمُؤَسَّسَاتٍ مَدِينِيَّةٍ، فِي المُدُنِ، بِحَيْثُ تَوَرَّمَتْ صُفُوفُ المُمَثِّلِينَ المُنْتَخَبِينَ فِي البِلَادِ... بِإِضَافَةِ ثَلَاثَةِ مَلَايِينَ آخَرِينَ مِنَ المُمَثِّلِينَ المُنْتَخَبِينَ حَدِيثًا، مِنَ الرِّجَالِ وَالنِّسَاءِ"
محمد العزير
حياة الديمقراطية وموتها، لجون كين: محمد العزير ، ترجمة: محمد العزير، المركز العربي للأبحاث ودراسة السياسات، بيروت، ط1، (2021م)، ‏818‏.

اسْتَوْرَمَ[لَازِم]

ن58هـ=678م

اسْتَوْرَمَ العُضْوُ وَنَحْوُهُ: انْتَفَخَ؛ مِنْ مَرَضٍ وَغَيْرِهِ.

"وَضَرَبَ بِيَدِهِ عَلَى رُكْبَتِي، فَاسْتَوْرَمَتْ ، حَتَّى صَارَتْ مِثْلَ رَأْسِ الشَّاةِ"
ابنة عوف بن عفراء
بذل الماعون في فضل الطاعون: ابن حجر العسقلاني (ت، 852هـ)، تحقيق: أحمد عصام عبد القادر الكاتب، دار العاصمة، الرياض، (د.ت)، ‏152‏.

وَرَم[اسْم]

ن60هـ=680م

الوَرَمُ : النُّتُوءُ وَالِانْتِفَاخُ فِي مَوْضِعٍ مِنَ الجَسَدِ؛ مِنْ مَرَضٍ أَوْ غَيْرِهِ.

"أَخْبَرَنَا عَفَّانُ بْنُ مُسْلِمٍ... عَن عُبَيدٍ، عَن عُمَيرٍ  أَنَّ أَسْمَاءَ كَانَ فِي عُنُقِهَا وَرَمٌ ، فَجَعَلَ النَّبِيُّ،  @ ، يَمْسَحَهَا"
عُمَيْر بن قتادة اللّيثيّ
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏10/‏239‏.

وَرَم[مَصْدَر]

ن175هـ=791م

وَرَمُ الشَّيْءِ: انْتِفَاخُهُ، مِنْ مَرَضٍ وَنَحْوِهِ.

"وَالمُنْدَعِصُ: الشَّيْءُ المَيِّتُ إِذَا انْفَسَخَ، وَشُبِّهَ بِالدِّعْصِ؛ لِوَرَمِهِ ، أَوْ ضَعْفِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏291‏.

وَرَّمَ[مُتَعَدٍّ]

ن125هـ=743م

وَرَّمَ العُضْوَ وَنَحْوَهُ: جَعَلَهُ مُنْتَفِخًا.

كُنَّا إِذَا عَامٌ أَلَحَّتْ أَزَمُهْ جَادَتْ بِمَطْحُونٍ لَهَا، لَا نَأْجَمُهْ لَا يَنْفُخُ البَطْنَ، وَلَا يُوَرِّمُهْ تَطْبُخُهُ ضُرُوعُهَا، وَتَأْدِمُهْ
أبو نَعامة مولى بني سَعْد
الموشَّح مآخذ العلماء على الشعراء في عدة أنواع من صناعة الشعر: أبو عُبيد الله المَرْزُبانيّ (ت، 384هـ)، تحقيق: علي محمّد البجاوي، نهضة مصر، القاهرة، د.ت. ‏274‏.

وَرَّمَ[مُتَعَدٍّ بِالحَرْف]

ن393هـ=1003م

وَرَّمَ بِأَنْفِهِ : شَمَخَ وَتَكّبَّرَ.

" وَرَّمَ فُلَانٌ بِأَنْفِهِ "
أبو نصر الجوهريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏5/‏2050‏.

تَوْرِيم[مَصْدَر]

ن399هـ=1008م

تَوْرِيمُ الشَّيْءِ: انْتِفَاخُهُ.

وَالأَخْدَعَانِ فَمَا زَالَا يُرَى بِهِمَا لِكَثْرَةِ المَزْحِ، تَوْرِيمٌ ، وَتَحْمِيرُ
أبو الرَّقَعْمَق الأنطاكيّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏1/‏393‏.
418هـ=1027م

تَوْرِيمُ العُضْوِ وَنَحْوِهِ: جَعْلُهُ يَتَوَرَّمُ وَيَنْتَفِخُ.

"فَنَحْنُ نَسْتَأْثِرُ وُقُوعَ الضَّرَرِ بِالعُضْوِ الخَسِيسِ... حَتَّى إِنَّا لَنَجْتَهِدُ فِي جَذْبِ المَادَّةِ إِلَى العُضْوِ الخَسِيسِ وَتَوْرِيمِهِ ، وَلَوْ بِالمَحَاجَمِ وَالأَضْمِدَةِ الجَاذِبَةِ الحَادَّةِ"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏1/‏304‏.

إِيرَام[مَصْدَر]

418هـ=1027م

إِيرَامُ العُضْوِ وَنَحْوِهِ: تَوْرِيمُهُ.

"وَلَا يَجِبُ أَنْ تَبْلُغَ بِالرِّبَاطِ الإِيلَامَ، ثُمَّ الإِيرَامَ "
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏3/‏229‏.

مَوْرَمَة[مَصْدَرٌ مِيمِيّ]

ن421هـ=1030م

المَوْرَمَةُ : الِانْتِفَاخُ مِنْ مَرَضٍ أَوْ غَيْرِهِ.

"نَوْمَةُ الصُّبْحَةِ مَعْجَزَةٌ، مَنْفَخَةٌ، مَكْسَلَةٌ، مَوْرَمَةٌ "
من رواية أبي الحسن الماورديّ
أدب الدّنيا والدّين: أبو الحسن الماورديّ (ت، 450هـ)، شرح وتعليق: محمّد كريم راجح، دار اقرأ، بيروت، ط4، (1405هـ/ 1985م)، ‏362‏.

وَرْمَة[اسْمُ مَرَّة]

970هـ=1562م

الوَرْمَةُ : الوَاحِدَةُ مِنَ الوَرَمِ.

"وَلِلْوَرْمَةِ فِي البَطْنِ وَرَقُ مُقَابَلَةِ الشَّمْسِ، يُدَقُّ وَيُعْجَنُ بِالخَلِّ، وَتُطْلَي بِهِ الوَرْمَةُ "
راشد بن عميرة بن هاشم العيني الرستاقي العماني
منهاج المتعلمين: راشد بن عميرة بن هاشم العيني الرستاقي العماني (ت، 1019هـ)، دراسة وتحقيق: عبد الله بن علي بن سعيد السعدي، وزارة التراث القومي والثقافة، مسقط، ط1، (1434ه/ 2013م)، ‏210‏.

مَوْرُوم[اسْمُ مَفْعُول]

1439هـ=2017م

المَوْرُومُ مِنَ الأَعْضَاءِ: المُصَابُ بِانْتِفَاخٍ؛ مِنْ سِمَنٍ وَغَيْرِهِ.

"يُحَدِّثُونَ بَعْضُهُمْ بَعْضًا، وَطَيَّ خَطْوِهِمُ العَجُولِ أَكْبَادٌ ضَامِرَةٌ، وَأَطْحَالٌ مَوْرُومَةٌ ، وَدِمَاءٌ مُتَحَجِّرَةٌ عِنْدَ حَافَةِ الِاضْطِجَاعِ لِلنَّوْمِ"
رضا البهات
شمعة البحر: رضا البهات، دار بدائل للنشر والتوزيع، الجيزة ـ مصر، ط 2، 2017م، ‏23‏.
