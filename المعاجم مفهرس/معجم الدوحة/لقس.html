لَقِسَ[لَازِم]

ن11هـ=632م

لَقِسَتِ النَّفْسُ: غَثَتْ وَخَبُثَتْ.

"لَا يَقُولَنَّ أَحَدُكُمْ: خَبُثَتْ نَفْسِي، وَلَكِنْ لِيَقُلْ: لَقِسَتْ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏40/‏289،،‏(رقم الحديث: 24244).
ن206هـ=821م

لَقِسَ الشَّخْصُ: أَكْثَرَ الكَلَامَ.

وَمَا الفَتْكُ بِالأَمْرِ الَّذِي أَنْتَ نَاظِرٌ بِهِ لَقِسَ الأَصْحَابُ مِمَّنْ تُشَاوِرُ
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏204‏.

لَقِسَ[مُتَعَدٍّ بِالحَرْف]

ن175هـ=791م

لَقِسَتِ النَّفْسُ إِلَى الشَّيْءِ: نَازَعَتْ إِلَيْهِ حِرْصًا.

" وَلَقِسَتْ نَفْسُهُ إِلَى الشَّيْءِ: نَازَعَتْهُ حِرْصًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏78‏.

لَقِس[صِفَةٌ مُشَبَّهَة]

ن11هـ=632م

اللَّقِسُ مِنَ الأَشْخَاصِ: الغَاثِي النَّفْسِ الخَبِيثُهَا.

"لَا يَقُولَنَّ أَحَدُكُمْ: نَفْسِي خَبِيثَةٌ، وَلَكِنْ يَقُولُ: نَفْسِي لَقِسَةٌ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏40/‏437،،‏(رقم الحديث: 24375).
23هـ=644م

اللَّقِسُ مِنَ الأَشْخَاصِ: الشَّرِسُ الضَّيِّقُ الخُلُقِ.

"وَعْقَةٌ لَقِسٌ "
عمر بن الخطّاب
كتاب غريب الحديث: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: عبد السلام هارون وآخرين، الهيئة العامة لشؤون المطابع الأميرية، القاهرة، (1404هـ/ 1984م- 1419هـ/ 1999م)، ‏4/‏229‏.
ن175هـ=791م

اللَّقِسُ مِنَ الأَشْخَاصِ: الشَّرِهُ النَّفْسِ، الحَرِيصُ عَلَى كُلِّ شَيْءٍ.

" اللَّقِسُ : الشَّرِهُ النَّفْسِ، الحَرِيصُ عَلَى كُلِّ شَيْءٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏78‏.

لَقْس[مَصْدَر]

ن26هـ=647م

اللَّقْسُ : شَرَهُ النَّفْسِ وَالحِرْصُ عَلَى كُلِّ شَيْءٍ.

وَجَرَّبْتُ الأُمُورَ وَجَرَّبَتْنِي وَأَحْكَمَنِي دَوَاهٍ مِنْ خِلَافِ
وَلَقْسٌ فِي الأُمُورِ وَمُضْلِعَاتٌ وَأَبوَابٌ تُطَارُ بِالِاكْتِنَافِ
كعب بن زهير بن أبي سُلْمى المُزَنيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏212‏.

لَقُوس[صِيغَةُ مُبَالَغَة]

82هـ=701م

اللَّقُوسُ مِنَ الأَشْخَاصِ: الشَّدِيدُ سُوءِ الخُلُقِ.

إِذَا اسْتَخَفَّ الخَلِطُ اللَّقُوسُ مَارَسَهَا مِنِّي وَلِي شَرِيسُ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏301‏.

لَقَس[مَصْدَر]

ن85هـ=704م

اللَّقَسُ : الشَّكَاسَةُ وَالتَّعْسِيرُ.

حُلْوَةٌ إِذَا تُكَلِّمُهَا تَمْنَعُ المَاعُونَ بِاللَّقَسِ
عُبيد الله بن قيس الرُّقَيّات
ديوان عُبيد الله بن قيس الرُّقَيّات: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، د.ت. ‏160‏.
ن675هـ=1276م

اللَّقَسُ : غَثَيَانُ النَّفْسِ وَخُبْثُهَا.

"وَالخُبْثُ وَاللَّقَسُ ، وَإِنْ كَانَ المَعْنَى المُرَادُ يَتَأَدَّى بِكُلٍّ مِنْهُمَا، لَكِنَّ لَفْظَ الخُبْثِ قَبِيحٌ، وَيَجْمَعُ أُمُورًا زَائِدَةً عَلَى المُرَادِ، بِخِلَافِ اللَّقَسِ؛ فَإِنَّهُ يَخْتَصُّ بِامْتِلَاءِ المَعِدَةِ"
عبد الله بن سعد بن أبي جمرة
فتح الباري بشرح صحيح البخاري: ابن حجر العسقلاني (ت، 852هـ)، تحقيق: عبد العزيز بن عبد الله بن باز، محمد فؤاد عبد الباقي، محبّ الدّين الخطيب، مكتبة المعرفة، بيروت، (د.ت)، ‏10/‏564‏.

مُلَاقِس[اسْمُ فَاعِل]

ن126هـ=744م

المُلَاقِسُ مِنَ الأَشْخَاصِ: المُصَابِرُ لِعَدُوِّهِ.

وَإِنْ أَدْعُ فِي حَيَّيْ رَبِيعَةَ تَأْتِنِي عَرَانِينُ يُشْجِينَ الأَلَدَّ المُلَاقِسَا
الكميت بن زيد الأسديّ
ديوان الكُمَيْت بن زيد الأسديّ: جمع وشرح وتحقيق: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (2000م)، ‏234‏.

لُقَّاس[اسْم]

ن646هـ=1248م

اللُّقَّاسُ : نَبْتَةٌ ذَاتُ سَاقٍ وَأَوْرَاقٍ شَبِيهَةٍ بِأَوْرَاقِ السَّبَانِخِ، يَضْرِبُ لَوْنُهَا إِلَى الصُّفْرَةِ، وَلَهَا رُؤُوسٌ صُفْرٌ تُؤْكَلُ عَسَالِيجُهَا.

"وَهِيَ شَجَرَةٌ تَنْبُتُ عَلَى سَاقٍ، وَلَهَا وَرَقٌ قَرِيبٌ مِنْ وَرَقِ الأَسْفَانَاخِ، وَلَوْنُهَا إِلَى الصُّفْرَةِ، وَلَهَا رُؤُوسٌ صُفْرٌ تُؤْكَلُ عَسَالِيجُهَا... وَيَعْرِفُهَا بَعْضُ أَهْلِ البَادِيَةِ بِاللُّقَّاسِ "
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏3/‏479‏.

لَقِيس[اسْم]

674هـ=1275م

اللَّقِيسُ : عِلَّةٌ تُصِيبُ البَقَرَ.

"وَأَيْضًا، لَمْ تُوجَدْ بِهَا عِلَّةُ البَقَرِ الَّتِي تُسَمَّى اللَّقِيسَ ، إِلَى أَنْ وُجِدَ فِي بَعْضِ الأَسَاسِ ثَوْرَانِ مِنْ صُفْرٍ، أَحَدُهُمَا قُدَّامَ الآخَرِ، يَلْتَفِتُ إِلَيْهِ، فَلَمَّا أُخِذَ مِنْ ذَلِكَ المَوْضِعِ وَقَعَتِ اللَّقِيسُ فِي ذَلِكَ العَامِ"
القَزْوينيّ
آثار البلاد وأخبار العباد: القَزْوينيّ (ت، 682هـ)، دار صادر، بيروت، (د.ت)، ‏556‏.

مَلْقُوس[اسْمُ مَفْعُول]

أَنْدَاهُمُ كَفًّا، أَكَفُّهُمُ عَنْ فِعْلِ مَحْظُورٍ وَمَلْقُوسِ

لَقِّيس[صِفَة]

1127هـ=1714م

اللَّقِّيسُ مِنَ الثَّمَرِ وَنَحْوِهِ: المُتَأَخِّرُ عَنْ وَقْتِهِ.

"فِيهِ الآنَ نَوْعٌ يُسَمَّى التَّدْمُرِيَّ كَالرُّمَّانِ ... وَنَوْعٌ لَقِّيسٌ ، بِزْرُهُ الدَّاخِلُ يَتَفَكَّكُ، وَطَعْمُ بِزْرِهِ فِي غَايَةِ الجَوْدَةِ غَالِبًا"
محمد بن عيسى ابن كنان الصالحي الدمشقي
المواكب الإسلاميَّة في الممالك والمحاسن الشّامية: محمد بن عيسى ابن كنان الصالحي الدمشقي (ت، 1153هـ)، تحقيق ودراسة: حكمت إسماعيل، مراجعة: محمَّد المِصْرِيّ، وزارة الثَّقَافَة، دمشق، (1993م)، ‏2/‏298‏.

لَاقِس[صِفَةٌ مُشَبَّهَة]

ن1201هـ=1786م

اللَّاقِسُ : الحَرِيصُ.

كَالأُقْحُوَانِ، تَرَى ثَنَايَاهُ، إِذَا مَا لَاحَ مُبْتَسِمًا بِثَغْرٍ أَلْعَسِ
بَلْ إِنَّها تَحْكِيِ لِسِمْطِ لآلِئٍ تَحْتَ العَقِيقِ مِنَ الشِّفَاهِ اللُّقَّسِ
عمر بن عبد السلام الداغستاني
اللآلئ الثمينة في أعيان شعراء المدينة: عمر بن عبد السلام الداغستاني (ت، 1201هـ)، دراسة وتحقيق وتعليق: عبد الرازق عبد الرازق عيسى، مكتبة زهراء الشرق، القاهرة، ط1، (2009م)، ‏104‏.
1433هـ=2011م

اللَّاقِسُ مِنَ النُّفُوسِ وَنَحْوِهَا: الغَاثِي الخَبِيثُ.

"أَمَّا أَنْتَ، أَيُّهَا الدَّعِيُّ الجَهُولُ، فَقَدْ هَانَتْ عَلَيْكَ نَفْسُك، فَهَانَتْ عَلَيْكَ لُغَتُك، وَلَمْ تَعْلَمْ أَنَّ فِي تَحْقِيرِ لُغَتِكَ تَحْقِيرًا لِذَاتِك، وَفِي عَدَاءِ لُغَتِكَ عَدَاءً لِأُمَّتِكَ وَوُجُودِك، وَفِي كَرَاهِيَتِكَ لِهَذِهِ اللُّغَةِ كَرَاهِيَةً لِنَفْسِك اللَّاقِسَة ، وَجِبِلَّتِكَ الثَّقِيلَة"

1222هـ=1806م

لَاقَسَ الشَّخْصَ: شَاتَمَهُ وَنَادَاهُ بِلَقَبٍ قَبِيحٍ.

"وَقَدْ كَثُرَ تَصَدِّي العَسْكَرِ بِالأَذِيَّةِ عَلَى العَامَّةِ أَرْبَابِ الحِرَفِ، فَيَأْتِي الشَّخْصُ مِنْهُمْ، وَيَجْلِسُ عَلَى بَعْضِ الحَوَانِيتِ، ثُمَّ يَقُومُ، فَيَدَّعِي ضَيَاعَ كِيسِهِ، أَوْ سُقُوطَ شَيْءٍ مِنْهُ، وَإِنْ أَمْكَنَهُ اخْتِلَاسُ شَيْءٍ فَعَلَ ... أَوْ يُلَاقِسُونَ النِّسَاءَ فِي مَجَامِعِ الأَسْوَاقِ، مِنْ غَيْرِ احْتِشَامٍ وَلَا حَيَاءٍ"
عبد الرحمن الجبرتي
شرح الأساس الكبير (شفاء صدور الناس بشرح الأساس): أحمد بن محمد الشرفي (ت 1055هـ)، تح: أحمد عطا الله عارف، دار الحكمة اليمانية، صنعاء، ط1، (1411هـ/ 1991م)، ‏1/‏35‏.

تَلْقِيس[مَصْدَر]

1270هـ=1853م

التَّلْقِيسُ : التَّعْيِيبُ عَلَى الشَّخْصِ، وَذِكْرُهُ بِلَقَبٍ قَبِيحٍ.

"وَأَوَّلُ مَا أَطَلَّ عَلَى السُّوقِ، أَمَرَ جَمِيعَ القُبَّعِيِّينَ أَنْ يَخْرُجُوا لِمُلَاقَاتِه بِالتَّقْلِيسِ، لَا بِالتَّلْقِيسِ "
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏163‏.

لِقَاس[مَصْدَر]

1288هـ=1870م

اللِّقَاسُ : المُشَاتَمَةُ وَالمُنَابَزَةُ بِالأَلْقَابِ.

"يُقَالُ: مَا هَذَا اللِّقَاسُ ، فَهُوَ عَيْبٌ لَا يَلِيقُ لِلْمُسْلِمِ؟!"
بطرس البستاني
محيط المحيط: بطرس البستاني، مكتبة لبنان، بيروت، (1987م)، ‏822‏.

تَمَلْقَسَ[مُتَعَدٍّ بِالحَرْف]

1384هـ=1964م

تَمَلْقَسَ بِالكَلِمَةِ عَلَى الشَّخْصِ: سَخِرَ بِهَا مِنْهُ.

"بَلْ عَلَى العَكْسِ، تَخْطُرُ لِأَحَدِ الحُضُورِ كَلِمَةٌ فَظَّةٌ يَتَمَلْقَسُ بِهَا عَلَيَّ"
ميشال نقولا
أحمر من الحمار؛ قصص جد قصص مزاح: ميشال نقولا، مطابع سيما، بيروت، (1964م)، ‏97‏.

مَلْقِس[مَصْدَرٌ مِيمِيّ]

1392هـ=1971م

المَلْقِس : العَيْبُ.

"فَإِنَّ مَجَالِسَ المُخَاصَمَات، وَمَلَامِسَ المُجَادَلَات، وَمَلَاقِسَ الكَلِمَات، قَدِ انْتَشَرَتْ وَتَنَاثَرَت، وَنَفَرَتْ وَتَكَاثَرَت"
عبد السّلام أبو النّجا سرحان
من طرائف الأدب العربي؛ مقالات وبحوث: عبد السلام سرحان، كلية اللغة العربية، جامعة الأزهر، (1971م)، ‏214‏.
