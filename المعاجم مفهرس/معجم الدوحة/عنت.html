مُعَنِّت[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

المُعَنِّتُ : المُفْسِدُ.

أُنْحِي صَبِيَّ السَّيْفِ وَسْطَ بُيُوتِهِمْ شَقَّ المُعَنِّتِ فِي أَدِيمِ المِلْطَمِ
عبد مَناف بن رِبْع الجُرَبيّ الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏688‏.
9هـ=630م

المُعَنِّتُ مِنَ الأَشْخَاصِ: المُشَدِّدُ المُكَلِّفُ غَيْرَهُ مَا يَكْرَهُ.

"إِنَّ اللهَ لَمْ يَبْعَثْنِي مُعَنِّتًا وَلَا مُتَعَنِّتًا، وَلَكِنْ بَعَثَنِي مُعَلِّمًا مُيَسِّرًا"
حديث نبويّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏3/‏1105،،‏(رقم الحديث: 1478).
336هـ=947م

المُعَنِّتُ : الَّذِي يُضَيِّقُ عَلَى غَيْرِهِ فِي السُّؤَالِ وَالمُنَاظَرَةِ طَلَبًا لِزَلَّاتِهِ.

"وَإِذَا تَفَطَّنَ لِمَا قُلْنَاهُ نَاظِرٌ مُنْصِفٌ، وَعَالِمٌ غَيْرُ مُعَنِّتٍ ، ظَهَرَ لَهُ عُوَارُهُ، وَتَبَيَّنَ حَقِيقَتَهُ"
أحمد بن محمّد العَروضيّ
الجامع في العَروض والقوافي: أحمد بن محمّد العَروضيّ (ت، 342هـ)، حققه وقدم له: زهير غازي زاهد، هلال ناجي، دار الجيل، بيروت، ط1، (1416هـ/ 1996م)، ‏71‏.
ن418هـ=1027م

المُعَنِّتُ : الَّذِي يُقَلِّلُ مِنْ شَأْنِ غَيْرِهِ، وَيَزْدَرِي بِهِ.

"ثُمَّ يَعُودُ إِلَيْهِ العَقْعَقُ، فَيُعَامِلُهُ بِمِثْلِ ذَلِكَ كَالمُسْتَهْزِئِ مِنْهُ... المُعَنِّتِ إِيَّاهُ"
ابن سينا
الشّفاء الطّبيعيّات: ابن سينا (ت، 428هـ)، ج(1- 3)، تحقيق: سعيد زايد وآخرين، تصدير ومراجعة: إبراهيم مدكور، منشورات مكتبة آية الله العظمى المرعشي النجفي، قم، دار الكتاب العربي- الهيئة المصرية العامة للتأليف والنشر، القاهرة، ط2، (1406هـ)، ‏112‏.
ن483هـ=1090م

المُعَنِّتُ : المُعَانِدُ.

"لَمْ تَكُنْ حِكْمَةُ النَّسْلِ إِلَّا بِهَذَا الفِعْلِ؛ وَإِنْ أَنَا مُتُّ تَارِكًا لَهُ أَصْلًا، كُنْتُ كَالسَّاخِطِ أَوِ المُعَنِّتِ لِمَا رَتَّبَهُ الرَّبُّ، وَعَسَى بِذَلِكَ نَسْتَوْجِبُ عِقَابَهُ"
عبد الله بن بلكين بن باديس الصُّنْهاجيّ
مذكرات الأمير عبد الله آخر ملوك بني زيري بغرناطة المسمّاة بكتاب "التّبيان": عبد الله بن بلكين بن باديس الصُّنْهاجيّ (ت، 483هـ)، نشر وتحقيق: إ. ليفي بروفنسال، دار المعارف، مصر، (1955م)، ‏199‏.

مُتَعَنَّت[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُتَعَنَّتُ عَلَيْهِ مِنَ الأُمُورِ: مَا يُسْتَجْلَبُ بِسَبَبِهِ التَّشَدُّدُ وَالمَشَقَّةُ.

نَمُشُّ بِأَعْرَافِ الجِيَادِ أَكُفَّنَا إِذَا نَحْنُ قُمْنَا عَنْ شِوَاءٍ مُضَهَّبِ
إِلَى أَنْ تَرَوَّحْنَا بِلَا مُتَعَنَّتٍ عَلَيْهِ كِسِيدِ الرُّدْهَةِ المُتَأَوِّبِ
امرؤ القيس بن حُجْر الكنديّ
الجزء الأول من سمط اللآلي ويحتوي على النصف الأول من اللآلي في شرح أمالي القالي: أبو عُبيد البَكْريّ الأندلسيّ (ت، 487هـ)، نسخه وصححه ونقحه وحقق ما فيه واستخرجه من بطون دواوين العلم: عبد العزيز الميمني، دار الكتب العلمية، بيروت، د.ت. ‏1/‏68‏-‏ 69‏.
