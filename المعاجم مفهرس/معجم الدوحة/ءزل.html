أَزْل[مَصْدَر]

قبل 50ق.هـ=573م

الأَزْلُ : الشِّدَّةُ وَالضِّيقُ.

إِنِّي لِعَمْرٍو مَا انْتَمَيْتُ، فَلَمْ أَعْدِلْ إِلَى بَدَلٍ، وَلَا مِثْلِ
وَلَمِثْلُ أَسْبَابٍ عَلِقْتُ بِهَا يَمْنَعْنَ مِنْ قَلَقٍ، وَمِنْ أَزْلِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏205‏.
قبل 13ق.هـ=609م

أَزْلُ النُّوقِ وَنَحْوِهَا: شِحَّةُ لَبَنِهَا.

نَشُقُّ بِهَا السِّنِينَ، وَلَا نُبَالِي بِهَا أَزْلَ المَخَاضِ، وَلَا اللِّقَاحِ
عبد الله بن ثور العامريّ
كتاب الوَحشيَّات وهو الحَماسة الصُّغْرى: أبو تمام (ت، 231هـ)، علق عليه وحققه: عبد العزيز الميمني، زاد في حواشيه: محمود محمّد شاكر، دار المعارف، القاهرة، ط3، (1987م)، ‏65‏.
ن110هـ=728م

الأَزْلُ : الحَبْسُ.

لَقَدْ رَجَعَتْ شَيْبَانُ، وَهْيَ أَذِلَّةٌ خَزَايَا، فَفَاظَتْ فِي الوِثَاقِ، وَفِي الأَزْلِ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏285‏.
ن175هـ=791م

أَزْلُ الفَرَسِ وَنَحْوِهِ: تَقْصِيرُ حَبْلِهِ، ثُمَّ إِرْسَالُهُ فِي المَرْعَى.

"أَزَلْتُ الفَرَسَ أَزْلًا : قَصَّرْتُ حَبْلَهُ، ثُمَّ أَرْسَلْتُهُ فِي المَرْعَى"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏385‏.

أَزْل[اسْم]

قبل 13ق.هـ=609م

الأَزْلُ : القَحْطُ وَالجَدْبُ.

إِلَى القَمَرِ السَّارِي المُنِيرِ دَعْوَتُهُ وَمُطْعِمِهِمْ فِي الأَزْلِ مِنْ قِمَعِ الجَزْرِ
مطرود بن كعب الخُزاعيّ
رسائل الجاحظ: الجاحظ (ت، 255هـ)، الرسائل السياسية، قدّم لها وبوبها وشرحها: علي أبو ملحم، دار ومكتبة الهلال، بيروت، (2002م)، ‏410‏.

آزِل[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الآزِلُ مِنَ الدَّوَابِّ: مَا حُبِسَ عَنِ المَرْعَى خَوْفًا عَلَيْهِ.

وَلَبُونِ مِعْزَابٍ حَوَيْتَ، فَأَصْبَحَتْ نُهْبَى، وَآزِلَةٍ قَضَبْتَ عِقَالَهَا
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏1/‏163‏.
142هـ=759م

الآزِلُ مِنَ الأُمُورِ: الشَّدِيدُ العُسْرِ.

وَابْنَا نِزَارٍ فَرَّجَا الزَّلَازِلَا عَنِ المُصَلِّينَ، وَأَزْلًا آزِلَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏353‏.

آزِل[اسْمُ فَاعِل]

ن11هـ=632م

الآزِلُ مِنَ النَّاسِ: الوَاقِعُ فِي شِدَّةٍ وَجَدْبٍ.

"وَعِلْمُ مَا فِي غَدٍ... وَعِلْمُ يَوْمِ الغَيْثِ، يُشْرِفُ عَلَيْكُمْ آزِلِينَ ، آزِلِينَ مُشْفِقِينَ، فَيَظَلُّ يَضْحَكُ، قَدْ عَلِمَ أَنَّ غِيَرَكُمْ إِلَى قُرْبٍ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏26/‏122،،‏(رقم الحديث: 16206).
ن449هـ=1057م

آزِلُ الشَّيْءِ: حَابِسُهُ.

دَعَتْ رَبَّهَا أَنْ يُهْلِكَ البِيضَ وَالقَنَا وَكُلٌّ لَهُ مِنْ قُدْرَةِ اللهِ آزِلُ
أبو العلاء المَعَرّيّ
اللّزوميّات: أبو العلاء المَعَرّيّ (ت، 449هـ)، تحقيق: أمين عبد العزيز الخانجي، مكتبة الهلال، بيروت، مكتبة الخانجي، القاهرة، (1342هـ)، ‏2/‏179‏.

مُؤْزِلَة[اسْمُ فَاعِل]

9هـ=630م

المُؤْزِلَةُ مِنَ السِّنِينَ: الَّتِي تَأْتِي بِالجَدْبِ وَالقَحْطِ.

"وَلَنَا نَعَمٌ... أَصَابَتْهَا سُنَيَّةٌ حَمْرَاءُ مُؤْزِلَةٌ "
طَهْفَة/طَخْفَة بن أبي زهير النَّهْديّ
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏2/‏54‏.

أَزَلَ[مُتَعَدٍّ]

ن11هـ=632م

أَزَلَ الشَّخْصَ: ضَيَّقَ عَلَيْهِ وَحَبَسَهُ لِقَحْطٍ أَوْ نَحْوِهِ.

" فَيُؤْزَلُونَ أَزْلًا شَدِيدًا، ثُمَّ يُهْلِكُهُ اللهُ وَجُنُودَهُ"
حديث نبويّ
المعجم الكبير: أبو القاسم الطَّبَرانيّ (ت، 360هـ)، حققه وخرج أحاديثه: حمدي عبد المجيد السلفي، مكتبة ابن تيمية، القاهرة، ط2، (1404هـ/ 1983م)، ‏7/‏231‏.
ن175هـ=791م

أَزَلَ الفَرَسَ وَنَحْوَهُ: قَصَّرَ حَبْلَهُ، ثُمَّ أَرْسَلَهُ فِي المَرْعَى.

" أَزَلْتُ الفَرَسَ أَزْلًا: قَصَّرْتُ حَبْلَهُ، ثُمَّ أَرْسَلْتُهُ فِي المَرْعَى"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏385‏.

إِزْل[اسْم]

ن60هـ=680م

الإِزْلُ : الكَذِبُ.

يَقُولُونَ: إِزْلٌ حُبُّ لَيْلَى وَوُدُّهَا وَقَدْ كَذَبُوا، مَا فِي مَوَدَّتِها إِزْلُ
عبد الرّحمن بن مسافع الغَطَفانيّ
إصلاح المنطق: ابن السّكّيت (ت، 244هـ)، شرح وتحقيق: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط4، (1368هـ/ 1949م)، ‏1/‏6‏.

مَأْزُول[صِفَةٌ مُشَبَّهَة]

105هـ=723م

المَأْزُولُ مِنَ الأَمَاكِنِ وَنَحْوِهَا: الضَّيِّقُ.

أَلَسْنَا نَحْنُ، قَدْ عَلِمَتْ مَعَدٌّ غَدَاةَ الرَّوْعِ، أَجْدَرَ أَنْ نَغَارَا
وَأَطْعَنَ حِينَ تَخْتَلِفُ العَوَالِي بِمَأْزُولٍ إِذَا مَا النَّقْعُ ثَارَا؟
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏888‏.
132هـ=749م

المَأْزُولُ مِنَ الأَمَاكِنِ: الخَشِنُ.

وَشَفَّهَا اللَّوْحُ بِمَأْزُولٍ ضَيَقْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏21‏.

مَأْزُول[اسْمُ مَفْعُول]

125هـ=743م

المَأْزُولُ مِنَ الدَّوَابِّ: الَّذِي قُصِّرَ حَبْلُهُ، ثُمَّ أُرْسِلَ فِي المَرْعَى.

يَسُفْنَ عِطْفَىْ سَنِمٍ هَمَرْجَلِ لَمْ يَرْعَ مَأْزُولًا ، وَلَمْ يَسْتَمْهِلِ
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏343‏.
375هـ=985م

المَأْزُولُ مِنَ الدَّوَابِ: المَحْبُوسُ عَنِ المَرْعَى.

" يُقَالُ: مَا المَعْذُولُ... وَالمَأْزُولُ ؟... أَمَّا المَعْذُولُ فَالمَلُومُ... وَأَمَّا المَأْزُولُ فَالمَحْبُوسُ، يُقَالُ: أَزَلُوا مَالَهُمْ؛ أَيْ حَبَسُوهُ عَنِ المَرْعَى"
أبو حيّان التَّوحيديّ
البصائر والذّخائر: أبو حيّان التَّوحيديّ (ت، 414هـ)، تحقيق: وداد القاضي، دار صادر، بيروت، ط1، (1408هـ/ 1988م)، ‏8/‏146‏-‏ 151‏.

مَأْزِل[اسْمُ مَكَان]

125هـ=743م

المَأْزِلُ : المَضِيقُ.

إِذَا دَنَتْ مِنْ عَضَدٍ لَمْ يُشْغَلِ عَنْهَا، وَلَوْ كَانَ بِضَيْقٍ مَأْزِلِ
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏352‏.
ن154هـ=771م

المَأْزِلُ : المَحْبِسُ.

"شُدُّونِي مَكَانَهُ حَتَّى تَأْتِيَكُمُ الإِبِلُ. وَأَرْسَلَ غُلَامَهُ، فَقَالَ: اعْجَلْ عَلَيَّ بِالفِدَاءِ. فَأَتَتْ حَاتِمًا أُمُّ مَأْزِلِهِ ...فَقَالَتْ: افْصِدْ لَنَا هَذِهِ النَّاقَةَ، فَلَتَبَ بِالشَّفْرَةِ فِي لَبَّتِهَا"
أبو عمرو بن العَلاء
ديوان المفضّليّات: المفضَّل الضَّبّيّ (ت، 178هـ)، شرح: أبي محمّد القاسم بن محمّد بن بشّار الأنباريّ، تحقيق: كارلوس يعقوب لايل، مطبعة الآباء اليسوعيّين، بيروت، كلارندون، أكسفورد، (1920م)، ‏748‏.

أَزَل[اسْم]

ن150هـ=767م

الأَزَلُ : القِدَمُ الَّذِي لَيْسَ لَهُ ابْتِدَاءٌ.

"لَمْ يَزَلْ، وَلَا يَزَالُ بِأَسْمَائِهِ وَصِفَاتِهِ. لَمْ يَحْدُثْ لَهُ اسْمٌ وَلَا صِفَةٌ. لَمْ يَزَلْ عَالِمًا بِعِلْمِهِ، وَالعِلْمُ صِفَةٌ فِي الأَزَلِ "
أبو حنيفة النّعمان
مِنَح الروض الأزهر في شرح الفقه الأكبر: المُلّا علي القاري (ت، 1014هـ)، تحقيق: وهبة سليمان الغاوجي، دار البشائر الإسلامية، بيروت، ط1، (1419هـ/ 1998م)، ‏87 ‏-‏88‏.

مُؤْزَل[اسْمُ مَفْعُول]

ن150هـ=767م

المُؤْزَلُ مِنَ الدَّوَابِّ: الَّذِي قُصِرَ حَبْلُهُ، وَأُرْسِلَ فِي المَرْعَى.

وَأُرِيحَتْ سَوَامُهُمْ مُؤْزَلَاتٍ فَسَوَاءٌ غُدُوُّهَا وَالرَّوَاحُ
الأجشّ الثّقفيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏57‏.

أَزَلِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن200هـ=815م

الأَزَلِيَّةُ : القِدَمُ الَّذِي لَا بِدَايَةَ لَهُ.

"وَهَذَا المُحَالُ وَالتَّنَاقُضُ الوَاضِحُ إِذَا كَانَا جِرْمَيْنِ، إِلَّا أَنْ تَدْخُلَ فِي ذَلِكَ الِاسْتِحَالَةُ، فَيَجِبُ مِنْ ذَلِكَ أَنَّ أَزَلِيَّتَهُمَا تَبْطُلُ، وَتَضْمَحِلُّ، وَالعِلَّةُ مَعَ المَعْلُولِ مُفَارِقَةٌ"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏123‏.

أَزَلِيَّة[اسْم]

1340هـ=1921م

الأَزَلِيَّةُ : فِرْقَةٌ مُتَفَرِّعَةٌ مِنَ البَهَائِيَّةِ، تَنْتَسِبُ إِلَى صُبْحِ أَزَلْ. [الأديان والمذاهب والفرق]

"وَانْشَقَّتِ البَابِيَّةُ إِلَى فِرْقَتَيْنِ، سُمِّيَتِ الأُولَى البَابِيَّةَ الأَزَلِيَّةَ نِسْبَةً إِلَى صُبْحِ أَزَلْ، وَسُمِّيَتِ الثَّانِيَةُ البَابِيَّةَ البَهَائِيَّةَ نِسْبَةً إِلَى البَهَاءِ"
أحمد رضا العاملي
مجلة العرفان: العرفان، صيدا، مج7، ج3، (ديسمبر- 1921م)، ‏149‏.

أَزَلِيّ[صِفَة]

ن200هـ=815م

الأَزَلِيُّ : القَدِيمُ الَّذِي لَا بِدَايَةَ لَهُ.

"فَلَنْ يَخْلُوَ مِنْ أَنْ يَكُونَ أَزَلِيًّا ، أَوْ لَا أَزَلِيًّا"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏132‏.
ن350هـ=961م

الأَزَلِيُّ : الأَبَدِيُّ.

"وَأَمَّا فِي القِسْمَةِ فَبِخِلَافِ ذَلِكَ يَكُونُ؛ لِأَنَّهَا تَأْخُذُ الحَدَّ الأَوْسَطَ أَكْبَرَ. فَلْيَكُنِ الحَيُّ (اَ) وَالمَائِتُ (بَ) وَالأَزَلِيُّ (حَ)، وَأَمَّا الإِنْسَانُ الَّذِي يَنْبَغِي أَنْ يُؤْخَذَ حَدُّهُ فَلْيَكُنْ (دَ). فَالمُسْتَعْمِلُ لِلْقِسْمَةِ يَأْخُذُ أَنَّ كُلَّ حَيٍّ إِمَّا أَنْ يَكُونَ مَائِتًا أَوْ أَزَلِيًّا. وَذَلِكَ هُوَ أَنَّ كُلَّ (اَ) إِمَّا أَنْ يَكُونَ (بَ) أَوْ (حَ)"
تَذاري بن باسل
منطق أرسطو: حققه وقدم له: عبد الرحمن بدويّ، وكالة المطبوعات، الكويت، دار القلم، بيروت، ط1، (1980م)، ‏1/‏225‏.

تَأَزُّل[مَصْدَر]

555هـ=1160م

التَّأَزُّلُ : التَّقَدُّمُ بِلَا بِدَايَةٍ.

"القَدِيمُ بِتَقَدُّمِهِ، الأَزَلِيُّ بِتَأَزُّلِهِ "
الهَمْدانيّ الحامديّ
كتاب كنز الولد: الهَمْدانيّ الحامديّ (ت، 597هـ)، عني بتحقيقه: مصطفى غالب، دار الأندلس، بيروت، (1979م)، ‏49‏.

مُتَأَزِّل[صِفَةٌ مُشَبَّهَة]

555هـ=1160م

المُتَأَزِّلُ مِنَ النُّفُوسِ وَنَحْوِهَا: الضَّيِّقُ.

"وَقَدْ صَارَتْ نَفْسُهُ مُتَأَزِّلَةً بِالبَقَاءِ السَّرْمَدِ مِمَّا عَلِمَهُ وَأَنْكَرَهُ"
الهَمْدانيّ الحامديّ
كتاب كنز الولد: الهَمْدانيّ الحامديّ (ت، 597هـ)، عني بتحقيقه: مصطفى غالب، دار الأندلس، بيروت، (1979م)، ‏299‏.

مُؤَزِّل[اسْمُ فَاعِل]

ن612هـ=1215م

المُؤَزِّلُ لِلنُّفُوسِ وَنَحْوِهَا: الَّذِي يَجْعَلُهَا بَاقِيَةً أَبَدًا.

"فَكَيْفَ لَا يَكُونُ ذَلِكَ كَذَلِكَ، وَهُوَ المِغْنَاطِيسُ الأَكْبَرُ الجَاذِبُ لِلنُّفُوسِ الشَّرِيفَةِ إِلَى دَارِ المَعَاد، المُحَافِظُ لِذَوَاتِهَا مِنَ التَّلَاشِي وَالفَسَاد... المُوصِلُ لَهَا إِلَى جَنَّةِ المَأْوَى آمِنَةً مِنَ الشَّقَاوَةِ وَالخَسَار، المُؤَزِّلُ لَهَا عَلَى مَمَرِّ الدُّهُورِ وَالأَعْصَارِ"
والد الجميع
الرسالة المفيدة في شرح القصيدة وهي: (القصيدة العينية لابن سينا): والد الجميع (ت، 612هـ)، تحقيق وتقديم وتعليق: الحبيب الفقي، حوليات الجامعة التونسية، تونس، ع7، (1979م)، (117- 182)، ‏173‏.

مُؤَزَّل[اسْمُ مَفْعُول]

ن765هـ=1363م

المُؤَزَّلُ (مَفْعُولٌ بِمَعْنَى فَاعِل): الدَّائِمُ أَبَدًا.

فَهَذَا بَقَاءٌ فِي فَنَاءٍ مُؤَبَّدٍ وَهَذَا فَنَاءٌ فِي بَقَاءٍ مُؤَزَّلُ
محمّد وفا الشاذليّ
ديوان محمد وفا الشاذلي: محمّد وفا الشاذليّ (ت، 765هـ)، تحقيق ودراسة وشرح: مهدي أسعد عرار، جامعة بيرزيت، القدس الشريف، (1436ه/ 2015م)، ‏362‏.

تَأْزِيل[مَصْدَر]

1437هـ=2015م

تَأْزِيلُ الوُجُودِ: جَعْلُهُ قَدِيمًا بِلَا حَدٍّ.

"لَمْ يَسْتَفِدْ مِنْ فَاعِلِيَّاتِهَا بِدَفْعِهَا بَعِيدًا عَنْ تَأْزِيلِ وُجُودِهَا"
عبيدو باشا
