مُغْدَوْدِن[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

المُغْدَوْدِنُ مِنَ الشَّعْرِ: النَّاعِمُ الطَّوِيلُ، الشَّدِيدُ السَّوَادِ.

وَقَامَتْ تُرَائِيكَ مُغْدَوْدِنًا إِذَا مَا تَنُوءُ بِهِ آدَهَا
حسّان بن ثابت الخزرجيّ الأنصاريّ
ديوان حسان بن ثابت: تحقيق: سيد حنفي حسنين، دار المعارف، القاهرة، (1983م)، ‏102‏.
ن90هـ=709م

المُغْدَوْدِنُ مِنَ النَّبَاتِ: المُلْتَفُّ، الشَّدِيدُ الِاخْضِرَارِ مِنْ شِدَّةِ رِيِّهِ.

مُغْدَوْدِنُ الأَرْطَى، غُدَانِيُّ الضَّالْ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏2/‏318‏.
126هـ=744م

المُغْدَوْدِنُ مِنَ الأَشْخَاصِ: المُسْتَرْخِي، اللَّيِّنُ فِي كَلَامِهِ.

وَدَغْيَةٍ مِنْ خَطِلٍ مُغْدَوْدِنِ قُرْبَانَ مُلْكٍ، أَوْ شَرِيفِ المَعْدِنِ قَامَتْ بِهِ شُدَّاكَ بَعْدَ الأَوْهَنِ بِدَرْءِ هَمَّازٍ، دُرُوءَ الضَّيْزَنِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏120‏-‏ 121‏.
ن203هـ=818م

المُغْدَوْدِنُ مِنَ الأَرْضِ: المُعْشِبُ، الكَثِيرُ الكَلَإِ المُلْتَفِّ.

إِذَا غَرَّدَ اللَّقَّاعُ فِيهِمَا لِعَنْتَرٍ بِمُغْدَوْدِنٍ ، مُسْتَأْسِدِ النَّبْتِ، ذِي خَبْرِ
من إنشاد النَّضْر بن شُميل المازنيّ التّميميّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏3/‏355‏.
ن426هـ=1035م

المُغْدَوْدِنُ مِنَ الحِبَالِ وَنَحْوِهَا: الطَّوِيلُ المُسْتَرْخِي.

وَفُتُوٌّ سَرَوْا، وَقَدْ عَكَفَ اللَّيْـ ـلُ، وَأَرْخَى مُغْدَوْدِنَ الأَطْنَابِ
ابن شُهيد الأندلسيّ
رسالة التّوابع والزّوابع: ابن شُهيد الأندلسيّ (ت، 426هـ)، صححها وحقق ما فيها وشرحها وبوبها وصدرها بدراسة تاريخية أدبية: بطرس البستاني، دار صادر، بيروت، ط2، (1416هـ/ 1996م)، ‏103‏.

غَدَن[مَصْدَر]

ن75هـ=694م

الغَدَنُ : الِاسْتِرْخَاءُ، وَالفُتُورُ.

عَلَى مَصَامِيدَ كَأَمْثَالِ الجُوَنْ وَلَمْ تَضَعْ أَوْلَادَهَا مِنَ البَطَنْ أَحْمَرَ لَمْ يُعْرَفْ بِبُؤْسٍ مُذْ مَهَنْ وَلَمْ تُصِبْهُ نَعْسَةٌ عَلَى غَدَنْ
القُلاخ بن حَزْن المِنْقَريّ السَّعْديّ
أراجيز المقلّين: تحقيق: محمّد يحيى زين الدّين، مجلّة مجمع اللُّغة العربيّة، دمشق، مج57، ج(1، 2)، 1982م، مج57، ج(1، 2)، 1982م، ج3، ج4، مج59، ج2، 1984م، ج4، مج68، ج2، 1993م، مج70، ج2، ‏4/‏402‏.
ن244هـ=858م

الغَدَنُ : اللِّينُ، وَالنِّعْمَةُ.

"إِنَّ فِيهِ لَغَدَنًا "
من رواية ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏13‏.

غُدَانِيّ[صِفَة]

ن90هـ=709م

الغُدَانِيُّ مِنَ النَّبَاتِ: الكَثِيرُ، الرَّيَّانُ، المُسْتَرْخِي.

مُغْدَوْدِنُ الأَرْطَى، غُدَانِيُّ الضَّالْ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏2/‏318‏.
ن97هـ=716م

الغُدَانِيُّ مِنَ الأَشْيَاءِ: الغَضُّ، المُمْتَلِئُ نَضَارَةً وَرِيًّا.

أَصَمُّ، غُدَانِيٌّ ، كَأَنَّ جَبِينَهُ لُطَاخَةُ نَقْسٍ فِي أَدِيمٍ مُمَجْمَجِ
الحُضين بن المنذر الرَّقاشيّ
شرح نقائض جرير والفرزدق: رواية: أبي عبد الله اليزيدي عن أبي سعيد السكري عن ابن حبيب عن أبي عبيدة، تحقيق وتقديم: محمّد حُوّر، وليد محمود خالص، منشورات المجمع الثقافي، أبو ظبي، ط2، (1998م)، ‏2/‏533‏.
ن145هـ=762م

غُدَانِيُّ الشَّبَابِ وَنَحْوِهِ: غَضَاضَتُهُ، وَنَضَارَتُهُ.

قَالَتْ أُبَيْلَى لِي وَلَمْ أُسَبَّهِ: مَا السِّنُّ إِلَّا غَفْلَةُ المُدَلَّهِ لَمَّا رَأَتْنِي خَلَقَ المُمَوَّهِ بَرَّاقَ أَصْلَادِ الجَبِينِ الأَجْلَهِ بَعْدَ غُدَانِيِّ الشَّبَابِ الأَبْلَهِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏61‏-‏ 62‏.

أَغْدَنَ[لَازِم]

ن126هـ=744م

أَغْدَنَ الشَّيْءُ: طَالَ، وَتَمَّ.

وَمُعْتَرَضٌ، فَوْقَ القُتُودِ، تَخَالُهُ مَتَاعًا مُعَلَّى أَوْ قَتِيلًا مُكَفَّنَا
جَلَوْتُ الكَرَى عَنْهُ بِذِكْرِكِ، بَعْدَمَا دَنَا اللَّيْلُ، وَالْتَجَّ الظَّلَامُ، فَأَغْدَنَا
يزيد ابن الطَّثَريّة القُشيريّ
شعر يزيد بن الطَّثْرِيّة: صنعة: حاتم الضامن، وزارة الإعلام- مطبعة أسعد، بغداد، (1973م)، ‏94‏.

اغْدَوْدَنَ[لَازِم]

ن701هـ=1301م

اغْدَوْدَنَ الشَّيْءُ: طَالَ، وَتَمَّ.

"فَلَمَّا احْلَوْلَكَتْ حَنَادِسُ الجَلَاء، وَاغْدَوْدَنَتْ غَلَائِلُ الغَلَاء، وَاسْتَدَّتِ المَسَالِكُ وَادْلَهَمَّتْ... انْسَلَلْتُ مِنَ القِرَاب، وَمَلِلْتُ حَلَاوَةَ الِاقْتِرَاب"
ابن الصيْقَل الجَزَريّ
المقامات الزّينية: ابن الصيْقَل الجَزَريّ (ت، 701هـ)، دراسة وتحقيق: عبّاس مُصطفى الصّالحيّ، دار المسيرة، بيروت، ط1، (1400ه/ 1980م)، ‏197‏-‏ 198‏.

غُدَّان[اسْم]

ن919هـ=1513م

الغُدَّانُ : نَوْعٌ مِنْ أَنْوَاعِ التِّينِ فِي المَغْرِبِ.

"وَغَيْرُ ذَلِكَ مِنْ أَنْوَاعِ التِّينِ، كَالأَشْكُوزِ، وَالشِّبْلِيِّ، وَالحَمْرَاءِ، وَالغُدَّانِ ، وَالحَافِرِ، وَالنَّقَّالِ، وَغَيْرِهَا"
أبو عبد الله محمد بن أحمد بن غازي العثماني المكناسي
الروض الهتون في أخبار مكناسة الزيتون: أبو عبد الله محمد بن أحمد بن غازي العثماني المكناسي (ت، 919هـ)، مطبعة الأمنية، الرباط، (1371ه/ 1952م)، ‏3‏.

غُدَّانِيّ[اسْمٌ مَنْسُوب]

994هـ=1585م

الغُدَّانِيُّ : المَنْسُوبٌ إِلَى غُدَّانٍ.

"وَمِنْهَا الغُدَّانِيُّ ، يَلِدُ مَرَّتَيْنِ فِي السَّنَةِ، وَهُوَ كَثِيرٌ بِتِلِمْسَانَ، مَشْهُورٌ بِهَا" .
أبو القاسم بن محمد الغساني
حديقة الأزهار في ماهية العشب والعقار: أبو القاسم بن محمد الغساني (ت، 1019هـ)، حققه وعلق حواشيه ووضع فهارسه: محمد العربي الخطابي، دار الغرب الإسلامي، بيروت، ط2، (1410ه/ 1990م)، ‏296‏.

تَغَدُّن[مَصْدَر]

1270هـ=1853م

التَّغَدُّنُ : التَّمَايُلُ وَالتَّعَطُّفُ.

"فَإِذَا بَكَرَتْ فِي اليَوْمِ القَابِلِ إِلَى الأَسْوَاقِ وَرَأَتِ النَّاسَ مُكِبِّينَ عَلَى أَشْغَالِهِمْ تَعَجَّبَتْ مِنْ بَقَائِهِمْ أَصِحَّاءَ قَادِرِينَ عَلَى السَّعْيِ وَالحَرَكَةِ؛ فَزَادَتْ لَهُمْ فِي كَشْفِ مَسَافِرِهَا ... وَتَمَايُلِهَا وَتَهَادِيهَا، وَتَغَدُّنِهَا وَتَعَاطُفِهَا"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏166‏.

غِدَان[اسْم]

1316هـ=1898م

الغِدَانُ : القَضِيبُ الَّذِي تُعَلَّقُ عَلَيْهِ الثِّيَابُ.

" وَالغُدْنِ ، وَالمَصَابِيحِ، وَالزُّهْرِيَّاتِ، وَسَائِرِ الآنِيَةِ، وَالمَاعُونِ النَّفِيسِ"
محمد رشيد رضا
الاتحاد: ‏محمد رشيد رضا،‏مجلة المنار: القاهرة: مج1، (1315-1316هـ)، ‏555‏.
