أَنَاضَ[لَازِم]

قبل 13ق.هـ=609م

أَنَاضَ النَّخْلُ: أَيْنَعَ وَأَثْمَرَ.

يَوْمَ أَرْزَاقُ مَنْ يُفَضِّلُ عُمٌّ مُوسَقَاتٌ، وَحُفَّلٌ أَبْكَارُ
فَاخِرَاتٌ ضُرُوعُهَا فِي ذُرَاهَا وَأَنَاضَ العَيْدَانُ وَالجَبَّارُ
لَبيد بن ربيعة العامريّ
شرح ديوان لبيد بن ربيعة العامري: حققه وقدم له: إحسان عبّاس، وزارة الإرشاد والأنباء، الكويت، (1962م)، ‏41‏-‏ 42‏.

نَوْض[اسْم]

11هـ=632م

النَّوْضُ : الوَادِي.

أَرْوَى الأَنَاوِيضَ ، وَأَرْوَى مِذْنَبَهْ
لَبيد بن ربيعة العامريّ
شرح ديوان لبيد بن ربيعة العامري: حققه وقدم له: إحسان عبّاس، وزارة الإرشاد والأنباء، الكويت، (1962م)، ‏356‏.
126هـ=744م

النَّوْضُ : مَخْرَجُ المَاءِ وَمَنْفَقُهُ.

أَرَّقَ عَيْنَيْكَ عَنِ الغَمَاضِ بَرْقٌ سَرَى فِي عَارِضٍ نَهَّاضِ غُرِّ الذُّرَى ضَوَاحِكِ الإِيمَاضِ يُسْقَى بِهِ مَدَافِعُ الأَنْوَاضِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏325‏.
ن145هـ=762م

النَّوْضُ : الوُصْلَةُ الَّتِي بَيْنَ العَجُزِ وَالمَتْنِ مِنَ الجَسَدِ.

إِذَا اعْتَزَمْنَ الرَّهْوَ فِي انْتِهَاضِ جَاذَبْنَ بِالأَصْلَابِ وَالأَنْوَاضِ
رؤبة بن العجّاج
مجموع أشعار العرب وهو مشتمل على ديوان رؤبة بن العجاج وعلى أبيات مفردات منسوبة إليه: اعتنى بتصحيحه وترتيبه: وليم بن الورد البروسيّ، دار ابن قتيبة، الكويت، د.ت. ‏176‏.

نَوْض[مَصْدَر]

ن175هـ=791م

النَّوْضُ : الحَرَكَةُ فِيهَا تَذَبْذُبٌ.

" وَالنَّوْضُ : الحَرَكَةُ كَالتَّذَبْذُبِ وَالتَّعَثْكُلِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏62‏.
300هـ=912م

نَوْضُ البَرْقِ وَنَحْوِهِ: تَلَأْلُؤُهُ.

نَفَى نَوْمَ عَيْنَيْكَ الخَيَالُ المُؤَرِّقُ وَنَوْضُ سَنًا بَاتَتْ لَهُ العَيْنُ تَأْرَقُ
أبو المسيَّب المُلَجَّمي الهُذَليّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏549‏.

نَاضَ[لَازِم]

ن125هـ=743م

نَاضَ الشَّخْصُ: ذَهَبَ فِي الأَرْضِ.

وَجَرَى بِالَّذِي أَخَافُ مِنَ البَيْـ ـنِ لَعِينٌ يَنُوضُ كُلَّ مَنَاضِ
الطِّرمّاح
ديوان الطِّرمَّاح: عني بتحقيقه: عِزّة حسن، دار الشرق العربي، بيروت- حلب، ط2، (1414هـ/ 1994م)، ‏171‏.
ن574هـ=1178م

نَاضَ الشَّيْءُ: تَلَأْلَأَ وَلَمَعَ.

صُبَيْحٌ يَنُوضُ البِشْرُ فِي قَسَمَاتِهِ كَمَا نَاضَ عُلْوِيُّ الغُيُومِ الحَوَافِلِ
حَيْصَ بَيْصَ
ديوان الأمير شهاب الدين أبي الفوارس سعد بن محمد بن سعد بن الصيفي التميمي البغدادي المعروف بـ(حَيْص بَيْص): حَيْصَ بَيْصَ (ت، 574هـ)، حققه وضبط كلماته وشرحها: مكي السَّيّد جاسم، شاكر هادي شكر، وزارة الإِعلام، بغداد، (1974م)، ‏2/‏10‏.

نَاضَ[مُتَعَدٍّ بِالحَرْف]

ن231هـ=845م

نَاضَ لِلْأَمْرِ: تَحَرَّكَ لَهُ.

"وَيُقَالُ: مَا يَنُوضُ لِحَاجَةٍ، وَمَا يَقْدِرُ أَنْ يَنُوضَ"
من رواية أبي عبد الله ابن الأعرابيّ اللُّغويّ
كتاب القلب والإبدال: ابن السِّكّيت (ت، 244هـ)- ضمن: الكنز اللغوي في اللَّسَن العربي: سعى في نشره وتعليق حواشيه: أوغست هفنر، المطبعة الكاثوليكية، بيروت، (1903م)، ‏49‏-‏ 50‏.

نَاضَ[مُتَعَدٍّ]

ن574هـ=1178م

نَاضَ الشَّيْءَ: عَالَجَهُ وَأَرَاغَهُ لِيَنْتَزِعَهُ.

وَأَرَيْتُهُ صُبْحَ المَسَرَّةِ بَعْدَمَا نَاضَ الهُمُومَ كَجُنْحِ لَيْلٍ أَلْيَلِ
حَيْصَ بَيْصَ
ديوان الأمير شهاب الدين أبي الفوارس سعد بن محمد بن سعد بن الصيفي التميمي البغدادي المعروف بـ(حَيْص بَيْص): حَيْصَ بَيْصَ (ت، 574هـ)، حققه وضبط كلماته وشرحها: مكي السَّيّد جاسم، شاكر هادي شكر، وزارة الإِعلام، بغداد، (1974م)، ‏1/‏302‏.

مَنَاض[مَصْدَرٌ مِيمِيّ]

ن125هـ=743م

المَنَاضُ : الذَّهَابُ فِي الأَرْضِ.

وَجَرَى بِالَّذِي أَخَافُ مِنَ البَيْـ ـنِ لَعِينٌ يَنُوضُ كُلَّ مَنَاضِ
الطِّرمّاح
ديوان الطِّرمَّاح: عني بتحقيقه: عِزّة حسن، دار الشرق العربي، بيروت- حلب، ط2، (1414هـ/ 1994م)، ‏171‏.
ن189هـ=805م

المَنَاضُ : المَلْجَأُ وَالمَفَرُّ.

"العَرَبُ تُبْدِلُ مِنَ الصَّادِ ضَادًا، فَتَقُولُ: مَا لَكَ مِنْ هَذَا مَنَاضٌ "
من رواية الكسائي
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏12/‏70‏.

مَنَاض[اسْمُ مَكَان]

772هـ=1370م

المَنَاضُ : المَلْجَأُ.

"أَعْنِي المَقَامَ الأَقْدَس، وَالإِمَامَ الأَنْدَس، مُرْشِدَ المُلُوكِ وَالسَّلَاطِين، مُرْصِدَ الهَلُوكِ لِلشَّيَاطِين، كَافِلَ أُمُورِ المُسْلِمِين، مَنَاضَ الإِسْلَام، مَنَاضَ الأَعْلَام"
مجد الدين محمد بن يعقوب الفيروزآبادي
كتاب الغرر المثلثة والدرر المبثّثة: مجد الدين محمد بن يعقوب الفيروزآبادي (ت، 817هـ)، تحقيق ودراسة: سليمان بن إبراهيم بن محمد العايد، مكتبة نزار مصطفى الباز، مكة المكرمة- الرياض، ط2، (1421ه/ 2000م)، ‏247‏.

نَوَّاض[صِيغَةُ مُبَالَغَة]

ن145هـ=762م

النَّوَّاضُ : الشَّدِيدُ النَّزْعِ وَالإِخْرَاجِ.

يَنْضُونَ مِنْ أَجْوَازِ لَيْلٍ غَاضِ نَضْوَ قِدَاحِ النَّابِلِ النَّوَّاضِ
رؤبة بن العجّاج
كتاب الأفعال: ابن الحدّاد المَعافريّ السَّرَقُسْطيّ (ت، 400هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: محمّد مهدي علام، مجمع اللغة العربية، القاهرة، ط2، (1398هـ/ 1978م- 1413هـ/ 1992م)، ‏3/‏170‏.
ن1274هـ=1857م

النَّوَّاضُ : الشَّدِيدُ التَّلَأْلُؤِ.

نَوَّاضَةٌ لِعَقْدِهَا انْحِلَالُ مُبْتَعِدٌ عَنْ خِصْبِهَا الأَمْحَالُ
ابن رزيق حميد بن محمد بن رزيق النخلي
الفتح المبين في سيرة السادة البُوسَعيديين: ابن رزيق حميد بن محمد بن رزيق النخلي (ت، 1291هـ)، تحقيق: محمد حبيب صالح، محمود بن مبارك السليمي، وزارة التراث القومي والثقافة، مسقط، ط6، (1437ه/ 2016م)، ‏1/‏441‏.

إِنَاض[مَصْدَر]

ن206هـ=821م

الإِنَاضُ : إِينَاعُ ثَمَرِ النَّخْلَةِ وَنُضْجُهُ.

"فَإِذَا أَرْطَبَ النَّخْلُ كُلُّهُ فَذَلِكَ المَعْوُ... وَإِذَا أَدْرَكَ حَمْلُ النَّخْلَةِ، فَهِيَ الإِنَاضُ "
أبو عمرو الشّيبانيّ
كتاب الجراثيم: ابن قتيبة (ت، 276هـ)، حققه: محمّد جاسم الحميدي، قدم له: مسعود بوبو، منشورات وزارة الثقافة، دمشق، (1997م)، ‏2/‏74‏.

إِنَاضَة[مَصْدَر]

ن216هـ=831م

الإِنَاضَةُ : الإِنَاضُ.

"وَإِذَا أَدْرَكَ حَمْلُ النَّخْلَةِ، فَهِيَ الإِنَاضَةُ "
الأصمعيّ
البُلْغة في شذور اللّغة: نشرها: أوغست هفنر، لويس شيخو، المطبعة الكاثوليكية، بيروت، ط2، (1914م)، ‏67‏.

مُنَوَّض[اسْمُ مَفْعُول]

ن231هـ=845م

المُنَوَّضُ بِالدَّمِ: المُضَرَّجُ بِهِ.

فِي غِيلِهِ جِيَفُ الرِّجَالِ كَأَنَّهُ بِالزَّعْفَرَانِ، مِنَ الدِّمَاءِ، مُنَوَّضُ
من إنشاد أبي عبد الله ابن الأعرابيّ اللُّغويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏12/‏70‏.

مُنْتَاض[اسْمُ فَاعِل]

ن283هـ=896م

المُنْتَاضُ : الذَّاهِبُ فِي البِلَادِ، الطَّالِبُ لِلْمَعْرُوفِ.

زُرْتُ ابْنَ يَحْيَى الَّذِي يُؤَمِّلُهُ كُلُّ أَجَبِّ السَّنَامِ مُنْتَاضِهْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏4/‏1378‏.

مَنُوض[اسْمُ مَفْعُول]

ن1227هـ=1811م

المَنُوضُ : المَتْرُوكُ المَنُوطُ.

"وَوَلِيَ بَعْدَهُ وَلَدُهُ أَبُو السَّعَادَاتِ أَحْمَدُ، وَعُمْرُهُ دُونَ سَنَتَيْنِ، وَكَانَ أَمْرُهُ مَنُوضًا إِلَى طَطْرَ ثُمَّ خَلَعَهُ طَطْرُ وَاسْتَقَلَّ بِالأَمْرِ تِلْكَ السَّنَةِ وَأَقَامَ ثَلَاثَةَ أَشْهُرٍ وَتُوُفِّيَ وَدُفِنَ بِجِوَارِ اللَّيْثِ بْنِ سَعْدٍ فِي القَرَّافَةِ"
عبد الله حجازي الشرقاوي
تحفة الناظرين فيمن ولي مصر من الملوك والسلاطين: عبد الله حجازي الشرقاوي (ت، 1227هـ)، تحقيق وتعليق: رحاب عبد الحميد القاري، مكتبة مدبولي، القاهرة، (1416ه/ 1996م)، ‏108‏.

نَيْض[مَصْدَر]

1270هـ=1853م

النَّيْضُ : الِاضْطِرَابُ وَالحَرَكَةُ.

"وَلَكِنَّ عُرُوقَ الإِنْسَانِ النَّابِضَةِ فِيهِ لَيْسَتْ فِي يَدِهِ وَقَلْبِهِ فَقَطْ بَلْ هِيَ فِي سَائِرِ أَعْضَائِهِ. فَيَنْبَغِي عَلَى هَذَا أَنْ نَجِسَّ كُلَّ عُضْوٍ فِينَا لِنَعْلَمَ أَيُّنَا أَكْثَرَ حَرَكَةً وَانْتِفَاضًا وَنَفَضَانًا وَنَبْضًا وَأَزًّا وَنَيْضًا وَأُزُوحًا وَحَبَضًا. إِذْ لَا يَصِحُّ الحُكْمُ عَلَى شَيْءِ إِلَّا بَعْدَ الِاسْتِقْرَاءِ وَالِاسْتِقْصَاءِ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏451‏.
