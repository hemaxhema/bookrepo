انْشَدَخَ[لَازِم]

قبل 50ق.هـ=573م

انْشَدَخَ الشَّيْءُ: انْكَسَرَ وَتَهَشَّمَ.

أَنَا الغُلَامُ ابْنُ الكِنَانِيِّ لَا بَذَخْ كَمْ مِنْ هِزَبْرٍ قَدْ رَآنِي فَانْشَدَخْ
ربيعة بن مُكَدَّم الفِراسيّ الكِنانيّ
مروج الذّهب ومعادن الجوهر: عليّ بن الحسين المسعوديّ (ت، 346هـ)، ج(1- 2)، تحقيق: محمّد محيي الدين عبد الحميد، دار الفكر، بيروت، ط5، (1393هـ/ 1973م)، ‏2/‏338‏.

شَدِيخ[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

الشَّدِيخُ : الَّذِي أُصِيبَ مَقْطَعُ عُنُقِهِ.

أَبَا الجُرَامِقِ، تَرْجُو أَنْ تَدِينَ لَكُمْ يَا بْنَ الشَّدِيخِ ، ضِبَاعٌ بَيْنَ أَجْبَاخِ
طَرَفة بن العبد البكريّ
ديوان طرفة بن العبد: تحقيق وتحليل ونقد: علي الجندي، مكتبة الأنجلو المصرية، القاهرة، (1378هـ/ 1958م)، ‏172‏.
ن1028هـ=1618م

الشَّدِيخُ مِنَ الأَشْيَاءِ: المُكَسَّرُ المُهَشَّمُ.

كَأَنَّكَ تَسْتَذْكِي بِهَا نَشْرَ عَنْبَرٍ شَدِيخٍ ومِسْكٍ فِي الكُؤُوسِ فَتِيتِ
أبوالبحر جعفر بن محمَّد الخطي
ديوان أبي البحر جعفر بن محمد الخطي: أبوالبحر جعفر بن محمَّد الخطي (ت، 1028هـ)، علق عليه وأخرجه: الخطيب علي بن الحسين الهاشمي، مطبعة الحيدري، طهران، (1373ه)، ‏13‏.

شَدِيخ[صِفَةٌ مُشَبَّهَة]

ن10هـ=631م

الشَّدِيخُ مِنَ الغُرَرِ: الَّتِي فَشَتْ فِي وَجْهِ الفَرَسِ مِنَ النَّاصِيَةِ إِلَى الأَنْفِ.

لَهُ رَعَثَاتٌ كَالشُّنُوفِ، وَغُرَّةٌ شَدِيخٌ ، وَلَوْنٌ كَالوَذِيلَةِ مُذْهَبُ
مُخارق بن شهاب المازنيّ التَّميميّ
عيون الأخبار: ابن قتيبة (ت، 276هـ)، تحقيق: منذر محمّد سعيد أبو شَعر، المكتب الإسلامي، بيروت، ط1، (1429هـ/ 2008م)، ‏02/‏97‏.

شَدِيخ[اسْم]

ن370هـ=980م

الشَّدِيخُ : مِنْ أَسْمَاءِ الأَسَدِ.

"وَالحَيْدَرَةُ، وَالخَيْتَعُورُ، وَالكَلْبُ، وَالسَّبْعُ، وَالسَّبُعُ، وَالعَمَيْثَلُ، وَالزَّأْرُ، وَالأَشْدَخُ، وَالشَّدِيخُ ، وَالشَّنْدَخُ"
ابن خالويه
أسماء الأسد: ابن خالَوَيْه (ت، 370هـ)، تحقيق: محمود جاسم الدرويش، مؤسسة الرسالة، بيروت، ط2، (1409هـ/ 1989م)، ‏12‏.

شَدُوخ[صِيغَةُ مُبَالَغَة]

قبل 13ق.هـ=609م

الشَّدُوخُ : الكَثِيرُ المَيْلِ وَالعُدُولِ عَنِ القَصْدِ.

عَنَنًا بَاطِلًا شَدُوخًا كَمَا تُعْـ ـتَرُ عَنْ حَجْرَةِ الرَّبِيضِ الظِّبَاءُ
الحارث بن حِلِّزَة اليَشْكريّ البكريّ
التّقفية في اللّغة: اليَمان بن أبي اليَمان البَنْدَنيجيّ (ت، 284هـ)، حققه: خليل إبراهيم العطية، مطبعة العاني، بغداد، (1976م)، ‏292‏.

شَدَخَ[مُتَعَدٍّ]

ن11هـ=632م

شَدَخَ الشَّيْءَ: شَجَّهُ وَكَسَّرَهُ.

"فَإِذَا رَجُلٌ مُسْتَلْقٍ عَلَى قَفَاهُ، وَرَجُلٌ قَائِمٌ بِيَدِهِ فِهْرٌ، أَوْ صَخْرَةٌ، فَيَشْدَخُ بِهَا رَأْسَهُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏33/‏335،،‏(رقم الحديث: 20165).
ن175هـ=791م

شَدَخَ الدِّمَاءَ وَنَحْوَهَا: أَهْدَرَهَا وَأَبْطَلَهَا.

"وَكَانَ حَكَمَ بَيْنَ خُزَاعَةَ وَقُصَيٍّ حِينَ اقْتَتَلُوا فِي أَمْرِ البَيْتِ، وَكَثُرَ القَتْلُ؛ فَشَدَخَ دِمَاءَ خُزَاعَةَ تَحْتَ قَدَمِهِ، أَيْ: أَبْطَلَهَا، وَقَضَى بِالبَيْتِ لِقُصَيٍّ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏167‏.

شَدَخَ[لَازِم]

ن69هـ=688م

شَدَخَتِ الغُرَّةُ: اتَّسَعَتْ وَفَشَتْ فِي وَجْهِ الفَرَسِ مِنَ النَّاصِيَةِ إِلَى الأَنْفِ.

شَدَخَتْ غُرَّةُ السَّوَابِقِ فِيهِمْ فِي وُجُوهٍ إِلَى اللِّمَامِ الجِعَادِ
ابن مُفَرِّغ الحِمْيريّ
ديوان يزيد بن مفرِّغ الحِميري (ت، 69هـ): جمعه وحققه: عبد القدوس أبو صالح، مؤسسة الرسالة، بيروت، ط2، (1402هـ/ 1982م)، ‏118‏.

شَدْخ[مَصْدَر]

ن40هـ=660م

الشَّدْخُ : الشَّجُّ وَالكَسْرُ.

"فَأَنَا أَبُو حَسَنٍ، قَاتِلُ جَدِّكَ وَأَخِيكَ وَخَالِكَ شَدْخًا يَوْمَ بَدْرٍ"
عليّ بن أبي طالب
شرح نهج البلاغة: ابن أبي الحَديد (ت، 656هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية، القاهرة، منشورات مكتبة آية الله العظمى المرعشي النجفي- مؤسسة إسماعيليان، قم، ط2، (1383هـ/ 1963م- 1378هـ/ 1967م)، ‏15/‏79‏.
ن204هـ=819م

شَدْخُ الدِّمَاءِ وَالدِّيَاتِ: إِبْطَالُهَا وَإِهْدَارُهَا.

"كَانَ سَبَبُ شَدْخِ يَعْمُرَ الدِّمَاءَ بَيْنَ قُرَيْشٍ وَخُزَاعَةَ... وَذَلِكَ أَنَّ خُزَاعَةَ أَخَذَتْ مِفْتَاحَ الكَعْبَةِ... وَحَكَّمُوا يَعْمُرَ بْنَ عَوْفٍ، فَقَالَ: قَضَيْتُ لِقُصَيٍّ بِالحِجَابَةِ، وَلِخُزَاعَةَ بِإِقَرَارِهِمْ بِالحَرَمِ وَأَنْ لَا يَخْرُجُوا مِنْهُ، وَقَدْ شَدَخْتُ الدِّمَاءَ، فَكَافَأَ بَيْنَهَا"
ابن الكلبيّ
جمهرة النّسب: ابن الكلبيّ (ت، 204هـ)، رواية: محمّد بن حَبيب، تحقيق: محمود فردوس العظم، قدم له: سهيل زكّار، دار اليقظة العربية، دمشق، ط2، (1983م)، ‏01/‏228‏.

شَدْخ[اسْم]

ن270هـ=883م

الشَّدْخُ : الرَّخْصُ الطَّرِيُّ مِنَ الشَّجَرِ.

"ثُمَّ أَمَرَ بِهِمْ فَشُقَّ عَنْ جَمَاعَتِهِمْ، وَأَخَذَتْهُمُ السِّيَاطُ حَتَّى سَقَطُوا، وَضُرِبُوا عَلَى رُؤُوسِهِمْ بِالشُّدُوخِ حَتَّى مَاتُوا جَمِيعًا"
نسيم
كتاب المكافأة وحُسْن العُقْبى: ابن الدّاية الكاتب (ت، 340هـ)، حققه وشرحه وصححه: محمود محمّد شاكر، مطبعة الاستقامة، القاهرة، ط1، (1359هـ/ 1940م)، ‏76‏.

مَشْدُوخ[اسْمُ مَفْعُول]

60هـ=680م

المَشْدُوخُ : المَشْجُوجُ المُكَسَّرُ.

"أَسْأَلُكُمْ بِاللهِ: أَتَعْلَمُونَ أَنَّ أَبِي حَوَارِيُّ رَسُولِ اللهِ  @  ... وَجَدِّيَ الصِّدِّيقُ، وَجَدَّهُ المَشْدُوخُ بِبَدْرٍ وَرَأْسُ الكُفْرِ؟"
عبد الله بن الزُّبير بن العَوّام
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏4/‏16‏.
ن175هـ=791م

المَشْدُوخُ مِنْ البُسْرِ: المُكَسَّرُ المُجَفَّفُ.

"وَالفَضِيخُ: شَرَابٌ يُتَّخَذُ مِنَ البُسْرِ المَفْضُوخِ، وَهُوَ المَشْدُوخُ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏178‏.
ن328هـ=940م

المَشْدُوخُ مِنَ الوُدِّ: المَقْطُوعُ غَيْرُ المَوْصُولِ.

وَاعْتَقِدْ مِنْ وُدِّ أَهْلِ الحِجَى كُلَّ وُدٍّ غَيْرَ مَشْدُوخِ
ابن عبد ربّه الأندلسيّ
ديوان ابن عبد ربّه: جمعه وحققه وشرحه: محمّد رضوان الداية، مؤسسة الرسالة، بيروت، ط1، (1399هـ/ 1979م)، ‏47‏.

شَدَّاخَة[صِيغَةُ مُبَالَغَة]

64هـ=684م

الشَّدَّاخَةُ : الشَّدِيدُ الشَّدْخِ.

إِنَّ لَنَا شَدَّاخَةً مُعَارِكَا قَرْمَ قُرُومٍ، صَلْهَبًا ضُبَارِكَا
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏126‏.

شَدَّاخَة[اسْمُ آلَة]

252هـ=866م

الشَّدَّاخَةُ : آلَةٌ حَرْبِيَّةٌ لِلتَّهْشِيمِ وَالتَّكْسِيرِ.

"وَجَعَلَ عَلَى بَابِ الشَّمَّاسِيَّةِ خَمْسَ شَدَّاخَاتٍ بِعَرْضِ الطَّرِيقِ؛ فِيهَا العَوَارِضُ وَالأَلْوَاحُ وَالمَسَامِيرُ الطِّوَالُ الظَّاهِرَةُ"
الطّبريّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏09/‏287‏-‏288‏.

شَدَخ[اسْم]

ن73هـ=692م

الشَّدَخُ : الجَنِينُ يُولَدُ لِغَيْرِ تَمَامٍ، وَلَا يَكُونُ إِلَّا سِقْطًا.

"إِذَا كَانَ شَدَخًا أَوْ مُضْغَةً فَادْفِنْهُ فِي بَيْتِكَ"
ابن عمر
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏07/‏76‏.

شَدَخ[صِفَةٌ مُشَبَّهَة]

ن216هـ=831م

الشَّدَخُ مِنَ الأَطْفَالِ: الصَّغِيرُ الَّذِي مَا يَزَالُ رَطْبًا.

"يُقَالُ لِلْمَوْلُودِ حِينَئِذٍ: وَلِيدٌ، ثُمَّ طِفْلٌ... وَيُقَالُ: طِفْلٌ وَطَفْلٌ، فَأَمَّا الطِّفْلُ فَهُوَ الصَّغِيرُ، وَأَمَّا الطَّفْلُ فَهُوَ الرَّخْصُ النَّاعِمُ، ثُمَّ شَدَخٌ إِذَا كَانَ صَغِيرًا رَطْبًا"
الأصمعيّ
كتاب خلق الإنسان: الأصمعيّ (ت، 216هـ)- ضمن: الكنز اللغوي في اللَّسَن العربي: سعى في نشره وتعليق حواشيه: أوغست هفنر، المطبعة الكاثوليكية، بيروت، (1903م)، ‏160‏.

مُشَدِّخ[اسْمُ فَاعِل]

90هـ=709م

المُشَدِّخُ : مَا يُسْتَعْمَلُ لِلتَّهْشِيمِ وَالتَّكْسِيرِ.

كُنَّا إِذَا نَزَلَتْ بِأَرْضِكَ حَيَّةٌ صَمَّاءُ تَخْرُجُ مِنْ صُدُوعِ جِبَالِ
