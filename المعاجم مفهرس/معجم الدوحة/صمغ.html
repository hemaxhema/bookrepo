صَمْغ[اسْمُ جِنْس]

قبل 50ق.هـ=573م

الصَّمْغُ : السَّائِلُ اللَّزِجُ يَنْضَحُهُ الشَّجَرُ وَيَسِيلُ مِنْهُ.

نَبِيتُ مَعَ الثَّعَالِبِ حَيْثُ بَاتَتْ وَنَجْعَلُ صَمْغَ عُرْفُطِهِنَّ زَادَا
مالك بن حَريم الهَمْدانيّ
شعر هَمْدان وأخبارها في الجاهلية والإسلام: جمع وتحقيق ودراسة: حسن عيسى أبو ياسين، دار العلوم، الرِّياض، ط1، (1403هـ/ 1983م)، ‏290‏.

صَمْغ[اسْم]

ن175هـ=791م

الصَّمْغُ : مَا يَتَيَبَّسُ فِي أَحَالِيلِ ضَرْعِ الشَّاةِ وَنَحْوِهَا.

" الصَّمْغُ : شَيْءٌ فِي أَحَالِيلِ ضَرْعِ الشَّاةِ، يَابِسٌ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏375‏.

صِمَاغ[اسْم]

ن40هـ=660م

صِمَاغُ الفَمِ: مُجْتَمَعُ الرِّيقِ فِي جَانِبِ الشَّفَةِ.

"نَظِّفُوا الصِّمَاغَيْنِ ؛ فَإِنَّهُمَا مَقْعَدَا المَلَكَيْنِ"
عليّ بن أبي طالب
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏2/‏131‏.

صِمْغ[اسْم]

ن175هـ=791م

الصِّمْغُ : مُلْتَقَى الشَّفَتَيْنِ مِمَّا يَلِي الشِّدْقَيْنِ.

" الصِّمْغَانِ : مُلْتَقَى الشَّفَتَيْنِ مِمَّا يَلِي الشِّدْقَيْنِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏375‏.

صَمَّغَ[لَازِم]

ن175هـ=791م

صَمَّغَ السَّائِلُ: ثَخُنَ وَلَزِجَ.

"وَالتَّعَيُّطُ: تَنَبُّعُ الشَّيْءِ مِنْ حَجَرٍ أَوْ عُودٍ يَخْرُجُ مِنْه شِبْهُ مَاءٍ، فيُصَمِّغُ أَوْ يَسِيلُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏2/‏211‏.

صَمَّغَ[مُتَعَدٍّ]

380هـ=990م

صَمَّغَ الشَّيْءَ: دَهَنَهُ بِالعَنْبَرِ وَنَحْوِهِ.

"وَنَحْنُ نَفْعَلُ بِطَاقَاتٍ كَثِيرَةً كَذَا، وَتُلَفُّ بَعْضُهَا عَلَى بَعْضٍ، ثُمَّ تُصَمَّغُ حَوَالَيْهَا بِشَيْءٍ مِنَ العَنْبَرِ"
الخِرقيّ أحَدُ بَني إسْحاقَ الشّيرازِيّ
نِشوار المحاضرة وأخبار المذاكرة: أبو عليّ التَّنوخيّ (ت، 384هـ)، تحقيق: عبّود الشالجي، دار صادر، بيروت، ط2، (1995م)، ‏1/‏294‏.

أَصْمَغَ[لَازِم]

ن175هـ=791م

أَصْمَغَ الشِّدْقُ: كَثُرَ بُصَاقُهُ.

" أَصْمَغَ شِدْقُهُ، أَيْ: كَثُرَ بُصَاقُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏375‏.
ن380هـ=990م

أَصْمَغَ الشَّيْءُ: خَرَجَ مِنْهُ الصَمْغُ وَالزَبَدُ.

"وَهُوَ حَارٌّ يَغْلِي، يَشْتَمُّهُ المَزْكُومُ، جَفَّفَ مَا يُخَدِّرُ مِنْ رَأْسِهِ إِلَى قَصَبَةِ رِئَتِهِ وَإِلَى مَنْخِرَيْهِ بِالحَرَارَةِ الَّتِي تَصِلُ إِلَى الدِّمَاغِ مِنْ مُقَاوَمَةِ اسْتِنْشَاقِهِ، فَكَذَلِكَ الخُيُوطُ الكَثِيرَةُ بِالأَلْوَانِ، وَخَاصَّةً إِذَا كَانَتْ مِنَ الأُرْجُوَانِ الَّذِي يَصْعَدُ مِنَ البَحْرِ وَيُصْمِغُ "
أحمد بن محمّد بن يحيى البلديّ
كتاب تدبير الحُبالى والأطفال والصّبيان وحفظِ صحتهم ومداواةِ الأمراض العارضة لهم: أحمد بن محمّد بن يحيى البلديّ (ت، 380هـ)، تحقيق: محمود الحاجّ قاسم محمّد، دار الرشيد، بغداد، (1980م)، ‏249‏-‏ 250‏.

أَصْمَغَ[مُتَعَدٍّ]

ن487هـ=1094م

أَصْمَغَ الدِّيبَاجَ وَنَحْوَهُ: طَلَاهُ بِالصَّمْغِ.

"وَفِيهِ أَشْجَارُ الصَّمْغِ الَّذِي يُجْلَبُ إِلَى الأَنْدَلُسِ، يُصْمَغُ بِهَا الدِّيبَاجُ"
أبو عُبيد البَكْريّ الأندلسيّ
كتاب المسالك والممالك: أبو عُبيد البَكْريّ الأندلسيّ (ت، 487هـ)، حققه وقدم له: أدريان فان ليوفن، أندري فيري، دار الغرب الإسلاميّ، بيروت، الدار العربية للكتاب- المؤسسة الوطنية للترجمة والتحقيق والدراسات، تونس، (1992م)، ‏2/‏848‏.
1393هـ=1972م

أَصْمَغَ الشَّيْءَ إِلَى غَيْرِهِ: أَلْصَقَهُ فِيهِ بِصَمْغٍ.

"وَيُصْبِحُ النَّسِيجُ رَقِيقًا وَمُتَّسِعًا، وَتُصْمَغُ قِطَعُ النَّسِيجِ إِلَى بَعْضِهَا بَعْدَ ذَلِكَ لِلْحُصُولِ عَلَى قِطْعَةِ نَسِيجٍ كَبِيرَةٍ يُمْكِنُ تَفْصِيلُهَا"
محمد رياض
الإنسان دراسة في النوع والحضارة: محمد رياض ، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2014م)، ‏362‏.

الصَّامِغُ : أَحَدُ جَانِبَيِ الفَمِ.

"زُبِّبَ شِدْقَاهُ، إِذَا اجْتَمَعَ الرِّيقُ فِي صَامِغَيْهِمَا "
ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏2/‏1000‏.

مُصَمَّغ[اسْمُ مَفْعُول]

ن313هـ=925م

المُصَمَّغُ مِنَ الأَشْيَاءِ: الَّذِي لُزِّجَ وَجُمِّدَ.

"مَطْبُوخُ اللِّفْتَةِ لِلسُّعَالِ وَالإِسْهَالِ:أَقْمَاعُ بَنَفْسَجٍ، لَا وَرَقُهُ، عَشْرَةُ دَرَاهِمَ تُرْبِذٍ مُصَمَّغٍ "
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏2/‏308‏.

تَصْمِيغ[مَصْدَر]

ن334هـ=945م

تَصْمِيغُ الشَّيْءِ: دَهْنُهُ بِالمِسْكِ وَنَحْوِهِ.

تَعَافُ الطِّرَارَ وَتَصْمِيغَهَا وَكَمْ طُرَّةٍ عَيْبُهَا صِمْغُهَا
الصَّنَوبريّ
ديوان الصّنوبريّ أحمد محمّد بن الحسن الضّبيّ: تحقيق: إحسان عبّاس، دار صادر، بيروت، ط2، (1998م)، ‏306‏.
517هـ=1123م

تَصْمِيغُ الدِّيبَاجِ وَنَحْوِهِ: تَصْقِيلُهُ وَتَلْبِيسُهُ بِالصَّمْغِ.

"وَهَذَا عَلَى مَا قَالَهُ ابْنُ حَبِيبٍ مِنْ أَنَّ مَا يَصْنَعُهُ حَاكَةُ الدِّيبَاجِ مِنْ تَصْمِيغِهَا غِشٌّ"
ابن رشد الجدّ
البيان والتحصيل والشرح والتوجيه والتعليل في مسائل المستخرجة: ابن رشد الجدّ (ت، 520هـ)، تحقيق: محمد حجي وآخرين، دار الغرب الإسلامي، بيروت، ط2، (1408ه/ 1988م)، ‏9/‏315‏.

مُتَصَمِّغ[صِفَةٌ مُشَبَّهَة]

ن380هـ=990م

المُتَصَمِّغُ مِنَ البُثُورِ وَنَحْوِهَا: الرَّطْبُ اللَّزِجُ.

"السَّعْفَةُ: بُثُورٌ تَحْدُثُ فِي الرَّأْسِ وَالوَجْهِ، مِنْهَا رَطْبَةٌ مُتَصَمِّغَةٌ ، وَمِنْهَا يَابِسَةٌ مَعَ خَشْكَرِيشَةٍ"
الحسن بن نوح القمريّ
التّنوير في الاصطلاحات الطّبّيّة: الحسن بن نوح القمري (ت، 380هـ)، تحقيق: غادة حسن الكرمي، مكتب التربية العربيّ لدول الخليج، الرياض، (1411هـ/ 1991م)، ‏61‏.

صِمْغَة[اسْم]

385هـ=995م

أَبُو صَمْغَةٍ : الَّذِي يُصْمِغُ فَمُهُ وَأُذُنُهُ وَعَيْنَاهُ وَأَنْفُهُ كَمَا يُصْمِغُ الشَّجَرُ.

مَصْمُوغ[اسْمُ مَفْعُول]

ن390هـ=1000م

صَمْغِيّ[صِفَة]
