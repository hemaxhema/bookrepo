أُعَيْمِش[صِفَة]

2هـ=623م

الأُعَيْمِشُ : تَصْغِيرُ الأَعْمَشِ.

"زَوَّجْتَنِيهِ أُعَيْمِشَ ، عَظِيمَ البَطْنِ"
فاطمة بنت محمّد رسول الله
كتاب المغازي: أبو عُروة مَعْمَر بن راشد البصريّ (ت، 153هـ)، رواية: عبد الرّزّاق بن هَمّام الصَّنْعانيّ (ت، 211هـ)، تحقيق: شون أنثوني، المكتبة العربيّة- مطبعة جامعة نيويورك، نيويورك، (2014م)، ‏141‏.

أَعْمَش[صِفَةٌ مُشَبَّهَة]

ن40هـ=660م

الأَعْمَشُ : الضَّعِيفُ الرُّؤْيَةِ، مَعَ سَيَلَانِ الدَّمْعِ.

"يُبْسُ الشِّفَاهِ مِنَ الظَّمَا، عُمْشُ العُيُونِ مِنَ البُكَا"
عليّ بن أبي طالب
أمالي المرتضى غُرر الفوائد ودُرر القلائد: الشّريف المرتضى (ت، 436هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، ط1، (1373هـ/ 1954م)، ‏1/‏18‏.
ن1409هـ=1988م

الأَعْمَشُ مِنَ الأَضْوَاءِ وَنَحْوِهَا: الضَّعِيفُ.

"كَانَ الظَّلَامُ كَثِيفًا أَصَمَّ، وَأَضْوَاءُ حَارَةِ المُهَاجِرِينَ عَمْشَاءَ ، تُوحِي بِمَكَانٍ مَهْجُورٍ"
غالب هلسا
سلطانة: غالب هلسا (ت، 1409هـ)، ضمن: الأعمال الروائية الكاملة، ج3، غالب هلسا، دار أزمنة- أمانة عمّان الكبرى، البنك الأهلي الأردني، عمّان، ط2، (2003م)، ‏356‏.

عُمَيْش[اسْمٌ مُصَغَّر]

ن110هـ=728م

العُمَيْشُ : تَصْغِيرُ الأَعْمَشِ.

"يَا عُمَيْشُ ، أَمَا تَعْلَمُ أَنَّ مَشْيَكَ فِي حَاجَةِ أَخِيكَ المُسْلِمِ، خَيْرٌ لَكَ مِنْ حَجَّةٍ بَعْدَ حَجَّةٍ؟!"
الحسن البصريّ
اصطناع المعروف: ابن أبي الدُّنيا (ت، 281هـ)، تحقيق: محمّد خير رمضان يوسف، دار ابن حزم، بيروت، ط1، (1422هـ/ 2002م)، ‏130‏.

عَمَش[اسْم]

ن125هـ=743م

العَمَشُ : وَسَخٌ تَبْذُلُهُ العَيْنُ بِسَبَبِ مَرَضٍ أَوْ تَعَبٍ وَنَحْوِهِمَا.

وَإِذَا الإِبْلُ مِنَ المَحْلِ غَدَتْ وَهْيَ فِي أَعْيُنِهَا مِثْلُ العَمَشْ
النّابغة الشَّيْبانيّ
ديوان نابغة بني شَيْبان: تحقيق: أحمد نسيم، مطبعة دار الكتب المصرية، القاهرة، ط1، (1351هـ/ 1932م)، ‏88‏.

عَمَش[مَصْدَر]

191هـ=807م

العَمَشُ : ضَعْفُ رُؤْيَةِ العَيْنِ.

"فَإِنْ أَصَابَهُ عَوَرٌ أَوْ عَمَشٌ أَوْ عَمًى؟"
سحنون
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏4/‏347‏.

عَمَّشَ[مُتَعَدٍّ]

ن140هـ=757م

عَمَّشَ عَيْنَهُ: أَضْعَفَ بَصَرَهَا، وَأَسَالَ دَمْعَهَا.

" عَمَّشَ اللهُ عَيْنَكَ"
صالح بن كَيسان المدنيّ
الموشَّح مآخذ العلماء على الشعراء في عدة أنواع من صناعة الشعر: أبو عُبيد الله المَرْزُبانيّ (ت، 384هـ)، تحقيق: علي محمّد البجاوي، نهضة مصر، القاهرة، د.ت. ‏44‏.

عَمِشَ[لَازِم]

ن148هـ=765م

عَمِشَتِ العَيْنُ: ضَعُفَ بَصَرُهَا، وَسَالَ دَمْعُهَا.

"قِيلَ لِلْأَعْمَشِ: مِمَّ عَمِشَتْ عَيْنَاكَ؟ قَالَ: مِنَ النَّظَرِ إِلَى الثُّقَلَاءِ"
سليمان بن مَهْران الأعمش
العزلة: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، حقّقه وعلّق عليه: ياسين محمّد السوّاس، دار ابن كثير، دمشق، ط2، (1410هـ/ 1990م)، ‏118‏.
ن198هـ=813م

عَمِشَ الشَّخْصُ: كَثُرَ سَيَلَانُ دَمْعِ عَيْنِهِ وَضَعُفَ إِبْصَارُهَا.

"قَدْ كَانَ عَمِشَ مِنَ البُكَاءِ"
سُفيان بن عُيَينة
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏8/‏456‏.
352هـ=963م

عَمِشَ جِسْمُ المَرِيضِ: ثَابَ إِلَيْهِ وَعَادَ.

" عَمِشَ جِسْمُ المَرِيضِ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏1/‏449‏.
ن899هـ=1493م

عَمِشَ الضَّوْءُ وَنَحْوُهُ: ضَعُفَ.

يَجْلِي بِهَا لِلْغَمِّ مِنْهَا حِنْدِسًا قَمَرٌ، عَلَيْهِ مِنَ الذُّؤَابَةِ حِنْدِسُ
حَتَّى إِذَا عَمِشَتْ مِرَاةُ البَدْرِ مِنْ صُبْحٍ، بَدَا تِلْقَاءَهُ يَتَنَفَّسُ
نَادَيْتُهُ، وَسَنَا الصَّبَاحِ مُحَصْحِصٌ يَنْجَابُ عَنْهُ مِنَ الظَّلَامِ مُعَسْعِسُ
محمّد بن علي بن محمّد الأصبحيّ الأندلسيّ
أزهار الرياض في أخبار عياض: أحمد بن محمد المقريّ التلسمانيّ (ت، 1041هـ)، ضبطه وحققه وعلق عليه: مصطفى السقا وآخرون، المعهد الخليفي للأبحاث المغربية- مطبعة فضالة، الرباط، (1398ه/ 1978م- 1400ه/ 1980م)، ‏3/‏320‏.

عَمِشَ[مُتَعَدٍّ بِالحَرْف]

ن428هـ=1037م

عَمِشَ عَنْهُ: تَغَافَلَ عَنْهُ، وَسَهَا.

"كَرِهْنَا أَنْ يَقِفَ الجُهَّالُ عَلَى مُخَالَفَةِ مَا هُوَ عِنْدَهُمْ مِنَ الشُّهْرَةِ بِحَيْثُ لَا يَشُكُّونَ فِيهِ، وَيَشُكُّونَ فِي النَّهَارِ الوَاضِحِ، وَبَعْضُهُ قَدْ كَانَ مِنَ الدِّقَّةِ بِحَيْثُ تَعْمَشُ عَنْهُ عُيُونُ عُقُولِ هَؤُلَاءِ الَّذِينَ فِي العَصْرِ "
ابن سينا
منطق المشرقيّين والقصيدة المزدوجة في المنطق: ابن سينا (ت، 428هـ)، منشورات مكتبة آية الله العظمى المرعشي النجفي- مطبعة الولاية، قم، ط2، (1405هـ)، ‏3‏.

عُمُوشَة[مَصْدَر]

ن148هـ=765م

عُمُوشَةُ العَيْنِ: عَمَشُهَا.

"لَا يَغُرَّنَّكِ عُمُوشَةُ عَيْنَيْه، وَلَا حُمُوشَةُ سَاقَيْه"
أبو البلاد الأعمى النّحْوي
المنتظم في تاريخ الملوك والأمم: ابن الجوزيّ (ت، 597هـ)، دراسة وتحقيق: محمد عبد القادر عطا، مصطفى عبد القادر عطا، راجعه وصححه: نعيم زرزور، دار الكتب العلمية، بيروت، ط2، (1415ه/ 1995م)، ‏8/‏114‏.

عَمْش[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

العَمْشُ مِنَ الأَطْعِمَةِ وَنَحْوِهَا: الصَّالِحُ لِلْبَدَنِ، المُوَافِقُ لَهُ.

"وَطَعَامٌ عَمْشٌ لَكَ: أَيْ مُوَافِقٌ صَالِحٌ، وَالعَمْشُ : مَا يَكُونُ فِيهِ صَلَاحٌ لِلْبَدَنِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏267‏.

عَمْش[مَصْدَر]

ن175هـ=791م

العَمْشُ : الإِصْلَاحُ وَإِزَالَةُ الفَسَادِ.

"وَالخِتَانُ عَمْشٌ لِلْغُلَامِ؛ لِأَنَّهُ يُرَى فِيهِ بَعْدَ ذَلِكَ زِيَادَةٌ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏367‏.

أَعْمَشَ[مُتَعَدٍّ]

ن243هـ=857م

أَعْمَشَ العَيْنَ: أَفْسَدَهَا، وَأَضْعَفَ رُؤْيَتَهَا.

"عَيْنٌ عَمْشَاءُ؛ إِذَا احْمَرَّتْ آمَاقُهَا وَحَمَالِيقُهَا، وَتَرَطَّبَتْ، وَتَنَاثَرَ شَعْرُ أَجْفَانِهَا، وَعَلَامَتُهَا: رُطُوبَةٌ لَزِجَةٌ بَوْرَقِيَّةٌ مَالِحَةٌ، فَأَكَّلَتْ لَحْمَ أَجْفَانِهَا، فَأَعْمَشَتْهُ "
ابن ماسويه
معرفة مِحْنة الكَحّالين: ابن ماسويه (ت، 243هـ)، تحقيق ودراسة: نشأت الحمارنة، اكتمال رجب، منشورات مجمع اللغة العربيّة الأردنيّ، ط1، (1440هـ/ 2018م)، ‏327‏-‏ 328‏.

عُمْشُوش[اسْم]

ن255هـ=869م

العُمْشُوشُ : العُنْقُودُ يُؤْكَلُ مَا عَلَيْهِ، وَيُتْرَكُ بَعْضُهُ.

"وَيُقَالُ لِمَا بَقِيَ مِنَ الثَّفَارِيقِ؛ يَعْنِي العَمَاشِيشَ إِذَا ضُرِبَتْ بِالخَشَبِ مِنَ الزَّبِيبِ أَوِالحَشَفِ أَوِالحَمْنَانِ: الحُفَالُ"
أبو حاتم السِّجِسْتانيّ
البُلْغة في شذور اللّغة: نشرها: أوغست هفنر، لويس شيخو، المطبعة الكاثوليكية، بيروت، ط2، (1914م)، ‏75‏.

إِعْمَاش[مَصْدَر]

ن283هـ=896م

إِعْمَاشُ العَيْنِ: إِضْعَافُ بَصَرِهَا وَإِسَالَةُ دَمْعِهَا.

لَمْ تَتْرُكِ البَغْيَ يَا حُذَيْفَتَهُ حَتَّى أَظَلَّتْكَ خَيْلُ قِرْوَاشِ
وَأَلْتَ جَهْلًا مِنَ المَرَاحِ إِلَى هَيْجَاءَ لَيْسَتْ بِذَاتِ إِفْرَاشِ
كَفَاقِئٍ عَيْنَهُ مُوَاءَلَةً مِنْ عَائِرٍ نَالَهَا بِإِعْمَاشِ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏1251 ‏-‏1252‏.

تَعَمُّش[مَصْدَر]

ن291هـ=904م

التَّعَمُّشُ : التَّغَافُلُ.

أُغَمِّضُ عَيْنِي عَنْ صَدِيقِي تَعَمُّشًا كَأَنِّي بِمَا يَأْتِي مِنَ الأَمْرِ جَاهِلُ
من إنشاد أبي العبّاس ثعلب
صُبابة المعاني وصبّابة المعاني: أبو بكر محمَّد محيي الدّين السَّلطي (ت، 1107هـ)، تحقيق: فاطمة بنت محمد السُّويدي، مركز الملك فيصل للبحوث والدّراسات الإسلاميَّة، الرياض، ط1، (1426ه/ 2005م)، ‏196‏.
ن1085هـ=1674م

التَّعَمُّشُ : النَّظَرُ بِبَصَرٍ ضَعِيفٍ.

"المُتَعَمِّشُ تَعَمُّشًا مَا فِي دِينِهِ، مُتَعَلِّقًا بِحِزْبِنَا، وَكَانَ مِنْ جُمْلَةِ فُقَرَائِنَا، خَيْرٌ مِنَ الحَازِمِ المُتَعَلِّقِ بِغَيْرِنَا"
ابن ناصر الدرعي
الأجوبة الناصرية في بعض مسائل البادية: ابن ناصر الدرعي (ت، 1085هـ)، اعتنى به: أبو الفضل الدمياطي أحمد بن علي، مركز التراث الثقافي المغربي، الدار البيضاء، دار ابن حزم، بيروت، ط1، (1433ه/ 2012م)، ‏192‏.

عَمَشَ[مُتَعَدٍّ]

352هـ=963م

عَمَشَ الوَلَدَ: خَتَنَهُ، وَطَهَّرَهُ.

"يُقَالُ: الخِتَانُ صَلَاحٌ لِلْوَلَدِ، فَاعْمِشُوهُ "
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏1/‏442‏.

" وَعَمَشْتُهُ بِالعَصَا"
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏1/‏297‏.

تَعَامَشَ[مُتَعَدٍّ بِالحَرْف]

436هـ=1044م

تَعَامَشَ عَنِ الأَمْرِ: تَغَافَلَ عَنْهُ.

"وَقِيلَ لِلَّذِي يَتَعَامَشُ عَنِ الأَمْرِ كَأَنَّهُ لَمْ يَشْعُرْ بِهِ: هُوَ يَتَعَاشَى"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
كتاب المخصَّص: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، المطبعة الكبرى الأميرية، بولاق، ط1، (1316- 1321هـ)، ‏11/‏31‏.

عُمَاش[اسْم]

ن620هـ=1223م

العُمَاشُ : العَمَشُ.

"أَتُنَاظِرُ بِعُمَاشِكَ عُيُونَ المِلَاح؟! مَا أَنْتِ يَا عُيُونَ النَّرْجِسِ إِلَّا وَقَاح!"
أبو الحسن، علاء الدين، علي بن محمد بن أبي بكر بن ‌مشرف ‌المارديني
كوكب الرّوضة في تاريخ النّيل وجزيرة الرّوضة: جلال الدّين السَّيوطي (ت، 911هـ)، تحقيق: محمَّد الشّشتاوي، دار الآفاق العربية، القاهرة، ط1، (1422ه/ 2002م)، ‏432‏.

عُمَاش[مَصْدَر]

ن911هـ=1505م

العُمَاشُ : ضَعْفُ رُؤْيَةِ العَيْنِ مَعَ سَيَلَانِ دَمْعِهَا.

"كُحْلٌ عَجِيبٌ نَافِعٌ لِلْجَرَبِ، وَالبَيَاضِ القَدِيمِ وَالحَادِثِ، وَالدَّمْعَةِ، وَالضَّبَابَةِ، وَالعُمَاشِ ، وَالحَرَارَةِ، صَحِيحٌ مُجَرَّبٌ"
جلال الدّين السَّيوطي
الرحمة في الطب والحكمة: جلال الدّين السَّيوطي (ت، 911هـ)، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، (د.ت)، ‏60‏.

انْعَمَشَ[لَازِم]

ن656هـ=1258م

انْعَمَشَتِ العَيْنُ: عَمِشَتْ.

قَدْ كُنْتُ، مِنْ قَبْلُ، فِي أَمْنٍ وَفِي دَعَةٍ طَرْفِي يَرُودُ لِقَلْبِي رَوْضَةَ الأَدَبِ
حَتَّى تَلَقَّبْتُ نُورَ الدِّينِ، فَانْعَمَشَتْ عَيْنِي، وَحُوِّلَ ذَاكَ النُّورُ لِلَّقَبِ
النور الأسعردي
كتاب الوافي بالوفيات: صلاح الدّين الصّفديّ (ت، 764هـ)، تحقيق واعتناء: أحمد الأرناؤوط، تركي مصطفى، دار إحياء التراث العربي، بيروت، ط1، (1420ه/ 2000م)، ‏1/‏155‏.

تَعَامُش[مَصْدَر]

ن743هـ=1342م

التَّعَامُشُ : التَّعَافِي وَالصَّلَاحُ بَعْدَ الفَسَادِ.

"وَهِيَ الَّتِي تَشْغَلُهُ بِلَوَازِمِ دُنْيَاهُ، عَنِ الِاسْتِعْدَادِ فِي دَارِ التَّحْصِيلِ لِآخِرَتِهِ... وَتُبْعِدُهُ عَنْ مَوَاهِبِ الفَتْحِ وَالذَّوْقِ وَالكَشْفِ، لِلتَّعَامُشِ بِأَنْوَارِ الحَقِّ وَخُضْرَتِهِ"
عزّ الدّين الجِلْدَكي
غاية السرور في شرح ديوان الشذور: عزّ الدّين الجِلْدَكي (ت، 743هـ)، تحقيق: عبد العزيز عبد الرحمن سعد آل سعد، الدار العربية للعلوم ناشرون، بيروت، ط1، (1439ه/ 2018م)، ‏59‏.

مُعَمَّش[اسْمُ مَفْعُول]

ن854هـ=1450م

المُعَمَّشُ مِنَ العُيُونِ: المُصَابُ بِالعَمَشَ.

"يُدْعَى (بَابَا تَرْمَش)، بِطَرْفٍ مُعَمَّش ، وَوَجْهٍ مُنَمَّش، وَصُورَةٍ قَبِيحَة، وَسِيرَةٍ غَيْرِ مَلِيحَة"
ابن عرب شاه الدمشقي
عجائب المقدور في نوائب تيمور: ابن عرب شاه الدمشقي (ت، 854هـ)، تحقيق: أحمد فايز الحمصي، مؤسسة الرسالة، بيروت، ط1، (1407ه/ 1986م)، ‏441‏.

مُعَمَّش[صِفَةٌ مُشَبَّهَة]

1270هـ=1853م

المُعَمَّشُ مِنَ العُيُونِ: ذَاتُ الرُّؤْيَةِ القَاصِرَةِ.

"كَانَ وُصُولُ الفَارِيَاقِ إِلَى هَذِهِ المَدِينَةِ الشَّهِيرَةِ، فِي لَيْلَةٍ ذَاتِ ضَبَابٍ، فَكَانَتْ عَيْنَاهُ مُعَمَّشَتَيْنِ عَنْ رُؤْيَةِ مَا فِيهَا مِنَ الخَصَائِصِ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏633‏.

عُمُوش[مَصْدَر]

ن1069هـ=1658م

العُمُوشُ : العَمَشُ.

"وَأَمَّا البَلْغَمُ، فَيَنْشَأُ عَنْهُ الدُّوَارُ، وَالشَّقِيقَةُ اليُمْنَى، وَغُبَارُ العَيْنِ، وَعُمُوشُهَا الرَّطْبُ، بِلَا حُمْرَةٍ"
شهاب الدين أحمد بن أحمد القليوبي
تذكرة القليوبي في الطب والحكمة: شهاب الدين أحمد بن أحمد القليوبي (ت، 1069هـ)، تحقيق: أحمد فريد المزيدي، منى شلبي، دار الكتب العلمية، بيروت، ط1، (1422ه/ 2001م)، ‏9‏.

مُتَعَمِّش[اسْمُ فَاعِل]

ن1085هـ=1674م

المُتَعَمِّشُ : النَّاظِرُ بِبَصَرٍ ضَعِيفٍ.

" المُتَعَمِّشُ تَعَمُّشًا مَا فِي دِينِهِ، مُتَعَلِّقًا بِحِزْبِنَا، وَكَانَ مِنْ جُمْلَةِ فُقَرَائِنَا، خَيْرٌ مِنَ الحَازِمِ المُتَعَلِّقِ بِغَيْرِنَا"
ابن ناصر الدرعي
الأجوبة الناصرية في بعض مسائل البادية: ابن ناصر الدرعي (ت، 1085هـ)، اعتنى به: أبو الفضل الدمياطي أحمد بن علي، مركز التراث الثقافي المغربي، الدار البيضاء، دار ابن حزم، بيروت، ط1، (1433ه/ 2012م)، ‏192‏.

تَعْمِيش[مَصْدَر]

1270هـ=1853م

تَعْمِيشُ العَيْنِ: ضَعْفُ بَصَرِهَا.

"قُلْتُ: أَيُّ نَفْعٍ فِي تَعْمِيشِ العَيْنَيْنِ؟ قَالَتْ: عَدَمُ رُؤْيَةِ الحِسَانِ لَيْلًا، فَإِنَّهُنَّ أَرْوَعُ فِيهِ وَأَفْتَنُ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏555‏.
