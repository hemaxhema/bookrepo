حَرْشَف[اسْمُ جِنْس]

قبل 150ق.هـ=476م

الحَرْشَفُ : الرَّجَّالَةُ مِنَ الجَيْشِ.

لَاقَى جَذِيمَةَ فِي جَأْوَاءَ مُشْعَلَةٍ فِيهَا حَرَاشِفُ بِالنِّيرَانِ تَرْتَشِقُ
منسوب إلى الأعور بن عمرو الأزديّ
شُعراء عُمان في الجاهلية وصدر الإسلام: جمع وتحقيق: أحمد محمّد عبيد، المجمع الثقافي، أبو ظبي، (1420هـ/ 2000م)، ‏42‏.
قبل 50ق.هـ=573م

الحَرْشَفُ : الجَرَادُ الكَثِيرُ.

وَغَارَةٍ قَدْ تَلَبَّبْتُ بِهَا كَأَنَّ أَسْرَابَهَا الرِّعَالُ
كَأَنَّهُمْ حَرْشَفٌ مَبْثُوثٌ بِالجَوِّ إذْ تَبْرُقُ النِّعَالُ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏192‏-‏193‏.
قبل 13ق.هـ=609م

حَرْشَفُ الدِّرْعِ وَنَحْوِهَا: حُبُكُهَا.

وَنَأْوِي إِلَى مَلْمُومَةٍ ذَاتِ حَرْشَفٍ تَقُودُ المَنَايَا، وَالمَنَايَا تَقُودُهَا
ثَوب بن سُحمة بن جُندب
كتاب جُمَل من أنساب الأشراف: أحمد بن يحيى البَلاذُري (ت، 279هـ)، ج(2- 13)، حققه وقدم له: سهيل زكّار، رياض زركلي، إشراف: مكتب البحوث والدراسات، دار الفكر، بيروت، ط1، (1417هـ/ 1996م)، ‏13/‏28‏.
ن175هـ=791م

الحَرْشَفُ : فُلُوسُ السَّمَكَةِ، وَهُوَ مَا يَلْمَعُ عَلَى جِلْدِهَا مِنَ القُشُورِ الرَّقِيقَةِ.

" الحَرْشَفُ : فُلُوسُ السَّمَكَةِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏330‏.
ن175هـ=791م

حَرْشَفُ السِّلَاحِ: مَا زُيِّنَ بِهِ.

" وَحَرْشَفُ السِّلَاحِ : مَا زُيِّنَ بِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏330‏.
ن200هـ=815م

الحَرْشَفُ : نَبَاتٌ شَائِكٌ عَرِيضُ الوَرَقِ، وَلَهُ زَهْرَةٌ حَمْرَاءُ، يُتَّخَذُ مِنْهُ صَمْغٌ، وَدَوَاءٌ.

"فَأَمَّا الصُّمُوغُ، فِبَأَنْ تُضَافَ إِلَى الأَسْمَاءِ الَّتِي هِيَ صُمُوغٌ لَهَا، مِثْلُ صَمْغِ السَّذَابِ، وَصَمْغِ اللَّوْزِ، وَالحَرْشَفِ ، وَمَا جَرَى مَجْرَاهَا"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏44‏.

حَرْشَف[اسْم]

206هـ=821م

الحَرْشَفُ : العَدَدُ الكَثِيرُ مِنَ الرِّجَالِ.

وَحَرْشَفٍ مِنَ الرِّجَالِ جُرْبِ
من إنشاد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏207‏.
ن1257هـ=1840م

الحَرْشَفُ : الطَّبَقَةُ الظَّاهِرَةُ الَّتِي تُغَلِّفُ بَعْضَ النَّبَاتِ وَالزُّهُورِ.

"الجِنْسُ الثَّالِثُ الثُّومِيُّ، رُؤُوسُ زَهْرِهِ عَلَى هَيْئَةِ حُزَمٍ كُرَوِيَّةٍ أَوْ صِيوَانِيَّةٍ، وَزَهْرُهُ مُنْحَصِرٌ قَبْلَ ابْتِسَامِهِ فِي كُوزٍ مُكَوَّنٍ مِنْ حَرْشَفَيْنِ غِشَائِيَّيْنِ، لِكُلِّ حَرْشَفٍ مِنْهُمَا سِتَّةُ أَقْسَامٍ مُسْتَطِيلَةٍ "
محمد بن عمر التونسي
الدر اللامع في النبات ومافيه من الخواص والمنافع: محمد بن عمر التونسي (ت، 1274هـ)، مطبعة بولاق، القاهرة، (1257ه)، ‏127‏.

حِرْشَاف[اسْمُ جِنْس]

6هـ=627م

الحِرْشَافَةُ : الوَاحِدَةُ مِنَ الحِرْشَافِ، وَهُوَ الجَرَادُ المَهْزُولُ الكَثِيرُ الأَكْلِ.

"وَاللهِ، يَا رَسُولَ اللهِ، مَا يَقْدِرُ عَلَى ذَلِكَ، إِنَّهُ لَيَشْرَبُ فِي اليَوْمِ كَذَا كَذَا مَرَّةً، قَدْ ذَهَبَ بَصَرُهُ مَعَ ضَعْفِ بَدَنِهِ، وَإِنَّمَا هُوَ كَالحِرْشَافَةِ "
خولة بنت ثعلبة الأنصاريّة
حياة الحيوان الكبرى: كمال الدين محمد بن موسى الدَّميري (ت، 808هـ)، عني بتحقيقه: إبراهيم صالح، دار البشائر، دمشق، ط1، (1426ه/ 2005م)، ‏2/‏31‏.

حَرْشَفَة[اسْمُ جَمْع]

ن175هـ=791م

الحَرْشَفَةُ مِنَ الجَيْشِ: الكَتِيبَةُ.

" وَحَرْشَفَةٌ مِنَ الجَيْشِ: كَتِيبَةٌ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏330‏.

حَرْشَفَة[صِفَة]

206هـ=821م

الحَرْشَفَةُ مِنَ النَّاسِ: صَاحِبُ الشَّرِّ.

"وَيُقَالَ: إِنَّهُ لَحَرْشَفَةُ شَرٍّ"
من رواية عمرو بن أبي عمرو الشَّيْبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏148‏.

حَرْشَفَة[اسْم]

ن275هـ=888م

الحَرْشَفَةُ : الأَرْضُ الغَلِيظَةُ.

"وَهِيَ حَرْشَفَةٌ حَرَّةٌ سَوْدَاءُ لَا تُنْبِتُ شَيْئًا"
عَرّام بن الأصبغ السُّلَميّ
كتاب أسماء جبال تهامة وسكّانها وما فيها من القرى وما ينبت عليها من الأشجار وما فيها من المياه: عَرّام بن الأصبغ السُّلَميّ (ت، 275هـ)، تحقيق: عبد السلام هارون، عني بنشره: يوسف زينل، محمّد نصيف، مطبعة أمين عبد الرحمن، القاهرة، ط1، (1373هـ)، ‏76‏.

حَرْشَفِيّ[صِفَة]

646هـ=1248م

الحَرْشَفِيُّ مِنَ النَّبَاتِ: الَّذِي تُوجَدُ عَلَيْهِ طَبَقَةٌ رَقِيقَةٌ خَشِنَةٌ مِنَ القُشُورِ المُتَرَاكِمَةِ.

"العَكُّوبُ تَأْكُلُهُ النَّاسُ بِالشَّامِ وَغَيْرِهَا، وَهُوَ نَوْعٌ مِنَ الشَّوْكِ الَّذِي تَرْتَعِيهِ الإِبِلُ... وَقَدْ يُثْمِرُ فِي رَأْسِ قَضِيبِهِ ثَمَرَةٌ مُسْتَدِيرَةٌ إِلَى الطُّولِ مَا هِيَ، حَرْشَفِيَّةٌ مُلْتَبِسَةٌ بِشَوْكٍ أَمْثَالِ مَا دَقَّ مِنَ الإِبَرِ"
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏3/‏265‏.
ن1260هـ=1843م

الحَرْشَفِيُّ مِنَ الجُذَامِ: الَّذِي لَهُ قُشُورٌ مُسْتَدِيرَةٌ عَرِيضَةٌ.

"أَمَّا الجُذَامُ الحَرْشَفِيُّ المُسَمَّى بِجُذَامِ اليُونَانِيِّينَ وَجُذَامِ العِبْرَانِيِّينَ، فَصِفَتُهُ أَنَّهُ فُلُوسٌ عَرِيضَةٌ مُسْتَدِيرَةٌ"
يوحنّا عَنْحُوري
كتاب منتهى الأغراض في علم الأمراض: يوحنّا عَنْحُوري (ت، 1260هـ)، صححه: محمد الهراوي، المطبعة الأميرية، (1834م)، ‏1/‏266‏.

حَرْشَفِيّ[اسْمٌ مَنْسُوب]

ن694هـ=1294م

الحَرْشَفِيُّ : المَنْسُوبُ إِلَى الحَرْشَفِ.

"مُسْتَعْجِلَةٌ: هُوَ نَبَاتٌ مَشْهُورٌ بِالدِّيَارِ المِصْرِيَّةِ جِدًّا، يَنْبُتُ بِظَاهِرِ الإِسْكَنْدَرِيَّةِ... وَرَقُهُ يُشْبِهُ وَرَقَ الطَّرَخْشَقُوقِ، حَرْشَفِيُّ الطَّعْمِ"
المُظَفَّر الرسوليّ
المعتمد في الأدوية المفردة: المُظَفَّر الرسوليّ (ت، 694هـ)، صححه وفهرسه: مصطفى السقا، دار المعرفة، بيروت، (1402ه/ 1982م)، ‏498‏.

مُحَرْشَف[اسْمُ مَفْعُول]

ن1369هـ=1949م

المُحَرْشَفُ مِنَ الحِيتَانِ وَنَحْوِهَا: الَّتِي لَهَا حَرَاشِيفُ.

"وَالبَرْمَائِيَّاتُ الحَالِيَّةُ، عَلَى وَجْهٍ عَامٍّ، مُلْسُ الجُسُومِ، فَاقِدَةُ الحَرَاشِفِ، وَلَكِنْ، كُلُّ عَشَائِرِهَا الأُولَى كَانَتْ مُحَرْشَفَةً "
إسماعيل مظهر
الحيتان: إسماعيل مظهر (ت، 1382هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2020م)، ‏30‏.

تَحَرْشَفَ[لَازِم]

1400هـ=1979م

تَحَرْشَفَ الشَّيْءُ عَلَى الجِسْمِ: كَوَّنَ عَلَيْهِ طَبَقَةً مِثْلَ الحَرْشَفِ.

" وَتَتَحَرْشَفُ قِطَعٌ مِنَ المَرَارَةِ عَلَى الجِسْمِ الدَّاخِلِيِّ لِلْعُمُرِ الَّذِي لَمْ يَمْضِ مِنْهُ أَرْبَعُونَ عَامًا"
محمد ملص
إعلانات عن مدينة كانت تعيش قبل الحرب، محمّد ملص، إصدار، جامعة متشغن، النّاشر، دار ابن رُشد، 1979م، ‏108‏.
1429هـ=2007م

تَحَرْشَفَ الجِلْدُ وَنَحْوُهُ: صَارَ خَشِنًا ذَا قُشُورٍ.

"كَانَتْ دَقِيقَةً وَمُنَمْنَمَةً، كَأَصَابِعِ طِفْلَةٍ لَمْ تَتَجَلَّدْ، وَلَمْ تَتَحَرْشَفْ بَعْدُ"
حُزامة حبايب
أصل الهوى: حُزامة حبايب، المؤسسة العربية للدراسات والنّشر، بيروت، الطّبعة الثّانية، 2009م، ‏17‏.

مُتَحَرْشِف[اسْمُ فَاعِل]

1409هـ=1988م

المُتَحَرْشِفُ مِنَ الجُلُودِ وَنَحْوِهَا: الَّذِي خَشُنَ وَتَقَشَّرَ.

"زُمْلَةُ الضَّعْفِ المِنْسَلِيِّ الحَرْشَفِيِّ اضْطِرَابٌ مُتَنَحٍّ يَرْتَبِطُ بِالصِّبْغِيِّ ( س أو X )، وَيَتَمَيَّزُ بِالأَعْضَاءِ التَّنَاسُلِيَّةِ الصَّغِيرَةِ، وَبِالجِلْدِ المُتَحَرْشِفِ فِي مَنَاطِقَ عَدِيدَةٍ مِنَ الجِسْمِ"
علاء الدّين كفافي
معجم علم النّفس والطّبّ النّفسيّ، إنجليزي -عربيّ: جابر عبد الحميد جابر، علاء الدّين كفافي، دار النّهضة العربيّة، القاهرة، 1988- 1996م، ‏4/‏؟‏.

حُرْشُوفِيّ[صِفَة]

1417هـ=1996م

الحُرْشُوفِيُّ: الخَشِنُ المُتَقَشِّرُ.

"يَظْهَرُ عَلَى النَّبَاتَاتِ المُصَابَةِ الحَسَّاسَةِ لِلْمَرَضِ تَشَقُّقَاتٌ عَمُودِيَّةٌ فِي القِلْفِ... وَالَّتِي تُعْطِي القِلْفَ مَظْهَرَ التَّقَرُّحِ، أَوِ المَظْهَرَ الحُرْشُوفِيَّ "
محمود موسى أبو عرقوب
الفيرويد والأمراض الفيرودية، محمود موسى أبو عرقوب، المكتبة الأكاديمية، 1996م، ‏323‏.

تَحَرْشُف[مَصْدَر]

1436هـ=2014م

تَحَرْشُفُ الجِلْدِ وَنَحْوِهِ: خُشُونَتُهُ وَتَقَشُّرُهُ.

"تَغَيُّرٌ فِي سَطْحِ الحَلَمَةِ، وَظُهُورُ الإِكْزِيمَا، وَتَحَرْشُفُهَا "
هنادي مزبودي
سرطان الثّدي: مايك ديكسون، ترجمة: هنادي مزبودي، مدينة الملك عبد العزيز للعلوم والتّقنيّة، الرياض، ط1، 1434هـ/ 2013م، ‏37‏.
