ثُكْل[مَصْدَر]

قبل 100ق.هـ=525م

الثُّكْلُ : المَوْتُ وَالهَلَاكُ.

ذَهَبَ الصُّلْحُ، أَوْ تَرُدُّوا كُلَيْبًا أَوْ أُذِيقَ الغَدَاةَ شَيْبَانَ ثُكْلَا
مُهَلْهِل بن ربيعة التَّغْلِبيّ
ديوان المُهَلْهِل: شرح وتحقيق: أنطوان محسن القوَّال، دار الجيل، بيروت، ط1، (1415هـ/ 1995م)، ‏63‏.
قبل 100ق.هـ=525م

الثُّكْلُ : الفَجْعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

يَشْتَفِي المُدْرِكُ بِالثَّأْرِ وَفِي دَرَكِي ثَأْرِيَ ثُكْلُ المُثْكِلِ
جليلة بنت مرّة الشَّيْبانيّة
شاعرات العرب في الجاهلية والإسلام: جمعه ورتبه ووقف على طبعه: بشير يموت، المكتبة الأهلية- المطبعة الوطنية، بيروت، ط1، (1353هـ/ 1934م)، ‏38‏.

مُثْكِل[اسْمُ فَاعِل]

قبل 100ق.هـ=525م

المُثْكِلُ مِنَ الأَشْخَاصِ: الَّذِي فُجِعَ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

يَشْتَفِي المُدْرِكُ بِالثَّأْرِ، وَفِي دَرَكِي ثَأْرِيَ ثُكْلُ المُثْكِلِ
جليلة بنت مرّة الشَّيْبانيّة
شاعرات العرب في الجاهلية والإسلام: جمعه ورتبه ووقف على طبعه: بشير يموت، المكتبة الأهلية- المطبعة الوطنية، بيروت، ط1، (1353هـ/ 1934م)، ‏38‏.
ن86هـ=705م

المُثْكِلُ مِنَ الأَشْيَاءِ: ما يَكُونُ سَببًا لِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

فَإِذَا قَرَأْتَ كِتَابِي هَذَا، فَكُنْ أَطْوَعَ لَهُ مِنْ خُفِّهِ وَنَعْلِهِ، وَإِلَّا أَتَاكَ مِنِّي سَهْمٌ مُثْكِلٌ بِحَتْفٍ قَاضٍ"
عبد الملك بن مروان بن الحكم
الجليس الصّالح الكافي والأنيس الناصح الشافي: أبو الفرج ابن طَرار الجَريريّ النَّهْروانيّ (ت، 390هـ)، دراسة وتحقيق: إحسان عبّاس، محمّد مرسي الخولي، عالم الكتب، بيروت، ط1، (1413هـ/ 1993م)، ‏3/‏153‏.

مُثْكِل[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

المُثْكِلُ مِنَ النِّسَاءِ: الَّتِي تَتَابَعَ عَلَيْهَا الثُّكْلُ، وَلَازَمَهَا.

"وَأَثْكَلَتِ المَرْأَةُ، فَهِيَ مُثْكِلٌ : لَازِمٌ لَهَا الثُّكْلُ، مِنْ غَيْرِ أَنْ يُقَالَ: أَثْكَلَتْ وَلَدَهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏350‏.

ثَكَّلَ[مُتَعَدٍّ]

قبل 100ق.هـ=525م

ثَكَّلَهُ : فَجَّعَهُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

تُثَكِّلُ دَانِيَاتُ البَغْيِ قَوْمًا وَتَدْعُو آخَرِينَ إِلَى الصَّلَاحِ
جَسّاس بن مرّة البَكْريّ
ديوان بني بكر في الجاهلية: جمع وشرح وتوثيق ودراسة: عبد العزيز نبوي، دار الزهراء، القاهرة، ط1، (1410هـ/ 1989م)، ‏381‏.

أَثْكَلَ[مُتَعَدٍّ]

قبل 100ق.هـ=525م

أَثْكَلَ الشَّخْصَ: أَفْقَدَهُ الوَلَدَ وَنَحْوَهُ، وَفَجَعَهُ فِيهِ.

إِذَا مَا إِخْوَةٌ كَثُرُوا، وَطَابُوا بِنَاسِبَةٍ، لِأُمِّهِمُ الهَبُولُ
سَتُثْكَلُ ، أَوْ يُفَارِقُهَا بَنُوهَا سَرِيعًا، أَوْ يَهُمُّ بِهِمْ قَبِيلُ
أُحَيحة بن الجُلاح الأوسيّ
جمهرة أشعار العرب في الجاهلية والإسلام: أبو زيد القرشي (ت، 170هـ)، حققه وضبطه وزاد في شرحه: علي محمّد البجاوي، نهضة مصر، د.ت. ‏520‏-‏ 521‏.

أَثْكَلَ[لَازِم]

ن175هـ=791م

أَثْكَلَتِ المَرْأَةُ: تَتَابَعَ عَلَيْهَا فَقْدُ الأَحِبَّةِ مِنْ وَلَدٍ وَنَحْوِهِ، وَلَازَمَهَا.

" وَأَثْكَلَتِ المَرْأَةُ، فَهِيَ مُثْكِلٌ: لَازِمٌ لَهَا الثُّكْلُ، مِنْ غَيْرِ أَنْ يُقَالَ: أَثْكَلَتْ وَلَدَهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏05/‏350‏.

مُثْكَل[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُثْكَلُ : المَفْجُوعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

وَهَزِئَتْ مِنِّيَ بِنْتُ مَوْءَلَهْ وَأَنْتِ لَا جُنِّبْتِ تَبْرِيحَ الوَلَهْ مَزْؤُودَةً أَوْ فَاقِدًا أَوْ مُثْكَلَهْ
صُحير بن عُمير التَّميميّ
الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏234‏-‏ 235‏.

ثَكْلَان[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الثَّكْلَانُ : الَّذِي فُجِعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

لَا أَسْمَعُ اللَّهْوَ فِي الحَدِيثِ، وَلَا يَنْفَعُنِي فِي الفِرَاشِ مُضْطَجَعُ
لَا وَجْدُ ثَكْلَى كَمَا وَجَدْتُ، وَلَا وَجْدُ عَجُولٍ أَضَلَّهَا رُبَعُ
مالك بن حَريم الهَمْدانيّ
ثلاثة شعراء مقلون جمع وتحقيق ودراسة: تحقيق: شريف راغب علاونه، دار المناهج، عمان، ط1، (1427هـ/ 2007م)، ‏48‏.

مَثْكَلَة[مَصْدَرٌ مِيمِيّ]

قبل 50ق.هـ=573م

المَثْكَلَةُ : سَبَبُ الفَقْدِ وَالهَلَكَةِ.

أَاُمَيَّ، إِنِّي لَوْ شَهِدْ تُكِ عِنْدَ مَثْكَلَةِ الرَّضَاعِ
لَحَمَيْتُكِ الأَعْدَاءَ، أَوْ لَأَذِنْتُ ثَمَّ إِلَى المِصَاعِ
عبد الله بن ثعلبة اليَشْكريّ
كتاب الوَحشيَّات وهو الحَماسة الصُّغْرى: أبو تمام (ت، 231هـ)، علق عليه وحققه: عبد العزيز الميمني، زاد في حواشيه: محمود محمّد شاكر، دار المعارف، القاهرة، ط3، (1987م)، ‏35‏.

ثَاكِل[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الثَّاكِلُ : الَّذِي فُجِعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

فَضَجَّ، وَضَجَّتْ بِالبَرَاحِ؛ كَأَنَّهَا وَإِيَّاهُ نَوْحٌ فَوْقَ عَلْيَاءَ ثُكَّلُ
فَأَغْضَى، وَأَغْضَتْ، وَابْتَسى، وَابْتَسَتْ بِهِ مَرَامِيلُ عَزَّاهَا، وَعَزَّتْهُ مُرْمِلُ
الشَّنْفَرَى الأزديّ
شرح شعر الشَّنفرى الأزدي: محاسن بن إسماعيل الحلبي، تحقيق وتعليق: خالد عبد الرؤوف الجبر، دار الينابيع، عمان، ط1، (2004م)، ‏76‏.

ثَكِلَ[مُتَعَدٍّ]

قبل 13ق.هـ=609م

ثَكِلَهُ : عَدِمَهُ، وَفُجِعَ بِفَقْدِهِ.

أَلَا ثَكِلَتْكَ أُمُّكَ، عَبْدَ عَمْرٍو أَبِالخِزْيَاتِ، آخَيْتَ المُلُوكَا؟
الخِرْنِق بنت بدر البَكْريّة
ديوان شعر الخرنق بنت بدر بن هفان: تحقيق: حسين نصار، مطبعة دار الكتب المصرية، القاهرة، ط2، (1416هـ/ 1996م)، ‏41‏.

مِثْكَال[صِيغَةُ مُبَالَغَة]

8هـ=629م

المِثْكَالُ مِنَ النِّسَاءِ وَنَحْوِهَا: الكَثِيرَةُ الثُّكْلِ.

كَأَنَّ أَوْبَ ذِرَاعَيْهَا، وَقَدْ عَرِقَتْ وَقَدْ تَلَفَّعَ بِالقُورِ العَسَاقِيلُ
شَدَّ النَّهَارِ، ذِرَاعَا عَيْطَلٍ نَصَفٍ قَامَتْ فَجَاوَبَهَا نُكْدٌ مَثَاكِيلُ
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏16‏-‏ 17‏.

مَثْكُول[اسْمُ مَفْعُول]

ن58هـ=678م

المَثْكُولُ مِنَ الأَشْخَاصِ: الَّذِي عَدِمَهُ المَرْءُ وَفُجِعَ بِفَقْدِهِ.

"يَا طُولَ حُزْنِي وَشَجَايَ، لَمْ أَلَعْ عَلَى مَثْكُولٍ ، بَعْدَ رَسُولِ اللهِ @ ، لَوْعِي عَلَى أَبِي"
عائشة بنت أبي بكر الصِّدّيق
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏1/‏7‏.

ثَكُول[صِفَةٌ مُشَبَّهَة]

ن68هـ=687م

الثَّكُولُ مِنَ الأَشْخَاصِ: مَنْ فُجِعَ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

كَأَنِّي وَالِهٌ بِفِرَاقِ لُبْنَى تَهِيمُ بِفَقْدِ وَاحِدِهَا، ثَكُولُ
أَلَا يَا قَلْبُ، وَيْحَكَ! كُنْ جَلِيدًا فَقَدْ رَحَلَتْ، وَفَاتَ بِهَا الذَّمِيلُ
قيس بن ذَريح الكنانيّ
قيس ولُبْنى شعر ودراسة: جمع وتحقيق وشرح: حسين نصّار، مكتبة مصر- دار مصر، القاهرة، (1979م)، ‏139‏.
ن117هـ=735م

الثَّكُولُ مِنَ الأَشْيَاءِ: مَا كَانَ سَبَبًا لِلْفَقْدِ وَالهَلَكَةِ.

إِذَا ذَاتُ أَهْوَالٍ ثَكُولٌ تَغَوَّلَتْ بِهَا الرُّبْدُ فَوْضَى، وَالنَّعَامُ السَّوَارِحُ
تَبَطَّنْتُهَا، وَالقَيْظُ مَا بَيْنَ جَالِهَا إِلَى جَالِهَا سِتْرًا مِنَ الآلِ، نَاصِحُ
ذو الرُّمَّة
ديوان ذي الرُّمّة غَيلان بن عقبة العَدَويّ: شرح: أبي نصر أحمد بن حاتم الباهلي صاحب الأصمعي، رواية: أبي العبّاس ثعلب، حققه وقدم له وعلق عليه: عبد القدوس أبو صالح، مؤسسة الرسالة، بيروت، ط3، (1414هـ/ 1993م)، ‏2/‏882‏.

إِثْكَال[مَصْدَر]

ن90هـ=709م

الإِثْكَالُ : الفَجْعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

إِذَا الصَّدَى جَاوَبَهُ، إِعْوَالَا هَامٌ يُنَادِي، مُثْكَلٌ إِثْكَالَا تَخَالُهُ مُؤَبِّنًا مُخْتَالَا
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏1/‏268‏.

مَثْكَل[مَصْدَرٌ مِيمِيّ]

ن100هـ=718م

المَثْكَلُ : الفَجْعُ بِفَقْدِ الحَبِيبِ مِنْ وَلَدٍ وَنَحْوِهِ.

لَعَمْرُكَ، إِنَّ المُسْتَثِيرَ عَدَاوَتِي لَكَالمُتَبَغِّي الثُّكْلَ مِنْ غَيْرِ مَثْكَلِ
الخَطيم المُحْرزيّ العَبْشَميّ العُكليّ
الخطيم المحرزي حياته وما تبقى من شعره: صنعة: نوري حمّودي القيسيّ، مجلة المورد، بغداد، مج3، ع4، (1394هـ/ 1974م)، ‏4/‏185‏.

تَثَكُّل[مَصْدَر]

ن597هـ=1200م

التَّثَكُّلُ : فَقْدُ الوَلَدِ.

"قَالَ رَجُلٌ لِوَلَدٍ لَهُ خَبِيثٍ: مَا أَطْيَبَ التَثَكُّلَ ! قَالَ: أَطْيَبُ مِنْهُ اليُتْمُ!"
ابن الجوزيّ
أخبار الأذكياء: ابن الجوزيّ (ت، 597هـ)، بعناية: بسّام عبد الوهّاب الجابي، الجفان والجابي، قبرص، دار ابن حزم، بيروت، ط1، (1424ه/ 2003م)، ‏264‏.

مُثَكَّل[اسْمُ مَفْعُول]

ن629هـ=1231م

المُثَكَّلُ مِنَ الأَشْخَاصِ: الَّذِي يُفْجَعُ فِي وَلَدِهِ وَمَالِهِ وَنَحْوِ ذَلِكَ.

وَمَنْ مَلَّكَ الأَعْدَاءَ تَدْبِيرَ أَمْرِهِ فَذَاكَ الَّذِي يُدْعَى العَدِيمَ المُثَكَّلَا
ابن المُقَرَّب العُيونيّ
شرح ديوان ابن المقرّب: ابن المُقَرَّب العُيونيّ (ت، 629هـ)، أعده وحققه وعلق عليه: عبد الخالق بن عبد الجليل الجنبي وآخرون، منشورات ملتقى الواحتين، الرويس، دار المحجة البيضاء، بيروت، ط2، (1433ه/ 2012م)، ‏3/‏1560‏.

مُثَكِّل[صِفَةٌ مُشَبَّهَة]

ن629هـ=1231م

المُثَكِّلُ مِنَ الأُمُورِ: المُهْلِكُ المُتْلِفُ.

"وَحَسْبُكَ بِدَارٍ لَا يَلَذُّ بِهَا لِعَاقِلٍ مَأْكَل، وَمَا لِوَالِدٍ فِيهَا مِنْ وَلَدِهِ إِلَّا العَنَاءُ المُنَكِّل، أَوِ الفَنَاءُ المُثَكِّلُ المُوَكِّل"
أبو الربيع الكَلاعيّ
نكتة الأمثال ونفْثة السِّحر الحلال: أبو الربيع الكَلاعيّ (ت، 634هـ)، حققه وقدم له: علي إِبراهيم كُردي، دار سعد الدِّين، دمشق، ط1، (1416ه/ 1995م)، ‏86‏.
