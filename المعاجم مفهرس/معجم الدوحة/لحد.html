مُلْحِد[اسْمُ فَاعِل]

قبل 150ق.هـ=476م

المُلْحِدُ : المَائِلُ عَنِ القَصْدِ وَسَوَاءِ الأَمْرِ.

"يَا نُفَيْرُ، وَفَيْتَ لِقَوْمِكَ، وَقَدْ يَكُونُ لَكَ مِنْهُمُ العَدُوُّ الكَاشِحُ، وَالحَسُودُ الضَّاغِنُ، وَالمُمَارِي المُلْحِدُ "
منسوب إلى تُبَّع أسعد أبي كرب الحميريّ
كتاب التيجان في ملوك حِمْيَر: وهب بن مُنَبِّه الصَّنْعانيّ (ت، 114هـ)، رواية: أبي محمد عبد الملك بن هشام عن أسد بن موسى عن أبي إدريس بن سنان عن جده لأمه وهب بن منبه: تحقيق ونشر: مركز الدراسات والأبحاث اليمنية، صنعاء، ط1، (1347هـ)، ‏240‏.
ن5هـ=626م

المُلْحِدُ : المُنْكِرُ وُجُودَ اللهِ تَعَالَى.

اعْلَمْ بِأَنَّ اللهَ لَيْسَ كَصُنْعِهِ صُنْعٌ، وَلَا يَخْفَى عَلَيْهِ مُلْحِدُ
أُمَيّة بن أبي الصَّلْت الثَّقَفيّ
ديوان أمية بن أبي الصلت: صنعة: عبد الحفيظ السطلي، المطبعة التعاونية، دمشق، (1974م)، ‏353‏.
11هـ=632م

المُلْحِدُ : الَّذِي يَدْفِنُ المَيَّتَ فِي اللَّحْدِ.

عَرَفْتُ بِهَا رَسْمَ الرَّسُولِ وَعَهْدَهُ وَقَبْرًا بِهَا وَارَاهُ فِي التُّرْبِ مُلْحِدُ
حسّان بن ثابت الخزرجيّ الأنصاريّ
ديوان حسان بن ثابت: حقّقه وعلّق عليه: وليد عرفات، دار صادر، بيروت، (2006م)، ‏455‏.
ن11هـ=632م

المُلْحِدُ فِي الحَرَمِ: المُنْتَهِكُ حُرْمَتَهُ الظَّالِمُ فِيهِ.

"أَبْغَضُ النَّاسِ إِلَى اللهِ ثَلَاثَةٌ: مُلْحِدٌ فِي الحَرَمِ، وَمُبْتَغٍ فِي الإِسْلَامِ سُنَّةَ الجَاهِلِيَّةِ، وَمُطَّلِبٌ دَمَ امْرِئٍ بِغَيْرِ حَقٍّ؛ لِيُهَرِيقَ دَمَهُ"
حديث نبويّ
الجامع الصحيح المسند من حديث رسول الله صلى الله عليه وسلم وسننه وأيامه: البُخاريّ (ت، 256هـ)، قام بشرحه وتصحيح تجاربه وتحقيقه: محبّ الدّين الخطيب، رقم كتبه وأبوابه وأحاديثه واستقصى أطرافه: محمّد فؤاد عبد الباقي، نشره وراجعه وقام بإخراجه وأشرف على طبعه: قصي محبّ الدّين الخطيب، الدار السلفية- ومكتبتها، القاهرة، ط1، (1400هـ)، ‏4/‏269،،‏(رقم الحديث: 6882).

لَاحِد[اسْمُ فَاعِل]

قبل 100ق.هـ=525م

اللَّاحِدُ : الَّذِي يَحْفِرُ اللَّحْدَ.

وَاللهِ، مَا هَاشِمٌ بِالنَّاقِصِ الفَاسِدْ وَالخَيْرُ فِي ثَوْبِهِ، وَحُفْرَةِ اللَّاحِدْ
الحارث بن حَنْش
الجامع في العَروض والقوافي: أحمد بن محمّد العَروضيّ (ت، 342هـ)، حققه وقدم له: زهير غازي زاهد، هلال ناجي، دار الجيل، بيروت، ط1، (1416هـ/ 1996م)، ‏61‏.
قبل 13ق.هـ=609م

اللَّاحِدُ : الَّذِي يَدْفِنُ المَيِّتَ فِي اللَّحْدِ.

وَمَاذَا يُعَدِّي المَالُ عَنْكَ وَجَمْعُهُ، إِذَا كَانَ مِيرَاثًا، وَوَارَاكَ لَاحِدُ ؟
حاتم بن عبد الله الطّائيّ
ديوان شعر حاتم بن عبد الله الطائي وأخباره: صنعة: يحيى بن مدرك الطائي، رواية: هشام بن محمّد الكلبي، دراسة وتحقيق: عادل سليمان جمال، مطبعة المدني، القاهرة، د.ت. ‏228‏.
ن58هـ=678م

اللَّاحِدُ : مَنْ حِرْفَتُهُ إِلْحَادُ القُبُورِ.

"اخْتَلَفُوا فِي اللَّحْدِ وَالشَّقِّ... فَأَرْسَلُوا إِلَى الشَّقَّاقِ وَاللاَّحِدِ جَمِيعًا، فَجَاءَ اللاَّحِدُ ، فَلَحَدَ لِرَسُولِ اللهِ @  ثُمَّ دُفِنَ @ "
عائشة بنت أبي بكر الصِّدّيق
سنن الحافظ أبي عبد الله محمّد بن يَزيد القَزويني ابن ماجه: ابن ماجَهْ (ت، 273هـ)، حقق نصوصه ورقم كتبه وأبوابه وأحاديثه وعلق عليه: محمّد فؤاد عبد الباقي،دار إحياء الكتب العربية، د.ت. ‏497‏.
ن145هـ=762م

اللَّاحِدُ مِنَ الأَشْخَاصِ: المَائِلُ عَنِ القَصْدِ وَسَوَاءِ الأَمْرِ.

بِالعَدْلِ حَتَّى انْضَمَّ كُلُّ عَانِدِ وَتَرَكَ الإِلْحَادَ كُلُّ لَاحِدِ
رؤبة بن العجّاج
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏421‏.

مُلْحَد[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُلْحَدُ مِنَ الأَشْخَاصِ: المَدْفُونُ فِي اللَّحْدِ.

وَأَيْأَسَنِي مِنْ كُلِّ خَيْرٍ طَلَبْتُهُ كَأَنَّا وَضَعْنَاهُ عَلَى رَمْسِ مُلْحَدِ
طَرَفة بن العبد البكريّ
ديوان طرفة بن العبد شرح الأعلم الشَّنْتَمَري وتليه طائفة من الشعر المنسوب إلى طرفة: تحقيق: درية الخطيب، لطفي الصقّال، إدارة الثقافة والفنون، البحرين، المؤسسة العربية، بيروت، ط2، (2000م)، ‏50‏.
قبل 13ق.هـ=609م

المُلْحَدُ مِنَ القُبُورِ: المَشْقُوقُ فِي جَانِبِهِ لَحْدٌ.

وَرَاجِي أُمُورٍ جَمَّةٍ لَنْ يَنَالَهَا سَتُشْعِبُهُ عَنْهَا شَعُوبٌ لِمُلْحَدِ
عَديّ بن زيد العِباديّ
ديوان عَديّ بن زيد العِباديّ: حققه وجمعه: محمّد جبار المعيبد، شركة دار الجمهورية، بغداد، (1385هـ/ 1965م)، ‏106‏.
ن125هـ=743م

المُلْحَدُ مِنَ الجُحُورِ وَنَحْوِهَا: المَحْفُورُ وَسَطُهُ كَاللَّحْدِ.

مِنْ كُلِّ ذَاقِنَةٍ يَعُومُ زِمَامُهَا عَوْمَ الخَشَاشِ عَلَى الصَّفَا يَتَرَأَّدُ
فُتْلٍ مَرَافِقُهَا، كَأَنَّ خَلِيفَهَا مَكْوٌ، أَبَنَّ بِهِ سِبَاعٌ، مُلْحَدُ
الطِّرمّاح
ديوان الطِّرمَّاح: عني بتحقيقه: عِزّة حسن، دار الشرق العربي، بيروت- حلب، ط2، (1414هـ/ 1994م)، ‏112‏.

"فَامْهَدْ لِنَفْسِكَ أَيُّهَا المَغْرُورُ... فَكَأَنِّي بِكَ، قَدْ أُدْرِجْتَ فِي أَطْمَارِكَ، وَأُودِعْتَ مُلْحَدَكَ "

مُلْحَد[اسْم]

ن176هـ=792م

المُلْحَدُ : مَا يُغَطَّى بِهِ اللَّحْدُ مِنَ الحِجَارَةِ وَنَحْوِهَا.

كَمْ أَخٍ صَالِحٍ وَعَمٍّ وَخَالٍ وَابْنِ عَمًّ كَالصَّارِمِ المَسْنُونِ

ابن هَرْمَة
شِعر إبراهيم بن هَرْمَة القُرشي (ت، 176هـ): تحقيق: محمّد نفّاع، حسين عَطوان، مطبوعات مجمع اللغة العربية، دمشق، (1969م)، ‏218‏.

إِلْحَادَة[اسْمُ مَرَّة]

710هـ=1310م

الإِلْحَادَةُ : المَرَّةُ مِنَ الإِلْحَادِ، وَهُوَ المَيْلُ عَنِ الحَقِّ وَمُخَالَفَتُهُ.

"فَهَذِهِ التَّفَاسِيرُ مِنْ جِنْسِ تِلْكَ التَّفَاسِيرِ، وَهِيَ أَمْثَلُ مِنْ إِلْحَادَاتِ الرَّافِضَةِ"
ابن تيميّة
منهاج السنة النبوية: ابن تيميّة (ت، 728هـ)، تحقيق: محمد رشاد سالم، جامعة الإمام محمد بن سعود الإسلامية، الرياض، ط1، (1406ه/ 1986م)، ‏7/‏246‏.

إِلْحَادِيّ[اسْمٌ مَنْسُوب]

718هـ=1318م

الإِلْحَادِيُّ : المَنْسُوبُ إِلَى الإِلْحَادِ، وَهُوَ المَيْلُ عَنِ الحَقِّ وَمُخَالَفَتُهُ.

"وَهَذِهِ الأَدِلَّةُ كُلَّمَا تَدَبَّرَهَا العَاقِلُ تَبَيَّنَ لَهُ قَطْعًا أَنَّ هَؤُلَاءِ النُّفَاةَ مُنَاقِضُونَ لِلرَّسُولِ... وَأَنَّ اسْتِدْلَالَ هَؤُلَاءِ بِنُصُوصِ الأَنْبِيَاءِ عَلَى نَفْيِهِمْ، مِنْ جِنْسِ اسْتِدْلَالِ القَرَامِطَةِ عَلَى شَرِيعَتِهِمُ الإِلْحَادِيَّةِ بِنُصُوصِ الأَنْبِيَاءِ"
ابن تيميّة
درء تعارض العقل والنقل: ابن تيميّة (ت، 728هـ)، تحقيق: محمد رشاد سالم، جامعة الإمام محمد بن سعود، الرياض، هجر، القاهرة، ط2، (1411ه/ 1991م)، ‏7/‏123‏.

إِلْحَادِيَّة[اسْم]

ن727هـ=1326م

الإِلْحَادِيَّةُ : جَمَاعَةُ المُلْحِدِينَ المُنْكِرِينَ وُجُودَ اللهِ تَعَالَى.

"وَالهُنُودُ بِجُمْلَتِهِمْ قَائِلُونَ بِالتَّنَاسُخِ، وَالإِسْمَاعِيلِيَّةُ كَذَلِكَ، وَالإِلْحَادِيَّةُ كَذَلِكَ، وَالقَرَامِطَةُ وَالنُّصَيْرِيَّةُ كَذَلِكَ"
شمس الدين الأنصاريّ
كتاب نخبة الدهر في عجائب البر والبحر: شمس الدين الأنصاريّ (ت، 727هـ)، Gebundene Ausgabe, von August F Mehren Herausgeber Frankfurt: Johann Wolfgang Goethe-Universität Nachdr. d. Ausg. Sankt Petersburg, (1865)، ‏174‏.

إِلْحَادِيَّة[مَصْدَرٌ صِنَاعِيّ]

ن1368هـ=1948م

الإِلْحَادِيَّةُ : إِنْكَارُ وُجُودِ اللهِ تَعَالَى. [الأديان والمذاهب والفرق]

"فَلَقِيَنِي بَعْدَ المُحَاضَرَةِ، وَتَحَدَّثْنَا طَوِيلًا عَنْ مِصْرَ، وَعَنِ اليَمَنِ وَعَنِ انْتِشَارِ الإِلْحَادِيَّةِ ، وَالإِبَاحِيَّةِ المُسْتَشْرِي فِي ذَلِكَ الوَقْتِ"
حسن البنا
مذكرات الدعوة والداعية: حسن البنا (ت، 1368هـ)، مكتبة آفاق، الكويت، ط1، (1433ه/ 2012م)، ‏136‏.

تَلَحُّد[مَصْدَر]

ن764هـ=1362م

التَّلَحُّدُ : الدُّخُولُ فِي اللَّحْدِ.

وَتَغْرِيدُهُ فِي الدَّوْحِ فِيهِ لَطِيفَةٌ يَعِيهَا اللَّطِيفُ الفَهْمِ دُونَ تَرَدُّدِ
بِتَكْرَارِهِ: هُدْ، هُدْ، وَذَلِكَ أَمْرُهُ بِتَأْكِيدٍ، أَيْ: تُبْ، تُبْ قُبَيْلَ التَّلَحُّدِ
صلاح الدّين الصّفديّ
ألحان السّواجع بين البادئ والمراجع: صلاح الدّين الصّفديّ (ت، 764هـ)، عني بتحقيقه: إِبراهيم صالح، دار البشائر، دمشق، ط1، (1425ه/ 2004م)، ‏2/‏355 ‏-‏ 356‏.

مُلَاحِد[اسْمُ فَاعِل]

1233هـ=1817م

المُلَاحِدُ : المُنْكِرُ وُجُودَ اللهِ تَعَالَى.

"فَيَا لَهَا مِنْ مَسَرَّاتٍ شَدَّتْ عَضُدَ الدِّين، وَفَتَّتَتْ سَوَاعِدَ المُلَاحِدِين ، وَقَصَمَتْ ظُهُورَ الكَافِرِين، وَقَلْقَلَتْ مَعَاقِلَ المُعَانِدِين!"
لطف الله بن أحمد جحاف
نصوص يمنية عن الحملة الفرنسية على مصر: لطف الله بن أحمد جحاف (ت، 1243هـ)، نشر وتحقيق: سيد مصطفى سالم: مركز الدراسات اليمنية، صنعاء، ط2، (1989م)، ‏136‏.

لِحَاد[مَصْدَر]

1270هـ=1853م

اللِّحَادُ : المِرَاءُ وَالجِدَالُ.

"فَإِنَّ الزَّوْجَيْنِ إِذَا كَانَا عَلَى حَالَةِ النُّفُورِ وَالعِنَاد، وَالخِلَافِ وَاللِّحِاد ، لَمْ تَكُنْ تَرْبِيَتُهُمَا لِوَلَدِهِمَا، إِلَّا إِغْرَاءً بِالِاقْتِدَاءِ بِهِمَا"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏234‏.

لَحْدِيّ[صِفَة]

1331هـ=1912م

اللَّحْدِيُّ مِنَ الأَمَاكِنِ: الَّذي يُشْبِهُ القَبْرَ.

"وَقَدْ بَقِيَتْ تِلْكَ المَخَادِعُ اللَّحْدِيَّةُ وَمَا يُجَاوِرُهَا بِزِينَتِهَا وَرُسُومِهَا وَصُوَرِهَا الكَثِيرَةِ الأَلْوَانِ"
بولس سلمان
العقائد المسيحية في الدياميس الرومانية: ‏بولس سلمان،‏مجلة المشرق: المطبعة اليسوعية، بيروت، س15، ع8، (1912م)، ‏563‏-‏ 564‏.
1428هـ=2006م

اللَّحْدِيُّ مِنَ العَتَمَاتِ وَنَحْوِهَا: الَّذي يُشْبِهُ ظَلَامَ القَبْرِ.

"وَحِينَ دَعَتْنِي لِلدُّخُولِ هَاجَتْ إِوَزَّاتٌ، وَفَزِعَتْ أَرَانِبُ، وَأَنَا أَتَقَدَّمُ نَحْوَ الأَرْضِ الزَّلِقَةِ، قَبْلَ أَنْ تَغْشَانِي عَتَمَةُ المَدْخَلِ البَارِدَةُ اللَّحْدِيَّةُ العَطِنَةُ"
سمير عبد الفتاح
تطهر الفارس القديم: سمير عبد الفتاح، كتب عربية، إصدار رقمي، (1428هـ/2006م)، ‏8‏.
ن1439هـ=2017م

اللَّحْدِيُّ مِنَ القُبُورِ: المُجَصَّصُ المَبْنِيُّ بِاللَّبِنِ.

"وَقَدْ شُيِّدَتِ القُبُورُ اللَّحْدِيَّةُ فِي القَرْنَيْنِ التَّاسِعَ عَشَرَ وَالعِشْرِينَ"
عبد السلام كمال
مظاهر حضارية من الثقافة العثمانية: عبد السلام كمال، دار بروج للنشر، مصر، القاهرة، (1439هـ/2017م)، ‏126‏.

لَحَّدَ[مُتَعَدٍّ]

ن1349هـ=1930م

لَحَّدَ الشَّخْصَ: دَفَنَهُ فِي اللَّحْدِ.

"لَا تُلَحِّدُوا سَيْفًا مُضَرَّجًا بِدَمِ الأَعْدَاءِ، وَلَا تَنْحَرُوا مُهْرًا يَخُوضُ المَنَايَا"
جبران خليل جبران
في عالم الرؤيا: مقالات مختارة لجبران خليل جبران: جبران خليل جبران (ت، 1350هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2019م)، ‏13‏.

لَحْدِيَّة[مَصْدَرٌ صِنَاعِيّ]

1446هـ=2024م

اللَّحْدِيَّةُ : الدُّخُولُ فِي حَالٍ مِنَ الجُمُودِ الفِكْرِيِّ، تُشْبِهُ حَالَ المَوْتَى فِي قُبُورِهِمْ.

"لِمَاذَا نَرْفُضُ التَّجْدِيدَ وَالتَّطْوِيرَ، وَنُصِرُّ عَلَى التَّنْمِيطِ، بَلِ التَّحْنِيطِ وَاللَّحْدِيَّةِ ؟"
المختار السالم
في ظلال الحروف (خواطر): المختار السالم أحمد السالم، دار إي كتب لمتد المحدودة، بريطانيا، (1445هـ/2024م)، ‏119‏.
