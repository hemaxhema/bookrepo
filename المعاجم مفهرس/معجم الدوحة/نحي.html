تَنَحَّى[لَازِم]

قبل 50ق.هـ=573م

تَنَحَّى عَنِ الشَّيْءِ: ابْتَعَدَ عَنْهُ.

أَلَا قُلْ لِمَنْ تَزْدَرِيهِ الحُرُوبُ تَنَحَّ ، وَخَلِّ لَهَا دَارَهَا
سَعد بن مالك بن ضُبَيْعة البَكْريّ
ديوان بني بكر في الجاهلية: جمع وشرح وتوثيق ودراسة: عبد العزيز نبوي، دار الزهراء، القاهرة، ط1، (1410هـ/ 1989م)، ‏545‏.

تَنَحَّى[مُتَعَدٍّ]

ن175هـ=791م

تَنَحَّى اللَّبَنَ: تَمَخَّضَهُ.

"نَحَى اللَّبَنَ، يَنْحَاهُ: مَخَضَهُ، وَتَنَحَّاهُ : تَمَخَّضَهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏303‏.

تَنَحَّى[مُتَعَدٍّ بِالحَرْف]

1318هـ=1900م

تَنَحَّى عَنِ الرِّئَاسَةِ وَنَحْوِهَا: تَرَكَهَا وَاعْتَزَلَهَا.

"وَأَطْلُبُ مِنَ الفَلَاسِفَةِ أَنْ يُخْضِعُوا عُقُولَهُمْ لِلْإِيمَانِ، وَمِنْ قَيْصَرَ أَنْ يَتَنَحَّى عَنِ الرِّئَاسَةِ الدِّينِيَّةِ"
جرمانوس معقّد
الدين المسيحي بإزاء العالم الوثني: ‏جرمانوس معقّد،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين، بيروت، س3، ع15، (06/ 04/ 1318هـ)، ‏689‏.

مُنْتَحِي[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

المُنْتَحِي مِنَ الدَّوَابِّ: المَائِلُ فِي سَيْرِهِ نَحْوَ شِقِّهِ الأَيْسَرِ.

لَا يَنْفَعُ الوَحْشَ مِنْهُ أَنْ تَحَذَّرَهُ كَأَنَّهُ مُعْلَقٌ مِنْهَا بِخُطَّافِ
إِذَا أُوَاضِعُ مِنْهُ مَرَّ مُنْتَحِيًا مَرَّ الأَتِيِّ عَلَى بَرْدِيِّهِ الطَّافِي
عبد المسيح بن عَسَلَة الشَّيْبانيّ
ديوان بني بكر في الجاهلية: جمع وشرح وتوثيق ودراسة: عبد العزيز نبوي، دار الزهراء، القاهرة، ط1، (1410هـ/ 1989م)، ‏427‏.

مُنْتَحِي[صِفَة]

ن216هـ=831م

المُنْتَحِي : المُبْتَعِدُ.

"لَمْ يَزَالُوا مُنْتَحِينَ "
الأصمعيّ
البُلْغة في شذور اللّغة: نشرها: أوغست هفنر، لويس شيخو، المطبعة الكاثوليكية، بيروت، ط2، (1914م)، ‏38‏.

نَحَّى[مُتَعَدٍّ]

قبل 13ق.هـ=609م

نَحَّى الشَّيْءَ: أَزَالَهُ.

أَطَعْتَ عُرَيِّبَ إِبْطَ الشِّمَالِ تُنَحِّي لِحَدِّ المَوَاسِي الحُلُوقَا
شُتيم بن خُويلد الفَزاريّ
شعر قبيلة ذُبيان في الجاهلية: جمع وتحقيق ودراسة: سلامة عبد الله السويدي، مطبوعات جامعة قطر، الدوحة، (1408هـ/ 1987م)، ‏393‏.
ن11هـ=632م

نَحَّاهُ عَنِ الطَّرِيقِ: أَبْعَدَهُ.

"كَانَتْ شَجَرَةٌ تُؤْذِي أَهْلَ الطَّرِيقِ، فَقَطَعَهَا رَجُلٌ، فَنَحَّاهَا عَنِ الطَّرِيقِ فَأُدْخِلَ بِهَا الجَنَّةَ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏13/‏407،،‏(رقم الحديث: 8039).

نَحَّى[مُتَعَدٍّ بِالحَرْف]

ن11هـ=632م

نَحَّى عَنِ الشَّيْءِ: مَالَ عَنْهُ، وَعَدَلَ.

"لَا أَرَانَا إِلَّا قَدْ شَقَقْنَا عَلَيْكَ، نَحِّ بِنَا عَنِ الطَّرِيقِ، أَوْ مِلْ بِنَا عَنِ الطَّرِيقِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏37/‏266،،‏(رقم الحديث: 22575).

نَحَى[مُتَعَدٍّ]

قبل 13ق.هـ=609م

نَحَى الشَّيْءَ: أَمَالَهُ.

يُعَارِضُ تِسْعًا، قَدْ نَحَاهَا لِمَوْرِدٍ يَجُورُ بِذَاتِ الضِّغْنِ مِنْهَا، وَيَعْدِلُ
امرؤ القيس بن جَبَلَة السَّكونيّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏8/‏347‏.
ن117هـ=735م

نَحَى الشَّيْءَ عَنْ غَيْرِهِ: أَبْعَدَهُ.

أَلَا أَيُّهَذَا البَاخِعُ الوَجْدُ نَفْسَهُ بِشَيْءٍ نَحَتْهُ عَنْ يَدَيْهِ المَقَادِرُ
ذو الرُّمَّة
ديوان ذي الرُّمّة غَيلان بن عقبة العَدَويّ: شرح: أبي نصر أحمد بن حاتم الباهلي صاحب الأصمعي، رواية: أبي العبّاس ثعلب، حققه وقدم له وعلق عليه: عبد القدوس أبو صالح، مؤسسة الرسالة، بيروت، ط3، (1414هـ/ 1993م)، ‏2/‏1037‏.
ن175هـ=791م

نَحَى اللَّبَنَ: مَخَضَهُ.

" نَحَى اللَّبَنَ، يَنْحَاهُ: مَخَضَهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏303‏.

مُتَنَحِّي[اسْمُ فَاعِل]

10ق.هـ=612م

المُتَنَحِّي عَنِ الشَّيْءِ: المُبْتَعِدُ عَنْهُ.

كَمْ قَدْ شَهِدْتُ الحَرْبَ فِي فِتْيَةٍ عِنْدَ الوَغَى، في عِثْيَرِ القَسْطَلِ
لَا مُتَنَحِّينَ إِذَا جِئْتَهُمْ وَفِي هِيَاجِ الحَرْبِ كَالأَشْبُلِ
أبو طالب
ديوان أبي طالب بن عبد المطلب: صنعة: أبي هَفّان المِهْزَميّ (ت، 257هـ)، عليّ بن حمزة البصريّ (ت، 375هـ): تحقيق: محمّد حسن آل ياسين، دار ومكتبة الهلال، بيروت، ط1، (1421هـ/ 2000م)، ‏110‏.
1403هـ=1982م

مُتَنَحِّي الأَلِّيلِ مِنَ الكُرُومُوسُومَاتِ: مَا يَكُونُ ضَعِيفًا غَيْرَ مُعْلَنٍ مِنْ صِفَاتِ زَوْجٍ مِنَ الأَلِّيلَاتِ ثُنَائِيَّةِ التَّفَرُّعِ (سَائِدَةٌ - مُتَنَحِّيَةٌ) الَّتِي لَيْسَ لَهَا أَيُّ تَأْثِيرٍ فِي النَّمَطِ الظَّاهِرِيِّ لِلْأَفْرَادِ المُتَخَالِفِينَ. [الأحياء]

"أَمَّا الرِّجَالُ الَّذِينَ يَرِثُونَ الكُرُومُوسُومَ X مُتَنَحِّي الأَلِّيلِ فَإِنَّهُمْ يُظْهِرُونَ المَرَضَ"
سامح محمد عبد الخالق التميمي
علم الأحياء: سامح محمد عبد الخالق التميمي، مكتبة العبيكان، الرياض، (1403هـ/1982م)، ‏240‏.

مُتَنَحِّي[صِفَةٌ مُشَبَّهَة]

ن368هـ=978م

المُتَنَحِّي مِنَ الأَمْكِنَةِ: البَعِيدُ.

"وَلَا يُسْتَعْمَلُ إِلَّا لِلْمَكَانِ المُتَنَحِّي ، أَوْ مَا يَجْرِي مَجْرَاهُ"
أبو سعيد السِّيرافيّ
شرح كتاب سيبويه: أبو سعيد السِّيرافيّ (ت، 368هـ)، تحقيق: أحمد حسن مهدلي، علي سيّد علي، دار الكتب العلميّة، بيروت، ط1، (1429هـ/ 2008م)، ‏1/‏100‏.

أَنْحَى[مُتَعَدٍّ]

ن2ق.هـ=620م

أَنْحَى الشَّيْءَ: صَبَّهُ وَأَسَالَهُ.

حَتَّى أُتِيحَ لَهُ أَخُو قَنَصٍ شَهْمٌ، يُطِرُّ ضَوَارِيًا كُثُبَا
يُنْحِي الدِّمَاءَ عَلَى تَرَائِبِهَا وَالقِدَّ مَعْقُودًا وَمُنْقَضِبَا
أوس بن حَجَر المازنيّ التَّميميّ
ديوان أوس بن حجر: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، ط3، (1399هـ/ 1979م)، ‏3‏.

تَنَحِّي[مَصْدَر]

38هـ=658م

التَّنَحِّي عَنِ الشَّيْءِ: الِابْتِعَادُ عَنْهُ.

"فَقَدْ أَتَانِي كِتَابُكَ تُذَكِّرُنِي مِنْ أَمْرِ عُثْمَانَ أَمْرًا لَا أَعْتَذِرُ إِلَيْكَ مِنْهُ، وَتَأْمُرُنِي بِالتَّنَحِّي عَنْكَ، كَأَنَّكَ لِي نَاصِحٌ"
محمّد بن أبي بكر الصِّدّيق
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏5/‏102‏.
1318هـ=1900م

التَّنَحِّي : إِنْهَاءُ المُوَكَّلِ أَوِ المُحَامِي لِعَقْدِ التَّوْكِيلِ. [القانون]

"لَا يَجُوزُ لِلْوَكِيلِ أَنْ يَتَنَحَّى عَنِ التَّوْكِيلِ فِي وَقْتٍ غَيْرِ لَائِقٍ ... غَيْرَ أَنَّ هَذَا القَيْدَ... لَيْسَ مَوْجُودًا فِي المَادَّةِ (23) مِنَ اللَّائِحَةِ؛ لِأَنَّهَا أَجَازَتِ التَّنَحِّيَ وَلَمْ تَشْتَرِطْ إِلَّا إِعْلَانَهُ لِلْمُوَكِّلِ"
أحمد فتحي زغلول
المحاماة: أحمد فتحي زغلول (ت، 1332هـ)، مطبعة المعارف، القاهرة، (1318ه/ 1900م)، ‏357‏.
1436هـ=2014م

التَّنَحِّي عَنِ الرِّئَاسَةِ وَنَحْوِهَا: تَرْكُهَا وَاعْتِزَالُهَا.

"وَمِنَ الجَدِيرِ بِالذِّكْرِ هُنَا أَنَّ المَشْهَدِيَّ كَانَ قَدِ اسْتَثَارَ نَظَرَاتِ الحَذَرِ... عِنْدَمَا لَمْ يَسْتَطِعْ كَبْحَ جِمَاحِ اسْتِيَائِهِ مِنْ قَرَارِ (أَحْمَد حَسَن البَكْر) بِالتَّنَحِّي عَنِ الرِّئَاسَةِ"
عبد الله فاضل
فلّاحو سورية- أبناء وجهائهم الريفيين الأقل شأنا وسياساتهم، لحنّا بطاطو: عبد الله فاضل ، رائد النقشبندي ، ترجمة: عبد الله فاضل، رائد النقشبندي، مراجعة: ثائر ديب، المركز العربي للأبحاث ودراسة السياسات، بيروت، ط1، (2014م)، ‏518‏.

تَنْحِيَة[مَصْدَر]

ن50هـ=670م

تَنْحِيَةُ الشَّيْءِ: إِزَالَتُهُ وَإِبْعَادُهُ.

وَلَوْحُ ذِرَاعَيْنِ فِي بِرْكَةٍ إِلَى جُؤْجُؤٍ رَهِلِ المَنْكِبِ
أُمِرَّ، وَنُحِّيَ مِنْ صُلْبِهِ كَتَنْحِيَةِ القَتَبِ المُجْلَبِ
النّابغة الجَعْديّ
شعر النابغة الجعدي: تحقيق: عبد العزيز رباح، منشورات المكتب الإسلامي، دمشق، ط1، (1384هـ/ 1964م)، ‏21 ‏-‏ 22‏.

نِحْي[اسْم]

ن78هـ=697م

النِّحْيُ : وِعَاءٌ مِنْ فَخَّارٍ، يُحْفَظُ فِيهِ السَّمْنُ وَنَحْوُهُ.

"فَبَيْنَمَا بَنُوهَا يَسْأَلُونَهَا عَنْ إِدَامٍ، وَلَيْسَ عِنْدَهَا شَيْءٌ؛ فَعَمَدَتْ إِلَى نِحْيِهَا الَّتِي كَانَتْ تُهْدِي فِيهِ السَّمْنَ إِلَى النَّبِيِّ @ ، فَوَجَدَتْ فِيهِ سَمْنًا"
جابر بن عبد الله الخزرجيّ الأنصاريّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏23/‏75‏.

1316هـ=1898م

المَنْحَاةُ عَنِ الشَّيْءِ: مَا تَبَاعَدَ عَنْهُ.

"ثُمَّ تُجْعَلُ العَنَاقِيدُ فِي حُجْرَةٍ قَلِيلَةِ الضَّوْءِ، خَالِيَةٍ مِنَ الرُّطُوبَةِ... وَلَا بُدَّ أَنْ تَكُونَ فِي مَنْحَاةٍ عَنِ التَّقَلُّبَاتِ الجَوِّيَّةِ"
مجلّة المشرق
حفظ العنب: ‏سليم أفندي أصفر،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين ،بيروت، س1، ع24 ،(02 /08/ 1316هـ)، ‏1128‏.

مَنْحَاة[اسْمُ مَكَان]

ن110هـ=728م

المَنْحَاةُ : مَكَانُ مُرُورِ الدَّابَّةِ الَّتِي يُسْتَقَى عَلَيْهَا.

قَدْ غَلَّبَتْنِي رُوَاةُ النَّاسِ كُلِّهِمُ إِلَّا حَنِيفَةَ تَفْسُو فِي مَنَاحِيهَا
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏544‏.

نَحْي[مَصْدَر]

ن175هـ=791م

النَّحْيُ : المُبَاعَدَةُ.

"نَحَيْتُهُ أَنْحَاهُ نَحْيًا ... أَيْ: بَاعَدْتَهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏03/‏303‏.

انْتَحَى[مُتَعَدٍّ]

ن276هـ=889م

انْتَحَى الشَّخْصَ: زَالَ عَنْهُ، وَابْتَعَدَ.

"يَبِيتُ مُنْتَحِيًا كَمَا يَنْتَحِي رَجُلٌ غَاضِبٌ أَهْلَهُ وَوَلَدَهُ، فَأَخَذَ كِسَاءَهُ وَبَاتَ وَحْدَهُ"
ابن قتيبة
كتاب المعاني الكبير في أبيات المعاني: ابن قتيبة (ت، 276هـ)، صححه: سالم الكرنكوى، دار النّهضة الحديثة، بيروت، د.ت. ‏2/‏729‏.

تَنَاحَى[مُتَعَدٍّ بِالحَرْف]

375هـ=985م

تَنَاحَى بِاللَّائِمَةِ وَنَحْوِهَا: أَلْقَى بِهَا عَلَى غَيْرِهِ.

"وَمَا أَسْتَقِلُّ بِحَمْلِ نِعْمَةٍ مِنْهُ إِلَّا أَثْقَلَنِي بِأُخْرَى، وَكَانَ، وَاللهِ، مَعَ هَذَا مِنْهَاجًا لِلْأُمُورِ المُشْكِلَةِ، إِذَا مَا تَنَاحَى ذَوُو الأَلْبَابِ بِاللَّائِمَةِ"
الجُنيد بن محمّد البغداديّ
البصائر والذّخائر: أبو حيّان التَّوحيديّ (ت، 414هـ)، تحقيق: وداد القاضي، دار صادر، بيروت، ط1، (1408هـ/ 1988م)، ‏1/‏152‏.

تَنَاحَى[لَازِم]

589هـ=1193م

تَنَاحَى الشَّيْءُ: تَبَاعَدَ.

"وَهَاجَتِ العَسَاكِر، وَمَاجَتِ الزَّوَاخِر، فَزَأَرَتِ القَسَاوِر، وَأَزْهَرَتِ الزَّوَاهِر، وَتَنَاحَتْ جَذَبَاتُ الحَدِيد وَعَذَبَاتُ الحَرِير"
عماد الدين الأصفهانيّ
حروب صلاح الدين: عماد الدين الأصفهانيّ (ت، 597هـ)، دار المنار، ط1، (1425ه/ 2004م)، ‏280‏.

مَنْحِيّ[صِفَة]

ن915هـ=1509م

المَنْحِيُّ مِنَ البُرُوقِ: البَعِيدُ.

أَشِيمُ البَارِقَ المَنْحِيَّ وَهْنًا وَمَنْ أَهْوَاهُ عَنْ سَهَرِي نَؤُومُ
سليمان بن سليمان النّبهاني
ديوان النّبهاني: سليمان بن سليمان النّبهاني (ت، 915هـ)، تحقيق: عز الدّين التّنوخيّ، وزارة التّراث والثَّقَافَة، مسقط، ط2، (1426ه/ 2005م)، ‏174‏.

مُتَنَاحِي[صِفَة]

ن1111هـ=1699م

المُتَنَاحِي مِنَ الأَمَاكِنِ: المُتَبَاعِدُ.

"ثُمَّ كَثُرَ الشِّجَارُ فِي كُلِّ نَاحِيَةٍ، وَطَلَبَهَا عِدَّةُ أَشْخَاصٍ فِي أَمَاكِنَ مُتَنَاحِيَةٍ "
عبد الملك بن حسين العصامي
سمط النجوم العوالي في أنباء الأوائل والتّوالي: عبد الملك بن حسين العصامي (ت، 1111هـ)، تحقيق وتعليق: عادل أحمد عبد الموجود، علي محمَّد معوض، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏4/‏200‏.

1411هـ=1990م
