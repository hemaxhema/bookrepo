حِقْد[مَصْدَر]

قبل 150ق.هـ=476م

الحِقْدُ : إِمْسَاكُ العَدَاوَةِ فِي القَلْبِ، وَالتَّرَبُّصُ بِفُرْصَتِهَا.

"وَالنَّصِيحَةُ تَجُرُّ الفَضِيحَةَ، وَالحِقْدُ يَمْنَعُ الرِّفْدَ"
منسوب إلى الحارث بن كعب المَذْحِجيّ
التّذكرة الحمدونيّة: ابن حمدون الكاتب (ت، 562هـ)، تحقيق: إحسان عبّاس، بكر عبّاس، دار صادر، بيروت، ط1، (1996م)، ‏3/‏342‏.

حِقْد[اسْم]

قبل 100ق.هـ=525م

الحِقْدُ : العَدَاوَةُ المُضْمَرَةُ.

تَقَطَّعَتِ الأَرْحَامُ مِنْهُمْ، وَبُدِّلَتْ ضَغَائِنَ حِقْدٍ ، بَعْدَ وُدٍّ، صُدُورُهَا
جليلة بنت مرّة الشَّيْبانيّة
ديوان بني بكر في الجاهلية: جمع وشرح وتوثيق ودراسة: عبد العزيز نبوي، دار الزهراء، القاهرة، ط1، (1410هـ/ 1989م)، ‏405‏.

أَحْقَدَ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

أَحْقَدَ الشَّخْصَ: حَمَلَهُ عَلَى الحِقْدِ.

لَا تَأْمَنَنَّ امْرَأً أَحْقَدْتَهُ أَبَدًا عَلَى ابْتِعَاثِ المَنَايَا الحُمْرِ، وَالسُّودِ
تأبَّط شرًّا الفَهْميّ
الأنس والعرس: أبو سعد الآبي الرّازيّ (ت، 421هـ)، إعداد وتحقيق: إيفلين فريد يارد، دار معد- دار النمير، دمشق، ط1، (1999م)، ‏193‏.

مَحْقِد[مَصْدَرٌ مِيمِيّ]

قبل 50ق.هـ=573م

المَحْقِدُ : المَدْعَاةُ إِلَى الحِقْدِ.

وَأُغْفِلُ لِلْمَوْلَى هَنَاةً تُرِيبُنِي فَأَظْلِمُهُ، مَا لَمْ يَنَلْنِي بِمَحْقِدِي
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: شرح: أشرف أحمد عدرة، دار الكتاب العربي، بيروت، ط1، (1414هـ/ 1994م)، ‏59‏.

مَحْقِد[اسْم]

ن385هـ=995م

المَحْقِدُ : الأَصْلُ وَالمَحْتِدُ.

"وَالمَحْقِدُ: المَحْتِدُ؛ رَجَعَ عَبْدُ سَوْءٍ إلى مَحْقِدِه "
الصّاحب ابن عَبّاد
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏2/‏339‏.

مُحْقِد[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

المُحْقِدُ : مَا يَدْعُو إِلَى الحِقْدِ.

وَأَغْفِرُ لِلْمَوْلَى هَنَاةً تَرِيبُنِي فَمَا ظُلْمُهُ، مَا لَمْ يَنَلْنِي، بِمُحْقِدِي
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏55‏.

حَقُود[صِيغَةُ مُبَالَغَة]

ن9هـ=630م

الحَقُودُ : الكَثِيرُ الحِقْدِ.

"أَقَلُّ النَّاسِ رَاحَةً الحَقُودُ "
أكثم بن صَيْفيّ الأُسَيِّديّ التّميميّ
كتاب جمهرة الأمثال: أبو هلال العسكريّ (ت، 395هـ)، حققه وعلق حواشيه ووضع فهارسه: محمّد أبو الفضل إبراهيم، عبد المجيد قطامش، دار الجيل- دار الفكر، بيروت، ط2، (1408هـ/ 1988م)، ‏1/‏495‏.

مَحْقُود[اسْمُ مَفْعُول]

ن11هـ=632م

المَحْقُودُ مِنَ الأَشْخَاصِ: المَمْلُوءُ حِقْدًا.

"لَا تَجُوزُ شَهَادَةُ ذِي الخُلَّةِ، وَلَا ذِي الجِنَّةِ، وَلَا ذِي الحِنَةِ المَحْقُودِ "
حديث نبويّ
السنن الكبير: أبو بكر البَيْهقيّ (ت، 458هـ)، تحقيق: مركز هجر للبحوث والدراسات العربية والإسلامية، هجر، القاهرة، ط1، (1432هـ/ 2011م)، ‏21/‏63،،‏(رقم الحديث: 20895).
ن538هـ=1143م

المَحْقُودُ عَليْهِ: المُمْتَلِئُ عَدُوُّهُ بِالغَيْظِ عَلَيْهِ.

"وَإِنَّ ذَا الشَّرَفِ مَحْسُودٌ، أَوْ حَاسِدٌ، وَمَحْقُودٌ عَلَيْهِ، أَوْ حَاقِدٌ"
جار الله الزمخشريّ
أطواق الذّهب في المواعظ والخطب: جار الله الزمخشريّ (ت، 538هـ)، دراسة وتحقيق: أَحمد عبد التواب عوض، دار الفضيلة، القاهرة، (د.ت)، ‏62‏.

حَاقَدَ[مُتَعَدٍّ]

ن11هـ=632م

حَاقَدَ الشَّخْصَ: جَارَاهُ فِي الحِقْدِ.

"وَإِذَا أَسَرُّوا عَلَيْكَ مِنَ الجَهَالَةِ، فَبِيِّنْ لَهُمْ حَتَّى يَعْرِفُوا، وَلَا تُحَاقِدْهُمْ ، وَأَمِتْ أَمْرَ الجَاهِلِيَّةِ"
حديث نبويّ
تاريخ مدينة دمشق وذكر فضلها وتسمية من حلها من الأماثل أو اجتاز بنواحيها من وارديها وأهلها: ابن عساكر (ت، 571هـ)، دراسة وتحقيق: عمر بن غرامة العمروي، دار الفكر، بيروت، ط1، (1415ه/ 1995م)، ‏58/‏412‏.

إِحْقَاد[مَصْدَر]

ن23هـ=644م

إِحْقَادُ الشَّخْصِ: حَمْلُهُ عَلَى الحِقْدِ.

"وَإِذَا عُومِلُوا بِخِلَافِ ذَلِكَ أَدَّاهُمْ إِلَى الحَنَقِ عَلَى السُّلْطَانِ، وَإِضْمَارِ السُّوءِ لَهُ؛ لِأَنَّ مَنْ رَأَى فِي نَفْسِهِ فَضْلًا مِنْ شَرَفٍ أَوْ عِلْمٍ أَوْ نَجْدَةٍ أَوْ مَجْدٍ أَوْ بَلَاءٍ أَوْ كِفَايَةٍ، فَجُهِلَ حَقُّهُ، وَحُرِمَ مِنْهُ مَا يَسْتَأْهِلُهُ وَيَسْتَحِقُّهُ، أَحْفَظَهُ ذَلِكَ إِحْفَاظًا، وَأَحْقَدَهُ إِحْقَادًا "
عمر بن الخطّاب
نصيحة الملوك: أبو الحسن الماورديّ (ت، 450هـ)، تحقيق: خضر محمّد خضر، مكتبة الفلاح، الكويت، ط1، (1403هـ/ 1983م)، ‏206‏.

حَقَدَ[مُتَعَدٍّ بِالحَرْف]

ن59هـ=679م

حَقَدَ عَلَيْهِ: أَضْمَرَ لَهُ العَدَاوَةَ فِي قَلْبِهِ، وَتَرَبَّصَ بِفُرْصَتِهَا.

"لَا تُمَازِحِ الشَّرِيفَ فَيَحْقِدَ عَلَيْك، وَلَا الدَّنِيءَ فَيَجْتَرِئَ عَلَيْك"
سعيد بن العاص الأُمويّ القُرَشيّ
كتاب الأمثال: أبو عُبيد الهَرَويّ (ت، 224هـ)، حقّقه وعلق عليه وقدّم له: عبد المجيد قطامش، دار المأمون للتراث، دمشق- بيروت، ط1، (1400هـ/ 1980م)، ‏86‏.

حَاقِد[اسْمُ فَاعِل]

65هـ=685م

الحَاقِدُ : المُضْمِرُ العَدَاوَةَ فِي قَلْبِهِ، المُتَرَبِّصُ بِفُرْصَتِهَا.

وَحَاقِدٍ أَزْمَعَ بِالتَّزَيُّلِ أَنْ يُخْبِرَ الإِمَامَ كُلَّ مَدْخَلِ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏1/‏312‏.

حَاقِد[صِفَةٌ مُشَبَّهَة]

200هـ=815م

الحَاقِدُ مِنَ المَعَادِنِ: المُنْقَطِعُ النَّيْلِ؛ فَلَا يَخْرُجُ مِنْهُ شَيْءٌ.

"وَإِذَا كَانَ المَعْدِنُ غَيْرَ حَاقِدٍ ، فَقَطَعَ العَامِلُ العَمَلَ فِيهِ ثُمَّ اسْتَأْنَفَهُ، لَمْ يَضُمَّ مَا أَصَابَ بِالعَمَلِ الآخَرِ إِلَى مَا أَصَابَ بِالعَمَلِ الأَوَّلِ قَلَّ قَطْعُهُ، أَوْ كَثُرَ"
الشّافعيّ
الأمّ: الشّافعيّ (ت، 204هـ)، تحقيق وتخريج: رفعت فوزي عبد المطلب، دار الوفاء، المنصورة، ط1، (1422هـ/ 2001م)، ‏3/‏112‏.

حَقِيدَة[اسْم]

ن80هـ=699م

الحَقِيدَةُ : العَدَاوَةُ الَّتِي تُضْمَرُ فِي القَلْبِ، وَيُتَرَبَّصُ بِفُرْصَتِهَا.

فَلَا تَأْسَ إِنْ صَدَّتْ سِوَاكَ، وَلَا تَكُنْ جَنِيبًا لِخَلَّاتٍ كَذُوبِ المَوَاعِدِ
وَعَدِّ إِلَى قَوْمٍ تَجِيشُ صُدُورُهُمْ بِغِشِّيَ لَا يُخْفُونَ حَمْلَ الحَقَائِدِ
أبو صخر الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏932‏-‏ 933‏.

أَحْقَد[اسْمُ تَفْضِيل]

ن96هـ=715م

الأَحْقَدُ : الأَشَدُّ حِقْدًا.

"وَمَا رَأَيْتُ عَظِيمَ الكِبْرِ صَاحِبَ حَرْبٍ إِلَّا كَانَ مَنْكُوبًا، لَا وَاللهِ، حَتَّى يَكُونَ عَدُوُّهُ عِنْدَهُ، وَخَصْمُهُ فِيمَا تَغَلَّبَ عَلَيْهِ، أَسْمَعَ مِنْ فَرَسٍ... وَأَحْقَدَ مِنْ جَمَلٍ"
قتيبة بن مسلم الباهليّ
جمهرة خطب العرب في عصور العربية الزاهرة: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1352هـ/ 1933م- 1352هـ/ 1934م)، ‏2/‏312‏-‏ 313‏.

تَحَقُّد[مَصْدَر]

ن110هـ=728م

التَّحَقُّدُ : إِضْمَارُ العَدَاوَةِ فِي القَلْبِ وَالتَّرَبُّصُ بِفُرْصَتِهَا.

بَاعَدْنَ، إِنَّ وِصَالَهُنَّ خِلَابَةٌ وَلَقَدْ جَمَعْنَ مَعَ البِعَادِ تَحَقُّدَا
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏2/‏376‏.

حَقِد[صِفَةٌ مُشَبَّهَة]

ن125هـ=743م

الحَقِدُ : المُضْمِرُ العَدَاوَةَ، المُتَرَبِّصُ بِفُرْصَتِهَا.

وَكَاشِحٍ مِنْ مَوَالِي السُّوءِ، ذِي حَسَدٍ يَقْتَاتُ لَحْمِي، وَمَا يَشفِيهِ مِنْ قَرَمِ
دَاوَيْتُ صَدْرًا، طَوِيلًا غِمْرُهُ، حَقِدًا مِنْهُ، وَقَلَّمْتُ أَظْفَارًا، بِلَا جَلَمِ
سالم بن وابصة الأسديّ
الوساطة بين المتنبّيّ وخصومه: القاضي الجُرْجانيّ (ت، 392هـ)، تحقيق وشرح: محمّد أبو الفضل إبراهيم، علي محمّد البجاوي، مطبعة عيسى البابي الحلبي وشركاه، القاهرة، د.ت. ‏392‏.

تَحَاقُد[مَصْدَر]

ن200هـ=815م

تَحَاقُدُ القَوْمِ: إِضْمَارُ بَعْضِهِمُ العَدَاوَةَ لِبَعْضٍ، وَتَرَبُّصُهُمْ بِفُرْصَتِهَا.

"وَالنَّكْبَةُ الرَّابِعَةُ: أَعْظَمُهَا عَلَى سَبِيلِ الظِّنَّةِ مِنَ الأَخِ الأَكْبَرِ؛ وَذَلِكَ أَنَّهُ سَيَمُرُّ بَيْنَهُمَا عَجَائِبُ مِنَ التَّحَاقُدِ وَالشُّرُورِ، فِي السِّرِّ وَالعَلَانِيَةِ بَيْنَهُمَا فَقَطْ"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏179‏.

حَقِدَ[لَازِم]

200هـ=815م

حَقِدَ المَعْدِنُ: انْقَطَعَ نَيْلُهُ؛ فَلَا يَخْرُجْ مِنْهُ شَيْءٌ.

"وَلَوْ تَابَعَ العَمَلَ فِي المَعْدِنِ فَحَقِدَ ، وَلَمْ يَقْطَعِ العَمَلَ فِيهِ، ضَمَّ مَا أَصَابَ مِنْهُ بِالعَمَلِ الآخَرِ إِلَى العَمَلِ الأَوَّلِ"
الشّافعيّ
الأمّ: الشّافعيّ (ت، 204هـ)، تحقيق وتخريج: رفعت فوزي عبد المطلب، دار الوفاء، المنصورة، ط1، (1422هـ/ 2001م)، ‏3/‏112‏.

مُحْتَقِد[اسْمُ فَاعِل]

ن207هـ=822م

المُحْتَقِدُ مِنَ الأَشْخَاصِ: الحَاقِدُ.

تَحَاقَدَ الشَّخْصُ: تَظَاهَرَ بِالحِقْدِ.

سَلِ الكَأْسَ لِمْ تُبْدِي لَنَا فِي خُدُودِنَا حَيَاءً، وَفِي أَفْعَالِنَا قِحَةً تُبْدِي؟
وَلَكِنْ إِذَا رَاحٌ وَرُوحٌ تَغَازَلَا، تَحَاقَدَ ذَاكَ الخَدُّ، وَاحْمَرَّ لِلْحِقْدِ

احْتَقَدَ[مُتَعَدٍّ]

352هـ=963م

احْتَقَدَ الضِّغْنَ وَنَحْوَهُ: أَضْمَرَهُ فِي قَلْبِهِ، وَتَرَبَّصَ بِفُرْصَتِهِ.

"إِذَا اسْتَوْحَشَ الرَّجُلُ مِنْ ذِي قَرَابَتِهِ، فَاضْطَغَنَ عَلَيْهِ سَخِيمَةً لِإِسَاءَةٍ كَانَتْ مِنْهُ إِلَيْهِ، فَأَوْحَشَتْهُ، ثُمَّ رَآهُ يُضَامُ، زَالَ عَنْ قَلْبِهِ مَا احْتَقَدَهُ عَلَيْهِ، وَغَضِبَ لَهُ، فَنَصَرَهُ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏4/‏460‏.

مُتَحَاقِد[اسْمُ فَاعِل]

369هـ=979م

المُتَحَاقِدُ مَعَ غَيْرِهِ: الَّذِي يُبَادِلُهُ العَدَاوَةَ المُضْمَرَةَ، وَيَتَرَبَّصُ لَهُ بِفُرْصَتِهَا.

"وَهِيَ خَرَابٌ بِكَثْرَةِ الفِتَنِ، وَاسْتِطَالَةِ العَامَّةِ، وَحُدُوثِ الحُرُوبِ فِيهَا، وَإِغَارَةِ بَعْضِهَا عَلَى بَعْضٍ، وَكَثْرَةِ رُؤَسَائِهِمُ النَّاجِمِينَ فِيهِمْ؛ حَتَّى حَصَلَ فِي كُلِّ مَحَلَّةٍ عِدَّةُ رُؤَسَاءَ مِنَ العَيَّارِينَ، يُحَامُونَ عَلَى مَحَلَّتِهِمْ، وَيَجْبُونَهُمُ الأَمْوَالَ، وَيُحَارِبُونَ مَنْ يَلِيهِمْ، فَهُمْ لِذَلِكَ مُتَحَاقِدُونَ "
ابن مسكويه
تجارِب الأُمَم وتعاقب الهِمَم: ابن مسكويه (ت، 421هـ)، حققه وقدم له: أبو القاسم إمامي، دار سروش، طهران، ط2، (1422هـ/ 2001م)، ‏6/‏347‏.
