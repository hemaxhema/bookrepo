ضَنَّ[مُتَعَدٍّ بِالحَرْف]

قبل 150ق.هـ=476م

ضَنَّ بِالشَّيْءِ: بَخِلَ بِهِ.

"إِذَا مِتُّ فَأَرْحِبُوا خَطَّ مَضْجَعِي، وَلَا تَضَنُّوا عَلَيَّ بِرُحْبِ الأَرْضِ"
دويد بن زيد القُضاعيّ
أمالي المرتضى غُرر الفوائد ودُرر القلائد: الشّريف المرتضى (ت، 436هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، ط1، (1373هـ/ 1954م)، ‏1/‏236‏.
ن11هـ=632م

ضَنَّ اللهُ بِالشَّيْءِ: اسْتَأْثَرَ بِهِ.

" ضَنَّ رَبُّكَ   بِمَفَاتِيحِ خَمْسٍ مِنَ الغَيْبِ لَا يَعْلَمُهَا إِلَّا اللهُ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏26/‏122،،‏(رقم الحديث: 16206).

ضِنّ[مَصْدَر]

قبل 150ق.هـ=476م

الضِّنُّ : الحِرْصُ الشَّدِيدُ.

فَإِنْ غُلِبْتُمْ، عَلَى ضِنٍّ بِدَارِكُمُ، فَقَدْ لَقِيتُمْ بِأَمْرٍ حَازِمٍ فَزَعَا
لَقيط بن يَعْمَر الإياديّ
ديوان لقيط بن يَعمر: حققه وقدّم له: عبد المعيد خان، دار الأمانة- مؤسّسة الرسالة، بيروت، (1391هـ/ 1971م)، ‏53‏.

ضِنّ[صِفَةٌ مُشَبَّهَة]

ن98هـ=717م

الضِّنُّ مِنَ النَّاسِ وَالأَشْيَاءِ: مَا يُحْرَصُ عَلَيْهِ وَيُمْسَكُ لِنَفَاسَتِهِ.

زَعُمْتَ، فَإِنْ تَلْحَقْ، فَضِنٌّ مُبَرِّزٌ جَوَادٌ، وَإِنْ تُسْبَقْ فَنَفْسَكَ أَعْوِلِ
عُبيد الله بن عبد الله بن عُتبة الهُذَليّ
مجالس ثعلب: أبو العبّاس ثعلب (ت، 291هـ)، شرح وتحقيق: عبد السلام هارون، دار المعارف، القاهرة، ط2، (1948م)، ‏1/‏15‏.
ن175هـ=791م

الضِّنُّ مِنَ النَّاسِ: الَّذِي يُخْتَصُّ بِهِ وَيُضَنُّ بِمَوَدَّتِهِ.

"وَهَذَا ضِنِّي مِنْ بَيْنِ إِخْوَانِي، أَيْ: أَخْتَصُّ بِهِ وَأَضِنُّ بِمَوَدَّتِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏7/‏11‏.

ضَنِين[اسْمُ مَفْعُول]

قبل 100ق.هـ=525م

الضَّنِينُ : مَا يُحْرَصُ عَلَيْهِ وَيُبْخَلُ بِهِ لِنَفَاسَتِهِ.

أَجِبْنِي، يَا كُلَيْبُ، خَلَاكَ ذَمٌّ ضَنِينَاتُ النُّفُوسِ لَهَا مَزَارُ
مُهَلْهِل بن ربيعة التَّغْلِبيّ
ديوان المُهَلْهِل: شرح وتحقيق: أنطوان محسن القوَّال، دار الجيل، بيروت، ط1، (1415هـ/ 1995م)، ‏29‏.
ن11هـ=632م

الضَّنِينُ : الَّذِي يُخْتَصُّ بِمَوَدَّتِهِ.

"إِنَّ للهِ ضَنَائِنَ مِنْ خَلْقِهِ، يُحْيِيهِمْ فِي عَافِيَةٍ، وَإِذَا تَوَفَّاهُمْ إِلَى جَنَّتِهِ، أُولَئِكَ الَّذِينَ يَمُرُّ عَلَيْهِمُ الفِتَنُ كَقِطَعِ اللَّيْلِ المُظْلِمِ، وَهُمْ فِيهَا فِي عَافِيَةٍ"
حديث نبويّ
المعجم الكبير: أبو القاسم الطَّبَرانيّ (ت، 360هـ)، حققه وخرج أحاديثه: حمدي عبد المجيد السلفي، مكتبة ابن تيمية، القاهرة، ط2، (1404هـ/ 1983م)، ‏12/‏385،،‏(رقم الحديث: 13425).

ضَنِين[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الضَّنِينُ : الحَرِيصُ الَّذِي اشْتَدَّ بُخْلُهُ.

قَدْ تُخْرِجُ الخَمْرُ مِنَ الضَّنِينِ
زهير بن جَناب الكلبيّ
ديوان زُهير بن جَناب الكلبيّ: صنعة: محمّد شفيق البيطار، دار صادر، بيروت، ط1، (1999م)، ‏107‏.
ن638هـ=1240م

الضَّنِينُ : مَا يُخْتَصُّ بِهِ مِنَ عِلْمٍ وَنَحْوِهِ.

وَأَنْتَ خَفِيٌّ فِي ضَنَائِنِ غَيْبِهِ وَلَسْتَ جَلِيًّا فَالمُبِينُ هُوَ اللهُ
محيي الدين ابن عربيّ
ديوان ابن عربي: محيي الدين ابن عربيّ (ت، 638هـ)، شرحه: أحمد حسن بسج، دار الكتب العلمية، بيروت، ط1، (1416ه/ 1996م)، ‏196‏.

ضَنِين[صِيغَةُ مُبَالَغَة]

1ق.هـ=621م

الضَّنِينُ : الكَتُومُ لِمَا أُوحِيَ إِلَيْهِ.

قرآن كريم
قرآن كريم، ‏[ التَّكْوِير : 24 ].

مَضِنَّة[مَصْدَرٌ مِيمِيّ]

قبل 100ق.هـ=525م

مَضِنَّةُ الشَّيْءِ: نَفَاسَتُهُ الدَّافِعَةُ إِلَى الحِرْصِ عَلَيْهِ وَالبُخْلِ بِهِ.

يَا كَعْبُ، إِنَّ أَخَاكَ مُنْحَمِقٌ إِنْ لَمْ تَكُنْ بِكَ مِرَّةٌ، كَعْبُ
أَتَجُودُ بِالدَّمِ ذِي المَضِنَّةِ فِي الـ ـجُلَّى، وَتُلْوَى النَّابُ وَالسَّقْبُ؟
ذؤيب بن كعب التَّميميّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏483‏.

مَضْنُون[اسْمُ مَفْعُول]

قبل 13ق.هـ=609م

المَضْنُونُ بِهِ: القَلِيلُ المَبْخُولُ بِهِ.

وَأُبْرِزُ قِدْرِي بِالفَضَاءِ، قَلِيلُهَا يُرَى غَيْرَ مَضْنُونٍ بِهِ، وَكَثِيرُهَا
حاتم بن عبد الله الطّائيّ
ديوان شعر حاتم بن عبد الله الطائي وأخباره: صنعة: يحيى بن مدرك الطائي، رواية: هشام بن محمّد الكلبي، دراسة وتحقيق: عادل سليمان جمال، مطبعة المدني، القاهرة، د.ت. ‏246‏.

مَضْنُون[اسْم]

ن244هـ=858م

المَضْنُونُ : نَوْعٌ مِنَ الطِّيبِ يُسَمَّى الغَالِيَةَ.

قَدْ أَكْنَبَتْ يَدَاكَ بَعْدَ لِينِ وَبَعْدَ دُهْنِ البَانِ وَالمَضْنُونِ
من شواهد ابن السِّكّيت
إصلاح المنطق: ابن السّكّيت (ت، 244هـ)، شرح وتحقيق: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط4، (1368هـ/ 1949م)، ‏2/‏412‏.

ضِنَّة[مَصْدَر]

ن2ق.هـ=620م

الضِّنَّةُ : شِدَّةُ الحِرْصِ وَالبُخْلِ.

يَجُودُ وَيُعْطِي المَالَ مِنْ غَيْرِ ضِنَّةٍ وَيَضْرِبُ أَنْفَ الأَبْلَخِ المُتَغَشِّمِ
أوس بن حَجَر المازنيّ التَّميميّ
ديوان أوس بن حجر: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، ط3، (1399هـ/ 1979م)، ‏118‏.

ضَنَانَة[مَصْدَر]

ن55هـ=675م

الضَّنَانَةُ بِالشَّيْءِ: الحِرْصُ عَلَيْهِ وَالبُخْلُ بِهِ لِنَفَاسَتِهِ.

وَمَنْ يُبْقِ مَالًا عُدَّةً وَضَنَانَةً فَلَا الشُّحُّ مُبْقِيهِ، وَلَا الدَّهْرُ وَافِرُهْ
توبة بن الحُمَيِّر الخَفاجيّ العُقيليّ
ديوان توبة بن الحُمَيِّر: عني بتحقيقه وشرحه: خليل إبراهيم العطية، دار صادر، بيروت، ط1، (1998م)، ‏46‏.

أَضَنّ[اسْمُ تَفْضِيل]

ن64هـ=684م

الأَضَنُّ بِالشَّيْءِ: الأَشَدُّ حِرْصًا عَلَيْهِ وَبُخْلًا بِهِ.

وَزَعَمْتِ أَهْلَكِ يَمْنَعُونَكِ رَغْبَةً عَنِّي، وَأَهْلِي بِي أَضَنُّ وَأَرْغَبُ
يزيد بن معاوية بن أبي سفيان
شعر يزيد بن معاوية بن أبي سفيان: جمعه وحقّقه: صلاح الدين المنجد، دار الكتاب الجديد، بيروت، ط1، (1982م)، ‏12‏.

مَضْنُونَة[اسْم]

ن90هـ=709م

المَضْنُونَةُ : نَوْعٌ مِنَ الطِّيِبِ يُسَمَّى الغَالِيَةَ.

تَضُمُّ عَلَى مَضْنُونَةٍ فَارِسِيَّةٍ ضَفَائِرَ لَا ضَاحِي القُرُونِ، وَلَا جَعْدِ
الرّاعي النُّمَيريّ
ديوان الراعي النُّمَيْري: جمعه وحققه: راينهرت فايبرت، المعهد الألماني للأبحاث الشرقيّة، بيروت، (1401هـ/ 1980م)، ‏74‏.

أَضَنَّ[مُتَعَدٍّ بِالحَرْف]

161هـ=778م

أَضَنَّ بِالشَّيْءِ: بَخِلَ بِهِ.

أَقُولُ لِلسَّاقِي وَقَدْ مَاثَهُ: أَضْنِنْ بِمَا ضَنَّ بِهِ المَائِثُ
بشّار بن برد
ديوان بشار بن برد: قرأه وقدم له: إحسان عبّاس، دار صادر، بيروت، ط1، (2000م)، ‏162‏.

ضَنَّان[صِيغَةُ مُبَالَغَة]

ن198هـ=813م

الضَّنَّانُ بِالشَّيْءِ: الشَّدِيدُ الضَّنِّ بِهِ لِنَفَاسَتِهِ.

قَدْ أَغْتَدِي بِزُرَّقٍ صَبِيحِ بِكَفِّ ضَنَّانٍ بِهِ شَحِيحِ مِمَّا اشْتَرَى بِالثَّمَنِ الرَّبِيحِ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏651‏.

ضَنَن[صِفَةٌ مُشَبَّهَة]

280هـ=893م

الضَّنَنُ مِنَ الرِّجَالِ: الشُّجَاعُ.

إِنِّي إِذَا ضَنَنٌ يَمْشِي إِلَى ضَنَنٍ أَيْقَنْتُ أَنَّ الفَتَى مُودٍ بِهِ المَوْتُ
من شواهد ابن دريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏2/‏1011‏.

تَضَنُّن[مَصْدَر]

ن656هـ=1258م

التَّضَنُّنُ : الحِرْصُ وَالبُخْلُ بِالشَّيْءِ النَّفِيسِ.

أَصُدُّ بِنَفْسِي عَنْ سِوَاكَ تَضَنُّنًا وَأَبْذُلُهَا لِلْوَجْدِ فِيكَ تَكَرُّمَا
الملك الناصر داود
الفوائِد الجليّة في الفرائِد النّاصريّة (رسائله وشعره): الملك الناصر داود (ت، 656هـ)، تحقيق: ناظم رشيد، دار الشُّؤون الثَّقافيَّة العامة، بغداد، ط1، (1416ه/ 1992م)، ‏322‏.

مُسْتَضَنّ[اسْمُ مَفْعُول]

ن1418هـ=1997م

المُسْتَضَنُّ : مَا يُحْرَصُ عَلَيْهِ لِنَفَاسَتِهِ.

أَنَا عِرْقٌ فِي جِسْمِكَ النَّابِضِ الحَيْـ يِ، وَلَمْحٌ مِنْ عِلْقِكَ المُسْتَضَنِّ
محمد مهدي الجواهري
الأعمال الشعرية الكاملة، محمد مهدي الجواهري: محمد مهدي الجواهري (ت، 1418هـ)، دراسة وتقديم: عصام عبد الفتاح، مكتبة جزيرة الورد، القاهرة، ط1، (1432ه/ 2011م)، ‏2/‏282‏.
