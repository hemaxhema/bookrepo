عِذْق[اسْم]

قبل 100ق.هـ=525م

العِذْقُ : النَّخْلَةُ بِحَمْلِهَا.

إِذَا مَا جِئْتُهَا قَدْ بِعْتُ عِذْقًا تُعَانِقُ، أَوْ تُقَبِّلُ، أَوْ تُفَدِّي
أُحَيحة بن الجُلاح الأوسيّ
الأصمعيّات: اختيار: الأصمعيّ (ت، 216هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السلام هارون، دار المعارف، القاهرة، ط3، (1387هـ/ 1967م)، ‏120‏.
قبل 100ق.هـ=525م

العِذْقُ : العُنْقُودُ مِنَ التَّمْرِ وَنَحْوِهِ.

قَتَلْنَا الحَارِثَ الوَضَّاحَ مِنْهُمْ فَخَرَّ، كَأَنَّ لِمَّتَهُ العُذُوقُ
المفضّل بن معشر النُّكْريّ
شعراء عبد القيس وشعرهم في العصر الجاهلي: جمع وتحقيق ودراسة: عبد الحميد المعيني، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري، الكويت، (2002م)، ‏327‏.
4هـ=625م

عِذْقُ زَيْدٍ : صِنْفٌ رَدِيءٌ مِنَ التَّمْرِ.

"اذْهَبْ، فَصَنِّفْ تَمْرَكَ أَصْنَافًا: العَجْوَةَ عَلَى حِدَةٍ، وَعِذْقَ زَيْدٍ عَلَى حِدَةٍ، وَأَصْنَافَهُ، ثُمَّ ابْعَثْ إِلَيَّ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏22/‏260،،‏(رقم الحديث: 14359).
ن37هـ=657م

العِذْقُ : الأَصْلُ العَرِيقُ.

وَفِي غَطَفَانَ عِذْقُ عِزٍّ مُمَنَّعٌ، عَلَى رَغْمِ أَقْوَامٍ مِنَ النَّاسِ، يَانِعُ
ابن مقبل العَجْلانيّ التَّميميّ
ديوان ابن مقبل: عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏260‏.
ن68هـ=687م

عِزْقُ التَّمْرِ وَنَحْوِهِ: عُنْقُودُهُ ( عِزْقٌ لُغَةٌ في: عِذْقٍ ).

"فَجَاءَ رَجُلٌ بِعِزْقٍ مِنْ تَمْرٍ، عَامَّتُهُ حَشَفٌ، فَوَضَعَهُ فِي المَسْجِدِ مَعَ التَّمْرِ"
ابن عبّاس
تفسير مقاتل بن سليمان: مقاتل بن سليمان الأزديّ (ت، 150هـ)، دراسة وتحقيق: عبد الله محمود شحاتة، مؤسسة التاريخ العربي، بيروت، ط1، (1423هـ/ 2002م)، ‏1/‏222‏.
86هـ=705م

عَذْقُ ابْنِ طَابٍ : ضَرْبٌ حُلْوٌ مِنَ التَّمْرِ.

وَهُمْ أَحْلَى إِذَا مَا لَمْ تُثِرْهُمْ عَلَى الأَحْنَاكِ مِنْ عَذْقِ ابْنِ طَابِ
كُثَيِّر عَزّة
ديوان كُثَيِّر عَزَّة: جمعه وشرحه: إحسان عبّاس، دار الثقافة، بيروت، (1391هـ/ 1971م)، ‏282‏.
ن86هـ=705م

العِذْقُ : الغُصْنُ الَّذِي لَهُ شُعَبٌ.

"إِنَّ أَخًا لَكُمْ أُرِيَ فِي المَنَامِ أَنَّ النَّاسَ يُسْلَكُونَ فِي صَدْعِ جَبَلٍ وَعْرٍ طَوِيلٍ، وَعَلَى رَأْسِ الجَبَلِ شَجَرَتَانِ خَضْرَاوَانِ تَهْتِفَانِ: هَلْ فِيكُمْ مَنْ يَقْرَأُ سُورَةَ البَقَرَةِ؟ هَلْ فِيكُمْ مَنْ يَقْرَأُ سُورَةَ آلِ عِمْرَانَ؟ فَإِذَا قَالَ الرَّجُلُ: نَعَمْ، دَنَتَا بِأَعْذَاقِهِمَا حَتَّى يَتَعَلَّقَ بِهِمَا، فَتَخْطُرَانِ بِهِ الجَبَلَ"
أبو أُمامة الباهليّ
مُسند الدّارمي المعروف بـ(سنن الدارمي): الدّارمي (ت، 255هـ)، تحقيق: حسين سليم أسد الداراني، دار المغني، الرياض، ط1، (1421هـ/ 2000م)، ‏4/‏2136،،‏(رقم الحديث: 3435).
352هـ=963م

عِذْقُ ابْنِ حُبَيْقٍ: صِنْفٌ رَدِيءٌ مِنَ التَّمْرِ.

"وَيُقَالُ لِنَخْلَتِهِ: عَذْقُ ابْنِ حُبَيْقٍ ، وَلَيْسَ بِشَيْصٍ، وَلَكِنَّهُ رَدِيءٌ مِنَ الدَّقَلِ"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏3/‏382‏.

عِذْق[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

العَذْقُ مِنَ النَّبَاتِ: ذُو الأَغْصَانِ.

" وَالعَذْقُ مِنَ النَّبَاتِ: ذُو الأَغْصَانِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏148‏.

مُعْذِق[اسْمُ فَاعِل]

12ق.هـ=610م

المُعْذِقُ مِنَ الأَشْجَارِ وَنَحْوِهَا: الكَثِيرُ الأَغْصَانِ وَالثِّمَارِ.

"وَاللهِ، إِنَّ لِقَوْلِهِ الَّذِي يَقُولُهُ لَحَلَاوَة، وَإِنَّ عَلَيْهِ لَطُلَاوَة، وَإِنَّهُ لَمُثْمِرٌ أَعْلَاهُ، مُعْذِقٌ أَسْفَلُهُ، وَإِنَّهُ لَيَحْطِمُ مَا تَحْتَهُ، وَإِنَّهُ لَيَعْلُو، وَمَا يُعْلَى"
الوليد بن المغيرة المخزوميّ
تفسير القرآن: عبد الرّزّاق بن هَمّام الصَّنْعانيّ (ت، 211هـ)، تحقيق: مصطفى مسلم محمد، مكتبة الرشد، ط1، (1410هـ/ 1989م)، ‏2/‏328‏.
36هـ=656م

مُعْذِقُ الإِبِلِ وَنَحْوِهَا: مَنْ يَضَعُ عَلَيْهَا عَلَامَةً مِنْ غَيْرِ لَوْنِهَا؛ يُمَيِّزُهَا بِهَا.

"بَيْنَمَا نَحْنُ جُلُوسٌ مَعَ رَسُولِ اللهِ  @  إِذْ مَرَّ بِنَا عُثْمَانُ بْنُ عَفَّانَ مُرَحِّلًا مُعْذِقًا ، فَقَالَ رَسُولُ اللهِ  @ :  لَتَخْرُجَنَّ فِتْنَةٌ تَحْتَ رِجْلَيْ، أَيْ: مِنْ تَحْتِ قَدَمَيْ هَذَا، وَهَذَا يَوْمَئِذٍ، وَمَنِ اتَّبَعَهُ، عَلَى الهُدَى"
مرّة بن كعب البَهْزيّ السُّلَميّ
كتاب تاريخ المدينة المنوّرة: ابن شَبَّة (ت، 262هـ)، حققه: فهيم محمّد شلتوت، (1399هـ/ 1979م)، ‏3/‏1103‏.
ن150هـ=767م

المُعْذِقُ مِنَ السُّحُبِ: الغَزِيرُ الوَدْقِ.

وَتَذَكَّرَتْ مُتَرَبَّعًا مِنْ أَرْضِهِ بَرَدَتْ شَمَائِمُهُ، وَجَالَ سَحَابُهُ
كَمْ قَدْ أَرَبَّ بِجَوِّهِ مِنْ مُعْذِقٍ مُتَهَزِّمٍ، قَرِدٍ، يَطِيرُ رَبَابُهُ
معن بن زائدة الشَّيْبانيّ
كتاب الأزمنة والأمكنة: أبو عليّ المرزوقيّ (ت، 421هـ)، ضبطه وخرج آياته: خليل المنصور، دار الكتب العلمية، بيروت، ط1، (1417هـ/ 1996م)، ‏451‏.

أَعْذَقَ[لَازِم]

6هـ=627م

أَعْذَقَ الإِذْخِرُ وَنَحْوُهُ: خَرَجَ ثَمَرُهُ.

"تَرَكْتُهَا وَقَدْ تَيَسَّرَتْ، قَدْ أَمْشَرَ عِضَاهُهَا، وَأَعْذَقَ إِذْخِرُهَا، وَأَسْلَبَ ثُمَامُهَا، وَأَبْقَلَ حَمْضُهَا"
غلام من خُزاعة
كتاب المغازي: الواقديّ (ت، 207هـ)، تحقيق: مارسدن جونس، عالم الكتب، القاهرة، ط3، (1404هـ/ 1984م)، ‏2/‏592‏.

عُذَيْق[اسْمٌ مُصَغَّر]

ن11هـ=632م

العُذَيْقُ : تَصْغِيرُ العِذْقِ.

"أَنَا جُذَيْلُهَا المُحَكَّكُ، وَعُذَيْقُهَا المُرَجَّبُ"
الحُباب بن المنذر الخزرجيّ الأنصاريّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏1/‏454،،‏(رقم الحديث: 391).

عُذَيْق[اسْم]

ن201هـ=816م

العُذَيْقُ : النَّخْلَةُ الَّتِي يَكْثُرُ حِمْلُهَا.

"أَكْرَمُ مِنَ العُذَيْقِ المُرَجَّبِ"
مثل
مجمع الأمثال: أبو الفضل الميدانيّ (ت، 518هـ)، حققه وفصله وضبط غرائبه وعلق حواشيه: محمّد محيي الدين عبد الحميد، مطبعة السنة المحمدية، القاهرة، (1374هـ/ 1955م)، ‏2/‏170‏.

عَاذِق[اسْمُ فَاعِل]

ن26هـ=647م

العَاذِقُ : مَنْ يُشَذِّبُ النَّخْلَ وَنَحْوَهُ.

تَنْجُو، وَتَقْطُرُ ذِفْرَاهَا عَلَى عُنُقٍ كَالجِذْعِ، شَذَّبَ عَنْهُ عَاذِقٌ سَعَفَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏81‏.

عَذَّقَ[مُتَعَدٍّ]

ن26هـ=647م

عَذَّقَ النَّخْلَةَ وَنَحْوَهَا: قَطَّعَ سَعَفَهَا، وَشَذَّبَهَا.

كَالجِذْعِ عَذَّقَ عَنْهُ عَاذِقٌ سَعَفَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏4/‏1522‏.

عَذَقَ[مُتَعَدٍّ]

ن337هـ=948م

" وَعَذَقَهُ بِمَسَبَّةٍ وَعَيْبٍ"
قُدامة بن جعفر البغداديّ
جواهر الألفاظ: أبو الفرج قُدامة بن جعفر البغداديّ (ت، 337هـ)، تحقيق: محمّد محيي الدين عبد الحميد، دار الكتب العلمية، بيروت، ط1، (1405هـ/ 1985م)، ‏44‏.
467هـ=1074م

عَذَقَ بِهِ أَمْرًا: عَلَّقَهُ بِهِ.

زَادَتْ حُلَاهَا، وَلَوْ جَاءَتْكَ عَاطِلَةً مِنْ خَالِصِ التِّبْرِ، مَا أَزْرَى بِهَا العَطَلُ
وَرَاءَهَا عَلَمَا النَّصْرِ اللَّذَا كَفَلَا لِمَنْ أَظَلَّا، بِعِزٍّ لَيْسَ يُنْتَحَلُ
مِنْ عَقْدِ مَنْ عُذِقَ النَّصْرُ العَزِيزُ بِهِ فَمَا لَهُ، أَبَدًا، عَنْ ظِلِّهُ حِوَلُ
ابن حَيّوس الغَنَويّ الدِّمشقيّ
ديوان ابن حيّوس: عني بنشره وتحقيقه: خليل مردم بك، دار صادر، بيروت، (1404هـ/ 1984م)، ‏2/‏511‏.
469هـ=1076م

عَذَقَ بِهِ عَمَلًا وَنَحْوَهُ: أَنَاطَ بِهِ عَمَلًا.

"فَإِنَّ أَمِيرَ المُؤْمِنِينَ لِمَا يَعْلَمُهُ مِنْ خُلُوصِ طَاعَتِكَ وَضَمِيرِك، وَيَعْرِفُهُ مِنْ تَحَلِّيكَ بِحُسْنِ الطَّرِيقَةِ فِي سِيَاسَتِكَ وَتَدْبِيرِك، لَا يَزَالُ يَعْذِقُ بِكَ مِنَ الأَعْمَالِ خَطِيرا، وَيَنُوطُ بِكَ مِنْ أَقْطَارِهَا فِي كُلِّ وَقْتٍ قُطْرًا كَبِيرا"
المستنصر بالله الفاطميّ
السِّجِلّات المستنصريّة: المستنصر بالله الفاطميّ (ت، 487هـ)، تقديم وتحقيق: عبد المنعم ماجد، دار الفكر العربي- مطبعة الاعتماد، القاهرة، (1374هـ/ 1954م)، ‏176‏.
478هـ=1085م

عَذَقَهُ : عَهِدَهُ.

"وَإِنْ كَانَ أَمِيرُ المُؤْمْنِينَ قَدْ فَوَّضَ إِلَى وَلَدِكِ الأَمْرَ، وَارْتَضَاهُ وَاسْتَرْجَحَهُ؛ لِمَا عَذَقَهُ بِهِ مِنْ ضَمِّ شَمْلِ المُؤْمْنِينَ وَاسْتَرْعَاهُ؛ فَإِنَّهُ اسْتَكْفَلَكِ بِشَدِيدِهِ"
المستنصر بالله الفاطميّ
السِّجِلّات المستنصريّة: المستنصر بالله الفاطميّ (ت، 487هـ)، تقديم وتحقيق: عبد المنعم ماجد، دار الفكر العربي- مطبعة الاعتماد، القاهرة، (1374هـ/ 1954م)، ‏164‏.
