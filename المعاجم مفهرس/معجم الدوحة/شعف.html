مَشْعُوف[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المَشْعُوفُ : المَذْعُورُ ذُعْرًا أَذْهَبَ لُبَّهُ.

وَحَثْحَثْتُ مَشْعُوفَ النَّجَاءِ، وَرَاعَنِي أُنَاسٌ بِفَيْفَانٍ، فَمِزْتُ القَرَائِنَا
تأبَّط شرًّا الفَهْميّ
ديوان تأبط شرًّا وأخباره: جمع وتحقيق وشرح: علي ذو الفقار شاكر، دار الغرب الإسلاميّ، بيروت، ط1، (1404هـ/ 1984م)، ‏216‏.
قبل 13ق.هـ=609م

المَشْعُوفُ بِالشَّيْءِ: المُولَعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

أَصَوْتَ مُنَادٍ مِنْ رُمَيْلَةَ تَسْمَعُ بِغَوْلٍ وَدُونِي بَطْنُ فَلْجٍ، فَلَعْلَعُ
أَمِ اسْتَحْقَبَ الشَّوْقَ الفُؤَادُ؟ فَإِنَّنِي، وَجَدِّكَ، مَشْعُوفٌ بِرَمْلَةَ، مُوجَعُ
بشر بن أبي خازم الأسديّ
ديوان بشر بن أبي خازم الأسدي: عني بتحقيقه: عزة حسن، وزارة الثقافة والإرشاد القومي، دمشق، (1379هـ/ 1960م)، ‏118‏.

شَعَفَ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

شَعَفَهُ الحُبُّ وَنَحْوُهُ: أَحْرَقَ قَلْبَهُ، وَأَذْهَبَ لُبَّهُ.

فَلَمْ تَدَعِ الأَرْوَاحُ وَالمَاءُ وَالبِلَى مِنَ الدَّارِ إِلَّا مَا يَشُوقُ وَيَشْعَفُ
عبيد بن عبد العُزَّى السَّلاميّ الأزديّ
منتهى الطلب من أشعار العرب: جمع: ابن ميمون البغدادي (ت، 597هـ)، تحقيق وشرح: محمّد نبيل طريفي، دار صادر، بيروت، ط1، (1999م)، ‏8/‏281‏.
قبل 50ق.هـ=573م

شَعَفَهُ : ذَعَرَهُ وَاسْتَخَفَّهُ.

لِتَقْتُلَنِي، وَقَدْ شَعَفْتُ فُؤَادَهَا كَمَا شَعَفَ المَهْنُوءَةَ الرَّجُلُ الطَّالِي
امرؤ القيس بن حُجْر الكنديّ
كتاب خلق الإنسان: ثابت بن أبي ثابت اللُّغويّ (ت، 250هـ)، تحقيق: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، ط2، د.ت. ‏261‏.

شَاعِف[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الشَّاعِفُ الشَّخْصَ: الَّذِي أَحْرَقَ قَلْبَهُ، وَأَذْهَبَ لُبَّهُ.

وَفِي الحَيِّ أَبْكَارٌ سَبَيْنَ فُؤَادَهُ عُلَالَةَ مَا زَوَّدْنَ، وَالحُبُّ شَاعِفِي
المُرَقِّش الأكبر البكريّ
المفضليات: المفضَّل الضَّبّيّ (ت، 178هـ)، تحقيق وشرح: أحمد محمد شاكر، عبد السّلام هارون، دار المعارف، القاهرة، ط6، د.ت. ‏231‏.
ن120هـ=738م

الشَّاعِفُ بِالشَّيْءِ: الوَلِعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

رَمِيٌّ بِذِكْرٍ مِنْ حَبِيبٍ أَصَابَهُ عَلَى النَّأْيِ وَالهِجْرَانِ، فَالقَلْبُ شَاعِفُ
مُزاحم العُقيليّ
شعر مزاحم العقيلي: تحقيق: نوري حمّودي القيسيّ، حاتم الضامن، مجلة معهد المخطوطات العربية، القاهرة، مج22، ج1، (1976م)، ‏105‏.

شَعَف[اسْمُ جِنْس]

قبل 13ق.هـ=609م

الشَّعَفُ (الشَّعْفُ: ضَرُورَةٌ): رُؤُوسُ الجِبَالِ وَنَحْوِهَا.

كُونُوا كَسَامَةَ إِذْ شَعْفٌ مَنَازِلُهُ إِذْ قِيلَ: جَيْشٌ، وَجَيْشٌ حَافِظٌ رَصَدُ
المُتَلَمِّس الضُّبَعيّ
ديوان شعر المتلمِّس الضُّبَعِيّ: رواية: الأثرم وأبي عُبيدة عن الأصمعيّ، عني بتحقيقه وشرحه والتّعليق عليه: حسن كامل الصيرفي، معهد المخطوطات العربية، القاهرة، (1390هـ/ 1970م)، ‏212‏.
ن11هـ=632م

الشَّعَفَ : خُصَلُ الشَّعْرِ فِي أَعْلَى الرَّأْسِ.

"وَإِنَّكُمْ لَا تَزَالُونَ تُقَاتِلُونَ عَدُوًّا، حَتَّى يَأْتِيَ يَأْجُوجُ وَمَأْجُوجُ، عِرَاضُ الوُجُوهِ، صِغَارُ العُيُونِ، صُهْبُ الشِّعَافِ ، مِنْ كُلِّ حَدَبٍ يَنْسِلُونَ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏37/‏19،،‏(رقم الحديث: 22331).

شَعَف[اسْم]

ن26هـ=647م

الشَعَفُ : أَوَّلُ مَا يَنْبُتُ مِنَ الرِّيشِ.

حُمْرٌ حَوَاصِلُهَا، كَالمَغْدِ قَدْ كُسِيَتْ فَوْقَ الحَوَاجِبِ مِمَّا سَبَّدَتْ شَعَفَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏79‏.

شَعَف[مَصْدَر]

ن50هـ=670م

الشَّعَفُ بِالشَّيْءِ: الوَلَعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

فَقَدْ تَرُوعُ الغَوَانِي طَلْعَتِي شَعَفًا يَنْصُصْنَ أَجْيَادَ أُدْمٍ تَرْتَعِي ضَالَا
النّابغة الجَعْديّ
شعر النابغة الجعدي: تحقيق: عبد العزيز رباح، منشورات المكتب الإسلامي، دمشق، ط1، (1384هـ/ 1964م)، ‏100‏.

اشْتَعَفَ[لَازِم]

قبل 13ق.هـ=609م

اشْتَعَفَ الشَّخْصُ: أَحْرَقَ الوَلَعُ قَلْبَهُ، وَأَذْهَبَ لُبَّهُ.

قَدْ مَلَّتِ الحَبْسَ بِالآطَامِ وَاشْتَعَفَتْ
النّابغة الذُّبْيانيّ
كتاب تاريخ المدينة المنوّرة: ابن شَبَّة (ت، 262هـ)، حققه: فهيم محمّد شلتوت، (1399هـ/ 1979م)، ‏1/‏289‏.

مُشْعِف[اسْمُ فَاعِل]

8هـ=629م

المُشْعِفُ : المُرْتَفِعُ الرَّأْسِ.

يُنَازِعْنَ الأَعِنَّةَ مُشْعِفَاتٍ يُلَطِّمُهُنَّ بِالخُمُرِ النِّسَاءُ
حسّان بن ثابت الخزرجيّ الأنصاريّ
أخبار مكّة في قديم الدّهر وحديثه: محمّد بن إسحاق الفاكهيّ (ت، 272هـ)، دراسة وتحقيق: عبد الملك بن عبد الله بن دهيش، دار خضر، بيروت، ط2، (1414هـ/ 1994م)، ‏5/‏215‏.

شَعِفَ[مُتَعَدٍّ بِالحَرْف]

ن20هـ=641م

شَعِفَ بِهِ: وَلِعَ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

"يَا أَبَا الوَلِيدِ، إِنِّي -وَاللهِ- شَعِفْتُ بِالخَمْرِ، فَاذْمُمْهَا لِي"
جَبَلَة بن الأيهم الغسّانيّ
تعليق من أمالي ابن دريد (ت، 321هـ): رواية: أبي مسلم محمّد بن أحمد بن علي الكتاب وآخرين، تحقيق: السيد مصطفى السنوسي، المجلس الوطني للثقافة والفنون والأدب، الكويت، ط1، (1404هـ/ 1984م)، ‏111‏.

شُعَيْفَة[اسْمٌ مُصَغَّر]

23هـ=644م

الشُّعَيْفَةُ : تَصْغِيرُ الشَّعَفَةِ، وَهِيَ: خُصْلَةُ الشَّعْرِ فِي أَعْلَى الرَّأْسِ.

"وَجَاءَ فِي الحَدِيثِ:عَلَانِي عُمَرُ بِالدِّرَّةِ، فَسَقَطَ البُرْنُسُ عَنْ رَأْسِي، فَأَفَاتَنِي اللهُ بِشُعَيْفَتَيْنِ فِي رَأْسِي"
رجل معاصر لعمر بن الخطّاب
كتاب خلق الإنسان: ثابت بن أبي ثابت اللُّغويّ (ت، 250هـ)، تحقيق: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، ط2، د.ت. ‏74‏.

شُعُوف[مَصْدَر]

ن26هـ=647م

الشُّعُوفُ بِالشَّيْءِ: الوَلَعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

أَنَّى أَلَمَّ بِكَ الخَيَالُ يَطِيفُ وَمَطَافُهُ لَكَ ذِكْرَةٌ وَشُعُوفُ ؟
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏113‏.

شَعْف[اسْمُ جِنْس]

53هـ=673م

الشَّعْفُ : المَطْرُ الهَيِّنُ.

تَصَرَّمَ عَنِّي وُدُّ بَكْرِ بْنِ وَائِلٍ وَمَا خِلْتُ عَنِّي وُدَّهُمْ يَتَصَرَّمُ
قَوَارِصُ تَأْتِينِي، وَيَحْتَقِرُونَهَا وَقَدْ يَمْلَأُ الشَّعْفُ الإِنَاءَ فَيُفْعَمُ
الفرزدق
ليس في كلام العرب: ابن خالَوَيْه (ت، 370هـ)، تحقيق: أحمد عبد الغفور عطّار، مكة المكرمة، ط2، (1399هـ/ 1979م)، ‏210‏.

شَعْف[مَصْدَر]

ن206هـ=821م

الشَّعْفُ : الذُّعْرُ الَّذِي يَذْهَبُ بِاللُّبِّ.

كَمَا اصْعَنْفَرَتْ مِعْزَى الجِبَالِ مِنَ الشَّعْفِ
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏145‏.

شَعُوف[صِيغَةُ مُبَالَغَة]

ن60هـ=680م

الشَّعُوفُ بِالشَّيْءِ: الشَّدِيدُ الوَلَعِ بِهِ.

شَعُوفٌ بِرَيَّا المِسْكِ تَشْفِي مِنَ الجَوَى وَمَنْ يَرَهَا، لَا يَشْتَكِي السُّقْمَ، يَسْقَمِ
كعب بن مشهور المُخَبَّليّ الخَثْعَميّ
كعب بن مشهور المُخَبَّليّ لا المُخبّل القيسيّ: تح: حمد الجاسر، مجلة مجمع اللغة العربي، دمشق، العدد2، أبريل 1991م، ‏223‏.
ن167هـ=783م

الشَّعُوفُ لِلُّبِّ وَنَحْوِهِ: الشَّدِيدُ الذَّهَابِ بِهِ.

وَجَارِيَةٍ يُغْلِي بِأَمْثَالِهَا الفَتَى شَعُوفٍ لِأَلْبَابِ الرِّجَالِ قَتُونِ
بشّار بن برد
ديوان بشار بن برد: جمع وتحقيق وشرح: محمّد الطاهر ابن عاشور، وزارة الثقافة، الجزائر، (2007م)، ‏4/‏219‏-‏ 220‏.

شُعَاف[اسْم]

ن90هـ=709م

الشُّعَافُ : الجُنُونُ.

كَأَنَّ فِي أَعْيُنِهِمْ مِنَ الكَمَنْ وَكْنًا وَدَاءً مِنْ شُعَافٍ وَحَبَنْ
جَنْدَل بن المُثَنَّى الطُّهَويّ
كتاب خلق الإنسان: ثابت بن أبي ثابت اللُّغويّ (ت، 250هـ)، تحقيق: عبد الستار أحمد فرّاج، مطبعة حكومة الكويت، الكويت، ط2، د.ت. ‏122‏.

مُشَعَّف[اسْمُ مَفْعُول]

ن110هـ=728م

المُشَعَّفُ : الَّذِي أَحْرَقَ الوَلَعُ قَلْبَهُ وَأَذْهَبَ لُبَّهُ.

بِمَا فِي فُؤَادَيْنَا مِنَ الشَّوْقِ وَالهَوَى فَيُجْبَرُ مُنْهَاضُ الفُؤَادِ، المُشَعَّفُ
الفرزدق
كتاب سيبويه: سيبويه (ت، 180هـ)، تحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط3، (1408هـ/ 1988م- 1416هـ/ 1996م)، ‏3/‏623‏.

مِشْعَف[صِيغَةُ مُبَالَغَة]

ن110هـ=728م

المِشْعَفُ : الشَّدِيدُ الإِذْهَابِ لِلُّبِّ وَنَحْوِهِ.

لَقَدْ كُنْتُ أَحْيَانًا صَبُورًا، فَهَاجَنِي مَشَاعِفُ بِالدَّيْرَيْنِ، رُجْحُ الرَّوَادِفِ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏95‏.

اشْتِعَاف[مَصْدَر]

ن213هـ=828م

الِاشْتِعَافُ بِالشَّيْءِ: الوَلَعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

وَاشْتِعَافِي بِصَفْعِ مَنْ يَدَّعِي الشِّعْـ ـرَ بِلَا خِبْرَةٍ، وَلَا إِحْسَانِ
أحمد بن يوسف الكاتب
كتاب الأوراق قسم أخبار الشّعراء: أبو بكر الصّوليّ (ت، 335هـ)، عني بنشره: ج. هيورث. دن، مطبعة الصاوي، القاهرة، ط1، (1934م)، ‏228‏.

تَشَاعَفَ[مُتَعَدٍّ بِالحَرْف]

269هـ=882م

تَشَاعَفَ بِهِ: تَوَلَّعَ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

يَتَشَاعَفْنَ بِالغَرِيرِ المُسَمَّى مِنْ فَتَاءٍ، دُونَ الجَلِيلِ المُكَنَّى
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏4/‏2144‏.

أَشْعَف[اسْمُ تَفْضِيل]

ن296هـ=909م

الأَشْعَفُ بِالشَّيْءِ: الأَشَدُّ تَعَلُّقًا، وَوَلَعًا بِهِ.

وَإِذَا العُيُوبُ تَعَرَّضَتْ كَانَتْ بِغَيْرِيَ أَشْعَفَا
ابن المعتزّ
ديوان أشعار الأمير أبي العبّاس عبد الله بن محمّد المعتزّ بالله الخليفة العباسي: دراسة وتحقيق: محمّد بديع شريف، دار المعارف، القاهرة، (1977م)، ‏2/‏85‏.

شَعِف[صِفَةٌ مُشَبَّهَة]

ن551هـ=1156م

الشَّعِفُ بِالشَّيْءِ: الوَلِعُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

لَهْفِي عَلَى نَفْحَةٍ مِنْ رِيحِ أَرْضِكُمُ أَبُلُّ مِنْهَا فُؤَادًا مُوقَرًا شَعِفَا
الحسن بن عليّ بن أبي جرادة
خريدة القصر وجريدة العصر: عماد الدين الأصفهانيّ (ت، 597هـ)، تحقيق: مبروك المناعي وآخرين، دار مداد يونيفارسيتي برس، قسنطينة، ط3، (2018م)، ‏14/‏230‏.

أَشْعَفَ[مُتَعَدٍّ]

ن911هـ=1505م

أَشْعَفَهُ بِالأَمْرِ: أَوْلَعَهُ بِهِ وَلَعًا يُذْهِبُ اللُّبَّ.

"المُقَامُ بِهَا سَعَادَةٌ، وَالخُرُوجُ مِنْهَا شِقْوَةٌ... ضَعَّفَ فِيهَا كُلَّ عَمَل، وَأَشْعَفَ بِمَأْمُولِهِ كُلَّ ذِي أَمَل"
جلال الدّين السَّيوطي
شرح مقامات جلال الدين السيوطي: جلال الدّين السَّيوطي (ت، 911هـ)، تحقيق: سمير محمود الدّروبي، مؤسسة الرّسالة، بيروت، ط1، (1409ه/ 1989م)، ‏1/‏506‏.

شَعَفَة[اسْم]

ن1019هـ=1610م

الشَّعَفَةُ مِنَ القَلْبِ: رَأْسُهُ وَأَعْلَاهُ.

يوسف بن زكرياء المغربي
دفع الإصر عن كلام أهل مصر: يوسف بن زكرياء المغربي (ت، 1019هـ)، حققه: عبد المحسن محمود جودة، قدم له وراجعه: محمد حسن عبد العزيز، مجمع اللغة العربية، القاهرة، ط1، (1435ه/ 2014م)، ‏114‏.
