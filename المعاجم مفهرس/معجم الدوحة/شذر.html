شَذْر[اسْمُ جِنْس]

قبل 50ق.هـ=573م

الشَّذْرُ : القِطَعُ الصَّغِيرَةُ مِنَ الذَّهَبِ وَاللُّؤْلُؤِ وَنَحْوِهِمَا.

غَرَائِرُ فِي كِنٍّ، وَصَوْنٍ، وَنَعْمَةٍ يُحَلَّيْنَ يَاقُوتًا وَشَذْرًا مُفَقَّرَا
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏59‏.

شَذْر[صِفَةٌ مُشَبَّهَة]

244هـ=858م

الشَّذْرُ مِنَ الأَنْظَارِ وَنَحْوِهَا: المُتَوَعِّدُ الغَاضِبُ.

فَقَدْ خِفْتُ، وَلَمْ أَظْلِمْ، سِهَامَ النَّظَرِ الشَّذْرِ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏1088‏.

شَذْر[مَصْدَر]

300هـ=912م

الشَّذْرُ : القَطْعُ وَالفَصْلُ.

وَلَنَا تُعْرَفُ المُشَوَّهَةُ النَّجْـ لَاءُ شَذْرًا ، وَالضَّرْبَةُ الأُخْدُودُ
عُفير بن جندل الحِماسيّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏2/‏742‏.

شَذْر[اسْم]

ن651هـ=1253م

الشَّذْرُ : حِلْيَةٌ صَوْتِيَّةٌ قَصِيرَةٌ تَتَخَلَّلُ نَغَمَاتِ اللَّحْنِ الأَصْلِيَّةَ. [الغناء والموسيقا]

"المُتَحَقِّقُونَ بِمَعْرِفَةِ الغِنَاءِ يَسْتَرْذِلُونَ مِنْهُ مَا كَانَ قَلِيلَ الشُّذُورِ وَالنَّغَمِ"
أبو العبّاس التِّيفَاشيّ
مُتعة الأسماع في علم السّماع: أحمد بن يوسف التّيفاشي القفصي (ت، 651هـ)، تحقيق: رشيد السلامي، المجمع التونسي للعلوم والآداب والفنون بيت الحكمة، تونس، (1440هـ/2019م)، ‏147‏.

تَشَذَّرَ[لَازِم]

قبل 50ق.هـ=573م

تَشَذَّرَتِ النَّاقَةُ: تَلَوَّتْ، وَضَرَبَتْ بِذَنَبِهَا نَشَاطًا وَمَرَحًا.

كَأَنَّ بِحَاذَيْهَا إِذَا مَا تَشَذَّرَتْ عَثَاكِيلَ عِذْقٍ مِنْ سُمَيْحَةَ مُرْطِبِ
علقمة الفَحْل
ديوان علقمة الفحل: شرح: الأعلم الشَّنْتَمَري (ت، 476هـ)، حقّقه: لطفي الصقال، درية الخطيب، راجعه: فخر الدين قباوة، دار الكتاب العربي- مطبعة الأصيل، حلب، ط1، (1389هـ/ 1969م)، ‏86‏.
قبل 13ق.هـ=609م

تَشَذَّرَ الشَّخْصُ: تَهَدَّدَ وَتَغَضَّبَ.

غُلْبٌ تَشَذَّرُ بِالذُّحُولِ كَأَنَّهَا جِنُّ البَدِيِّ رَوَاسِيًا أَقْدَامُهَا
لَبيد بن ربيعة العامريّ
شرح ديوان لبيد بن ربيعة العامري: حققه وقدم له: إحسان عبّاس، وزارة الإرشاد والأنباء، الكويت، (1962م)، ‏317‏.
ن40هـ=660م

تَشَذَّرَ القَوْمُ وَغَيْرُهُمْ: تَفَرَّقُوا.

"وَلَئِنْ أَذِنَ اللهُ فِي الكَرَّةِ عَلَيْهِمْ، لَأُدِيلَنَّ مِنْهُمْ، إِلَّا مَا يَتَشَذَّرُ فِي أَطْرَافِ البِلَادِ تَشَذُّرًا"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏2/‏156‏.

تَشَذَّرَ[مُتَعَدٍّ بِالحَرْف]

ن130هـ=748م

تَشَذَّرَ لِلْأَمْرِ: تَشَمَّرَ لَهُ وَتَهَيَّأَ.

مَا أُبَالِي أَنْ تَشَذَّرْتَ لَنَا عَادِيًا، أَمْ بَالَ فِي البَحْرِ قُبَعْ
خلف بن خليفة الأقطع الثّعلبيّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏183‏.

شَذَّرَ[مُتَعَدٍّ]

ن20هـ=641م

شَذَّرَ الشَّيْءَ: فَرَّقَهُ.

يُشَذِّرُ بِالسَّيْفِ أَقْرَانَهُ كَمَا فَرَّقَ اللِّمَّةَ الغَيْلَمُ
البُريق الخُناعيّ الهُذَليّ
البارع في اللُّغة: أبو عليّ القالي (ت، 356هـ)، تحقيق: هاشم الطعّان، مكتبة النهضة، بغداد، دار الحضارة العربية، بيروت، ط1، (1975م)، ‏278‏.

تَشَذُّر[مَصْدَر]

ن35هـ=655م

التَّشَذُّرُ : التَّهَدُّدُ وَالتَّغَضُّبُ.

"وَإِنَّمَا أَنْهَاكَ عَنِ التَّشَذُّرِ عَلَى أَكْفَائِكَ، فَإِنَّ ذَلِكَ لَا يَرْفَعُكَ عَلَيْهِمْ، وَلَا يَضَعُهُمْ لَكَ"
الحارث بن نوفل بن الحارث الهاشميّ
جمهرة خطب العرب في عصور العربية الزاهرة: أحمد زكي صفوت، المكتبة العلميّة، بيروت، ط1، (1352هـ/ 1933م- 1352هـ/ 1934م)، ‏2/‏229‏.
ن40هـ=660م

التَّشَذُّرُ : التَّفَرُّقُ وَالتَّنَاثُرُ.

"وَلَئِنْ أَذِنَ اللهُ فِي الكَرَّةِ عَلَيْهِمْ، لَأُدِيلَنَّ مِنْهُمْ، إِلَّا مَا يَتَشَذَّرُ فِي أَطْرَافِ البِلَادِ تَشَذُّرًا "
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏2/‏156‏.
ن175هـ=791م

التَّشَذُّرُ بِالثَّوْبِ: إِدْخَالُهُ بَيْنَ الفَخِذَيْنِ مَلْوِيًّا، ثُمَّ إِخْرَاجُهُ.

" وَالتَّشذُّرُ : الِاسْتِثْفَارُ بِالثَّوْبِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏249‏.
ن175هـ=791م

التَّشَذُّرُ إِلَى الأَمْرِ: النَّشَاطُ وَالتَّسَرُّعُ إِلَيْهِ.

" وَالتَّشَذُّرُ : النَّشَاطُ وَالتَّسَرُّعُ إِلَى الأَمْرِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏249‏.
178هـ=794م

تَشَذُّرُ النَّاقَةِ: عَقْدُهَا ذَنَبَهَا، بَعْدَ أَنْ تَلْقَحَ، وَنَصْبُهُ عَلَى عَجُزِهَا مِنَ التَّخَيُّلِ.

لَهَا ذَنَبٌ كَالقِنْوِ قَدْ مَذِلَتْ بِهِ وَأَسْمَحَ لِلتَّخْطَارِ بَعْدَ التَّشَذُّرِ
من شواهد أبي زيد الأنصاريّ
كتاب النّوادر في اللّغة: أبو زيد الأنصاريّ (ت، 215هـ)، تحقيق ودراسة: محمّد عبد القادر أحمد عطا، دار الشروق، بيروت- القاهرة، ط1، (1401هـ/ 1981م)، ‏491‏.

شَذَر[مَصْدَر]

ن58هـ=678م

شَذَرَ مَذَرَ : الذَّهَابُ وَالتَّفَرُّقُ فِي كُلِّ وَجْهٍ.

"لِلَّهِ دَرُّ أُمٍّ حَفَلَتْ لَهُ، وَدَرَّتْ عَلَيْهِ... فَفَنَّخَ الكَفَرَةَ، وَدَيَّخَهَا، وَشَرَّدَ الشِّرْكَ شَذَرَ مَذَرَ "
عائشة بنت أبي بكر الصِّدّيق
كتاب بلاغات النّساء وطرائف كلامهن وملح نوادرهن وأخبار ذوات الرأي منهن وأشعارهن في الجاهلية وصدر الإسلام: ابن طَيْفور البغداديّ (ت، 280هـ)، تحقيق: أحمد الألفي، مطبعة مدرسة والدة عباس الأول، القاهرة، (1326هـ/ 1908م)، ‏5‏.

شَذَرَان[مَصْدَر]

ن198هـ=813م

شَذَرَانُ النَّاقَةِ: تَلَوِّيهَا، وَضَرْبُهَا بِذَنَبِهَا نَشَاطًا وَمَرَحًا.

تَثْنِي عَلَى الحَاذَيْنِ ذَا خُصَلٍ تَعْمَالُهُ الشَّذَرَانُ وَالخَطْرُ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏478‏.

شَذْرَة[اسْم]

235هـ=849م

الشَّذْرَةُ فِي الغِنَاءِ: نَغْمَةٌ قَصِيرَةٌ نَاعِمَةٌ، تَقْتَرِنُ بِهَا مُصَوِّتَاتٌ مُنْخَفِضَةٌ وَإِمَالَاتٌ. [الغناء والموسيقا]

وَمُنتَشِرُ الحَلْقِ، وَاهِي اللَّهَاةِ إِذَا مَا شَدَا، فَاحِشُ الغَلْصَمَهْ
فَكَمْ شَذْرَةٍ ثَمَّ مَنْسِيَّةٍ أُطِيحَتْ، وَكَمْ نَغْمَةٍ مُدْغَمَهْ!
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏4/‏2077‏.

مُشَذَّر[اسْمُ مَفْعُول]

ن283هـ=896م

المُشَذَّرُ مِنَ الأَلْحَانِ: المُوَزَّعُ فِي نَغَمَاتٍ قَصِيرَةٍ نَاعِمَةٍ، تَقْتَرِنُ بِهَا مُصَوِّتَاتٌ مُنْخَفِضَةٌ وَإِمَالَاتٌ. [الغناء والموسيقا]

وَاكْسُ شِعْرِي مِنَ النَّشِيدِ نَشِيدًا كَالغِنَاءِ المُشَذَّرَاتِ لُحُونُهْ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏6/‏2482‏.
ن603هـ=1206م

المُشَذَّرُ مِنَ الأَشْيَاءِ: المُفَرَّقُ.

"وَيُتَّخَذُ عَلَى الأَسَاطِينِ الأَرْبَعِ قَصْرٌ مُرَبَّعُ الشَّكْلِ... وَعَلَى أَرْكَانِ أَسْفَلِهِ كَالأَرْجُلِ مُشَذَّرَةٍ ، وَفِي أَسْفَلِ كُلِّ رِجْلٍ ثُقْبٌ يَدْخُلُ فِيهِ ذَكَرُ رَأْسِ أُسْطُوَانَةٍ"
ابن الرّزّاز الجَزَريّ
الجامع بين العلم والعمل النافع في صناعة الحيل: ابن الرّزّاز الجَزَريّ (ت، 603هـ)، تحقيق: أحمد يوسف الحسن وآخرين، معهد التراث العلمي العربي، حلب، (1979م)، ‏103‏.

شَذَارَة[مَصْدَر]

ن327هـ=939م

الشَّذَارَةُ : التَّفْرِيقُ وَالتَّشْتِيتُ.

"وَهُمْ أَهْلُ الزَّعَارَةِ، وَالنَّكَارَةِ، وَالشَّذَارَةِ ، وَأَهْلُ الرِّيَبِ"
عبد الرّحمن بن عيسى الهَمَذانيّ
كتاب الألفاظ: عبد الرّحمن بن عيسى الهَمَذانيّ (ت، 327هـ)، حققه وضبطه وعلق عليه وقدم له: البدراوي زهران، دار المعارف، القاهرة، ط3، (1989م)، ‏226‏.

شَاذُورَة[اسْم]

ن328هـ=940م

الشَّاذُورَةُ : الحَبَّةُ مِنْ صِغَارِ اللُّؤْلِؤِ وَنَحْوِهِ.

مَنْظُومَةٌ هُذِّبَ أَلْفَاظُهَا لَيْسَتْ مِنَ الشِّعْرِ الحِجَازِيِّ
كَأَنَّهَا شَاذُورَةٌ عُلِّقَتْ بِوَجْهِ دِينَارٍ هِرَقْلِيِّ
ابن عبد ربّه الأندلسيّ
ديوان ابن عبد ربّه: جمعه وحققه وشرحه: محمّد رضوان الداية، مؤسسة الرسالة، بيروت، ط1، (1399هـ/ 1979م)، ‏177‏.

مُتَشَذِّر[صِفَةٌ مُشَبَّهَة]

330هـ=941م

المُتَشَذِّرُ مِنَ الذَّهَبِ وَنَحْوِهِ: المُتَفَرِّقُ.

"فَإِذَا بَقِيَتْ قَالُونَاتُ بُوطَقِ الفَحْمِ، وَذَلِكَ الشَّذْرُ بَيْنَ أَجْزَاءِ تُرَابِ التِّبْرِ، دُقَّ فِي مَوْقِعِ حِجَارَةٍ، أَوْ عَلَى مَدَاكٍ، فَانْسَحَقَ ذَلِكَ التُّرَابُ المُنْسَبِكُ، وَتَبَرَّأَ مِنْهُ الذَّهَبُ المُتَشَذِّرُ "
الحسن بن أحمد الهَمْدانيّ
