لَذْع[مَصْدَر]

قبل 50ق.هـ=573م

اللَّذْعُ بِالنَّارِ: الإِحْرَاقُ وَالكَيُّ بِهَا.

فَدَمْعِيَ مِنْ ذِكْرِهَا مُسْبَلٌ وَفِي الصَّدْرِ لَذْعٌ كَجَمْرِ الغَضَا
أبو دُواد الإياديّ
ديوان أبي دواد الإياديّ: جمعه وحققه: أنوار محمود الصالحي، أحمد هاشم السامرائي، دار العصماء، دمشق، ط1، (1431هـ/ 2010م)، ‏187‏.
ن117هـ=735م

لَذْعُ الشَّوْكِ وَنَحْوِهِ: وَخْزُهُ وَلَسْعُهُ.

رَمَى أُمَّهَاتِ القُرْدِ لَذْعٌ مِنَ السَّفَا وَأَحْصَدَ مِنْ قُرْيَانِهِ الزَّهَرُ النَّضْرُ
ذو الرُّمَّة
التّقفية في اللّغة: اليَمان بن أبي اليَمان البَنْدَنيجيّ (ت، 284هـ)، حققه: خليل إبراهيم العطية، مطبعة العاني، بغداد، (1976م)، ‏102‏.
ن198هـ=813م

لَذْعُ الهِجَاءِ: شِدَّتُهُ.

اذْهَبْ نَجَوتَ مِنَ الهِجَاءِ وَلَذعِهِ وَأَمَا وَلَثْغَةِ رَحْمَةَ بْنِ نَجاحِ
لَوْلَا فُتُورٌ فِي كَلَامِكَ يُشْتَهَى وَتَرَفُّقِي بِكَ بَعدُ وَاسْتِمْلَاحِي
لَعَلِمْتَ أَنَّكَ لَا تُمَازِحُ شَاعِرًا فِي سَاعَةٍ لَيْسَتْ بِحِينِ مُزَاحِ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏387‏.
ن198هـ=813م

لَذْعُ الطَّعْمِ: حَرَارَتُهُ وَحَرَافَتُهُ.

اسْقِيَانِي مِنْ شَمُولٍ فِي مَدَى الدَّهْرِ الطَّوِيلِ
فِي لِسَانِ الشَّرَبِ مِنْهَا مِثْلَ لَذْعِ الزَّنْجَبِيلِ
أبو نُواسٍ
ديوان أبي نواس الحسن بن هانئ: حققه وضبطه وشرحه: أحمد عبد المجيد الغزالي، دار الكتاب العربي، بيروت، د.ت. ‏5/‏233‏.
ن200هـ=815م

لَذْعُ العَقَارِبِ وَنَحْوِهَا: لَسْعُهَا وَلَدْغُهَا.

"وَالدَّلِيلُ عَلَى ذَلِكَ العَقَارِبُ الطَّيَّارَةُ؛ إِذْ قَدْ نُعَالِجُهَا بِالبَارِدِ المَحْضِ، فَإِنَّ الثَّلْجَ فِي لَذْعِ هَذِهِ العَقَارِبِ شِفَاءٌ"
منسوب إلى جابر بن حَيّان
رسائل جابر بن حيان: جابر بن حيان (ت، 200هـ)، إعداد: أحمد فريد المزيدي، دار الكتب العلميّة، بيروت، ط1، (1427هـ/ 2006م)، ‏240‏.
ن224هـ=839م

اللَّذْعُ : الضَّرْبُ وَالإِيلَامُ.

وَيَلْذَعُ مُهْجَتِي ذُو العَذْلِ فِيهَا كَلَذْعِ السَّوْطِ خَاصِرَةَ البَطِيِّ
إبراهيم بن المهدي بن المنصور
شعر إبراهيم بن المهديّ وأخباره ونثره: جمع وتحقيق ودراسة: محمّد مصطفى أبو شوارب، مؤسسة جائزة عبد العزيز سعود البابطين للإبداع الشعري- دار الوفاء، الإسكندرية، ط1، (2007م)، ‏239‏.

لَوْذَعِيّ[صِفَة]

قبل 50ق.هـ=573م

اللَّوْذَعِيُّ : الذَّكِيُّ، الحَدِيدُ الفُؤَادِ، الفَصِيحُ.

فَلَيْسَ مِثْلِي عَنْ زُهَيْرٍ بِغَنِي هُوَ الشُّجَاعُ، وَالخَطِيبُ اللَّوْذَعِي
حُصين بن عمرو الكِلابيّ العامريّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏11/‏104‏.

لَذْعَة[اسْمُ مَرَّة]

قبل 13ق.هـ=609م

اللَّذْعَةُ : كَيَّةُ النَّارِ وَنَحْوِهَا.

مَاوِيَّ، بَلْ رُبَّتَمَا غَارَةٍ شَعْوَاءَ، كَاللَّذْعَةِ بِالمِيسَمِ
ضَمْرة بن ضَمْرة النَّهْشليّ
شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏286‏.
60هـ=680م

لذْعَةُ العَيْنِ: نَظْرَتُهَا الضَّارَةُ.

أَفْنَى الشَّبَابَ الَّذِي فَارَقْتُ بَهْجَتَهُ كَرُّ الجَدِيدَينِ مِن آتٍ وَمُنْطَلِقِ
لَمْ يَترُكَا لِيَ فِي طُولِ اخْتِلَافِهِمَا شَيْئًا أَخَافُ عَلَيْهِ لَذْعَةَ الحَدَقِ
أبو الأسود الدُّؤَليّ
ديوان أبي الأسود الدؤلي: صنعة: أبي سعيد الحسن السكري، تحقيق: محمّد حسن آل ياسين، دار ومكتبة الهلال، بيروت، ط2، (1418هـ/ 1998م)، ‏399‏.
ن425هـ=1034م

لَذْعَةُ الكَلَامِ: قَسْوَتُهُ وَشِدَّتُهُ.

وَمَا لُذِعَتْ أُنْثَى مِنَ الدَّهْرِ لَذْعَةً أَشَدَّ عَلَيْهَا مِنْ طَلَاقٍ تَزَوَّدُ
من إنشاد الرّاغب الأصفهانيّ
محاضرات الأدباء ومحاورات الشّعراء والبلغاء: الرّاغب الأصفهانيّ (ت، 425هـ)، تحقيق: رياض عبد الحميد مراد، دار صادر، بيروت، ط1، (1425ه/ 2004م)، ‏3/‏434‏.

لَذْعَة[اسْم]

3هـ=624م

اللَّذْعَةُ : الحُرْقَةُ وَالأَلَمُ.

أَذْهَبَ عَنِّي ذَاكَ مَا كُنْتُ أَجِدْ مِنْ لَذْعَةِ الحُزْنِ الشَّدِيدِ المُعْتَمِدْ
هند بنت عتبة
السّيرة النّبويّة: ابن هشام المَعافريّ الحِمْيريّ (ت، 213هـ)، حققها وضبطها وشرحها ووضع فهارسها: مصطفى السّقا وآخرون، شركة مكتبة ومطبعة البابي الحلبي وأولاده، القاهرة، ط2، (1375هـ/ 1955م)، ‏2/‏92‏.

لَذْعَة[مَصْدَر]

ن247هـ=861م

لَذْعَةُ العَقْرَبِ وَنَحْوِهَا: لَسْعَتُهَا.

"وَأَمَّا الرَّصَاصُ، فَإِنَّهُ إِذَا كَانَ عَلَى وَجْهِهِ، وَدُلِكَ عَلَى لَذْعَةِ العَقْرَبِ البَحْرِيِّ وَتِنِّينِ البَحْرِ، نَفَعَ مِنْهَا"
ابن بَسيل
المقالات السبع من كتاب دياسقوريدوس وهو هيولى الطب في الحشائش والسموم: حُنين بن إسحاق العِباديّ (ت، 260هـ)، تحقيق: سيزار دوبلر. أمبوريوم، برشلونة، تطوان، (1957م)، ‏409‏.

لَاذِع[اسْمُ فَاعِل]

3هـ=624م

اللَّاذِعُ مِنَ الضَّرْبِ: المُوجِعُ، المُؤْلِمُ.

وَمَا انْتَمَيْتُ إِلَى خُورٍ وَلَا كُشُفٍ وَلَا لِئَامٍ، غَدَاةَ البَأْسِ، أَوْرَاعِ
بَلْ ضَارِبِينَ حَبِيكَ البَيْضِ، إِذْ لَحِقُوا شُمَّ العَرَانِينِ، عِنْدَ المَوْتِ لُذَّاعِ
ضرار بن الخطّاب الفِهْريّ
ضِرار بن الخطّاب الفِهري: دراسة وجمع وتحقيق: عبد الله سليمان الجربوع، مطبوعات نادي مكة الثقافي الأدبي- دار المدني، جدة، (1409هـ/ 1989م)، ‏97‏.
ن20هـ=641م

اللَّاذِعُ مِنَ الإِبَرِ وَنَحْوِهَا: الوَاخِزُ اللَّاسِعُ.

أُسُودٌ تُلَكِّعُ أَفْوَاهَهَا وَآذَانَهَا إِبْرَةٌ لَاذِعَهْ
مسعود بن مُعَتِّب التُّجَيْبِيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏220‏.
126هـ=744م

اللَّاذِعُ مِنَ الكَلَامِ: المُؤْذِي، الجَارِحُ.

وَإِنْ رَأَيْتَ الخَصْمَ ذَا اعْتِرَاضِ وَاشْتَقَّ مِنْ لَوَاذِعِ الإِمْعَاضِ فَأَنْتَ، يَا ابْنَ القَاضِيَيْنِ، قَاضِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏336‏-‏ 337‏.
ن220هـ=835م

اللَّاذِعُ : الحَارُّ الحَارِقُ.

فَالقَلْبُ فِيهِ كُلُومٌ مِنْ لَاعِجِ الأَحْزَانِ
وَفِي الحَشَا لَاذِعَاتٌ كَمُشْعَلِ النِّيرَانِ
القاسم بن يوسف
كتاب الأوراق قسم أخبار الشّعراء: أبو بكر الصّوليّ (ت، 335هـ)، عني بنشره: ج. هيورث. دن، مطبعة الصاوي، القاهرة، ط1، (1934م)، ‏193‏.
ن369هـ=979م

اللَّاذِعُ مِنَ الأَطْعِمَةِ: الحَارُّ فِي طَعْمِهِ الحِرِيِّفُ.

"وَأَمَّا تَحَرُّكُ الدَّافِعَةِ لِمَنَافَاةِ المَعِدَةِ الغِذَاءَ فِي كَيْفِيتِهِ فَيَكُونُ إِمَّا لِمَخَالَطَةِ الغِذَاءِ شَيْئًا مِنَ الأَغْذِيَةِ اللَّاذِعَةِ أَوْ مِنَ الأَخْلَاطِ اللَّاذِعَةِ"
ابن الجزّار القيروانيّ
الفروق بين الاشتباهات في العلل: ابن الجزّار القيروانيّ (ت، 369هـ)، تحقيق: رمزيّة محمّد الأطرقجي، بيت الحكمة، بغداد، (1410هـ/ 1989م)، ‏60‏.
1432هـ=2010م

اللَّاذِعُ مِنَ الرَّوَائِحِ: المُثِيرُ المُعَبِّقُ فِي الأَنْفِ.

"ثُمَّ مَا تَلْبَثُ أَنْ تَتَلَاشَى، وَتَعُودُ الرَّائِحَةُ اللَّاذِعَةُ لِلْوَرَقَةِ الخَضْرَاءِ"
حسن كلاوي وآخرون
كيمياء الغذاء: حسن كلاوي وآخرون ، إشراف: زيد العساف، مراجعة: كرم عودة، تقديم: حسين عبد الرزاق الجزائري، المركز العربي للتعريب، دمشق، (2010م)، ‏1201‏.

لَاذِع[صِفَةٌ مُشَبَّهَة]

ن313هـ=925م

اللَّاذِعُ مِنَ النَّبَاتِ وَنَحْوِهِ: الحَارُّ مَذَاقُهُ، الحَارِقُ فِي اللِّسَانِ كَالفُلْفُلِ وَالبَصَلِ وَنَحْوِهِمَا.

"صِفَةُ مَعْجُونٍ بَدِيعٍ لِلْحُمَّى المُطْبَقَةِ: بَذْرُ هِنْدِبَاءٍ، وَلِسَانُ الحَمَلِ، وَبَذْرُ وَرْدٍ... دَارُصِينِيٍّ لَاذِعٌ مِثْقَالٌ وَنِصْفٌ، يُسْحَقُ الجَمِيعُ، وَيُعْجَنُ بِثَلَاثَةِ أَمْثَالِهِ مِنْ رُبِّ الأُتْرُجِّ السَّاذَجِ وَالرُّمَانِ"
أبو بكر الرّازيّ
كتاب سر صناعة الطّب: أبو بكر الرّازيّ (ت، 313هـ)، دراسة وتحقيق: خالد حربي، دار الثقافة العلميّة، الإسكندريّة، د.ت. ‏115‏.

مُتَلَذِّع[اسْمُ فَاعِل]

ن10هـ=631م

المُتَلَذِّعُ : المُتَوَقِّدُ الذَّكَاءِ.

وَمَا اخْتَصَّنِي مِنْ بَيْنِ ظَهْرَيْ مُحَارِبٍ بِصَحْنَةَ غَيْرُ الأَرْوَعِ المُتَلَذِّعِ
المُزَرِّد بن ضرار الغَطَفانيّ
ديوان المُزرِّد بن ضِرار الغطفاني: رواية: ابن السكيت وغيره، شرح: ثعلب، عني بتحقيقه: خليل إبراهيم العطية، قدم له: محمّد رضا الشبيبي، مطبعة أسعد، بغداد، ط1، (1382هـ/ 1962م)، ‏66‏.

لَذَّعَ[مُتَعَدٍّ]

ن30هـ=651م

لَذَّعَهُ : أَحْرَقَهُ وَكَوَاهُ.

فَوَ اللهِ، لَا أَنْسَاكِ، مَا هَبَّتِ الصَّبَا وَمَا عَقَبَتْهَا فِي الرِّيَاحِ جَنُوبُ

كتاب في أصناف الحمّيات: جالينوس، ترجمة: حُنين بن إسحاق العِباديّ (ت، 260هـ)، تحقيق: ماتياس فيرنهارد، رسالة دكتوراه، جامعة لودفيج ماكسيميليانس، ميونيخ، (2004م)، ‏80‏.
