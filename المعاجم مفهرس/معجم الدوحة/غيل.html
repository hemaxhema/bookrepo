غَيْل[صِفَةٌ مُشَبَّهَة]

قبل 150ق.هـ=476م

الغَيْلُ مِنَ السَّوَاعِدِ وَنَحْوِهَا: الرَّيَّانُ المُمْتَلِئُ.

يَا رُبَّ نَهْبٍ صَالِحٍ حَوَيْتُهُ وَرُبَّ قِرْنٍ بَطَلٍ أَرْدَيْتُهُ وَرُبَّ غَيْلٍ حَسَنٍ لَوَيْتُهُ
دويد بن زيد القُضاعيّ
أمالي المرتضى غُرر الفوائد ودُرر القلائد: الشّريف المرتضى (ت، 436هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار إحياء الكتب العربية- عيسى البابي الحلبي وشركاه، القاهرة، ط1، (1373هـ/ 1954م)، ‏1/‏237‏.

غَيْل[اسْم]

قبل 50ق.هـ=573م

الغَيْلُ : اللَّبَنُ الَّذِي تُرْضِعُهُ المَرْأَةُ وَلَدَهَا وَهِيَ حَامِلٌ أَوْ حِينَ تُجَامَعُ.

"وَاللهِ، مَا حَمَلْتُهُ وُضْعًا، وَلَا وَلَدْتُهُ يَتْنًا، وَلَا أَرْضَعْتُهُ غَيْلًا "
أُميمة القَيْنيّة
كتاب الإبل: الأصمعيّ (ت، 216هـ)، تحقيق: حاتم الضامن، دار البشائر، دمشق، ط1، (1424هـ/ 2003م)، ‏51‏.
قبل 50ق.هـ=573م

الغَيْلُ : المَاءُ الجَارِي عَلَى وَجْهِ الأَرْضِ.

وَيَخْطُو عَلَى صُمٍّ صِلَابٍ كَأَنَّهَا حِجَارَةُ غَيْلٍ وَارِسَاتٌ بِطُحْلَبِ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏47‏.
816هـ=1413م

الغَيْلُ : المَوْضِعُ فِيهِ المَاءُ.

"الرَّشِيدِيَّةُ أَصْحَابُ رَجُلٍ يُسَمَّى رَشِيدًا، انْفَرَدُوا بِأَنْ أَوْجَبُوا فِيمَا تُسْقَى بِالغُيُولِ وَالأَنْهَارِ نِصْفَ الْعُشْرِ، فَبَرِئَتْ مِنْهُمُ الثَّعَالِبَةُ مِنَ الخَوَارِجِ"
أحمد بن يحيى بن المرتضى
كتاب البحر الزخَّار الجامع لمذاهب علماء الأمصار: أحمد بن يحيى بن المرتضى (ت، 840هـ)، تحقيق: عبد الله بن عبد الكريم الجرافي، دار الحكمة اليمانية، صنعاء، ط2، (1409ه/ 1988م)، ‏1/‏3‏.

غَيْل[اسْمُ جِنْس]

قبل 13ق.هـ=609م

الغَيْلُ : ما يَكُونُ فِي الثَّوْبِ وَنَحْوِهِ مِنْ نَقْشٍ وَخُطُوطٍ.

فَأُبْنَا، وَنَازَعْنَا الحَدِيثَ أَوَانِسًا عَلَيْهِنَّ جَيْشَانِيَّةٌ ذَاتُ أَغْيَالِ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏114‏.

غَيْل[مَصْدَر]

ن11هـ=632م

الغَيْلُ : وَطْءُ المُرْضِعِ.

"لَا تَقْتُلُوا أَوْلادَكُمْ سِرًّا؛ فَإِنَّ قَتْلَ الغَيْلِ يُدْرِكُ الفَارِسَ فَيُدَعْثِرُهُ عَنْ ظَهْرِ فَرَسِهِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏45/‏543،،‏(رقم الحديث: 27562).
ن21هـ=642م

الغَيْلُ : إِرْضَاعُ المَرْأَةِ وَلَدَهَا وَهِيَ حَامِلٌ.

وَكُنْتُمْ أَعْبُدًا أَوْلادَ غَيْلٍ بَنِي آمٍ مَرَنَّ عَلَى السِّفَادِ
عمرو بن مَعْدي كَرِب الزُّبَيديّ
شعر عَمرو بن مَعْدي كَرب الزُّبَيْدي: جمعه ونسّقه: مطاع الطرابيشي، مطبوعات مجمع اللغة العربية، دمشق، ط2، (1405هـ/ 1985م)، ‏113‏.

غِيل[اسْمُ جِنْس]

قبل 50ق.هـ=573م

الغِيلَةُ : الوَاحِدَةُ مِنَ الغِيلِ، وَهُوَ الشَّجَرُ الكَثِيفُ المُلْتَفُّ الَّذِي يُسْتَتَرُ فِيهِ.

كَأَنَّ بِبَطْنِ الشِّعْبِ غِرْبَانَ غِيلَةٍ وَمِنْ فَوْقِنَا مِنْهُمْ رِجَالٌ عَصَائِبُ
مالك بن خالد الخُناعيّ الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏3/‏12‏.
قبل 50ق.هـ=573م

الغِيلُ : مَوْضِعُ الأَسَدِ.

وَشَبَابٌ كَأَنَّهُمْ أُسْدُ غِيلٍ خَالَطَتْ فَرْطَ حَدِّهِمْ أَحْلَامُ
أبو دُواد الإياديّ
ديوان أبي دواد الإياديّ: جمعه وحققه: أنوار محمود الصالحي، أحمد هاشم السامرائي، دار العصماء، دمشق، ط1، (1431هـ/ 2010م)، ‏164‏.

غَيَّال[اسْم]

قبل 50ق.هـ=573م

الغَيَّالُ : الأَسَدُ.

لَمَّا رَأَيْتُ أَبَا عَمْرٍو رَزَمْتُ لَهُ مِنِّي كَمَا رَزَمَ الغَيَّالُ فِي الغُرُفِ
عبد مَناف بن رِبْع الجُرَبيّ الهُذَليّ
تاج العروس من جواهر القاموس المؤلف: محمّد بن محمّد بن عبد الرزّاق الحسيني، أبو الفيض، الملقّب بمرتضى، الزَّبيدي: تحقيق: مجموعة من المحققين الناشر: دار الهداية، الكويت،، ‏30/‏139‏.

مُغْيَل[اسْمُ مَفْعُول]

قبل 50ق.هـ=573م

المُغْيَلُ : المُرْضَعُ مِنْ لَبَنِ المَأْتِيَّةِ أَوْ لَبَنِ الحُبْلَى.

وَمِثْلِكِ بِكْرًا قَدْ طَرَقْتُ وَثَيِّبًا فَأَلهَيْتُهَا عَنْ ذِي تَمَائِمَ مُغْيَلِ
امرؤ القيس بن حُجْر الكنديّ
كتاب سيبويه: سيبويه (ت، 180هـ)، تحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط3، (1408هـ/ 1988م- 1416هـ/ 1996م)، ‏2/‏163‏.

مُغْيِل[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

المُغْيِلُ مِنَ الشَّجَرِ: النَّابِتُ فِي الغِيلِ.

كَالأَيْمِ ذِي الطُّرَّةِ أَوْ نَاشِئِ الـ ـبَرْدِيِّ تَحْتَ الحَفَأِ المُغْيِلِ
المُتَنَخِّل الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏4‏.
ن11هـ=632م

المُغْيِلُ مِنَ الأَدْوَاءِ: الَّذِي يَكُونُ بِسَبَبِ رَضَاعِ لَبَنِ الأُمِّ الحَامِلِ.

فَأَتَتْ بِهِ حُوشَ الجَنَانِ مُبَطَّنًا سُهُدًا، إِذَا مَا نَامَ لَيْلُ الهَوْجَلِ
وَمُبَرَّأً مِنْ كُلِّ غُبَّرِ حَيْضَةٍ وَفَسَادِ مُرْضِعَةٍ وَدَاءٍ مُغْيِلِ
أبو كبير الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏92 ‏-‏ 93‏.
464هـ=1071م

المُغْيِلُ مِنَ النِّسَاءِ: الَّتِي تُرْضِعُ وَلَدَهَا وَهِيَ حُبْلَى.

تَرَعْرَعَ كَالشِّبْلِ فِي غِيلِهِ وَلَمْ يَشْكُ مُرْضِعَةً مُغْيِلَا
أسعد بن عليّ البارع الزّوزنيّ
دُمية القصر وعُصرة أهل العصر: عليّ بن الحسن الباخَرْزيّ (ت، 467هـ)، تحقيق ودراسة: محمّد ألتونجي، دار الجيل، بيروت، ط1، (1414هـ/ 1993م)، ‏2/‏1406‏.

مُغْتَال[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

المُغْتَالُ : الرَّيَّانُ المُمْتَلِئُ.

عَرَفْتُ بِأَجْدُثٍ فِنِعَافِ عِرْقٍ عَلَامَاتٍ كَتَحْبِيرِ النِّمَاطِ
كَوَشْمِ المِعْصَمِ المُغْتَالِ عُلَّتْ نَوَاشِرُهُ بِوَشْمٍ مُسْتَشَاطِ
المُتَنَخِّل الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏18‏.

غَيُول[صِفَةٌ مُشَبَّهَة]

قبل 13ق.هـ=609م

الغَيُولُ : السَّمِينُ المُمْتِلِئُ.

إِنِّي، لَعَمْرُ الَّذِي خَطَّتْ مَنَاسِمُهَا لَهُ وَسِيقَ إِلَيْهِ البَاقِرُ الغُيُلُ
لَئِنْ قَتَلْتُمْ عَمِيدًا لَمْ يَكُنْ صَدَدًا، لَنَقْتُلَنْ مِثْلَهُ مِنْكُمْ فَنَمْتَثِلُ
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏1/‏221‏-‏ 222‏.

مُتَغَيِّل[صِفَةٌ مُشَبَّهَة]

ن2ق.هـ=620م

المُتَغَيِّلُ مِنَ الشَّجَرِ: الكَثِيفُ المُتَشَابِكُ.

تَعَلَّمَهَا فِي غِيلِهَا وَهِيَ حَظْوَةٌ بِوَادٍ بِهِ نَبْعٌ طِوَالٌ وَحِثْيَلُ
وَبَانٌ وَظَيَّانٌ وَرَنْفٌ وَشَوْحَطٌ أَلَفُّ أَثِيثٌ نَاعِمٌ مُتَغَيِّلُ
أوس بن حَجَر المازنيّ التَّميميّ
ديوان أوس بن حجر: تحقيق وشرح: محمّد يوسف نجم، دار صادر، بيروت، ط3، (1399هـ/ 1979م)، ‏97‏.

مُغَيَّل[اسْمُ مَفْعُول]

ن7هـ=628م

المُغَيَّلُ مِنَ الثِّيَابِ: المُزَخْرَفُ بِالخُطُوطِ وَنَحْوِهَا، مَعَ اتِّسَاعٍ فِيهِ.

إِذَا مَا عَلَاهَا فَارِسٌ مَتَبَذِّلٌ فَنِعْمَ فِرَاشُ الفَارِسِ المُتَبَذِّلِ
يَنُوءُ بِهَا بُوصٌ إِذَا مَا تَفَضَّلَتْ تَوَعَّبَ عَرْضَ الشَّرْعَبِيِّ المُغَيَّلِ
الأعشى الكبير
ديوان الأعشى الكبير ميمون بن قيس بن جَندل: تحقيق: محمود إبراهيم محمّد الرضواني، وزارة الثقافة والفنون والتراث، الدوحة، ط1، (2010م)، ‏2/‏237‏.

غِيلَة[اسْم]

86هـ=705م

الغِيلَةُ : ثَمَرَةُ شَجَرِ الأَرَاكِ الرَّطْبَةُ، وَهِيَ ثَمَرَةٌ فِي هَيْئَةِ حَبٍّ كَالعِنَبِ الأَسْوَدِ.

وَلَيْتَ لَنَا، بِالجَوْزِ وَاللَّوْزِ، غِيلَةً جَنَاهَا لَنَا، مِنْ بَطْنِ حَلْيَةَ، جَانِ

أشعار اللصوص وأخبارهم: جمع وتحقيق: عبد المعين الملّوحي، دار أسامة، دمشق، د.ت. ‏16‏.

أَصْهَبُ هَدَّارٌ لِكُلِّ أَرْكَبِ بِغِيلَةٍ تَنْسَلُّ نَحْوَ الأَنْبُبِ

أَغَالَ الشَّخْصُ وَلَدَهُ: وَطِئَ أُمَّهُ وَهِيَ تُرْضِعُهُ.

"لَقَدْ هَمَمْتُ أَنْ أَنْهَى عَنِ الغِيلَةِ، فَنَظَرْتُ فِي الرُّومِ وَفَارِسَ فَإِذَا هُمْ يُغِيلُونَ أَوْلادَهُمْ، فَلَا يَضُرُّ أَوْلَادَهُمْ ذَلِكَ شَيْئًا"
حديث نبويّ
صحيح مسلم: مسلم بن الحجّاج النّيسابوريّ (ت، 261هـ)، تحقيق: محمّد فؤاد عبد الباقي، دار إحياء الكتب العربية- دار الحديث، القاهرة، ط1، (1412هـ/ 1991م)، ‏2/‏1067،،‏(رقم الحديث: 141).

مالك بن أنس الأصبحيّ الحِمْيريّ
المدونة الكبرى: مالك بن أنس الأصبحيّ الحِمْيريّ (ت، 179هـ)، رواية: سحنون بن سعيد التنوخي، مطبعة السعادة، مصر، (1323هـ)، ‏4/‏407‏.

تَغَيَّلَ[مُتَعَدٍّ]

ن30هـ=651م

شعر بني تميم في العصر الجاهلي: جمع وتحقيق: عبد الحميد المُعيني، منشورات نادي القصيم الأدبي، بريدة، (1402هـ/ 1982م)، ‏364‏.

تَغَيَّلَ[لَازِم]

اغْتَالَ[مُتَعَدٍّ]

105هـ=723م

تَعَلَّقَ هَذَا القَلْبُ لِلْحَيْنِ مَعْلَقًا غَزَالًا تَحَلَّى عِقْدَ دُرٍّ وَيَارَقَا
شَجِي الحِجْلِ يَغْتَالُ العَجِيزَةَ مِرْطُهُ وَأَمَّا وِشَاحَاهُ عَلَيْهِ فَأَمْلَقَا
عبد الله بن عمر العَرْجيّ
ديوان العَرجي: رواية: أبي الفتح ابن جنّي (ت، 392هـ)، شرحه وحققه: خضر الطائي، رشيد العبيدي، الشركة الإسلامية، بغداد، ط1، (1375هـ/ 1956م)، ‏163‏-‏ 164‏.

"فَأَخَذَ بِيَدِهِ شَجَرَةَ أُمِّ غَيْلَانَ ، وَأَخَذَ بِيَمِينِهِ خَشَبَةً عَظِيمَةً، فَإِذَا هَجَمَ عَلَيْهِ التِّنِّينُ، أَدْخَلَ الشَّجَرَةَ فِي فَمِهِ، وَضَرَبَهُ بِالخَشَبَةِ فِي الرَّأْسِ، فَلَمْ يَزَلْ يُقَاتِلُهُ حَتَّى كَلَّ التِّنِّينُ، وَانْصَرَفَ"
