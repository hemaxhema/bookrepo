أَسْنَع[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الأَسْنَعُ مِنْ كُلِّ شَيْءٍ: الطَّوِيلُ.

فَإِنْ تَكُ جَارَيْتَ الظِّلَالَ فَرُبَّمَا سُبِقْتَ، وَيَوْمُ القِرْنِ عُرْيَانُ أَسْنَعُ
حاجز بن عوف الأزديّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏21/‏109‏.
ن145هـ=762م

الأَسْنَعُ : العَالِي المُرْتَفِعُ.

إِنَّ لَنَا عِزًّا رَسَا أَنْ يُنْزَعَا فِي الخِنْدِفِيِّينَ وَمَجْدًا أَسْنَعَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏239‏.
ن198هـ=813م

الأَسْنَعُ : الحَسَنُ الجَمِيلُ.

سُنْعِ الأَسَامِي، مُسْبِلِي أُزُرٍ حُمْرٍ، تَمَسُّ الأَرْضَ بِالهُدُبِ
أبو نُواسٍ
تفسير الكشاف عن حقائق التنزيل وعيون الأقاويل في وجوه التأويل: جار الله الزمخشريّ (ت، 538هـ)، اعتنى به وخرج أحاديثه وعلق عليه: خليل مأمون شيحا، دار المعرفة، بيروت، ط3، (1430ه/ 2009م)، ‏632‏.

أَسْنَع[اسْمُ تَفْضِيل]

ن1305هـ=1887م

الأَسْنَعُ : الأَكْثَرُ سَنَاعَةً وَحُسْنًا.

عُوجُوا عَلَى رَبْعِ الرُّبُوعِ الأَسْنَعِ وَتَعَجَّبُوا مِنْ حُسْنِ صُنْعِ الإِصْبَعِ
أحمد فارس الشدياق
كنز الرغائب في منتخبات الجوائب: أحمد فارس الشدياق (ت، 1305هـ)، جمعها: سليم بن أحمد فارس، مطبعة الجوائب، الأستانة، ط1، (1289- 1298ه)، ‏3/‏212‏.

سَنِيعَة[اسْم]

قبل 50ق.هـ=573م

السَّنِيعَةُ : الطَّرِيقَةُ فِي الجَبَلِ.

إِذَا صَدَرَتْ عَنْهُ تَمَشَّتْ، مَخَاضُهَا إِلَى السَّرْوِ، تَدْعُوهَا إِلَيْهِ السَّنَائِعُ
قيس بن العَيزارَة الهُذَليّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏2/‏104‏.

سَنِيعَة[صِفَةٌ مُشَبَّهَة]

1270هـ=1853م

السَّنِيعَةُ مِنَ النِّسَاءِ: اللَّيِّنَةُ المَفَاصِلِ، اللَّطِيفَةُ العِظَامِ.

"وَهَذِهِ سَيِّدَتِي الجَمِيلَةُ البَضَّةُ الغَرَّاء، السَّنِيعَةُ الغَضَّةُ الفَرَّاء، الصَّبِيحَةُ الزَّهْرَاء"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏501‏-‏502‏.

سِنْع[اسْم]

78هـ=697م

السِّنْعُ : العَظْمُ الَّذِي يَصِلُ بَيْنَ الأَصَابِعِ وَالرُّسْغِ فِي جَوْفِ الكَفِّ.

نَجَا قَطَرِيٌّ، وَالرِّمَاحُ تَنُوشُهُ عَلَى سَابِحٍ، نَهْدِ التَّلِيلِ مُقَرَّعِ
يَلُفُّ بِهِ السَّاقَيْنِ رَكْضًا، وَقَدْ بَدَا لِأَسْنَاعِهِ ، يَوْمٌ مِنَ الشَّرِّ أَشْنَعِ
كعب بن مَعْدان الأشقريّ
معجم البلدان: ياقوت الحَمَويّ (ت، 626هـ)، دار صادر، بيروت، (1397ه/ 1977م)، ‏2/‏198‏.

سَنِيع[اسْم]

100هـ=718م

السَّنِيعُ : الرِّفْعَةُ وَالعُلُوُّ.

أَنْتَ ابْنُ كُلِّ مُنْتَضًى قَرِيعِ تَمَّ تَمَامَ البَدْرِ فِي سَنِيعِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏145‏.

سَنِيع[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

السَّنِيعُ : التَّامُّ الخَلْقِ الشَّدِيدُ.

" وَالسَّنِيعُ : التَّامُّ الضَّلِيعُ مِنْ كُلِّ شَيْءٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏338‏.
ن175هـ=791م

السَّنِيعُ : الجَمِيلُ اللَّطِيفُ.

"امْرَأَةٌ سَنِيعَةٌ ... وَهِيَ الجَمِيلَةُ اللَّيِّنَةُ المَفَاصِلِ، اللَّطِيفَةُ العِظَامِ فِي كَمَالٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏338‏.
ن540هـ=1145م

السَّنِيعُ : المُرْتَفِعُ العَالِي.

"الفَقِيهُ الأَجَلُّ... لَا أَخْمَدَ اللهُ شِهَابَه، وَلَا أَعْدَمَ المُقْتَبِسِينَ ثُنَارَهُ وَنِهَابَه، عُصْبَةً لِأَهْلِ الأَدَبِ سَنِيعَة "
ابن أبي الخِصال
رسائل ابن أبي الخصال: ابن أبي الخِصال (ت، 540هـ)، تحقيق: محمَّد رضوان الدّاية، دار الفكر، دمشق، ط1، (1408ه/ 1988م)، ‏529‏-‏530‏.

مِسْنَاع[صِيغَةُ مُبَالَغَة]

125هـ=743م

المِسْنَاعُ مِنَ النُّوقِ: السَّرِيعَةُ الوَاسِعَةُ الخَطْوِ.

"يَا أَمِيرَ المُؤْمِنِينَ، جَعَلَنِي اللهُ فِدَاكَ، لِمَ رَدَدْتَ نَاقَتِي، وَهِيَ هِلْوَاعٌ مِرْيَاع، مِرْبَاعٌ مِقْرَاع، مِسْيَاعٌ مِسْنَاع ، حَلْبَانَةٌ رَكْبَانَةٌ؟"
بَكّار بن عبد الله بن مصعب الزُّبيريّ
البصائر والذّخائر: أبو حيّان التَّوحيديّ (ت، 414هـ)، تحقيق: وداد القاضي، دار صادر، بيروت، ط1، (1408هـ/ 1988م)، ‏2/‏197‏.

سَنُعَ[لَازِم]

ن175هـ=791م

سَنُعَ الشَّخْصُ: تَمَّ فِي الخَلْقِ وَالحُسْنِ.

"امْرَأَةٌ سَنِيعَةٌ، قَدْ سَنُعَتْ سَنَاعَةً ، وَهِيَ الجَمِيلَةُ اللَّيِّنَةُ المَفَاصِلِ، اللَّطِيفَةُ العِظَامِ فِي كَمَالٍ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏338‏.

سُنُوع[مَصْدَر]

ن238هـ=852م

سُنُوعُ الحَاجِبِ وَنَحْوِهِ: طُولُهُ.

"وَأَمَّا قَوْلُهُ: (أَزَجُّ الحَاجِبَيْنِ)؛ يَعْنِي دِقَّتَهُمَا وَسُنُوعَهُمَا مَعَ تَقْوِيسِهِمَا إِلِى مُؤَخَّرِ العَيْنَيْنِ"
ابن حَبيب الإلبيريّ الأندلسيّ
كتاب التاريخ: ابن حَبيب الإلبيريّ الأندلسيّ (ت، 238هـ)، اعتنى به: عبد الغني مستو، المكتبة العصرية- شركة أبناء شريف الأنصاري- الدار النموذجية- المطبعة العصرية، صيدا- بيروت، ط1، (1429هـ/ 2008م)، ‏84‏.
1313هـ=1895م

السُّنُوعُ : الحُسْنُ وَالجَمَالُ.

قَرِيحَتُهُ سَيَّالَةٌ لَا قَرِيحَةٌ لَقَدْ طَابَ مَجْنَاهَا، وَجَادَ صَنِيعُهَا
بَدَائِعُهَا بِالتِّبْرِ لَا الحِبْرِ حَقَّ أَنْ تُسَطَّرَ حَتَّى يَسْتَبِينَ سُنُوعُهَا
جرجس شِلْحُت
نظم اللآلي للحبر الشمالي: جرمانوس الشمالي ميخائيل بن منصور بن يوسف الشمالي (ت، 1313هـ)، المطبعة المارونية، حلب، (1895م)، ‏207‏.

مُسَنَّع[صِفَةٌ مُشَبَّهَة]

278هـ=891م

المُسَنَّعُ : الحَسَنُ التَّامُّ الخِلْقَةِ.

لَجَّ الفُؤَادُ فَلَيْسَ يَلْذَعُهُ عَذْلٌ، وَلَا النَّكَبَاتُ تَرْدَعُهُ
بِمُسَنَّعٍ ظَلَّتْ مَحَاسِنُهُ تُخْفِي بِهَا بَدْرًا، وَتُطْلِعُهُ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏4/‏1539‏.

أَسْنَعَ[مُتَعَدٍّ]

330هـ=941م

أَسْنَعَ العَطَاءَ: أَجْزَلَهُ وَكَثَّرَهُ.

"وَصَحِبْتُ الحَيَا المُحْسِبَ، وَالجَوَادَ المُفْضِلَ، الَّذِى إِذَا وَعَدَ وَفَّى، وَإِذَا أَوْعَدَ عَفَا، وَإِذَا وَهَبَ أَسْنَع ، وَإِذَا أَعْطَى أَقْنَع"
أبو عليّ القالي
كتاب الأمالي: أبو عليّ القالي (ت، 356هـ)، تحقيق: محمّد عبد الجواد الأصمعي، الهيئة المصرية العامة للكتاب، القاهرة، (1975- 1976م)، ‏1/‏23‏.

مُسْنَع[صِفَةٌ مُشَبَّهَة]

352هـ=963م

المُسْنَعُ : الكَثِيرُ.

مُفَرَّكٌ مُجْتَوًى، لَمْ تُرْضَ طَلَّتُهُ وَلَوْ أَتَاهَا بِمَهْرٍ مُسْنَعٍ رُغُبِ

تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏2/‏104‏.

سَنَاعِيَّة[اسْم]

ن530هـ=1135م

السَّنَاعِيَّةُ : طَائِفَةٌ مِنَ الشِّيعَةِ تَنْتَمِي لِلزَّيْدِيَّةِ، وَأَصْلُهَا مِنْ مَدِينَةِ سَنَاعَ.

"ثُمَّ لَمْ يَتَهَيَّأْ لِعَطَّافٍ إِلَيْهِ رُجُوعٌ حَتَّى اجْتَمَعَ إِلَيْهِ قَوْمٌ مِنَ المُخْتَرِعَةِ وَالسَّنَاعِيَّةِ ، وَتَكَلَّمُوا فِيمَا بَيْنَهُمْ مِنَ الخِلَافِ، وَاحْتَجَّ كُلُّ فَرِيقٍ بِحُجَّتِهِ"
اللَّحْجيّ
من كتاب أخبار الزيدية [المُطَرِّفِيَّة]: اللَّحْجيّ (ت، 530هـ)، اعتنى به وضبطه: مقبل التام عامر الأحمدي، الجيل الجديد ناشرون، صنعاء، (2020م)، ‏44‏.

إِسْنَاع[مَصْدَر]

1282هـ=1865م

الإِسْنَاعُ : الحُسْنُ وَالجَمَالُ.

"وَمِنَ العَجِيبِ أَنَّ الوَالِدَيْنِ مِنَ الإِنْكِلِيزِ إِذَا كَانَا قَبِيحَيْنِ تَأْتِي أَوْلَادُهُمْ مِلَاحًا، فَإِذَا دَامَ هَذَا الإِسْنَاعُ حِقْبَةً فَلَا يُرَى فِيهِمْ بَعْدُ مِنْ قَبِيحٍ"
أحمد فارس الشدياق
كشف المخبّا عن فنون أوربّا: أحمد فارس الشدياق (ت، 1305هـ)، تقديم: عصمت نصار، دار الكتاب المصري، مكتبة الإسكندرية، القاهرة، دار الكتاب اللبناني، بيروت، (1433ه/ 2012م)، ‏193‏.
