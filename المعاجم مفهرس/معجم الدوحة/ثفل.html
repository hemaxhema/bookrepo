ثِفَال[اسْم]

قبل 150ق.هـ=476م

الثِّفَالُ : الحَجَرُ الأَسْفَلُ مِنَ الرَّحَى.

أَتَاهُمْ بِالفُيُولِ مُجَلَّلَاتٍ وَبِالأَبْطَالِ سَابُورُ الجُنُودِ
فَهَدَّمَ مِنْ أَوَاسِي الحِصْنِ صَخْرًا كَأَنَّ ثِفَالَهُ زُبَرُ الحَدِيدِ
منسوب إلى جُدَيّ بن الدِّلْهاث القُضاعيّ
تاريخ الطّبريّ تاريخ الرسل والملوك: الطّبريّ (ت، 310هـ)، تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، دار سويدان، بيروت، ط2، (1387هـ/ 1967م)، ‏2/‏49‏.
قبل 50ق.هـ=573م

ثِفَالُ الرَّحَى: بِسَاطٌ يُوضَعُ تَحْتَهَا يَقِي الطَّحِينَ مِنَ التُّرَابِ.

أَعَامِرُ، إِنَّا لَوْ نَشَاءُ لَغُرْتُمُ كَمَا غَارَ مِنْ شَمْسِ النَّهَارِ نُجُومُهَا
إِلَى أَيِّمَا الحَيَّيْنِ تُرْكُوا، فَإِنَّكُمْ ثِفَالُ الرَّحَى مِنْ تَحْتِهَا، لَا يَرِيمُهَا
الجُميح الأسديّ
الشاعر الجاهلي الجُميح بن الطماح الأسدي أخباره وأشعاره: شرح وتحقيق: محمّد علي دقة، مجلة جامعة الملك سعود، الرياض، مج5، الآداب2، (1413هـ/ 1993م)، ‏489‏-‏ 490‏.
ن73هـ=692م

الثِّفَالُ : الإِبْرِيقُ.

"أَكَلَ الدَّجْرَ... ثُمَّ غَسَلَ يَدَهُ بِالثِّفَالِ "
ابن عمر
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏15/‏90‏-‏ 91‏.

ثَفَال[صِفَةٌ مُشَبَّهَة]

قبل 50ق.هـ=573م

الثَّفَالُ مِنَ الإِبِلِ وَنَحْوِهَا: البَطِيءُ الثَّقِيلُ الِانْبِعَاثِ.

كَرَّ المَنُونُ عَلَيْكَ دَهْرًا قُلَّبًا كَرَّ الثَّفَالِ بِقَيْدِهِ أَنْ يَهْرُبَا
جابر بن رالان السِّنْبِسيّ الطّائيّ
الفاخر: المُفَضَّل بن سَلَمة بن عاصم (ت، 291هـ)، تحقيق: عبد العليم الطحاوي، مراجعة: محمّد علي النجار، الهيئة المصرية العامة للكتاب، القاهرة، (1974م)، ‏90‏.

ثُفْل[اسْم]

6هـ=627م

الثُّفْلُ : الدَّقِيقُ، وَالسَّوِيقُ، وَنَحْوُهُمَا.

"مَنْ كَانَ مَعَهُ ثُفْلٌ ، فَلْيَصْطَنِعْ"
حديث نبويّ
غريب الحديث: أبو سليمان الخَطّابيّ البُسْتيّ (ت، 388هـ)، تحقيق: عبد الكريم إبراهيم العزباوي، جامعة أم القرى، مكة المكرمة، ط2، (1422هـ/ 2001م)، ‏1/‏720‏.
9هـ=630م

الثُّفْلُ : مَا يَرْسُبُ مِنْ كَدَرٍ، وَخُثَارَةٍ فِي المَاءِ، وَنَحْوِهِ.

"إِنَّا نُصِيبُ مِنَ الثُّفْلِ ، فَأَيُّ الأَسْقِيَةِ؟"
وفد عبد القيس على رسول الله
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏4/‏279،،‏(رقم الحديث: 2476).
23هـ=644م

ثُفْلُ الرَّحَى: ثِفَالُهَا، وَهُوَ حَجَرُهَا السُّفْلِيُّ.

"وَتَدُقُّهُمُ الفِتَنُ كَمَا تَدُقُّ الرَّحَا ثُفْلَهَا "
عليّ بن أبي طالب
كتاب الجامع: مَعْمَر بن راشد البصريّ (ت، 153هـ)- ضمن: المصنف: عبد الرّزّاق بن هَمّام الصَّنْعانيّ (ت، 211هـ)، عني بتحقيق نصوصه وتخريج أحاديثه والتعليق عليه: حبيب الرحمن الأعظمي، المجلس العلمي، كراتشي، ط2، (1403هـ/ 1983م)، ‏11/‏360،،‏(رقم الحديث: 20743).
ن93هـ=712م

ثُفْلُ الطَّعَامِ وَنَحْوِهِ: مَا تَرَسَّبَ مِنْهُ، وَبَقِيَ أَسْفَلَ الوِعَاءِ.

"كَانَ رَسُولُ اللهِ @  يُعْجِبُهُ الثُّفْلُ "
أنس بن مالك الخزرجيّ الأنصاريّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏21/‏26،،‏(رقم الحديث: 13300).
125هـ=743م

الثُّفْلُ : الثَّرِيدُ.

يَحْلِفُ بِاللهِ وَإِنْ لَمْ يُسْأَلِ مَا ذَاقَ ثُفْلًا بَعْدَ عَامِ أَوَّلِ
أبو النَّجْم العِجْليّ
ديوان أبي النجم العِجلي الفضل بن قُدامة: جمعه وشرحه وحققه: محمّد أديب عبد الواحد جمران، مطبوعات مجمع اللغة العربية، دمشق، (1427هـ/ 2006م)، ‏359‏.
ن200هـ=815م

ثُفْلُ العَسَلِ : الشَّمْعُ الَّذِي يُهَيِّئُهُ النَّحْلُ، وَيَصْنَعُهُ.

"وَلِلنَّحْلِ غِذَاءٌ آخَرُ، وَهُوَ ثُفْلُ العَسَلِ : لَيْسَ يَحْلُو جِدًّا، وَحَلَاوَتُهُ شَبِيهَةٌ بِحَلَاوَةِ التِّينِ، وَالنَّحْلُ يَجْلِبُهُ عَلَى سَاقَيْهِ"
ابن البطريق
كتاب الحيوان: أرسطوطاليس: ترجمة: ابن البطريق (ت، 200هـ)، حقّقها وقدّم لها: لاوروس فيليوس وآخرون، بريل، ليدن- بوسطن، (2018م)، ‏232‏.
235هـ=849م

ثُفْلُ الغِذَاءِ: فَضَلَاتُهُ.

"وَمَا بَقِيَ فِي المَعِدَةِ مِنْ أَثْفَالِ الأَغْذِيَةِ تَنْزِلُ إِلَى الأَمْعَاءِ"
ابن سهل الطَّبريّ
فردوس الحكمة في الطّب: ابن سهل الطَّبريّ (ت، 247هـ)، اعتنى بنسخه وتصحيحه: محمّد زبير الصديقي، مطبع آفتاب، برلين، (1928م)، ‏41‏.
ن478هـ=1085م

الثُّفْلُ : الغَائِطُ.

"عَنْ مَالِكٍ... أَنَّهُ قَالَ: لَا يُغْسَلُ الثَّوْبُ مِنْ بَوْلِ الصَّبِيِّ وَلَا الصَّبِيَّةِ حَتَّى يَأْكُلَا الطَّعَامَ. وَقَالَ ابْنُ وَهْبٍ مِثْلَ ذَلِكَ فِي الصَّبِيِّ دُونَ الصَّبِيَّةِ. وَلَمْ يُخْتَلَفْ فِي أَثْفَالِهِمَا أَنَّهَا نَجِسَةٌ، تُغْسَلُ وَإِنْ لَمْ يَأْكُلَا الطَّعَامَ"
عليّ بن محمّد اللَّخْميّ
التّبصرة: عليّ بن محمّد اللَّخْميّ (ت، 478هـ)، دراسة وتحقيق: أحمد عبد الكريم نجيب، وزارة الأوقاف والشؤون الإسلامية، الدّوحة، ط1، د.ت. ‏1/‏116‏.

ثُفَالَة[اسْم]

ن40هـ=660م

الثُّفَالَةُ : مَا يَتَرَسَّبُ فِي قَعْرِ الإِنَاءِ مِنَ الخُثَارَةِ.

"فَلَا يَبْقَى يَوْمَئِذٍ مِنْكُمْ إِلَّا ثُفَالَةٌ كَثُفَالَةِ القِدْرِ، أَوْ نُفَاضَةٌ كَنُفَاضَةِ العِكْمِ"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏1/‏208‏.
ن40هـ=660م

ثُفَالَةُ النَّاسِ: مَنْ لَا خَيْرَ فِيهِ مِنَ الأَرَاذِلِ.

"فَلَا يَبْقَى يَوْمَئِذٍ مِنْكُمْ إِلَّا ثُفَالَةٌ كَثُفَالَةِ القِدْرِ، أَوْ نُفَاضَةٌ كَنُفَاضَةِ العِكْمِ"
عليّ بن أبي طالب
نهج البلاغة: الشّريف الرّضيّ (ت، 406هـ)، شرح: محمّد عبده، دار المعرفة، بيروت، د.ت. ‏1/‏208‏.

ثَافِل[صِفَةٌ مُشَبَّهَة]

ن100هـ=718م

الثَّافِلُ مِنَ الدَّوَابِّ: المُتَثَاقِلُ البَطِيءُ الِانْبِعَاثِ.

جَرُورُ القِيَادِ، ثَافِلٌ ، لَا يَرُوعُهُ صِيَاحُ المُنَادِي، وَاحْتِثَاثُ المُرَاهِنِ
مدرك بن حصن الفَقْعَسيّ الأسديّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏1/‏111‏.
1214هـ=1799م

الثَّافِلُ مِنَ الأَشْخَاصِ: السَّاقِطُ الرَّذْلُ.

"أَمَّا بَعْدُ، فَقَدْ نَقَضْتَ عَهْدِي، وَتَرَكْتَ مَوَدَّةَ آلِ بَيْتِ جَدِّي، وَأَطَعْتَ الظَّلَمَةَ السَّفَلَة، وَامْتَثَلْتَ أَمْرَ المَارِقِينَ الثَّفَلَة "
محمّد أبو الأنوار السّادات
عجائب الآثار في التراجم والأخبار: عبد الرحمن الجبرتي (ت، 1240هـ)، تحقيق: عبد الرحيم عبد الرحمن عبد الرحيم، تقديم: عبد العظيم رمضان، دار الكتب المصرية، القاهرة، (1997- 1998م)، ‏3/‏171‏.

وَرَازِقِيٍّ مُخْطَفٌ خُصُورُهُ قَدْ أَيْنَعَتْ أَنْصَافُهُ الأَسَافِلُ

ثُفْلِيّ[اسْمٌ مَنْسُوب]

الثُّفْلِيُّ مِنَ القُولَنْجَاتِ: الَّذِي يُمْسِكُ ثُفْلَ الأَغْذِيَةِ، وَيَنْتَفِخُ بِسَبَبِ تَعَفُّنِهِ فِيهِ. [الطّبّ والصّيدلة]

تَثَافُل[مَصْدَر]

1420هـ=1998م

التَّثَافُلُ : الِاسْتِغْنَاءُ فِي الجَدْبِ عَنِ اللَّبَنِ بِالتَّمْرِ، وَالزَّبِيبِ، وَنَحْوِهِمَا.

"وَكَانُوا يُسَمُّونَ هَذَا الوَضْعَ المَعِيشِيَّ -كَمَا سَبَقَتِ الإِيمَاءَةُ إِلَى بَعْضِ ذَلِكَ- التَّثَافُلَ ؛ فَكَانُوا، إِذَنْ، فِي تِلْكَ الحَالِ، هُمُ المُتَثَافِلِينَ. وَلَكِنَّهُمْ -كَانُوا إِذَا وَقَعُوا فِي المُثَافَلَةِ- كَانُوا أَشَدَّ مَا تَكُونُ عَلَيْهِ حَالُهُمْ مِنَ الشَّظَفِ، وَسُوءِ الغِذَاءِ"
عبد الملك مرتاض
السّبع المعلّقات، مُقاربة سيميائيّة-أنثروبولوجيّة لنُصوصها: عبد الملك مرتاض، اتّحاد الكتّاب العرب، دمشق، ط1، 1999م، ‏476‏.

تَثْفِيلِيّ[اسْمٌ مَنْسُوب]

1432هـ=2010م

التَّثْفِيلِيُّ مِنَ الأَدَوَاتِ: المَنْسُوبُ إِلَى التَّثْفِيلِ.

"يُسْحَقُ اللَّاكْتُوزُ الأَلْفَا الأَبْيَضُ الثَّلْجِيُّ أُحَادِيُّ المَاءِ فِي مِطْحَنَةٍ صَغِيرَةٍ، وَيُفَصَّلُ بِحَسَبِ حَجْمِ الجُسَيْمَاتِ فِي مُصَنِّفٍ/ فَرَّازٍ تَثْفِيلِيٍّ "
حسن كلاوي وآخرون
كيمياء الغذاء: حسن كلاوي وآخرون ، إشراف: زيد العساف، مراجعة: كرم عودة، تقديم: حسين عبد الرزاق الجزائري، المركز العربي للتعريب، دمشق، (2010م)، ‏1099‏.

مِثْفَلَة[اسْمُ آلَة]

1439هـ=2017م

المِثْفَلَةُ : آلَةُ التَّثْفِيلِ، وَهُوَ تَرْسِيبُ مَا يُخَالِطُ المَاءَ وَنَحْوَهُ مِنْ شَوَائِبَ.

"طَرِيقَةُ التَّثْفِيلِ: تُثَفَّلُ عَيِّنَاتُ المِيَاهِ بِاسْتِعْمَالِ مِثْفَلَةٍ ذَاتِ تَسَارُعٍ... مُدَّةَ 15 دَقِيقَةً، وَيُفْصَلُ المَاءُ عَنِ الرَّاسِبِ"
نصر الحايك
مدخل إلى كيمياء المياه: تلوث، معالجة، تحليل: نصر الحايك ، المعهد العالي للعلوم التطبيقية والتكنولوجيا، دمشق، (2017م)، ‏407‏.
