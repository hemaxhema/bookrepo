صَحْن[اسْم]

قبل 50ق.هـ=573م

الصَّحْنُ : إِنَاءٌ كَبِيرٌ عَرِيضٌ قَرِيبُ القَعْرِ، يُتَّخَذُ لِلشُّرْبِ وَنَحْوِهِ.

كَأَنَّ التِّجَارَ أَصْعَدُوا بِسَبِيئَةٍ مِنَ الخُصِّ حَتَّى أَنْزَلُوهَا عَلَى يُسُرْ
فَلَمَّا اسْتَطَابُوا صُبَّ فِي الصَّحْنِ نِصْفُهُ وَشُجَّتْ بِمَاءٍ غَيْرِ طَرْقٍ وَلَا كَدِرْ
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: تحقيق: محمّد أبو الفضل إبراهيم، دار المعارف، القاهرة، ط5، (1990م)، ‏111‏.
قبل 13ق.هـ=609م

صَحْنُ الدَّارِ وَنَحْوِهَا: السَّاحَةُ فِي وَسَطِهَا.

هُوَ الرَّئِيسُ الَّذِي لَا خَلْقَ يَقْدُمُهُ غَدَاةَ يَحْمِي عَنِ الأَبْطَالِ بِالعَلَمِ
رَبُّ الفِرَاشِ بِصَحْنِ البَيْتِ تَكْرِمَةً بِذَاكَ فُضِّلَ أَهْلُ الفَخْرِ وَالقَدَمِ
أبو طالب
ديوان أبي طالب بن عبد المطلب: صنعة: أبي هَفّان المِهْزَميّ (ت، 257هـ)، عليّ بن حمزة البصريّ (ت، 375هـ): تحقيق: محمّد حسن آل ياسين، دار ومكتبة الهلال، بيروت، ط1، (1421هـ/ 2000م)، ‏97‏.
قبل 13ق.هـ=609م

الصَّحْنُ : المُسْتَوِي مِنَ الأَرْضِ.

أَقْفَرَ مِنْ مَيَّةَ الدَّوَافِعُ مِنْ خَبْتٍ، فَلُبْنَى فَيْحَانَ، فَالرِّجَلُ
فَالجُمُدُ الحَافِظُ الطَّرِيقَ مِنَ الزْ زَيْغِ، فَصَحْنُ الشَّقِيقِ، فَالأُمُلُ
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏95‏.
ن209هـ=824م

الصَّحْنُ : جَوْفُ حَافِرِ الدَّابَّةِ.

"وَفِي الحَافِرِ: الإِطَارُ، وَالدَّخِيسُ، وَالضِّفْدَعُ، وَالأَخْلَقُ، وَالسُّنْبُكُ، وَالأَمْعَرُ، وَالسَّلِيمُ، وَالصَّحْنُ "
أبو عُبيدة
كتاب الخيل: أبو عبيدة (ت، 209هـ)، رواية: أبي حاتم سهل بن محمّد السجستاني عنه رواية أبي يوسف الأصبهاني عنه، مطبعة دائرة المعارف العثمانية، حيدر آباد الدكن، ط1، (1358هـ)، ‏29‏.
ن235هـ=849م

صَحْنُ الخَدِّ وَنَحْوُهُ: وَاجِهَتُهُ.

أَمَا، وَاللهِ، لَوْ عَايَنْتَ وَجْدِي إِذَا اسْتَعْبَرْتُ فِي الظَّلْمَاءِ وَحْدِي
وَجَدَّ تَنَفُّسِي، وَعَلَا زَفِيرِي وَفَاضَتْ عَبْرَتِي فِي صَحْنِ خَدِّي
إِذَنْ لَعَلِمْتَ أَنِّي عَنْ قَرِيبٍ سَتُحْفَرُ حُفْرَتِي وَيُشَقُّ لَحْدِي
ديك الجنّ الحمصيّ
ديوان ديك الجنّ الحمصيّ عبد السلام بن رغبان: جمع وتحقيق ودراسة: مظهر رشيد الحجي، منشورات اتحاد الكتاب العرب، دمشق، (2004م)، ‏121‏.
ن393هـ=1003م

الصَّحْنُ : آلَةٌ مُوسِيقِيَّةٌ مُكَوَّنَةٌ مِنْ قِطْعَتَيْنِ نُحَاسِيَّتَيْنِ تُشْبِهَانِ الطَّسْتَ، تُضْرَبُ إِحْدَاهُمَا بِالأُخْرَى.

سَامَرَنِي أَصْوَاتُ صَنْجِ مُلْهِيَهْ وَصَوْتُ صَحْنَيْ قَيْنَةٍ مُغَنِّيَهْ
من شواهد الجوهريّ
الصّحاح تاج اللغة وصِحاح العربيّة: أبو نصر الجوهريّ (ت، 393هـ)، تحقيق: أحمد عبد الغفور عطّار، دار العلم للملايين، بيروت، ط4، (1990م)، ‏6/‏2151‏.
ن420هـ=1029م

صَحْنُ الأُذُنِ: مَا اتَّسَعَ فِي دَاخِلِهَا.

"فَمِنْ أَعْضَائِهِ: الأُذُنَانِ، وَهُمَا الخُذُنَّتَانِ، وَالأُنْثَيَانِ، وَالسَّامِعَتَانِ... وَفِيهِمَا الصَّحْنَانِ "
الخطيب الإسكافيّ الأصبهانيّ
مبادئ اللُّغة مع شرح أبياته: الخطيب الإسكافيّ الأصبهانيّ (ت، 420هـ)، دراسة وتحقيق: عبد المجيد دياب، دار الفضيلة، القاهرة، (1419هـ/ 1999م)، ‏188‏.

صَحْن[مَصْدَر]

1345هـ=1926م

صَحْنُ الشَّيْءِ: دَقُّهُ وَطَحْنُهُ.

"صُنْعُ الوَرَقِ الكَامِلِ: تُؤْخَذُ الكُتْلَةُ الجَافَّةُ مِنَ الوَرَقِ، وَتُصْحَنُ صَحْنًا جَيِّدًا فِي نَوْعٍ مِنَ الرَّحَى مَصْنُوعٍ مِنَ الجَرَانِيتِ"
عبد القادر حمزة
صناعة الورق، صناعة يمكن أن تكون مصرية: ‏عبد القادر حمزة،‏جريدة البلاغ الأسبوعي: البلاغ الأسبوعي، القاهرة، س1، ع1، (نوفمبر- 1926م)، ‏27‏.

صَحُون[صِيغَةُ مُبَالَغَة]

ن95هـ=714م

الصَّحُونُ مِنَ الدَّوَابِّ: الكَثِيرُ الضَّرْبِ بِالرِّجْلِ.

قَوْدَاءَ لَا تَضْغَنُ أَوْ ضَغُونِ مُلِحَّةٍ لِنَحْرِهِ صَحُونِ
حُميد الأرقط التَّميميّ
كتاب المأثور من اللّغة (ما اتفق لفظه واختلف معناه): أبو العَمَيْثَل الأعرابيّ (ت، 240هـ)، تحقيق: محمّد عبد القادر أحمد عطا، مكتبة النهضة المصرية، القاهرة، ط1، (1408هـ/ 1988م)، ‏116‏.

مَصْحَن[اسْمُ مَكَان]

126هـ=744م

المَصْحَنُ : المَكَانُ الوَاسِعُ فِي وَسَطِ الدَّارِ.

وَحَقِّ أَضْيَافٍ عِطَاشِ الأَعْيُنِ أَمْكَنْتَهُمْ مِنْ حَاجَةِ المُسْتَمْكِنِ حَتَّى بَدَا أَصْحَانُ كُلِّ مَصْحَنِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏114‏-‏ 116‏.

صَحَنَ[مُتَعَدٍّ]

132هـ=749م

صَحَنَهُ : ضَرَبَهُ عَلَى صَحْنِ وَجْهِهِ.

لَوْ صُحِنَتْ حَوْلًا وَحَوْلًا لَمْ تُفِقْ
رؤبة بن العجّاج
مجموع أشعار العرب وهو مشتمل على ديوان رؤبة بن العجاج وعلى أبيات مفردات منسوبة إليه: اعتنى بتصحيحه وترتيبه: وليم بن الورد البروسيّ، دار ابن قتيبة، الكويت، د.ت. ‏107‏.
589هـ=1193م

صَحَنَ الشَّيْءَ: مَحَاهُ وَأَزَالَهُ.

"وَاسْتَأْنَفَ الفِرِنْجُ عَمَلَ دَبَّابَةٍ هَائِلَة، وَآلَةٍ لِلْغَوَائِلِ غَائِلَة... فَكَمْ سُورٍ إِذَا نَطَحَتْهُ طَحَنَتْه، وَكَمْ مَعْقِلٍ حَصَّنَهُ الدَّهْرُ حَصَّتْهُ وَصَحَنَتْه "
عماد الدين الأصفهانيّ
حروب صلاح الدين: عماد الدين الأصفهانيّ (ت، 597هـ)، دار المنار، ط1، (1425ه/ 2004م)، ‏227‏.
ن660هـ=1261م

صَحَنَ الشَّيْءَ: دَقَّهُ وَطَحَنَهُ.

"فَإِذَا احْتَرَقَ البَخُورُ تُفْتَحُ القِدْرُ، وَتُسْتَخْرَجُ القُرْصُ، وَتُنَشَّفُ فِي الظِّلِّ، وَتُصْحَنُ نَاعِمًا"
ابن العديم
كتاب الوصلة إلى الحبيب في وصف الطيّبات والطيب: ابن العديم (ت، 660هـ)، تحقيق: تشارلز بيري، مطبعة جامعة نيويورك، نيويورك، (2016م)، ‏16‏.

صَحَانَة[اسْم]

ن139هـ=756م

الصَّحَانَةُ : سَاحَةُ الدَّارِ فِي وَسَطِهَا.

"مَنِ اسْتَطَاعَ مِنْكُمْ أَنْ يَكُونَ لَهُ فِي مَدِينَةِ مَرْوَ دَارٌ فِيهَا بِئِرٌ وَصَحَانَةٌ فَلْيَفْعَلْ"
الرّبيع بن أنس البَكْريّ
الأنساب: السَّمْعانيّ (ت، 562هـ)، اعتنى بتصحيح والتعليق عليه: عبد الرحمن بن يحيى المعلمي اليماني، مطبعة مجلس دائرة المعارف العثمانية، حيدر آباد، الفاروق الحديثة، القاهرة، ط1، (1397ه/ 1977م)، ‏2/‏160‏.

تَصَحَّنَ[مُتَعَدٍّ]

ن175هـ=791م

تَصَحَّنَ النَّاسَ: سَأَلَهُمُ العَطَاءَ فِي قَصْعَةٍ وَنَحْوِهَا.

"وَالسَّائِلُ يَتَصَحَّنُ النَّاسَ، أَيْ: يَسْأَلُ فِي قَصْعَةٍ وَنَحْوِهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏3/‏119‏.

مَصْحُون[اسْمُ مَفْعُول]

ن250هـ=864م

المَصْحُونُ : المَضْرُوبُ.

"فَانْصَرَفْتُ إِلَى رَحْلِي مَصْحُونًا مَرْضُوضًا"
أبو صدقة
أخبار مكّة في قديم الدّهر وحديثه: محمّد بن إسحاق الفاكهيّ (ت، 272هـ)، دراسة وتحقيق: عبد الملك بن عبد الله بن دهيش، دار خضر، بيروت، ط2، (1414هـ/ 1994م)، ‏3/‏19‏.
ن651هـ=1253م

المَصْحُونُ مِنَ الأَشْيَاءِ: المَدْقُوقُ.

"صِفَةٌ أُخْرَى فِي جَلَاءِ اللُّؤْلُؤَةِ: يُجْعَلُ كَافُورٌ مَصْحُونٌ فِي خِرْقَةِ كَتَّانٍ رَقِيقَةٍ، وَيُتْرَكُ الحَبُّ فِي وَسَطِ الكَافُورِ بَعْدَ أَنْ يُدَقَّ الكَافُورُ نَاعِمًا"
أبو العبّاس التِّيفَاشيّ
كتاب أزهار الأفكار في جواهر الأحجار: أبو العبّاس التِّيفَاشيّ (ت، 651هـ)، حققه وعلق عليه وشرحه: محمد يوسف حسن، محمود بسيوني خفاجي، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1436ه/ 2015م)، ‏58‏.

صَحَّنَ[مُتَعَدٍّ]

ن575هـ=1179م

صَحَّنَ الشَّيْءَ: أَخْلَاهُ وَأَتْلَفَهُ.

لَوْ أَنَّ النَّارَ كَلَّسَتِ الكَلَّاسَةَ... لَمَا اقْتَصَرَتْ عَلَى المَقْصُورَةِ، وَلَا بَرَّدَتْهَا البَرَّادَةُ، حَتَّى تُصَحِّنَ الصُّحْنَ، وَتُنَسِّرَ النَّسْرَ"
الوَهْرانيّ
منامات الوَهْرانيّ ومقاماته ورسائله: الوَهْرانيّ (ت، 575هـ)، تحقيق: إبراهيم شعلان، محمَّد نغش، مراجعة: عبد العزيز الأهواني، منشورات الجمل، كولونيا، ألمانيا، ط1، (1998م)، ‏18‏.

صُحَيْن[اسْمٌ مُصَغَّر]

ن894هـ=1488م

الصُّحَيْنُ : تَصْغِيرُ الصَّحْنِ، وَهُوَ إِنَاءُ الشُّرْبِ وَنَحْوِهِ.

وَحَقِّ مَنْ مُزِجَتْ صَرْفُ الشَّمُولِ لَهُ مَا ذُقْتُ خَمْرًا، وَلَمْ أَشْرَبْ كَمَنْ شَرِبَا
وَلَا اعْتَرَفْتُ صُحَيْنَ الزِّيهِ فِي عُمُرِي لَكِنَّ أَوْصَافَهُمْ مِنْ عَادَةِ الأُدَبَا
أبو البقاء عبد الله البدري الدِّمشْقِيّ
راحة الأرواح في الحشيش والرّاح: أبو البقاء عبد الله البدري الدِّمشْقِيّ (ت، 894هـ)، Raconter le haschich dans l’époque mamlouke – Etude et édition critique partielle de la Râḥat al-arwâḥ fi-l-ḥašîš wa-l-râḥ, de Badr al-Dîn Abû l-Tuqâ al-Badrî (847 – 894/1443 – 1489), par : Danilo Morino, La Sorbonne, Paris, (2015)، ‏73‏.

صَحْنَة[اسْم]

ن1288هـ=1870م

الصَّحْنَةُ : الإِنَاءُ الوَاسِعُ.

"وَمِنْهُمْ مَنْ يَدُقُّ الأَفَاوِيهَ ثُمَّ يُلْزِقُ عَلَى رَأْسِ كَأْسٍ قِرْطَاسًا ثَخِينًا، وَيُشْبِكُهُ بِالإِبْرَةِ، ثُمَّ يَجْعَلُ عَلَى فَمِ الكَأْسِ حَلْقَةً مِنَ العَجِينِ، ثُمَّ يَضَعُ عَلَيْهِ صَحْنَةَ نُحَاسٍ أَوْ طَابِقَةَ حَدِيدٍ، وَيُوقِدُ عَلَيْهَا، فَيَقْطُرُ الدُّهْنُ نَازِلًا عَنِ الشَّبَكَةِ فِي الكَأْسِ"
محمد كريم بن ابراهيم كرماني
دقائق العلاج في الطب البدني: محمد كريم بن ابراهيم كرماني (ت، 1288هـ)، مطبعة محمد رشيد السعدي، بومباي، (1315ه)، ‏296‏.
ن1307هـ=1889م

صَحْنَةُ الدَّارِ وَنَحْوِهَا: السَّاحَةُ فِي وَسَطِهَا.

"وَفِي كُلِّ أُسْطُوَانَةٍ تَسَامَتِ الأُخْرَى، وَأَمَامَهَا جِدَارٌ سَتْرُهُ يَفْصِلُ بَيْنَ الطَّرِيقِ وَبَيْنَ صَحْنَتَيِ الدَّارَيْنِ"
حسين بن أحمد المرصفيّ
دليل المسترشد في الإنشاء للشيخ حسين المرصفي (تحقيق ودراسة): حسين بن أحمد المرصفيّ (ت، 1307هـ)، إعداد: يوسف عبد اللطيف ناصر، رسالة ماجستير، جامعة الأزهر، (1437ه/ 2016م)، ‏33‏.

مِصْحَن[اسْمُ آلَة]

1310هـ=1892م

المِصْحَنُ : إِنَاءٌ لِلشُّرْبِ قَرِيبُ القَعْرِ.

"وَبِهَذَا الِاجْتِهَادِ تَوَصَّلُوا لِعَمَلِ المُلَاءَاتِ مِنَ الحَرِيرِ وَالقُطْنِ... وَعَمَلِ القُلَلِ وَالدَّوَارِقِ وَالمَحَالِبِ... وَالمِسْرَجَةِ وَالمِصْحَنِ ، كُلُّ ذَلِكَ يُعْمَلُ مِنَ الفَخَّارِ فِي جُرَيْسَ وَقَنَا"
عبد الله النديم
المرافعة الوطنية: ‏عبد الله النديم،‏الأستاذ: مطبعة المحروسة، مصر، س1، ع1، ج5، (29/ 02/ 1310ه)، ‏102‏.
