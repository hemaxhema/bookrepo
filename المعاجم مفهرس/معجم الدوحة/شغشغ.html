شَغْشَغَة[اسْم]

قبل 50ق.هـ=573م

الشَّغْشَغَةُ : حِكَايَةُ صَوْتِ الطَّعْنِ بِالسِّنَانِ إِذَا حُرِّكَ فِي الجَوْفِ.

فَالطَّعْنُ شَغْشَغَةٌ ، وَالضَّرْبُ هَيْقَعَةٌ ضَرْبَ المُعَوِّلِ تَحْتَ الدِّيمَةِ العَضَدَا
عبد مَناف بن رِبْع الجُرَبيّ الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏674‏.

شَغْشَغَة[مَصْدَر]

ن11هـ=632م

شَغْشَغَةُ الفَارِسِ: تَحْرِيكُهُ اللِّجَامَ وَتَرْدِيدُهُ فِي فَمِ الدَّابَّةِ المُمْتَنِعَةِ عَلَيْهِ تَأْدِيبًا.

ذُو غَيِّثٍ بَثْرٍ، يَبُذُّ قَذَالُهُ إِذْ كَانَ شَغْشَغَةٌ ، سِوَارَ المُلْجِمِ
أبو كبير الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏2/‏1092‏.
ن175هـ=791م

الشَّغْشَغَةُ فِي الشُّرْبِ: التَّقْلِيلُ فِيهِ.

" الشَّغْشَغَةُ فِي الشُّرْبِ: التَّصْرِيدُ، أَيِ التَّقْلِيلُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏340‏.

شَغْشَغَ[مُتَعَدٍّ]

136هـ=753م

شَغْشَغَ المَاءَ: كَدَّرَهُ.

لَوْ كُنْتُ أَسْطِيعُكَ لَمْ يُشَغْشَغِ شِرْبِي، وَمَا المَشْغُولُ مِثْلُ الأَفْرَغِ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏2/‏49‏-‏ 50‏.
280هـ=893م

شَغْشَغَ السِّنَانَ وَنَحْوَهُ: أَدْخَلَهُ وَأَخْرَجَهُ وَحَرَّكَهُ لِيَتَمَكَّنَ.

" شَغْشَغَ السِّنَانَ فِي الطَّعْنَةِ"
ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏1/‏206‏.
280هـ=893م

شَغْشَغَ الإِنَاءَ: صَبَّ فِيهِ مَاءً وَنَحْوَهُ وَلَمْ يَمْلَأْهُ.

"وَيُقَالُ: شَغْشَغْتُ الإِنَاءَ"
من رواية ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏1/‏206‏.
352هـ=963م

شَغْشَغَ اللِّجَامَ فِي فَمِ الدَّابَّةِ: رَدَّدَهُ فِيهِ إِذَا امْتَنَعَتْ عَلَيْهِ.

"وَيُقَالُ: شَغْشَغَ المُلْجِمُ اللِّجَامَ فِي فَمِ الدَّابَّةِ"
من رواية الأزهريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏16/‏33‏.

مُشَغْشَغ[اسْمُ مَفْعُول]

ن656هـ=1258م

المُشَغْشَغُ مِنَ الدَّوَّابِ: المُعْجَلُ، الَّذِي يُحَرَّكُ اللِّجَامُ فِي فَمِهِ.

وَأَجَلُّ مَنْ تُحْدَى إِلَيْهِ عُذَافِرٌ مَرِحٌ عَلَى بُعْدِ الفَلَاةِ مُشَغْشَغُ
الصَّرْصَريّ
ديوان الصّرصريّ: الصَّرْصَريّ (ت، 656هـ)، تحقيق وتقديم: مخيمر صالح، منشورات جامعة اليرموك، إِربد، (د.ت)، ‏300‏.
