مَغْص[مَصْدَر]

2هـ=623م

المَغْصُ : الوَجَعُ وَالتَّقَطُّعُ فِي الأَمْعَاءِ.

"إِنَّ أَبَا حَسَنٍ وَجَدَ مَغْصًا فِي بَطْنِهِ"
حديث نبويّ
المعجم الكبير: أبو القاسم الطَّبَرانيّ (ت، 360هـ)، حققه وخرج أحاديثه: حمدي عبد المجيد السلفي، مكتبة ابن تيمية، القاهرة، ط2، (1404هـ/ 1983م)، ‏5/‏46،،‏(رقم الحديث: 4548).

مُمْغِص[اسْمُ فَاعِل]

13هـ=634م

المُمْغِصُ : المُسَبِّبُ لِلْمَغْصِ.

"فَتَبَلَّغْتُ بِصَحْفَتِهِم، وَتَعَلَّلْتُ بِدَرَّةِ لَقْحَتِهِم... لَمْ أَعْدُ سَدَّ الجَوْعَةِ، وَتَوْرِيَةَ العَوْرَةِ، طَوًى مُمْغِصٌ تَهْفُو لَهُ الأَحْشَاء، وَتَجِبُ لَهُ الأَمْعَاء"
أبو بكرٍ الصِّدّيق
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏3/‏231‏.

مَغَص[اسْمُ جِنْس]

ن90هـ=709م

المَغَصُ مِنَ الإِبِلِ: البِيضُ الكِرَامُ.

أَنْتَ وَهـَبْتَ هـَجْمَةً جُرْجُورَا أُدْمًا وَعِيسًا مَغَصًا خُبُورَا
العجّاج
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏3/‏1302‏.

مَغِص[صِفَةٌ مُشَبَّهَة]

ن130هـ=748م

المَغِصُ : الَّذِي يَشْتَكِي المَغْصَ.

جَوْفَاءُ مَحْشُوَّةٌ فِي مُوجَحٍ مَغِصٍ أَضْيَافُهُ جُوَّعٌ، مِنْهُ، مَهَازيلُ
أبو وَجْزة السُّلَميّ السَّعْديّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏5/‏137‏.

مَمْغُوص[اسْمُ مَفْعُول]

ن175هـ=791م

المَمْغُوصُ مِنَ الأَشْخَاصِ: المُصَابُ بِالمَغْصِ.

"المَغْصُ: غِلَظٌ فِي المِعَى وَتَقْطِيعٌ، وَرَجُلٌ مَمْغُوصٌ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏4/‏375‏.
ن646هـ=1248م

المَمْغُوصُ مِنَ الأَمْعَاءِ وَنَحْوِهَا: المُصَابُ بِالمَغْصِ.

"وَإِذَا ضُمِّدَتْ بِهِ المَعِدَةُ المَمْغُوصَةُ مَعَ عُصَارَةِ الأَفْسَنْتِينِ، أَذْهَبَ مَغْصَهَا، وَهُوَ مُقَوٍّ لِلْأَبْدَانِ المُسْتَرْخِيَةِ"
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏3/‏130‏.
761هـ=1359م

المَمْغُوصُ مِنَ الأَشْخَاصِ: الَّذِي فِيهِ تَوَجُّعٌ وَأَلَمٌ، مِنْ حَسَدٍ وَنَحْوِهِ.

لَوْ رَامَ قَصْدًا فُوَيْقَ النَّجْمِ، أَدْرَكَهُ حَتْمًا، وَكُلُّ حَسُودٍ فِيهِ مَمْغُوصُ
ابن جابر الأندلسيّ
ديوان المقصد الصّالِح في مدح الملك الصّالِح: ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: أحمد فوزي الهيب، دار سعد الدِّين، دمشق، ط1، (1429ه/ 2008م)، ‏108‏.

مُتَمَغِّص[اسْمُ فَاعِل]

203هـ=818م

المُتَمَغِّصُ مِنْ خَبَرٍ: المُتَوَجِّسُ مِنْهُ الخَائِفُ أَلَّا يَكُونَ حَقًّا.

"يُقَالُ: أَنَا مُتَمَغِّصٌ مِنْ هَذَا الخَبَرِ"
من رواية النَّضْر بن شُميل المازنيّ التّميميّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏31‏.

إِمْغَاص[مَصْدَر]

268هـ=881م

إِمْغَاصُ الشَّخْصِ: إِصَابَتُهُ بِآلَامٍ وَتَقْطِيعٍ فِي أَمْعَائِهِ.

فِي رِبَاعٍ تَرْتَادُ عَيْنُكَ فِيهَا حِلَلَ المُلْكِ مُفْضِيَاتٍ عِرَاصُهْ
شَرَفٌ يُمْغِصُ الحَسُودَ، وَمِنْ أَدْ نَى جَزَاءٍ لِحَاسِدٍ إِمْغَاصُهْ
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏2/‏1189‏.

أَمْغَصَ[مُتَعَدٍّ]

268هـ=881م

أَمْغَصَهُ : جَعَلَهُ مُتَوَجِّعًا مُتَأَلِّمًا.

فِي رِبَاعٍ تَرْتَادُ عَيْنُكَ فِيهَا حِلَلَ المُلْكِ مُفْضِيَاتٍ عِرَاصُهْ
شَرَفٌ يُمْغِصُ الحَسُودَ، وَمِنْ أَدْ نَى جَزَاءٍ لِحَاسِدٍ إِمْغَاصُهْ
البُحْتُريّ
ديوان البُحْتُريّ: عني بتحقيقه وشرحه والتعليق عليه: حسن كامل الصيرفي، دار المعارف، القاهرة، ط3، (1963م)، ‏2/‏1189‏.
ن313هـ=925م

أَمْغَصَهُ : أَصَابَهُ بِآلَامٍ وَتَقْطِيعٍ فِي أَمْعَائِهِ.

"فَإِنَّ الدَّوَاءَ إِذَا لَقِيَ طَبِيعَةً يَابِسَةً قَوِيَّةً أَكْرَبَ، وَأَمْغَصَ ، وَقَلَّ فِعْلُهُ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏2/‏326‏.

تَمَغَّصَ[لَازِم]

329هـ=941م

تَمَغَّصَ الشَّخْصُ: تَوَجَّعَ وَتَأَلَّمَ مِنْ جَوْفِهِ.

"فَعَمَدَتَا إِلَى خَشَبِ العُشَرِ، فَدَقَّتَاهُ، وَهُوَ مُسَهِّلٌ، وَطَرَحَتَاهُ فِي شَرَابِهِ، ... فَلَمَّا أَخَذَ الشَّرَابُ مَأْخَذَهُ مِنْهُ، تَنَاوَمَ المَوْلَى، وَتَمَغَّصَ الفَتَى مِنْ جَوْفِهِ"
أحمد بن محمّد العَروضيّ
مروج الذهب ومعادن الجوهر: عليّ بن الحسين المسعوديّ (ت، 346هـ)، عني بتنقيحها وتصحيحها: شارل بلا، طبعة بربيه دي مينار، بافيه دي كرتاي، منشورات الجامعة اللّبنانيّة، بيروت، (1965- 1974م)، ‏5/‏224‏-‏ 225‏.

مَغَصَ[مُتَعَدٍّ]

418هـ=1027م

مَغَصَ الشَّخْصَ: أَصَابَهُ بِالمَغْصِ.

"وَإِذَا لَمْ يُسْتَمْرَأْ مَغَصَ ، وَنَفَخَ"
ابن سينا
القانون في الطّبّ: ابن سينا (ت، 428هـ)، وضع حواشيه: محمّد أمين الضنّاوي، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏1/‏391‏.

مَغَّصَ[مُتَعَدٍّ]

ن555هـ=1160م

مَغَّصَهُ : أَوْجَعَهُ وَآلَمَهُ.

" يُمَغِّصُ وَيُسْحِجُ"
العلائيّ المغربيّ
تقويم الأدوية المفردة أو المنجِح في التداوي من جميع الأمراض والشكاوي: العلائيّ المغربيّ (ت، 555هـ)، تحقيق ودراسة: هشام الأحمد، مراجعة: محمود مصري، معهد المخطوطات العربية، القاهرة، ط1، (1432ه/ 2011م)، ‏236‏.

مَغْصِيّ[اسْمٌ مَنْسُوب]

ن610هـ=1213م

المَغْصِيُّ : المَنْسُوبُ إِلَى المَغْصِ.

"وَأَمَّا عَلَامَةُ مَا كَانَ سَبَبَهُ كَثْرَةُ انْصِبَابِ المَرَارِ، فَلَذْعُهُ المِعَى وَالإِسْهَالُ، ... وَأَنْ يَكُونَ الأَلَمُ مَغْصِيًّا "
ابن هَبَل
المختارات في الطب: ابن هَبَل (ت، 610هـ)، مطبعة جمعية دائرة المعارف العثمانية، حيدر آباد، ط1، (1362ه)، ‏3/‏319‏.

تَمَغُّص[مَصْدَر]

ن687هـ=1288م

التَّمَغُّصُ : الشُّعُورُ بِالمَغْصِ.

"فَإنَّمَا يَكُونُ ذَلِكَ، بِمَا يَتْبَعُ الِاسْتِفْرَاغَ مِنَ الإِفْرَاطِ فِي الإِسْهَالِ، أَوْ حُدُوثِ سَحْجٍ أَوْ تَمَغُّصٍ ، وَنَحْوِ ذَلِكَ"
ابن النفيس
الشامل في الصناعة الطبية: ابن النفيس (ت، 687هـ)، دراسة وتحقيق: يوسف زيدان، المجمع الثقافي، أبوظبي، (1421ه/ 2000م)، ‏10-1/‏262‏.

تَمْغِيص[مَصْدَر]

ن748هـ=1347م

التَّمْغِيصُ : الأَلَمُ الَّذِي يَأْخُذُ بِالأَمْعَاءِ.

"وَيَنْفَعُ مِنْ حَرْقِ النَّارِ ضِمَادًا، وَيَمْنَعُ التَّمْغِيصَ ، وَيُسَكِّنُ أَوْجَاعَ العَيْنِ"
الذهبيّ
الطب النبوي: شمس الدين الذهبيّ (ت، 748هـ)، تحقيق: أحمد رفعت البدراوي، دار إحياء العلوم، بيروت، ط3، (1410هـ/1990م)، ‏86‏.
761هـ=1359م

تَمْغِيصُ الشَّخْصِ: تَجْرِيعُهُ غُصَصَ الأَلَمِ.

لَا زِلْتَ تَلْبَسُ بُرْدَ الحَمْدِ مُحْتَكِمًا وَلِلْعُدَاةِ بِمَا خُوِّلْتَ تَمْغِيصُ
ابن جابر الأندلسيّ
ديوان المقصد الصّالِح في مدح الملك الصّالِح: ابن جابر الأندلسيّ (ت، 780هـ)، تحقيق: أحمد فوزي الهيب، دار سعد الدِّين، دمشق، ط1، (1429ه/ 2008م)، ‏108‏.

مَاغِص[اسْمُ فَاعِل]

927هـ=1520م

المَاغِصُ : مَا يُسَبِّبُ المَغْصَ فِي الأَمْعَاءِ.

"وَيُحْكَى أَنَّ أَحَدَهُمَا أَكَلَ حَبَّةَ مَوْزٍ أَوْ قِثَّاءٍ، فَمَغَصَهُ بَاطِنُهُ، فَتَقَيَّأَهَا الآخَرُ عَنْهُ، فَبَرِئَ مِنَ المَاغِصِ "
أبو محمد الطيب بن عبد الله بن أحمد بن علي بامخرمة الهجراني الحضرمي الشافعي
قلادة النحر في وفيات أعيان الدهر: أبو محمد الطيب بن عبد الله بن أحمد بن علي بامخرمة الهجراني الحضرمي الشافعي (ت، 947هـ)، عني به: بوجمعة مكري، خالد زواري، دار المنهاج، بيروت- جدة، ط1، (1428ه/ 2008م)، ‏5/‏69‏.

مَاغِص[اسْم]

ن1008هـ=1599م

المَاغِصُ : دَاءٌ يُصِيبُ فَمَ المَعِدَةِ.

"وَالأَوْجَاعُ الَّتِي تَهِيجُ مِنْهُ فَهِيَ: الفَالِجُ، وَالنِّقْرِسُ، وَالمَاغِصُ الَّذِي يَأْخُذُ عَلَى فَمِ المَعِدَةِ، وَالغِشَاءِ"
داود بن عمر الأنطاكي
الملكي في الطب: داود بن عمر الأنطاكي (ت، 1008هـ)، تحقيق: أحمد فريد المزيدي، دار الكتب العلمية، بيروت، ط1، (1424ه/ 2002م)، ‏1/‏220‏.
