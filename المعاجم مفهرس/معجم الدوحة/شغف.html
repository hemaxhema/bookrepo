شُغِفَ[مُلَازِمٌ لِلْبِنَاءِ لِلْمَجْهُول]

قبل 150ق.هـ=476م

شُغِفَ بِهِ: أُولِعَ بِهِ.

أَشَمُّ كَغُصْنِ البَانِ جَعْدٌ مُرَجَّلٌ شُغِفْتُ بِهِ لَوْ كَانَ شَيْئًا مُدَانِيَا
منسوب إلى هند بنت الخُسّ الإياديّة
التذكرة الحمدونية: ابن حَمدون (ت، 562هـ)، تحقيق: إحسان عبّاس، بكر عبّاس، دار صادر، بيروت، ط1، (1996م)، ‏9/‏417‏.

شَغَفَ[مُتَعَدٍّ]

قبل 50ق.هـ=573م

شَغَفَهُ الحُبُّ وَنَحْوُهُ: بَلَغَ شَغَافَ قَلْبِهِ.

أَيَقْتُلُنِي أَنِّي شَغَفْتُ فُؤَادَهَا كَمَا شَغَفَ المَهْنُوءَةَ الرَّجُلُ الطَّالِي؟
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏137‏.
قبل 50ق.هـ=573م

شَغَفَ النَّاقَةَ وَنَحْوَهَا: أَصَابَهَا بِمَا يُؤْلِمُهَا.

أَيَقْتُلُنِي وَقَدْ شَغَفْتُ فُؤَادَهَا كَمَا شَغَفَ المَهْنُوءَةَ الرَّجُلُ الطَّالِي؟
امرؤ القيس بن حُجْر الكنديّ
ديوان امرئ القيس: اعتنى به وشرحه: عبد الرحمن المصطاوي، دار المعرفة، بيروت، ط2، (1425هـ/ 2004م)، ‏137‏.
ن50هـ=670م

شَغَفَهُ : أَثَارَ وَلَعَهُ وَتَعَلُّقَهُ بِهِ.

تَذَكَّرْتُ لَيْلَى يَوْمَ أَصْبَحَتُ قَافِلًا بِزِيزَاءَ وَالذِّكْرَى تَشُوقُ وَتَشْغَفُ
مُلَيح بن الحَكَم الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏3/‏1042‏.

شَغَف[مَصْدَر]

قبل 50ق.هـ=573م

الشَّغَفُ بِالشَّيْءِ: الوَلَعُ بِهِ وَتَعَلُّقُهُ.

لِمَنْ طَلَلٌ دَاثِرٌ آيُهُ أَضَرَّ بِهِ سَالِفُ الأَحْرُسِ
تُنْكِرُهُ العَيْنُ مِنْ جَادِثٍ وَيَعْرِفُهُ شَغَفُ الأَنْفُسِ؟
امرؤ القيس بن حُجْر الكنديّ
حلية المحاضرة في صناعة الشعر: محمّد بن الحسن الحاتميّ (ت، 388هـ)، تحقيق: جعفر الكتاني، دار الرشيد، بغداد، (1979م)، ‏1/‏208‏.

الشَّغَفُ : غِشَاءُ القَلْبِ.

إِنِّي لَأَهْوَاكِ غَيْرَ ذِي كَذِبٍ قَدْ شُفَّ مِنِّي الأَحْشَاءُ وَالشَّغَفُ
قيس بن الخَطيم الأوسيّ
ديوان قيس بن الخطيم: تحقيق: ناصر الدين الأسد، دار صادر، بيروت، (1967م)، ‏112‏.

شَاغِف[اسْمُ فَاعِل]

قبل 50ق.هـ=573م

الشَّاغِفُ مِنَ الحُبِّ وَنَحْوِهِ: البَالِغُ شَغَافَ القَلْبِ.

وَفِي الحَيِّ أَبْكَارٌ سَبَيْنَ فُؤَادَهُ عُلَالَةَ مَا زَوَّدْنَ وَالحُبُّ شَاغِفِي
المُرَقِّش الأكبر البكريّ
ديوان المرَقِّشين المرقِّش الأكبر عمرو بن سعد، المرقِّش الأصغر عمرو بن حَرملة: تحقيق: كارين صادر، دار صادر، بيروت، ط1، (1998م)، ‏59‏.
ن110هـ=728م

الشَّاغِفُ مِنَ الضُّلُوعِ وَنَحْوِهَا: ذُو الشَّغَافِ.

أَبَى القَلْبُ إِلَّا أَنْ يُسَلَّى بِحَاجَةٍ أَتَى ذِكْرُهَا بَيْنَ الحَشَا وَالشَّوَاغِفِ
الفرزدق
شرح ديوان الفرزدق: ضبط معانيه وشروحه وأكملها: إيليّا الحاوي، دار الكتاب اللبناني- مكتبة المدرسة، بيروت، ط1، (1983م)، ‏2/‏89‏.
126هـ=744م

الشَّاغِفُ مِنَ الأَعْضَاءِ: ذُو الدَّاءِ القَاتِلِ.

وَقَاصِدٌ إِنْ قَصَدُوا مُنَاصِفُ يَشْتَدُّ حَتَّى تَبْرَأَ النَّكَائِفُ مِنَ المِرَاضِ، وَالطِّحَالُ الشَّاغِفُ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏196‏.
ن350هـ=961م

الشَّاغِفُ مِنَ الصَّيْدِ وَنَحْوِهِ: المُثِيرُ لِلْوَلَعِ وَالتَّعَلُّقِ بِهِ.

وَمِنْ شَغَفِي بِالصَّيْدِ، وَالصَّيْدُ شَاغِفٌ مُطَارَدَتِي لِلْوَحْشِ، وَالفَهْدُ لِي رِدْفُ
من شواهد بازيار العزيز بالله الفاطميّ
البَيْزَرة: بازيار العزيز بالله الفاطميّ (ت، 350هـ)، نظر فيه وعلق عليه: محمّد كرد علي، مطبوعات المجمع العلمي العربي، دمشق، د.ت. ‏128‏.

كتاب المغازي: الواقديّ (ت، 207هـ)، تحقيق: مارسدن جونس، عالم الكتب، القاهرة، ط3، (1404هـ/ 1984م)، ‏1/‏186‏.

اشْتِغَاف[مَصْدَر]

ن1283هـ=1866م

الِاشْتِغَافُ بِالشَّيْءِ: الوَلَعُ وَالتَّعَلُّقُ بِهِ.

"فَلِذَلِكَ أَرْجُو كُلَّ مُطَّلِعٍ عَلَى هَذَا المُؤَلَّفِ مِنْ إِخْوَتِي المَسِيحِيِّينَ الأَرْثُودُوكْسِيِّينَ مُتَوَسِّلًا إِلَيْهِ قَبُولَهُ بِرَغْبَةٍ وَاشْتِغَافٍ "
أغابيوس بن فارس صليبا
صخرة شك: أغابيوس بن فارس صليبا (ت، 1283هـ)، الأكاديمية الإفطوكراطورية، بطرسبرج، (1859م)، ‏1/‏6‏.

شَغَافِيّ[اسْمٌ مَنْسُوب]

1369هـ=1949م

الشَّغَافِيُّ : المَنْسُوبُ إِلَى الشَّغَافِ.

"إِنَّ القَوْلُونَ أَوْ بِالحَرِيِّ تَجْوِيفَ الجِسْمِ المُكَوَّنَ مِنَ الطَّبَقَةِ الجُرْثُومِيَّةِ الوُسْطَى يَنْقَسِمُ فِي الفَقَارِيَّاتِ تَجْوِيفَيْنِ أَوْ ثَلَاثَةَ تَجَاوِيفَ، فَأَوَّلُ تَحْدِيدٍ يُصِيبُ القَوْلُونَ نُشُوءُ الحِجَابِ المُسْتَعْرِضِ فِي الأَسْمَاكِ، وَهُوَ الَّذِي يُؤَلِّفُ التَّجْوِيفَ التَّامُورِيَّ أَوِ الشَّغَافِيَّ "
إسماعيل مظهر
الحيتان: إسماعيل مظهر (ت، 1382هـ)، مؤسسة هنداوي، وندسور، المملكة المتحدة، (2020م)، ‏35‏.
