شِدْق[اسْم]

قبل 50ق.هـ=573م

الشِّدْقُ : جَانِبُ الفَمِ مِمَّا تَحْتَ الخَدِّ.

صَعْبُ البَدِيهَةِ، مَشْبُوبٌ أَظَافِرُهُ مُوَاثِبٌ، أَهْرَتُ الشِّدْقَيْنِ ، هِرْمَاسُ
مالك بن خالد الخُناعيّ الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏3/‏5‏.
ن538هـ=1143م

شِدْقُ الوَادِي: نَاحِيَتُهُ وَجَانِبُهُ.

"وَأَقْبَلَ سَيْلٌ، فَأَفْعَمَ أَشْدَاقَ الأَوْدِيَةِ"
جار الله الزمخشريّ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، تحقيق: محمد باسل عيون السود، دار الكتب العلمية، بيروت، ط1، (1419ه/ 1998م)، ‏1/‏499‏.

مُتَشَدِّق[اسْمُ فَاعِل]

ن11هـ=632م

المُتَشَدِّقُ : المُتَفَاصِحُ، الَّذِي يَلْوِي شِدْقَهُ بِالكَلَامِ تَكَلُّفًا.

"أَلاَ أُنَبِّئُكُمْ بِشِرَارِكُمْ؟... هُمُ الثَّرْثَارُونَ المُتَشَدِّقُونَ "
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏14/‏418،،‏(رقم الحديث: 8822).

مُتَشَدِّق[صِفَةٌ مُشَبَّهَة]

ن190هـ=806م

المُتَشَدِّقُ مِنَ الخُطَبَاءِ وَنَحْوِهِمْ: الفَصِيحُ المُفَوَّهُ.

فَيَوْمًا يُبَارُونَ الرِّيَاحَ سَمَاحَةً وَيَوْمًا لِبَذْلِ الخَاطِبِ المُتَشَدِّقِ
مروان بن أبي حفصة
شعر مروان بن أبي حفصة (ت، 190هـ): جمعه وحققه وقدم له: حسين عطوان، دار المعارف، القاهرة، ط3، (1982م)، ‏69‏.

تَشَدَّقَ[لَازِم]

ن11هـ=632م

تَشَدَّقَ الشَّخْصُ: تَفَاصَحَ، وَلَوَى شِدْقَهُ بِالكَلَامِ تَكَلُّفًا.

"سَيَكُونُ نَشْوٌ مِنْ أُمَّتِي يُولَدُونَ، فِي النَّعِيمِ وَيَغْذُونَ بِهِ، هِمَّتُهُمْ أَلْوَانُ الطَّعَامِ وَأَلْوَانُ الثِّيَابِ، يَتَشَدَّقُونَ بِالْقَوْلِ، أُولَئِكَ شِرَارُ أُمَّتِي"
حديث نبويّ
الزّهد: أحمد ابن حَنْبَل (ت، 241هـ)، وضع حواشيه: محمّد عبد السّلام شاهين، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏319،،‏(رقم الحديث: 2330).
ن1327هـ=1908م

تَشَدَّقَ الحَيَوَانُ: حَرَّكَ شِدْقَيْهِ لِمَضْغِ الطَّعَامِ، أَوِ اجْتِرَارِهِ.

"يَتَشَدَّقُ الكَلَامَ بِلِسَانِهِ، كَمَا تَتَشَدَّقُ البَقَرُ"
محمد بن عبد الكبير الكتاني
كتاب الذب عن التصوف: محمد بن عبد الكبير الكتاني (ت، 1327هـ)، تحقيق: عدنان عبد الله زُهار، دار الكتب العلمية، بيروت، ط1، (1428ه/ 2007م)، ‏406‏.

مُتَشَادِق[اسْمُ فَاعِل]

ن11هـ=632م

المُتَشَادِقُ : المُتَفَاصِحُ، الَّذِي يَلْوِي شِدْقَهُ بِالكَلَامِ تَكَلُّفًا.

"وَإِنَّ أَبْغَضَكُمْ إِلَى اللهِ، وَأَبْعَدَكُمْ مِنِّي يَوْمَ القِيَامَةِ الثَّرْثَارُونَ، المُفَيْهِقُونَ، المُتَشَادِقُونَ "
حديث نبويّ
الجامع في الحديث: عبد الله بن وهب بن مسلم القُرَشيّ (ت، 197هـ)، ضبط وتخريج وتحقيق: مصطفى حسن حسين محمّد أبو الخير، دار ابن الجوزي، الرياض، ط1، (1416هـ/ 1995م)، ‏1/‏580،،‏(رقم الحديث: 479).

تَشَادَقَ[لَازِم]

ن11هـ=632م

"إِنَّ مِنْ شِرَارِ أُمَّتِي الَّذِينَ غُذُّوا بِالنَّعِيمِ، الَّذِينَ يَطْلُبُونَ أَلْوَانَ الطَّعَامِ، وَأَلْوَانَ الثِّيَابِ، يَتَشَادَقُونَ بِالكَلَامِ"
حديث نبويّ
الزّهد: أحمد ابن حَنْبَل (ت، 241هـ)، وضع حواشيه: محمّد عبد السّلام شاهين، دار الكتب العلميّة، بيروت، ط1، (1420هـ/ 1999م)، ‏66،،‏(رقم الحديث: 402).

أَشْدَق[صِفَةٌ مُشَبَّهَة]

ن30هـ=651م

الأَشْدَقُ : الوَاسِعُ الشِّدْقَيْنِ.

ثُمَّ انْكَفَأْنَا إِلَى حِرْزٍ لَنَا جَبَلٌ صُلْنَا عَلَيْهِمْ صُوَالَ الأَشْدَقِ الضَّارِي
سارية بن زُنيم الدُّؤَليّ
تاريخ مدينة دمشق وذكر فضلها وتسمية من حلها من الأماثل أو اجتاز بنواحيها من وارديها وأهلها: ابن عساكر (ت، 571هـ)، دراسة وتحقيق: أبي سعيد عمر بن غرامة العَمروي، دار الفكر، بيروت، ط1، (1415هـ/ 1995م- 1431هـ/ 2000م)، ‏20/‏28‏.
ن60هـ=680م

الأَشْدَقُ مِنَ الأَشْخَاصِ: الفَصِيحُ المُفَوَّهُ.

"إِنَّ ابْنَ سَعِيدٍ هَذَا لَأَشْدَقُ "
معاوية بن أبي سفيان
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏1/‏316‏.
425هـ=1034م

الأَشْدَقُ مِنَ الأَلْفَاظِ: الجَهِيرُ، الَّذِي يَمْلَأُ الفَمَ.

جَاحِظُ العَيْنِ، وَلَمَّا يُخْنَقِ بِلَحْظِ مَخْنُوقٍ، وَلَفْظٍ أَشْدَقِ
من شواهد الرّاغب الأصفهانيّ
مجْمع البلاغة: الرّاغب الأصفهانيّ (ت، 425هـ)، تحقيق: عُمر عبد الرَّحمن السّاريسيّ، مكتبة الأَقصى، عمّان، ط1، (1406ه/ 1986م)، ‏2/‏800‏.

شَدَقَ[مُتَعَدٍّ]

ن58هـ=678م

شَدَقَ الشَّخْصَ: آلَمَهُ بِكَلَامِهِ وَنَحْوِهِ.

"فَلَمَّا ذَكَرَ خَدِيجَةَ، قُلْتُ: قَدْ أَبْدَلَكَ اللهُ مِنْ كَبِيرَةِ السِّنِّ حَدِيثَةَ السِّنِّ، فَشَدَقَنِي ، وَقَالَ: مَا عَلَيَّ، أَوْ نَحْوَ هَذَا، أَنْ كَانَ اللهُ رَزَقَهَا مِنِّي الوَلَدَ، وَلَمْ يَرْزُقْكِيهِ"
عائشة بنت أبي بكر الصِّدّيق
كتاب الدّلائل في غريب الحديث: قاسم بن ثابت السَّرَقُسْطيّ (ت، 302هـ)، تحقيق: محمّد بن عبد الله القناص، مكتبة العبيكان، الرياض، ط1، (1422هـ/ 2001م)، ‏2/‏720‏.
ن175هـ=791م

شَدَقَ اللِّجَامَ: أَدْخَلَهُ فِي فَمِ الدَّابَةِ.

"وَاللِّجَامُ الشَّادِقُ: الدَّاخِلُ الفَمِ، وَشَدَقَهُ يَشْدِقُهُ شَدْقًا وَأَشْدَقْتُهُ أَنَا إِيَّاهُ إِشْدَاقًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏34‏.

اشَّدَّقَ[لَازِم]

ن120هـ=738م

اشَّدَّقَ الشَّخْصُ: تَفَاصَحَ، وَلَوَى شِدْقَهُ بِالكَلَامِ تَكَلُّفًا.

"وَالمُتَشَادِقُونَ: الَّذِينَ يَشَّدَّقُونَ فِي الحَدِيثِ"
عقبة بن مسلم التُّجَيبيّ
الجامع في الحديث: عبد الله بن وهب بن مسلم القُرَشيّ (ت، 197هـ)، ضبط وتخريج وتحقيق: مصطفى حسن حسين محمّد أبو الخير، دار ابن الجوزي، الرياض، ط1، (1416هـ/ 1995م)، ‏1/‏580‏.

تَشْدِيق[مَصْدَر]

126هـ=744م

التَّشْدِيقُ : التَّفَاصُحُ، وَلَيُّ الشِّدْقِ بِالكَلَامِ تَكَلُّفًا.

وَأَلْحَنُ النَّاسِ، كُلِّ النَّاسِ قَاطِبَةً وَكَانَ يُولَعُ بِالتَّشْدِيقِ فِي الخُطَبِ
يحيى بن نوفل الحِمْيريّ
الكامل في اللغة والأدب: أبو العباس، تح: محمد أحمد الدالي ، الرسالة ، 1992م، ‏1/‏46‏.
ن444هـ=1052م

التَّشْدِيقُ فِي قِرَاءَةِ القُرْآنِ: التَّكَلُّفُ بِتَطْوِيلِ الحُرُوفِ، حَتَّى يَمِيلَ مَعَهَا أَحَدُ شِدْقَيِ القَارِئِ. [علوم القرآن والقراءات]

"وَأَمَّا صِفَةُ مَنْ يَنْتَحِلُ قِرَاءَةَ حَمْزَةَ فَأَكْثَرُ مَنْ رَأَيْنَا مِنْهُمْ مَا يَنْبَغِي أَنْ تُحْكَى قِرَاءَتُهُ؛ لِفَسَادِهَا؛ وَلِأَنَّهَا مَصْنُوعَةٌ مِنْ تِلْقَاءِ أَنْفُسِهِمْ، وَأَمَّا مَنْ كَانَ مِنْهُمْ يَعْدِلُ فِي قِرَاءَتِهِ حَدْرًا، أَوْ تَحْقِيقًا فَصِفَتُهَا المَدُّ العَدْلُ، وَالقَصْرُ وَالهَمْزُ المُقَوَّمُ، وَالتَّشْدِيدُ المُجَوَّدُ، بِلَا تَمْطِيطٍ، وَلَا تَشْدِيقٍ ، وَلَا تَعْلِيَةِ صَوْتٍ، وَلَا تَرْعِيدٍ، فَهَذِهِ صِفَةُ التَّحْقِيقِ"
أبو عمرو الدّانيّ الأندلسيّ
التّحديد في الإتقان والتّجويد: أبو عمرو الدّانيّ الأندلسيّ (ت، 444هـ)، دراسة وتحقيق: غانم قدُّوري الحمد، دار عمار، عمان، ط1، (1421هـ/ 2000م)، ‏93‏.

كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏34‏.

شَدْق[مَصْدَر]

ن175هـ=791م

شَدْقُ اللِّجَامِ وَنَحْوِهِ: دُخُولُهُ فِي فَمِ الدَّابَّةِ.

"وَاللِّجَامُ الشَّادِقُ: الدَّاخِلُ الفَمِ، وَشَدَقَهُ يَشْدِقُهُ شَدْقًا ، وَأَشْدَقْتُهُ أَنَا إِيَّاهُ إِشْدَاقًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏34‏.

تَشَادُق[مَصْدَر]

ن180هـ=796م

التَّشَادُقُ : التَّفَاصُحُ، وَلَيُّ الشِّدْقِ بِالكَلَامِ تَكَلُّفًا.

لَهُ حَنْجَرٌ رَحْبٌ، وَقَوْلٌ مُنَقَّحٌ وَفَصْلُ خِطَابٍ، لَيْسَ فِيهِ تَشَادُقُ
خلف الأحمر
البيان والتّبيين: الجاحظ (ت، 255هـ)، بتحقيق وشرح: عبد السلام هارون، مكتبة الخانجي- مطبعة المدني، القاهرة، ط7، (1418هـ/ 1998م)، ‏1/‏129‏.

تَشَدُّق[مَصْدَر]

ن302هـ=914م

التَّشَدُّقُ : التَّفَاصُحُ، وَلَيُّ الشِّدْقِ بِالكَلَامِ تَكَلُّفًا.

"فَإِنْ لَمْ يَكُ مِنْ هَذَا؛ فَإِنَّ مَعْنَاهُ تَشْقِيقُ الكَلَامِ، وَالتَّشَدُّقُ فِيهِ"
قاسم بن ثابت السَّرَقُسْطيّ
كتاب الدّلائل في غريب الحديث: قاسم بن ثابت السَّرَقُسْطيّ (ت، 302هـ)، تحقيق: محمّد بن عبد الله القناص، مكتبة العبيكان، الرياض، ط1، (1422هـ/ 2001م)، ‏2/‏525‏.

مُتَمَشْدِق[اسْمُ فَاعِل]

346هـ=957م

المُتَمَشْدِقُ : المُتَقَعِّرُ فِي كَلَامِهِ تَفَاصُحًا.

"إِنَّ هَذَا المُتَمَشْدِقَ كَثِيرُ الإِدْلَالِ عَلَيْكَ، وَأَنْتَ تُعْطِيهِ كُلَّ سَنَةٍ ثَلَاثَةَ آلَافِ دِينَارٍ، عَنْ ثَلَاثِ قَصَائِدَ، وَيُمْكِنُ أَنْ تُفَرِّقَ مِئَتَيْ دِينَارٍ عَلَى عِشْرِينَ شَاعِرًا، يَأْتُونَ بِمَا هُوَ خَيْرٌ مِنْ شِعْرِهِ"
أبو فراس الحَمْدانيّ
الصّبح المنبي عن حيثية المتنبي: يوسف البديعيّ الدِّمشقيّ (ت، 1073هـ)، تحقيق: مصطفى السَّقا وآخرين، دار المعارف، القاهرة، ط3، (1994م)، ‏87‏-‏88‏.

شَدَّاق[صِيغَةُ مُبَالَغَة]

360هـ=971م

الشَّدَّاقُ : المُتَفَاصِحُ، الَّذِي يَلْوِي شِدْقَهُ بِالكَلَامِ تَكَلُّفًا.

"لَا تَنْكِحُوا مِنَ النِّسَاءِ سِتَّةً: أَنَّانَةً، وَلَا مَنَّانَةً، وَلَا حَنَّانَةً، وَلَا حَدَّاقَةً، وَلَا بَرَّاقَةً، وَلَا شَدَّاقَةً "
بعض العرب
قوت القلوب في معاملة المحبوب ووصف طريق المريد إلى مقام التوحيد: أبو طالب المكّيّ (ت، 386هـ)، حققه وقدم له وعلق حواشيه: محمود إبراهيم محمّد الرضواني، مكتبة دار التراث، القاهرة، ط1، (1422هـ/ 2001م)، ‏3/‏1639‏.

شَدَّقَ[مُتَعَدٍّ]

ن391هـ=1001م

شَدَّقَ الصَّبِيَّ وَنَحْوَهُ: أَصَابَ فَتْحَةَ دُبُرِهِ، فَجَعَلَهَا كَالشِّدْقِ.

فَشَدَّقْتُ الصَّبِيَّ، فَدَتْهُ نَفْسِي، بِدُودِيكِي، وَتِيمَرْدَمْ دَرَسْتِ
ابن الحجّاج النِّيلي البغداديّ
يتيمة الدّهر في محاسن أهل العصر: أبو منصور الثّعالبيّ (ت، 429هـ)، شرح وتحقيق: مفيد محمّد قميحة، دار الكتب العلمية، بيروت، ط1، (1403هـ/ 1983م)، ‏3/‏107‏.
671هـ=1272م

"الَّذِي خَلَقَ الرَّحَى يَأْتِيهَا بِالطَّحِينِ، وَالَّذِي شَدَّقَ الأَشْدَاقَ هُوَ خَالِقُ الأَرْزَاقِ"
رجل من العرب
الجامع لأحكام القرآن والمبيِّن لما تضمنه من السنة وآي الفرقان: القرطبيّ (ت، 671هـ)، تحقيق: عبد الله بن عبد المحسن التركي، محمد رضوان عرقسوسي، مؤسسة الرسالة، بيروت، ط1، (1427ه/ 2006م)، ‏11/‏72‏.

شَدَّقَ[لَازِم]

425هـ=1034م

"لِأَنَّكَ تُعَلِّقُ، وَتُشَدِّقُ ، وَتُحَدِّقُ"

محاضرات الأدباء ومحاورات الشّعراء والبلغاء: الرّاغب الأصفهانيّ (ت، 425هـ)، تحقيق: رياض عبد الحميد مراد، دار صادر، بيروت، ط1، (1425ه/ 2004م)، ‏2/‏607‏.
ن701هـ=1301م
