لَقَّاعَة[اسْمُ جِنْس]

قبل 13ق.هـ=609م

اللَّقَّاعَةُ : ضَرْبٌ مِنَ الذُّبَابِ أَخْضَرُ يَلْسَعُ.

فَسَلِّ طِلَابَهَا، وَتَعَزَّ عَنْهَا بِنَاجِيَةٍ تَخَيَّلُ فِي الرِّكَابِ
طَوَتْ قَرْنًا، وَلَمْ تَطْعَمْ خَبِيًّا، وَأَظْهَرَ كَشْحَهَا لَقَعُ الذُّبَابِ
أوفى بن مطر المازنيّ
معجم البلدان: ياقوت الحَمَويّ (ت، 626هـ)، دار صادر، بيروت، (1397ه/ 1977م)، ‏5/‏27‏.

لَقَعَ[مُتَعَدٍّ]

32هـ=653م

لَقَعَهُ : أَصَابَهُ بِالعَيْنِ.

"جَاءَتْ جَارِيَةٌ إِلَى سَيِّدِهَا، وَقَالَتْ: مَا يُقْعِدُكَ؟ قُمْ فَابْتَغِ رَاقِيًا، فَإِنَّ فُلَانًا قَدْ لَقَعَ فَرَسَكَ، فَتَرَكَهُ يَدُورُ كَأَنَّهُ فَلَكٌ"
إحدى الجواري
كتاب الدعاء: محمّد بن فُضيل الضَّبّيّ (ت، 195هـ)، تحقيق ودراسة وتخريج: عبد العزيز بن سليمان بن إبراهيم البعيمي، مكتبة الرشد، الرياض، ط1، (1419هـ/ 1999م)، ‏298،،‏(رقم الحديث: 117).
ن175هـ=791م

لَقَعَ الشَّيْءَ: رَمَى بِهِ.

" لَقَعْتُ الشَّيْءَ: رَمَيْتُ بِهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏167‏.
ن175هـ=791م

لَقَعَهُ بِبَعْرَةٍ وَنَحْوِهَا: رَمَاهُ بِهَا.

" لَقَعَهُ بِبَعْرَةٍ: رَمَاهُ بِهَا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏167‏.

لَقَعَ[لَازِم]

352هـ=963م

لَقَعَ الشَّخْصُ: مَرَّ مُسْرِعًا.

صَلَنْقَعٌ بَلَنْقَعُ وَسْطَ الرِّكَابِ يَلْقَعُ
من شواهد الأزهريّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏1/‏248‏.

لَقْعَة[اسْمُ مَرَّة]

40هـ=660م

اللَّقْعَةُ : المَرَّةُ مِنَ الرَّمْيِ.

"يَا أَمِيرَ المُؤْمِنِينَ، بَلَغَنِي أَنَّ هَذِهِ الذِّئَابَ مِنَ الكُوفَةِ يَأْتُونَ يُعَنِّفُونَكَ فِي قَتْلِ المُقَاتِلَةِ... وَلَنْ يَكُفَّهُمْ عَنْكَ إِلَّا شَيْءٌ تَقْذِفُهُ فِي أَفْوَاهِهِمْ... وَايْمُ اللهِ، لَئِنْ رِمْتَ مَا يُرِيدُونَكَ عَلَيْهِ، لَا يَكُونُ الأُولَى عِنْدَ الأُخْرَى إِلَّا كَلَقْعَةٍ بِبَعْرَةٍ"
الأحنف بن قيس التَّميميّ
كتاب جُمَل من أنساب الأشراف: أحمد بن يحيى البَلاذُري (ت، 279هـ)، ج(2- 13)، حققه وقدم له: سهيل زكّار، رياض زركلي، إشراف: مكتب البحوث والدراسات، دار الفكر، بيروت، ط1، (1417هـ/ 1996م)، ‏12/‏322‏-‏ 323‏.

لَقْعَة[مَصْدَر]

ن64هـ=684م

لَقْعَةُ الحَدَقِ: الإِصَابَةُ بِالعَيْنِ.

لَمْ يُبْقِيَا لِيَ فِي طُولِ اخْتِلَافِهِمَا شَيْئًا يُخَافُ عَلَيْهِ لَقْعَةُ الحَدَقِ
أبو الأسود الدُّؤَليّ
معجم الأدباء إرشاد الأريب إلى معرفة الأديب: ياقوت الحَمَويّ (ت، 626هـ)، تحقيق: إحسان عبّاس، دار الغرب الإِسلاميّ، بيروت، ط1، (1993م)، ‏4/‏1468‏.
ن216هـ=831م

اللَّقْعَةُ : الرَّمْيَةُ.

"لَا تُعْجَبْ بِهِمَا، فَمَا يُسَاوِيَانِ لَقْعَةً بِبَعَرَةٍ، وَأَجْوَدُ الشِّعْرِ مَا صُدِقَ فِيهِ"
الأصمعيّ
الموشَّح مآخذ العلماء على الشعراء في عدة أنواع من صناعة الشعر: أبو عُبيد الله المَرْزُبانيّ (ت، 384هـ)، تحقيق: علي محمّد البجاوي، نهضة مصر، القاهرة، د.ت. ‏281‏.
ن375هـ=985م

لَقْعَةُ اللِّسَانِ وَنَحْوِهِ: الدَّهَاءُ فِي رَمْيِ الكَلَامِ وَالمُغَالَبَةُ بِهِ.

"وَكَانَ مَلِيًّا بِهَذَا النَّمَطِ، وَيُفْرِغُ فِي قَالَبِهِ، وَلَكِنْ لَمْ يَكُنْ لَهُ مِنْهُ إِلَّا لَقْعَةُ اللِّسَانِ وَصَدَى الصَّوْتِ وَتَقْطِيعُ اللَّفْظِ، فَأَمَّا التَّحَلِّي وَالعَمَلُ فَكَانَ مِنْهُمَا عَلَى بُعْدٍ"
أبو حيّان التَّوحيديّ
أخلاق الوزيرين "مثالب الوزيرين الصاحب ابن عباد وابن العميد": أبو حيّان التَّوحيديّ (ت، 414هـ)، حققه وعلق حواشيه: محمّد بن تاويت الطنجي، دار صادر، بيروت، (1412هـ/ 1992م)، ‏390‏.

لُقَّاعَة[صِيغَةُ مُبَالَغَة]

ن110هـ=728م

اللُّقَّاعَةُ مِنَ النَّاسِ: الكَثِيرُ الكَلَامِ، الَّذِي يَرْمِي بِهِ فِي دَهَاءٍ.

فَبَاتَ يُمَنِّينَا الرَّبِيعَ وَصَوْبَهُ وَسَطَّرَ، مِنْ لُقَّاعَةٍ ، وَهْوَ كَاذِبُ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏747‏.
ن206هـ=821م

اللُّقَّاعَةُ : الهَمَّازُ، الَّذِي يُلَقِّبُ النَّاسَ.

لَقَدْ لَاعَ مِمَّا كَانَ بَيْنِي وَبَيْنَهُ وحَدَّثَ عَنْ لُقَّاعَةٍ ، وَهْوَ كَاذِبُ
أبو جُهيمة الذُّهْليّ
لسان العرب: ابن منظور (ت، 711هـ)، دار صادر، بيروت، ط1، (د.ت)، ‏8/‏322‏.

لَقَّاع[اسْمُ جِنْس]

ن140هـ=757م

اللَّقَّاعُ : الذُّبَابُ الأَخْضَرُ الَّذِي يَلْسَعُ النَّاسَ.

كَأَنَّ تَجَاوُبَ اللَّقَّاعِ فِيهَا وَعَنْتَرِهِ وَأَهْمُجِهِ، رِعَالُ
شُبيل بن عَزْرة الضُّبَعيّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏1/‏128‏.

تَلَقَّعَ[مُتَعَدٍّ بِالحَرْف]

ن175هـ=791م

تَلَقَّعَ بِالكَلَامِ: رَمَاهُ فِي دَهَاءٍ؛ يُغَالِبُ بِهِ.

"وَاللُّقَّاعَةُ، عَلَى بِنَاءِ شُدَّاخَةٍ: الرَّجُلُ الدَّاهِيَةُ الَّذِي يَتَلَقَّعُ بِالكَلَامِ، يَرْمِي بِهِ رَمْيًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏167‏.

الْتُقِعَ[مُلَازِمٌ لِلْبِنَاءِ لِلْمَجْهُول]

ن200هـ=815م

الْتُقِعَ لَوْنُهُ: تَغَيَّرَ وَحَالَ عَنْ حَالِهِ.

"وَيُقَالُ: امْتُقِعَ لَوْنُهُ، وَانْتُقِعَ، وَالْتُقِعَ "
أبو مِسْحَل الأعرابيّ
كتاب النّوادر: أبو مِسْحَل الأعرابيّ (ت، 230هـ)، عني بتحقيقه: عزة حسن، مطبوعات مجمع اللغة العربية، دمشق، (1380هـ/ 1961م)، ‏78‏.

مِلْقَع[صِيغَةُ مُبَالَغَة]

ن209هـ=824م

المِلْقَعُ مِنَ الأَشْخَاصِ: الشَّدِيدُ الفُحْشِ فِي القَوْلِ.

وَإِنْ تَكَلَّمْتِ فَكُونِي مِلْقَعَهْ
من إنشاد أبي عُبيدة
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏1/‏248‏.

لَقْع[مَصْدَر]

ن215هـ=830م

اللَّقْعُ : الرَّمْيُ بِبَعَرَةٍ أَوْ حَصَاةٍ.

"وَلَا يَكُونُ اللَّقْعُ فِي غَيْرِ البَعْرَةِ مِمَّا يُرْمَى بِهِ"
أبو زيد الأنصاريّ
كتاب السلاح: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حاتم الضامن، مؤسسة الرسالة، بيروت، ط2، (1405هـ/ 1985م)، ‏37‏.
ن646هـ=1248م

اللَّقْعُ : الإِصَابَةُ بِالعَيْنِ.

وَعَوِّدْنِي التَّعَمُّدَ بِاعْتِنَاءٍ وَعَوِّذْ عَهْدَنَا عَنْ لَقْعِ عَيْنِ
ابن الجنّان الأنصاريّ الأندلسيّ
ديوان ابن الجنّان الأنصاريّ الأندلسيّ: ابن الجنّان الأنصاريّ الأندلسيّ (ت، 646هـ)، جمع وتحقيق ودراسة: مُنجد مُصطفى بهجت، جامعة الموصل، الموصل، (1410ه/ 1990م)، ‏163‏.

تِلِقَّاعَة[صِفَةٌ مُشَبَّهَة]

ن328هـ=940م

التِّلِقَّاعَةُ مِنَ الأَشْخَاصِ: الكَثِيرُ الكَلَامِ، الَّذِي يَرْمِي بِهِ فِي دَهَاءٍ.

"إِنَّ فُلَانًا لَلُقَّاعَةٌ وَتِلِقَّاعَةٌ "
أبو بكر ابن الأنباريّ
المذكّر والمؤنّث: أبو بكر ابن الأنباريّ (ت، 328هـ)، تحقيق: محمّد عبد الخالق عضيمة، راجعه وصنع فهارسه: رمضان عبد التوّاب، وزارة الأوقاف، القاهرة، (1401هـ/ 1981م- 1419هـ/ 1999م)، ‏2/‏163‏.
ن395هـ=1005م

التِّلِقَّاعَةُ مِنَ النَّاسِ: الدَّاهِيَةُ، الَّذِي يَرْمِي بِالكَلَامِ مِنْ أَقْصَى حَلْقِهِ.

"وَاللُّقَّاعَةُ: الدَّاهِيَةُ الَّتِي يَتَلَقَّعُ بالْكَلَامِ، يَرْمِي بِهِ مِنْ أَقْصَى حَلْقِهِ، وَكَذَا التِّلِقَّاعَةُ "
ابن فارس
معجم مقاييس اللُّغة: ابن فارس(ت، 395هـ)، تحقيق وضبط: عبد السلام هارون، دار الفكر، القاهرة، (1399هـ/ 1979م)، ‏5/‏263‏.
ن542هـ=1147م

التِّلِقَّاعَةُ مِنَ الأَشْخَاصِ: الَّذِي يُصِيبُ الآخَرِينَ بِالعَيْنِ.

"وَكَانَ عَلِيُّ بْنُ حَمُودٍ تِلِقَّاعَةً ، شَدِيدَ الإِصَابَةِ بِعَيْنِهِ"
ابن بسّام الشّنْتَرِيني
الذخيرة في محاسن أهل الجزيرة: ابن بسّام الشّنْتَرِيني (ت، 542هـ)، تحقيق: إحسان عبّاس، دار الثَّقافة، بيروت، ط1، (1417ه/ 1997م)، ‏1/‏98‏.

مَلْقُوع[اسْمُ مَفْعُول]

"وَرُوِيَ عَنْهُ @  أَنَّهُ أَمَرَ رَجُلًا لُقِعَ صَعَدًا أَنْ يُسْقَى لَهُ... وَإِنَّمَا أَمَرَ الرَّجُلَ أَنْ يُغْسَلَ لَهُ؛ لِيَزُولَ عَنِ المَلْقُوعِ مَا أَثَّرَتْ فِيهِ العَيْنُ بِمَا بَدَرَ مِنْهَا، وَأَنْ يَزُولَ ذَلِكَ بِمَا يَبْدُرُ مِنْهُ"

مِلْقَاع[صِيغَةُ مُبَالَغَة]

1270هـ=1853م

المِلْقَاعُ مِنَ الأَشْخَاصِ: الكَثِيرُ الفُحْشِ فِي الكَلَامِ.

"وَهَذِهِ سَيِّدَتِي الزَّنَمْرَدَةُ، العَنْجَرِدُ، الدَّلَعُوسُ، الإِلْقَةُ، السِّلْقَةُ، المِلْقَاعُ "
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏505‏.

لُقَعَة[صِفَةٌ مُشَبَّهَة]

ن1270هـ=1853م

اللُّقَعَةُ : الَّذِي يَرْمِي الكَلَامَ بِدَهَاءٍ، وَيُغَالِبُ بِهِ.

"قَالَتْ: أَيُّ لُقَعَةٍ أَنْتَ! مَا عَنَيْتُ إِلَّا المَعْنَى"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏589‏.
