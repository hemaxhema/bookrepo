لَسَق[مَصْدَر]

ن145هـ=762م

لَسَقُ الشَّيْءِ بِالشَّيْءِ: لُصُوقُهُ بِهِ.

وَبَلَّ بَرْدُ المَاءِ أَعْضَادَ اللَّسَقْ
رؤبة بن العجّاج
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏78‏.

لَسَق[اسْم]

ن203هـ=818م

اللَّسَقُ : لُصُوقُ الرِّئَةِ بِالجَنْبِ مِنَ العَطَشِ.

"إِنَّ اللَّسَقَ يَسْحَرُ أَلْبَانَ الغَنَمِ، وَهُوَ أَنْ يَنْزِلَ اللَّبَنُ قَبْلَ الوِلَادِ"
النَّضْر بن شُميل المازنيّ التّميميّ
لسان العرب: ابن منظور (ت، 711هـ)، دار صادر، بيروت، ط1، (د.ت)، ‏4/‏350‏.

لُسُوق[مَصْدَر]

ن175هـ=791م

لُسُوقُ الشَّيْءِ بِالشَّيْءِ: لُصُوقُهُ بِهِ.

"اللَّسَقُ: إِذَا الْتَزَقَتِ الرِّئَةُ بِالجَنْبِ مِنْ شِدَّةِ العَطَشِ... وَاللُّسُوقُ كَاللُّزُوقِ فِي كُلِّ التَّصْرِيفِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏77‏-‏ 78‏.

لَسِقَ[مُتَعَدٍّ بِالحَرْف]

ن175هـ=791م

لَسِقَ الشَّيْءُ بِالشَّيْءِ: لَصِقَ بِهِ.

"اللَّسَقُ: إِذَا الْتَزَقَتِ الرِّئَةُ بِالجَنْبِ مِنْ شِدَّةِ العَطَشِ قِيلَ: لَسِقَتْ "
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏77‏.

مَلْسُوق[اسْمُ مَفْعُول]

ن550هـ=1155م

المَلْسُوقُ بِالشَّيْءِ: المَشْدُودُ إِلَيْهِ، المَوْصُولُ بِهِ.

"فَتَبْقَى الثُّرَيَّاتُ مُعَلَّقَةً بَيْنَ السَّمَاءِ وَالأَرْضِ، مَلْسُوقَةً إِلَى بَعْضِ تِلْكَ الأَحْجَارِ، دُونَ لِسَاقٍ وَلَا مِسْمَارٍ، يَجْذِبُهَا حَجَرُ المَغْنَطِيسِ"
الزُّهْري الغرناطيّ
كتاب الجغرافية: الزُّهْري الغرناطيّ (ت، 550هـ)، اعتنى بتحقيقه: محمد حاج صادق، مكتبة الثقافة الدينية- المركز الإسلامي للطباعة، القاهرة، (د.ت)، ‏74‏.

لِسَاق[اسْم]

ن550هـ=1155م

اللِّسَاقُ : مَادَّةٌ كَالصَّمْغِ وَنَحْوِهِ، تُتَّخَذُ فِي لَحْمِ الأَشْيَاءِ.

"فَتَبْقَى الثُّرَيَّاتُ مُعَلَّقَةً بَيْنَ السَّمَاءِ وَالأَرْضِ، مَلْسُوقَةً إِلَى بَعْضِ تِلْكَ الأَحْجَارِ، دُونَ لِسَاقٍ وَلَا مِسْمَارٍ، يَجْذِبُهَا حَجَرُ المَغْنَطِيسِ"
الزُّهْري الغرناطيّ
كتاب الجغرافية: الزُّهْري الغرناطيّ (ت، 550هـ)، اعتنى بتحقيقه: محمد حاج صادق، مكتبة الثقافة الدينية- المركز الإسلامي للطباعة، القاهرة، (د.ت)، ‏74‏.
1438هـ=2016م

اللِّسَاقُ : التُّرَابُ المُبَلَّلُ بِالمَاءِ.

"وَيُوضَعُ القَصَبُ أَوْ جَرِيدُ النَّخْلِ بَعْدَ تَجْرِيدِهِ مِنْ سَعَفِهِ فَوْقَ هَذِهِ الأَعْمِدَةِ، ثُمَّ يُغَطَّى الكُلُّ بِالتُّرَابِ المُبَلَّلِ، وَيُسَمَّى اللِّسَاقَ ، ثُمَّ يُطْمَرُ بِالتُّرَابِ"
عبد الله حمادي الإدريسي
قاعدة المغرب الأقصى قبل فاس سجلماسة ووريثتها تافيلالت تاريخًا وأمجادًا وجهادًا: عبد الله حمادي الإدريسي، دار الكتب العلمية، بيروت، ط1، (2016م)، ‏27‏.

لَاسِق[اسْمُ فَاعِل]

578هـ=1182م

اللَّاسِقُ بِالشَّيْءِ: المُلْتَحِمُ، المُعَلَّقُ بِهِ.

"وَلَقِيَنِي فِي ذَلِكَ اليَوْمِ وَاحِدٌ مِنْهُمْ فِي يَدِهِ سِيخٌ وَفِي يَدِي سَيْفٌ لِي، فَهَجَمَ عَلَيَّ بِالسِّيخِ فَضَرَبْتُهُ فِي وَسَطِ سَاعِدِهِ، وَالسِّيخُ فِي يَدِهِ قَبْضَتُهُ وَنَصْلُهُ لَاسِقٌ بِسَاعِدِهِ"
أسامة بن مُنْقِذ
كتاب الاعتبار: أسامة بن مُنْقِذ (ت، 584هـ)، دقق نصوصها وفصل فقرها وقدم لها وعلق عليها: عبد الكريم الأَشتر، المكتب الإِسلاميّ، بيروت- دمشق- عمان، ط2، (1424ه/ 2003م)، ‏200‏.

الْتَسَقَ[مُتَعَدٍّ بِالحَرْف]

ن726هـ=1325م

الْتَسَقَ بِالشَّيْءِ: الْتَحَمَ بِهِ، وَتَعَلَّقَ.

"فَإِنَّ الجِسْمَ لَوِ الْتَسَقَ بِالعَيْنِ، لَمْ يُمْكِنْ رُؤْيَتُهُ"
ابن المطَهَّر الحِلِّيّ
نهج الحق وكشف الصدق: الحسن بن يوسف المطهر الحلي، تعليق: عين الله الحسني الأرموي، تقديم: رضا الصدر، دار الهجرة، قم، إيران، ط1، (1986م)، ‏40‏.

مُتَلَاسِق[اسْمُ فَاعِل]

1074هـ=1663م

المُتَلَاسِقُ مَعَ غَيْرِهِ: المُتَّصِلُ بِهِ، المُتَلَاحِمُ مَعَهُ.

"جِئْنَا ظُهْرًا لِقَصْرِ المُخَيْلِفِ وَوَجَدْنَا فِيهِ مِيَاهًا كَثِيرَةً فِي مَآجِلَ، وَوَجَدْنَا بِهِ جَابِيَتَيْنِ مُتَلَاسِقَتَيْنِ مَبْنِيَتَيْنِ بِالحِجَارَةِ المَرْصُوصَةِ بِنَاءً مُتْقَنًا"
أبو سالم العياشي
الرّحلة العياشية: أبو سالم العياشي (ت، 1090هـ)، حققها وقدم لها: سعيد الفاضلي، سليمان القرشي، دار السُّويدي، أبو ظبي، ط1، (1426ه/ 2006م)، ‏1/‏204‏.

لِسْق[اسْم]

ن1117هـ=1705م

لِسْقُ الشَّيْءِ: جَنْبُهُ.

"وَهَكَذَا وَرَدَ أَيْضًا فِي النُّبْذَةِ التَّارِيخِيَّةِ فِي المُقَاطَعَةِ الكِسْرَوَانِيَّةِ نَقْلًا عَنِ الدُّوَيْهِيِّ، قَالَ: وَبِلِسْقِهَا (أَيْ بِجَانِبِ الكَنِيسَةِ) مِنَ الجِهَةِ الشَّرْقِيَّةِ عَمَّرْنَا حَارَةً لِسُكْنَى البَطَارِكَةِ إِذَا تَوَجَّهُوا لِذَاكَ الجَانِبِ"
اسطفان الدويهيّ
الأديار القديمة في كسروان: ‏إبراهيم حرفوش،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين، بيروت، س5، ع15، (27/ 04/ 1320هـ)، ‏696‏.

تَلَاسُق[مَصْدَر]

ن1323هـ=1904م

التَّلَاسُقُ بَيْنَ الشَّخْصَيْنِ: الِاتِّصَالُ بَيْنَهُمَا وَالتَّلَاحُمُ.

"فَنَظَرْتُ فَأَلْفَيْتُهُمْ يَتَزَاحَمُونَ وَيَتَرَاكَمُونَ عَلَى مَرْكَبَاتِ القِطَارِ... وَمَا تَخَيَّلْتَهُمْ، بِعِلْمِ اللهِ فِي تَزَاحُمِهِمْ وَتَرَاكُمِهِمْ وَتَلَاسُقِهِمْ وَتَلَاحُمِهِمْ، إِلَّا أَنَّهُمْ فِي إِحْدَى الحَالَتَيْنِ الهَائِلَتَيْنِ؛ حَالَتَهِمْ فِي مَيْدَانِ الطِّعَانِ عِنْدَ إِقْدَامِهِمْ أَوْ حَالَةِ أَعْدَائِهِمْ فِي انْهِزَامِهِمْ"
إبراهيم المويلحي
مرآة العالم: إبراهيم المويلحي (ت، 1323هـ)، ضمن: الأعمال الكاملة (إبراهيم المويلحي)، تقديم وتحرير: روجر آلن، المجلس الأعلى للثقافة، القاهرة، ط1، (1428ه/ 2007م)، (161- 202)، ‏176‏.

مُلَسَّق[اسْمُ مَفْعُول]

ن1329هـ=1910م

المُلَسَّقُ بِغَيْرِهِ: المُضَافُ إِلَيْهِ، المُخَالِطُ لَهُ ادِّعَاءً.

أَمِثْلَ ابْنِ هِنْدٍ رَافِلٌ فِي حَرِيرِهِ عَلَى دَسْتِهِ، وَهْوَ الطَّلِيقُ المُلَسَّقُ ؟
آماد كاظم البرواري
ديوان الشيخ يعقوب الحاج جعفر النجفي الحلّي: يعقوب الحاج جعفر التبريزي (ت، 1329هـ)، جمعه وعلق عليه: محمد علي اليعقوبي، جمعية الرابطة الأدبية، النجف، ط1، (1962م)، ‏134‏.

أَلْسَق[صِفَةٌ مُشَبَّهَة]

1373هـ=1953م

الأَلْسَقُ مِنَ الأَشْخَاصِ: الأَكْثَرُ اتِّصَالًابِغَيْرِهِ، وَالْتِحَامًا بِهِ.

"وَلِكَيْ يَتَّصِلَ جُمْهُورُ القُرَّاءِ بَعْضُهُمْ بِبَعْضٍ فِي الأَفْكَارِ وَالنَّظَرِيَاتِ وَالأَبْحَاثِ وَالمُعْتَقَدَاتِ أَوَّلًا، وَأَنْ تُعْرَفَ بِلَادُنَا فِي الأَقْطَارِ الشَّرْقِيَّةِ خَاصَّةً وَبِلَادِ العَالَمِ عَامَّةً، لِأَنِّي أَعْلَمُ عِلْمَ اليَقِينِ أَنَّ أَلْسَقَ الأُمَمِ بِنَا تَكَادُ تَجْهَلُنَا كُلَّ الجَهْلِ"
عبد المجيد عابدين
تاريخ الثقافة العربية في السودان منذ نشأتها إلى العصر الحديث: عبد المجيد عابدين، دار الثقافة، القاهرة، (1967)، ‏152‏.

إِلْسَاق[مَصْدَر]

1427هـ=2005م

إِلْسَاقُ الشَّيْءِ بِغَيْرِهِ: جَعْلُهُ مُلْتَحِمًا بِهِ، مُتَّصِلًا.

"تُصَارِعُ مِنْ أَجْلِ البَقَاءِ فِي ظِلِّ هَذِهِ الهَجْمَةِ الشَّرِسَةِ عَلَى العُرُوبَةِ، وَمُحَاوَلَةِ إِلْسَاقِ التُّهَمِ لَهَا بِالتَّخَلُّفِ وَعَدَمِ القُدْرَةِ عَلَى التَّكَيُّفِ"
يعقوب حسين نشوان
التربية في الوطن العربي في مشارف القرن الحادي والعشرين: يعقوب حسين نشوان، دار الفرقان للنشر والتوزيع، عمان، ط1، (2005م)، ‏49‏.

أَلْسَقَ[مُتَعَدٍّ]

1428هـ=2006م

أَلْسَقَ الشَّيْءَ بِغَيْرِهِ: لَحَمَهُ بِهِ، وَضَمَّهُ إِلَيْهِ.

"لَقَدْ عَرَفَ المَغْرِبُ، قَبْلَ الِاسْتِقْلَالِ وَبَعْدَهُ، جَمَاعَةً مِنَ الَّذِينَ لَا خَلَاقَ لَهُمْ، تَجَمَّعَتْ فِيهِمْ كُلُّ الرَّذَائِلِ... حَتَّى أَصْبَحَ الشَّعْبُ مَنْعُوتًا فِي بُلْدَانِ العَالَمِ بِمَا أَلْسَقَ بِهِ مِنَ الفَظَائِعِ وَالفَضَائِحِ الَّتِي كَتَبَ عَنْهَا الآخَرُونَ فِي الشَّرْقِ وَالغَرْبِ"
عبد الكريم فيلالي
التاريخ السياسي للمغرب العربي الكبير: عبد الكريم الفيلالي، شركة ناس للطباعة، القاهرة، ط1، (2006م)، ‏12/‏361‏.

مُلْتَسِق[اسْمُ فَاعِل]

1436هـ=2014م

المُلْتَسِقُ بِالشَّيْءِ: المُتَّصِلُ بِهِ، المُلْتَحِمُ.

"كَانَ هَذَا التَّرْكِيزُ عَلَى اللَّوْنِ الأَحْمَرِ نَتِيجَةً لِمَا ارْتَبَطَ بِهِ مِنْ إِيحَاءَاتٍ عَنِيفَةٍ كَوْنَهُ قَدْ دَلَّ عَلَى مَشْهَدِ المَوْتِ فَهُوَ مُلْتَسِقٌ وَمُعَبِّرٌ عَنِ الدَّمِ فِي المَعَارِكِ"
صالح ويس
الصورة اللونية في الشعر الأندلسي: صالح ويس، دار مجدلاوي، عمان، ط1، (2014م)، ‏181‏.

الْتِسَاق[مَصْدَر]

1440هـ=2018م

الِالْتِسَاقُ بِالشَّيْءِ: الِاتِّصَالُ بِهِ، وَالِالْتِحَامُ.

"وَخِدَاعُ وَإِغْوَاءُ الحَيَوَانِ مُهِمٌّ لِيَتِمَّ الِالْتِسَاقُ مَعَ حَيَوَانٍ آخَرَ، وَهَذَا مَا يَجْعَلُ النَّشَاطَ الجِنْسِيَّ البَشَرِيَّ عُنْصُرًا مُتَخَيَّلًا قَوِيًّا، فَالسُّلُوكُ الجِنْسِيُّ، بِوَجْهٍ خَاصٍّ، عُرْضَةٌ لِلْإِغْوَاءِ"
محمد أحمد محمود خطاب
قاموس لاكان التمهيدي في التحليل النفسي: ديلان إيفانز، ترجمة: محمد أحمد محمود خطاب، مكتبة الأنجلو المصرية، القاهرة، (2018م)، ‏240‏.
