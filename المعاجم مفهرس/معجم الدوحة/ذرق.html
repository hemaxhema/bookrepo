أَذْرَقَ[لَازِم]

قبل 13ق.هـ=609م

أَذْرَقَ الطَّائِرُ وَنَحْوُهُ: رَمَى خُرْأَهُ.

أَلَا تِلْكَ الثَّعَالِبُ قَدْ تَوَالَتْ عَلَيَّ، وَحَالَفَتْ عُرْجًا ضِبَاعَا
لِتَأْكُلَنِي، ‌فَمَرَّ ‌لَهُنَّ ‌لَحْمِي فَأَذْرَقَ مِنْ حِذَارِي، أَوْ أَتَاعَا
عَديّ بن زيد العِباديّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏15/‏197‏.
132هـ=749م

أَذْرَقَتِ الأَرْضُ: أَنْبَتَتِ الذُّرَقَ.

وَاصْفَرَّ مِنْ حُجْرَانِهِ مَا أَذْرَقَا وَحَتَّ فِيمَا حَتَّ، إِذْ تَحَرَّقَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏1/‏158‏.

أَذْرَقَ[مُتَعَدٍّ]

ن90هـ=709م

أَذْرَقَهُ : جَعَلَهُ يُلْقِي خُرْأَهُ.

إِنَّ لَنَا قَرْمًا، إِذَا مَا صَالَا هَدَّ الصُّوَى، وَأَذْرَقَ الفِحَالَا
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، تحقيق: عبد الحفيظ السطلي، مكتبة أطلس- المطبعة التعاونية، دمشق، (1971م)، ‏1/‏271‏.

ذَارِق[اسْمُ فَاعِل]

قبل 13ق.هـ=609م

الذَّارِقُ : الَّذِي يُلْقِي خُرْأَهُ.

لَا بُدَّ أَنْ يُغْمَزَ مِنْكَ الفَائِقُ غَمْزًا، تَرَى أَنَّكَ مِنْهُ ذَارِقُ
لَبيد بن ربيعة العامريّ
ديوان لَبيد بن ربيعة: اعتنى به: حمدو طمّاس، دار المعرفة، بيروت، ط1، (1425هـ/ 2004م)، ‏63‏.

ذَرْق[اسْم]

3هـ=624م

ذَرْقُ الطَّائِرِ وَنَحْوِهِ: خُرْؤُهُ.

وَأَسَرْنَا مِنْكُمُ أَعَدَادَهُمْ فَانْصَرَفْتُمْ مِثْلَ إِفْلَاتِ الحَجَلْ
يَخْرُجُ الأَكْدَرُ مِنْ أَسْتَاهِكُمْ مِثْلَ ذَرْقِ النِّيبِ يَأْكُلْنَ العَصَلْ
حسّان بن ثابت الخزرجيّ الأنصاريّ
ديوان حسان بن ثابت: حقّقه وعلّق عليه: وليد عرفات، دار صادر، بيروت، (2006م)، ‏1/‏67‏.

ذَرَقَ[مُتَعَدٍّ بِالحَرْف]

23هـ=644م

ذَرَقَ عَلَيْهِ: أَلْقَى خُرْأَهُ عَلَيْهِ.

"مَا هَجَاهُ، وَلَكِنْ ذَرَقَ عَلَيْهِ"
حسّان بن ثابت الخزرجيّ الأنصاريّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏143‏.
ن175هـ=791م

ذَرَقَ بِخُرْئِهِ: وَضَعَهُ مُسْتَوِيًا غَيْرَ مُتَجَمِّعٍ.

" وَذَرَقَ بِهِ إِذَا وَضَعَهُ بِخِرَاءَةٍ مُسْتَوِيًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏1/‏303‏.

ذَرَقَ[لَازِم]

ن26هـ=647م

ذَرَقَ : رَمَى خُرْأَهُ.

كَالكَلْبِ، لَا يَسْأَمُ الكَلْبُ الهَرِيرَ، وَلَوْ لَاقَيْتَ بِالكَلْبِ لَيْثًا مُخْدِرًا، ذَرَقَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏239‏.

ذُرَق[اسْمُ جِنْس]

ن26هـ=647م

الذُّرَقُ : نَبَاتٌ بَرِّيٌّ، شَبِيهٌ بِالكُرَّاثِ الجَبَلِيِّ، لَهُ حَبٌّ أَغْبَرُ حُلْوٌ، يُؤْكَلُ رَطْباً، تُحِبُّهُ المَاشِيَةُ.

فَأَنْبَتَ الفَغْوَ وَالرَّيْحَانَ وَابِلُهُ وَالأَيْهُقَانَ مَعَ المُكْنَانِ وَالذُّرَقَا
كعب بن زهير بن أبي سُلْمى المُزَنيّ
شرح ديوان كعب بن زهير: صنعة: أبي سعيد السُّكَّريّ(ت، 275هـ)، تحقيق: عباس عبد القادر، دار الكتب والوثائق القومية، القاهرة، ط3، (1423هـ/ 2002م)، ‏234‏.

ذِرَاقَة[مَصْدَر]

ن313هـ=925م

الذِّرَاقَةُ : إِخْرَاجُ فَضْلَاتِ البَطْنِ.

"أَرَى أَنَّ الذِّرَاقَةَ هَهُنَا أَجْوَدُ"
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏4/‏146‏.

أَذَارَقِي[اسْم]

ن313هـ=925م

أَذَارَقِي : نَوْعٌ مِنْ زَبَدِ البَحْرِ، حَارٌّ، يُعَالِجُ المِزَاجَ الرَّدِيءَ.

" أَذَارَقِي ... يَصْلُحُ لِقَلْعِ الجَرَبِ المُتَقَرِّحِ وَالكَلَفِ وَالقَوَابِي وَالبُثُورِ اللَّيِّنَةِ وَمَا أَشْبَهَ ذَلِكَ "
أبو بكر الرّازيّ
الحاوي في الطّبّ: أبو بكر الرّازيّ (ت، 313هـ)، اعتنى به: هيثم خليفة طعيمى، دار إحياء التراث العربي، بيروت، ط1، (1422هـ/ 2002م)، ‏6/‏39‏.

مُذَرَّق[اسْمُ مَفْعُول]

ن334هـ=945م

المُذَرَّقُ مِنَ الأَشْيَاءِ: المَلْفُوظُ المُلْقَى.

في خَنَزِ التَّيْسِ، فِي سُهُوكَتِهِ أَسْهَكُ تَيْسٍ بُرِيٍّ، وَأَخْنَزُهُ
ذُو نَغَفٍ، مَا يَنِي مُذَرَّقُهُ عَلَى سِبَالَيْهِ، أَوْ مُلَوَّزُهُ
الصَّنَوبريّ
ديوان الصّنوبريّ أحمد محمّد بن الحسن الضّبيّ: تحقيق: إحسان عبّاس، دار صادر، بيروت، ط2، (1998م)، ‏132‏.

مُذَرَّق[صِفَة]

ن456هـ=1064م

المُذَرَّقُ مِنَ الأَدْوِيَةِ: الَّذِي فَقَدَ فَاعِلِيَتَهُ لِقِدَمٍ، وَغَيْرِهِ.

"وَدَوَاءٌ مُذَرَّقٌ : لَيْسَتْ لَهُ فَاعِلِيَّةٌ، إِمَّا عَنْ قِدَمٍ، وَإِمَّا عَنْ غَلَطٍ فِي تَرْكِيبِهِ، أَوْ فِي مَعْرِفَةِ العِلَّةِ"
عبد الله بن محمّد الأزديّ الصُّحاريّ
الماء: عبد الله بن محمّد الأزديّ الصُّحاريّ (ت، 456هـ)، حققه: هادي حسن حمّودي، وزارة التراث والثقافة، مسقط، ط2، (1436هـ/ 2015م)، ‏2/‏136‏.

مَذْرَق[اسْمُ مَكَان]

ن350هـ=961م

مَذْرَقُ الطَّائِرِ وَنَحْوِهِ: مَخْرَجُ ذَرْقِهِ.

"وَالعَنَقُ، ‌فَإِنَّهُ ‌يُدَسِّمُ ‌جَوْفَهُ ‌وُيُلَيِّنُهُ، وَيُوَسِّعُ مَذْرَقَهُ ، وَالَّذِي لَا مُخَّ فِيهِ، يُخْرِجُ أَمْعَاءَهُ"
بازيار العزيز بالله الفاطميّ
البَيْزَرة: بازيار العزيز بالله الفاطميّ (ت، 350هـ)، نظر فيه وعلق عليه: محمّد كرد علي، مطبوعات المجمع العلمي العربي، دمشق، د.ت. ‏79‏.

الأَذْرَقُ مِنَ الطُّيُورِ وَنَحْوِهَا: الأَكْثَرُ رَمْيًا لِخُرْئِهِ.

" أَذْرَقُ مِنْ حُبَارَى، وَأَسْلَحُ مِنْ حُبَارَى"

ذَرْقَة[اسْمُ مَرَّة]

384هـ=994م

الذَّرْقَةُ : المَرَّةُ مِنَ الذَّرْقِ.

"وَمِنْ عَادَةِ الحُبَارَى أَنْ تَذْرِقَ عَلَى الجَارِحِ الَّذِي يَصِيدُهَا لِتَجْرَحَ جَنَاحَهُ ... فَذَرَقَتِ الحُبَارَى إِلَى فَوْقَ حَتَّى صَعَدَتْ ذَرْقَتُهَا "
فارس بن مشغف
حياة الحيوان الكبرى: كمال الدين محمد بن موسى الدَّميري (ت، 808هـ)، عني بتحقيقه: إبراهيم صالح، دار البشائر، دمشق، ط1، (1426ه/ 2005م)، ‏1/‏508‏.

تَذْرِقَة[مَصْدَر]

ن646هـ=1248م

تَذْرِقَةُ الطَّعَامِ وَنَحْوِهِ: تَلْيِينُهُ، وَتَسْهِيلُ امْتِصَاصِهِ، أَوْ إِخْرَاجِهِ ذُرَاقًا وَنَحْوَهُ.

"المَاءُ جَوْهَرٌ يُعِينُ فِي تَسْهِيلِ الغِذَاءِ وَتَذْرِقَتِهِ نَافِذًا بِهِ إِلَى العُرُوقِ وَإِلَى المَخَارِجِ"
ابن البَيْطار
الجامع لمفردات الأدوية والأغذية: ابن البَيْطار (ت، 646هـ)، تحقيق: محمود مهدي بدوي، لجنة المراجعة والإشراف: حسيني الجندي وآخرين، دار الكتب والوثائق القوميّة، القاهرة، (2015م)، ‏4/‏268‏.

"وَإِذَا كَانَ القَصْدُ اسْتِعمَالَ حَقْنٍ يُدْخِلُ طَرَفَ أُنْبُوبَةِ الذَّارِقَةِ بِاسْتِحْكَامٍ فِي طَرَفِ القَثَاطِيرِ فَإِنْ وَقَفَ السَّائِلُ المَحْقُونُ بِهِ في البُوقِ، تُرْفَعُ الذَّارِقَةُ"

كتاب غرر النجاح في أعمال الجراح: محمد علي باشا البقلي (ت، 1293هـ)، مطبعة بولاق، القاهرة، (1263ه)، ‏2/‏92‏.

ذُرَاق[اسْم]

1437هـ=2015م

ذُرَاقُ الطّائِرِ وَنَحْوِهِ: خُرْؤُهُ.

"إِذْ بِمُجَرَّدِ اسْتِيقَاظِ سَانْتِيَاغُو مِنَ النَّوْمِ شَعَرَ كَمَا لَوْ أَنَّهُ مُلَوَّثٌ بِكَامِلِهِ بِذُرَاقِ العَصَافِيرِ"
مصطفى الحمداوي
غابرييل غارسيا ماركيز في دائرة الواقعية السحرية: مصطفى الحمداوي، دار الأدهم للنشر والتوزيع، القاهرة، (2015م)، ‏100‏.

مِذْرَق[اسْمُ آلَة]

1439هـ=2017م

المِذْرَقُ : آلَةٌ لِقَلْعِ الحِجَارَةِ وَنَحْوِهَا، وَإِطْلَاقِهَا.

"لَمْ أَسْتَطِعْ ـ بِسَبَبِ بَصَرِي الضَّعِيفِ ـ إِصَابَةَ أَيَّةِ جَوْلَبَةٍ خِلَالَ مُبَارَيَاتِ الصَّيْدِ الَّتِي خُضْنَاهَا أَنَا وَأَوْلَادُ عَمِّي، مُسْتَخْدِمِينَ مِذْرَقًا لِإِطْلَاقِ الحِجَارَةِ"
جميل جمال السروري
الملكة المغدورة: حبيب عبد الرب سروري، دار الساقي ، ط1، (2017م)، ‏79‏.
