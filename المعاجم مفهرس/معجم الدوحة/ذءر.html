ذَئِرَ[لَازِم]

قبل 13ق.هـ=609م

ذَئِرَ الشَّخْصُ: اغْتَاظَ وَغَضِبَ.

وَلَقَدْ أَتَانِي عَنْ تَمِيمٍ أَنَّهُمْ ذَئِرُوا لِقَتْلَى عَامِرٍ، وَتَغَضَّبُوا
عَبيد بن الأبرص الأسديّ
ديوان عَبيد بن الأبرص: تحقيق وشرح: حسين نصار، شركة مكتبة ومطبعة مصطفى البابي الحلبي وأولاده، مصر، ط1، (1377هـ/ 1957م)، ‏6‏.
11هـ=632م

ذَئِرَتِ المَرْأَةُ عَلَى زَوْجِهَا: نَشَزَتْ وَتَغَيَّرَ خُلُقُهَا.

"قَدْ ذَئِرْنَ عَلَى أَزْوَاجِهِنَّ"
عمر بن الخطّاب
مُسند الدّارمي المعروف بـ(سنن الدارمي): الدّارمي (ت، 255هـ)، تحقيق: حسين سليم أسد الداراني، دار المغني، الرياض، ط1، (1421هـ/ 2000م)، ‏3/‏1424،،‏(رقم الحديث: 2265).

ذَئِرَ[مُتَعَدٍّ بِالحَرْف]

15هـ=636م

ذَئِرَ عَلَى الشَّخْصِ: تَجَرَّأَ عَلَيْهِ.

"هَلْ لَكَ فِي الشَّامِ؟ فَإِنَّ المُسْلِمِينَ قَدْ نُزِفُوا بِهِ، وَإِنَّ العَدُوَّ قَدْ ذَئِرُوا عَلَيْهِمْ"
عمر بن الخطّاب
كتاب الطّبقات الكبير: ابن سعد البَغداديّ (ت، 230هـ)، تحقيق: علي محمّد عمر، مكتبة الخانجي، القاهرة، ط1، (1421هـ/ 2001م)، ‏3/‏434‏.
15هـ=636م

ذَئِرَ بِالشَّيْءِ: أُولِعَ بِهِ وَاعْتَادَهُ.

"هَلْ لَكَ فِي الشَّامِ؟ فَإِنَّ المُسْلِمِينَ قَدْ رَقُّوا بِهَا، وَإِنَّ العَدُوَّ قَدْ ذَئِرُوا بِهِمْ"
عمر بن الخطّاب
غريب الحديث: إبراهيم بن إسحاق الحربيّ (ت، 285هـ)، تحقيق ودراسة: سليمان بن إبراهيم بن محمّد العايد، جامعة أم القرى، مكة المكرمة، دار المدني، جدة، ط1، (1405هـ/ 1985م)، ‏1/‏250‏.

أَذْأَرَ[مُتَعَدٍّ]

ن9هـ=630م

أَذْأَرَ الشَّخْصَ: جَرَّأَهُ وَحَرَّضَهُ.

"سُوءُ حَمْلِ الفَاقَةِ يُحْرِضُ الحَسَبَ، وَيُذْئِرُ العَدُوَّ"
أكثم بن صَيْفيّ الأُسَيِّديّ التّميميّ
المحيط في اللُّغة: الصّاحب ابن عَبّاد (ت، 385هـ)، تحقيق: محمّد حسن آل ياسين، عالم الكتب، بيروت، ط1، (1414هـ/ 1994م)، ‏10/‏08‏.
ن108هـ=726م

أَذْأَرَ الشَّخْصَ: أَغَاظَهُ وَأَغْضَبَهُ.

"وَكَرِهَ رَسُولُ اللهِ  @  أَنْ يَبْلُغَ قَوْمَهُ عَنْهُ، فَيُذْئِرَهُمْ ذَلِكَ عَلَيْهِ"
محمّد بن كعب القُرَظيّ
السّيرة النّبويّة: ابن هشام المَعافريّ الحِمْيريّ (ت، 213هـ)، حققها وضبطها وشرحها ووضع فهارسها: مصطفى السّقا وآخرون، شركة مكتبة ومطبعة البابي الحلبي وأولاده، القاهرة، ط2، (1375هـ/ 1955م)، ‏1/‏419‏.
ن175هـ=791م

أَذْأَرَهُ إِلَى الشَّيْءِ: أَلْجَأَهُ إِلَيْهِ.

" وَأَذْأَرْتُهُ : أَلْجَأتُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏197‏.
ن175هـ=791م

أَذْأَرَهُ بِالشَّيْءِ: أَوْلَعَهُ بِهِ وَحَرَّشَهُ.

" وَأَذْأَرْتُهُ بِالشَّيْءِ: أَوْلَعْتُهُ وَحَرَّشْتُهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏197‏.

ذَاءَرَ[لَازِم]

23هـ=644م

ذَاءَرَتِ النَّاقَةُ وَنَحْوُهَا: (ذَارَتْ: ضَرُورَةٌ): نَفَرَتْ مِنْ وَلَدِهَا، وَلَمْ تَشَمَّهُ عِنْدَ وِلَادَتِهِ.

وَأَكْرَمْتُ نَفْسِي اليَوْمَ مِنْ سُوءِ طِعْمَةٍ وَيَقْنَى الحَيَاءَ المَرْءُ، وَالرُّمْحُ شَاجِرُهْ
وَكُنْتُ كَذَاتِ البَعْلِ ذَارَتْ بِأَنْفِهَا فَمِنْ ذَاكَ تَبْغِي غَيْرَهُ، وَتُهَاجِرُهْ
الحطيئة
ديوان الحُطَيْئة: رواية وشرح: ابن السِّكّيت (ت، 244هـ)، تحقيق: نعمان محمّد أمين طه، مكتبة الخانجي- مطبعة المدني، القاهرة، ط1، (1407هـ/ 1987م)، ‏26‏.

ذِئَار[اسْم]

ن27هـ=648م

الذِّئَارُ : بَعَرٌ رَطْبٌ مُخْتَلِطٌ بِتُرَابٍ، يُوضَعُ عَلَى ضُرُوعِ النَّاقَةِ حَتَّى لَا تُرْضَعَ.

غَدَتْ، وَهْيَ مَحْشُوكَةٌ حَافِلٌ، فَرَاحَ الذِّئَارُ عَلَيْهَا صَحِيحَا
أبو ذؤيب الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏3/‏1308‏.

ذَائِر[اسْمُ فَاعِل]

98هـ=717م

الذَّائِرُ : الغَاضِبُ المُغْتَاظُ.

دَعَانِي إِلَى جُرْجَانَ، وَالرَّيُّ دُونَهُ، لِآتِيَهُ، إِنِّي إِذَنْ لَزَؤُورُ
لِآتِيَ مِنْ آلِ المُهَلَّبِ ذَائِرًا بِأَعْرَاضِهِمْ، وَالدَّائِرَاتُ تَدُورُ
الفرزدق
طبقات فحول الشّعراء: محمّد بن سلّام الجُمَحيّ (ت، 231هـ)، قرأه وشرحه: محمود محمّد شاكر، دار المدني، جدة، (1974م)، ‏2/‏338‏-‏ 339‏.
125هـ=743م

الذَّائِرُ مِنَ النُّوقِ: الَّتِي كَرِهَتْ وَلَدَهَا، وَلَمْ تَشَمَّهُ عِنْدَ وِلَادَتِهِ.

صَدَّتْ، وَيُبْدِي الكِبَرُ المَقَاذِرَا صُدُودَ أُمِّ البَوِّ، أَمْسَتْ ذَائِرَا مِنْ أَنْ رَأَتْ، فِي لِحْيَتِي، القَتَائِرَا
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏159‏.
244هـ=858م

الذَّائِرُ مِنَ الأَشْخَاصِ: الَّذِي يَأْبَى عَلَى غَيْرِهِ وَلَا يَنْعَطِفُ.

لَئِنْ كُنْتَ لِي يَوْمًا بِنَجْرَانَ ذَائِرًا لَقَدْ كُنْتَ لِي يَوْمًا بِمَنْدَدَ رَائِمَا
من إنشاد ابن السِّكّيت
المذكّر والمؤنّث: أبو بكر ابن الأنباريّ (ت، 328هـ)، تحقيق: محمّد عبد الخالق عضيمة، راجعه وصنع فهارسه: رمضان عبد التوّاب، وزارة الأوقاف، القاهرة، (1401هـ/ 1981م- 1419هـ/ 1999م)، ‏2/‏110‏.

ذَأَّرَ[مُتَعَدٍّ]

ن110هـ=728م

ذَأَّرَ النَّاقَةَ: طَلَى ضُرُوعَهَا بِالذِّئَارِ.

أَعَيَّاشُ، قَدْ آوَتْ قُفَيْرَةُ نَسْلَهَا إِلَى بَيْتِ لُؤْمٍ مَا لَهُ مِنْ مُحَوَّلِ
تُذَئِّرُ أَبْكَارَ اللِّقَاحِ وَلَمْ تَكُنْ قُفَيْرَةُ تَدْرِي مَا جَنَاةُ القَرَنْفُلِ
جرير
ديوان جرير: شرح: محمّد بن حبيب: تحقيق: نعمان محمّد أمين طه، دار المعارف، القاهرة، ط3، (1986م)، ‏3/‏946‏.

ذِئْرَة[اسْم]

ن175هـ=791م

الذِّئْرَةُ : البَعَرُ الرَّطْبُ المُخْتَلِطُ بِالتُّرَابِ.

"السِّرْقِينُ المُخْتَلِطُ بِالتُّرَابِ يُسَمَّى ذِئْرَةً ، فَإِذَا طُلِيَ عَلَى أَطْبَاءِ النَّاقَةِ لِئَلَّا يَرْضَعَهَا الفَصِيلُ، فَهُوَ الذِّئَارُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏197‏.

ذَئِر[صِفَةٌ مُشَبَّهَة]

ن175هـ=791م

الذَّئِرُ : المُغْتَاظُ الغَضْبَانُ.

"السَّبُعُ ذَئِرٌ عَلَى عَدُوِّهِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏8/‏196‏.
206هـ=821م

الذَّئِرُ مِنَ الكَلَامِ: المُتَضَمِّنُ وَعِيدًا وَتَهْدِيدًا، مِنْ دُونِ أَنْ يَصْنَعَ صَاحِبُهُ شَيْئًا.

"وَإِنَّهُ لَذَئِرُ الكَلَامِ"
العُذْريّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏222‏.

ذَأَر[مَصْدَر]

ن283هـ=896م

الذَّأَرُ : الِاغْتِيَاظُ وَالغَضَبُ.

لَوْ كَانَ فِعْلَ الوَرَى، لَقَدْ ذَئِرَتْ لَهُ المَسَاعِيرُ، أَيَّمَا ذَأَرِ
لَكِنَّهُ وِتْرُ مَالِكٍ مَلِكٍ يَعْلُو، عَلَى الطَّالِبِينَ، بِالثُّؤَرِ
ابن الرّوميّ
ديوان ابن الرّوميّ أبي الحسين علي بن العباس بن جريح: تحقيق: حسين نصار، مطبعة دار الكتب والوثائق القومية، القاهرة، ط3، (1424هـ/ 2003م)، ‏3/‏922‏.

ذَأَرَ[مُتَعَدٍّ]

ن304هـ=916م

ذَأَرَ النَّاقَةَ: طَلَى ضُرُوعَهَا بِالذِّئَارِ.

"إِذَا لَهِجَ الفَصِيلُ بِالرَّضَاعِ، فَإِمَّا أَنْ يُشَقَّ وَسَطُ لِسَانِهِ وَيُخَلَّ بِخِلَالٍ، وَإِمَّا أَنْ تُصَرَّ أُمُّهُ؛ وَذَلِكَ أَنْ يَفُتُّوا بَعَرًا عَلَى كُلِّ خِلْفٍ مِنْ ضَرْعِهَا فَتَذْأَرُهُ بِذَلِكَ الذِّئَارِ"
أبو محمّد الأنباريّ
ديوان المفضّليّات: المفضَّل الضَّبّيّ (ت، 178هـ)، شرح: أبي محمّد القاسم بن محمّد بن بشّار الأنباريّ، تحقيق: كارلوس يعقوب لايل، مطبعة الآباء اليسوعيّين، بيروت، كلارندون، أكسفورد، (1920م)، ‏526‏.

ذَأْرَة[اسْمُ مَرَّة]

1302هـ=1884م

الذَّأْرَةُ : المَرَّةُ مِنَ الذَّأَرِ.

فَتًى يَتَحَرَّى المَجْدَ بِالجِدِّ، فَهْوَ لَا يُوَدِّعُ يَوْمًا، لَمْ يُفَاخِرْ وَيَمْجُدِ
تُحَوِّلُهُ، عَنْ لَفْتَةِ الظَّبِي، ذَأْرَةٌ وَيَشْغَلُهُ الضِّرْغَامُ عَنْ لَثْمِ أَغْيَدِ
حسن حسني الطويراني
النشر الزهري في رسائل النسر الدهري: حسن حسني الطويراني (ت، 1315هـ)، مطبعة محمود بك، دار الخلافة، (1306ه)، ‏75‏.
