أُمْلُوج[اسْم]

9هـ=630م

الأُمْلُوجُ : ضَرْبٌ مِنْ وَرَقِ الشَّجَرِ.

"قَدْ نَشِفَ المُدْهُن، ويَبِسَ الجِعْثِن، وَمَاتَ العُسْلُوج، وَسَقَطَ الأُمْلُوج ، وَهَلَكَ الهَدِيّ، وَمَاتَ الوَدِيّ"
طَهْفَة/طَخْفَة بن أبي زهير النَّهْديّ
كتاب العقد الفريد: ابن عبد ربّه الأندلسيّ (ت، 328هـ)، شرحه وضبطه وصححه وعنون موضوعاته ورتب فهارسه: أحمد أمين وآخرون، مطبعة لجنة التأليف والترجمة والنشر، القاهرة، ط3، (1363هـ/ 1944م- 1393هـ/ 1973م)، ‏2/‏53‏.

إِمْلَاجَة[اسْمُ مَرَّة]

ن11هـ=632م

الإِمْلَاجَةُ : المَرَّةُ مِنَ الإِمْلَاجِ، وَهُوَ الإِرْضَاعُ.

"لَا تُحَرِّمُ الإِمْلَاجَةُ ، وَلَا الإِمْلَاجَتَانِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏44/‏443،،‏(رقم الحديث: 26873).

أَمْلَج[صِفَةٌ مُشَبَّهَة]

ن50هـ=670م

الأَمْلَجُ مِنَ النَّبَاتِ: الأَصْفَرُ الَّذِي بَيْنَ الأَخْضَرِ وَالأَبْيَضِ.

عَوَارِضُ مِنْ نَوْءِ السِّمَاكَيْنِ مُزْنُهُ يُنَحَّرُ فِي البِيضِ الدِّمَاثِ وَيُنْتَجُ
هَمَلْنَ بِهِ حَتَّى دَنَا الصَّيْفُ وَانْقَضَى رَبِيعٌ، وَحَتَّى هَائِجُ البَقْلِ أَمْلَجُ
مُلَيح بن الحَكَم الهُذَليّ
كتاب شرح أشعار الهُذَليّين: صنعة: أبي سعيد السُّكَّريّ (ت، 275هـ)، رواية: أبي الحسن علي بن عيسى بن علي النحوي بن أبي بكر أحمد بن محمّد الحُلواني عن السكري، حققه: عبد الستار أحمد فرّاج، راجعه: محمود محمّد شاكر، مكتبة دار العروبة- مطبعة المدني، القاهرة، (1384هـ/ 1965م)، ‏3/‏1032‏.
ن206هـ=821م

الأَمْلَجُ مِنَ الأَغْصَانِ وَنَحْوِهَا: العَارِي مِنَ الأَوْرَاقِ.

دَافَعْنَ عَنَّا فِي السَّلِيقِ الأَمْلَجِ حَتَّى انْجَلَى طَبْخُ الشِّتَاءِ المُنْضِجِ
الأسلميّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏98‏.

أَمْلَج[صِفَة]

ن1323هـ=1904م

الأَمْلَجُ : الأَسْمَرُ.

"أَوْفَى لِي حُكْمُ القَدَرِ بِالِاطِّلَاعِ عَلَى كِتَابِ (نَهْجُ البَلَاغَةِ) مُصَادَفَةً بِلَا تَعَمُّلٍ... فَكَانَ يُخَيَّلُ لِي فِي كُلِّ مَقَامٍ أَنَّ حُرُوبًا شُبَّت، وَغَارَاتٍ شُنَّت، وَأَنَّ لِلْبَلَاغَةِ دَوْلَة، وِلِلْفَصَاحَةِ صَوْلَة، وَأَنَّ لِلْأَوْهَامِ عَرَامَة، وِلِلرِّيَبِ دَعَارَة، وَأَنَّ جَحَافِلَ الخَطَابَة، وَكَتَائِبَ الذَّرَابَة، فِي عُقُودِ النِّظَام، وَصُفُوفِ الِانْتِظَام، تُنَافِحُ بِالصَّفِيحِ الأَبْلَج، وَالقَوِيمِ الأَمْلَج "
محمد عبده
الأعمال الكاملة للإمام الشيخ محمد عبده: محمد عبده (ت، 1323هـ)، تحقيق: محمد عمارة، دار الشروق، بيروت، القاهرة، ط1، (1414ه/ 1993م)، ‏2/‏409‏.

مَلْج[مَصْدَر]

70هـ=689م

المَلْجُ : الإِرْضَاعُ.

"أُذْكِرُكَ مَلْجَ فُلَانَةَ"
عمرو بن سعيد بن العاص الأموي
النهاية في غريب الحديث والأثر: ابن الأثير أبو السعادات (ت، 606هـ)، تحقيق: محمود محمد الطناحي، طاهر أحمد الزاوي، المكتبة الإسلامية، بيروت، ط1، (1383ه/ 1963م)، ‏4/‏353‏.
ن175هـ=791م

المَلْجُ : تَنَاوُلُ الضَّرْعِ وَالثَّدْيِ بِأَدْنَى الفَمِ.

" المَلْجُ : تَنَاوُلُ الضَّرْعِ وَالثَّدْيِ بِأَدْنَى الفَمِ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏140‏.

مَلْج[اسْم]

640هـ=1242م

المَلْجُ : لَحْمُ العَضَلِ.

"يُؤْخَذُ مِنْ أَطَايِبِ اللَّحْمِ البَقَرِيِّ مِثْلِ الصُّدُورِ، وَالأَمْلَاجِ ، وَالسُّدُسِيَّةِ، وَغَلِيظِ الكِرْشِ"
ابن رزين التَّجيبي
فضالة الخوان في طيبات الطعام والألوان: ابن رزين التَّجيبي (ت، 692هـ)، حققه وقدم له: محمد بن شقرون، أشرف على إعداده: إحسان عباس، دار الغرب الإسلامي، بيروت، ط2، (1984م)، ‏41‏.

مَلَجَ[مُتَعَدٍّ]

ن74هـ=693م

مَلَجَ الدَّمَ وَنَحْوَهُ: مَصَّهُ.

"فَلَمَّا نُزِعَتَا جَعَلَ الدَّمُ يَسْرُبُ كَمَا يَسْرُبُ الشَّنُّ، فَجَعَلَ مَالِكُ بْنُ سِنَانٍ يَمْلُجُ الدَّمَ بِفِيهِ، ثُمَّ ازْدَرَدَهُ"
أبو سعيد الخُدْريّ الخزرجيّ
كتاب المغازي: الواقديّ (ت، 207هـ)، تحقيق: مارسدن جونس، عالم الكتب، القاهرة، ط3، (1404هـ/ 1984م)، ‏1/‏247‏.

مَلْجَة[اسْمُ مَرَّة]

ن175هـ=791م

المَلْجَةُ : المَرَّةُ مِنَ المَلْجِ.

"المَلْجُ: تَنَاوُلُ الضَّرْعِ وَالثَّدْيِ بِأَدْنَى الفَمِ، وَفِي الحَدِيثِ: لَا بَأْسَ بِالإِمْلَاجَةِ وَالإِمْلَاجَتَيْنِ، وَهُوَ أَنْ يَتَناوَلَ الصَّبِيُّ مِنْ ثَدْيِ أُمِّهِ مَلْجَةً ، أَوْ مَلْجَتيْنِ، شُرْبًا يَسِيرًا، ثُمَّ تَقْطَعَ ذَلِكَ عَنْهُ"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏6/‏140‏.

تَمَلَّجَ[مُتَعَدٍّ بِالحَرْف]

ن206هـ=821م

تَمَلَّجَ مِنَ المَاءِ وَنَحْوِهِ: امْتَصَّ مِنْهُ.

كُلُّ بَنِي مُجَاشِعٍ تَمَلَّجَا مِنْ نَاطِفٍ يَسْلُجُ مِنْهُ سُلَّجَا
الكلبيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏2/‏106‏.

مُمَالَجَة[مَصْدَر]

218هـ=833م

المُمَالَجَةُ : تَنَاوُلُ الضَّرْعِ بِأَدْنَى الفَمِ.

" المُمَالَجَةُ أَوَّلُ الرَّضَاعِ"
جارية روى عنها إسحاق بن إبراهيم الموصلي
شرح مقامات الحريريّ: الشّريشيّ (ت، 619هـ)، تحقيق: محمَّد أَبو الفضل إِبراهيم، المكتبة العصريَّة- شركة أبناء شريف الأنصاري- الدار النموذجية- المطبعة العصرية، بيروت- صيدا، (1413ه/ 1992م)، ‏4/‏336‏.

أَمْلَجَ[مُتَعَدٍّ]

ن224هـ=839م

أَمْلَجَتِ الصَّبِيَّ: أَرْضَعَتْهُ.

"قَدْ أَمْلَجَتْ صَبِيَّهَا إِمْلَاجًا"
أبو عُبيد الهَرَويّ
كتاب غريب الحديث: أبو عُبيد الهَرَويّ (ت، 224هـ)، تحقيق: حسين محمّد محمّد شرف، مراجعة: عبد السلام هارون وآخرين، الهيئة العامة لشؤون المطابع الأميرية، القاهرة، (1404هـ/ 1984م- 1419هـ/ 1999م)، ‏2/‏394‏.
1040هـ=1630م

أَمْلَجَ العِلْمَ وَنَحْوَهُ: قَعَّدَهُ، وَأَظْهَرَهُ تَامًّا مُكْتَمِلًا.

"وَبَلَغَ نِصَابِي إِلَى حَيْثُ فَكَكْتُ عُقَدَ الأَفْكَار، وَنَحَلْتُ العُلُومَ وَنَقَدْتُ الأَنْظَار، ...، وَأَبْرَحْتُ الحِكْمَةَ وَأَبْلَجْتُهَا، وَأَمْلَجْتُ الفَلْسَفَةَ وَأَنْضَجْتُهَا"
محمد باقر الحسيني الآستربادي المعروف بالميرداماد
مصنفات ميرداماد: محمد باقر الحسيني الآستربادي المعروف بالميرداماد (ت، 1041هـ)، اهتمام: عبدالله نوراني، سلسله انتشارات، طهران، (2003- 2006م)، ‏2/‏2‏.

مُلْج[اسْمُ جِنْس]

352هـ=963م

المُلْجُ : نَوَى ثَمَرِ شَجَرِ الدَّوْمِ.

"الأُمْلُوجُ عِنْدِي نَوَى المُقْلِ، مِثْلُ المُلْجِ سَوَاءً"
أبو منصور الأزهريّ الهَرَويّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏11/‏105‏.

إِمْلَاج[مَصْدَر]

846هـ=1442م

الإِمْلَاجُ : الإِرْضَاعُ.

فَاسْلَمْ وَدُمْ فِي رِفْعَةٍ مَعَ نَاظِرٍ لِجُيُوشِنَا فِي صُحْبَةِ الأَمْشَاجِ
وَعَلَيْكُمَا أَزْكَى السَّلَامِ وَرَحْمَةٍ مَا حَنَّ مَوْلُودٌ إِلَى الإِمْلَاجِ
سراج الدين عمر بن موسى الحمصيّ المخزوميّ
عنوان الزَّمان بتراجم الشُّيوخ والأقران: إبراهيم بن حسن البِقاعِيّ (ت، 885هـ)، حققه وقدم له وعلق عليه: حسن حبشي، دار الكتب والوثائق القومية، القاهرة، ط1، (1422ه/ 2001م- 1430ه/ 2009م)، ‏4/‏125‏.
1010هـ=1601م

الإِمْلَاجُ : الشُّرْبُ وَالِارْتِشَافُ.

"وَقَدْ أُحْضِرَ مِنْ سَمِينِ الدَّجَاج، وَلَحْمِ طَيْرٍ مِمَّا تَشْتَهِي الأَنْفُسُ كَالإِوَزِّ وَالسُّمْنِ وَالدُّرَّاج، مَعَ مَا يَلِي ذَلِكَ مِنْ نَفِيسِ الأَطْعِمَةِ... وَنَفَائِسِ الأَشْرِبَةِ السُّكَّرِيَّة مَعَ الكُلَّاج، مَصْفُوفَةً فِي صُحُونِ العِجَاج، عَلَيْهَا مَعَالِقُ مِنْ فِضَّةٍ قَدَّرُوهَا لِلإِمْلَاج "
الطالوي
سانحات دمى القصر في مطارحات بني العصر: الطالوي (ت، 1014هـ)، تحقيق: محمَّد مرسي الخولي، عالم الكتب، بيروت، ط1، (1403ه/ 1983م)، ‏2/‏183‏-‏ 184‏.

مُمْلِج[اسْمُ فَاعِل]

ن1102هـ=1690م

المُمْلِجُ : المُرْضِعُ.

مَا لِي سِوَاكَ، فَلَا تَكِلْنِي طَرْفَةً لِسِوَاكَ، وَاكْلَأْنِي كَلَاءَةَ مُمْلِجِ
الحسن بن مسعود اليوسي
ديوان اليوسي: الحسن بن مسعود اليوسي (ت، 1102هـ)، جمع وتحقيق: عبد الجواد السقاط، تقديم: عباس الجراري، مطبعة دار المناهل، الرباط، (2016م)، ‏96‏.

امْتَلَجَ[مُتَعَدٍّ]

ن1120هـ=1708م

امْتَلَجَ الثَّدْيَ: امْتَصَّهُ وَارْتَضَعَهُ.

"كَأَنَّهُ القَمَرُ فِي اسْتِهْلَالِه، أَوِ الهِلَالُ فِي إِهْلَالِه، رَضَعَ أَلْبَانَ الجَلَالَة، وَامْتَلَجَ ثَدْيَ الرِّسَالَة"
أبو العباس أحمد بن عبد الحي الحلبي الفاسي
الدر النفيس والنور الأنيس في مناقب الإمام إدريس بن إدريس رضي الله عنهما: أبو العباس أحمد بن عبد الحي الحلبي الفاسي (ت، 1120هـ)، تقديم وتحقيق: محمد بوخنيفي، دار الكتب العلمية، بيروت، ط1، (1438ه/ 2017م)، ‏1/‏319‏.

امْتِلَاج[مَصْدَر]

1270هـ=1853م

الِامْتِلَاجُ : الِامْتِصَاصُ.

"وَمِنْهُنَّ مَنْ يُؤْثِرُ التَّجَضُّمَ الكَمَرِيَّ، أَوِ الِامْتِلَاجَ القُنْبِيَّ"
أحمد فارس الشدياق
كتاب الساق على الساق في ما هو الفارياق: أحمد فارس الشدياق (ت، 1305هـ)، بنجامين دوبرات، مكتبة المعهد، باريس، (1270ه/ 1855م)، ‏637‏.

1327هـ=1908م

المَالِجُ الثَّدْيَ وَنَحْوَهُ: المَاصُّ لَهُ الرَّاضِعُ.

"سَمْلَجَهُ فِي حَلْقِهِ: جَرَعَهُ جَرْعًا سَهْلًا، مِنْ مَلَجَ الصَّبِيُّ أُمَّهُ: إِذَا تَنَاوَلَ ثَدْيَهَا بِأَدْنَى فَمِهِ، فَرَضَعَهَا، كَأَنَّ المُسَمْلِجَ يَفْعَلُ فِعْلَ الطِّفْلِ المَالِجِ "
أنستاس ماري الكرملي
خصائص الموازين العربية وأصل كيفية نشوئها: ‏أنستاس ماري الكرملي،‏المشرق: المطبعة الكاثولوكية للآباء اليسوعيين، بيروت، س11، ع7، (03/ 06/ 1326هـ)، ‏490‏.
