زُغْلَة[اسْم]

قبل 100ق.هـ=525م

الزُّغْلَةُ : الدُّفْعَةُ مِنَ المَاءِ وَنَحْوِهِ.

أَنَا قُصَيٌّ، وَحَفَرْتُ سَجْلَهْ تُرْوِي الحَجِيجَ زُغْلَةً فَزُغْلَهْ
قصيّ بن كلاب
معجم ما اسْتَعْجَم من أسماء البلاد والمواضع: أبو عُبيد البَكْريّ الأندلسيّ (ت، 487هـ)، حققه وضبطه: مصطفى السقا، عالم الكتب، بيروت، (1364هـ/ 1945م)، ‏3/‏724‏.
300هـ=912م

الزُّغْلَةُ : الِاسْتُ.

"وَمِنْ سَبِّهِمْ: يَا زُغْلَةَ الثَّوْرِ"
من رواية أبي عليّ الهَجَريّ
التّعليقات والنّوادر دراسة ومختارات: أبو عليّ الهَجَريّ (ت، 300هـ)، ترتيب: حَمَد الجاسر، ط1، (1413هـ/ 1992م)، ‏1137‏.
352هـ=963م

الزُّغْلَةُ : قَدْرُ مَا يَمْلَأُ الفَمَ مِنْ لَبَنٍ وَنَحْوِهِ.

"اسْقِنِي زُغْلَةً مِنَ اللَّبَنِ"
أعرابيٌّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏50‏.

أَزْغَلَ[لَازِم]

ن10ق.هـ=612م

أَزْغَلَتِ الطَّعْنَةُ وَنَحْوُهَا: دَفَعَتْ بِالدَّمِ دُفْعَةً بَعْدَ دُفْعَةٍ.

وَلَقَدْ دَفَعْتُ إِلَى دُرَيْدٍ طَعْنَةً نَجْلَاءَ تُزْغِلُ مِثْلَ عَطِّ المَنْحَرِ
صَخر بن عمرو بن الشَّريد الرِّياحيّ السُّلَميّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏15/‏70‏.
ن127هـ=745م

أَزْغَلَ الشَّخْصُ: فَعَلَ فِعْلَ الصِّغَارِ، وَدَخَلَ فِي حُكْمِهِمْ.

" أَزْغَلْتَ ، يَا أَبَا سَلَمَةَ"
عاصم بن أبي النُّجود القارئ
أساس البلاغة: جار الله الزمخشريّ (ت، 538هـ)، مطبعة دار الكتب المصرية، القاهرة، (1341ه/ 1923م)، ‏1/‏415‏.

أَزْغَلَ[مُتَعَدٍّ]

ن75هـ=694م

أَزْغَلَ الطَّائِرُ فَرْخَهُ: أَطْعَمَهُ بِفِيهِ دُفْعَةً بَعْدَ دُفْعَةٍ.

ﻓَﺄَزْغَلَتْ ﻓِﻲ ﺣَﻠْﻘِﻪِ زُﻏْﻠَﺔً ﻟَﻢْ تُخْطِئِ اﻟﺠِﻴﺪَ، وَﻟَﻢْ ﺗَﺸْﻔَﺘِﺮْ
عمرو بن أحمر الباهليّ
شعر عَمرو بن أحمر الباهلي: جمعه وحققه: حسين عطوان، مطبوعات مجمع اللغة العربية، دمشق، د.ت. ‏69‏.
ن200هـ=815م

أَزْغَلَ الشَّيْءَ: صَبَّهُ.

"وَثَلُّوا فِي القَصْعَةِ ثَلَّةً مِنَ الدَّقِيقِ، ثُمَّ أَزْغَلُوا فِيهَا زُغْلَةً مِنْ زَيْتِهِمْ"
أبو زياد الكلابيّ
التّنبيهات: عليّ بن حمزة البصريّ (ت، 375هـ)، تحقيق: عبد العزيز الميمني، دار المعارف، القاهرة، ط3، (1967م)، ‏268‏.

زَغَلَ[لَازِم]

ن41هـ=661م

زَغَلَ الشَّيْءُ: انْصَبَّ مِنْهُ السَّائِلُ دُفْعَةً بَعْدَ دُفْعَةٍ.

فِي كُلِّ مُعْتَرَكٍ يُرَى مِنَّا فَتًى يَهْوِي، كَعَزْلَاءِ المَزَادَةِ، يَزْغَلُ
أبو العِيال بن أبي غُثير/عُتير الهُذَليّ
ديوان الهُذَليّين: تحقيق: أحمد الزين وآخرين، مطبعة دار الكتب المصرية، القاهرة، ط2، (1995م)، ‏2/‏254‏.

زَغَلَ[مُتَعَدٍّ]

ن685هـ=1286م

زَغَلَ الشَّيْءَ: غَشَّهُ وَزَيَّفَهُ.

"فَإِنَّهُ يَقْطُرُ فِيهِ مِنْ ذَلِكَ النَّبَاتِ شَيْءٌ شَبِيهٌ بِاللَّبَنِ، وَيُتْرَكُ إِلَى حِينِ يَجْمُدُ، هَذَا هُوَ الخَالِصُ، وَقَدْ يُزْغَلُ "
ابن القُفّ الكَرَكي
كتاب العمدة في الجراحة: ابن القُفّ الكَرَكي (ت، 685هـ)، دائرة المعارف العثمانية، حيدر آباد، ط1، (1997م)، ‏1/‏260‏.

زُغْلُول[صِفَة]

49هـ=669م

الزُّغْلُولُ : الصَّبِيُّ الصَّغِيرُ.

إِنْ سَابَقُوا سُبِقُوا، أَوْ نَافَرُوا نُفِروا أَوْ كَاثَرُوا وَاحِدًا، مِنْ غَيْرِهِمْ، كُثِرُوا
شِبْهُ الزَّغَالِيلِ ، لَا دِينٌ، وَلَا حَسَبٌ لَوْ قَامَرُوا الزِّنْجَ، عَنْ أَحْسَابِهِمْ، قُمِرُوا
حسّان بن ثابت الخزرجيّ الأنصاريّ
الأخبار الموفّقيّات: الزُّبير بن بَكّار الأسديّ القُرَشيّ (ت، 256هـ)، تحقيق: سامي مكي العاني، عالم الكتب، بيروت، ط2، (1416هـ/ 1996م)، ‏211‏.
ن90هـ=709م

الزُّغْلُولُ : الخَفِيفُ السَّرِيعُ.

إِذَا بَدَتْ عَوْرَةٌ مِنْهَا أَضَرَّ بِهَا بَادِي الكَرَادِيسِ خَاظِي اللَّحْمِ زُغْلُولُ
الأخطل
شعر الأخطل أبي مالك غياث بن غوث التغلبي: صنعة: السُّكّري، رواية: أبي جعفر محمّد بن حبيب، تحقيق: فخر الدين قباوة، دار الفكر المعاصر، بيروت، دار الفكر، دمشق، ط4، (1416هـ/ 1996م)، ‏53‏.

زُغْلُول[اسْم]

ن398هـ=1007م

الزُّغْلُولُ : فَرْخُ الطَّيْرِ وَنَحْوِهِ الَّذِي تُزْغِلُهُ أُمُّهُ.

"أَنَا رَجُلٌ مِنْ أَهْلِ الإِسْكَنْدَرِيَّةِ، مِنَ الثُّغُورِ الأُمَوِيَّةِ، قَدْ وَطَّأَ لِيَ الفَضْلُ كَنَفَهُ، وَرَحَّبَ بِي عَيْشٌ، وَنَمَانِي بَيْتٌ، ثُمَّ جَعْجَعَ بِي الدَّهْرُ عَنْ ثُمِّهِ وَرُمِّهِ، وَأَتْلَانِي زَغَالِيلَ حُمْرَ الحَوَاصِلِ"
بديع الزّمان الهَمَذانيّ
مقامات بديع الزّمان الهَمَذانيّ: بديع الزّمان الهَمَذانيّ (ت، 398هـ)، قدم لها وشرح غوامضها: الشّيخ محمّد عبده، دار الكتب العلميّة، بيروت، ط3، (1426هـ/ 2005م)، ‏76‏.
ن1098هـ=1686م

الزُّغْلُولُ : فَرْخُ الحَمَامِ الفَتِيُّ.

"وَارْضَ اللهم عَنِ السِّتَّةِ البَاقِينَ مِنَ العَشَرَةِ الأَطْعِمَةِ المُفْتَخَرَةِ: المَاوَرْدِيَّةِ، وَالمُهَلَّبِيَّةِ، وَالشَّعْرِيَّةِ بِالزَّغَالِيلِ المَرْبِيَّةِ... مَتَّعَنَا اللهُ وَإِيَّاكُمْ بِهَا أَجْمَعِينَ"
الشّربيني
هز القحوف في شرح قصيدة أبي شادوف: الشّربيني (ت، 1098هـ)، اعتنى به: عبد العزيز جمال الدّين، الهيئة العامة لقصور الثَّقَافَة، القاهرة، ط1، (1440ه/ 2019م)، ‏340‏.
1305هـ=1887م

الزُّغْلُولُ : ضَرْبٌ مِنَ التَّمْرِ.

"وَفِيهَا نَخِيلٌ [مَشْهُورٌ] بِكَثْرَةِ ثَمَرِهِ، فِي غَايَةِ الجَوْدَةِ، وَيَتَأَخَّرُ نُضْجُهُ عَنْ مُعْتَادِ نَخِيلِ القُطْرِ أَكْثَرَ مِنْ شَهْرٍ... وَهُوَ أَصْنَافٌ فَمِنْهُ الزُّغْلُولُ ، وَمِنْهُ السَّمَّانِيُّ، وَمِنْهُ الحَيَّانِيُّ"
علي باشا مبارك
الخطط التوفيقية الجديدة لمصر القاهرة ومدنها وبلادها القديمة والشهيرة: علي باشا مبارك (ت، 1311هـ)، المطبعة العامرة، بولاق، مصر، ط1، (1306ه)، ‏11/‏76‏.

تَزَغُّل[مَصْدَر]

65هـ=685م

التَّزَغُّلُ : النَّشَاطُ.

يَرْمِي بِأَجْوَازِ المَهَارَى النُّحَّلِ ذَوَابِلًا مِثْلَ القِسِيِّ الذُّبَّلِ يَنْتُقْنَ بِالقَوْمِ مِنَ التَّزَغُّلِ وَهِزَّةِ المِرَاحِ وَالتَّخَيُّلِ
العجّاج
ديوان العجاج: رواية: عبد الملك بن قُرَيب الأصمعيّ وشرحه، عني بتحقيقه: عزة حسن، دار الشرق العربي، بيروت- حلب، (1416هـ/ 1995م)، ‏211‏.

إِزْغَالَة[مَصْدَر]

ن75هـ=694م

الإِزْغَالَةُ : الدُّفْعَةُ مِمَّا يُصَبُّ وَيُقْذَفُ.

فَأَزْغَلَتْ فِي الحَلْقِ إِزْغَالَةً
عمرو بن أحمر الباهليّ
كتاب الأغاني: أبو الفرج الأصفهانيّ (ت، 356هـ)، تحقيق: إحسان عبّاس وآخرين، دار صادر، بيروت، ط3، (1429هـ/ 2008م)، ‏15/‏70‏.

مُزْغَل[اسْمُ مَفْعُول]

ن110هـ=728م

المُزْغَلُ مِنَ النَّقْدِ: المُزَيَّفُ.

"وَالدَّرَاهِمُ المُزْغَلَةُ غِشٌّ، وَكَذِبٌ، وَخِلَافٌ، وَخِيَانَةٌ فِي المَعِيشَةِ، وَاجْتِرَاءٌ عَلَى الكَبَائِرِ"
ابن سِيرين
منتخب الكلام في تفسير الأحلام: محمد بن سيرين (ت: 110هـ)، اعتنى به ورتبه: محمود طعمة حلبي، الناشر: دار المعرفة، بيروت، ط5، (1423هـ/2002م)، ‏140‏.
ن1244هـ=1828م

المُزْغَلُ : المَنْقُوصُ.

وَيَكُونُ لَوْنُ حُبُوبِهَا كَعَوَارِضٍ فِي وَجْهِ خِلِّي بِالبَيَاضِ مُجَمَّلَهْ
وَتَكُونُ قُفَّتُهَا مِنَ الخُوصِ النَّقِي وَبِوَزْنِهَا مَتْمُومَةً لَا مُزْغَلَهْ
نقولا بن يوسف الترك الإسطمبولي
ديوان المعلم نقولا الترك: نقولا بن يوسف الترك الإسطمبولي (ت، 1244هـ)، ضبط نصوصه ووضع مقدمته وفهارسه: فؤاد أفرام البستاني، وزارة التربية الوطنية والفنون الجميلة، بيروت، (1949م)، ‏201‏.
ن1244هـ=1828م

المُزْغَلُ مِنَ الأَنْسَابِ: المَطْعُونُ فِيهِ وَغَيْرُ الصَّرِيحِ.

وَبِأَنَّهَا عَرَبِيَّةٌ مَشْهُورَةٌ بَيْنَ القَبَائِلِ؛ حُرَّةٌ لَا مُزْغَلَهْ
فَعَلَامَ تَظْهَرُ أَنَّهَا عَجَمِيَّةٌ وَأُصُولُهَا مَنْكُورَةٌ مُسْتَجْهَلَهْ؟
نقولا بن يوسف الترك الإسطمبولي
ديوان المعلم نقولا الترك: نقولا بن يوسف الترك الإسطمبولي (ت، 1244هـ)، ضبط نصوصه ووضع مقدمته وفهارسه: فؤاد أفرام البستاني، وزارة التربية الوطنية والفنون الجميلة، بيروت، (1949م)، ‏202‏.

مُزْغَل[اسْمُ مَكَان]

1271هـ=1854م

المُزْغَلُ : الكُوَّةُ تُجْعَلُ فِي جُدْرَانِ الحُصُونِ لِمُرَاقَبَةِ العَدُوِّ وَرَمْيِهِ.

وَكَمْ دَرْوَةٍ مِنْ مُزْغَلِ المَوْتِ أَحْرَقَتْ بِنِيرَانِهَا جَيْشَ العُدَاةِ المُلِمَّةِ!
محمد صالح مجدي بك
ديوان المرحوم السيد صالح مجدي بك: محمد صالح مجدي بك (ت، 1298هـ)، المطبعة الأميرية، بولاق، مصر، (1311ه)، ‏54‏.
ن1274هـ=1857م

المُزْغَلَةُ : المُزْغَلُ.

"وَصَنَعَ بَيْتًا جَمِيلًا، وَسَوَّرَهُ بِسُورٍ، وَجَعَلَ السُّورَ عَرِيضًا، وَجَعَلَ فِيهِ مُزْغَلَتَيْنِ مُقَابِلَتَيْنِ لِبَيْتِ السُّلْطَانِ، يَضَعُ فِي إِحْدَاهُمَا المِدْفَعَ، وَفِي الأُخْرَى هَاوُنَ البُنْبِ"
محمد بن عمر التونسي
تشحيذ الأذهان بسيرة بلاد العرب والسودان: محمد بن عمر التونسي (ت، 1274هـ)، حققه وكتب حواشيه: خليل محمود عساكر، مصطفى محمد مسعد، راجعه: محمد مصطفى زيادة، المؤسسة المصرية العامة للتأليف والأنباء والنشر، القاهرة، ط1، (1965م)، ‏118‏.

زَغْل[اسْم]

زَاغِل[اسْمُ فَاعِل]

ن1061هـ=1650م

الزَّاغِلُ : مُزَيِّفُ النَّقْدِ مِنَ الذَّهَبِ وَالفِضَّةِ.

"مِنَ الفَسَادِ فِي الأَرْضِ الزَّغَلُ، وَضَرْبُ مِثْلِ السِّكَّةِ الَّتِي يَضْرِبُهَا الإِمَامُ، إِلَّا أَنَّ الزَّاغِلَ يُدْخِلُ فِي الذَّهَبِ وَالفِضَّةِ غَيْرَهُمَا مِنَ المَعَادِنِ؛ كَالفِضَّةِ فِي الذَّهَبِ وَالنَّحَاسِ فِيهِمَا"
نجم الدّين الغزي
حسن التّنبُّه لما ورد في التّشبّه: نجم الدّين الغزّي (ت، 1061هـ)، تحقيق: لجنة مختصة من المحققين بإشراف نور الدين طالب، دار النوادر، دمشق، بيروت، الكويت، ط1، 1432هـ/2011م، ‏139‏.

تَزْغِيل[مَصْدَر]

1305هـ=1887م

التَّزْغِيلُ : الغِشُّ وَالتَّزْيِيفُ.

"وَلَمْ أَصْنَعْ - وَالحَمْدُ للهِ - صَنِيعَهُ فِي انْتِقَادِ مُسْتَقِيمِ القَوْلِ وَأَعْوَجِه، بَلْ غَايَرْتُهُ فِيمَا جَنَحَ إِلَيْهِ مِنْ تَزْغِيلِ إِبْرِيزِ الكَلَامِ وَبَهْرَجِه"
أبو بكر بن شهاب العلوي الحسيني الحضرمي
إقامة الحجة على التقي ابن حجة: أبو بكر بن عبد الرحمن بن شهاب الدين العلوي الحسيني الحضرمي، مطبعة نخبة الأخبار، بومبي، 1305هـ/1887م، ‏4‏.
