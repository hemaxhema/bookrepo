قَسْقَاسَة[مَصْدَر]

ن11هـ=632م

قَسْقَاسَةُ العَصَا وَنَحْوِهَا: تَحْرِيكُهَا.

"أَبُو جَهْمٍ أَخَافُ عَلَيْكِ قَسْقَاسَتَهُ لِلْعَصَا... وَأَمَّا مُعَاوِيَةُ فَرَجُلٌ أَخْلَقُ مِنَ المَالِ"
حديث نبويّ
مسند الإمام أحمد ابن حنبل: أحمد ابن حَنْبَل (ت، 241هـ)، حققه وخرج أحاديثه وعلق عليه: شعيب الأرنؤوط وآخرون، مؤسسة الرسالة، بيروت، ط1، (1416هـ/ 1995م- 1421هـ/ 2001م)، ‏45/‏321،،‏(رقم الحديث: 27336).

قَسْقَاس[صِيغَةُ مُبَالَغَة]

ن22هـ=643م

القَسْقَاسُ : الدَّؤُوبُ السُّرَى.

ودَلَجُ اللَّيْلِ، وَهَادٍ قَسْقَاسْ
الشَّمّاخ بن ضرار الذُّبْيانيّ
الغريبين في القرآن والحديث: أبو عُبيد الهَرَويّ الفاشانيّ (ت، 401هـ)، تحقيق ودراسة: أحمد فريد المزيدي، قدم له وراجعه: فتحي حجازي، مكتبة نزار مصطفى الباز، مكّة المكرّمة- الرّياض، ط1، (1419هـ/ 1999م)، ‏5/‏1540‏.
120هـ=738م

القَسْقَاسُ مِنَ الحُدَاةِ: الشَّدِيدُ السَّوْقِ.

وَضُمَّرٍ فِي لِينِهِنَّ أَشْرَاسْ يَحْفِزُهَا لَيْلٌ، وَحَادٍ قَسْقَاسْ
رؤبة بن العجّاج
شرح ديوان رُؤبة بن العجّاج: عالم لُغوي قديم، تحقيق: ضاحي عبد الباقي محمّد وآخرين، مراجعة: محمود علي مكي وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1429هـ/ 2008م- 1432هـ/ 2011م)، ‏3/‏200‏-‏ 201‏.

قَسْقَاس[اسْم]

ن145هـ=762م

القَسْقَاسُ : نَبَاتٌ يُشْبِهُ الكَرَفْسَ، لَهُ زَهْرَةٌ بَيْضَاءُ، ذُو رَائِحَةٍ خَبِيثَةٍ، يَنْبُتُ فِي مَسِيلِ المَاءِ.

إِنْ كُنْتَ مِنْ أَمْرِكَ فِي مِسْمَاسِ فَاسْطُ عَلَى أَمِّكَ سَطْوَ المَاسِي وَكُنْتَ مِنْ دَائِكَ ذَا إِفْلَاسِ فَاسْتَقِئَنْ بِثَمَرِ القَسْقَاسِ
رؤبة بن العجّاج
مجموع أشعار العرب وهو مشتمل على ديوان رؤبة بن العجاج وعلى أبيات مفردات منسوبة إليه: اعتنى بتصحيحه وترتيبه: وليم بن الورد البروسيّ، دار ابن قتيبة، الكويت، د.ت. ‏175‏.
206هـ=821م

القَسْقَاسُ : الجُوعُ.

أَتَانَا بِهِ القَسْقَاسُ لَيْلًا وَدُونَهُ جَرَاثِيمُ رَمْلٍ بَيْنَهُنَّ قِفَافُ
أبو جُهيمة الذُّهْليّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏259‏.
772هـ=1370م

القَسْقَاسُ : العَصَا.

"الرَّاءُ (رِ): أَمْرٌ مِنْ وَرَى الزَّنْدَ يَرِي: إِذَا خَرَجَتْ نَارُهُ لِلْإِيقَادِ، وَمِنْ وَرَى جَوْفَهُ: إِذَا آذَاهُ وَأَفْسَدَهُ غَايَةَ الإِفْسَادِ، وَمَضْمُونُهَا نَوَّرَ بِنِبْرَاسِ التَّدْبِيرِ أَطْرَافَ البِلَاد، وَبَوَّرَ بِقَسْقَاسِ التَّدْمِيرِ أَجْوَافَ ذَوِي العِنَاد"
مجد الدين محمد بن يعقوب الفيروزآبادي
كتاب الغرر المثلثة والدرر المبثّثة: مجد الدين محمد بن يعقوب الفيروزآبادي (ت، 817هـ)، تحقيق ودراسة: سليمان بن إبراهيم بن محمد العايد، مكتبة نزار مصطفى الباز، مكة المكرمة- الرياض، ط2، (1421ه/ 2000م)، ‏254‏.

قَسْقَاس[صِفَة]

ن145هـ=762م

القَسْقَاسُ مِنَ اللَّيَالِي: الشَّدِيدُ الظُّلْمَةِ.

كَمْ جُبْنَ مِنْ بِيدٍ، وَلَيْلٍ قَسْقَاسْ
رؤبة بن العجّاج
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏13‏.
ن206هـ=821م

القَسْقَاسُ : سَيْرُ اللَّيْلِ السَّرِيعُ.

لَيْلُ المَطِيِّ الدَّائِبِ القَسْقَاسِ عَلَى الغُلَامِ الغِرِّ ذِي مِرَاسِ
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏99‏.
ن244هـ=858م

القَسْقَاسُ مِنَ السَّيْرِ وَنَحْوِهِ: الشَّدِيدُ الطُّولِ المُتْعِبُ.

"وَيُقَالُ: قَرَبٌ قَسْقَاسٌ "
ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏188‏.

قُسَاقِسَة[اسْم]

ن80هـ=699م

القُسَاقِسَةُ : الأَسَدُ.

جَهْمُ المُحَيَّا عَبُوسٌ بَاسِلٌ شَرِسٌ وَرْدٌ قُسَاقِسَةٌ رِئْبَالَةٌ شَكِمُ
أبو صخر الهُذَليّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏10/‏35‏.

قَسْقَس[اسْم]

ن175هـ=791م

القَسْقَسُ : الدَّلِيلُ الهَادِي المُتَفَقِّدُ الَّذِي لَا يَغْفُلُ.

" وَالقَسْقَسُ : الدَّلِيلُ الهَادِي المُتَفَقِّدُ الَّذِي لَا يَغْفُلُ، إِنَّمَا هُوَ تَلَفُّتًا وَنَظَرًا"
الخليل بن أحمد الفراهيديّ
كتاب العين: الخليل بن أحمد الفراهيديّ (ت، 175هـ)، تحقيق: مهدي المخزومي، إبراهيم السامرائي، دار ومكتبة الهلال، القاهرة، د.ت. ‏5/‏12‏.

قُسْقُوس[صِفَة]

ن206هـ=821م

القُسْقُوسُ : السَّائِرُ اللَّيْلَ كُلَّهُ.

مَاذَا لَقِينَا مِنْهُمُ، يَا قِنْدِيسْ مِنْ بَيْنِ بَاغِي مَأْكَلٍ، أَوْ قُسْقُوسْ ؟
من شواهد أبي عمرو الشّيبانيّ
كتاب الجيم: أبو عمرو الشّيبانيّ (ت، 206هـ)، تحقيق: إبراهيم الأبياريّ وآخرين، مراجعة: محمّد خلف الله أحمد وآخرين، مجمع اللغة العربية، القاهرة، ط1، (1394هـ/ 1974م- 1403هـ/ 1983م)، ‏3/‏103‏.

قَسْقَسَ[لَازِم]

ن206هـ=821م

قَسْقَسَ الشَّخْصُ: سَهِرَ اللَّيْلَ كُلَّهُ، وَلَمْ يَنَمْ.

"وَقَدْ قَسْقَسَ لَيْلَهُ أَجْمَعَ، إِذَا لَمْ يَنَمْ"
أبو عمرو الشّيبانيّ
تهذيب اللُّغة: أبو منصور الأزهريّ الهَرَويّ (ت، 370هـ)، تحقيق: عبد السّلام هارون وآخرين، دار القومية العربية- المؤسسة المصرية العامة- الدار المصرية- مطابع سجل العرب- دار الكتاب العربي- مكتبة الخانجي، القاهرة، (1384هـ/ 1964م- 1396هـ/ 1976م)، ‏8/‏259‏.
ن244هـ=858م

قَسْقَسَ الشَّخْصُ وَنَحْوُهُ: سَاقَ الإِبِلَ سَوْقًا شَدِيدًا.

"وَيُقَالُ: قَسْقَسَ لَيْلَتَهُ"
من رواية ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏188‏.
ن401هـ=1010م

قَسْقَسَ فِي مِشْيَتِهِ: أَسْرَعَ فِيهَا.

"يُقَالُ: قَسْقَسَ الرَّجُلُ فِي مِشْيَتِهِ"
من رواية أبي عُبيد الهَرَويّ الفاشانيّ
الغريبين في القرآن والحديث: أبو عُبيد الهَرَويّ الفاشانيّ (ت، 401هـ)، تحقيق ودراسة: أحمد فريد المزيدي، قدم له وراجعه: فتحي حجازي، مكتبة نزار مصطفى الباز، مكّة المكرّمة- الرّياض، ط1، (1419هـ/ 1999م)، ‏5/‏1540‏.
ن774هـ=1372م

قَسْقَسَ القَوْمُ: صَخِبُوا وَرَفَعُوا أَصْوَاتَهَمْ.

"فَلَمَّا تَرَاءَى الجَمْعَانِ بَاتَ الرُّومُ يُقَسْقِسُونَ وَيُصَلِّبُونَ، وَبَاتَ المُسْلِمُونَ يَقْرَؤُونَ وَيُصَلُّونَ"
ابن كثير
البداية والنهاية: ابن كثير (ت، 774هـ)، تحقيق: عبد الله عبد المحسن التركي، مركز البحوث والدراسات العربية والإسلامية، دار هجر، القاهرة، ط1، (1417ه/ 1997م- (1420ه/ 1999م)، ‏10/‏237‏.

قَسْقَسَ[مُتَعَدٍّ]

280هـ=893م

قَسْقَسَ مَا عَلَى العَظْمِ مِنَ اللَّحْمِ: أَكَلَهُ.

"يُقَالُ: قَسْقَسْتُ مَا عَلَى العَظْمِ مِنَ اللَّحْمِ إِذَا أَكَلْتُهُ"
من رواية ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏1/‏203‏.
280هـ=893م

قَسْقَسَ مَا عَلَى المَائِدَةِ: أَكَلَهُ كُلَّهُ.

"يُقَالُ: ... قَسْقَسْتُ مَا عَلَى المَائِدَةِ إِذَا أَكَلْتُ كُلَّ مَا عَلَيْهَا"
من رواية ابن دُريد
كتاب جمهرة اللّغة: ابن دُريد (ت، 321هـ)، حققه وقدم له: رمزي منير بعلبكي، دار العلم للملايين، بيروت، ط1، (1987م)، ‏1/‏203‏.

قَسْقَسَ[مُتَعَدٍّ بِالحَرْف]

ن309هـ=921م

قَسْقَسَ بِالكَلْبِ: زَجَرَهُ وَأَبْعَدَهُ، وَقَالَ لَهُ: قُوسْ قُوسْ.

" وَقَسْقَسْتُ بِالكَلْبِ، وَقَرْقَسْتُ بِالجَرْوِ"
كُراع النَّمْل الهُنائيّ الأزديّ
المُنتَخَب من غريب كلام العَرَب: كُراع النَّمْل الهُنائيّ الأزديّ (ت، 309هـ)، تحقيق: محمّد بن أحمد العمري، جامعة أم القرى، مكة المكرمة، ط1، (1409هـ/ 1989م)، ‏299‏.
ن379هـ=989م

قَسْقَسَ بِالكَلْبِ: دَعَاهُ.

" وَقَسْقَسْتُ بِالكَلْبِ"
أبو بكر الزُّبيديّ الأندلسيّ
مختصر العين: أبو بكر الزُّبيديّ الأندلسيّ (ت، 379هـ)، إعداد: عبد العزيز بن حميد الحميد، رسالة ماجستير، جامعة محمّد بن سعود، الرياض، (1412هـ)، ‏2/‏820‏.

قِسْقِيس[صِفَة]

ن244هـ=858م

القِسْقِيسُ : السَّيْرُ السَّرِيعُ الشَّدِيدُ.

قَدْ عَلِمَ الصُّهْبُ المَهَارَى، والعِيسْ النَّافِخَاتُ، فِي البُرَى المَدَاعِيسْ أَنْ لَيْسَ بَيْنَ الحَفَرَيْنِ تَعْرِيسْ إِذَا حَدَاهُنَّ النَّجَاءُ القِسْقِيسْ
من إنشاد ابن السِّكّيت
كتاب الألفاظ: ابن السِّكّيت (ت، 244هـ)، تحقيق: فخر الدين قباوة، مكتبة لبنان ناشرون، بيروت، ط1، (1998م)، ‏202‏.

قَسْقَسَة[مَصْدَر]

436هـ=1044م

قَسْقَسَةُ الشَّيْءِ: تَحْرِيكُهُ.

"القَسْقَاسَةُ: العَصَا؛ قِيلَ فِي تَفْسِيرِهِ قَوْلَانِ: أَحَدُهُمَا أَنَّهُ أَرَادَ قَسْقَسَتَهُ : أَيْ تَحْرِيكَهُ إِيَّاهَا لِضَرْبِكِ"
ابن سِيدَهْ المُرْسيّ الأندلسيّ
المحكَم والمحيط الأعظم في اللُّغة: ابن سِيدَهْ المُرْسيّ الأندلسيّ (ت، 458هـ)، تحقيق: عبد الفتاح السيد سليم وآخرين، معهد المخطوطات العربية، القاهرة، ط2، (1424هـ/ 2003م)، ‏6/‏69‏.
